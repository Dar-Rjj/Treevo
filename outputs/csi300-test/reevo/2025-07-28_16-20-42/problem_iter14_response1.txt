```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 2-day simple moving average of the close price
    sma_2 = df['close'].rolling(window=2).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with equal weights
    weighted_prices = (0.25 * df['open'] + 0.25 * df['high'] + 0.25 * df['low'] + 0.25 * df['close'])
    
    # Calculate the ratio of (high - low) to the 2-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_2 + 1e-7)
    
    # Calculate the 2-day exponential moving average of the close price
    ema_2 = df['close'].ewm(span=2, adjust=False).mean()
    
    # Calculate the 2-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=2).std()
    
    # Final alpha factor as a simplified combination of the above metrics
    alpha_factor = (weighted_prices - sma_2) / (volatility_ratio + 1e-7) + (ema_2 - df['close']) / (realized_volatility + 1e-7)
    return alpha_factor
```
