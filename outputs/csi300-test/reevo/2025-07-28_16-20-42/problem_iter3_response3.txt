```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving average of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Momentum factor: difference between 5-day and 20-day moving averages
    momentum = ma_5 - ma_20
    
    # Volatility factor: standard deviation of the last 5 days' close prices
    volatility = df['close'].rolling(window=5).std()
    
    # Volume trend: 5-day moving average of volume
    volume_trend = df['volume'].rolling(window=5).mean()
    
    # Liquidity factor: ratio of amount to volume
    liquidity = df['amount'] / (df['volume'] + 1e-7)
    
    # Market trend: difference between today's close and the 20-day moving average
    market_trend = df['close'] - ma_20
    
    # Adaptive weighting based on the volatility
    weight_momentum = 1.0 / (volatility + 1e-7)
    weight_market_trend = 1.0 / (volatility + 1e-7)
    weight_volume_trend = 1.0 / (volatility + 1e-7)
    
    # Final alpha factor as a weighted combination of the above metrics
    alpha_factor = (weight_momentum * momentum + 
                    weight_market_trend * market_trend + 
                    weight_volume_trend * volume_trend + 
                    liquidity)
    return alpha_factor
```
