def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving averages of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close'])
    
    # Calculate the ratio of volume to the moving amount
    volatility_ratio = (df['high'] - df['low']) / (ma_5 + 1e-7)
    
    # Calculate the momentum factor as the difference between the 5-day and 20-day moving averages
    momentum_factor = ma_5 - ma_20
    
    # Calculate the liquidity factor as the ratio of volume to amount
    liquidity_factor = df['volume'] / (df['amount'] + 1e-7)
    
    # Calculate the trend strength as the percentage change in the 20-day moving average
    trend_strength = (ma_20 - ma_20.shift(5)) / (ma_20.shift(5) + 1e-7)
    
    # Adaptive weighting based on trend strength
    adaptive_weight_momentum = np.tanh(trend_strength)
    adaptive_weight_volatility = 1 - adaptive_weight_momentum
    
    # Combine the factors with adaptive weighting
    alpha_factor = (weighted_prices - ma_5) / (volatility_ratio + 1e-7) * adaptive_weight_volatility + \
                   adaptive_weight_momentum * momentum_factor + 0.3 * liquidity_factor
    return alpha_factor
