```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day moving average of the close price for trend indication
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Relative strength factor: 10 day return
    rs_factor = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Aroon indicator to measure the trend
    aroon_up = ((df['high'].rolling(window=14).apply(lambda x: x.argmax(), raw=True) - 13) / 13) * 100
    aroon_down = ((13 - df['low'].rolling(window=14).apply(lambda x: x.argmin(), raw=True)) / 13) * 100
    aroon_oscillator = aroon_up - aroon_down
    
    # Liquidity measure: Volume relative to the 5-day average volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    liquidity_ratio = df['volume'] / (avg_volume_5d + 1e-7)
    
    # Price momentum adjusted by volatility
    price_momentum = (df['close'] - df['close'].shift(5))
    volatility = df['close'].rolling(window=5).std()
    adj_momentum = price_momentum / (volatility + 1e-7)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (rs_factor * 0.4) + (aroon_oscillator * 0.2) + (liquidity_ratio * 0.2) + (adj_momentum * 0.2)
    
    return alpha_factor
```
