def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day simple moving average of the close price for a longer trend
    sma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the 3-day simple moving average of the close price for a shorter trend
    sma_3 = df['close'].rolling(window=3).mean()
    
    # Adaptive weights based on the ratio of volume to its 10-day mean, capturing liquidity
    volume_mean_10 = df['volume'].rolling(window=10).mean()
    adaptive_weights = (df['volume'] / volume_mean_10)
    
    # Weighted sum of the open, high, low, and close prices with adaptive weights
    weighted_prices = (adaptive_weights * 0.15 * df['open'] + 
                       adaptive_weights * 0.35 * df['high'] + 
                       adaptive_weights * 0.35 * df['low'] + 
                       adaptive_weights * 0.15 * df['close'])
    
    # Volatility ratio using the 5-day simple moving average
    volatility_ratio = (df['high'] - df['low']) / (sma_5 + 1e-7)
    
    # 3-day exponential moving average of the close price
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    
    # 3-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=3).std()
    
    # Mean reversion component: difference between the current close and the 5-day SMA
    mean_reversion = (df['close'] - sma_5) / (sma_5 + 1e-7)
    
    # Momentum component: difference between the 3-day EMA and the 5-day SMA
    momentum = (ema_3 - sma_5) / (sma_5 + 1e-7)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (weighted_prices - sma_3) / (volatility_ratio + 1e-7) + \
                   (ema_3 - df['close']) / (realized_volatility + 1e-7) + \
                   mean_reversion + momentum
    return alpha_factor
