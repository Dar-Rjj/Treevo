```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving averages of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()

    # Calculate a weighted sum of the open, high, low, and close prices
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close'])

    # Calculate the ratio of volume to the moving amount
    volatility_ratio = (df['high'] - df['low']) / (ma_5 + 1e-7)

    # Calculate the momentum factor
    momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)

    # Calculate the volume trend over 5 days
    volume_trend = df['volume'].rolling(window=5).mean() / df['volume'].rolling(window=20).mean()

    # Adaptive weighting based on the market trend
    trend_weight = (ma_5 - ma_20) / (ma_5 + 1e-7)
    trend_weight = trend_weight.clip(lower=-1, upper=1)

    # Final alpha factor as a combination of the above metrics
    alpha_factor = (weighted_prices - ma_5) / (volatility_ratio + 1e-7) + momentum + volume_trend * trend_weight
    return alpha_factor
```
