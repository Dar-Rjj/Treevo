def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day moving average of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with simplified weights
    weighted_prices = (0.2 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.2 * df['close'])
    
    # Calculate the ratio of (high - low) to the 5-day moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (ma_5 + 1e-7)
    
    # Calculate the 5-day exponential moving average of the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 5-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()
    
    # Calculate the momentum factor as the difference between the current close price and the 5-day moving average
    momentum_factor = df['close'] - ma_5
    
    # Calculate the liquidity factor as the ratio of volume to the amount
    liquidity_factor = df['volume'] / (df['amount'] + 1e-7)
    
    # Final alpha factor as a balanced combination of the above metrics
    alpha_factor = ((weighted_prices - ma_5) / (volatility_ratio + 1e-7) + 
                    (ema_5 - df['close']) / (realized_volatility + 1e-7) +
                    0.5 * momentum_factor + 0.5 * liquidity_factor)
    return alpha_factor
