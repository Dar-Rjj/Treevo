```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and the open price, suggesting momentum
    momentum = df['close'] - df['open']
    
    # Compute the average of high and low for a typical price representation
    avg_price = (df['high'] + df['low']) / 2
    
    # Consider the volume weighted amount, reflecting transaction activity
    vol_weighted_amount = df['amount'] / df['volume']
    
    # Calculate the daily range to measure volatility
    daily_range = df['high'] - df['low']
    
    # Compute a 5-day moving average of the close price
    moving_avg_5 = df['close'].rolling(window=5).mean()
    
    # Compute the ratio of the current close to the moving average
    ma_ratio = (df['close'] / moving_avg_5)
    
    # Create a factor that combines momentum with the relative position of the close to the day's average price
    # This factor aims to capture the strength of the price movement adjusted by where the closing occurred in relation to the day's average price
    # Incorporate volatility and trading volume intensity
    factor = (momentum * vol_weighted_amount * daily_range) / ((avg_price - df['low']) + 1e-7) * ma_ratio
    
    return factor
```
