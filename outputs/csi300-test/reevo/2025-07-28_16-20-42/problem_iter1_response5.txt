```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price
    price_diff = df['close'] - df['open']
    # Calculate the average of high and low
    avg_price = (df['high'] + df['low']) / 2
    # Calculate the ratio of volume to amount (assuming amount is in dollars)
    vol_to_amt_ratio = df['volume'] / df['amount']
    # Generate the alpha factor by combining the above calculations
    factor = (price_diff / avg_price) * vol_to_amt_ratio
    return factor
```
