```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day moving average of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with adaptive weights
    adaptive_weights_open = 0.1 + (df['open'] - df['open'].shift(1)) / (df['high'] - df['low'] + 1e-7)
    adaptive_weights_high = 0.3 + (df['high'] - df['high'].shift(1)) / (df['high'] - df['low'] + 1e-7)
    adaptive_weights_low = 0.3 + (df['low'] - df['low'].shift(1)) / (df['high'] - df['low'] + 1e-7)
    adaptive_weights_close = 0.3 + (df['close'] - df['close'].shift(1)) / (df['high'] - df['low'] + 1e-7)
    weighted_prices = (adaptive_weights_open * df['open'] + adaptive_weights_high * df['high'] + 
                       adaptive_weights_low * df['low'] + adaptive_weights_close * df['close'])
    
    # Calculate the ratio of (high - low) to the 5-day moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (ma_5 + 1e-7)
    
    # Calculate the 5-day exponential moving average of the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 5-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()
    
    # Incorporate market trend by calculating the 20-day momentum
    momentum_20 = df['close'] / df['close'].shift(20) - 1
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (weighted_prices - ma_5) / (volatility_ratio + 1e-7) + (ema_5 - df['close']) / (realized_volatility + 1e-7) + momentum_20
    return alpha_factor
```
