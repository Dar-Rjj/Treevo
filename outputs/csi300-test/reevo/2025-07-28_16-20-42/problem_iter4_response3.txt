def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the exponentially weighted moving averages of the close price
    ewma_5 = df['close'].ewm(span=5).mean()
    ewma_20 = df['close'].ewm(span=20).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close'])
    
    # Calculate the volatility ratio using exponential weighting
    volatility_ratio = (df['high'] - df['low']) / (ewma_5 + 1e-7)
    
    # Calculate the momentum factor as the difference between the 5-day and 20-day exponential moving averages
    momentum_factor = ewma_5 - ewma_20
    
    # Calculate the liquidity factor as the ratio of volume to amount
    liquidity_factor = df['volume'] / (df['amount'] + 1e-7)
    
    # Calculate the market trend as the difference between today's close and the 20-day exponential moving average
    market_trend = df['close'] - ewma_20
    
    # Adaptive weighting based on the volatility
    weight_momentum = 1.0 / (volatility_ratio + 1e-7)
    weight_market_trend = 1.0 / (volatility_ratio + 1e-7)
    weight_liquidity = 1.0 / (volatility_ratio + 1e-7)
    
    # Combine the factors with adaptive weighting
    alpha_factor = (
        (weighted_prices - ewma_5) / (volatility_ratio + 1e-7) +
        0.5 * weight_momentum * momentum_factor +
        0.3 * weight_liquidity * liquidity_factor +
        0.2 * weight_market_trend * market_trend
    )
    return alpha_factor
