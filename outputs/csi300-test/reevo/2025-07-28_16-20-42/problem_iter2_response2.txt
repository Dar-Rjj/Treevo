```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the true range for a given day
    true_range = df['high'] - df['low']
    
    # Calculate the average volume over the past 5 days
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate momentum based on the close price of 5 days difference
    momentum = df['close'] - df['close'].shift(5)
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up = up.rolling(window=14).mean()
    roll_down = down.abs().rolling(window=14).mean().abs()
    RS = roll_up / roll_down
    RSI = 100.0 - (100.0 / (1.0 + RS))
    
    # Adaptive weighting based on the RSI
    weight_momentum = 0.7 if RSI[-1] > 70 else 0.3
    weight_true_range = 1.0 - weight_momentum
    
    # Combine true range, average volume, and momentum with adaptive weights
    factor_values = (true_range * weight_true_range + avg_volume) * (momentum * weight_momentum)
    
    return factor_values
```
