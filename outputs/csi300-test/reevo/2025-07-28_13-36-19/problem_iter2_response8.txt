```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the simple moving average (SMA) of the close price over 10 days as a momentum indicator
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the difference between the closing price and the 10-day SMA to capture momentum
    momentum = df['close'] - sma_10
    
    # Calculate the average true range (ATR) over 14 days as a volatility measure
    tr = df['high'] - df['low']
    atr = tr.rolling(window=14).mean()
    
    # Calculate the percentage change in volume compared to the previous day to capture volume shocks
    volume_shock = df['volume'].pct_change()
    
    # Calculate the ratio of today's close price to the 5-day maximum high, capturing short-term price trends
    max_high_5 = df['high'].rolling(window=5).max()
    short_trend = df['close'] / max_high_5
    
    # Combine the factors: momentum, volatility, volume shock, and short-term price trend
    alpha_factor = (momentum / atr) * volume_shock * short_trend
    
    return alpha_factor
```
