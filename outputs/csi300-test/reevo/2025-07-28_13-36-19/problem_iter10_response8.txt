```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price movement as a percentage change from open to close
    price_movement = (df['close'] - df['open']) / df['open']

    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2

    # Calculate the relative strength index (RSI) for the close prices
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the volume trend
    volume_trend = df['volume'].pct_change()

    # Calculate the exponential moving average of the price movement
    ema_price_movement = price_movement.ewm(span=14, adjust=False).mean()

    # Calculate the seasonal adjusted price movement
    seasonal_adj_price_movement = price_movement.rolling(window=5).mean() - price_movement.rolling(window=20).mean()

    # Calculate the volatility
    volatility = df['close'].rolling(window=14).std()

    # Incorporate dynamic weights
    dynamic_weight_ema = (price_movement * df['amount']).ewm(span=14, adjust=False).mean()
    dynamic_weight_rsi = (rsi * df['volume']).ewm(span=14, adjust=False).mean()
    dynamic_weight_volume_trend = (volume_trend * avg_price).ewm(span=14, adjust=False).mean()
    dynamic_weight_seasonal_adj = (seasonal_adj_price_movement * (df['amount'] + df['volume'])).ewm(span=14, adjust=False).mean()

    # Sum of the dynamic weights
    total_dynamic_weights = dynamic_weight_ema + dynamic_weight_rsi + dynamic_weight_volume_trend + dynamic_weight_seasonal_adj

    # Normalize the dynamic weights
    normalized_ema = dynamic_weight_ema / total_dynamic_weights
    normalized_rsi = dynamic_weight_rsi / total_dynamic_weights
    normalized_volume_trend = dynamic_weight_volume_trend / total_dynamic_weights
    normalized_seasonal_adj = dynamic_weight_seasonal_adj / total_dynamic_weights

    # Include more macroeconomic data
    macro_indicator = (df['amount'] * df['volume']) / avg_price

    # Combine the factors with dynamic weights
    factor = (normalized_ema * ema_price_movement + 
              normalized_rsi * rsi + 
              normalized_volume_trend * volume_trend + 
              normalized_seasonal_adj * seasonal_adj_price_movement + 
              (macro_indicator * (df['amount'] + df['volume'])) / (2 * avg_price)) / 5

    return factor
```
