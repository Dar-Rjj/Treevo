```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the average of the open and close prices
    avg_open_close = (df['open'] + df['close']) / 2
    
    # Calculate the difference between the high and low prices
    diff_high_low = df['high'] - df['low']
    
    # Calculate the moving average of the volume over the last 5 days
    ma_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the factor as the ratio of the average of open and close to the difference between high and low,
    # scaled by the 5-day moving average of the volume, adding a small constant to avoid division by zero.
    factor = (avg_open_close / (diff_high_low + 1e-7)) * (ma_volume_5d + 1e-7)
    
    return factor
```
