def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price movement as a percentage change from open to close
    price_movement = (df['close'] - df['open']) / df['open']

    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2

    # Calculate the relative strength index (RSI) for the close prices with adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Adaptive RSI window based on volatility
    volatility = df['close'].pct_change().rolling(window=14).std()
    adaptive_rsi_window = 14
    if volatility[-1] > df['close'].pct_change().std():
        adaptive_rsi_window = 10
    else:
        adaptive_rsi_window = 20
    gain_adaptive = (delta.where(delta > 0, 0)).rolling(window=adaptive_rsi_window).mean()
    loss_adaptive = (-delta.where(delta < 0, 0)).rolling(window=adaptive_rsi_window).mean()
    rs_adaptive = gain_adaptive / loss_adaptive
    rsi_adaptive = 100 - (100 / (1 + rs_adaptive))

    # Calculate the volume trend
    volume_trend = df['volume'].pct_change()

    # Exponential moving average of the price movement
    ema_price_movement = price_movement.ewm(span=14, adjust=False).mean()

    # Dynamic indicator using the difference between EMA and simple moving average
    sma_price_movement = price_movement.rolling(window=14).mean()
    dynamic_indicator = ema_price_movement - sma_price_movement

    # Enhanced seasonal adjusted price movement
    seasonal_adj_price_movement = (price_movement.rolling(window=5).mean() - price_movement.rolling(window=20).mean()) * 2

    # Dynamic weighting based on volatility and stability
    weight_volatility = 1 / (1 + np.exp(-volatility))
    weight_stability = 1 - weight_volatility

    # Incorporate seasonality in the dynamic indicator
    seasonal_dynamic_indicator = dynamic_indicator * (1 + seasonal_adj_price_movement)

    # Adaptive amount and volume weighted by volatility
    adaptive_amount = df['amount'] * weight_volatility
    adaptive_volume = df['volume'] * weight_stability

    # Integrate MACD
    exp1 = df['close'].ewm(span=12, adjust=False).mean()
    exp2 = df['close'].ewm(span=26, adjust=False).mean()
    macd = exp1 - exp2
    signal = macd.ewm(span=9, adjust=False).mean()
    macd_diff = macd - signal

    # Integrate Bollinger Bands
    bb_middle = df['close'].rolling(window=20).mean()
    bb_std = df['close'].rolling(window=20).std()
    bb_upper = bb_middle + 2 * bb_std
    bb_lower = bb_middle - 2 * bb_std
    bb_percent = (df['close'] - bb_lower) / (bb_upper - bb_lower)
    
    # Integrate ATR (Average True Range)
    tr = pd.DataFrame(index=df.index)
    tr['h-l'] = df['high'] - df['low']
    tr['h-pc'] = abs(df['high'] - df['close'].shift(1))
    tr['l-pc'] = abs(df['low'] - df['close'].shift(1))
    tr['tr'] = tr[['h-l', 'h-pc', 'l-pc']].max(axis=1)
    atr = tr['tr'].rolling(window=14).mean()

    # Calculate the weighted average of the price movement, RSI, volume trend, and dynamic indicator with enhanced seasonality
    factor = (ema_price_movement * adaptive_amount +
              rsi_adaptive * adaptive_volume +
              volume_trend * avg_price * weight_volatility +
              seasonal_dynamic_indicator * (adaptive_amount + adaptive_volume) +
              macd_diff * df['volume'] * weight_volatility +
              bb_percent * df['amount'] * weight_stability +
              atr * df['volume'] * weight_volatility) / (7 * (adaptive_amount + adaptive_volume))

    return factor
