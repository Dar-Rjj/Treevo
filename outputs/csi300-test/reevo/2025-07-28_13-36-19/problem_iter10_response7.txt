```python
import pandas as pd
import numpy as np

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price movement as a percentage change from open to close
    price_movement = (df['close'] - df['open']) / df['open']

    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2

    # Calculate the relative strength index (RSI) for the close prices
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the volume trend
    volume_trend = df['volume'].pct_change()

    # Calculate the exponential moving average of the price movement
    ema_price_movement = price_movement.ewm(span=14, adjust=False).mean()

    # Calculate the seasonal adjusted price movement
    seasonal_adj_price_movement = price_movement.rolling(window=5).mean() - price_movement.rolling(window=20).mean()

    # Calculate the dynamic indicator using the difference between EMA and simple moving average
    sma_price_movement = price_movement.rolling(window=14).mean()
    dynamic_indicator = ema_price_movement - sma_price_movement

    # Calculate the volatility
    volatility = df['close'].rolling(window=20).std()

    # Integrate macroeconomic indicators (e.g., interest rate, GDP growth)
    # For simplicity, we'll use a dummy macroeconomic indicator here
    macro_indicator = df['close'].rolling(window=30).mean()

    # Adaptive window size for the dynamic indicator based on volatility
    adaptive_window = (volatility * 10).astype(int) + 1
    adaptive_sma_price_movement = price_movement.rolling(window=adaptive_window, min_periods=1).mean()
    adaptive_dynamic_indicator = ema_price_movement - adaptive_sma_price_movement

    # Calculate the weighted average of the price movement, RSI, volume trend, and dynamic indicator with seasonality and macroeconomic indicators
    factor = (
        ema_price_movement * df['amount'] * (1 + macro_indicator) +
        rsi * df['volume'] * (1 + macro_indicator) +
        volume_trend * avg_price * (1 + macro_indicator) +
        adaptive_dynamic_indicator * (df['amount'] + df['volume']) * (1 + macro_indicator) +
        seasonal_adj_price_movement * (df['amount'] + df['volume']) * (1 + macro_indicator)
    ) / (4 * (df['amount'] + df['volume']))

    return factor
```
