def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price movement as a percentage change from open to close
    price_movement = (df['close'] - df['open']) / df['open']

    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2

    # Calculate the relative strength index (RSI) for the close prices with adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Adaptive window for RSI based on volatility
    adaptive_rsi_window = 14
    if df['close'].std() > df['close'].mean():
        adaptive_rsi_window = 10
    else:
        adaptive_rsi_window = 20
    gain = (delta.where(delta > 0, 0)).rolling(window=adaptive_rsi_window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=adaptive_rsi_window).mean()
    rs = gain / loss
    rsi_adaptive = 100 - (100 / (1 + rs))

    # Calculate the volume trend
    volume_trend = df['volume'].pct_change()

    # Calculate the exponential moving average of the price movement
    ema_price_movement = price_movement.ewm(span=14, adjust=False).mean()

    # Calculate the seasonal adjusted price movement
    seasonal_adj_price_movement = price_movement.rolling(window=5).mean() - price_movement.rolling(window=20).mean()

    # Calculate the dynamic indicator using the difference between EMA and simple moving average
    sma_price_movement = price_movement.rolling(window=14).mean()
    dynamic_indicator = ema_price_movement - sma_price_movement

    # Dynamic weighting based on volatility
    volatility = df['close'].pct_change().rolling(window=14).std()
    weight_volatility = 1 / (1 + np.exp(-volatility))
    weight_stability = 1 - weight_volatility

    # Incorporate market sentiment (e.g., positive or negative news sentiment)
    # For simplicity, assume a hypothetical sentiment score is available in the DataFrame
    sentiment_score = df.get('sentiment_score', 0)

    # Use more adaptive windows for smoothing
    adaptive_weight_ema = weight_volatility.ewm(span=10, adjust=False).mean()
    adaptive_weight_rsi = weight_stability.ewm(span=10, adjust=False).mean()
    adaptive_weight_vol_trend = weight_volatility.ewm(span=10, adjust=False).mean()
    adaptive_weight_dynamic = weight_volatility.ewm(span=10, adjust=False).mean()
    adaptive_weight_seasonal = weight_stability.ewm(span=10, adjust=False).mean()

    # Consider macroeconomic indicators (e.g., GDP growth, interest rates, etc.)
    # For simplicity, assume a hypothetical macroeconomic indicator is available in the DataFrame
    macro_indicator = df.get('macro_indicator', 0)

    # Calculate the weighted average of the factors with adaptive weights and market sentiment
    factor = (
        adaptive_weight_ema * ema_price_movement +
        adaptive_weight_rsi * rsi_adaptive +
        adaptive_weight_vol_trend * volume_trend +
        adaptive_weight_dynamic * dynamic_indicator +
        adaptive_weight_seasonal * seasonal_adj_price_movement
    )

    # Adjust the factor based on market sentiment and macroeconomic indicators
    factor = factor * (1 + sentiment_score / 100) * (1 + macro_indicator / 100)

    return factor
