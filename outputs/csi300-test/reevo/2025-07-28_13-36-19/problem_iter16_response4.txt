```python
import pandas as pd
import numpy as np

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price movement as a percentage change from open to close
    price_movement = (df['close'] - df['open']) / df['open']

    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2

    # Calculate the relative strength index (RSI) for the close prices with adaptive window
    def calculate_rsi(prices, window=14):
        delta = prices.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        return rsi

    # Adaptive RSI window based on market volatility
    adaptive_rsi_window = 14
    if df['close'].std() > df['close'].mean():
        adaptive_rsi_window = 10
    else:
        adaptive_rsi_window = 20
    rsi_adaptive = calculate_rsi(df['close'], window=adaptive_rsi_window)

    # Calculate the volume trend
    volume_trend = df['volume'].pct_change()

    # Calculate the exponential moving average (EMA) of the price movement
    ema_price_movement = price_movement.ewm(span=14, adjust=False).mean()

    # Calculate the seasonal adjusted price movement
    seasonal_adj_price_movement = price_movement.rolling(window=5).mean() - price_movement.rolling(window=20).mean()

    # Dynamic indicator using the difference between EMA and simple moving average
    sma_price_movement = price_movement.rolling(window=14).mean()
    dynamic_indicator = ema_price_movement - sma_price_movement

    # Dynamic weighting based on volatility
    volatility = df['close'].pct_change().rolling(window=14).std()
    weight_volatility = 1 / (1 + np.exp(-volatility))
    weight_stability = 1 - weight_volatility

    # Momentum factor
    momentum = df['close'].pct_change(periods=12).rolling(window=3).mean()

    # Additional factors
    amount_to_volume_ratio = df['amount'] / df['volume']
    money_flow_index = (df['amount'] * df['close']).rolling(window=14).sum() / df['amount'].rolling(window=14).sum()

    # Adaptive window for EMA and SMA based on market conditions
    adaptive_ema_sma_window = 14
    if df['close'].std() > df['close'].mean():
        adaptive_ema_sma_window = 10
    else:
        adaptive_ema_sma_window = 20
    ema_price_movement_adaptive = price_movement.ewm(span=adaptive_ema_sma_window, adjust=False).mean()
    sma_price_movement_adaptive = price_movement.rolling(window=adaptive_ema_sma_window).mean()

    # Combine the factors with dynamic weighting
    factor = (
        ema_price_movement_adaptive * df['amount'] * weight_volatility +
        rsi_adaptive * df['volume'] * weight_stability +
        volume_trend * avg_price * weight_volatility +
        (ema_price_movement_adaptive - sma_price_movement_adaptive) * (df['amount'] + df['volume']) * weight_volatility +
        seasonal_adj_price_movement * (df['amount'] + df['volume']) * weight_stability +
        momentum * (df['amount'] + df['volume']) * weight_volatility +
        amount_to_volume_ratio * df['amount'] * weight_stability +
        money_flow_index * df['volume'] * weight_volatility
    ) / (8 * (df['amount'] + df['volume']))

    return factor
```
