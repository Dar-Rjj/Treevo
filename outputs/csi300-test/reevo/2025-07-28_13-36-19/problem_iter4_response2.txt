```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    daily_return = df['close'].pct_change().fillna(0)
    
    # Calculate the 5-day momentum (difference between the current close and the close 5 days ago)
    momentum = df['close'] - df['close'].shift(5).fillna(0)
    
    # Calculate the relative strength (current close divided by the 5-day moving average of the close)
    relative_strength = df['close'] / df['close'].rolling(window=5).mean()
    relative_strength.fillna(1, inplace=True)
    
    # Calculate the liquidity (ratio of volume to the 5-day average volume)
    liquidity = df['volume'] / df['volume'].rolling(window=5).mean()
    liquidity.fillna(1, inplace=True)
    
    # Factor: Combine daily return, momentum, relative strength, and liquidity
    factor = 0.4 * daily_return + 0.3 * momentum + 0.2 * relative_strength + 0.1 * liquidity
    
    return factor
```
