```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    import pandas as pd

    # Calculate the price change from open to close
    price_change = df['close'] - df['open']

    # Calculate the average price between high and low
    avg_price = (df['high'] + df['low']) / 2

    # Calculate the volatility as the range between high and low
    volatility = df['high'] - df['low']

    # Calculate the momentum using the percentage change in close prices
    momentum = df['close'].pct_change()

    # Calculate the money flow multiplier
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])

    # Calculate the money flow volume
    money_flow_volume = money_flow_multiplier * df['volume']

    # Calculate the accumulation/distribution line
    ad_line = money_flow_volume.cumsum()

    # Calculate the 10-day and 30-day moving averages of the closing price
    ma_close_10 = df['close'].rolling(window=10).mean()
    ma_close_30 = df['close'].rolling(window=30).mean()

    # Calculate the relative strength, which is the ratio of the closing price to the 10-day moving average of the closing price
    relative_strength = df['close'] / ma_close_10

    # Calculate the long-term momentum, which is the ratio of the 10-day moving average to the 30-day moving average
    long_term_momentum = ma_close_10 / ma_close_30

    # Calculate the daily return
    daily_return = (df['close'] - df['open']) / df['open']

    # Calculate the weighted average of the price change, average price, and volume
    factor = (price_change * df['volume'] +
              avg_price * df['amount'] +
              volatility * df['volume'] +
              momentum * df['amount'] +
              ad_line * df['volume'] +
              relative_strength * df['amount'] +
              long_term_momentum * df['amount'] +
              daily_return * df['volume']) / (df['volume'] + df['amount'])

    return factor
```
