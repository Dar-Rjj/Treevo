```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Momentum - Calculate the 10-period simple moving average of the typical price
    momentum = typical_price.rolling(window=10).mean()
    
    # Volatility - Calculate the standard deviation of the typical price over a 10-period window
    volatility = typical_price.rolling(window=10).std()
    
    # Liquidity - Calculate the dollar volume (typical price * volume)
    liquidity = typical_price * df['volume']
    
    # Relative strength - Calculate the ratio of the current close to the 10-period simple moving average of the close
    relative_strength = df['close'] / df['close'].rolling(window=10).mean()
    
    # Combine momentum, volatility, and liquidity with relative strength
    factor = (momentum / (volatility + 1e-7)) * (liquidity / df['volume']) * relative_strength
    
    return factor
```
