```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open prices (momentum indicator)
    price_diff = df['close'] - df['open']
    
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the weighted average of the price difference and average price with volume
    weighted_avg = (price_diff * df['volume'] + avg_price * df['amount']) / (df['volume'] + df['amount'])
    
    # Calculate the rolling standard deviation of the close price (volatility indicator)
    volatility = df['close'].rolling(window=10).std()
    
    # Calculate the rolling mean of the volume (volume trend indicator)
    volume_trend = df['volume'].rolling(window=10).mean()
    
    # Combine the weighted average, volatility, and volume trend
    factor = (weighted_avg / (volatility + 1e-7)) * (volume_trend / (df['volume'] + 1e-7))
    
    return factor
```
