```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price movement as a percentage change from open to close
    price_movement = (df['close'] - df['open']) / df['open']

    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2

    # Calculate the relative strength index (RSI) for the close prices
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the volume trend
    volume_trend = df['volume'].pct_change()

    # Calculate the exponential moving average of the price movement with adaptive window
    ema_price_movement = price_movement.ewm(span=df['volume'].rolling(window=14).mean(), adjust=False).mean()

    # Calculate the seasonal adjusted price movement
    seasonal_adj_price_movement = price_movement.rolling(window=5).mean() - price_movement.rolling(window=20).mean()

    # Calculate the dynamic indicator using the difference between EMA and simple moving average
    sma_price_movement = price_movement.rolling(window=14).mean()
    dynamic_indicator = ema_price_movement - sma_price_movement

    # Calculate the weighted average of the price movement, RSI, volume trend, and dynamic indicator with seasonality
    weight_price_movement = df['amount'].rolling(window=14).mean() / df['amount'].rolling(window=14).mean().max()
    weight_rsi = df['volume'].rolling(window=14).mean() / df['volume'].rolling(window=14).mean().max()
    weight_volume_trend = avg_price.rolling(window=14).mean() / avg_price.rolling(window=14).mean().max()
    weight_dynamic_indicator = (df['amount'] + df['volume']).rolling(window=14).mean() / (df['amount'] + df['volume']).rolling(window=14).mean().max()
    weight_seasonal_adj_price_movement = (df['amount'] + df['volume']).rolling(window=14).mean() / (df['amount'] + df['volume']).rolling(window=14).mean().max()

    factor = (ema_price_movement * weight_price_movement + rsi * weight_rsi + 
              volume_trend * weight_volume_trend + dynamic_indicator * weight_dynamic_indicator + 
              seasonal_adj_price_movement * weight_seasonal_adj_price_movement)

    return factor
```
