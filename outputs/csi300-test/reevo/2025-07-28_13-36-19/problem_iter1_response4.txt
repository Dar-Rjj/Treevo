```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between closing price and opening price
    diff_close_open = df['close'] - df['open']
    
    # Calculate the average of high and low for the day
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the normalized volume, which is the volume divided by the average volume over the last 5 days
    norm_volume = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Calculate the alpha factor, which combines the difference between close and open, 
    # with how much the closing price deviates from the average of high and low, adjusted by the normalized volume.
    alpha_factor = (diff_close_open * (df['close'] - avg_high_low)) * norm_volume
    
    return alpha_factor
```
