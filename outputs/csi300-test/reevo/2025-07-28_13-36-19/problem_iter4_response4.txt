```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day and 50-day moving averages of the close price to capture short-term and long-term trends.
    df['ma_20'] = df['close'].rolling(window=20).mean()
    df['ma_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate the relative strength (RS) as the ratio of the 20-day moving average to the 50-day moving average.
    df['relative_strength'] = df['ma_20'] / df['ma_50']
    
    # Calculate the rate of change (ROC) for the close price over a 10-day period to capture momentum.
    df['roc_10'] = df['close'] / df['close'].shift(10) - 1
    
    # Calculate the volume momentum as the log of the ratio of today's volume to the 20-day moving average volume.
    df['volume_momentum'] = np.log(df['volume'] / df['volume'].rolling(window=20).mean())
    
    # Calculate the price range as the difference between the high and low prices.
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the relative price range as the ratio of the price range to the 20-day average price range.
    df['relative_price_range'] = df['price_range'] / df['price_range'].rolling(window=20).mean()
    
    # Combine the factors to create a new alpha factor that reflects the relationship between trend, momentum, and volume changes.
    alpha_factor = df['relative_strength'] * df['roc_10'] * df['volume_momentum'] * df['relative_price_range']
    return alpha_factor
```
