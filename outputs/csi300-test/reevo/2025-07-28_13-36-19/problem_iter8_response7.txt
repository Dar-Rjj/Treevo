```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price movement as a percentage change from open to close
    price_movement = (df['close'] - df['open']) / df['open']

    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2

    # Calculate the relative strength index (RSI) for the close prices with an 8-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=8).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=8).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the volume trend using a 5-day rolling average
    volume_trend = df['volume'].pct_change().rolling(window=5).mean()

    # Calculate the exponential moving average of the price movement with a 15-day span
    ema_price_movement = price_movement.ewm(span=15, adjust=False).mean()

    # Calculate the seasonal adjusted price movement using a 10-day and 30-day rolling mean
    seasonal_adj_price_movement = price_movement.rolling(window=10).mean() - price_movement.rolling(window=30).mean()

    # Calculate the weighted average of the price movement, RSI, and volume trend with seasonality
    factor = (ema_price_movement * df['amount'] + rsi * df['volume'] + volume_trend * avg_price + seasonal_adj_price_movement * (df['amount'] + df['volume'])) / (2 * (df['amount'] + df['volume']))

    return factor
```
