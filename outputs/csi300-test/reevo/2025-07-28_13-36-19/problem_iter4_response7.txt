```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price movement as a percentage change from open to close
    price_movement = (df['close'] - df['open']) / df['open']
    
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the relative strength index (RSI) for the close prices with adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the volume trend with adaptive window
    volume_trend = df['volume'].pct_change().rolling(window=14).mean()
    
    # Calculate the exponential moving average of the price movement with adaptive window
    ema_price_movement = price_movement.ewm(span=14, adjust=False).mean()
    
    # Calculate the volatility using standard deviation with adaptive window
    volatility = df['close'].rolling(window=14).std()
    
    # Integrate macroeconomic indicators (e.g., interest rates, GDP growth, etc.)
    # For this example, let's assume we have a column 'interest_rate' in the DataFrame
    macro_indicator = df['interest_rate']
    
    # Calculate the weighted average of the price movement, RSI, volume trend, and macroeconomic indicator
    factor = (ema_price_movement * df['amount'] + rsi * df['volume'] + volume_trend * avg_price + macro_indicator * volatility) / (df['amount'] + df['volume'] + avg_price + volatility)
    
    return factor
```
