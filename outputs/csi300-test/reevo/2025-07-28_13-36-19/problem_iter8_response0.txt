```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price movement as a percentage change from open to close
    price_movement = (df['close'] - df['open']) / df['open']

    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2

    # Adaptive window for RSI based on the volatility
    vol_window = df['close'].pct_change().rolling(window=10).std() * 10 + 10
    vol_window = vol_window.fillna(10).astype(int)

    # Calculate the relative strength index (RSI) for the close prices with an adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=vol_window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=vol_window).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the volume trend using an adaptive rolling average
    volume_trend_window = (df['volume'].pct_change().abs() * 5) + 5
    volume_trend = df['volume'].pct_change().rolling(window=volume_trend_window.astype(int)).mean()

    # Calculate the exponential moving average of the price movement with an adaptive span
    ema_span = (df['close'].pct_change().abs() * 20) + 20
    ema_price_movement = price_movement.ewm(span=ema_span, adjust=False).mean()

    # Calculate the seasonal adjusted price movement using a 10-day and 30-day rolling mean
    seasonal_adj_price_movement = price_movement.rolling(window=10).mean() - price_movement.rolling(window=30).mean()

    # Integrate macroeconomic indicators (e.g., GDP growth, inflation rate)
    gdp_growth = df['gdp_growth']  # Assuming 'gdp_growth' is a column in the DataFrame
    inflation_rate = df['inflation_rate']  # Assuming 'inflation_rate' is a column in the DataFrame

    # Calculate the weighted average of the price movement, RSI, and volume trend with seasonality and macroeconomic indicators
    factor = (
        ema_price_movement * df['amount'] +
        rsi * df['volume'] +
        volume_trend * avg_price +
        seasonal_adj_price_movement * (df['amount'] + df['volume']) +
        gdp_growth * df['volume'] +
        inflation_rate * df['amount']
    ) / (4 * (df['amount'] + df['volume']))

    return factor
```
