```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price movement ratio, which is the difference between today's close and yesterday's close,
    # divided by the average of today's high and low to standardize the price change.
    df['price_movement_ratio'] = (df['close'] - df['close'].shift(1)) / ((df['high'] + df['low']) / 2)
    # Calculate the volume momentum as the log of the ratio of today's volume to the 5-day moving average volume.
    df['volume_momentum'] = np.log(df['volume'] / df['volume'].rolling(window=5).mean())
    # Combine the two factors to create a new factor that reflects the relationship between price movements and volume changes.
    alpha_factor = df['price_movement_ratio'] * df['volume_momentum']
    return alpha_factor
```
