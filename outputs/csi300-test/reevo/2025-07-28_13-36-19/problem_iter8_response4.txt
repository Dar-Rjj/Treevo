```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price movement as a percentage change from open to close
    price_movement = (df['close'] - df['open']) / df['open']

    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2

    # Calculate the relative strength index (RSI) for the close prices with adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the volume trend
    volume_trend = df['volume'].pct_change()

    # Calculate the exponential moving average of the price movement with adaptive window
    ema_price_movement = price_movement.ewm(span=14, adjust=False).mean()

    # Calculate the seasonal adjusted price movement with adaptive window
    seasonal_adj_price_movement = price_movement.rolling(window=5).mean() - price_movement.rolling(window=20).mean()

    # Calculate the volatility of the price movement
    volatility = df['close'].rolling(window=20).std()

    # Use machine learning for dynamic factor weighting
    from sklearn.linear_model import LinearRegression
    model = LinearRegression()
    X = df[['amount', 'volume', 'avg_price']].values
    y = df['close'].pct_change().shift(-1).fillna(0).values
    model.fit(X, y)
    weights = model.coef_

    # Apply the learned weights to the factors
    weighted_ema_price_movement = ema_price_movement * weights[0]
    weighted_rsi = rsi * weights[1]
    weighted_volume_trend = volume_trend * weights[2]
    weighted_seasonal_adj_price_movement = seasonal_adj_price_movement * (weights[0] + weights[1])

    # Combine the factors
    factor = (weighted_ema_price_movement + weighted_rsi + weighted_volume_trend + weighted_seasonal_adj_price_movement) / (2 * (df['amount'] + df['volume'] + avg_price))

    return factor
```
