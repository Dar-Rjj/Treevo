```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    diff_close_open = df['close'] - df['open']
    
    # Calculate the difference between the high and low price
    diff_high_low = df['high'] - df['low']
    
    # Calculate a smoothed version of the volume
    smooth_volume = df['volume'].rolling(window=10).mean()
    
    # Heuristic factor: Ratio of the close-open difference to the high-low difference,
    # scaled by the smoothed volume
    factor_value = (diff_close_open / (diff_high_low + 1e-7)) * (smooth_volume / df['volume'])
    
    return factor_value
```
