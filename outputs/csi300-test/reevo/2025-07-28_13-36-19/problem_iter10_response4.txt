def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price movement as a percentage change from open to close
    price_movement = (df['close'] - df['open']) / df['open']

    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2

    # Calculate the relative strength index (RSI) for the close prices
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the volume trend
    volume_trend = df['volume'].pct_change()

    # Calculate the exponential moving average of the price movement
    ema_price_movement = price_movement.ewm(span=14, adjust=False).mean()

    # Calculate the seasonal adjusted price movement
    seasonal_adj_price_movement = price_movement.rolling(window=5).mean() - price_movement.rolling(window=20).mean()

    # Calculate the Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))

    # Calculate the Accumulation/Distribution Line (ADL)
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    adl = adl * df['volume']
    adl = adl.cumsum()

    # Calculate the On-Balance Volume (OBV)
    obv = (df['volume'] * (df['close'] > df['close'].shift(1))).cumsum()

    # Calculate the adaptive weights based on the volatility of each factor
    volatility_price_movement = price_movement.rolling(window=14).std()
    volatility_rsi = rsi.rolling(window=14).std()
    volatility_volume_trend = volume_trend.rolling(window=14).std()
    volatility_ema_price_movement = ema_price_movement.rolling(window=14).std()
    volatility_seasonal_adj_price_movement = seasonal_adj_price_movement.rolling(window=14).std()
    volatility_mfi = mfi.rolling(window=14).std()
    volatility_adl = adl.rolling(window=14).std()
    volatility_obv = obv.rolling(window=14).std()

    # Normalize the volatilities to sum to 1
    total_volatility = (volatility_price_movement + volatility_rsi + volatility_volume_trend + 
                       volatility_ema_price_movement + volatility_seasonal_adj_price_movement + 
                       volatility_mfi + volatility_adl + volatility_obv)
    weight_price_movement = volatility_price_movement / total_volatility
    weight_rsi = volatility_rsi / total_volatility
    weight_volume_trend = volatility_volume_trend / total_volatility
    weight_ema_price_movement = volatility_ema_price_movement / total_volatility
    weight_seasonal_adj_price_movement = volatility_seasonal_adj_price_movement / total_volatility
    weight_mfi = volatility_mfi / total_volatility
    weight_adl = volatility_adl / total_volatility
    weight_obv = volatility_obv / total_volatility

    # Combine the factors with adaptive weights
    factor = (price_movement * weight_price_movement + 
              rsi * weight_rsi + 
              volume_trend * weight_volume_trend + 
              ema_price_movement * weight_ema_price_movement + 
              seasonal_adj_price_movement * weight_seasonal_adj_price_movement + 
              mfi * weight_mfi + 
              adl * weight_adl + 
              obv * weight_obv)

    return factor
