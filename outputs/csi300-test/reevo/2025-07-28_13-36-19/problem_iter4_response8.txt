```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    daily_return = df['close'].pct_change().fillna(0)
    
    # Calculate the average of high and low as a proxy for the midpoint of the day's trading range
    midpoint = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of amount to volume, which could act as a proxy for the average price per share
    avg_price_per_share = df['amount'] / df['volume']
    
    # Calculate the difference between the close and open prices (daily movement)
    daily_movement = df['close'] - df['open']
    
    # Calculate the volatility (standard deviation of daily returns over a 5-day window)
    volatility = df['close'].rolling(window=5).std().fillna(0)
    
    # Calculate the trend strength (difference between the 5-day moving average and the current close)
    trend_strength = df['close'].rolling(window=5).mean() - df['close']
    
    # Calculate the trading volume intensity (current volume divided by the 5-day average volume)
    volume_intensity = df['volume'] / df['volume'].rolling(window=5).mean()
    volume_intensity.fillna(1, inplace=True)
    
    # Incorporate seasonality by calculating the average daily return for each month
    monthly_seasonality = df.groupby(df.index.month)['close'].pct_change().groupby(df.index.month).transform('mean').fillna(0)
    
    # Calculate the 50-day moving average and its slope as a long-term trend indicator
    ma_50 = df['close'].rolling(window=50).mean()
    ma_50_slope = ma_50.diff().fillna(0)
    
    # Calculate the 200-day moving average and its slope as a long-term trend indicator
    ma_200 = df['close'].rolling(window=200).mean()
    ma_200_slope = ma_200.diff().fillna(0)
    
    # Factor: Combine daily return, daily movement, distance from the closing price to the midpoint, 
    # average price per share, volatility, trend strength, trading volume intensity, monthly seasonality,
    # and long-term trend indicators
    factor = (
        daily_return * 0.3 +
        daily_movement * 0.15 -
        0.1 * (df['close'] - midpoint) +
        0.1 * avg_price_per_share -
        0.1 * volatility +
        0.1 * trend_strength +
        0.1 * volume_intensity +
        0.1 * monthly_seasonality +
        0.05 * ma_50_slope +
        0.05 * ma_200_slope
    )
    
    return factor
```
