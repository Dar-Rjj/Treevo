```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices, and close and open prices
    price_range = (df['high'] - df['low'])
    daily_return = (df['close'] - df['open'])

    # Determine the percentage change in volume relative to its moving average
    volume_change_ratio = df['volume'] / df['volume'].rolling(window=10).mean() - 1
    
    # Combine both factors with a focus on stocks that have small daily ranges but large relative changes in volumes and positive daily returns
    factor_value = ((price_range + 1e-7)**-1) * daily_return * abs(volume_change_ratio)

    return factor_value
```
