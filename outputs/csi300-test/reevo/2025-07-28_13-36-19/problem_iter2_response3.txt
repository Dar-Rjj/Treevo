```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change
    price_change = df['close'] - df['open']
    
    # Calculate the average price
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the volatility
    volatility = df['high'] - df['low']
    
    # Calculate the momentum using the close price
    momentum = df['close'].pct_change()
    
    # Calculate the weighted average of the price change, average price, and volume
    factor = (price_change * df['volume'] + avg_price * df['amount'] + volatility * df['volume'] + momentum * df['amount']) / (df['volume'] + df['amount'])
    
    return factor
```
