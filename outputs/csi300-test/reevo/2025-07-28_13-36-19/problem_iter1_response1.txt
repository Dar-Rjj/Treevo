```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open prices
    price_diff = df['close'] - df['open']
    
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the weighted average of the price difference and average price with volume
    factor = (price_diff * df['volume'] + avg_price * df['amount']) / (df['volume'] + df['amount'])
    
    return factor
```
