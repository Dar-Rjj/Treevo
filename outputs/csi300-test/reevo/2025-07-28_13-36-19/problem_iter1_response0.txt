```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate average price
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in the average price
    pct_change_avg = avg_price.pct_change().fillna(0)
    
    # Calculate the money flow, which is the average price multiplied by volume
    money_flow = avg_price * df['volume']
    
    # Calculate the difference between current and previous money flow
    delta_money_flow = money_flow.diff().fillna(0)
    
    # Combine the percentage change in the average price and the difference in money flow
    factor = pct_change_avg * delta_money_flow
    
    return factor
```
