```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    daily_return = df['close'].pct_change().fillna(0)
    
    # Calculate the average of high and low as a proxy for the midpoint of the day's trading range
    midpoint = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of amount to volume, which could act as a proxy for the average price per share
    avg_price_per_share = df['amount'] / df['volume']
    
    # Calculate the difference between the close and open prices (daily movement)
    daily_movement = df['close'] - df['open']
    
    # Calculate the volatility (standard deviation of daily returns over a 5-day window)
    volatility = df['close'].rolling(window=5).std().fillna(0)
    
    # Calculate the trend strength (difference between the 5-day moving average and the current close)
    trend_strength = df['close'].rolling(window=5).mean() - df['close']
    
    # Calculate the trading volume intensity (current volume divided by the 5-day average volume)
    volume_intensity = df['volume'] / df['volume'].rolling(window=5).mean()
    volume_intensity.fillna(1, inplace=True)
    
    # Calculate the rolling average of the daily return
    rolling_daily_return = daily_return.rolling(window=5).mean().fillna(0)
    
    # Calculate the rolling average of the daily movement
    rolling_daily_movement = daily_movement.rolling(window=5).mean().fillna(0)
    
    # Calculate the rolling average of the distance from the closing price to the midpoint
    rolling_midpoint_distance = (df['close'] - midpoint).rolling(window=5).mean().fillna(0)
    
    # Calculate the rolling average of the average price per share
    rolling_avg_price_per_share = avg_price_per_share.rolling(window=5).mean().fillna(0)
    
    # Calculate the rolling average of the trend strength
    rolling_trend_strength = trend_strength.rolling(window=5).mean().fillna(0)
    
    # Factor: Combine rolling daily return, rolling daily movement, rolling distance from the closing price to the midpoint, 
    # rolling average price per share, volatility, rolling trend strength, and trading volume intensity
    factor = (rolling_daily_return * 0.3 + 
              rolling_daily_movement * 0.2 - 
              0.1 * rolling_midpoint_distance + 
              0.1 * rolling_avg_price_per_share - 
              0.1 * volatility + 
              0.1 * rolling_trend_strength + 
              0.1 * volume_intensity)
    
    return factor
```
