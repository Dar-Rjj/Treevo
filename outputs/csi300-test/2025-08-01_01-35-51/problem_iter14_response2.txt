```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'].diff(1)
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['close'].diff(5)
    
    # Define dynamic weighting factor based on recent price levels
    def dynamic_weighting_factor(price_diff):
        if price_diff > 0:
            return 1.2  # Higher weight for positive momentum
        else:
            return 0.8  # Lower weight for negative momentum
    
    # Apply dynamic weighting to Short-Term Momentum
    df['weighted_short_term_momentum'] = df['short_term_momentum'].apply(
        lambda x: x * dynamic_weighting_factor(x)
    )
    
    # Apply dynamic weighting to Long-Term Momentum
    df['weighted_long_term_momentum'] = df['long_term_momentum'].apply(
        lambda x: x * dynamic_weighting_factor(x)
    )
    
    # Combine Intraday Return, Adjusted Volatility, and Weighted Momentum
    df['combined_value'] = (
        df['intraday_return'] 
        - df['adjusted_volatility'] 
        + df['weighted_short_term_momentum'] 
        + df['weighted_long_term_momentum']
    )
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = (df['combined_value'] > 0).astype(int)
    
    return df['alpha_factor']
```
