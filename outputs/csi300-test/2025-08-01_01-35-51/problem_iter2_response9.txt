```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High-Low_Range'] = df['high'] - df['low']
    
    # Calculate Range Momentum
    df['Prev_High-Low_Range'] = df['High-Low_Range'].shift(1)
    df['Range_Momentum'] = (df['High-Low_Range'] - df['Prev_High-Low_Range']) * (df['close'] > df['close'].shift(1)).astype(int)

    # Integrate Volatility
    df['HL_Range_20_EMA'] = df['High-Low_Range'].ewm(span=20, adjust=False).mean()
    df['Volatility_Trend'] = df['High-Low_Range'] - df['HL_Range_20_EMA']

    # Refine Volume Trends
    df['Volume_50_SMA'] = df['volume'].rolling(window=50).mean()
    df['Volume_Trend'] = df['volume'] - df['Volume_50_SMA']

    # Combine Factors
    conditions = (df['Range_Momentum'] > 0) & (df['Volatility_Trend'] > 0) & (df['Volume_Trend'] > 0)
    df['Indicator'] = conditions.astype(int)

    return df['Indicator']
```
