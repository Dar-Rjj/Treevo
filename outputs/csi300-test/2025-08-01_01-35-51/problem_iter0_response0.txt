```python
import pandas as pd

def heuristics_v2(market_data: pd.DataFrame) -> pd.Series:
    # Calculate the close-to-open return
    close_to_open_return = (market_data['open'] - market_data['close'].shift(1)) / market_data['close'].shift(1)
    
    # Weight the return by volume
    volume_weighted_return = close_to_open_return * market_data['volume']
    
    return volume_weighted_return
```
