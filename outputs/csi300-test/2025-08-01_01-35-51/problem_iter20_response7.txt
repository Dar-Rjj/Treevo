```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(1)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'] - df['close'].shift(5)
    
    # Incorporate Moving Averages
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    moving_average_crossover = (ma_5 > ma_20).astype(int)
    
    # Analyze Trading Volume Trends
    volume_ma_5 = df['volume'].rolling(window=5).mean()
    volume_ma_20 = df['volume'].rolling(window=20).mean()
    volume_trend = (volume_ma_5 > volume_ma_20).astype(int)
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    combined_value = intraday_return - adjusted_volatility + short_term_momentum + long_term_momentum
    
    # Integrate Moving Average Crossover and Volume Trend
    integrated_value = combined_value + moving_average_crossover + volume_trend
    
    # Evaluate Historical Volatility
    historical_volatility = df['close'].rolling(window=20).std()
    normalized_volatility = historical_volatility / intraday_volatility
    
    # Analyze Amount Traded
    amount_ma_5 = df['amount'].rolling(window=5).mean()
    amount_ma_20 = df['amount'].rolling(window=20).mean()
    amount_trend = (amount_ma_5 > amount_ma_20).astype(int)
    
    # Final Integration
    final_integrated_value = integrated_value + normalized_volatility + amount_trend
    
    # Generate Final Alpha Factor
    alpha_factor = (final_integrated_value > 0).astype(int)
    
    return alpha_factor
```
