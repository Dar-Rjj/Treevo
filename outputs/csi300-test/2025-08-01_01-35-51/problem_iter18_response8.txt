import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Calculate Intraday Price Range
    data['intraday_range'] = data['high'] - data['low']

    # Compute Intraday Return
    data['intraday_return'] = data['close'] - data['open']

    # Determine Intraday Volatility
    data['intraday_volatility'] = (data['high'] - data['low']) / data['close']

    # Adjust Volatility by Volume
    data['adjusted_volatility'] = data['intraday_volatility'] * data['volume']

    # Calculate Short-Term Momentum
    data['short_term_momentum'] = data['close'] - data['close'].shift(1)

    # Calculate Medium-Term Momentum
    data['medium_term_momentum'] = data['close'] - data['close'].shift(3)

    # Calculate Long-Term Momentum
    data['long_term_momentum'] = data['close'] - data['close'].shift(5)

    # Apply Exponential Weights to Momentum
    data['exp_short_term_momentum'] = 0.7 * data['short_term_momentum']
    data['exp_medium_term_momentum'] = 0.2 * data['medium_term_momentum']
    data['exp_long_term_momentum'] = 0.1 * data['long_term_momentum']

    # Calculate Additional Price Transformations
    data['log_return'] = np.log(data['close']) - np.log(data['close'].shift(1))
    data['price_to_volume_ratio'] = data['close'] / data['volume']

    # Combine Intraday Return, Adjusted Volatility, Weighted Momentum, and Additional Transformations
    data['combined_value'] = (
        data['intraday_return'] - 
        data['adjusted_volatility'] + 
        data['exp_short_term_momentum'] + 
        data['exp_medium_term_momentum'] + 
        data['exp_long_term_momentum'] + 
        data['log_return'] - 
        data['price_to_volume_ratio']
    )

    # Generate Final Alpha Factor
    data['alpha_factor'] = (data['combined_value'] > 0).astype(int)

    return data['alpha_factor']
