import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Return
    df['high_low_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Open-Close Return
    df['open_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate True Range
    df['true_range'] = df.apply(lambda row: max(row['high'] - row['low'], 
                                                abs(row['high'] - row['close'].shift(1)), 
                                                abs(row['low'] - row['close'].shift(1))), axis=1)
    
    # Calculate Average True Range over 14 days
    df['avg_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Combine High-Low and Open-Close Returns
    df['combined_returns'] = (df['high_low_return'] + df['open_close_return']) / 2
    
    # Volume-Weighted Movement
    df['volume_weighted_movement'] = df['combined_returns'] * df['volume']
    df['sum_volume_weighted_movement'] = df['volume_weighted_movement'].rolling(window=10).sum()
    
    # Adjust by Volatility
    df['volatility_adjusted_movement'] = df['sum_volume_weighted_movement'] / df['avg_true_range']
    
    # Incorporate Price Trend using MACD
    df['ema_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['macd'] = df['ema_12'] - df['ema_26']
    
    # Final Alpha Score
    df['alpha_score'] = df['volatility_adjusted_movement'] * df['macd']
    
    return df['alpha_score'].dropna()
