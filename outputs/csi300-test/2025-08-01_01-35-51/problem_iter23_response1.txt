import pandas as pd

def heuristics_v2(df):
    # Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Volume-Adjusted Volatility
    volume_adjusted_volatility = intraday_volatility * df['volume']
    
    # Short-Term Momentum
    short_term_momentum = df['close'].diff()
    
    # Long-Term Momentum
    long_term_momentum = df['close'] - df['close'].shift(5)
    
    # Combined Alpha Factor
    combined_alpha_factor = (intraday_return 
                             - volume_adjusted_volatility 
                             + short_term_momentum 
                             + long_term_momentum)
    
    # Final Alpha Factor
    final_alpha_factor = (combined_alpha_factor > 0).astype(int)
    
    return final_alpha_factor
