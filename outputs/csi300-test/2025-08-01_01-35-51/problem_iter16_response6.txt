```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Short-Term Weighted Momentum
    df['short_momentum'] = 0.5 * (df['close'] - df['close'].shift(1))
    
    # Calculate Medium-Term Weighted Momentum
    df['medium_momentum'] = 0.3 * (df['close'] - df['close'].shift(3))
    
    # Calculate Long-Term Weighted Momentum
    df['long_momentum'] = 0.2 * (df['close'] - df['close'].shift(5))
    
    # Calculate Ultra Long-Term Weighted Momentum
    df['ultra_long_momentum'] = 0.1 * (df['close'] - df['close'].shift(10))
    
    # Combine Intraday Return, Adjusted Volatility, and Weighted Momentum
    df['combined_value'] = (
        df['intraday_return'] -
        df['adjusted_volatility'] +
        df['short_momentum'] +
        df['medium_momentum'] +
        df['long_momentum'] +
        df['ultra_long_momentum']
    )
    
    # Consider Dynamic Factor: Price Change Rate
    df['price_change_rate'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine All Factors
    df['combined_value_with_price_change'] = df['combined_value'] + df['price_change_rate']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = (df['combined_value_with_price_change'] > 0).astype(int)
    
    return df['alpha_factor']
```
