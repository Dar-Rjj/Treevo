```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Daily Price Range
    df['daily_range'] = df['high'] - df['low']

    # Average of Recent Daily Price Ranges
    df['avg_recent_range'] = df['daily_range'].rolling(window=n).mean()

    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    
    # Previous Day's Range (True Range)
    df['prev_day_range'] = (df['high'].shift(1) - df['low'].shift(1)).fillna(0)
    
    # Adjust by Previous Day's Range
    df['intraday_momentum'] = df['high_low_diff'] / (df['prev_day_range'] + 1e-6)  # add small constant to avoid division by zero

    # Composite Score
    df['composite_score'] = df['intraday_momentum'] * df['avg_recent_range']

    return df['composite_score']
```
