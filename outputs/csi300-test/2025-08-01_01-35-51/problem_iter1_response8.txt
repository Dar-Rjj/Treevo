import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['prev_day_range'] = (df['close'].shift(1) - df['open'].shift(1)).abs()
    df['intraday_momentum'] = df['high_low_diff'] / df['prev_day_range']

    # Calculate Weighted Average Price Change
    df['intraday_price_move'] = df['close'] - df['open']
    df['volume_averaged_adj'] = df['intraday_price_move'] / df['volume']
    df['weighted_avg_price_change'] = df['volume_averaged_adj'].rolling(window=5).sum()

    # Composite Score
    df['composite_score'] = df['intraday_momentum'] * df['weighted_avg_price_change']

    # Final Alpha Score
    alpha_factor = df['composite_score']

    return alpha_factor
