import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']

    # Compute Intraday Return
    intraday_return = df['close'] - df['open']

    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']

    # Adjust Intraday Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']

    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].diff(1).fillna(0)

    # Calculate Relative Strength
    relative_strength = df['close'] / df['close'].shift(5).fillna(1)

    # Combine Intraday Metrics
    combined_value = (
        intraday_return
        - adjusted_volatility
        + short_term_momentum
    ) * relative_strength

    # Generate Final Alpha Factor
    alpha_factor = (combined_value > 0).astype(int)

    return alpha_factor
