import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']
    
    # Calculate Short-Term Weighted Momentum
    short_term_momentum = (df['close'] - df['close'].shift(1)) * 0.5
    
    # Calculate Medium-Term Weighted Momentum
    medium_term_momentum = (df['close'] - df['close'].shift(3)) * 0.3
    
    # Calculate Long-Term Weighted Momentum
    long_term_momentum = (df['close'] - df['close'].shift(5)) * 0.2
    
    # Combine Intraday Return, Adjusted Volatility, and Weighted Momentum
    combined_value = (intraday_return - adjusted_volatility + 
                      short_term_momentum + medium_term_momentum + long_term_momentum)
    
    # Generate Final Alpha Factor
    alpha_factor = (combined_value > 0).astype(int)
    
    return alpha_factor
