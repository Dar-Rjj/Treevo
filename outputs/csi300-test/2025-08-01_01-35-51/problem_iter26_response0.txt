import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust Intraday Volatility by Volume
    adjusted_intraday_volatility = intraday_volatility * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].diff().fillna(0)
    
    # Combine Key Factors
    combined_factor = intraday_return - adjusted_intraday_volatility + short_term_momentum
    
    # Refine Final Alpha Factor
    refined_factor = combined_factor.apply(lambda x: 1 if x > 0.5 else 0)
    
    return refined_factor
