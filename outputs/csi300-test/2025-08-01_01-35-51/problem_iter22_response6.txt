```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']

    # Compute Intraday Return
    df['intraday_return'] = df['close'] - df['open']

    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']

    # Adjust Volatility by Volume
    df['vol_adj_intraday_volatility'] = df['intraday_volatility'] * df['volume']

    # Calculate Short-Term Momentum (5 days, exponentially weighted)
    short_term_momentum = df['close'].pct_change().ewm(span=5).mean()

    # Calculate Long-Term Momentum (20 days, exponentially weighted)
    long_term_momentum = df['close'].pct_change().ewm(span=20).mean()

    # Integrate Relative Volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    df['relative_volume'] = df['volume'] / avg_volume_5d

    # Additional Volatility Adjustment
    df['adj_volatility'] = df['vol_adj_intraday_volatility'] * df['relative_volume']

    # Combine Intraday Return, Adjusted Volatility, and Momentum
    combined_value = (df['intraday_return'] 
                      - df['adj_volatility'] 
                      + short_term_momentum 
                      + long_term_momentum)

    # Generate Final Alpha Factor
    df['alpha_factor'] = np.where(combined_value > 0, 1, 0)

    return df['alpha_factor']
```
