import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'].diff(1)
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['close'].diff(5)
    
    # Incorporate Weighted Averages
    weights_3_day = [0.5, 0.3, 0.2]
    weights_7_day = [0.15, 0.12, 0.10, 0.09, 0.08, 0.07, 0.06]
    
    df['weighted_avg_close_3_day'] = df['close'].rolling(window=3).apply(lambda x: (x * weights_3_day).sum(), raw=True)
    df['weighted_avg_close_7_day'] = df['close'].rolling(window=7).apply(lambda x: (x * weights_7_day).sum(), raw=True)
    
    # Consider Liquidity Measures
    df['avg_volume_5_day'] = df['volume'].rolling(window=5).mean()
    df['avg_volume_10_day'] = df['volume'].rolling(window=10).mean()
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    df['combined_value'] = (
        df['intraday_return'] 
        - df['adjusted_volatility'] 
        + df['short_term_momentum'] 
        + df['long_term_momentum'] 
        - df['weighted_avg_close_3_day'] 
        + df['weighted_avg_close_7_day']
    )
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = (df['combined_value'] > 0).astype(int)
    
    return df['alpha_factor']
