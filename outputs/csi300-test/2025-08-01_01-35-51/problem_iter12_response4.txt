import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].diff(1)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].diff(20)
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    combined_value = (intraday_return - adjusted_volatility 
                      + short_term_momentum + long_term_momentum)
    
    # Integrate Market Trends
    # Calculate 5-Day Moving Average of Close Price
    ma_5 = df['close'].rolling(window=5).mean()
    # Calculate 20-Day Moving Average of Close Price
    ma_20 = df['close'].rolling(window=20).mean()
    # Calculate Trend Indicator
    trend_indicator = (ma_5 > ma_20).astype(int)
    
    # Generate Final Alpha Factor
    final_alpha_factor = (combined_value * trend_indicator).apply(lambda x: 1 if x > 0 else 0)
    
    return final_alpha_factor
