import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'].diff(1)
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['close'].diff(5)
    
    # Refine Short-Term Momentum with Exponential Weighting
    short_term_ewm = df['close'].ewm(span=6, adjust=False).mean().diff()
    df['refined_short_term_momentum'] = short_term_ewm
    
    # Refine Long-Term Momentum with Moving Average
    long_term_ma = df['close'].rolling(window=20).mean()
    df['refined_long_term_momentum'] = df['close'] - long_term_ma
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    df['combined_value'] = (df['intraday_return'] 
                            - df['adjusted_volatility'] 
                            + df['refined_short_term_momentum'] 
                            + df['refined_long_term_momentum'])
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = df['combined_value'].apply(lambda x: 1 if x > 0 else 0)
    
    return df['alpha_factor']
