import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].diff(1)
    
    # Calculate Medium-Term Momentum
    medium_term_momentum = df['close'].diff(3)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].diff(5)
    
    # Calculate Ultra-Short-Term Momentum
    ultra_short_term_momentum = df['close'].diff(1)
    
    # Calculate Ultra-Long-Term Momentum
    ultra_long_term_momentum = df['close'].diff(20)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['open'] * df['volume'] + df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume']) / (4 * df['volume'])
    
    # Adjust VWAP for Intraday Dynamics
    vwap_adjustment = vwap - vwap.shift(1)
    
    # Integrate Relative Volume
    relative_volume = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Additional Volatility Adjustment
    additional_volatility = adjusted_volatility * relative_volume
    
    # Combine Key Factors
    combined_value = (intraday_return - additional_volatility + 
                      short_term_momentum + medium_term_momentum + 
                      long_term_momentum + ultra_short_term_momentum + 
                      ultra_long_term_momentum + vwap_adjustment)
    
    # Generate Final Alpha Factor
    alpha_factor = (combined_value > 0).astype(int)
    
    # Refine Final Alpha Factor with a threshold
    refined_alpha_factor = (combined_value > 0.5).astype(int)
    
    return refined_alpha_factor
