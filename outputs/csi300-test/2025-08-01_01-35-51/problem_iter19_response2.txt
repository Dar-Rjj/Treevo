```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Relative Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    df['adjusted_volatility'] = df['intraday_volatility'] * (df['volume'] / avg_volume)
    
    # Calculate Exponentially Weighted Short-Term Momentum
    weights = [0.5, 0.3, 0.2, 0.1]
    df['ewm_momentum'] = df['close'].shift(1) * weights[0] + \
                         df['close'].shift(2) * weights[1] + \
                         df['close'].shift(3) * weights[2] + \
                         df['close'].shift(4) * weights[3]
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    df['combined_value'] = df['intraday_return'] - df['adjusted_volatility'] + \
                           df['ewm_momentum'] + df['long_term_momentum']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = (df['combined_value'] > 0).astype(int)
    
    return df['alpha_factor']
```
