import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    adjusted_volatility = df['volume'] * intraday_volatility
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'].shift(-1) - df['open'].shift(1)
    
    # Calculate Short-Term Moving Average
    short_term_ma = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Moving Average
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Calculate Relative Strength
    relative_strength = short_term_ma / long_term_ma
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    combined_value = intraday_return + intraday_momentum - adjusted_volatility
    
    # Incorporate Relative Strength
    combined_value_with_rs = combined_value * relative_strength
    
    # Generate Final Alpha Factor
    alpha_factor = (combined_value_with_rs > 0).astype(int)
    
    return alpha_factor
