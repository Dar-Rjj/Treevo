```python
import pandas as pd

def heuristics_v2(df, n_days=10, m_days=5):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Average Intraday Range over the last N days
    avg_intraday_range = intraday_range.rolling(window=n_days).mean()
    
    # Calculate Close-Open Momentum
    close_open_momentum = df['close'] - df['open']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combine Intraday Range, Momentum, and Volume
    combined_factor = (avg_intraday_range * 
                       close_open_momentum * 
                       volume_change)
    
    # Weighted Moving Average of Combined Factor over the last M days
    wma_combined_factor = combined_factor.rolling(window=m_days).mean()
    
    return wma_combined_factor
```
