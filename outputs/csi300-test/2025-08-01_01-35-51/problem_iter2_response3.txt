import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    df['open_shifted'] = df['open'].shift(1)
    intraday_momentum = (df['close'] - df['open_shifted']) / df['open_shifted']
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    combined_value = intraday_return - adjusted_volatility + intraday_momentum
    
    # Generate Final Alpha Factor
    alpha_factor = (combined_value > 0).astype(int)
    
    return alpha_factor
