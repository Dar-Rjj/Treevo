```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] / df['open']) - 1
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Intraday Volatility by Volume
    df['adjusted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['close'] / df['open'].shift(1)) - 1
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    df['combined_value'] = df['intraday_return'] - df['adjusted_volatility'] + df['intraday_momentum']
    
    # Generate Initial Alpha Factor
    df['initial_alpha_factor'] = df['combined_value'].apply(lambda x: 1 if x > 0 else 0)
    
    # Refine and Simplify for Clearer Signal
    # Calculate Short-Term Moving Average (e.g., 5 days)
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Moving Average (e.g., 20 days)
    df['long_term_ma'] = df['close'].rolling(window=20).mean()
    
    # Integrate Granular Price Data
    # Calculate ATR (Average True Range)
    df['true_range'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Consider External Market Sentiment
    # Assume an external sentiment score is available
    # For this example, we will generate a random sentiment score between -1 and 1
    np.random.seed(0)
    df['sentiment_score'] = np.random.uniform(-1, 1, size=len(df))
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (df['initial_alpha_factor'] 
                                + df['short_term_ma'] 
                                - df['long_term_ma']) * df['atr'] * df['sentiment_score']
    
    return df['final_alpha_factor']
```
