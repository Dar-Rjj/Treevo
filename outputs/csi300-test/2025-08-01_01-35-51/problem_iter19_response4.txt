```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Exponentially Weighted Short-Term Momentum
    df['ewm_momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Integrate Relative Volume
    df['average_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['relative_volume'] = df['volume'] / df['average_volume_5d']
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    df['combined_value'] = (df['intraday_return'] 
                            - df['adjusted_volatility'] 
                            + df['ewm_momentum'])
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_value'] * df['relative_volume']
    df['alpha_factor_indicator'] = (df['alpha_factor'] > 0).astype(int)
    
    return df['alpha_factor_indicator']
```
