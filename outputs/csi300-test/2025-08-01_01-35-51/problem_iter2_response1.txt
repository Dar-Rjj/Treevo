```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['range'] = df['high'] - df['low']
    
    # Calculate Average of Recent Daily Price Ranges (N = 10 days)
    N = 10
    df['avg_range'] = df['range'].rolling(window=N).mean()
    
    # Calculate Exponential Moving Average (EMA) of Daily Range (M = 20 days)
    M = 20
    alpha = 2 / (M + 1)
    df['ema_range'] = df['range'].ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate Relative Strength Indicator (RSI)
    df['price_change'] = df['close'].diff()
    df['gain'] = np.where(df['price_change'] > 0, df['price_change'], 0)
    df['loss'] = np.where(df['price_change'] < 0, -df['price_change'], 0)
    
    avg_gain = df['gain'].rolling(window=N).mean()
    avg_loss = df['loss'].rolling(window=N).mean()
    
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Final Alpha Factor: Normalized EMA of Range and RSI
    df['alpha_factor'] = (df['ema_range'] / df['avg_range']) * (df['rsi'] / 100)
    
    return df['alpha_factor'].dropna()
```
