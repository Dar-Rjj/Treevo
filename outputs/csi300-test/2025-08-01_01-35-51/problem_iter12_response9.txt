import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']

    # Compute Intraday Return
    df['intraday_return'] = df['close'] - df['open']

    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']

    # Smooth Intraday Volatility
    df['smoothed_intraday_volatility'] = df['intraday_volatility'].rolling(window=2).mean()

    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['smoothed_intraday_volatility'] * df['volume']

    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(1)

    # Calculate Medium-Term Momentum
    df['medium_term_momentum'] = df['close'] - df['close'].shift(3)

    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(5)

    # Calculate Weights
    def calculate_weight(series, window=20):
        std_dev = series.rolling(window=window).std()
        max_std_dev = std_dev.rolling(window=window).max()
        return (std_dev / max_std_dev).clip(0, 1)

    df['weight_intraday_return'] = calculate_weight(df['intraday_return'])
    df['weight_adjusted_volatility'] = calculate_weight(df['adjusted_volatility'])
    df['weight_short_term_momentum'] = calculate_weight(df['short_term_momentum'])
    df['weight_medium_term_momentum'] = calculate_weight(df['medium_term_momentum'])
    df['weight_long_term_momentum'] = calculate_weight(df['long_term_momentum'])

    # Apply Dynamic Weights
    df['weighted_intraday_return'] = df['intraday_return'] * df['weight_intraday_return']
    df['weighted_adjusted_volatility'] = df['adjusted_volatility'] * df['weight_adjusted_volatility']
    df['weighted_short_term_momentum'] = df['short_term_momentum'] * df['weight_short_term_momentum']
    df['weighted_medium_term_momentum'] = df['medium_term_momentum'] * df['weight_medium_term_momentum']
    df['weighted_long_term_momentum'] = df['long_term_momentum'] * df['weight_long_term_momentum']

    # Sum the Weighted Values
    df['sum_weighted_values'] = (
        df['weighted_intraday_return'] 
        - df['weighted_adjusted_volatility'] 
        + df['weighted_short_term_momentum'] 
        + df['weighted_medium_term_momentum'] 
        + df['weighted_long_term_momentum']
    )

    # Generate Final Alpha Factor
    df['alpha_factor'] = (df['sum_weighted_values'] > 0).astype(int)

    return df['alpha_factor']

# Example usage:
# alpha_factor = heuristics_v2(your_dataframe)
