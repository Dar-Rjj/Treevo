import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust Intraday Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].diff(1)
    
    # Combine Intraday Return, Adjusted Volatility, and Short-Term Momentum
    combined_value = intraday_return - adjusted_volatility + short_term_momentum
    
    # Evaluate Medium-Term Momentum
    medium_term_momentum = df['close'].diff(3)
    
    # Integrate Medium-Term Momentum into Combined Value
    integrated_value = combined_value + medium_term_momentum
    
    # Generate Final Alpha Factor
    alpha_factor = (integrated_value > 0).astype(int)
    
    return alpha_factor
