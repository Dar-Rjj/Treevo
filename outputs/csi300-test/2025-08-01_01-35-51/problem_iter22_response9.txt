import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['Intraday_Price_Range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['Intraday_Return'] = df['close'] - df['open']
    
    # Determine Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    df['Adjusted_Volatility'] = df['Intraday_Volatility'] * df['volume']
    
    # Calculate Short-Term Momentum
    df['Short_Term_Momentum'] = df['close'].diff(1)
    
    # Calculate Medium-Term Momentum
    df['Medium_Term_Momentum'] = df['close'].diff(3)
    
    # Calculate Long-Term Momentum
    df['Long_Term_Momentum'] = df['close'].diff(5)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    total_volume = df['open'] + df['high'] + df['low'] + df['close']
    df['VWAP'] = (df['open'] * df['volume'] + df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume']) / (4 * df['volume'])
    
    # Adjust VWAP for Intraday Dynamics
    df['VWAP_Adjustment'] = df['VWAP'].diff(1)
    
    # Integrate Relative Volume
    df['Relative_Volume'] = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Additional Volatility Adjustment
    df['Additional_Volatility'] = df['Adjusted_Volatility'] * df['Relative_Volume']
    
    # Combine Intraday Return, Adjusted Volatility, Momentum, and VWAP Adjustment
    df['Combined_Value'] = (df['Intraday_Return'] 
                            - df['Additional_Volatility'] 
                            + df['Short_Term_Momentum'] 
                            + df['Medium_Term_Momentum'] 
                            + df['Long_Term_Momentum'] 
                            + df['VWAP_Adjustment'])
    
    # Generate Final Alpha Factor
    df['Alpha_Factor'] = df['Combined_Value'].apply(lambda x: 1 if x > 0 else 0)
    
    return df['Alpha_Factor']
