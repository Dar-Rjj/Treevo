```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'].diff(1)
    
    # Calculate Medium-Term Momentum
    df['medium_term_momentum'] = df['close'].diff(5)
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['close'].diff(20)
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    combined_value = (
        df['intraday_return'] 
        - df['adjusted_volatility'] 
        + df['short_term_momentum'] 
        + df['medium_term_momentum'] 
        + df['long_term_momentum']
    )
    
    # Refine Combined Value with Weights
    weights = [0.3, 0.2, 0.1, 0.2, 0.2]  # Weights for intraday return, adjusted volatility, short, medium, long term momentum
    refined_combined_value = (
        weights[0] * df['intraday_return'] 
        - weights[1] * df['adjusted_volatility'] 
        + weights[2] * df['short_term_momentum'] 
        + weights[3] * df['medium_term_momentum'] 
        + weights[4] * df['long_term_momentum']
    )
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = (refined_combined_value > 0).astype(int)
    
    return df['alpha_factor']
```
