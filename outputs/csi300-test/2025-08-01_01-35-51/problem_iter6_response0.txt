import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'].shift(-1) - df['open']
    
    # Calculate Short-Term Moving Average (5 days)
    short_term_ma = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Moving Average (20 days)
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Calculate Relative Strength
    relative_strength = short_term_ma / long_term_ma
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    combined_value = intraday_return - adjusted_volatility + intraday_momentum
    
    # Incorporate Relative Strength
    combined_value_rs = combined_value * relative_strength
    
    # Generate Final Alpha Factor
    alpha_factor = (combined_value_rs > 0).astype(int)
    
    return alpha_factor
