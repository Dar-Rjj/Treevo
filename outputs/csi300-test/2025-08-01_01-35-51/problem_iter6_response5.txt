```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']

    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Determine Intraday Volatility using Average True Range over 14 days
    df['true_range'] = df.apply(lambda x: max(x['high'] - x['low'], 
                                              abs(x['high'] - df.loc[x.name - pd.Timedelta(days=1), 'close']), 
                                              abs(df.loc[x.name - pd.Timedelta(days=1), 'close'] - x['low'])), axis=1)
    df['avg_true_range'] = df['true_range'].rolling(window=14).mean()

    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['avg_true_range'] * df['volume']

    # Calculate Short-Term, Medium-Term, and Long-Term Momentum
    df['short_term_momentum'] = df['close'].diff(1)
    df['medium_term_momentum'] = df['close'].diff(5)
    df['long_term_momentum'] = df['close'].diff(20)

    # Combine Intraday Return, Adjusted Volatility, and Momentum
    df['combined_value'] = (df['intraday_return'] - 
                            df['adjusted_volatility'] + 
                            df['short_term_momentum'] + 
                            df['medium_term_momentum'] + 
                            df['long_term_momentum'])

    # Generate Final Alpha Factor
    df['alpha_factor'] = df['combined_value'].apply(lambda x: 1 if x > 0 else 0)

    return df['alpha_factor']
```
