```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'].shift(-1) - df['open']
    
    # Incorporate Recent Price Changes
    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    df['recent_price_change'] = df['close'] - df['5_day_sma']
    
    # Combine Intraday Return, Adjusted Volatility, Intraday Momentum, and Recent Price Change
    df['combined_value'] = df['intraday_return'] - df['adjusted_volatility'] + df['intraday_momentum'] + df['recent_price_change']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = np.where(df['combined_value'] > 0, 1, 0)
    
    return df['alpha_factor']
```
