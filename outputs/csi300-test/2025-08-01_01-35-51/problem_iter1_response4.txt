```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=20):
    # Compute Daily Price Change
    df['price_change'] = df['close'].pct_change()
    
    # Compute Moving Average of Price Changes
    df['moving_avg_price_change'] = df['price_change'].rolling(window=N).mean()
    
    # Subtract Moving Average from Latest Price Change
    df['momentum'] = df['price_change'] - df['moving_avg_price_change']
    
    # Calculate Volatility Score
    df['volatility_score'] = np.log(df['high'] - df['low'])
    
    # Calculate moving average of Volatility Scores over M days
    df['historical_volatility'] = df['volatility_score'].rolling(window=M).mean()
    
    # Compute difference between current and historical volatility
    df['volatility_diff'] = df['volatility_score'] - df['historical_volatility']
    
    # Normalize Momentum by Volatility Difference
    df['adjusted_momentum'] = df.apply(
        lambda row: row['momentum'] * (1 + row['volatility_diff']) if row['momentum'] > 0 else row['momentum'] / (1 - row['volatility_diff']),
        axis=1
    )
    
    # Output the combined score
    alpha_factor = df['adjusted_momentum']
    
    return alpha_factor
```
