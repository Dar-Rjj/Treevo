import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    high_low_range = df['high'] - df['low']
    close_high_range = abs(df['close'] - df['high'])
    close_low_range = abs(df['close'] - df['low'])
    df['intraday_volatility'] = high_low_range.combine(close_high_range, max).combine(close_low_range, max)
    
    # Adjust Volatility by Volume
    df['weighted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'].shift(-1) - df['open'].shift(1)
    
    # Incorporate Market Sentiment
    df['avg_daily_volume'] = df['volume'].rolling(window=10).mean()
    df['relative_volume'] = df['volume'] / df['avg_daily_volume']
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    combined_value = df['intraday_return'] - df['weighted_volatility'] + df['intraday_momentum']
    
    # Adjust Combined Value by Relative Volume
    df['adjusted_combined_value'] = combined_value * df['relative_volume']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = df['adjusted_combined_value'].apply(lambda x: 1 if x > 0 else 0)
    
    return df['alpha_factor']
