```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Intraday Volatility by Volume
    df['adjusted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'].diff(1)
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['close'].diff(5)
    
    # Integrate Market Liquidity
    df['average_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['liquidity_indicator'] = df['volume'] / df['average_volume_5d']
    
    # Combine Intraday Return, Adjusted Volatility, Momentums, and Liquidity
    df['combined_value'] = (
        df['intraday_return'] 
        - df['adjusted_volatility'] 
        + df['short_term_momentum'] 
        + df['long_term_momentum']
    ) * df['liquidity_indicator']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = df['combined_value'].apply(lambda x: 1 if x > 0 else 0)
    
    return df['alpha_factor']
```
