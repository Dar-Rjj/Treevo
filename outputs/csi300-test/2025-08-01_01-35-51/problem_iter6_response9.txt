```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = df[['high', 'low', 'close']].std(axis=1)
    
    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'] - df['open'].shift(1)
    
    # Incorporate Recent Price Changes
    df['5_day_sma_close'] = df['close'].rolling(window=5).mean()
    df['recent_price_change'] = df['close'] - df['5_day_sma_close']
    
    # Combine Intraday Return, Adjusted Volatility, Intraday Momentum, and Recent Price Change
    df['combined_value'] = (df['intraday_return'] 
                            - df['adjusted_volatility'] 
                            + df['intraday_momentum'] 
                            + df['recent_price_change'])
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = (df['combined_value'] > 0).astype(int)
    
    return df['alpha_factor']
```
