```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Return
    high_low_return = df['high'] - df['low']
    
    # Calculate Open-Close Return
    open_close_return = df['close'] - df['open']
    
    # Combine High-Low and Open-Close Returns
    combined_movement = (high_low_return + open_close_return) / 2
    
    # Calculate True Range
    df['previous_close'] = df['close'].shift(1)
    true_range = df[['high', 'low', 'previous_close']].apply(
        lambda x: max(x['high'] - x['low'], 
                      abs(x['high'] - x['previous_close']), 
                      abs(x['low'] - x['previous_close'])), 
        axis=1
    )
    
    # Calculate Average True Range over 14 days
    average_true_range = true_range.rolling(window=14).mean()
    
    # Volume-Weighted Movement
    volume_weighted_movement = combined_movement * df['volume']
    volume_weighted_movement_sum = volume_weighted_movement.rolling(window=10).sum()
    
    # Adjust by Volatility
    final_alpha_score = volume_weighted_movement_sum / average_true_range
    
    return final_alpha_score
```
