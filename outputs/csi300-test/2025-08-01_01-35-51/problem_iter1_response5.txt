```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['daily_range'] = df['high'] - df['low']
    
    # Define N (e.g., 10 days)
    N = 10
    
    # Average of Recent Daily Price Ranges
    df['avg_daily_range'] = df['daily_range'].rolling(window=N).mean()
    
    # The factor is the average daily price range
    return df['avg_daily_range']
```
