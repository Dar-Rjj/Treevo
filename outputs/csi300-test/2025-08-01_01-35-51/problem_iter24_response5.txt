import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Intraday Volatility by Volume
    df['adjusted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'].diff()
    
    # Combine Key Factors
    df['combined_factors'] = df['intraday_return'] - df['adjusted_intraday_volatility'] + df['short_term_momentum']
    
    # Refine Final Alpha Factor
    df['refined_factor'] = df['combined_factors'].apply(lambda x: 1 if x > 0.5 else 0)
    
    # Generate Final Alpha Factor
    alpha_factor = df['refined_factor']
    
    return alpha_factor
