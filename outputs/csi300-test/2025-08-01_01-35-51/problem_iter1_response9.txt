import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Range Momentum
    df['range_momentum'] = (df['high_low_range'] - df['high_low_range'].shift(1)) * (df['close'] > df['close'].shift(1)).astype(int)
    
    # Apply Volume Filter
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['filtered_momentum'] = df['range_momentum'] * (df['volume'] > df['avg_volume']).astype(int)
    
    return df['filtered_momentum']
