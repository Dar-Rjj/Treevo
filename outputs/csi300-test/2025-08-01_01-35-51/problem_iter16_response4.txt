import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].diff(1)
    
    # Calculate Medium-Term Momentum
    medium_term_momentum = df['close'].diff(5)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].diff(20)
    
    # Weighted Momentum Combination
    weighted_momentum = 0.4 * short_term_momentum + 0.3 * medium_term_momentum + 0.3 * long_term_momentum
    
    # Combine Intraday Return, Adjusted Volatility, and Weighted Momentum
    combined_value = intraday_return - adjusted_volatility + weighted_momentum
    
    # Generate Final Alpha Factor
    alpha_factor = (combined_value > 0).astype(int)
    
    return alpha_factor
