import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust Intraday Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'].diff(1) / df['close'].shift(1)).fillna(0)
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'].diff(5) / df['close'].shift(5)).fillna(0)
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    combined_value = (
        intraday_return
        - adjusted_volatility
        + short_term_momentum
        + long_term_momentum
    )
    
    # Generate Final Alpha Factor
    alpha_factor = (combined_value > 0).astype(int)
    
    return alpha_factor
