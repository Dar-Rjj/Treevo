```python
import pandas as pd

def heuristics_v2(df, window=10, atr_window=14):
    # Calculate High-Low Return
    high_low_return = df['high'] - df['low']
    
    # Calculate Open-Close Return
    open_close_return = df['close'] - df['open']
    
    # Combine High-Low and Open-Close Returns
    combined_movements = (high_low_return + open_close_return) / 2
    
    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    true_range = df[['high', 'low', 'prev_close']].apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - x['prev_close']), abs(x['low'] - x['prev_close'])), axis=1)
    
    # Calculate Average True Range
    average_true_range = true_range.rolling(window=atr_window).mean()
    
    # Volume-Weighted Movement
    volume_weighted_movement = combined_movements * df['volume']
    volume_weighted_sum = volume_weighted_movement.rolling(window=window).sum()
    volume_weighted_adjusted = volume_weighted_sum / average_true_range
    
    # Amount-Weighted Movement
    amount_weighted_movement = combined_movements * df['amount']
    amount_weighted_sum = amount_weighted_movement.rolling(window=window).sum()
    amount_weighted_adjusted = amount_weighted_sum / average_true_range
    
    # Aggregate Dynamic Weights
    dynamic_weights = (volume_weighted_adjusted + amount_weighted_adjusted) / 2
    
    # Final Alpha Score
    alpha_factor = dynamic_weights
    
    return alpha_factor
```
