import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'] - df['open'].shift(1)
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    df['combined_value'] = df['intraday_return'] - df['adjusted_volatility'] + df['intraday_momentum']
    
    # Generate Short-Term Moving Average
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    
    # Generate Long-Term Moving Average
    df['long_term_ma'] = df['close'].rolling(window=20).mean()
    
    # Dynamic Weighting
    df['ma_diff'] = df['short_term_ma'] - df['long_term_ma']
    df['dynamic_weight'] = (df['short_term_ma'] - df['long_term_ma']) / df['close']
    df['weighted_combined_value'] = df['combined_value'] * (1 + df['dynamic_weight'])
    
    # Integrate Market Sentiment (For simplicity, we assume a dummy sentiment factor here)
    df['sentiment_factor'] = 1  # Replace with actual sentiment data if available
    df['integrated_value'] = df['weighted_combined_value'] * df['sentiment_factor']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = df['integrated_value'].apply(lambda x: 1 if x > 0 else 0)
    
    return df['alpha_factor']
