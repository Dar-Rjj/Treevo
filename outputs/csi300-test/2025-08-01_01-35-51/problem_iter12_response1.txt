```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Smooth Intraday Volatility
    smoothed_intraday_volatility = intraday_volatility.rolling(window=2).mean()
    
    # Adjust Volatility by Volume
    adjusted_volatility = smoothed_intraday_volatility * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].diff(periods=1)
    
    # Calculate Medium-Term Momentum
    medium_term_momentum = df['close'].diff(periods=3)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].diff(periods=5)
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    combined_value = (intraday_return - adjusted_volatility + 
                      short_term_momentum + medium_term_momentum + 
                      long_term_momentum)
    
    # Generate Final Alpha Factor
    alpha_factor = (combined_value > 0).astype(int)
    
    return alpha_factor
```
