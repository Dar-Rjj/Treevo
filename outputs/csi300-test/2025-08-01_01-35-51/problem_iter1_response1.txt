```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the short-term (5 days) and long-term (20 days) EMAs
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()

    # Determine the trend direction
    df['trend_direction'] = np.where(df['short_ema'] > df['long_ema'], 1, -1)

    # Measure the trend strength
    df['trend_strength'] = (df['short_ema'] - df['long_ema']) * df['trend_direction']

    # Return the factor values
    return df['trend_strength']
```
