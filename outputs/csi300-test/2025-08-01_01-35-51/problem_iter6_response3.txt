import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['volume'] * df['intraday_volatility']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'].shift(-1) - df['open']
    
    # Calculate Short-Term Moving Average
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Moving Average
    df['long_term_ma'] = df['close'].rolling(window=20).mean()
    
    # Calculate Relative Strength
    df['relative_strength'] = df['short_term_ma'] / df['long_term_ma']
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    df['combined_intraday_value'] = df['intraday_return'] + df['intraday_momentum'] - df['adjusted_volatility']
    
    # Incorporate Relative Strength
    df['integrated_value'] = df['combined_intraday_value'] * df['relative_strength']
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate Medium-Term Momentum
    df['medium_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Adjust Short-Term Momentum by Volume
    df['adjusted_short_term_momentum'] = df['short_term_momentum'] * df['volume']
    
    # Adjust Medium-Term Momentum by Volume
    df['adjusted_medium_term_momentum'] = df['medium_term_momentum'] * df['volume']
    
    # Combine All Momentum Factors
    df['combined_momentum_factors'] = df['adjusted_short_term_momentum'] + df['adjusted_medium_term_momentum']
    
    # Integrate All Components
    df['final_integrated_value'] = df['integrated_value'] + df['combined_momentum_factors']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = (df['final_integrated_value'] > 0).astype(int)
    
    return df['alpha_factor']
