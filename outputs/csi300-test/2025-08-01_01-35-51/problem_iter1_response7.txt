```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on intraday momentum and volume-adjusted volatility.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series representing the factor values indexed by (date).
    """
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']

    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']

    # Adjust Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']

    # Combine Intraday Return and Adjusted Volatility
    alpha_factor = intraday_return - adjusted_volatility

    return alpha_factor
```
