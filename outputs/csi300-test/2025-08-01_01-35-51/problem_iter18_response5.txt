```python
import pandas as pd
from statsmodels.tsa.api import ExponentialSmoothing

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].diff(1)
    
    # Calculate Medium-Term Momentum
    medium_term_momentum = df['close'].diff(3)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].diff(5)
    
    # Exponentially Smooth Short-Term Momentum
    smoothed_short_term_momentum = ExponentialSmoothing(short_term_momentum, trend=None, seasonal=None, initialization_method='estimated').fit(smoothing_level=0.7).fittedvalues
    
    # Exponentially Smooth Medium-Term Momentum
    smoothed_medium_term_momentum = ExponentialSmoothing(medium_term_momentum, trend=None, seasonal=None, initialization_method='estimated').fit(smoothing_level=0.5).fittedvalues
    
    # Exponentially Smooth Long-Term Momentum
    smoothed_long_term_momentum = ExponentialSmoothing(long_term_momentum, trend=None, seasonal=None, initialization_method='estimated').fit(smoothing_level=0.3).fittedvalues
    
    # Combine Intraday Return, Adjusted Volatility, and Smoothed Momentum
    combined_value = (
        intraday_return
        - adjusted_volatility
        + smoothed_short_term_momentum
        + smoothed_medium_term_momentum
        + smoothed_long_term_momentum
    )
    
    # Generate Final Alpha Factor
    alpha_factor = (combined_value > 0).astype(int)
    
    return alpha_factor
```
