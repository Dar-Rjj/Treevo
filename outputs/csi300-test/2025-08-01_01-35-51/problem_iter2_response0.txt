```python
import pandas as pd

def heuristics_v2(df, n=10, m=5, p=10):
    # Calculate Daily Price Range
    df['daily_range'] = df['high'] - df['low']
    
    # Average of Recent Daily Price Ranges (Last N Days)
    recent_ranges = df['daily_range'].rolling(window=n).mean()
    
    # Calculate Short-Term Implied Volatility (Last M Days)
    short_term_volatility = df['daily_range'].rolling(window=m).mean()
    
    # Calculate Volume Weighted Average Price (VWAP) for Last P Days
    vwap = (df['close'] * df['volume']).rolling(window=p).sum() / df['volume'].rolling(window=p).sum()
    
    # Calculate Volume Momentum
    volume_momentum = (vwap - df['close']) / df['close']
    
    # Integrate all components to create the alpha factor
    alpha_factor = (recent_ranges + short_term_volatility + volume_momentum) / 3
    
    return alpha_factor
```
