import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Average of Recent Daily Price Ranges
    df['avg_price_range'] = df['price_range'].rolling(window=n).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Average Volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Adjust Average Volatility by Volume
    df['adjusted_volatility'] = df['avg_price_range'] * df['avg_volume']
    
    # Combine Intraday Return and Adjusted Average Volatility
    df['alpha_factor'] = df['intraday_return'] - df['adjusted_volatility']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']

# Example usage:
# Assuming df is a pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
