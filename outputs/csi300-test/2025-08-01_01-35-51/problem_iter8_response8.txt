import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    high_low_range = df['high'] - df['low']
    close_high_range = abs(df['close'] - df['high'])
    close_low_range = abs(df['close'] - df['low'])
    df['intraday_volatility'] = high_low_range.combine(close_high_range, max).combine(close_low_range, max)
    
    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'].shift(-1) - df['open']
    
    # Incorporate Market Sentiment
    avg_volume = df['volume'].rolling(window=10).mean()
    df['relative_volume'] = df['volume'] / avg_volume
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    combined_value = df['intraday_return'] - df['adjusted_volatility'] + df['intraday_momentum']
    
    # Adjust Combined Value by Relative Volume
    adjusted_combined_value = combined_value * df['relative_volume']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = (adjusted_combined_value > 0).astype(int)
    
    return df['alpha_factor']
