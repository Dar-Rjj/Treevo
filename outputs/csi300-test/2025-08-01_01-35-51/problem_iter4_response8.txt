import pandas as pd

def heuristics_v2(df, N=14):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    df['prev_close'] = df['close'].shift(1)
    true_range = (df[['high', 'low', 'prev_close']]
                  .max(axis=1) - 
                  df[['high', 'low', 'prev_close']]
                  .min(axis=1))
    average_true_range = true_range.rolling(window=N).mean()
    
    # Adjust Volatility by Volume
    adjusted_volatility = average_true_range * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open'].shift(1)
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    combined_value = (intraday_return - adjusted_volatility + intraday_momentum)
    
    # Generate Final Alpha Factor
    alpha_factor = (combined_value > 0).astype(int)
    
    return alpha_factor
