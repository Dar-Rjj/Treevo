```python
def heuristics_v2(data):
    # Calculate Close-High Distance
    close_high_distance = data['high'] - data['close']
    
    # Calculate Open-Low Distance
    open_low_distance = data['open'] - data['low']
    
    # Combine Distances
    combined_distance = close_high_distance - open_low_distance
    
    # Weight by Volume Change
    volume_change = data['volume'] - data['volume'].shift(1)
    weighted_distance = combined_distance * volume_change
    
    return weighted_distance
```
