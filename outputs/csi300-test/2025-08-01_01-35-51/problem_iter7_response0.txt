```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Intraday Volatility by Volume
    df['adjusted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    df['combined_value'] = df['intraday_return'] - df['adjusted_volatility'] + df['intraday_momentum']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = np.where(df['combined_value'] > 0, 1, 0)
    
    # Refine and Simplify for Clearer Signal
    # Calculate Short-Term Moving Average (e.g., 5 days)
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Moving Average (e.g., 20 days)
    df['long_term_ma'] = df['close'].rolling(window=20).mean()
    
    # Integrate Granular Price Data
    # Calculate ATR (Average True Range)
    df['tr1'] = abs(df['high'] - df['low'])
    df['tr2'] = abs(df['high'] - df['close'].shift())
    df['tr3'] = abs(df['low'] - df['close'].shift())
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Consider External Market Sentiment (if available)
    # For the purpose of this example, we assume an external sentiment score is available in the dataframe
    if 'sentiment_score' in df.columns:
        df['final_alpha_factor'] = df['alpha_factor'] * (1 + df['sentiment_score'] * 0.5)
    else:
        df['final_alpha_factor'] = df['alpha_factor']
    
    return df['final_alpha_factor']
```
