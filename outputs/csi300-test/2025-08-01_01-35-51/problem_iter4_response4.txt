import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    df['previous_close'] = df['close'].shift(1)
    df['true_range'] = df.apply(lambda row: max(row['high'] - row['low'], 
                                                row['high'] - row['previous_close'], 
                                                row['previous_close'] - row['low']), axis=1)
    
    M = 20  # Define M
    alpha = 2 / (M + 1)  # Smoothing Factor
    
    # Calculate Exponential Moving Average (EMA) of True Range
    df['ema_true_range'] = df['true_range'].ewm(alpha=alpha, adjust=False).mean()
    
    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['ema_true_range'] * df['volume']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'] - df['open'].shift(1)
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    df['combined_value'] = df['intraday_return'] - df['adjusted_volatility'] + df['intraday_momentum']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = (df['combined_value'] > 0).astype(int)
    
    return df['alpha_factor']
