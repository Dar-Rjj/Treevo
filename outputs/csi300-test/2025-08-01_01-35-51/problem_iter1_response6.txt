import pandas as pd

def heuristics_v2(df):
    # Compute 10-day Simple Moving Average (SMA) of Close prices
    df['10_day_SMA'] = df['close'].rolling(window=10).mean()
    
    # Calculate Daily Price Momentum
    df['momentum'] = df['close'] - df['10_day_SMA']
    
    # Calculate 20-day Median Volume
    df['20_day_median_volume'] = df['volume'].rolling(window=20).median()
    
    # Identify Volume Spike
    df['volume_spike'] = df['volume'] > 2 * df['20_day_median_volume']
    
    # Combine Momentum and Spike
    def combine_momentum_and_spike(row):
        if row['momentum'] > 0 and row['volume_spike']:
            return 1
        else:
            return 0
    
    df['indicator'] = df.apply(combine_momentum_and_spike, axis=1)
    
    # Return the indicator as a pandas Series
    return df['indicator']
