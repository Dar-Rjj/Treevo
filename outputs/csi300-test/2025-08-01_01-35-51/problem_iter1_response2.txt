```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate Intraday Open-Close Difference
    open_close_diff = df['close'] - df['open']
    
    # Combine Differences by averaging
    combined_diff = (high_low_diff + open_close_diff) / 2
    
    # Get the sign of the combined differences (negative indicates reversal)
    intraday_reversal = combined_diff.apply(lambda x: -1 if x < 0 else 1 if x > 0 else 0)
    
    # Calculate Volume Momentum
    volume_change = df['volume'].diff().fillna(0)  # Fill first value with 0 to handle the first NaN
    
    # Multiply Volume Change by Intraday Reversal
    factor_value = intraday_reversal * volume_change
    
    return factor_value
```
