import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    high_low_range = df['high'] - df['low']
    close_high_range = abs(df['close'] - df['high'])
    close_low_range = abs(df['close'] - df['low'])
    df['intraday_volatility'] = np.max([high_low_range, close_high_range, close_low_range], axis=0)
    
    # Adjust Volatility by Volume
    df['vol_adj_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'].shift(-1) - df['open'].shift(1)
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    df['combined_value'] = (
        df['intraday_return'] 
        - df['vol_adj_intraday_volatility'] 
        + df['intraday_momentum']
        + df['short_term_momentum']
        + df['long_term_momentum']
    )
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = (df['combined_value'] > 0).astype(int)
    
    return df['alpha_factor']
