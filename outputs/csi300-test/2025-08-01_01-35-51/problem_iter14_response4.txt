import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].diff(1)
    
    # Calculate Medium-Term Momentum
    medium_term_momentum = df['close'].diff(3)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].diff(5)
    
    # Refine Momentum with Exponential Weights
    weighted_short_term_momentum = short_term_momentum * 0.7
    weighted_medium_term_momentum = medium_term_momentum * 0.2
    weighted_long_term_momentum = long_term_momentum * 0.1
    
    # Combine Intraday Return, Adjusted Volatility, and Weighted Momentum
    combined_value = (
        intraday_return
        - adjusted_volatility
        + weighted_short_term_momentum
        + weighted_medium_term_momentum
        + weighted_long_term_momentum
    )
    
    # Generate Final Alpha Factor
    alpha_factor = (combined_value > 0).astype(int)
    
    return alpha_factor
