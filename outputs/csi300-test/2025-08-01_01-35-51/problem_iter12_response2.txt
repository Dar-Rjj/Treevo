import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].diff(1)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].diff(5)
    
    # Incorporate Market Sentiment
    average_daily_volume = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / average_daily_volume
    volume_trend = df['volume'].diff(1)
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    combined_value = (intraday_return 
                      - adjusted_volatility 
                      + short_term_momentum 
                      + long_term_momentum)
    
    # Adjust Combined Value by Relative Volume and Volume Trend
    adjusted_combined_value = (combined_value * relative_volume) + volume_trend
    
    # Generate Final Alpha Factor
    alpha_factor = (adjusted_combined_value > 0).astype(int)
    
    return alpha_factor
