```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Cumulative Volume-Weighted Intraday Momentum.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       and index as dates.
    
    Returns:
    pd.Series: Series of the factor values, indexed by date.
    """
    # Calculate High-Low Return
    high_low_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low and Open-Close Returns
    combined_movement = (high_low_return + open_close_return) / 2
    
    # Weight Averaged Movement by Volume
    volume_weighted_movement = combined_movement * df['volume']
    
    # Sum Over a Specified Time Window (e.g., 10 days)
    rolling_window = 10
    cumulative_volume_weighted_momentum = volume_weighted_movement.rolling(window=rolling_window).sum()
    
    return cumulative_volume_weighted_momentum
```
