```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Determine Intraday Volatility
    high_low_range = df['high'] - df['low']
    close_high_range = np.abs(df['close'] - df['high'])
    close_low_range = np.abs(df['close'] - df['low'])
    intraday_volatility = pd.concat([high_low_range, close_high_range, close_low_range], axis=1).max(axis=1)
    
    # Adjust Volatility by Volume
    adjusted_volatility = intraday_volatility * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].diff(1)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].diff(5)
    
    # Combine Intraday Return, Adjusted Volatility, and Momentum
    combined_value = intraday_return - adjusted_volatility + short_term_momentum + long_term_momentum
    
    # Generate Final Alpha Factor
    alpha_factor = (combined_value > 0).astype(int)
    
    return alpha_factor
```
