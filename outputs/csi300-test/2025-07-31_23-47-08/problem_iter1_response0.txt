```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Overnight Return
    df['overnight_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Compute Rolling Average of Intraday Returns
    intraday_rolling_avg = df['intraday_return'].rolling(window=5).mean()
    
    # Compute Rolling Average of Overnight Returns
    overnight_rolling_avg = df['overnight_return'].rolling(window=5).mean()
    
    # Calculate Intraday Momentum
    intraday_momentum = intraday_rolling_avg * 1.2
    
    # Calculate Overnight Reversal
    overnight_reversal = overnight_rolling_avg * -0.8
    
    # Combine Intraday Momentum and Overnight Reversal
    alpha_factor = intraday_momentum + overnight_reversal
    
    return alpha_factor
```
