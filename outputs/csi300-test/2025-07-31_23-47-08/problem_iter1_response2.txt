```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Determine lookback period
    lookback_period = 10
    
    # Compute Exponential Moving Average of Close Price
    decay_factor = 2 / (lookback_period + 1)
    ema_close = df['close'].ewm(alpha=decay_factor, adjust=False).mean()
    
    # Calculate Price Momentum
    price_momentum = (df['close'] - ema_close) / ema_close
    
    # Compute Ratio of Current Volume to Average Volume
    avg_volume = df['volume'].rolling(window=lookback_period).mean()
    volume_change = df['volume'] / avg_volume
    
    # Combine Price Momentum and Volume Change
    composite_factor = price_momentum * volume_change - 0  # Baseline is zero
    
    return composite_factor
```
