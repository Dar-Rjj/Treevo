import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = df['Intraday_Range'] / df['close']
    
    # Calculate Overnight Return
    df['Previous_Close'] = df['close'].shift(1)
    df['Overnight_Return'] = (df['open'] - df['Previous_Close']) / df['Previous_Close']
    
    # Combine Intraday and Overnight Returns
    df['Combined_Return'] = df['Intraday_Return'] + df['Overnight_Return']
    
    # Compute Volume Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume_Weighted_Price'] = df['Typical_Price'] * df['volume']
    df['Total_Volume_Weighted_Price'] = df['Volume_Weighted_Price'].rolling(window=1).sum()
    df['Total_Volume'] = df['volume'].rolling(window=1).sum()
    df['VWAP'] = df['Total_Volume_Weighted_Price'] / df['Total_Volume']
    
    # Calculate VWAP Reversal Indicator
    df['Reversal_Indicator'] = (df['VWAP'] > df['close']).astype(int) * 2 - 1
    
    # Incorporate Short-Term Price Movement
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['Short_Term_Price_Movement'] = df['close'] - df['5_day_MA']
    
    # Integrate Reversal Indicator with Combined Return and Short-Term Price Movement
    df['Final_Alpha_Factor'] = df['Combined_Return'] * df['Reversal_Indicator'] + df['Short_Term_Price_Movement']
    
    return df['Final_Alpha_Factor']
