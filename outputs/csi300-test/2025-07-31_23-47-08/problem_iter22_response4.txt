```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = df['Intraday_Range'] / df['close']

    # Calculate Overnight Return
    df['Overnight_Return'] = (df['open'].shift(-1) - df['close']) / df['close']

    # Incorporate Transaction Amount
    df['Amount-Weighted_Intraday_Return'] = df['Intraday_Return'] * df['amount']
    df['Amount-Weighted_Overnight_Return'] = df['Overnight_Return'] * df['amount']
    
    total_amount = df['amount'].sum()
    amount_weighted_intraday_return = (df['Amount-Weighted_Intraday_Return']).sum() / total_amount
    amount_weighted_overnight_return = (df['Amount-Weighted_Overnight_Return']).sum() / total_amount
    amount_weighted_combined_return = amount_weighted_intraday_return + amount_weighted_overnight_return

    # Combine Intraday and Overnight Returns
    df['Combined_Return'] = df['Intraday_Return'] + df['Overnight_Return']

    # Compute Volume-Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume-Weighted_Typical_Price'] = df['Typical_Price'] * df['volume']
    vwap = df['Volume-Weighted_Typical_Price'].sum() / df['volume'].sum()

    # Calculate VWAP Reversal Indicator
    df['Reversal_Indicator'] = np.where(vwap > df['close'], 1, -1)

    # Integrate Reversal Indicator with Amount-Weighted Combined Return
    df['Integrated_Reversal_Return'] = amount_weighted_combined_return * df['Reversal_Indicator']

    # Calculate Volume-Weighted Return
    df['Return'] = (df['close'] - df['open']) / df['open']
    df['Volume-Weighted_Return'] = df['Return'] * df['volume']
    volume_weighted_return = df['Volume-Weighted_Return'].sum() / df['volume'].sum()

    # Integrate Volume-Weighted Return with Integrated Reversal Return
    df['Initial_Alpha_Factor'] = df['Integrated_Reversal_Return'] + volume_weighted_return

    # Calculate Rolling Averages
    df['Short_Term_Rolling_Average'] = df['close'].rolling(window=5).mean()
    df['Long_Term_Rolling_Average'] = df['close'].rolling(window=20).mean()

    # Calculate Dynamic Volatility
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Short_Term_Volatility'] = df['Daily_Return'].rolling(window=10).std()
    df['Medium_Term_Volatility'] = df['Daily_Return'].rolling(window=20).std()
    df['Long_Term_Volatility'] = df['Daily_Return'].rolling(window=50).std()

    # Adjust for Volatility
    df['Short_Term_Adjusted_Alpha_Factor'] = df['Initial_Alpha_Factor'] / df['Short_Term_Volatility']
    df['Medium_Term_Adjusted_Alpha_Factor'] = df['Initial_Alpha_Factor'] / df['Medium_Term_Volatility']
    df['Long_Term_Adjusted_Alpha_Factor'] = df['Initial_Alpha_Factor'] / df['Long_Term_Volatility']

    # Integrate Volatility-Adjusted Alpha Factors
    df['Final_Alpha_Factor'] = (df['Short_Term_Adjusted_Alpha_Factor'] +
                                df['Medium_Term_Adjusted_Alpha_Factor'] +
                                df['Long_Term_Adjusted_Alpha_Factor'])

    return df['Final_Alpha_Factor']
```
