```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_return'] = df['intraday_range'] / df['close']

    # Calculate Overnight Return
    df['overnight_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Intraday and Overnight Returns
    df['combined_return'] = df['intraday_return'] + df['overnight_return']

    # Compute Volume-Weighted Return
    df['volume_weighted_return'] = (df['close'] - df['open']) / df['open'] * df['volume']
    total_volume = df['volume'].sum()
    volume_weighted_return = df['volume_weighted_return'].sum() / total_volume
    df['volume_weighted_return'] = volume_weighted_return

    # Integrate Volume-Weighted Return with Combined Return
    df['integrated_return'] = df['combined_return'] + df['volume_weighted_return']

    # Calculate VWAP
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['vwap'] = (df['typical_price'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Calculate VWAP Reversal Indicator
    df['reversal_indicator'] = (df['vwap'] > df['close']).astype(int) * 2 - 1

    # Integrate Reversal Indicator with Integrated Return
    df['final_alpha_factor'] = df['integrated_return'] * df['reversal_indicator']

    return df['final_alpha_factor']
```
