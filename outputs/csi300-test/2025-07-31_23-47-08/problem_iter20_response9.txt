```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = df['Intraday_Range'] / df['close']

    # Calculate Overnight Return
    df['Overnight_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Intraday and Overnight Returns
    df['Combined_Return'] = df['Intraday_Return'] + df['Overnight_Return']

    # Compute Volume-Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume_Weighted_Typical_Price'] = df['Typical_Price'] * df['volume']
    total_volume = df['volume'].sum()
    vwap = df['Volume_Weighted_Typical_Price'].sum() / total_volume
    df['VWAP'] = vwap

    # Calculate VWAP Reversal Indicator
    df['Reversal_Indicator'] = df.apply(lambda row: 1 if row['VWAP'] > row['close'] else -1, axis=1)

    # Integrate Reversal Indicator with Combined Return
    df['Integrated_Reversal_Return'] = df['Combined_Return'] * df['Reversal_Indicator']

    # Calculate Volume-Weighted Return
    df['Return'] = (df['close'] - df['open']) / df['open']
    df['Volume_Weighted_Return'] = df['Return'] * df['volume']
    volume_weighted_return = df['Volume_Weighted_Return'].sum() / total_volume
    df['Volume_Weighted_Return'] = volume_weighted_return

    # Integrate Volume-Weighted Return with Integrated Reversal Return
    df['Initial_Alpha_Factor'] = df['Integrated_Reversal_Return'] + df['Volume_Weighted_Return']

    # Calculate Rolling Averages
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['20_day_MA'] = df['close'].rolling(window=20).mean()

    # Calculate Volatility
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['20_day_Volatility'] = df['Daily_Return'].rolling(window=20).std()

    # Adjust for Volatility
    df['Volatility_Adjusted_Alpha_Factor'] = df['Initial_Alpha_Factor'] / df['20_day_Volatility']

    # Incorporate Transaction Amount
    df['Amount_Weighted_Return'] = (df['close'] - df['open']) / df['open'] * df['amount']
    total_amount = df['amount'].sum()
    amount_weighted_return = df['Amount_Weighted_Return'].sum() / total_amount
    df['Amount_Weighted_Return'] = amount_weighted_return

    # Integrate Amount-Weighted Return with Volatility-Adjusted Alpha Factor
    df['Final_Alpha_Factor'] = df['Volatility_Adjusted_Alpha_Factor'] + df['Amount_Weighted_Return']

    # Incorporate Multi-Term Volatility
    df['5_day_Volatility'] = df['Daily_Return'].rolling(window=5).std()
    df['60_day_Volatility'] = df['Daily_Return'].rolling(window=60).std()
    df['Multi_Term_Volatility'] = 0.4 * df['20_day_Volatility'] + 0.3 * df['5_day_Volatility'] + 0.3 * df['60_day_Volatility']

    # Adjust for Multi-Term Volatility
    df['Multi_Term_Volatility_Adjusted_Alpha_Factor'] = df['Final_Alpha_Factor'] / df['Multi_Term_Volatility']

    # Final Alpha Factor
    final_alpha_factor = df['Multi_Term_Volatility_Adjusted_Alpha_Factor']

    return final_alpha_factor
```
