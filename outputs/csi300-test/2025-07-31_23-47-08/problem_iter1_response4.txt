```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_momentum=20, lookback_atr=14):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=lookback_momentum).mean()
    
    # Calculate Price Momentum
    df['PriceMomentum'] = df['SMA'] - df['SMA'].shift(1)
    
    # Weight by Volume
    df['VolumeAdjustedMomentum'] = df['PriceMomentum'] * df['volume']
    
    # Calculate True Range
    df['TrueRange'] = df[['high', 'low', 'close']].apply(lambda x: max(x['high'] - x['low'],
                                                                        abs(x['high'] - df['close'].shift(1)),
                                                                        abs(x['low'] - df['close'].shift(1))), axis=1)
    
    # Calculate Average True Range (ATR)
    df['ATR'] = df['TrueRange'].rolling(window=lookback_atr).mean()
    
    # Enhance with Price Volatility
    df['CombinedMomentum'] = df['VolumeAdjustedMomentum'] + df['ATR']
    
    # Normalize the Combined Momentum
    df['NormalizedMomentum'] = df['CombinedMomentum'] / df['ATR'].replace(0, 0.0001)
    
    # Return the alpha factor
    return df['NormalizedMomentum']
```
