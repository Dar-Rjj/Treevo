```python
import pandas as pd
import numpy as np

def heuristics_v2(data, n_days=5, vol_window=20):
    # Calculate Intraday High-Low Spread
    intraday_spread = data['high'] - data['low']
    
    # Create Volume Adjusted Factor
    volume_sum_n_days = data['volume'].rolling(window=n_days).sum()
    volume_adjusted_factor = data['volume'] / volume_sum_n_days
    
    # Weight Intraday Spread by Volume Adjusted Factor
    weighted_intraday_spread = intraday_spread * volume_adjusted_factor
    
    # Calculate Intraday Log Return
    intraday_log_return = np.log(data['close'] / data['open'])
    
    # Combine Intraday Momentum with Volume-Adjusted Intraday Spread
    intraday_momentum = weighted_intraday_spread + intraday_log_return
    
    # Calculate Short-Term Volatility
    daily_log_returns = np.log(data['close'] / data['close'].shift(1))
    short_term_volatility = daily_log_returns.rolling(window=vol_window).std()
    
    # Final Alpha Factor
    alpha_factor = intraday_momentum / short_term_volatility
    
    return alpha_factor
```
