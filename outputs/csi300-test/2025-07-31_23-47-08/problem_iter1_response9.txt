```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day price momentum
    df['price_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate 5-day average volume and identify volume spikes
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = (df['volume'] > 1.5 * df['5_day_avg_volume']).astype(int)
    
    # Combine price momentum and volume spike
    df['alpha_factor'] = df['price_momentum'] * df['volume_spike']
    
    return df['alpha_factor']
```
