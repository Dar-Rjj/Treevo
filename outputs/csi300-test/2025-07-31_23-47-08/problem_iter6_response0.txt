```python
import pandas as pd

def heuristics_v2(df, N=20, M=10):
    # Intraday Return Calculation
    df['Intraday Range'] = df['high'] - df['low']
    df['Intraday Return'] = df['Intraday Range'] / df['close']

    # Overnight Return Calculation
    df['Overnight Difference'] = df['open'] - df['close'].shift(1)
    df['Overnight Return'] = df['Overnight Difference'] / df['close'].shift(1)

    # Combine Intraday and Overnight Returns
    df['Combined Return'] = df['Intraday Return'] + df['Overnight Return']

    # Volume-Weighted Average Price (VWAP) Calculation
    df['Typical Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume-Weighted Typical Price'] = df['Typical Price'] * df['volume']
    
    df['VWAP'] = df['Volume-Weighted Typical Price'].rolling(window=N).sum() / df['volume'].rolling(window=N).sum()

    # Reversal Indicator
    df['Reversal Indicator'] = df.apply(lambda row: 1 if row['VWAP'] > row['close'] else -1, axis=1)

    # Integrated Reversal Return
    df['Integrated Reversal Return'] = df['Combined Return'] * df['Reversal Indicator']

    # Volume-Weighted Return
    df['Daily Return'] = (df['close'] - df['open']) / df['open']
    df['Volume-Weighted Return'] = (df['Daily Return'] * df['volume']).rolling(window=M).sum() / df['volume'].rolling(window=M).sum()

    # Final Alpha Factor
    df['Final Alpha Factor'] = df['Integrated Reversal Return'] + df['Volume-Weighted Return']

    return df['Final Alpha Factor']
```
