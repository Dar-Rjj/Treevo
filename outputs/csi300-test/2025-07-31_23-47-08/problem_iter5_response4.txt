import pandas as pd

def heuristics_v2(data):
    # Calculate Intraday Return
    data['intraday_range'] = data['high'] - data['low']
    data['intraday_return'] = data['intraday_range'] / data['close']

    # Calculate Overnight Return
    data['overnight_return'] = (data['open'] - data['close'].shift(1)) / data['close'].shift(1)

    # Combine Intraday and Overnight Returns
    data['combined_return'] = data['intraday_return'] + data['overnight_return']

    # Compute Volume Weighted Average Price (VWAP)
    data['typical_price'] = (data['high'] + data['low'] + data['close']) / 3
    data['volume_weighted_price'] = data['typical_price'] * data['volume']
    vwap = data['volume_weighted_price'].sum() / data['volume'].sum()
    data['vwap'] = vwap

    # Calculate VWAP Reversal Indicator
    data['reversal_indicator'] = data.apply(lambda row: 1 if row['vwap'] > row['close'] else -1, axis=1)

    # Integrate Reversal Indicator with Combined Return
    data['integrated_reversal_return'] = data['combined_return'] * data['reversal_indicator']

    # Calculate Volume-Weighted Return
    data['return'] = (data['close'] - data['open']) / data['open']
    data['volume_weighted_return'] = (data['return'] * data['volume']).sum() / data['volume'].sum()

    # Integrate Volume-Weighted Return with Integrated Reversal Return
    data['final_alpha_factor'] = data['integrated_reversal_return'] + data['volume_weighted_return']

    return data['final_alpha_factor']
