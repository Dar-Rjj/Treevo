```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = df['Intraday_Range'] / df['close']

    # Calculate Overnight Return
    df['Overnight_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Intraday and Overnight Returns
    df['Combined_Return'] = df['Intraday_Return'] + df['Overnight_Return']

    # Compute Volume-Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume_Weighted_Typical_Price'] = df['Typical_Price'] * df['volume']
    total_volume = df['volume'].sum()
    vwap = df['Volume_Weighted_Typical_Price'].sum() / total_volume

    # Calculate VWAP Reversal Indicator
    df['Reversal_Indicator'] = np.where(vwap > df['close'], 1, -1)

    # Integrate Reversal Indicator with Combined Return
    df['Integrated_Reversal_Return'] = df['Combined_Return'] * df['Reversal_Indicator']

    # Calculate Volume-Weighted Return
    df['Return'] = (df['close'] - df['open']) / df['open']
    df['Volume_Weighted_Return'] = df['Return'] * df['volume']
    volume_weighted_return = df['Volume_Weighted_Return'].sum() / total_volume

    # Integrate Volume-Weighted Return with Integrated Reversal Return
    df['Final_Alpha_Factor'] = df['Integrated_Reversal_Return'] + volume_weighted_return

    # Incorporate Volatility
    df['Daily_Log_Return'] = np.log(df['open'] / df['close'].shift(1))
    df['Volatility'] = df['Daily_Log_Return'].rolling(window=20).std()

    # Incorporate Simple Moving Average (SMA)
    df['SMA'] = df['close'].rolling(window=20).mean()

    # Calculate Trend Indicator
    df['Trend_Indicator'] = np.where(df['close'] > df['SMA'], 1, -1)

    # Integrate Trend Indicator with Final Alpha Factor
    df['Integrated_Alpha_Factor'] = df['Final_Alpha_Factor'] * df['Trend_Indicator']

    # Incorporate Transaction Amount
    df['Amount_Weighted_Close'] = df['close'] * df['amount']
    total_amount = df['amount'].sum()
    amount_weighted_average_price = df['Amount_Weighted_Close'].sum() / total_amount
    df['Amount_Weighted_Delta'] = amount_weighted_average_price - df['close']

    # Integrate Amount-Weighted Delta with Integrated Alpha Factor
    df['Enhanced_Final_Alpha_Factor'] = df['Integrated_Alpha_Factor'] + df['Amount_Weighted_Delta']

    # Final Alpha Factor
    final_alpha_factor = df['Enhanced_Final_Alpha_Factor']

    return final_alpha_factor
```
