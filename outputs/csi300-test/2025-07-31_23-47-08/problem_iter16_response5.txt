```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = df['Intraday_Range'] / df['close']

    # Calculate Overnight Return
    df['Overnight_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Intraday and Overnight Returns
    df['Combined_Return'] = df['Intraday_Return'] + df['Overnight_Return']

    # Incorporate Transaction Amount
    df['Amount_Weighted_Intraday_Return'] = df['Intraday_Return'] * df['amount']
    df['Amount_Weighted_Overnight_Return'] = df['Overnight_Return'] * df['amount']
    
    total_amount = df['amount'].sum()
    df['Amount_Weighted_Combined_Return'] = (df['Amount_Weighted_Intraday_Return'].sum() + df['Amount_Weighted_Overnight_Return'].sum()) / total_amount

    # Compute Volume-Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume_Weighted_Typical_Price'] = df['Typical_Price'] * df['volume']
    total_volume = df['volume'].sum()
    df['VWAP'] = df['Volume_Weighted_Typical_Price'].sum() / total_volume

    # Calculate VWAP Reversal Indicator
    df['Reversal_Indicator'] = df.apply(lambda row: 1 if row['VWAP'] > row['close'] else -1, axis=1)

    # Integrate Reversal Indicator with Amount-Weighted Combined Return
    df['Integrated_Reversal_Return'] = df['Amount_Weighted_Combined_Return'] * df['Reversal_Indicator']

    # Calculate Volume-Weighted Return
    df['Return'] = (df['close'] - df['open']) / df['open']
    df['Volume_Weighted_Return'] = df['Return'] * df['volume']
    df['Volume_Weighted_Return'] = df['Volume_Weighted_Return'].sum() / total_volume

    # Integrate Volume-Weighted Return with Integrated Reversal Return
    df['Initial_Alpha_Factor'] = df['Integrated_Reversal_Return'] + df['Volume_Weighted_Return']

    # Calculate Rolling Averages
    df['Short_Term_Rolling_Average'] = df['close'].rolling(window=5).mean()
    df['Long_Term_Rolling_Average'] = df['close'].rolling(window=20).mean()

    # Calculate Dynamic Volatility
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Short_Term_Volatility'] = df['Daily_Return'].rolling(window=10).std()
    df['Medium_Term_Volatility'] = df['Daily_Return'].rolling(window=20).std()
    df['Long_Term_Volatility'] = df['Daily_Return'].rolling(window=50).std()

    # Adjust for Volatility
    df['Short_Term_Adjusted_Alpha'] = df['Initial_Alpha_Factor'] / df['Short_Term_Volatility']
    df['Medium_Term_Adjusted_Alpha'] = df['Initial_Alpha_Factor'] / df['Medium_Term_Volatility']
    df['Long_Term_Adjusted_Alpha'] = df['Initial_Alpha_Factor'] / df['Long_Term_Volatility']

    # Integrate Volatility-Adjusted Alpha Factors
    df['Final_Alpha_Factor'] = (df['Short_Term_Adjusted_Alpha'] + 
                                df['Medium_Term_Adjusted_Alpha'] + 
                                df['Long_Term_Adjusted_Alpha'])

    return df['Final_Alpha_Factor']
```
