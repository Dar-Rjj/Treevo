```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = df['Intraday_Range'] / df['close']

    # Calculate Overnight Return
    df['Overnight_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Intraday and Overnight Returns
    df['Combined_Return'] = df['Intraday_Return'] + df['Overnight_Return']

    # Compute Volume Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume_Typical_Price'] = df['Typical_Price'] * df['volume']
    total_volume = df['volume'].sum()
    VWAP = df['Volume_Typical_Price'].sum() / total_volume
    df['VWAP'] = VWAP

    # Calculate VWAP Reversal Indicator
    df['Reversal_Indicator'] = np.where(df['VWAP'] > df['close'], 1, -1)

    # Calculate Volume Momentum
    df['Volume_MA_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Momentum'] = np.where(df['volume'] > df['Volume_MA_20'], 1, -1)

    # Calculate Intraday Volatility
    df['True_Range'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()

    # Adjust Combined Return by Intraday Volatility
    df['Adjusted_Combined_Return'] = df['Combined_Return'] / df['ATR_14']

    # Integrate Reversal Indicator, Adjusted Combined Return, and Volume Momentum
    df['Final_Alpha_Factor'] = df['Adjusted_Combined_Return'] * df['Reversal_Indicator'] * df['Volume_Momentum']

    return df['Final_Alpha_Factor']
```
