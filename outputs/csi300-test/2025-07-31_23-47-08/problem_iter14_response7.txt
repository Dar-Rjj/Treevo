import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_range = df['high'] - df['low']
    intraday_return = intraday_range / df['close']

    # Calculate Overnight Return
    overnight_diff = df['open'] - df['close'].shift(1)
    overnight_return = overnight_diff / df['close'].shift(1)

    # Combine Intraday and Overnight Returns
    combined_return = intraday_return + overnight_return

    # Compute Volume-Weighted Return
    return_daily = (df['close'] - df['open']) / df['open']
    volume_weighted_return = (return_daily * df['volume']).sum() / df['volume'].sum()

    # Integrate Volume-Weighted Return with Combined Return
    final_alpha_factor = combined_return * volume_weighted_return

    # Incorporate Trading Volume Trends
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_ema = df['volume'].ewm(span=5, adjust=False).mean()  # 5-day EMA for example
    volume_trend_indicator = np.where(df['volume'] > volume_ema, 1, -1)

    # Integrate Volume Trend with Final Alpha Factor
    ultimate_alpha_factor = final_alpha_factor * volume_trend_indicator

    return pd.Series(ultimate_alpha_factor, index=df.index, name='ultimate_alpha_factor')
