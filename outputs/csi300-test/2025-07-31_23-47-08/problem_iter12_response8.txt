```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_range = df['high'] - df['low']
    intraday_return = intraday_range / df['close']

    # Calculate Overnight Return
    overnight_return = (df['open'].shift(-1) - df['close']) / df['close']

    # Combine Intraday and Overnight Returns
    combined_return = intraday_return + overnight_return

    # Compute Volume-Weighted Return
    return_close_open = (df['close'] - df['open']) / df['open']
    volume_weighted_return = (return_close_open * df['volume']).sum() / df['volume'].sum()

    # Integrate Volume-Weighted Return with Combined Return
    final_alpha_factor = combined_return * volume_weighted_return

    return final_alpha_factor
```
