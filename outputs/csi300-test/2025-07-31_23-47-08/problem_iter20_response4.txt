import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = df['Intraday_Range'] / df['close']

    # Calculate Overnight Return
    df['Overnight_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Incorporate Transaction Amount
    df['Amount_Weighted_Intraday_Return'] = df['Intraday_Return'] * df['amount']
    df['Amount_Weighted_Overnight_Return'] = df['Overnight_Return'] * df['amount']
    total_amount = df['amount'].sum()
    amount_weighted_intraday_return = df['Amount_Weighted_Intraday_Return'].sum() / total_amount
    amount_weighted_overnight_return = df['Amount_Weighted_Overnight_Return'].sum() / total_amount
    df['Amount_Weighted_Combined_Return'] = amount_weighted_intraday_return + amount_weighted_overnight_return

    # Combine Intraday and Overnight Returns
    df['Combined_Return'] = df['Intraday_Return'] + df['Overnight_Return']

    # Compute Volume-Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume_Weighted_Typical_Price'] = df['Typical_Price'] * df['volume']
    total_volume = df['volume'].sum()
    vwap = df['Volume_Weighted_Typical_Price'].sum() / total_volume

    # Calculate VWAP Reversal Indicator
    df['Reversal_Indicator'] = (vwap > df['close']).astype(int) * 2 - 1

    # Integrate Reversal Indicator with Amount-Weighted Combined Return
    df['Integrated_Reversal_Return'] = df['Amount_Weighted_Combined_Return'] * df['Reversal_Indicator']

    # Calculate Volume-Weighted Return
    df['Return'] = (df['close'] - df['open']) / df['open']
    df['Volume_Weighted_Return'] = df['Return'] * df['volume']
    volume_weighted_return = df['Volume_Weighted_Return'].sum() / total_volume

    # Integrate Volume-Weighted Return with Integrated Reversal Return
    df['Initial_Alpha_Factor'] = df['Integrated_Reversal_Return'] + volume_weighted_return

    # Calculate Rolling Averages
    df['Short_Term_Rolling_Average'] = df['close'].rolling(window=5).mean()
    df['Long_Term_Rolling_Average'] = df['close'].rolling(window=20).mean()

    # Calculate Dynamic Volatility
    df['Daily_Returns'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Short_Term_Volatility'] = df['Daily_Returns'].rolling(window=10).std()
    df['Medium_Term_Volatility'] = df['Daily_Returns'].rolling(window=20).std()
    df['Long_Term_Volatility'] = df['Daily_Returns'].rolling(window=50).std()

    df['Short_Term_Volatility_Adjusted_Alpha_Factor'] = df['Initial_Alpha_Factor'] / df['Short_Term_Volatility']
    df['Medium_Term_Volatility_Adjusted_Alpha_Factor'] = df['Initial_Alpha_Factor'] / df['Medium_Term_Volatility']
    df['Long_Term_Volatility_Adjusted_Alpha_Factor'] = df['Initial_Alpha_Factor'] / df['Long_Term_Volatility']
