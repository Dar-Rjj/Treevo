```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['Intraday Range'] = df['high'] - df['low']
    df['Intraday Return'] = df['Intraday Range'] / df['close']

    # Calculate Overnight Momentum
    df['Overnight Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Intraday and Overnight Returns
    df['Combined Return'] = df['Intraday Return'] + df['Overnight Return']

    # Compute Dynamic Volume-Weighted Average Price (VWAP)
    df['Typical Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume-Weighted Typical Price'] = df['Typical Price'] * df['volume']
    df['VWAP'] = df['Volume-Weighted Typical Price'].cumsum() / df['volume'].cumsum()

    # Calculate Adaptive VWAP Reversal Indicator
    df['VWAP Deviation'] = df['VWAP'] - df['close']
    dynamic_threshold = 1.5 * df['VWAP Deviation'].rolling(window=20).std()
    df['Reversal Indicator'] = 0
    df.loc[df['VWAP Deviation'] > dynamic_threshold, 'Reversal Indicator'] = 1
    df.loc[df['VWAP Deviation'] < -dynamic_threshold, 'Reversal Indicator'] = -1

    # Integrate Reversal Indicator with Combined Return
    df['Integrated Reversal Return'] = df['Combined Return'] * df['Reversal Indicator']

    # Calculate Volume-Weighted Return
    df['Return'] = (df['close'] - df['open']) / df['open']
    df['Volume-Weighted Return'] = (df['Return'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Integrate Volume-Weighted Return with Integrated Reversal Return
    df['Final Alpha Factor'] = df['Integrated Reversal Return'] + df['Volume-Weighted Return']

    # Apply Adaptive Weighting to Intraday Momentum
    df['Recent Volatility'] = df['close'].pct_change().rolling(window=20).std()
    df['Volatility-Based Weight'] = 1 / df['Recent Volatility']
    df['Average Volume'] = df['volume'].rolling(window=20).mean()
    df['Volume-Based Weight'] = df['volume'] / df['Average Volume']
    df['Weighted Intraday Momentum'] = df['Intraday Return'] * df['Volatility-Based Weight'] * df['Volume-Based Weight']

    # Combine Weighted Intraday Momentum with Final Alpha Factor
    df['Enhanced Final Alpha Factor'] = df['Final Alpha Factor'] + df['Weighted Intraday Momentum']

    return df['Enhanced Final Alpha Factor'].dropna()
```
