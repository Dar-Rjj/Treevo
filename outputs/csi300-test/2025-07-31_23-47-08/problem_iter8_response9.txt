import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor using a combination of price and volume data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    
    Returns:
    pd.Series: A series indexed by date representing the alpha factor values.
    """
    
    # Calculate the relative range of the day's trading
    df['range'] = df['high'] - df['low']
    df['relative_range'] = df['range'] / df['close']
    
    # Calculate the money flow index
    df['money_flow'] = df['close'] * df['volume']
    
    # Calculate the normalized money flow index
    df['norm_money_flow'] = df['money_flow'] / df['money_flow'].rolling(window=10).mean()
    
    # Combine the factors to create the alpha
    df['alpha_factor'] = df['relative_range'] * df['norm_money_flow']
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
