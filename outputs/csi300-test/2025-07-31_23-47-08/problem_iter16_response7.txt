```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_range = df['high'] - df['low']
    intraday_return = intraday_range / df['close'].shift(1)
    
    # Calculate Overnight Return
    overnight_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Intraday and Overnight Returns
    combined_return = intraday_return + overnight_return
    
    # Calculate Volume-Weighted Return
    daily_return = (df['close'] - df['open']) / df['open']
    volume_weighted_return = (daily_return * df['volume']).sum() / df['volume'].sum()
    
    # Integrate Volume-Weighted Return with Combined Return
    initial_alpha_factor = combined_return + volume_weighted_return
    
    # Calculate Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility = daily_returns.rolling(window=20).std()
    
    # Adjust for Volatility
    volatility_adjusted_alpha_factor = initial_alpha_factor / volatility
    
    # Incorporate Transaction Amount
    amount_weighted_return = ((df['close'] - df['open']) / df['open'] * df['amount']).sum() / df['amount'].sum()
    
    # Integrate Amount-Weighted Return with Volatility-Adjusted Alpha Factor
    final_alpha_factor = volatility_adjusted_alpha_factor + amount_weighted_return
    
    return final_alpha_factor
```
