```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Smoothed Intraday Volatility (EMA for 5 days)
    df['smoothed_volatility'] = df['price_range'].ewm(span=5, adjust=False).mean()
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Weighted Long-Term Momentum
    df['weighted_momentum'] = df['daily_return'] * df['smoothed_volatility']
    
    # Momentum over 20 Days
    df['momentum_20_days'] = df['weighted_momentum'].rolling(window=20).sum()
    
    # Adjust for Volatility
    df['volatility'] = df['daily_return'].rolling(window=20).std()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['momentum_20_days'] / df['volatility']
    
    return df['alpha_factor']
```
