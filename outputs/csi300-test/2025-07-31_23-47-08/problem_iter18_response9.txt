```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the price change from the previous day
    price_change = df['close'].pct_change()
    
    # Calculate the volume change from the previous day
    volume_change = df['volume'].pct_change()
    
    # Calculate the amount change from the previous day
    amount_change = df['amount'].pct_change()
    
    # Calculate the weighted moving average of the price change
    wma_price_change = price_change.rolling(window=5).mean()
    
    # Calculate the weighted moving average of the volume change
    wma_volume_change = volume_change.rolling(window=5).mean()
    
    # Calculate the weighted moving average of the amount change
    wma_amount_change = amount_change.rolling(window=5).mean()
    
    # Combine the factors into a single alpha factor
    alpha_factor = (avg_price * wma_price_change) + (df['volume'] * wma_volume_change) + (df['amount'] * wma_amount_change)
    
    return alpha_factor
```
