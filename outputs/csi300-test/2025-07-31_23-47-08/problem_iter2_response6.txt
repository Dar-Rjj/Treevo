```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=10, short_term_vol_window=20):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Create Volume Adjusted Factor
    volume_adjusted_factor = df['volume'].rolling(window=window).sum()
    volume_adjusted_factor = df['volume'] / volume_adjusted_factor
    
    # Weight Intraday Spread by Volume Adjusted Factor
    weighted_high_low_spread = high_low_spread * volume_adjusted_factor
    
    # Calculate Intraday Log Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # Combine Intraday Momentum with Volume-Adjusted Intraday Spread
    intraday_momentum = weighted_high_low_spread + intraday_log_return
    
    # Calculate Daily Log Returns
    daily_log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Short-Term Volatility
    short_term_volatility = daily_log_returns.rolling(window=short_term_vol_window).std()
    
    # Final Alpha Factor
    alpha_factor = intraday_momentum / short_term_volatility
    
    return alpha_factor
```
