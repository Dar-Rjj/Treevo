```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the simple moving average of the close price over 5 days
    sma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the exponential moving average of the close price over 10 days
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) over a 14 day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Heuristic alpha factor: Combine the SMA, EMA, and RSI
    alpha_factor = (sma_5 - ema_10) / (rsi - 50)
    
    return alpha_factor
```
