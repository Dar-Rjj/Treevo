```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Overnight Return
    df['overnight_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Compute Rolling Average of Intraday Returns (5 days window)
    df['rolling_intraday_return'] = df['intraday_return'].rolling(window=5).mean()
    
    # Compute Rolling Average of Overnight Returns (5 days window)
    df['rolling_overnight_return'] = df['overnight_return'].rolling(window=5).mean()
    
    # Calculate Intraday Volatility
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['true_range'] = df['true_range'].fillna(0)
    df['average_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Liquidity Measure
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    df['liquidity_measure'] = df['volume'] / df['average_volume']
    
    # Calculate Intraday Momentum
    momentum_factor = 1.2
    df['intraday_momentum'] = (df['rolling_intraday_return'] * momentum_factor) / df['average_true_range']
    
    # Calculate Overnight Reversal
    reversal_factor = -0.8
    df['overnight_reversal'] = (df['rolling_overnight_return'] * reversal_factor) / df['average_true_range']
    
    # Combine Intraday Momentum, Overnight Reversal, and Liquidity
    df['combined_factor'] = (df['intraday_momentum'] + df['overnight_reversal']) * df['liquidity_measure']
    
    # Final Alpha Factor
    final_alpha_factor = df['combined_factor']
    
    return final_alpha_factor
```
