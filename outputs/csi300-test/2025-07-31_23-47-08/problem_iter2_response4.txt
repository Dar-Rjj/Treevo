```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_range = df['high'] - df['low']
    intraday_return = intraday_range / df['close']

    # Calculate Overnight Return
    overnight_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Intraday and Overnight Returns
    combined_return = intraday_return + overnight_return

    # Compute Volume Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    volume_weighted_typical_price = typical_price * df['volume']
    vwap = volume_weighted_typical_price.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Calculate VWAP Reversal Indicator
    reversal_indicator = (vwap > df['close']).astype(int) - (vwap < df['close']).astype(int)

    # Calculate Volume Momentum
    volume_moving_average = df['volume'].rolling(window=20).mean()
    volume_momentum = (df['volume'] > volume_moving_average).astype(int) - (df['volume'] < volume_moving_average).astype(int)

    # Integrate Reversal Indicator, Combined Return, and Volume Momentum
    final_alpha_factor = combined_return * reversal_indicator * volume_momentum

    return final_alpha_factor
```
