```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = df['Intraday_Range'] / df['close']

    # Calculate Overnight Return
    df['Overnight_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Intraday and Overnight Returns
    df['Combined_Return'] = df['Intraday_Return'] + df['Overnight_Return']

    # Compute Volume-Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume_Weighted_Price'] = df['Typical_Price'] * df['volume']
    df['VWAP'] = df['Volume_Weighted_Price'].rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Calculate VWAP Reversal Indicator
    df['Reversal_Indicator'] = np.where(df['VWAP'] > df['close'], 1, -1)

    # Integrate Reversal Indicator with Combined Return
    df['Integrated_Reversal_Return'] = df['Combined_Return'] * df['Reversal_Indicator']

    # Calculate Volume-Weighted Return
    df['Volume_Weighted_Return'] = ((df['close'] - df['open']) / df['open']) * df['volume']
    df['Total_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Integrate Volume-Weighted Return with Integrated Reversal Return
    df['Initial_Alpha_Factor'] = df['Integrated_Reversal_Return'] + df['Total_Volume_Weighted_Return']

    # Calculate Adaptive Rolling Averages
    df['Rolling_Avg_5'] = df['close'].rolling(window=5).mean()
    df['Rolling_Avg_20'] = df['close'].rolling(window=20).mean()
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    recent_volatility = df['Daily_Return'].rolling(window=20).std()
    df['Adaptive_Rolling_Averages'] = np.where(recent_volatility > recent_volatility.mean(), df['Rolling_Avg_5'], df['Rolling_Avg_20'])

    # Calculate Volatility
    df['Volatility'] = df['Daily_Return'].rolling(window=20).std()
    df['Volatility_Adjusted'] = np.where(recent_volatility > recent_volatility.mean(), df['Daily_Return'].rolling(window=10).std(), df['Daily_Return'].rolling(window=30).std())

    # Adjust for Volatility
    df['Volatility_Adjusted_Alpha_Factor'] = df['Initial_Alpha_Factor'] / df['Volatility_Adjusted']

    # Incorporate Transaction Amount
    df['Amount_Weighted_Return'] = ((df['close'] - df['open']) / df['open']) * df['amount']
    df['Total_Amount_Weighted_Return'] = df['Amount_Weighted_Return'].rolling(window=20).sum() / df['amount'].rolling(window=20).sum()

    # Integrate Amount-Weighted Return with Volatility-Adjusted Alpha Factor
    df['Final_Alpha_Factor'] = df['Volatility_Adjusted_Alpha_Factor'] + df['Total_Amount_Weighted_Return']

    return df['Final_Alpha_Factor']
```
