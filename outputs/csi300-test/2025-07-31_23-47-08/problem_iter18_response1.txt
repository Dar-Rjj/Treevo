import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change from open to close
    df['pct_change'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the range of the day (High - Low)
    df['range'] = df['high'] - df['low']
    
    # Calculate the average price
    df['average_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the normalized volume
    df['norm_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Alpha factor: Combine the above features with weights
    df['alpha_factor'] = 0.4 * df['pct_change'] + 0.3 * (df['average_price'] / df['range']) + 0.3 * df['norm_volume']
    
    return df['alpha_factor']
