import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Opening Gap
    df['Opening_Gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Momentum Indicator
    df['Daily_Returns'] = df['close'] - df['open']
    df['Cumulative_Return'] = df['Daily_Returns'].rolling(window=n).sum()
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    df['Volume_Weighted_Momentum'] = df['Cumulative_Return'] * df['Average_Volume']
    
    # Calculate Intraday Volume Change
    df['Average_Past_Volume'] = df['volume'].rolling(window=n).mean()
    df['Intraday_Volume_Change'] = df['volume'] / df['Average_Past_Volume']
    
    # Combine Subcomponents
    df['Composite_Index'] = (df['High_Low_Spread'] + 
                             df['Opening_Gap'] + 
                             df['Volume_Weighted_Momentum']) * df['Intraday_Volume_Change']
    
    # Return the factor values
    return df['Composite_Index']
