```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Overnight Return
    df['overnight_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Compute Rolling Average of Intraday Returns (5 days)
    df['rolling_intraday_return'] = df['intraday_return'].rolling(window=5).mean()
    
    # Compute Rolling Average of Overnight Returns (5 days)
    df['rolling_overnight_return'] = df['overnight_return'].rolling(window=5).mean()
    
    # Calculate Intraday Volatility
    df['daily_price_range'] = df['high'] - df['low']
    df['avg_daily_price_range'] = df['daily_price_range'].rolling(window=5).mean()
    df['intraday_volatility'] = df['avg_daily_price_range'] / df['daily_price_range']
    
    # Adjusted Intraday Momentum
    momentum_factor = 1.2
    df['adjusted_intraday_momentum'] = df['rolling_intraday_return'] * momentum_factor * df['intraday_volatility']
    
    # Adjusted Overnight Reversal
    reversal_factor = -0.8
    df['adjusted_overnight_reversal'] = df['rolling_overnight_return'] * reversal_factor * df['intraday_volatility']
    
    # Incorporate Volume Impact
    df['volume_weighted_return'] = df['intraday_return'] * df['volume']
    df['rolling_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=5).mean()
    
    # Adjust Intraday and Overnight Factors based on Volume Impact
    df['adjusted_factors'] = (df['adjusted_intraday_momentum'] + df['adjusted_overnight_reversal']) * df['rolling_volume_weighted_return']
    
    # Combine Adjusted Intraday Momentum, Adjusted Overnight Reversal, and Volume Impact
    df['alpha_factor'] = df['adjusted_intraday_momentum'] + df['adjusted_overnight_reversal'] + df['adjusted_factors']
    
    return df['alpha_factor']
```
