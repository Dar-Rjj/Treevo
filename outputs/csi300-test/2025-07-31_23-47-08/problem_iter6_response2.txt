import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_range = df['high'] - df['low']
    intraday_momentum = intraday_range / df['close']

    # Calculate Overnight Momentum
    overnight_change = df['open'] - df['close'].shift(1)
    overnight_momentum = overnight_change / df['close'].shift(1)

    # Combine Intraday and Overnight Momentum
    combined_momentum = intraday_momentum + overnight_momentum

    # Adaptive Volatility Calculation
    daily_log_returns = np.log(df['close'] / df['close'].shift(1))
    adaptive_volatility = daily_log_returns.ewm(span=20).std()

    # Market Sentiment Indicator
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    sentiment_indicator = np.where(vwap > df['close'], 1, -1)

    # Integrate Sentiment with Combined Momentum
    integrated_momentum = combined_momentum * sentiment_indicator

    # Calculate Volume-Weighted Return
    return_value = (df['close'] - df['open']) / df['open']
    volume_weighted_return = (return_value * df['volume']).sum() / df['volume'].sum()

    # Integrate Volume-Weighted Return with Integrated Momentum
    final_alpha_factor = integrated_momentum + volume_weighted_return

    # Normalize Final Alpha Factor by Adaptive Volatility
    final_alpha_factor = final_alpha_factor / adaptive_volatility

    return final_alpha_factor
