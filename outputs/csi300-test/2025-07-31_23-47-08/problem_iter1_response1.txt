```python
import pandas as pd

def heuristics_v2(df, n=5):
    # Calculate Intraday High-Low Spread
    df['Intraday_Spread'] = df['high'] - df['low']
    
    # Calculate Opening Gap
    df['Opening_Gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate Momentum Indicator
    df['Daily_Return'] = df['close'] - df['open']
    df['Cumulative_Return'] = df['Daily_Return'].rolling(window=n).sum()
    df['Momentum_Indicator'] = df['Cumulative_Return'] * df['volume']
    
    # Aggregate Subcomponents
    df['Intraday_Sentiment_Index'] = (df['Intraday_Spread'] + 
                                       df['Opening_Gap'] + 
                                       df['Momentum_Indicator']) / 3
    
    # Interpret Factor: Higher values indicate positive sentiment, lower values indicate negative sentiment
    
    return df['Intraday_Sentiment_Index']
```
