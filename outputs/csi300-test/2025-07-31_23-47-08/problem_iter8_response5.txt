```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_range = df['high'] - df['low']
    intraday_return = intraday_range / df['close']

    # Calculate Overnight Return
    overnight_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Intraday and Overnight Returns
    combined_return = intraday_return + overnight_return

    # Compute Volume Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()

    # Calculate VWAP Reversal Indicator
    reversal_indicator = (vwap > df['close']).astype(int) * 2 - 1

    # Integrate Reversal Indicator with Combined Return
    integrated_reversal_return = combined_return * reversal_indicator

    # Calculate Volume-Weighted Return
    volume_weighted_return = ((df['close'] - df['open']) / df['open']) * df['volume']
    volume_weighted_return = volume_weighted_return.cumsum() / df['volume'].cumsum()

    # Integrate Volume-Weighted Return with Integrated Reversal Return
    final_alpha_factor = integrated_reversal_return + volume_weighted_return

    # Calculate Volume-Weighted Moving Average (VWMA)
    close_volume_product = df['close'] * df['volume']
    vwma = close_volume_product.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Calculate VWMA Reversal Indicator
    vwma_reversal_indicator = (vwma > df['close']).astype(int) * 2 - 1

    # Integrate VWMA Reversal Indicator with Final Alpha Factor
    enhanced_final_alpha_factor = final_alpha_factor * vwma_reversal_indicator

    # Calculate Volume-Weighted Momentum (VWM)
    momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    vwm = (momentum * df['volume']).cumsum() / df['volume'].cumsum()

    # Integrate VWM with Enhanced Final Alpha Factor
    final_integrated_alpha_factor = enhanced_final_alpha_factor + vwm

    return final_integrated_alpha_factor
```
