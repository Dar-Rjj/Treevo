```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the positive and negative money flow
    positive_money_flow = money_flow[df['close'] > df['close'].shift(1)].fillna(0)
    negative_money_flow = money_flow[df['close'] < df['close'].shift(1)].fillna(0)
    
    # Calculate the ratio of positive to negative money flow
    pmf_sum = positive_money_flow.rolling(window=14).sum()
    nmf_sum = negative_money_flow.rolling(window=14).sum()
    money_ratio = pmf_sum / nmf_sum
    
    # Generate the alpha factor
    alpha_factor = money_ratio - money_ratio.rolling(window=14).mean()
    
    return alpha_factor
```
