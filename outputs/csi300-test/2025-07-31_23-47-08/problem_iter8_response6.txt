import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_range = df['high'] - df['low']
    intraday_momentum = intraday_range / df['close']

    # Calculate Overnight Momentum
    overnight_momentum = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Intraday and Overnight Momentum
    combined_momentum = intraday_momentum + overnight_momentum

    # Adaptive Volatility Calculation
    daily_log_returns = np.log(df['close'] / df['close'].shift(1))
    span = 20
    adaptive_volatility = daily_log_returns.ewm(span=span, min_periods=span).std()

    # Market Sentiment Indicator
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    volume_weighted_typical_price = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    sentiment_indicator = np.where(volume_weighted_typical_price > df['close'], 1, -1)

    # Integrate Sentiment with Combined Momentum
    integrated_momentum = combined_momentum * sentiment_indicator

    # Calculate Volume-Weighted Return
    return_series = (df['close'] - df['open']) / df['open']
    volume_weighted_return = (return_series * df['volume']).cumsum() / df['volume'].cumsum()

    # Integrate Volume-Weighted Return with Integrated Momentum
    final_alpha_factor = integrated_momentum + volume_weighted_return

    return final_alpha_factor
