import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = df['Intraday_Range'] / df['close']
    
    # Calculate Overnight Momentum
    df['Overnight_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Intraday and Overnight Returns
    df['Combined_Return'] = df['Intraday_Return'] + df['Overnight_Return']
    
    # Compute Dynamic Volume-Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume_Weighted_Price'] = df['Typical_Price'] * df['volume']
    df['VWAP'] = df['Volume_Weighted_Price'].rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate Adaptive VWAP Reversal Indicator
    df['VWAP_Deviation'] = df['VWAP'] - df['close']
    df['Threshold'] = 1.5 * df['VWAP_Deviation'].rolling(window=20).std()
    conditions = [
        (df['VWAP_Deviation'] > df['Threshold']),
        (df['VWAP_Deviation'] < -df['Threshold'])
    ]
    choices = [1, -1]
    df['Reversal_Indicator'] = np.select(conditions, choices, default=0)
    
    # Integrate Reversal Indicator with Combined Return
    df['Integrated_Reversal_Return'] = df['Combined_Return'] * df['Reversal_Indicator']
    
    # Calculate Volume-Weighted Return
    df['Return'] = (df['close'] - df['open']) / df['open']
    df['Volume_Weighted_Return'] = (df['Return'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Integrate Volume-Weighted Return with Integrated Reversal Return
    df['Final_Alpha_Factor'] = df['Integrated_Reversal_Return'] + df['Volume_Weighted_Return']
    
    # Apply Weighting to Intraday Momentum
    df['Weighted_Intraday_Momentum'] = df['Intraday_Return'] * (df['volume'].rolling(window=20).mean() / df['volume'].mean())
    
    # Combine Weighted Intraday Momentum with Final Alpha Factor
    df['Enhanced_Final_Alpha_Factor'] = df['Final_Alpha_Factor'] + df['Weighted_Intraday_Momentum']
    
    return df['Enhanced_Final_Alpha_Factor']
