import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate log returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Take moving sum of log returns over a 20-day window
    df['moving_sum_log_return'] = df['log_return'].rolling(window=20).sum()
    
    # Weight by volume
    df['volume_weighted_moving_sum_log_return'] = df['moving_sum_log_return'] * df['volume']
    
    # Return the factor values
    return df['volume_weighted_moving_sum_log_return']
