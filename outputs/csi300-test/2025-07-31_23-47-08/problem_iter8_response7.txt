import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = df['Intraday_Range'] / df['close']
    
    # Calculate Overnight Return
    df['Overnight_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Intraday and Overnight Returns
    df['Combined_Return'] = df['Intraday_Return'] + df['Overnight_Return']
    
    # Compute Volume Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume_Weighted_Typical_Price'] = df['Typical_Price'] * df['volume']
    vwap_sum = df['Volume_Weighted_Typical_Price'].cumsum()
    volume_sum = df['volume'].cumsum()
    df['VWAP'] = vwap_sum / volume_sum
    
    # Calculate VWAP Reversal Indicator
    conditions = [
        (df['VWAP'] > df['close']),
        (df['VWAP'] < df['close'])
    ]
    choices = [1, -1]
    df['Reversal_Indicator'] = pd.Series(pd.np.select(conditions, choices, default=0), index=df.index)
    
    # Integrate Reversal Indicator with Combined Return
    df['Integrated_Reversal_Return'] = df['Combined_Return'] * df['Reversal_Indicator']
    
    # Calculate Volume-Weighted Intraday Return
    df['Volume_Weighted_Intraday_Return'] = ((df['close'] - df['open']) / df['open']) * df['volume']
    df['Volume_Weighted_Intraday_Return'] = (df['Volume_Weighted_Intraday_Return'].cumsum() / volume_sum)
    
    # Integrate Volume-Weighted Intraday Return with Integrated Reversal Return
    df['Final_Alpha_Factor'] = df['Integrated_Reversal_Return'] + df['Volume_Weighted_Intraday_Return']
    
    # Incorporate Relative Volume
    df['Volume_Moving_Average'] = df['volume'].rolling(window=20).mean()
    df['Relative_Volume'] = df['volume'] / df['Volume_Moving_Average']
    df['Adjusted_Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * df['Relative_Volume']
    
    return df['Adjusted_Final_Alpha_Factor']
