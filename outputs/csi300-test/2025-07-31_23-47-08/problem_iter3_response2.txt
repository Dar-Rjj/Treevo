```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = df['Intraday_Range'] / df['close']

    # Calculate Overnight Return
    df['Previous_Close'] = df['close'].shift(1)
    df['Overnight_Return'] = (df['open'] - df['Previous_Close']) / df['Previous_Close']

    # Combine Intraday and Overnight Returns
    df['Combined_Return'] = df['Intraday_Return'] + df['Overnight_Return']

    # Compute Volume Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume_Weighted_Typical_Price'] = df['Typical_Price'] * df['volume']
    df['VWAP'] = df['Volume_Weighted_Typical_Price'].cumsum() / df['volume'].cumsum()

    # Calculate VWAP Reversal Indicator
    df['Reversal_Indicator'] = df.apply(lambda row: 1 if row['VWAP'] > row['close'] else -1, axis=1)

    # Integrate Reversal Indicator with Combined Return
    df['Final_Alpha_Factor'] = df['Combined_Return'] * df['Reversal_Indicator']

    return df['Final_Alpha_Factor']
```
