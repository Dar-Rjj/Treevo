```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Average Intraday Volatility over 20 Days
    df['avg_intraday_volatility'] = df['daily_price_range'].rolling(window=20).mean()
    
    # Calculate Close-to-Previous Close Return
    df['close_to_prev_close_return'] = df['close'].pct_change()
    
    # Weighted Sum of Returns Based on Intraday Volatility
    df['weighted_sum_returns'] = (df['close_to_prev_close_return'] * df['avg_intraday_volatility']).rolling(window=20).sum()
    
    # Incorporate Volume Spike
    df['volume_t'] = df['volume']
    df['volume_t_minus_5'] = df['volume'].shift(5)
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = (df['volume_t'] > 1.5 * df['5_day_avg_volume']).astype(int)
    
    # Combine Weighted Sum and Volume Spike
    df['alpha_factor'] = df['weighted_sum_returns'] * df['volume_spike']
    
    return df['alpha_factor']
```
