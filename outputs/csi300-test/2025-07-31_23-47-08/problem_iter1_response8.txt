import pandas as pd

def heuristics_v2(df, spread_window=5, volatility_window=20):
    # Calculate Intraday High-Low Spread
    df['Intraday_Spread'] = df['high'] - df['low']
    
    # Create Volume Adjusted Factor
    df['Volume_Adjusted_Factor'] = df['volume'].rolling(window=spread_window).sum()
    df['Volume_Adjusted_Factor'] = df['volume'] / df['Volume_Adjusted_Factor']
    
    # Weight Intraday Spread by Volume Adjusted Factor
    df['Weighted_Spread'] = df['Intraday_Spread'] * df['Volume_Adjusted_Factor']
    
    # Calculate Daily Returns
    df['Daily_Returns'] = df['close'].pct_change()
    
    # Calculate Standard Deviation of Recent Returns (Volatility)
    df['Volatility'] = df['Daily_Returns'].rolling(window=volatility_window).std()
    
    # Combine Factors for Final Alpha
    df['Alpha_Factor'] = df['Weighted_Spread'] / df['Volatility']
    
    # Return the Alpha Factor as a pandas Series
    return df['Alpha_Factor']
