```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday Range'] = df['high'] - df['low']
    df['Intraday Return'] = df['Intraday Range'] / df['close']

    # Calculate Overnight Return
    df['Overnight Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Incorporate Transaction Amount
    df['Amount-Weighted Intraday Return'] = df['Intraday Return'] * df['amount']
    df['Amount-Weighted Overnight Return'] = df['Overnight Return'] * df['amount']
    
    total_amount = df['amount'].sum()
    amount_weighted_intraday_return = df['Amount-Weighted Intraday Return'].sum() / total_amount
    amount_weighted_overnight_return = df['Amount-Weighted Overnight Return'].sum() / total_amount
    amount_weighted_combined_return = amount_weighted_intraday_return + amount_weighted_overnight_return

    # Calculate Dynamic Weights
    df['5d Volume Avg'] = df['volume'].rolling(window=5).mean()
    df['20d Volume Avg'] = df['volume'].rolling(window=20).mean()
    df['Dynamic Weight'] = df['5d Volume Avg'] / df['20d Volume Avg']

    # Compute Volume-Weighted Average Price (VWAP)
    df['Typical Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume-Weighted Typical Price'] = df['Typical Price'] * df['volume']
    vwap = df['Volume-Weighted Typical Price'].sum() / df['volume'].sum()

    # Calculate VWAP Reversal Indicator
    df['Reversal Indicator'] = df.apply(lambda row: 1 if vwap > row['close'] else -1, axis=1)

    # Integrate Reversal Indicator with Amount-Weighted Combined Return
    df['Integrated Reversal Return'] = amount_weighted_combined_return * df['Reversal Indicator']

    # Calculate Volume-Weighted Return
    df['Return'] = (df['close'] - df['open']) / df['open']
    df['Volume-Weighted Return'] = df['Return'] * df['volume']
    volume_weighted_return = df['Volume-Weighted Return'].sum() / df['volume'].sum()

    # Integrate Volume-Weighted Return with Integrated Reversal Return
    initial_alpha_factor = volume_weighted_return + df['Integrated Reversal Return']

    # Adjust for Dynamic Weights
    df['Weight-Adjusted Alpha Factor'] = initial_alpha_factor * df['Dynamic Weight']

    # Calculate Dynamic Volatility
    df['Daily Returns'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Medium-Term Volatility'] = df['Daily Returns'].rolling(window=20).std()

    # Adjust for Volatility
    df['Volatility-Adjusted Alpha Factor'] = df['Weight-Adjusted Alpha Factor'] / df['Medium-Term Volatility']

    # Final Alpha Factor
    final_alpha_factor = df['Volatility-Adjusted Alpha Factor']

    return final_alpha_factor
```
