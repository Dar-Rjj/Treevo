```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday Range'] = df['high'] - df['low']
    df['Intraday Return'] = df['Intraday Range'] / df['close']

    # Calculate Overnight Return
    df['Overnight Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Intraday and Overnight Returns
    df['Combined Return'] = df['Intraday Return'] + df['Overnight Return']

    # Compute Volume Weighted Average Price (VWAP)
    df['Typical Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume-Weighted Typical Price'] = df['Typical Price'] * df['volume']
    vwap = df['Volume-Weighted Typical Price'].sum() / df['volume'].sum()
    df['VWAP'] = vwap

    # Calculate VWAP Reversal Indicator
    df['Reversal Indicator'] = 0
    df.loc[df['VWAP'] > df['close'], 'Reversal Indicator'] = 1
    df.loc[df['VWAP'] < df['close'], 'Reversal Indicator'] = -1

    # Calculate Volume Momentum
    df['20-Day MA Volume'] = df['volume'].rolling(window=20).mean()
    df['Volume Momentum'] = 0
    df.loc[df['volume'] > df['20-Day MA Volume'], 'Volume Momentum'] = 1
    df.loc[df['volume'] < df['20-Day MA Volume'], 'Volume Momentum'] = -1

    # Integrate Reversal Indicator with Combined Return and Volume Momentum
    df['Final Alpha Factor'] = df['Combined Return'] * df['Reversal Indicator'] * df['Volume Momentum']

    return df['Final Alpha Factor']
```
