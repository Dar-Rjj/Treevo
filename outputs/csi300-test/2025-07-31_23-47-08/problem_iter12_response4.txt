```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday Range'] = df['high'] - df['low']
    df['Intraday Return'] = df['Intraday Range'] / df['close']

    # Calculate Overnight Return
    df['Overnight Return'] = (df['open'].shift(-1) - df['close']) / df['close']

    # Combine Intraday and Overnight Returns
    df['Combined Return'] = df['Intraday Return'] + df['Overnight Return']

    # Compute Volume-Weighted Average Price (VWAP)
    df['Typical Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volume-Weighted Typical Price'] = df['Typical Price'] * df['volume']
    total_volume = df['volume'].sum()
    df['VWAP'] = df['Volume-Weighted Typical Price'].sum() / total_volume

    # Calculate VWAP Reversal Indicator
    df['Reversal Indicator'] = np.where(df['VWAP'] > df['close'], 1, -1)

    # Integrate Reversal Indicator with Combined Return
    df['Integrated Reversal Return'] = df['Combined Return'] * df['Reversal Indicator']

    # Calculate Volume-Weighted Return
    df['Return'] = (df['close'] - df['open']) / df['open']
    df['Volume-Weighted Return'] = (df['Return'] * df['volume']).sum() / total_volume

    # Integrate Volume-Weighted Return with Integrated Reversal Return
    df['Final Alpha Factor'] = df['Integrated Reversal Return'] + df['Volume-Weighted Return']

    # Incorporate Volatility
    df['Daily Log Return'] = np.log(df['open'] / df['close'].shift(1))
    df['Volatility'] = df['Daily Log Return'].rolling(window=20).std()

    # Incorporate Simple Moving Average (SMA)
    df['SMA'] = df['close'].rolling(window=20).mean()

    # Calculate Trend Indicator
    df['Trend Indicator'] = np.where(df['close'] > df['SMA'], 1, -1)

    # Integrate Trend Indicator with Final Alpha Factor
    df['Integrated Alpha Factor'] = df['Final Alpha Factor'] * df['Trend Indicator']

    # Incorporate Transaction Amount
    df['Amount-Weighted Close'] = df['close'] * df['amount']
    total_amount = df['amount'].sum()
    df['Amount-Weighted Average Price'] = df['Amount-Weighted Close'].sum() / total_amount
    df['Amount-Weighted Delta'] = df['Amount-Weighted Average Price'] - df['close']

    # Integrate Amount-Weighted Delta with Integrated Alpha Factor
    df['Enhanced Final Alpha Factor'] = df['Integrated Alpha Factor'] + df['Amount-Weighted Delta']

    # Final Alpha Factor
    final_alpha_factor = df['Enhanced Final Alpha Factor']

    return final_alpha_factor
```
