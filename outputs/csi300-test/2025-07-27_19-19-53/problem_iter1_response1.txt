def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow_multiplier = (typical_price - df['close'].shift(1)) / (df['high'] - df['low'] + 1e-7)
    money_flow_volume = money_flow_multiplier * df['volume']
    positive_money_flow = money_flow_volume.where(money_flow_volume > 0, 0).rolling(window=14).sum()
    negative_money_flow = money_flow_volume.where(money_flow_volume < 0, 0).rolling(window=14).sum().abs()
    money_ratio = positive_money_flow / negative_money_flow
    return 100 - (100 / (1 + money_ratio))
