def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the relative change in volume compared to a 10-day moving average of volume
    vol_relative_change = (df['volume'] - df['volume'].rolling(window=10).mean()) / (df['volume'].rolling(window=10).std() + 1e-7)
    # Calculate the price momentum by taking the difference between current close price and 20-day moving average of close prices
    mom = df['close'] - df['close'].rolling(window=20).mean()
    # Combine the two factors in a non-linear way
    return (mom * (1 + vol_relative_change)) / (vol_relative_change.abs() + 1)
