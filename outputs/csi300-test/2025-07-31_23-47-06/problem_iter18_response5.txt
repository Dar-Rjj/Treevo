import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    intraday_price_movement = high_low_range + close_open_diff

    # Incorporate Volume Influence
    volume_adjusted_momentum = intraday_price_movement * df['volume']
    volume_adjusted_momentum_smoothed = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()

    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].pct_change()
    absolute_daily_return = daily_return.abs()
    dynamic_market_volatility = absolute_daily_return.rolling(window=15).std()
    modified_volume_adjusted_momentum = volume_adjusted_momentum / dynamic_market_volatility

    # Integrate Non-Linear Transformation
    non_linear_transformed = np.sqrt(np.abs(modified_volume_adjusted_momentum)) + \
                             np.log1p(np.abs(modified_volume_adjusted_momentum))

    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    trend_reversal_signal = momentum_reversal.apply(lambda x: 1 if x > 0 else -1)

    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    smoothed_trend_reversal_signal = trend_reversal_signal.ewm(span=5, adjust=False).mean() + \
                                     trend_reversal_signal.ewm(span=10, adjust=False).mean()
    combined_signal = smoothed_trend_reversal_signal + non_linear_transformed

    # Refine Final Alpha Factor
    interim_alpha_factor = combined_signal
    final_alpha_factor = interim_alpha_factor.ewm(span=5, adjust=False).mean() + \
                         interim_alpha_factor.ewm(span=10, adjust=False).mean()

    return final_alpha_factor
