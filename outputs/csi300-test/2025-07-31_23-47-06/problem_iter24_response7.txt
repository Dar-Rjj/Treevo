import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    intraday_price_movement = (high_low_range + close_open_diff) / 2

    # Incorporate Volume Influence
    volume_adjusted_momentum = intraday_price_movement * df['volume']
    volume_adjusted_momentum_ema = volume_adjusted_momentum.ewm(span=10).mean()

    # Adjust for Market Volatility
    daily_return = df['close'].pct_change()
    abs_daily_return = daily_return.abs()
    market_volatility = daily_return.rolling(window=30).std()
    cross_sectional_volatility = abs_daily_return.median()
    dynamic_market_volatility = market_volatility * cross_sectional_volatility

    modified_volume_adjusted_momentum = volume_adjusted_momentum_ema - dynamic_market_volatility

    # Integrate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5).mean()
    long_term_momentum = df['close'].ewm(span=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum

    reversal_signal = np.sign(momentum_reversal)

    # Combine Modified Momentum with Reversal Signal
    combined_alpha_factor = modified_volume_adjusted_momentum + reversal_signal

    # Integrate Non-Linear Transformation
    sqrt_transformed_alpha = np.sqrt(combined_alpha_factor)
    log_transformed_alpha = np.log(combined_alpha_factor + 1e-6)  # Add small constant to avoid log(0)

    # Refine Final Alpha Factor
    final_alpha_factor = (sqrt_transformed_alpha + log_transformed_alpha) / 2
    final_alpha_factor_smoothed = final_alpha_factor.ewm(span=5).mean()

    return final_alpha_factor_smoothed
