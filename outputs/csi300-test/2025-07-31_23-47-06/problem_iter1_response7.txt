```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Intraday Volatility
    df['intraday_vol'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Detect Volume Spikes
    df['vol_mean'] = df['volume'].rolling(window=n).mean()
    df['vol_std'] = df['volume'].rolling(window=n).std()
    df['is_volume_spike'] = (df['volume'] > (df['vol_mean'] + 2 * df['vol_std'])).astype(int)
    
    # Calculate Raw Close-to-Open Return
    df['raw_c_to_o_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Adjust by Intraday Volatility
    df['adjusted_c_to_o_return'] = df['raw_c_to_o_return'] / df['intraday_vol']
    
    # Multiply by Indicator for Volume Spike
    df['final_factor'] = df['adjusted_c_to_o_return'] * df['is_volume_spike']

    return df['final_factor']
```
