```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (high_low_range + close_open_diff) * df['volume']
    smoothed_volume_adjusted_momentum = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].pct_change()
    abs_daily_return = daily_return.abs()
    dynamic_market_volatility = abs_daily_return.rolling(window=10).std()
    modified_volume_adjusted_momentum = smoothed_volume_adjusted_momentum / dynamic_market_volatility
    
    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(modified_volume_adjusted_momentum)
    log_transformed_momentum = np.log(modified_volume_adjusted_momentum)
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_signal = momentum_reversal.apply(lambda x: 1 if x > 0 else -1)
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    advanced_adaptive_smoothing = (reversal_signal.ewm(span=5, adjust=False).mean() + 
                                   reversal_signal.ewm(span=10, adjust=False).mean()) / 2
    combined_signal = sqrt_transformed_momentum + log_transformed_momentum + advanced_adaptive_smoothing
    
    # Refine Final Alpha Factor
    interim_alpha_factor = combined_signal
    final_alpha_factor = (interim_alpha_factor.ewm(span=5, adjust=False).mean() + 
                          interim_alpha_factor.ewm(span=10, adjust=False).mean()) / 2
    
    return final_alpha_factor
```
