```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum'] = df['Close_Open_Diff'] * df['volume']
    df['Volume_Adjusted_Smoothed'] = df['Volume_Adjusted_Momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    df['Daily_Return'] = df['close'].diff()
    df['Absolute_Daily_Return'] = df['Daily_Return'].abs()
    df['Market_Volatility'] = df['Daily_Return'].rolling(window=30).std()
    df['Volatility_Adjusted_Momentum'] = df['Volume_Adjusted_Smoothed'] / df['Market_Volatility']
    
    # Incorporate Trend Reversal Signal
    df['Short_Term_Momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_Term_Momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    df['Reversal_Signal'] = np.sign(df['Momentum_Reversal'])
    
    # Integrate Non-Linear Transformation
    df['Sqrt_Transformed'] = np.sqrt(np.abs(df['Volatility_Adjusted_Momentum']))
    df['Log_Transformed'] = np.log1p(np.abs(df['Volatility_Adjusted_Momentum']))
    
    # Enhance Reversal Signal with Adaptive Smoothing
    df['Smoothed_Reversal_Signal'] = df['Reversal_Signal'].ewm(span=5, adjust=False).mean()
    df['Combined_Factor'] = df['Sqrt_Transformed'] + df['Log_Transformed'] + df['Smoothed_Reversal_Signal']
    
    # Final Adaptive Smoothing
    df['Final_Alpha_Factor'] = df['Combined_Factor'].ewm(span=5, adjust=False).mean()
    
    return df['Final_Alpha_Factor']
```
