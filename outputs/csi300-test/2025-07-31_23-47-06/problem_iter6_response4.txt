```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    df['intraday_price_movement'] = df['high_low_range'] + df['close_open_diff']

    # Incorporate Volume Influence
    df['volume_adjusted_momentum'] = df['intraday_price_movement'] * df['volume']
    df['sma_volume_adj_mom'] = df['volume_adjusted_momentum'].rolling(window=10).mean()

    # Adjust for Market Volatility
    df['daily_return'] = df['close'].pct_change()
    df['abs_return_diff'] = df['daily_return'].abs()
    df['market_volatility'] = df['daily_return'].rolling(window=30).std()
    df['adjusted_momentum'] = df['sma_volume_adj_mom'] - df['market_volatility']

    # Incorporate Trend Reversal Signal
    df['short_term_mom'] = df['close'] - df['close'].rolling(window=5).mean()
    df['long_term_mom'] = df['close'] - df['close'].rolling(window=20).mean()
    df['momentum_reversal'] = df['short_term_mom'] - df['long_term_mom']
    df['reversal_signal'] = df['momentum_reversal'].apply(lambda x: 1 if x > 0 else -1)
    df['interim_alpha_factor'] = df['adjusted_momentum'] + df['reversal_signal']

    # Incorporate Relative Strength
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['ema_60'] = df['close'].ewm(span=60, adjust=False).mean()
    df['relative_strength'] = (df['ema_20'] - df['ema_60']) / df['ema_60']

    # Incorporate Liquidity Measures
    df['turnover_ratio'] = df['volume'] / df['floating_shares']
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    df['liquidity_measure'] = df['turnover_ratio'] * df['atr_14']

    # Refine Final Alpha Factor
    df['final_alpha_factor'] = (df['interim_alpha_factor'] + 
                                df['relative_strength'] + 
                                df['liquidity_measure'])
    df['final_alpha_factor'] = df['final_alpha_factor'].rolling(window=5).mean()

    return df['final_alpha_factor']
```
