```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = close_open_diff * df['volume']
    volume_adjusted_momentum_sma = volume_adjusted_momentum.rolling(window=10).mean()
    
    # Adjust for Market Volatility
    daily_returns = df['close'].pct_change()
    market_volatility = daily_returns.rolling(window=30).std()
    
    # Final Alpha Factor
    alpha_factor = volume_adjusted_momentum_sma - market_volatility
    
    return alpha_factor
```
