```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['volume_adjusted_momentum'] = (df['high_low_range'] + df['close_open_diff']) * df['volume']
    df['ema_10_vol_adj_mom'] = df['volume_adjusted_momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    df['daily_return'] = df['close'].pct_change()
    df['abs_return_diff'] = df['daily_return'].abs()
    df['std_dev_30'] = df['daily_return'].rolling(window=30).std()
    df['adj_momentum'] = df['ema_10_vol_adj_mom'] - df['std_dev_30']
    
    # Enhance Trend Reversal Signal
    df['ema_5_close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20_close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_reversal'] = df['ema_5_close'] - df['ema_20_close']
    df['reversal_signal'] = np.where(df['momentum_reversal'] > 0, 1, -1)
    
    # Integrate Reversal Signal into Final Factor
    df['interim_alpha_factor'] = df['adj_momentum'] + df['reversal_signal']
    
    # Refine Final Alpha Factor
    df['final_alpha_factor'] = df['interim_alpha_factor'].ewm(span=5, adjust=False).mean()
    
    return df['final_alpha_factor']
```
