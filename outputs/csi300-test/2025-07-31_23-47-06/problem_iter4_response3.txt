```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (close_open_diff * df['volume']).rolling(window=10).mean()
    
    # Adjust for Market Volatility
    daily_return = df['close'].pct_change()
    abs_return_diff = daily_return.abs().rolling(window=30).std()
    adjusted_momentum = volume_adjusted_momentum - abs_return_diff
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'] - df['close'].rolling(window=5).mean()
    long_term_momentum = df['close'] - df['close'].rolling(window=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_signal = (momentum_reversal > 0).astype(int) - (momentum_reversal < 0).astype(int)
    
    # Integrate Reversal Signal into Final Factor
    final_factor = adjusted_momentum + reversal_signal
    
    return final_factor
```
