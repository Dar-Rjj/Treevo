```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Volume Adjusted Momentum
    volume_adjusted_momentum = df['volume'] * close_open_spread
    
    # Combine Factors
    combined_factors = high_low_range + volume_adjusted_momentum
    
    # Normalize by Close price
    factor_values = combined_factors / df['close']
    
    return factor_values
```
