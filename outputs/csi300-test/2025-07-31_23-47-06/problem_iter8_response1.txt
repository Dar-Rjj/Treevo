```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (high_low_range + close_open_diff) * df['volume']
    smoothed_momentum = volume_adjusted_momentum.rolling(window=10).mean()
    
    # Adjust for Market Volatility
    daily_return = df['close'].pct_change()
    market_volatility = daily_return.rolling(window=30).std()
    adjusted_momentum = smoothed_momentum - market_volatility
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].rolling(window=5).mean()
    long_term_momentum = df['close'].rolling(window=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    
    # Identify Reversal Points
    reversal_signal = momentum_reversal.apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Integrate Reversal Signal into Final Factor
    final_factor = adjusted_momentum + reversal_signal
    
    return final_factor
```
