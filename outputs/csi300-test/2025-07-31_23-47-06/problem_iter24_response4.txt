import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (high_low_range + close_open_diff) * df['volume']
    smoothed_volume_momentum = volume_adjusted_momentum.ewm(span=5, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].diff()
    abs_daily_return = daily_return.abs()
    dynamic_market_volatility = daily_return.rolling(window=10).std()
    modified_momentum = smoothed_volume_momentum / dynamic_market_volatility
    
    # Integrate Non-Linear Transformation
    sqrt_transformed = np.sqrt(modified_momentum)
    log_transformed = np.log(modified_momentum)
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    advanced_smoothed_reversal = 0.5 * momentum_reversal.ewm(span=5, adjust=False).mean() + \
                                0.5 * momentum_reversal.ewm(span=10, adjust=False).mean()
    combined_signal = advanced_smoothed_reversal + 0.5 * (sqrt_transformed + log_transformed)
    
    # Refine Final Alpha Factor
    final_alpha_factor = 0.5 * combined_signal.ewm(span=5, adjust=False).mean() + \
                        0.5 * combined_signal.ewm(span=10, adjust=False).mean()
    
    return final_alpha_factor.dropna()
