```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    intraday_price_movement = (high_low_range + close_open_diff) / 2
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = intraday_price_movement * df['volume']
    volume_adjusted_ema = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].pct_change()
    absolute_daily_return = daily_return.abs()
    dynamic_market_volatility = absolute_daily_return.rolling(window=20).std()
    modified_volume_adjusted_momentum = volume_adjusted_ema / dynamic_market_volatility
    
    # Integrate Non-Linear Transformation
    sqrt_transformed = np.sqrt(modified_volume_adjusted_momentum)
    log_transformed = np.log(modified_volume_adjusted_momentum + 1e-10)
    
    # Incorporate Early Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=30, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_points = np.sign(momentum_reversal)
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    reversal_signal_ema_short = momentum_reversal.ewm(span=5, adjust=False).mean()
    reversal_signal_ema_long = momentum_reversal.ewm(span=10, adjust=False).mean()
    smoothed_reversal_signal = (reversal_signal_ema_short + reversal_signal_ema_long) / 2
    combined_signal = smoothed_reversal_signal + (sqrt_transformed + log_transformed) / 2
    
    # Refine Final Alpha Factor
    final_alpha_factor = (combined_signal.ewm(span=5, adjust=False).mean() + 
                          combined_signal.ewm(span=10, adjust=False).mean()) / 2
    
    return final_alpha_factor
```
