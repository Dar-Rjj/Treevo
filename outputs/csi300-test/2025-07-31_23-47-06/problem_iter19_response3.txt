```python
import pandas as pd
import numpy as np
from scipy.stats import median_abs_deviation
from talib import EMA

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    df['intraday_price_movement'] = df['high_low_range'] + df['close_open_diff']

    # Incorporate Volume Influence
    df['volume_adjusted_momentum'] = df['intraday_price_movement'] * df['volume']
    
    # Adaptive Smoothing via Moving Average
    def dynamic_ema_period(std_dev, min_periods=5, max_periods=30):
        return int(np.clip((std_dev * 10), min_periods, max_periods))

    df['daily_return'] = df['close'].pct_change()
    recent_volatility = df['daily_return'].rolling(window=30).std()
    ema_period = recent_volatility.apply(dynamic_ema_period)
    df['smoothed_volume_adj_momentum'] = df[['volume_adjusted_momentum']].apply(lambda x: EMA(x, timeperiod=ema_period.iloc[-1]))

    # Adjust for Market Volatility
    df['abs_daily_return'] = df['daily_return'].abs()
    df['robust_market_volatility'] = df['abs_daily_return'].rolling(window=30).apply(median_abs_deviation, raw=False)
    df['vol_adjusted_smoothed_momentum'] = df['smoothed_volume_adj_momentum'] / df['robust_market_volatility']

    # Incorporate Trend Reversal Signal
    df['short_term_momentum'] = EMA(df['close'], timeperiod=5)
    df['long_term_momentum'] = EMA(df['close'], timeperiod=20)
    df['momentum_reversal'] = df['short_term_momentum'] - df['long_term_momentum']
    df['reversal_signal'] = np.sign(df['momentum_reversal'])

    # Integrate Non-Linear Transformation
    df['sqrt_transformed_momentum'] = np.sqrt(np.abs(df['vol_adjusted_smoothed_momentum']))
    df['log_transformed_momentum'] = np.log(np.abs(df['vol_adjusted_smoothed_momentum']) + 1)

    # Enhance Reversal Signal with Adaptive Smoothing
    df['smoothed_reversal_signal'] = df[['reversal_signal']].apply(lambda x: EMA(x, timeperiod=ema_period.iloc[-1]))
    df['combined_signal'] = df['sqrt_transformed_momentum'] + df['log_transformed_momentum'] + df['smoothed_reversal_signal']

    # Final Adaptive Smoothing
    df['final_alpha_factor'] = EMA(df['combined_signal'], timeperiod=ema_period.iloc[-1])

    return df['final_alpha_factor']
```
