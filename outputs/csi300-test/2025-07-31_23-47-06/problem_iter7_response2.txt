```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (close_open_diff * df['volume']).ewm(span=10).mean()
    
    # Adjust for Market Volatility
    daily_return = df['close'].pct_change()
    abs_return_diff = daily_return.abs()
    std_dev_daily_return = daily_return.rolling(window=30).std()
    adjusted_momentum = volume_adjusted_momentum / std_dev_daily_return
    
    # Enhance Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5).mean()
    long_term_momentum = df['close'].ewm(span=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_signal = np.where(momentum_reversal > 0, 1, -1)
    
    interim_alpha_factor = adjusted_momentum + reversal_signal
    
    # Refine Final Alpha Factor
    final_alpha_factor = interim_alpha_factor.ewm(span=5).mean()
    
    return final_alpha_factor
```
