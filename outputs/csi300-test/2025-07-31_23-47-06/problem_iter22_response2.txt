import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (close_open_diff * df['volume']).rolling(window=10).mean()
    
    # Adjust for Market Volatility
    daily_return = df['close'].diff()
    abs_daily_return = daily_return.abs()
    market_volatility = daily_return.rolling(window=30).std()
    volatility_adjusted_momentum = volume_adjusted_momentum / market_volatility
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_points = np.sign(momentum_reversal).diff().ne(0)
    
    # Combine Signals Before Non-Linear Transformations
    combined_signal = (volatility_adjusted_momentum + momentum_reversal).ewm(span=5, adjust=False).mean()
    
    # Integrate Non-Linear Transformation
    sqrt_transformed = np.sqrt(combined_signal)
    log_transformed = np.log(combined_signal + 1)  # Adding 1 to avoid log of negative or zero
    
    # Refine Final Alpha Factor
    interim_alpha_factor = sqrt_transformed + log_transformed
    final_alpha_factor = interim_alpha_factor.ewm(span=5, adjust=False).mean()
    
    return final_alpha_factor
