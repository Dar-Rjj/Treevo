```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High-Low Range'] = df['high'] - df['low']
    df['Close-Open Difference'] = df['close'] - df['open']

    # Incorporate Volume Influence
    df['Volume-Adjusted Momentum'] = (df['High-Low Range'] + df['Close-Open Difference']) * df['volume']
    df['Volume-Adjusted Momentum EMA'] = df['Volume-Adjusted Momentum'].ewm(span=10).mean()

    # Adjust for Market Volatility
    df['Daily Return'] = df['close'].diff()
    df['Absolute Return Difference'] = df['Daily Return'].abs()
    df['Volatility_30d'] = df['Daily Return'].rolling(window=30).std()
    df['Adjusted Momentum'] = df['Volume-Adjusted Momentum EMA'] / df['Volatility_30d']

    # Enhance Trend Reversal Signal
    df['Short-Term Momentum'] = df['close'].ewm(span=5).mean()
    df['Long-Term Momentum'] = df['close'].ewm(span=20).mean()
    df['Momentum Reversal'] = df['Short-Term Momentum'] - df['Long-Term Momentum']
    
    # Integrate Reversal Signal into Final Factor
    df['Final Alpha Factor'] = df['Momentum Reversal'] + df['Adjusted Momentum']
    
    # Refine Final Alpha Factor
    df['Final Alpha Factor EMA'] = df['Final Alpha Factor'].ewm(span=5).mean()

    return df['Final Alpha Factor EMA']
```
