```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Calculate Intraday Price Movement
    intraday_movement = (high_low_range + close_open_diff) / 2
    
    # Volume-Adjusted Momentum
    volume_adjusted_momentum = intraday_movement * df['volume']
    
    # Smoothing via 10-day SMA
    volume_adjusted_momentum_sma10 = volume_adjusted_momentum.rolling(window=10).mean()
    
    # Daily Return
    daily_return = df['close'].pct_change()
    
    # Absolute Return Difference
    abs_return_diff = daily_return.abs()
    
    # Market Volatility (Standard Deviation of Daily Returns over a 30-day period)
    market_volatility = daily_return.rolling(window=30).std()
    
    # Adjust Intraday Momentum with Market Volatility
    adjusted_momentum = volume_adjusted_momentum_sma10 - market_volatility
    
    # Short-Term Momentum (5-day SMA of Close prices)
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Long-Term Momentum (20-day SMA of Close prices)
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Compute Momentum Reversal
    momentum_reversal = short_term_momentum - long_term_momentum
    
    # Integrate Reversal Signal into Final Factor
    reversal_signal = momentum_reversal.apply(lambda x: 1 if x > 0 else -1)
    interim_alpha_factor = adjusted_momentum + reversal_signal
    
    # Refine Final Alpha Factor with 5-day SMA
    final_alpha_factor = interim_alpha_factor.rolling(window=5).mean()
    
    return final_alpha_factor
```
