```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    vol_adjusted_high_low = high_low_range * df['volume']
    vol_adjusted_close_open = close_open_diff * df['volume']
    vol_adjusted_momentum = (vol_adjusted_high_low + vol_adjusted_close_open) / 2
    
    # Smoothing via Moving Average
    smoothed_momentum = vol_adjusted_momentum.rolling(window=10).mean()
    
    # Adjust for Market Volatility
    daily_returns = df['close'].pct_change()
    market_volatility = daily_returns.rolling(window=30).std()
    adjusted_momentum = smoothed_momentum - market_volatility
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'] - df['close'].rolling(window=5).mean()
    long_term_momentum = df['close'] - df['close'].rolling(window=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    
    # Compute Reversal Points
    reversal_signal = np.sign(momentum_reversal)
    
    # Integrate Reversal Signal into Final Factor
    final_factor = adjusted_momentum + reversal_signal
    
    return final_factor
```
