import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_high_low = df['high'] - df['low']
    intraday_close_open = df['close'] - df['open']

    # Incorporate Volume Influence
    volume_adjusted_momentum = (intraday_high_low + intraday_close_open) * df['volume']
    volume_adjusted_momentum_smoothed = volume_adjusted_momentum.ewm(span=10).mean()

    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].pct_change()
    abs_daily_return = daily_return.abs()
    dynamic_market_volatility = daily_return.rolling(window=10).std()
    
    modified_volume_adjusted_momentum = volume_adjusted_momentum_smoothed / dynamic_market_volatility

    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(modified_volume_adjusted_momentum)
    log_transformed_momentum = np.log(modified_volume_adjusted_momentum)

    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5).mean()
    long_term_momentum = df['close'].ewm(span=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum

    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    advanced_smoothed_reversal = (momentum_reversal.ewm(span=5).mean() + momentum_reversal.ewm(span=10).mean()) / 2

    # Combine Smoothed Reversal Signal and Non-Linearly Transformed Momentum
    interim_alpha_factor = (sqrt_transformed_momentum + log_transformed_momentum) + advanced_smoothed_reversal

    # Final Advanced Adaptive Smoothing
    final_alpha_factor = (interim_alpha_factor.ewm(span=5).mean() + interim_alpha_factor.ewm(span=10).mean()) / 2

    return final_alpha_factor
