```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']

    # Volume-Adjusted Momentum
    volume_adjusted_momentum = (df['volume'] * (high_low_range + close_open_diff)) / 2

    # Adaptive Smoothing via Moving Average
    volume_adjusted_momentum_ema = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()

    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].pct_change()
    absolute_daily_return = abs(daily_return)
    dynamic_market_volatility = absolute_daily_return.rolling(window=10).std()

    adjusted_volume_momentum = volume_adjusted_momentum_ema / dynamic_market_volatility

    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum

    reversal_signal = np.sign(momentum_reversal)

    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    reversal_signal_5_ema = reversal_signal.ewm(span=5, adjust=False).mean()
    reversal_signal_10_ema = reversal_signal.ewm(span=10, adjust=False).mean()
    smoothed_reversal_signal = (reversal_signal_5_ema + reversal_signal_10_ema) / 2

    # Refine Final Alpha Factor
    combined_signal = smoothed_reversal_signal * adjusted_volume_momentum
    final_alpha_factor_5_ema = combined_signal.ewm(span=5, adjust=False).mean()
    final_alpha_factor_10_ema = combined_signal.ewm(span=10, adjust=False).mean()
    final_alpha_factor = (final_alpha_factor_5_ema + final_alpha_factor_10_ema) / 2

    return final_alpha_factor
```
