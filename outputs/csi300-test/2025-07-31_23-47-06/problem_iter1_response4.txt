```python
def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Average Range over 5 days
    df['average_range_5d'] = df['daily_range'].rolling(window=5).mean()
    
    # Current Day Range (same as daily range, but for clarity we rename it)
    df['current_day_range'] = df['daily_range']
    
    # Compute Momentum
    df['momentum'] = (df['current_day_range'] - df['average_range_5d']) * df['volume']
    
    return df['momentum']
```
