```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Cumulative Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Smoothed Cumulative Return
    smoothing_factor = 0.1
    df['smoothed_return'] = df['daily_return'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Identify Volume Spikes
    df['volume_change'] = df['volume'].diff()
    spike_multiplier = 2
    df['volume_spike_threshold'] = spike_multiplier * df['volume_change'].std()
    df['is_volume_spike'] = df['volume_change'] > df['volume_spike_threshold']
    
    # Combine Momentum and Volume Breakout
    large_positive_value = 5
    df['combined_factor'] = df['smoothed_return']
    df.loc[df['is_volume_spike'], 'combined_factor'] += large_positive_value
    
    return df['combined_factor']
```
