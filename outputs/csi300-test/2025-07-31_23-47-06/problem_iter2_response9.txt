```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Intraday Price Movement
    intraday_movement = (high_low_range + close_open_diff) / 2
    
    # Volume Adjusted Momentum
    volume_adjusted_momentum = intraday_movement * df['volume']
    
    # Smoothing via 10-day EMA
    ema_10_days = volume_adjusted_momentum.ewm(span=10).mean()
    
    # Calculate Absolute Return Difference
    abs_return_diff = df['close'].pct_change().abs()
    
    # Market Volatility (30-day Standard Deviation of Daily Returns)
    market_volatility = df['close'].pct_change().rolling(window=30).std()
    
    # Adjust for Market Volatility
    adjusted_momentum = ema_10_days - market_volatility
    
    # Calculate Short-Term and Long-Term Trends
    short_term_trend = df['close'].ewm(span=5).mean()
    long_term_trend = df['close'].ewm(span=20).mean()
    
    # Trend Reversal Signal
    trend_signal = np.where(short_term_trend > long_term_trend, 1,
                            np.where(short_term_trend < long_term_trend, -1, 0))
    trend_cross = trend_signal - trend_signal.shift(1)
    
    # Incorporate Trend Reversal into the Final Factor
    final_factor = adjusted_momentum * (1 + 0.5 * trend_cross)
    
    return final_factor
```
