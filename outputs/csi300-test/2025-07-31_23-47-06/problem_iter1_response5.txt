import pandas as pd

def heuristics_v2(df, n_days=10):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate N-day Cumulative Price Range
    df['cumulative_price_range'] = df['daily_price_range'].rolling(window=n_days).sum()
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open'].shift(1)
    
    # Compute the Ratio
    df['ratio'] = df['cumulative_price_range'] / df['close_to_open_return'].abs()
    
    return df['ratio']
