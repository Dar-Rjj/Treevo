```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday Price Indicators
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum_HL'] = df['High_Low_Range'] * df['volume']
    df['Volume_Adjusted_Momentum_CO'] = df['Close_Open_Diff'] * df['volume']
    df['VAM_10_EMA_HL'] = df['Volume_Adjusted_Momentum_HL'].ewm(span=10, adjust=False).mean()
    df['VAM_10_EMA_CO'] = df['Volume_Adjusted_Momentum_CO'].ewm(span=10, adjust=False).mean()

    # Adjust for Dynamic Market Volatility
    df['Daily_Returns'] = df['close'].pct_change()
    df['Absolute_Daily_Returns'] = df['Daily_Returns'].abs()
    df['Dynamic_Volatility'] = df['Daily_Returns'].rolling(window=5).std()
    df['Modified_VAM_HL'] = df['VAM_10_EMA_HL'] / df['Dynamic_Volatility']
    df['Modified_VAM_CO'] = df['VAM_10_EMA_CO'] / df['Dynamic_Volatility']

    # Integrate Non-Linear Transformations
    df['Sqrt_Modified_VAM_HL'] = np.sqrt(df['Modified_VAM_HL'])
    df['Log_Modified_VAM_CO'] = np.log(df['Modified_VAM_CO'])

    # Incorporate Early Reversal Signals
    df['Short_Term_Momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_Term_Momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    df['Reversal_Signal'] = np.sign(df['Momentum_Reversal'])

    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    df['Reversal_Signal_5_EMA'] = df['Reversal_Signal'].ewm(span=5, adjust=False).mean()
    df['Reversal_Signal_10_EMA'] = df['Reversal_Signal'].ewm(span=10, adjust=False).mean()
    df['Smoothed_Reversal_Signal'] = (df['Reversal_Signal_5_EMA'] + df['Reversal_Signal_10_EMA']) / 2
    df['Combined_Signal'] = df['Sqrt_Modified_VAM_HL'] + df['Log_Modified_VAM_CO'] + df['Smoothed_Reversal_Signal']

    # Refine Final Alpha Factor
    df['Interim_Alpha_Factor'] = df['Combined_Signal']
    df['Final_Alpha_Factor_5_EMA'] = df['Interim_Alpha_Factor'].ewm(span=5, adjust=False).mean()
    df['Final_Alpha_Factor_10_EMA'] = df['Interim_Alpha_Factor'].ewm(span=10, adjust=False).mean()
    df['Final_Alpha_Factor'] = (df['Final_Alpha_Factor_5_EMA'] + df['Final_Alpha_Factor_10_EMA']) / 2

    return df['Final_Alpha_Factor']
```
