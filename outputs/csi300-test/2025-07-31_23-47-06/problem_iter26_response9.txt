```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday Price Indicators
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']

    # Incorporate Volume Influence
    df['vol_adj_mom_high_low'] = df['high_low_range'] * df['volume']
    df['vol_adj_mom_close_open'] = df['close_open_diff'] * df['volume']
    
    # Apply Adaptive Smoothing
    df['ema_vol_adj_mom_high_low'] = df['vol_adj_mom_high_low'].ewm(span=10, adjust=False).mean()
    df['ema_vol_adj_mom_close_open'] = df['vol_adj_mom_close_open'].ewm(span=10, adjust=False).mean()

    # Adjust for Dynamic Market Volatility
    df['daily_returns'] = df['close'].pct_change()
    df['abs_daily_returns'] = df['daily_returns'].abs()
    df['dynamic_market_volatility'] = df['daily_returns'].rolling(window=10).std()
    
    # Modify Volume-Adjusted Momentum with Dynamic Market Volatility
    df['modified_vol_adj_mom_high_low'] = df['ema_vol_adj_mom_high_low'] / df['dynamic_market_volatility']
    df['modified_vol_adj_mom_close_open'] = df['ema_vol_adj_mom_close_open'] / df['dynamic_market_volatility']

    # Integrate Non-Linear Transformations
    df['sqrt_modified_vol_adj_mom'] = np.sqrt((df['modified_vol_adj_mom_high_low'] + df['modified_vol_adj_mom_close_open']) / 2)
    df['log_modified_vol_adj_mom'] = np.log((df['modified_vol_adj_mom_high_low'] + df['modified_vol_adj_mom_close_open']) / 2)

    # Incorporate Early Reversal Signals
    df['short_term_mom'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_mom'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_reversal'] = df['short_term_mom'] - df['long_term_mom']
    df['early_signal_indicator'] = df['short_term_mom'].ewm(span=3, adjust=False).mean() - df['long_term_mom'].ewm(span=3, adjust=False).mean()

    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    df['smoothed_reversal_signal_5'] = df['momentum_reversal'].ewm(span=5, adjust=False).mean()
    df['smoothed_reversal_signal_10'] = df['momentum_reversal'].ewm(span=10, adjust=False).mean()
    df['combined_smoothed_reversal'] = (df['smoothed_reversal_signal_5'] + df['smoothed_reversal_signal_10']) / 2

    df['interim_alpha_factor'] = df['combined_smoothed_reversal'] + df['sqrt_modified_vol_adj_mom'] + df['log_modified_vol_adj_mom']

    # Refine Final Alpha Factor
    df['final_alpha_factor_5'] = df['interim_alpha_factor'].ewm(span=5, adjust=False).mean()
    df['final_alpha_factor_10'] = df['interim_alpha_factor'].ewm(span=10, adjust=False).mean()
    df['final_alpha_factor'] = (df['final_alpha_factor_5'] + df['final_alpha_factor_10']) / 2

    return df['final_alpha_factor']
```
