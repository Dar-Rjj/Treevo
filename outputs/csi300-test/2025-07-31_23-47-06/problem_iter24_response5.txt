import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (high_low_range + close_open_diff) * df['volume']
    volume_adjusted_momentum_smoothed = volume_adjusted_momentum.ewm(span=10).mean()
    
    # Adjust for Dynamic Market Volatility
    daily_returns = df['close'].pct_change()
    abs_daily_returns = daily_returns.abs()
    dynamic_volatility = daily_returns.rolling(window=5).std()
    modified_volume_adjusted_momentum = volume_adjusted_momentum / (dynamic_volatility + 1e-8)
    
    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(modified_volume_adjusted_momentum)
    log_transformed_momentum = np.log(modified_volume_adjusted_momentum + 1e-8)
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5).mean()
    long_term_momentum = df['close'].ewm(span=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_points = momentum_reversal.apply(lambda x: 1 if x > 0 else -1)
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    smoothed_reversal_5_day = reversal_points.ewm(span=5).mean()
    smoothed_reversal_10_day = reversal_points.ewm(span=10).mean()
    combined_smoothed_reversal = (smoothed_reversal_5_day + smoothed_reversal_10_day) / 2
    
    # Combine Smoothed Reversal Signal with Non-Linearly Transformed Momentum
    interim_alpha_factor = combined_smoothed_reversal + (sqrt_transformed_momentum + log_transformed_momentum) / 2
    
    # Refine Final Alpha Factor
    final_alpha_factor_5_day = interim_alpha_factor.ewm(span=5).mean()
    final_alpha_factor_10_day = interim_alpha_factor.ewm(span=10).mean()
    final_alpha_factor = (final_alpha_factor_5_day + final_alpha_factor_10_day) / 2
    
    return final_alpha_factor
