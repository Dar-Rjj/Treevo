```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Intraday Price Movement: combine high-low range and close-open difference
    intraday_movement = 0.5 * (high_low_range + close_open_diff)
    
    # Volume Adjusted Momentum: multiply intraday movement by volume
    volume_adjusted_momentum = intraday_movement * df['volume']
    
    # Smoothing via 10-day Simple Moving Average
    sma_10 = volume_adjusted_momentum.rolling(window=10).mean()
    
    # Calculate Absolute Return Difference
    absolute_return_diff = df['close'].pct_change().abs()
    
    # Calculate Market Volatility (30-day standard deviation of daily returns)
    market_volatility = df['close'].pct_change().rolling(window=30).std()
    
    # Modify Intraday Momentum with Market Volatility
    adjusted_intraday_momentum = sma_10 - market_volatility
    
    return adjusted_intraday_momentum
```
