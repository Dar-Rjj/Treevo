```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (high_low_range + close_open_diff) * df['volume']
    volume_adjusted_momentum_smoothed = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    daily_returns = df['close'].pct_change()
    absolute_daily_returns = abs(daily_returns)
    market_volatility = absolute_daily_returns.rolling(window=30).std()
    volatility_adjusted_momentum = volume_adjusted_momentum_smoothed - market_volatility
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_signal = momentum_reversal.apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    smoothed_reversal_signal = reversal_signal.ewm(span=5, adjust=False).mean()
    
    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(volatility_adjusted_momentum)
    log_transformed_momentum = np.log1p(volatility_adjusted_momentum)
    
    # Combine Smoothed Reversal Signal with Non-Linearly Transformed Momentum
    interim_alpha_factor = smoothed_reversal_signal + sqrt_transformed_momentum + log_transformed_momentum
    
    # Final Adaptive Smoothing
    final_alpha_factor = interim_alpha_factor.ewm(span=5, adjust=False).mean()
    
    return final_alpha_factor
```
