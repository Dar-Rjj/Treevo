```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday Price Indicators
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']

    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum_HL'] = df['volume'] * df['High_Low_Range']
    df['Volume_Adjusted_Momentum_CO'] = df['volume'] * df['Close_Open_Diff']
    df['Volume_Adjusted_Momentum'] = (df['Volume_Adjusted_Momentum_HL'] + df['Volume_Adjusted_Momentum_CO']) / 2

    # Apply Adaptive Smoothing
    df['Smoothed_VAM_10d'] = df['Volume_Adjusted_Momentum'].ewm(span=10, adjust=False).mean()

    # Adjust for Dynamic Market Volatility
    df['Daily_Returns'] = df['close'].pct_change()
    df['Abs_Daily_Returns'] = df['Daily_Returns'].abs()
    df['Dynamic_Volatility'] = df['Daily_Returns'].rolling(window=10).std()

    # Modify Volume-Adjusted Momentum with Dynamic Market Volatility
    df['Modified_VAM'] = df['Smoothed_VAM_10d'] / df['Dynamic_Volatility']

    # Integrate Non-Linear Transformations
    df['Sqrt_Modified_VAM'] = np.sqrt(df['Modified_VAM'])
    df['Log_Modified_VAM'] = np.log(df['Modified_VAM'])

    # Incorporate Early Reversal Signals
    df['Short_Term_Momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_Term_Momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']

    # Label Positive and Negative Reversals
    df['Reversal_Signal'] = np.where(df['Momentum_Reversal'] > 0, 1, -1)

    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    df['Smoothed_Reversal_5d'] = df['Reversal_Signal'].ewm(span=5, adjust=False).mean()
    df['Smoothed_Reversal_10d'] = df['Reversal_Signal'].ewm(span=10, adjust=False).mean()
    df['Combined_Smoothed_Reversal'] = (df['Smoothed_Reversal_5d'] + df['Smoothed_Reversal_10d']) / 2

    # Combine Smoothed Reversal Signal with Non-Linearly Transformed Momentum
    df['Interim_Alpha_Factor'] = df['Sqrt_Modified_VAM'] + df['Combined_Smoothed_Reversal']

    # Apply Final Advanced Adaptive Smoothing
    df['Final_Alpha_Factor_5d'] = df['Interim_Alpha_Factor'].ewm(span=5, adjust=False).mean()
    df['Final_Alpha_Factor_10d'] = df['Interim_Alpha_Factor'].ewm(span=10, adjust=False).mean()
    df['Final_Alpha_Factor'] = (df['Final_Alpha_Factor_5d'] + df['Final_Alpha_Factor_10d']) / 2

    return df['Final_Alpha_Factor'].dropna()
```
