```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    intraday_price_movement = (high_low_range + close_open_diff) / 2
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = intraday_price_movement * df['volume']
    smoothed_volume_momentum = volume_adjusted_momentum.ewm(span=10).mean()
    
    # Adjust for Market Volatility
    daily_return = df['close'].pct_change()
    absolute_daily_return = daily_return.abs()
    market_volatility = absolute_daily_return.rolling(window=30).std()
    adjusted_momentum = smoothed_volume_momentum / market_volatility
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5).mean()
    long_term_momentum = df['close'].ewm(span=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_points = np.sign(momentum_reversal - momentum_reversal.shift())
    
    # Combine Signals Before Non-Linear Transformations
    combined_signal = adjusted_momentum + reversal_points
    smoothed_combined_signal = combined_signal.ewm(span=5).mean()
    
    # Integrate Non-Linear Transformation
    sqrt_transformed = np.sqrt(np.abs(smoothed_combined_signal))
    log_transformed = np.log1p(np.abs(smoothed_combined_signal))
    
    # Refine Final Alpha Factor
    interim_alpha_factor = sqrt_transformed + log_transformed
    final_alpha_factor = interim_alpha_factor.ewm(span=5).mean()
    
    return final_alpha_factor
```
