import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum'] = df['Close_Open_Diff'] * df['volume']
    df['EMA_10_VAM'] = df['Volume_Adjusted_Momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Abs_Daily_Return'] = df['Daily_Return'].abs()
    df['Market_Volatility'] = df['Daily_Return'].rolling(window=30).std()
    df['Modified_VAM'] = df['EMA_10_VAM'] - df['Market_Volatility']
    
    # Incorporate Trend Reversal Signal
    df['Short_Term_Momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_Term_Momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    df['Reversal_Signal'] = np.where(df['Momentum_Reversal'] > 0, 1, -1)
    
    # Enhance Reversal Signal with Adaptive Smoothing
    df['Smoothed_Reversal'] = df['Momentum_Reversal'].ewm(span=5, adjust=False).mean()
    df['Combined_Momentum'] = df['Modified_VAM'] + df['Smoothed_Reversal']
    
    # Integrate Non-Linear Transformation
    df['Sqrt_Transformed'] = np.sqrt(np.abs(df['Combined_Momentum']))
    df['Log_Transformed'] = np.log1p(np.abs(df['Combined_Momentum']))
    
    # Refine Final Alpha Factor
    df['Interim_Alpha_Factor'] = df['Sqrt_Transformed'] + df['Log_Transformed']
    df['Final_Alpha_Factor'] = df['Interim_Alpha_Factor'].ewm(span=5, adjust=False).mean()
    
    return df['Final_Alpha_Factor']
