```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = (df['close_open_diff'] + df['high_low_range']) * df['volume']
    
    # Adaptive Smoothing via 10-day EMA
    df['ema_10'] = df['volume_adjusted_momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    df['daily_return'] = df['close'].diff()
    df['abs_daily_return'] = df['daily_return'].abs()
    df['market_volatility'] = df['daily_return'].rolling(window=10).std()
    df['modified_momentum'] = df['ema_10'] / df['market_volatility']
    
    # Integrate Non-Linear Transformation
    df['sqrt_transformed_momentum'] = np.sqrt(df['modified_momentum'])
    df['log_transformed_momentum'] = np.log(df['modified_momentum'])
    
    # Incorporate Trend Reversal Signal
    df['short_term_momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_reversal'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    df['ema_5_reversal'] = df['momentum_reversal'].ewm(span=5, adjust=False).mean()
    df['ema_10_reversal'] = df['momentum_reversal'].ewm(span=10, adjust=False).mean()
    df['smoothed_reversal'] = (df['ema_5_reversal'] + df['ema_10_reversal']) / 2
    
    # Combine Smoothed Reversal Signal and Non-Linearly Transformed Momentum
    df['interim_alpha_factor'] = df['sqrt_transformed_momentum'] + df['log_transformed_momentum'] + df['smoothed_reversal']
    
    # Final Advanced Adaptive Smoothing
    df['final_ema_5'] = df['interim_alpha_factor'].ewm(span=5, adjust=False).mean()
    df['final_ema_10'] = df['interim_alpha_factor'].ewm(span=10, adjust=False).mean()
    df['final_alpha_factor'] = (df['final_ema_5'] + df['final_ema_10']) / 2
    
    return df['final_alpha_factor']
```
