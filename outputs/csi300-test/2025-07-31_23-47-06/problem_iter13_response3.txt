```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['Intraday_Price_Movement'] = (df['High_Low_Range'] + df['Close_Open_Diff']) / 2
    df['Volume_Adjusted_Momentum'] = df['volume'] * df['Intraday_Price_Movement']
    df['EMA_10_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Absolute_Daily_Return'] = df['Daily_Return'].abs()
    df['Market_Volatility'] = df['Daily_Return'].rolling(window=30).std()
    df['Modified_Volume_Adjusted_Momentum'] = df['EMA_10_Volume_Adjusted_Momentum'] / df['Market_Volatility']
    
    # Incorporate Trend Reversal Signal
    df['EMA_5_Close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20_Close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Reversal'] = df['EMA_5_Close'] - df['EMA_20_Close']
    df['Trend_Reversal_Signal'] = np.where(df['Momentum_Reversal'] > 0, 1, -1)
    
    # Integrate Non-Linear Transformation
    df['Sqrt_Modified_Volume_Adjusted_Momentum'] = np.sqrt(np.abs(df['Modified_Volume_Adjusted_Momentum']))
    df['Log_Modified_Volume_Adjusted_Momentum'] = np.log(1 + np.abs(df['Modified_Volume_Adjusted_Momentum']))
    
    # Enhance Reversal Signal with Adaptive Smoothing
    df['Smoothed_Reversal_Signal'] = df['Trend_Reversal_Signal'].ewm(span=5, adjust=False).mean()
    df['Interim_Alpha_Factor'] = (df['Sqrt_Modified_Volume_Adjusted_Momentum'] + 
                                  df['Log_Modified_Volume_Adjusted_Momentum'] + 
                                  df['Smoothed_Reversal_Signal'])
    
    # Final Adaptive Smoothing
    df['Final_Alpha_Factor'] = df['Interim_Alpha_Factor'].ewm(span=5, adjust=False).mean()
    
    return df['Final_Alpha_Factor']
```
