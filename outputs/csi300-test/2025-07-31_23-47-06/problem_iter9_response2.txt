```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']

    # Incorporate Volume Influence
    volume_adjusted_momentum = (high_low_range + close_open_diff) * df['volume']
    ema_volume_adjusted_momentum = volume_adjusted_momentum.ewm(span=10, min_periods=10).mean()

    # Adjust for Market Volatility
    daily_return = df['close'].diff()
    abs_return_diff = np.abs(daily_return)
    std_dev_daily_returns = daily_return.rolling(window=30).std()
    adjusted_momentum = ema_volume_adjusted_momentum / std_dev_daily_returns

    # Enhance Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, min_periods=5).mean()
    long_term_momentum = df['close'].ewm(span=20, min_periods=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum

    # Integrate Reversal Signal into Final Factor
    final_factor = momentum_reversal + adjusted_momentum
    final_factor_smoothed = final_factor.ewm(span=5, min_periods=5).mean()

    return final_factor_smoothed
```
