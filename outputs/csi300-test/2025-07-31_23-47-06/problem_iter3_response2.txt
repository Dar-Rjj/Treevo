```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    
    # Intraday Price Movement
    df['intraday_movement'] = (df['high_low_range'] + df['close_open_diff']) / 2
    
    # Volume Adjusted Momentum
    df['volume_adjusted_momentum'] = df['intraday_movement'] * df['volume']
    
    # Smoothing via Moving Average (10-day SMA)
    df['sma_10_volume_adjusted'] = df['volume_adjusted_momentum'].rolling(window=10).mean()
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Absolute Daily Return
    df['abs_daily_return'] = df['daily_return'].abs()
    
    # Calculate Average Market Volatility (30-day Standard Deviation of Daily Returns)
    df['market_volatility'] = df['daily_return'].rolling(window=30).std()
    
    # Modify Volume Adjusted Momentum with Market Volatility
    df['adjusted_momentum'] = df['sma_10_volume_adjusted'] - df['market_volatility']
    
    # Calculate Short-Term Momentum (5-day SMA of Close prices)
    df['short_term_momentum'] = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Momentum (20-day SMA of Close prices)
    df['long_term_momentum'] = df['close'].rolling(window=20).mean()
    
    # Compute Momentum Reversal
    df['momentum_reversal'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Identify Reversal Points
    df['reversal_signal'] = df['momentum_reversal'].apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Integrate Reversal Signal into Final Factor
    df['final_alpha_factor'] = df['adjusted_momentum'] + df['reversal_signal']
    
    return df['final_alpha_factor'].dropna()
```
