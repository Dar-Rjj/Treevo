```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Momentum
    high_to_low_momentum = (df['high'] - df['low']) / df['low']
    
    # Calculate Short-Term Trend
    short_term_trend_3 = high_to_low_momentum.rolling(window=3).mean()
    short_term_trend_10 = high_to_low_momentum.rolling(window=10).mean()
    short_term_trend = (short_term_trend_3 + short_term_trend_10) / 2
    
    # Calculate Long-Term Trend
    long_term_trend_20 = high_to_low_momentum.rolling(window=20).mean()
    long_term_trend_50 = high_to_low_momentum.rolling(window=50).mean()
    long_term_trend = (long_term_trend_20 + long_term_trend_50) / 2
    
    # Compute Trend Difference
    trend_difference = short_term_trend - long_term_trend
    
    # Detect Volume Spikes
    volume = df['volume']
    rolling_std_volume = volume.rolling(window=20).std()
    rolling_mean_volume = volume.rolling(window=20).mean()
    spike_threshold = 2 * rolling_std_volume
    volume_spike_indicator = (volume > (rolling_mean_volume + spike_threshold)).astype(int)
    
    # Weight by Volume and Adjust for Volume Spikes
    weighted_trend = trend_difference * volume * volume_spike_indicator
    
    return weighted_trend
```
