import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['volume_adjusted_momentum'] = (df['high_low_range'] + df['close_open_diff']) * df['volume']
    df['ema_10_vol_adj_mom'] = df['volume_adjusted_momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    df['daily_return'] = df['close'].pct_change()
    df['abs_daily_return'] = df['daily_return'].abs()
    df['dynamic_market_volatility'] = df['daily_return'].rolling(window=20).std()
    df['modified_vol_adj_mom'] = df['ema_10_vol_adj_mom'] / df['dynamic_market_volatility']
    
    # Incorporate Trend Reversal Signal
    df['short_term_momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_reversal'] = df['short_term_momentum'] - df['long_term_momentum']
    df['reversal_signal'] = np.where(df['momentum_reversal'] > 0, 1, -1)
    
    # Integrate Non-Linear Transformation
    df['sqrt_modified_vol_adj_mom'] = np.sqrt(df['modified_vol_adj_mom'])
    df['log_modified_vol_adj_mom'] = np.log(df['modified_vol_adj_mom'])
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    df['ema_5_reversal_signal'] = df['reversal_signal'].ewm(span=5, adjust=False).mean()
    df['ema_10_reversal_signal'] = df['reversal_signal'].ewm(span=10, adjust=False).mean()
    df['smoothed_reversal_signal'] = (df['ema_5_reversal_signal'] + df['ema_10_reversal_signal']) / 2
    
    # Combine Smoothed Reversal Signal with Non-Linearly Transformed Momentum
    df['interim_alpha_factor'] = df['smoothed_reversal_signal'] + (df['sqrt_modified_vol_adj_mom'] + df['log_modified_vol_adj_mom']) / 2
    
    # Final Advanced Adaptive Smoothing
    df['ema_5_interim_alpha'] = df['interim_alpha_factor'].ewm(span=5, adjust=False).mean()
    df['ema_10_interim_alpha'] = df['interim_alpha_factor'].ewm(span=10, adjust=False).mean()
    df['final_alpha_factor'] = (df['ema_5_interim_alpha'] + df['ema_10_interim_alpha']) / 2
    
    return df['final_alpha_factor']
