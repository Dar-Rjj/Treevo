```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    df['intraday_price_movement'] = df['close_open_diff'] + 0.5 * df['high_low_range']  # Using a simple heuristic for intraday price movement

    # Incorporate Volume Influence
    df['volume_adjusted_momentum'] = df['intraday_price_movement'] * df['volume']
    df['vol_adj_mom_10day_ema'] = df['volume_adjusted_momentum'].ewm(span=10, adjust=False).mean()

    # Adjust for Market Volatility
    df['daily_return'] = df['close'].diff()
    df['std_dev_daily_returns'] = df['daily_return'].rolling(window=30).std()
    df['adjusted_momentum'] = df['vol_adj_mom_10day_ema'] - df['std_dev_daily_returns']

    # Enhance Trend Reversal Signal
    df['short_term_momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_reversal'] = df['short_term_momentum'] - df['long_term_momentum']

    # Integrate Reversal Signal into Final Factor
    df['final_factor'] = df['momentum_reversal'] + df['adjusted_momentum']

    # Refine Final Alpha Factor
    df['final_alpha_factor'] = df['final_factor'].ewm(span=5, adjust=False).mean()

    return df['final_alpha_factor']
```
