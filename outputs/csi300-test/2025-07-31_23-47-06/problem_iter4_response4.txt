```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    intraday_movement = (high_low_range + close_open_diff) / 2  # Average of the two movements
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = intraday_movement * df['volume']
    smoothed_momentum = volume_adjusted_momentum.rolling(window=10).mean()
    
    # Adjust for Market Volatility
    daily_return = df['close'].pct_change()
    absolute_daily_return = daily_return.abs()
    avg_market_volatility = absolute_daily_return.rolling(window=30).std()
    adjusted_momentum = smoothed_momentum - avg_market_volatility
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'] - df['close'].rolling(window=5).mean()
    long_term_momentum = df['close'] - df['close'].rolling(window=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_points = (momentum_reversal > 0).astype(int) - (momentum_reversal < 0).astype(int)
    
    # Integrate Reversal Signal into Final Factor
    final_factor = adjusted_momentum + reversal_points
    
    return final_factor
```
