```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Volume-Adjusted Breakout Intensity
    volume_adjusted_intensity = (df['volume'] / intraday_range) * intraday_range
    
    # Calculate Trend Direction
    trend_direction = (df['close'] > df['open']).astype(int) - (df['close'] < df['open']).astype(int)
    
    # Combine the components to form the final factor
    factor = volume_adjusted_intensity * trend_direction
    
    return factor
```
