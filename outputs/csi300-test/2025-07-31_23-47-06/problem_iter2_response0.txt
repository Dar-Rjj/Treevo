import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    
    # Calculate Intraday Price Movement
    df['intraday_movement'] = (df['high_low_range'] + df['close_open_diff']) / 2
    
    # Volume Adjusted Momentum
    df['volume_adjusted_momentum'] = df['intraday_movement'] * df['volume']
    
    # Smoothing via 10-day SMA
    df['sma_10_volume_adjusted_momentum'] = df['volume_adjusted_momentum'].rolling(window=10).mean()
    
    # Calculate Absolute Return Difference for Market Volatility
    df['return_diff'] = df['close'].pct_change()
    df['market_volatility'] = df['return_diff'].rolling(window=30).std()
    
    # Modify Intraday Momentum with Market Volatility
    df['adjusted_momentum'] = df['sma_10_volume_adjusted_momentum'] - df['market_volatility']
    
    # Calculate Short-Term and Long-Term Momentum
    df['short_term_momentum'] = df['close'] - df['close'].rolling(window=5).mean()
    df['long_term_momentum'] = df['close'] - df['close'].rolling(window=20).mean()
    
    # Compute Momentum Reversal
    df['momentum_reversal'] = df['short_term_momentum'] - df['long_term_momentum']
    df['reversal_signal'] = df['momentum_reversal'].apply(lambda x: 1 if x > 0 else -1)
    
    # Integrate Reversal Signal into Final Factor
    df['final_alpha_factor'] = df['adjusted_momentum'] + df['reversal_signal']
    
    return df['final_alpha_factor'].dropna()
