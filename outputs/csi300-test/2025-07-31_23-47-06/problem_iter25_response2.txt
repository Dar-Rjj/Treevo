import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']

    # Incorporate Volume Influence
    volume_adjusted_momentum = close_open_diff * df['volume']
    ema_volume_momentum = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()

    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].diff()
    abs_daily_return = daily_return.abs()
    dynamic_volatility = daily_return.rolling(window=10).std()
    
    modified_volume_momentum = ema_volume_momentum / dynamic_volatility

    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(modified_volume_momentum)
    log_transformed_momentum = np.log(modified_volume_momentum + 1e-6)  # Add small constant to avoid log(0)

    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_signal = momentum_reversal.apply(lambda x: 1 if x > 0 else -1)

    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    ema_5_reversal = momentum_reversal.ewm(span=5, adjust=False).mean()
    ema_10_reversal = momentum_reversal.ewm(span=10, adjust=False).mean()
    smoothed_reversal = (ema_5_reversal + ema_10_reversal) / 2

    # Combine Smoothed Reversal Signal with Non-Linearly Transformed Momentum
    interim_alpha_factor = smoothed_reversal + log_transformed_momentum

    # Final Advanced Adaptive Smoothing
    final_ema_5 = interim_alpha_factor.ewm(span=5, adjust=False).mean()
    final_ema_10 = interim_alpha_factor.ewm(span=10, adjust=False).mean()
    final_alpha_factor = (final_ema_5 + final_ema_10) / 2

    return final_alpha_factor
