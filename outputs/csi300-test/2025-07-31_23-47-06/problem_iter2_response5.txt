import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    intraday_movement = high_low_range + close_open_diff
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = intraday_movement * df['volume']
    volume_adjusted_momentum_smoothed = volume_adjusted_momentum.rolling(window=10).mean()
    
    # Adjust for Dynamic Market Volatility
    daily_returns = df['close'].pct_change()
    dynamic_volatility = daily_returns.ewm(span=30).std()
    adjusted_momentum = volume_adjusted_momentum_smoothed - dynamic_volatility
    
    # Incorporate Liquidity Factor
    dollar_amount = df['volume'] * df['close']
    liquidity_ratio = df['volume'] / dollar_amount
    final_alpha_factor = adjusted_momentum * liquidity_ratio
    
    return final_alpha_factor
