import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (intraday_high_low_range + close_open_diff) * df['volume']
    volume_adjusted_momentum_ema_10 = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].pct_change()
    abs_daily_return = np.abs(daily_return)
    dynamic_market_volatility = abs_daily_return.rolling(window=10).std()
    modified_volume_adjusted_momentum = volume_adjusted_momentum_ema_10 / dynamic_market_volatility
    
    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(modified_volume_adjusted_momentum)
    log_transformed_momentum = np.log(modified_volume_adjusted_momentum)
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_points = np.sign(momentum_reversal.diff())
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    smoothed_reversal_signal = 0.5 * (momentum_reversal.ewm(span=5, adjust=False).mean() + 
                                       momentum_reversal.ewm(span=10, adjust=False).mean())
    combined_signal = sqrt_transformed_momentum + log_transformed_momentum + smoothed_reversal_signal
    
    # Refine Final Alpha Factor
    final_alpha_factor = 0.5 * (combined_signal.ewm(span=5, adjust=False).mean() + 
                                 combined_signal.ewm(span=10, adjust=False).mean())
    
    return final_alpha_factor
