```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Momentum
    high_to_low_momentum = df['high'] / df['low']
    
    # Calculate Short-Term Trend (5 and 10 days moving averages)
    short_term_trend_5 = high_to_low_momentum.rolling(window=5).mean()
    short_term_trend_10 = high_to_low_momentum.rolling(window=10).mean()
    short_term_trend = (short_term_trend_5 + short_term_trend_10) / 2
    
    # Calculate Long-Term Trend (20 and 50 days moving averages)
    long_term_trend_20 = high_to_low_momentum.rolling(window=20).mean()
    long_term_trend_50 = high_to_low_momentum.rolling(window=50).mean()
    long_term_trend = (long_term_trend_20 + long_term_trend_50) / 2
    
    # Compute Trend Difference
    trend_difference = short_term_trend - long_term_trend
    
    # Weight by Volume
    volume_weighted_trend_difference = trend_difference * df['volume']
    
    # Adjust for Market Sentiment
    daily_return = df['close'] / df['open'] - 1
    ten_day_moving_average_of_daily_returns = daily_return.rolling(window=10).mean()
    
    # Final Alpha Factor
    alpha_factor = volume_weighted_trend_difference * ten_day_moving_average_of_daily_returns
    
    return alpha_factor
```
