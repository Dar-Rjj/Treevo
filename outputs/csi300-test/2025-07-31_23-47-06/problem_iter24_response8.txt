```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']

    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum'] = (df['Close_Open_Diff'] + df['High_Low_Range']) * df['volume']
    df['Volume_Adjusted_Momentum_10d_EMA'] = df['Volume_Adjusted_Momentum'].ewm(span=10, adjust=False).mean()

    # Adjust for Dynamic Market Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Absolute_Daily_Return'] = df['Daily_Return'].abs()
    df['Dynamic_Market_Volatility'] = df['Daily_Return'].rolling(window=20).std()

    df['Modified_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum_10d_EMA'] / df['Dynamic_Market_Volatility']

    # Integrate Non-Linear Transformation
    df['Sqrt_Modified_Volume_Adjusted_Momentum'] = np.sqrt(np.abs(df['Modified_Volume_Adjusted_Momentum']))
    df['Log_Modified_Volume_Adjusted_Momentum'] = np.log1p(np.abs(df['Modified_Volume_Adjusted_Momentum']))

    # Incorporate Trend Reversal Signal
    df['Short_Term_Momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_Term_Momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    df['Reversal_Signal_5d_EMA'] = df['Momentum_Reversal'].ewm(span=5, adjust=False).mean()
    df['Reversal_Signal_10d_EMA'] = df['Momentum_Reversal'].ewm(span=10, adjust=False).mean()
    df['Smoothed_Reversal_Signal'] = (df['Reversal_Signal_5d_EMA'] + df['Reversal_Signal_10d_EMA']) / 2

    # Combine Smoothed Reversal Signal and Non-Linearly Transformed Momentum
    df['Interim_Alpha_Factor'] = df['Sqrt_Modified_Volume_Adjusted_Momentum'] + df['Log_Modified_Volume_Adjusted_Momentum'] + df['Smoothed_Reversal_Signal']

    # Final Advanced Adaptive Smoothing
    df['Final_Alpha_Factor_5d_EMA'] = df['Interim_Alpha_Factor'].ewm(span=5, adjust=False).mean()
    df['Final_Alpha_Factor_10d_EMA'] = df['Interim_Alpha_Factor'].ewm(span=10, adjust=False).mean()
    df['Final_Alpha_Factor'] = (df['Final_Alpha_Factor_5d_EMA'] + df['Final_Alpha_Factor_10d_EMA']) / 2

    return df['Final_Alpha_Factor']
```
