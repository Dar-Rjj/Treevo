```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']
    df['Intraday_Price_Movement'] = (df['High_Low_Range'] + df['Close_Open_Diff']) / 2
    
    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum'] = df['Intraday_Price_Movement'] * df['volume']
    df['SMA_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].rolling(window=10).mean()
    
    # Adjust for Dynamic Market Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Abs_Return_Diff'] = df['Daily_Return'].abs()
    df['Std_Daily_Return'] = df['Daily_Return'].rolling(window=20).std()
    df['EMA_Daily_Return'] = df['Daily_Return'].ewm(span=20, adjust=False).mean()
    df['Dynamic_Volatility_Adjustment'] = df['EMA_Daily_Return'] / df['Std_Daily_Return']
    df['Adjusted_Intraday_Momentum'] = df['SMA_Volume_Adjusted_Momentum'] * df['Dynamic_Volatility_Adjustment']
    
    # Incorporate Enhanced Trend Reversal Signal
    df['Short_Term_Momentum'] = df['close'].rolling(window=5).mean()
    df['Long_Term_Momentum'] = df['close'].rolling(window=20).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    df['Reversal_Signal'] = np.sign(df['Momentum_Reversal'])
    df['Interim_Alpha_Factor'] = df['Adjusted_Intraday_Momentum'] + df['Reversal_Signal']
    
    # Enhance Final Alpha Factor
    df['Smoothed_Final_Alpha_Factor'] = df['Interim_Alpha_Factor'].rolling(window=5).mean()
    
    # Apply Advanced Trend Confirmation
    df['60_day_SMA'] = df['close'].rolling(window=60).mean()
    df['Trend_Strength'] = df['60_day_SMA'].pct_change().rolling(window=60).mean()
    df['Final_Alpha_Factor'] = df['Smoothed_Final_Alpha_Factor'] * df['Trend_Strength']
    
    return df['Final_Alpha_Factor']
```
