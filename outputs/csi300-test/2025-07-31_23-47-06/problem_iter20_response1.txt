```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    df['intraday_movement'] = df[['high_low_range', 'close_open_diff']].max(axis=1)
    
    # Incorporate Volume Influence
    df['volume_adjusted_momentum'] = df['intraday_movement'] * df['volume']
    df['ema_10_vol_adj_mom'] = df['volume_adjusted_momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    df['daily_return'] = df['close'].pct_change()
    df['abs_daily_return'] = df['daily_return'].abs()
    df['dynamic_volatility'] = df['daily_return'].rolling(window=15).std()
    df['vol_adj_mom_normalized'] = df['ema_10_vol_adj_mom'] / df['dynamic_volatility']
    
    # Integrate Non-Linear Transformation
    df['sqrt_transformed'] = np.sqrt(df['vol_adj_mom_normalized'])
    df['log_transformed'] = np.log(df['vol_adj_mom_normalized'])
    
    # Incorporate Trend Reversal Signal
    df['short_term_momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_reversal'] = df['short_term_momentum'] - df['long_term_momentum']
    df['reversal_signal'] = np.where(df['momentum_reversal'] > 0, 1, -1)
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    df['ema_5_reversal'] = df['reversal_signal'].ewm(span=5, adjust=False).mean()
    df['ema_10_reversal'] = df['reversal_signal'].ewm(span=10, adjust=False).mean()
    df['smoothed_reversal'] = (df['ema_5_reversal'] + df['ema_10_reversal']) / 2
    
    # Refine Final Alpha Factor
    df['interim_alpha'] = (df['sqrt_transformed'] + df['log_transformed']) * df['smoothed_reversal']
    df['final_alpha_ema_5'] = df['interim_alpha'].ewm(span=5, adjust=False).mean()
    df['final_alpha_ema_10'] = df['interim_alpha'].ewm(span=10, adjust=False).mean()
    df['final_alpha'] = (df['final_alpha_ema_5'] + df['final_alpha_ema_10']) / 2
    
    return df['final_alpha']
```
