```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_high_low = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (intraday_high_low + close_open_diff) * df['volume']
    volume_adjusted_ema = volume_adjusted_momentum.ewm(span=10).mean()
    
    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].diff()
    absolute_daily_return = daily_return.abs()
    dynamic_market_volatility = absolute_daily_return.rolling(window=15).std()
    modified_volume_adj_momentum = volume_adjusted_ema / dynamic_market_volatility
    
    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(modified_volume_adj_momentum)
    log_transformed_momentum = np.log(modified_volume_adj_momentum)
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5).mean()
    long_term_momentum = df['close'].ewm(span=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_signal = np.sign(momentum_reversal.diff())
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    reversal_signal_ema_5 = reversal_signal.ewm(span=5).mean()
    reversal_signal_ema_10 = reversal_signal.ewm(span=10).mean()
    smoothed_reversal_signal = (reversal_signal_ema_5 + reversal_signal_ema_10) / 2
    combined_signal = sqrt_transformed_momentum + log_transformed_momentum + smoothed_reversal_signal
    
    # Refine Final Alpha Factor
    interim_alpha_factor = combined_signal
    final_alpha_ema_5 = interim_alpha_factor.ewm(span=5).mean()
    final_alpha_ema_10 = interim_alpha_factor.ewm(span=10).mean()
    final_alpha_factor = (final_alpha_ema_5 + final_alpha_ema_10) / 2
    
    return final_alpha_factor
```
