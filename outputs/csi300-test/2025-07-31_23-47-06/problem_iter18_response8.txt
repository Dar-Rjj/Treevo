```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['volume_adjusted_momentum'] = (df['high_low_range'] + df['close_open_diff']) * df['volume']
    df['smoothed_vol_adj_mom'] = df['volume_adjusted_momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    df['daily_return'] = df['close'].pct_change()
    df['absolute_daily_return'] = df['daily_return'].abs()
    df['market_volatility'] = df['daily_return'].rolling(window=30).std()
    df['vol_adj_mom_normalized'] = df['smoothed_vol_adj_mom'] / df['market_volatility']
    
    # Incorporate Trend Reversal Signal
    df['short_term_mom'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_mom'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_reversal'] = df['short_term_mom'] - df['long_term_mom']
    df['reversal_signal'] = np.sign(df['momentum_reversal'])
    
    # Combine Signals Before Non-Linear Transformations
    df['combined_signal'] = df['vol_adj_mom_normalized'] + df['reversal_signal']
    df['smoothed_combined_signal'] = df['combined_signal'].ewm(span=5, adjust=False).mean()
    
    # Integrate Non-Linear Transformation
    df['sqrt_transformed_signal'] = np.sqrt(np.abs(df['smoothed_combined_signal']))
    df['log_transformed_signal'] = np.log(np.abs(df['smoothed_combined_signal']) + 1)
    
    # Refine Final Alpha Factor
    df['interim_alpha_factor'] = df['sqrt_transformed_signal'] + df['log_transformed_signal']
    final_alpha_factor = df['interim_alpha_factor'].ewm(span=5, adjust=False).mean()
    
    return final_alpha_factor
```
