```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (high_low_range + close_open_diff) * df['volume']
    
    # Smoothing via Moving Average
    volume_adjusted_momentum_sma_10 = volume_adjusted_momentum.rolling(window=10).mean()
    
    # Adjust for Market Volatility
    daily_return = df['close'].pct_change()
    absolute_return_diff = daily_return.abs()
    market_volatility = absolute_return_diff.rolling(window=30).std()
    
    adjusted_momentum = volume_adjusted_momentum_sma_10 - market_volatility
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'] - df['close'].rolling(window=5).mean()
    long_term_momentum = df['close'] - df['close'].rolling(window=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    
    # Label Positive and Negative Reversals
    reversal_signal = momentum_reversal.apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Integrate Reversal Signal into Final Factor
    interim_alpha_factor = adjusted_momentum + reversal_signal
    
    # Refine Final Alpha Factor
    final_alpha_factor = interim_alpha_factor.rolling(window=5).mean()
    
    return final_alpha_factor
```
