```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday Price Indicators
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']

    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum'] = (df['high'] - df['low']) * df['volume']
    
    # Apply Adaptive Smoothing
    df['EMA_10_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].ewm(span=10, adjust=False).mean()

    # Adjust for Dynamic Market Volatility
    df['Daily_Returns'] = df['close'].pct_change()
    df['Absolute_Daily_Returns'] = df['Daily_Returns'].abs()
    df['Dynamic_Market_Volatility'] = df['Daily_Returns'].rolling(window=10).std()
    
    df['Modified_Volume_Adjusted_Momentum'] = df['EMA_10_Volume_Adjusted_Momentum'] / df['Dynamic_Market_Volatility']
    
    # Integrate Non-Linear Transformations
    df['Sqrt_Modified_VAM'] = np.sqrt(df['Modified_Volume_Adjusted_Momentum'])
    df['Log_Modified_VAM'] = np.log(df['Modified_Volume_Adjusted_Momentum'])
    
    # Incorporate Early Reversal Signals
    df['Short_Term_Momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_Term_Momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    df['Early_Signal_Indicator'] = df['Short_Term_Momentum'].ewm(span=3, adjust=False).mean() - df['Long_Term_Momentum'].ewm(span=3, adjust=False).mean()
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    df['EMA_5_Reversal_Signal'] = df['Momentum_Reversal'].ewm(span=5, adjust=False).mean()
    df['EMA_10_Reversal_Signal'] = df['Momentum_Reversal'].ewm(span=10, adjust=False).mean()
    df['Smoothed_Reversal_Signal'] = (df['EMA_5_Reversal_Signal'] + df['EMA_10_Reversal_Signal']) / 2
    
    df['Interim_Alpha_Factor'] = df['Smoothed_Reversal_Signal'] + df['Sqrt_Modified_VAM'] + df['Log_Modified_VAM']
    
    # Refine Final Alpha Factor
    df['Final_EMA_5'] = df['Interim_Alpha_Factor'].ewm(span=5, adjust=False).mean()
    df['Final_EMA_10'] = df['Interim_Alpha_Factor'].ewm(span=10, adjust=False).mean()
    df['Final_Alpha_Factor'] = (df['Final_EMA_5'] + df['Final_EMA_10']) / 2
    
    return df['Final_Alpha_Factor']
```
