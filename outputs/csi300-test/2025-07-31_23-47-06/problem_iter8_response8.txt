```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']

    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum'] = (df['High_Low_Range'] + df['Close_Open_Diff']) * df['volume']
    df['EMA_10_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].ewm(span=10).mean()

    # Adjust for Market Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Abs_Daily_Return'] = df['Daily_Return'].abs()
    df['Std_Daily_Return'] = df['Daily_Return'].rolling(window=30).std()
    df['Adjusted_Intraday_Momentum'] = df['EMA_10_Volume_Adjusted_Momentum'] - df['Std_Daily_Return']

    # Enhance Trend Reversal Signal
    df['Short_Term_Momentum'] = df['close'].ewm(span=5).mean()
    df['Long_Term_Momentum'] = df['close'].ewm(span=20).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    df['Reversal_Signal'] = np.where(df['Momentum_Reversal'] > 0, 1, -1)
    
    # Integrate Reversal Signal into Final Factor
    df['Interim_Alpha_Factor'] = df['Adjusted_Intraday_Momentum'] + df['Reversal_Signal']

    # Refine Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Interim_Alpha_Factor'].ewm(span=5).mean()

    # Introduce Dynamic Volatility Adjustment
    if 'market_index_close' in df.columns:
        df['Rolling_Beta'] = df.rolling(window=30).cov()['close'].div(df['market_index_close'].rolling(window=30).var())
        df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * df['Rolling_Beta']

    return df['Final_Alpha_Factor'].dropna()
```
