```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum'] = (df['Close_Open_Diff'] + df['High_Low_Range']) * df['volume']
    
    # Adaptive Smoothing via Moving Average
    df['Vol_Adj_Mom_EMA10'] = df['Volume_Adjusted_Momentum'].ewm(span=10, adjust=False).mean()
    
    # Integrate Trend Reversal Signal Early
    df['Short_Term_Mom'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_Term_Mom'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Reversal'] = df['Short_Term_Mom'] - df['Long_Term_Mom']
    
    # Adjust for Market Volatility
    df['Daily_Return'] = df['close'].diff()
    df['Abs_Daily_Return'] = df['Daily_Return'].abs()
    df['Market_Volatility'] = df['Daily_Return'].rolling(window=30).std()
    df['Modified_Vol_Adj_Mom'] = df['Vol_Adj_Mom_EMA10'] - df['Market_Volatility']
    
    # Combine Modified Momentum with Reversal Signal
    df['Combined_Alpha_Factor'] = df['Modified_Vol_Adj_Mom'] + df['Momentum_Reversal']
    
    # Integrate Non-Linear Transformation
    df['Sqrt_Combined_Factor'] = np.sqrt(np.abs(df['Combined_Alpha_Factor']))
    df['Log_Combined_Factor'] = np.log1p(np.abs(df['Combined_Alpha_Factor']))
    
    # Refine Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Sqrt_Combined_Factor'] + df['Log_Combined_Factor']) * np.sign(df['Combined_Alpha_Factor'])
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(span=5, adjust=False).mean()
    
    return df['Final_Alpha_Factor']
```
