```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['volume_adjusted_momentum'] = df['volume'] * (df['high_low_range'] + df['close_open_diff'])
    df['volume_adjusted_momentum_ema10'] = df['volume_adjusted_momentum'].ewm(span=10).mean()
    
    # Adjust for Market Volatility
    df['daily_return'] = df['close'].pct_change()
    df['abs_daily_return'] = df['daily_return'].abs()
    df['market_volatility'] = df['daily_return'].rolling(window=30).std()
    df['volatility_adjusted_mom'] = df['volume_adjusted_momentum_ema10'] / df['market_volatility']
    
    # Incorporate Trend Reversal Signal
    df['short_term_mom'] = df['close'].ewm(span=5).mean()
    df['long_term_mom'] = df['close'].ewm(span=20).mean()
    df['momentum_reversal'] = df['short_term_mom'] - df['long_term_mom']
    df['reversal_points'] = np.sign(df['momentum_reversal'])
    
    # Integrate Non-Linear Transformation
    df['sqrt_transformed_mom'] = np.sqrt(np.abs(df['volatility_adjusted_mom']))
    df['log_transformed_mom'] = np.log(np.abs(df['volatility_adjusted_mom']) + 1)
    
    # Enhance Reversal Signal with Adaptive Smoothing
    df['smoothed_reversal_signal'] = df['reversal_points'].ewm(span=5).mean()
    df['interim_alpha_factor'] = df['sqrt_transformed_mom'] + df['log_transformed_mom'] + df['smoothed_reversal_signal']
    
    # Final Adaptive Smoothing
    df['final_alpha_factor'] = df['interim_alpha_factor'].ewm(span=5).mean()
    
    return df['final_alpha_factor']
```
