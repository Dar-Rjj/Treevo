import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (high_low_range + close_open_diff) * df['volume']
    volume_adjusted_momentum_ema = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].diff()
    abs_daily_return = daily_return.abs()
    dynamic_volatility = daily_return.rolling(window=20).std()
    modified_volume_adjusted_momentum = volume_adjusted_momentum_ema / dynamic_volatility
    
    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(modified_volume_adjusted_momentum)
    log_transformed_momentum = np.log(modified_volume_adjusted_momentum)
    
    # Incorporate Early Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=30, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_points = np.sign(momentum_reversal).diff().fillna(0)
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    smoothed_reversal_signal = (reversal_points.ewm(span=5, adjust=False).mean() + 
                                reversal_points.ewm(span=10, adjust=False).mean()) / 2
    combined_signal = log_transformed_momentum + smoothed_reversal_signal
    
    # Refine Final Alpha Factor
    interim_alpha_factor = combined_signal
    final_alpha_factor = (interim_alpha_factor.ewm(span=5, adjust=False).mean() + 
                          interim_alpha_factor.ewm(span=10, adjust=False).mean()) / 2

    return final_alpha_factor
