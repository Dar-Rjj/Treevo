```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_price_movement = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = intraday_price_movement * df['volume']
    volume_adjusted_ema = volume_adjusted_momentum.ewm(span=10).mean()
    
    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].diff()
    abs_daily_return = daily_return.abs()
    dynamic_market_volatility = abs_daily_return.rolling(window=5).std()
    modified_volume_adjusted_momentum = volume_adjusted_ema / dynamic_market_volatility
    
    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(modified_volume_adjusted_momentum)
    log_transformed_momentum = np.log(modified_volume_adjusted_momentum + 1)  # Adding 1 to avoid log(0)
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5).mean()
    long_term_momentum = df['close'].ewm(span=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    advanced_adaptive_smoothing = (momentum_reversal.ewm(span=5).mean() + momentum_reversal.ewm(span=10).mean()) / 2
    combined_signal = advanced_adaptive_smoothing + (sqrt_transformed_momentum + log_transformed_momentum) / 2
    
    # Refine Final Alpha Factor
    final_alpha_factor = (combined_signal.ewm(span=5).mean() + combined_signal.ewm(span=10).mean()) / 2

    return final_alpha_factor
```
