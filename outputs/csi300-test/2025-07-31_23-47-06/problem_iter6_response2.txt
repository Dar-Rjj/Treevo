```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Momentum
    df['High_to_Low_Momentum'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Short-Term Trend
    short_term_3 = df['High_to_Low_Momentum'].rolling(window=3).mean()
    short_term_10 = df['High_to_Low_Momentum'].rolling(window=10).mean()
    df['Short_Term_Trend'] = (short_term_3 + short_term_10) / 2
    
    # Calculate Long-Term Trend
    long_term_20 = df['High_to_Low_Momentum'].rolling(window=20).mean()
    long_term_50 = df['High_to_Low_Momentum'].rolling(window=50).mean()
    df['Long_Term_Trend'] = (long_term_20 + long_term_50) / 2
    
    # Compute Trend Difference
    df['Trend_Difference'] = df['Short_Term_Trend'] - df['Long_Term_Trend']
    
    # Detect Volume Spikes
    volume_std = df['volume'].rolling(window=30).std()
    volume_mean = df['volume'].rolling(window=30).mean()
    spike_threshold = volume_mean + 2 * volume_std
    df['Volume_Spike_Indicator'] = (df['volume'] > spike_threshold).astype(int)
    
    # Calculate Adaptive Volatility
    df['Absolute_Returns'] = df['close'].diff().abs()
    adaptive_volatility = df['Absolute_Returns'].rolling(window=30).std()
    
    # Weight by Volume and Adjust for Volume Spikes
    df['Weighted_Adjusted_Trend'] = (df['volume'] * df['Trend_Difference']) * df['Volume_Spike_Indicator']
    
    # Integrate Adaptive Volatility
    df['Final_Alpha_Factor'] = df['Weighted_Adjusted_Trend'] / adaptive_volatility
    
    return df['Final_Alpha_Factor']
```
