import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['volume_adjusted_momentum'] = df['volume'] * df['close_open_diff']
    df['ema_10_volume_adjusted_momentum'] = df['volume_adjusted_momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    df['daily_return'] = df['close'].pct_change()
    df['abs_daily_return'] = df['daily_return'].abs()
    df['volatility_30'] = df['daily_return'].rolling(window=30).std()
    df['ema_5_abs_daily_return'] = df['abs_daily_return'].ewm(span=5, adjust=False).mean()
    df['ema_30_abs_daily_return'] = df['abs_daily_return'].ewm(span=30, adjust=False).mean()
    df['dynamic_volatility'] = (df['ema_5_abs_daily_return'] / df['ema_30_abs_daily_return']) * df['volatility_30']
    df['vol_adj_momentum'] = df['ema_10_volume_adjusted_momentum'] / df['dynamic_volatility']
    
    # Incorporate Trend Reversal Signal
    df['ema_5_close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20_close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_reversal'] = df['ema_5_close'] - df['ema_20_close']
    df['threshold'] = df['momentum_reversal'].ewm(span=5, adjust=False).mean()
    df['reversal_signal'] = df['momentum_reversal'] - df['threshold']
    
    # Integrate Non-Linear Transformation
    df['sqrt_vol_adj_momentum'] = np.sqrt(np.abs(df['vol_adj_momentum']))
    df['log_vol_adj_momentum'] = np.log(np.abs(df['vol_adj_momentum']) + 1)
    
    # Enhance Reversal Signal with Adaptive Smoothing
    df['smoothed_reversal_signal'] = df['reversal_signal'].ewm(span=5, adjust=False).mean()
    df['interim_alpha_factor'] = df['sqrt_vol_adj_momentum'] + df['log_vol_adj_momentum'] + df['smoothed_reversal_signal']
    
    # Final Adaptive Smoothing
    df['final_alpha_factor'] = df['interim_alpha_factor'].ewm(span=5, adjust=False).mean()
    
    return df['final_alpha_factor']
