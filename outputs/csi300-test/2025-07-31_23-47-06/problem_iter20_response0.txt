import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum'] = (df['Close_Open_Diff'] + df['High_Low_Range']) * df['volume']
    
    # Adaptive Smoothing via Moving Average
    df['Volume_Adjusted_Momentum_EMA10'] = df['Volume_Adjusted_Momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Absolute_Daily_Return'] = df['Daily_Return'].abs()
    df['Market_Volatility'] = df['Daily_Return'].rolling(window=30).std()
    df['Modified_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum_EMA10'] - df['Market_Volatility']
    
    # Integrate Trend Reversal Signal
    df['Short_Term_Momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_Term_Momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    df['Reversal_Signal'] = np.where(df['Momentum_Reversal'] > 0, 1, -1)
    
    # Combine Modified Momentum with Reversal Signal
    df['Combined_Factor'] = df['Modified_Volume_Adjusted_Momentum'] + df['Reversal_Signal']
    
    # Apply Non-Linear Transformation
    df['Square_Root_Transformed'] = np.sqrt(np.abs(df['Combined_Factor']))
    df['Log_Transformed'] = np.log(np.abs(df['Combined_Factor']) + 1)  # Adding 1 to handle log(0)
    
    # Refine Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Square_Root_Transformed'] + df['Log_Transformed']) * df['Reversal_Signal']
    df['Final_Alpha_Factor_EMA5'] = df['Final_Alpha_Factor'].ewm(span=5, adjust=False).mean()
    
    return df['Final_Alpha_Factor_EMA5']
