```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday Price Indicators
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']

    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum'] = (df['High_Low_Range'] + df['Close_Open_Diff']) * df['volume']
    df['Volume_Adjusted_Momentum_EMA10'] = df['Volume_Adjusted_Momentum'].ewm(span=10).mean()

    # Adjust for Dynamic Market Volatility
    df['Daily_Returns'] = df['close'].pct_change()
    df['Absolute_Daily_Returns'] = df['Daily_Returns'].abs()
    df['Dynamic_Volatility'] = df['Daily_Returns'].rolling(window=10).std()
    df['Adjusted_Momentum'] = df['Volume_Adjusted_Momentum_EMA10'] / df['Dynamic_Volatility']

    # Integrate Non-Linear Transformations
    df['Sqrt_Adjusted_Momentum'] = np.sqrt(df['Adjusted_Momentum'])
    df['Log_Adjusted_Momentum'] = np.log(df['Adjusted_Momentum'])

    # Incorporate Early Reversal Signals
    df['Short_Term_Momentum'] = df['close'].ewm(span=5).mean()
    df['Long_Term_Momentum'] = df['close'].ewm(span=20).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    df['Early_Signal_Indicator'] = df['Short_Term_Momentum'].ewm(span=3).mean() - df['Long_Term_Momentum'].ewm(span=3).mean()

    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    df['Smoothed_Reversal_Signal'] = 0.5 * (df['Momentum_Reversal'].ewm(span=5).mean() + df['Momentum_Reversal'].ewm(span=10).mean())
    df['Combined_Factor'] = df['Sqrt_Adjusted_Momentum'] + df['Log_Adjusted_Momentum'] + df['Smoothed_Reversal_Signal']

    # Refine Final Alpha Factor
    df['Interim_Alpha_Factor'] = df['Combined_Factor']
    df['Final_Alpha_Factor'] = 0.5 * (df['Interim_Alpha_Factor'].ewm(span=5).mean() + df['Interim_Alpha_Factor'].ewm(span=10).mean())

    return df['Final_Alpha_Factor']
```
