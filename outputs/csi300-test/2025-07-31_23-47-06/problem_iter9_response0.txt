```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = close_open_diff * df['volume']
    ema_volume_adjusted_momentum = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    daily_return = df['close'].diff()
    abs_daily_return = daily_return.abs()
    std_dev_daily_return = daily_return.rolling(window=30).std()
    adjusted_momentum = ema_volume_adjusted_momentum / std_dev_daily_return
    
    # Enhance Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    
    # Integrate Reversal Signal into Final Factor
    final_factor = momentum_reversal + adjusted_momentum
    
    # Refine Final Alpha Factor
    final_alpha_factor = final_factor.ewm(span=5, adjust=False).mean()
    
    return final_alpha_factor
```
