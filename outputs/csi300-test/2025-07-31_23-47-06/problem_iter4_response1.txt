import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Momentum
    df['high_to_low_momentum'] = df['High'] - df['Low']
    
    # Calculate Short-Term Trend
    df['short_term_trend_3d'] = df['high_to_low_momentum'].rolling(window=3).mean()
    df['short_term_trend_10d'] = df['high_to_low_momentum'].rolling(window=10).mean()
    df['short_term_trend'] = (df['short_term_trend_3d'] + df['short_term_trend_10d']) / 2
    
    # Calculate Long-Term Trend
    df['long_term_trend_20d'] = df['high_to_low_momentum'].rolling(window=20).mean()
    df['long_term_trend_50d'] = df['high_to_low_momentum'].rolling(window=50).mean()
    df['long_term_trend'] = (df['long_term_trend_20d'] + df['long_term_trend_50d']) / 2
    
    # Compute Trend Difference
    df['trend_difference'] = df['long_term_trend'] - df['short_term_trend']
    
    # Detect Volume Spikes
    rolling_vol_std = df['Volume'].rolling(window=20).std()
    rolling_vol_mean = df['Volume'].rolling(window=20).mean()
    spike_threshold = 2 * rolling_vol_std
    df['volume_spike_indicator'] = (df['Volume'] > (rolling_vol_mean + spike_threshold)).astype(int)
    
    # Weight by Volume and Adjust for Volume Spikes
    df['volume_weighted_trend_difference'] = df['Volume'] * df['trend_difference'] * df['volume_spike_indicator']
    
    # Return the factor values
    return df['volume_weighted_trend_difference']
