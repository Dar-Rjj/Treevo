```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['volume_adjusted_momentum'] = (df['high_low_range'] + df['close_open_diff']) * df['volume']
    df['ema_10_vol_adj_mom'] = df['volume_adjusted_momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    df['daily_return'] = df['close'].pct_change()
    df['abs_daily_return'] = df['daily_return'].abs()
    df['dynamic_market_volatility'] = df['daily_return'].rolling(window=10).std()
    df['modified_vol_adj_mom'] = df['ema_10_vol_adj_mom'] / df['dynamic_market_volatility']
    
    # Integrate Non-Linear Transformation
    df['sqrt_transformed_mom'] = np.sqrt(df['modified_vol_adj_mom'])
    df['log_transformed_mom'] = np.log(df['modified_vol_adj_mom'] + 1)  # +1 to avoid log(0)
    
    # Incorporate Trend Reversal Signal
    df['short_term_mom'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_mom'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_reversal'] = df['short_term_mom'] - df['long_term_mom']
    df['reversal_points'] = np.sign(df['momentum_reversal'])
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    df['advanced_smoothed_reversal'] = (df['reversal_points'].ewm(span=5, adjust=False).mean() + 
                                       df['reversal_points'].ewm(span=10, adjust=False).mean()) / 2
    df['combined_signal'] = df['log_transformed_mom'] + df['advanced_smoothed_reversal']
    
    # Refine Final Alpha Factor
    df['interim_alpha_factor'] = df['combined_signal']
    df['final_alpha_factor'] = (df['interim_alpha_factor'].ewm(span=5, adjust=False).mean() + 
                                df['interim_alpha_factor'].ewm(span=10, adjust=False).mean()) / 2
    
    return df['final_alpha_factor']
```
