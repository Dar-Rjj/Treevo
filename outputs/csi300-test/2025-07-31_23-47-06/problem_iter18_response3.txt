import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (close_open_diff + high_low_range) * df['volume']
    volume_adjusted_momentum_smoothed = volume_adjusted_momentum.ewm(span=10).mean()
    
    # Adjust for Market Volatility
    daily_return = df['close'].diff()
    abs_daily_return = daily_return.abs()
    market_volatility = daily_return.rolling(window=30).std()
    modified_volume_momentum = volume_adjusted_momentum_smoothed - market_volatility
    
    # Integrate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5).mean()
    long_term_momentum = df['close'].ewm(span=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_signal = np.where(momentum_reversal > 0, 1, -1)
    
    # Combine Modified Momentum with Reversal Signal
    combined_alpha_factor = modified_volume_momentum + reversal_signal
    
    # Integrate Non-Linear Transformation
    sqrt_transformed = np.sqrt(np.abs(combined_alpha_factor))
    log_transformed = np.log(np.abs(combined_alpha_factor) + 1e-10)  # Adding small value to avoid log(0)
    
    # Refine Final Alpha Factor
    final_alpha_factor = (sqrt_transformed + log_transformed) * reversal_signal
    final_alpha_factor_smoothed = final_alpha_factor.ewm(span=5).mean()
    
    return pd.Series(final_alpha_factor_smoothed, index=df.index, name='alpha_factor')
