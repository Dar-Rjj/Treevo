```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday Price Indicators
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum_high_low = high_low_range * df['volume']
    volume_adjusted_momentum_close_open = close_open_diff * df['volume']

    # Apply Adaptive Smoothing (10-day EMA)
    ema_10 = lambda x: x.ewm(span=10, adjust=False).mean()
    smoothed_volume_adjusted_momentum_high_low = ema_10(volume_adjusted_momentum_high_low)
    smoothed_volume_adjusted_momentum_close_open = ema_10(volume_adjusted_momentum_close_open)

    # Adjust for Dynamic Market Volatility
    daily_returns = df['close'].pct_change()
    abs_daily_returns = daily_returns.abs()
    dynamic_market_volatility = abs_daily_returns.rolling(window=10).std()

    modified_volume_adjusted_momentum_high_low = smoothed_volume_adjusted_momentum_high_low / dynamic_market_volatility
    modified_volume_adjusted_momentum_close_open = smoothed_volume_adjusted_momentum_close_open / dynamic_market_volatility

    # Integrate Non-Linear Transformations
    sqrt_transformed_momentum_high_low = np.sqrt(modified_volume_adjusted_momentum_high_low)
    log_transformed_momentum_high_low = np.log(modified_volume_adjusted_momentum_high_low + 1)  # Adding 1 to avoid log(0)

    sqrt_transformed_momentum_close_open = np.sqrt(modified_volume_adjusted_momentum_close_open)
    log_transformed_momentum_close_open = np.log(modified_volume_adjusted_momentum_close_open + 1)

    # Incorporate Early Reversal Signals
    short_term_momentum = ema_10(df['close'])
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    early_signal_indicator = short_term_momentum.ewm(span=3, adjust=False).mean() - long_term_momentum.ewm(span=3, adjust=False).mean()

    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    advanced_smoothed_reversal_signal = (short_term_momentum.ewm(span=5, adjust=False).mean() +
                                         short_term_momentum.ewm(span=10, adjust=False).mean()) / 2

    combined_factor_high_low = sqrt_transformed_momentum_high_low + log_transformed_momentum_high_low + advanced_smoothed_reversal_signal
    combined_factor_close_open = sqrt_transformed_momentum_close_open + log_transformed_momentum_close_open + advanced_smoothed_reversal_signal

    # Refine Final Alpha Factor
    interim_alpha_factor_high_low = ema_10(combined_factor_high_low)
    final_alpha_factor_high_low = (interim_alpha_factor_high_low.ewm(span=5, adjust=False).mean() +
                                    interim_alpha_factor_high_low.ewm(span=10, adjust=False).mean()) / 2

    interim_alpha_factor_close_open = ema_10(combined_factor_close_open)
    final_alpha_factor_close_open = (interim_alpha_factor_close_open.ewm(span=5, adjust=False).mean() +
                                      interim_alpha_factor_close_open.ewm(span=10, adjust=False).mean()) / 2

    final_alpha_factor = (final_alpha_factor_high_low + final_alpha_factor_close_open) / 2
    return final_alpha_factor
```
