```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    intraday_movement = (high_low_range + close_open_diff) / 2
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = intraday_movement * df['volume']
    volume_adjusted_momentum_smoothed = volume_adjusted_momentum.ewm(span=10).mean()
    
    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].pct_change()
    abs_daily_return = daily_return.abs()
    dynamic_volatility = abs_daily_return.rolling(window=10).std()
    
    modified_volume_adjusted_momentum = volume_adjusted_momentum_smoothed / (dynamic_volatility + 1e-8)
    
    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(np.abs(modified_volume_adjusted_momentum))
    log_transformed_momentum = np.log(modified_volume_adjusted_momentum + 1e-8)
    
    non_linear_transformed_momentum = (sqrt_transformed_momentum + log_transformed_momentum) / 2
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5).mean()
    long_term_momentum = df['close'].ewm(span=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_signal = momentum_reversal.apply(lambda x: 1 if x > 0 else -1)
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    smoothed_reversal_5d = reversal_signal.ewm(span=5).mean()
    smoothed_reversal_10d = reversal_signal.ewm(span=10).mean()
    advanced_smoothed_reversal = (smoothed_reversal_5d + smoothed_reversal_10d) / 2
    
    interim_alpha_factor = non_linear_transformed_momentum + advanced_smoothed_reversal
    
    # Final Advanced Adaptive Smoothing
    final_alpha_factor_5d = interim_alpha_factor.ewm(span=5).mean()
    final_alpha_factor_10d = interim_alpha_factor.ewm(span=10).mean()
    final_alpha_factor = (final_alpha_factor_5d + final_alpha_factor_10d) / 2
    
    return final_alpha_factor
```
