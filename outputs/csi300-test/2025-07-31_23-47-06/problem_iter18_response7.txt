import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    intraday_movement = df['high_low_range'] + df['close_open_diff']

    # Incorporate Volume Influence
    volume_adjusted_momentum = intraday_movement * df['volume']
    df['vol_adj_mom_ema'] = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()

    # Adjust for Market Volatility
    df['daily_return'] = df['close'].pct_change()
    df['abs_daily_return'] = df['daily_return'].abs()
    df['market_volatility'] = df['daily_return'].rolling(window=30).std()
    df['mod_vol_adj_mom'] = df['vol_adj_mom_ema'] / df['market_volatility']

    # Integrate Non-Linear Transformation
    df['sqrt_transform'] = np.sqrt(df['mod_vol_adj_mom'])
    df['log_transform'] = np.log(df['mod_vol_adj_mom'])

    # Incorporate Trend Reversal Signal
    df['short_term_mom'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_mom'] = df['close'].ewm(span=20, adjust=False).mean()
    df['mom_reversal'] = df['short_term_mom'] - df['long_term_mom']
    df['reversal_signal'] = np.sign(df['mom_reversal'])

    # Enhance Reversal Signal with Adaptive Smoothing
    df['smoothed_reversal'] = df['reversal_signal'].ewm(span=5, adjust=False).mean()
    df['combined_signal'] = df['sqrt_transform'] + df['smoothed_reversal']

    # Refine Final Alpha Factor
    df['interim_alpha_factor'] = df['combined_signal']
    df['final_alpha_factor'] = df['interim_alpha_factor'].ewm(span=5, adjust=False).mean()

    return df['final_alpha_factor']
