```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['volume_adjusted_momentum'] = (df['close_open_diff'] * df['volume']).rolling(window=10).mean()
    
    # Adjust for Market Volatility
    df['daily_return'] = df['close'].pct_change()
    df['market_volatility'] = df['daily_return'].rolling(window=30).std()
    df['adjusted_momentum'] = df['volume_adjusted_momentum'] - df['market_volatility']
    
    # Incorporate Trend Reversal Signal
    df['short_term_mom'] = df['close'].rolling(window=5).mean()
    df['long_term_mom'] = df['close'].rolling(window=20).mean()
    df['momentum_reversal'] = df['short_term_mom'] - df['long_term_mom']
    
    # Label Positive and Negative Reversals
    df['reversal_signal'] = df['momentum_reversal'].apply(lambda x: 1 if x > 0 else -1)
    
    # Integrate Reversal Signal into Final Factor
    df['final_factor'] = df['adjusted_momentum'] + df['reversal_signal']
    
    return df['final_factor'].dropna()
```
