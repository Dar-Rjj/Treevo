```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['intraday_movement'] = df['high'] - df['low']
    
    # Compute Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine Intraday and Opening Movement
    df['combined_movement'] = df['intraday_movement'] + df['opening_gap']
    
    # Volume Shock Calculation
    df['volume_shock'] = df['volume'] / df['volume'].shift(1)
    
    # Relative Strength Calculation
    df['relative_strength'] = df['close'] / df['close'].shift(5)
    
    # Multi-Period Intraday Momentum
    df['multi_period_momentum'] = df['intraday_movement'].rolling(window=5).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['combined_movement'] 
                          * df['volume_shock'] 
                          * df['relative_strength'] 
                          * df['multi_period_momentum'])
    
    return df['alpha_factor']
```
