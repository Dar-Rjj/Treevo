```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (high_low_range + close_open_diff) * df['volume']
    volume_adjusted_ema = volume_adjusted_momentum.ewm(span=10).mean()
    
    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].diff()
    abs_daily_return = daily_return.abs()
    dynamic_volatility = daily_return.rolling(window=10).std()
    modified_volume_momentum = volume_adjusted_ema / (dynamic_volatility + 1e-8)
    
    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(modified_volume_momentum)
    log_transformed_momentum = np.log(modified_volume_momentum + 1e-8)
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5).mean()
    long_term_momentum = df['close'].ewm(span=20).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    advanced_smoothed_reversal = momentum_reversal.ewm(span=5).mean() + momentum_reversal.ewm(span=10).mean()
    interim_alpha_factor = advanced_smoothed_reversal + sqrt_transformed_momentum + log_transformed_momentum
    
    # Refine Final Alpha Factor
    final_alpha_factor = interim_alpha_factor.ewm(span=5).mean() + interim_alpha_factor.ewm(span=10).mean()
    
    return final_alpha_factor
```
