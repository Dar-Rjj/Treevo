```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Daily Close-to-Open Momentum
    daily_momentum = df['close'] - df['open']
    
    # Calculate Volume Spike Indicator
    prev_5_days_avg_volume = df['volume'].rolling(window=5).mean().shift(1)
    volume_spike = (df['volume'] > 1.5 * prev_5_days_avg_volume).astype(int)
    
    # Combine Factors
    combined_intraday = intraday_return * volume_spike
    combined_daily_momentum = daily_momentum * volume_spike
    alpha_factor = combined_intraday + combined_daily_momentum
    
    return alpha_factor
```
