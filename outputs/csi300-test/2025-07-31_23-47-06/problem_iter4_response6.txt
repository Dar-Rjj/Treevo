```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Momentum
    high_to_low_momentum = df['high'] - df['low']
    
    # Calculate Short-Term Trend
    short_term_5_day_avg = high_to_low_momentum.rolling(window=5).mean()
    short_term_10_day_avg = high_to_low_momentum.rolling(window=10).mean()
    short_term_trend = (short_term_5_day_avg * 0.7) + (short_term_10_day_avg * 0.3)
    
    # Calculate Long-Term Trend
    long_term_20_day_avg = high_to_low_momentum.rolling(window=20).mean()
    long_term_50_day_avg = high_to_low_momentum.rolling(window=50).mean()
    long_term_trend = (long_term_20_day_avg * 0.6) + (long_term_50_day_avg * 0.4)
    
    # Compute Trend Difference
    trend_difference = short_term_trend - long_term_trend
    
    # Incorporate Volatility
    daily_return = df['close'].pct_change()
    daily_abs_return = abs(daily_return)
    volatility = daily_abs_return.rolling(window=30).std()
    volatility_adjusted_trend = trend_difference / volatility
    
    # Weight by Volume
    volume_weighted_trend = volatility_adjusted_trend * df['volume']
    
    # Adjust for Market Sentiment
    daily_market_sentiment = df['close'] - df['open']
    market_sentiment_10_day_avg = daily_market_sentiment.rolling(window=10).mean()
    final_alpha_factor = volume_weighted_trend * market_sentiment_10_day_avg
    
    return final_alpha_factor
```
