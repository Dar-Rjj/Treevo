```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_movement = df['high'] - df['low']
    
    # Compute Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday and Opening Movements
    combined_movement = intraday_movement + opening_gap
    
    # Volume Shock Calculation
    volume_shock = df['volume'] / df['volume'].shift(1)
    
    # Multiply Combined Movement by Volume Shock
    factor = combined_movement * volume_shock
    
    return factor
```
