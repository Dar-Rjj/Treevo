import pandas as pd
import numpy as np
from scipy.stats import zscore

def heuristics_v2(df):
    # Compute Intraday Price Indicators
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['volume_adjusted_momentum_high_low'] = df['volume'] * df['high_low_range']
    df['volume_adjusted_momentum_close_open'] = df['volume'] * df['close_open_diff']
    
    # Apply Adaptive Smoothing
    df['ema_10_vol_adj_mom_high_low'] = df['volume_adjusted_momentum_high_low'].ewm(span=10, adjust=False).mean()
    df['ema_10_vol_adj_mom_close_open'] = df['volume_adjusted_momentum_close_open'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    df['daily_returns'] = df['close'].pct_change()
    df['abs_daily_returns'] = df['daily_returns'].abs()
    df['dynamic_market_volatility'] = df['daily_returns'].rolling(window=10).std()
    
    df['vol_adj_mom_high_low_normalized'] = df['volume_adjusted_momentum_high_low'] / df['dynamic_market_volatility']
    df['vol_adj_mom_close_open_normalized'] = df['volume_adjusted_momentum_close_open'] / df['dynamic_market_volatility']
    
    # Integrate Non-Linear Transformations
    df['sqrt_vol_adj_mom_high_low'] = np.sqrt(np.abs(df['vol_adj_mom_high_low_normalized']))
    df['log_vol_adj_mom_high_low'] = np.log1p(np.abs(df['vol_adj_mom_high_low_normalized']))
    
    df['sqrt_vol_adj_mom_close_open'] = np.sqrt(np.abs(df['vol_adj_mom_close_open_normalized']))
    df['log_vol_adj_mom_close_open'] = np.log1p(np.abs(df['vol_adj_mom_close_open_normalized']))
    
    # Incorporate Early Reversal Signals
    df['short_term_momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    
    df['momentum_reversal'] = df['short_term_momentum'] - df['long_term_momentum']
    df['reversal_points'] = np.where(df['momentum_reversal'] > 0, 1, -1)
    
    df['early_signal_short_term'] = df['short_term_momentum'].ewm(span=3, adjust=False).mean()
    df['early_signal_long_term'] = df['long_term_momentum'].ewm(span=3, adjust=False).mean()
    df['early_reversal_points'] = np.where(df['early_signal_short_term'] > df['early_signal_long_term'], 1, -1)
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    df['smoothed_reversal_signal_5'] = df['reversal_points'].ewm(span=5, adjust=False).mean()
    df['smoothed_reversal_signal_10'] = df['reversal_points'].ewm(span=10, adjust=False).mean()
    df['smoothed_reversal_signal'] = (df['smoothed_reversal_signal_5'] + df['smoothed_reversal_signal_10']) / 2
    
    df['final_factor_non_linear'] = df['sqrt_vol_adj_mom_high_low'] + df['log_vol_adj_mom_high_low']
    df['interim_alpha_factor'] = df['final_factor_non_linear'] + df['smoothed_reversal_signal']
    
    # Refine Final Alpha Factor
    df['final_alpha_ema_5'] = df['interim_alpha_factor'].ewm(span=5, adjust=False).mean()
    df['final_alpha_ema_10'] = df['interim_alpha_factor'].ewm(span=10, adjust=False).mean()
    df['final_alpha_factor'] = (df['final_alpha_ema_5'] + df['final_alpha_ema_10']) / 2
    
    return df['final_alpha_factor']
