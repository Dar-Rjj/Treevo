import pandas as pd

def heuristics_v2(df):
    """
    Calculate a Volume Weighted Close-to-Open Return factor.

    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and indexed by date.

    Returns:
    pd.Series: A Series of the calculated factor values, indexed by date.
    """
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume'].shift(1)
    
    return volume_weighted_return
