```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = df['volume'] * (df['high_low_range'] + df['close_open_diff'])
    
    # Adaptive Smoothing via 10-day EMA
    df['vol_adj_mom_ema'] = df['volume_adjusted_momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    df['daily_return'] = df['close'].pct_change()
    df['abs_daily_return'] = df['daily_return'].abs()
    df['market_volatility'] = df['daily_return'].rolling(window=30).std()
    
    df['modified_momentum'] = df['vol_adj_mom_ema'] - df['market_volatility']
    
    # Integrate Trend Reversal Signal
    df['short_term_mom'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_mom'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_reversal'] = df['short_term_mom'] - df['long_term_mom']
    df['reversal_signal'] = np.sign(df['momentum_reversal'])
    
    # Combine Modified Momentum with Reversal Signal
    df['combined_alpha_factor'] = df['modified_momentum'] + df['reversal_signal']
    
    # Integrate Non-Linear Transformation
    df['sqrt_transformed'] = np.sqrt(df['combined_alpha_factor'].abs())
    df['log_transformed'] = np.log1p(df['combined_alpha_factor'].abs())
    df['non_linear_combined'] = df['sqrt_transformed'] + df['log_transformed']
    
    # Refine Final Alpha Factor
    df['final_alpha_factor'] = df['non_linear_combined'].ewm(span=5, adjust=False).mean()
    
    return df['final_alpha_factor']
```
