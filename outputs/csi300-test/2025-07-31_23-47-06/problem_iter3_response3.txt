import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Compute Close-Open Difference
    df['Close_Open_Diff'] = df['close'] - df['open']
    
    # Calculate Intraday Price Movement
    df['Intraday_Price_Movement'] = (df['High_Low_Range'] + df['Close_Open_Diff']) / 2
    
    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum'] = df['Intraday_Price_Movement'] * df['volume']
    
    # Smoothing via Moving Average
    df['Volume_Adjusted_Momentum_SMA10'] = df['Volume_Adjusted_Momentum'].rolling(window=10).mean()
    
    # Adjust for Market Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Market_Volatility'] = df['Daily_Return'].rolling(window=30).std()
    df['Adjusted_Momentum'] = df['Volume_Adjusted_Momentum_SMA10'] - df['Market_Volatility']
    
    # Incorporate Trend Reversal Signal
    df['Short_Term_Momentum'] = df['close'] - df['close'].rolling(window=5).mean()
    df['Long_Term_Momentum'] = df['close'] - df['close'].rolling(window=20).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    df['Reversal_Signal'] = df['Momentum_Reversal'].apply(lambda x: 1 if x > 0 else -1)
    
    # Integrate Reversal Signal into Final Factor
    df['Final_Alpha_Factor'] = df['Adjusted_Momentum'] + df['Reversal_Signal']
    
    return df['Final_Alpha_Factor']
