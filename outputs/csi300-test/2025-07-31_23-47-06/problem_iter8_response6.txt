```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Calculate Intraday Price Movement
    intraday_price_movement = (high_low_range + close_open_diff) / 2
    
    # Volume-Adjusted Momentum
    volume_adjusted_momentum = intraday_price_movement * df['volume']
    
    # Smoothing via Exponential Moving Average
    ema_10_vol_adj_mom = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    daily_return = df['close'].pct_change()
    abs_daily_return = daily_return.abs()
    volatility = daily_return.rolling(window=30).std()
    
    # Modify Intraday Momentum with Market Volatility
    adjusted_momentum = ema_10_vol_adj_mom - volatility
    
    # Enhance Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    
    # Identify Reversal Points
    reversal_signal = np.sign(momentum_reversal).diff().fillna(0)
    
    # Integrate Reversal Signal into Final Factor
    interim_alpha_factor = adjusted_momentum + reversal_signal
    
    # Refine Final Alpha Factor
    final_alpha_factor = interim_alpha_factor.ewm(span=5, adjust=False).mean()
    
    return final_alpha_factor
```
