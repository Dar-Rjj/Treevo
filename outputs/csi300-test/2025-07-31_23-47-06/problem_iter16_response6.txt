```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (high_low_range + close_open_diff) * df['volume']
    
    # Adaptive Smoothing via Moving Average
    ema_period = 20  # This can be made dynamic based on recent volatility
    volume_adjusted_momentum_ema = volume_adjusted_momentum.ewm(span=ema_period, adjust=False).mean()
    
    # Adjust for Market Volatility
    daily_return = df['close'].pct_change()
    absolute_daily_return = daily_return.abs()
    robust_market_volatility = absolute_daily_return.rolling(window=30).apply(lambda x: np.median(np.abs(x - np.median(x))), raw=True)
    adjusted_volume_momentum = volume_adjusted_momentum_ema / robust_market_volatility
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_points = np.sign(momentum_reversal) - np.sign(momentum_reversal.shift())
    
    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(adjusted_volume_momentum)
    log_transformed_momentum = np.log(adjusted_volume_momentum)
    
    # Enhance Reversal Signal with Adaptive Smoothing
    smoothed_reversal_signal = reversal_points.ewm(span=ema_period, adjust=False).mean()
    combined_signal = (sqrt_transformed_momentum + log_transformed_momentum) + smoothed_reversal_signal
    
    # Refine Final Alpha Factor
    final_alpha_factor = combined_signal.ewm(span=ema_period, adjust=False).mean()
    
    return final_alpha_factor
```
