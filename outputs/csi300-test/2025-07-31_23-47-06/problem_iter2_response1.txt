```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute exponentially smoothed returns
    alpha = 0.1  # Smoothing factor
    df['smoothed_return'] = df['daily_return'].ewm(alpha=alpha, adjust=False).mean()
    
    # Detect volume spikes
    df['volume_change'] = df['volume'].diff()
    rolling_window = 20  # Rolling window for standard deviation
    multiplier = 2  # Multiplier for the threshold
    df['volume_change_std'] = df['volume_change'].rolling(window=rolling_window).std()
    df['volume_spike_threshold'] = df['volume_change_std'] * multiplier
    df['volume_spike'] = (df['volume_change'] > df['volume_spike_threshold']).astype(int)
    
    # Combine momentum and volume breakout
    boost_value = 10  # Define the boost value
    df['combined_factor'] = df['smoothed_return']
    df.loc[df['volume_spike'] == 1, 'combined_factor'] += boost_value * df['smoothed_return']
    
    return df['combined_factor']
```
