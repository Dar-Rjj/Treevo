```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']

    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum'] = (df['Close_Open_Diff'] * df['volume'])
    df['Volume_Adjusted_EMA'] = df['Volume_Adjusted_Momentum'].ewm(span=10, adjust=False).mean()

    # Adjust for Dynamic Market Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Absolute_Daily_Return'] = df['Daily_Return'].abs()
    df['Dynamic_Volatility'] = df['Daily_Return'].rolling(window=10).std()
    df['Modified_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_EMA'] / df['Dynamic_Volatility']

    # Integrate Non-Linear Transformation
    df['Sqrt_Transformed_Momentum'] = np.sqrt(np.abs(df['Modified_Volume_Adjusted_Momentum']))
    df['Log_Transformed_Momentum'] = np.log(np.abs(df['Modified_Volume_Adjusted_Momentum']))

    # Incorporate Trend Reversal Signal
    df['Short_Term_Momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_Term_Momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    df['Reversal_Signal'] = df['Momentum_Reversal'].apply(lambda x: 1 if x > 0 else -1)

    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    df['Reversal_Signal_EMA_5'] = df['Reversal_Signal'].ewm(span=5, adjust=False).mean()
    df['Reversal_Signal_EMA_10'] = df['Reversal_Signal'].ewm(span=10, adjust=False).mean()
    df['Smoothed_Reversal_Signal'] = 0.5 * (df['Reversal_Signal_EMA_5'] + df['Reversal_Signal_EMA_10'])

    # Combine Smoothed Reversal Signal with Non-Linearly Transformed Momentum
    df['Interim_Alpha_Factor'] = df['Log_Transformed_Momentum'] + df['Smoothed_Reversal_Signal']

    # Final Advanced Adaptive Smoothing
    df['Final_Alpha_Factor_EMA_5'] = df['Interim_Alpha_Factor'].ewm(span=5, adjust=False).mean()
    df['Final_Alpha_Factor_EMA_10'] = df['Interim_Alpha_Factor'].ewm(span=10, adjust=False).mean()
    df['Final_Alpha_Factor'] = 0.5 * (df['Final_Alpha_Factor_EMA_5'] + df['Final_Alpha_Factor_EMA_10'])

    return df['Final_Alpha_Factor']
```
