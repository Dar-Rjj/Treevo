```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['Intraday_Price_Movement'] = (df['High_Low_Range'] + df['Close_Open_Diff']) / 2
    df['Volume_Adjusted_Momentum'] = df['Intraday_Price_Movement'] * df['volume']
    df['Volume_Adjusted_Momentum_10EMA'] = df['Volume_Adjusted_Momentum'].ewm(span=10).mean()
    
    # Adjust for Market Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Std_30day_Daily_Returns'] = df['Daily_Return'].rolling(window=30).std()
    df['Modified_Intraday_Momentum'] = df['Volume_Adjusted_Momentum_10EMA'] - df['Std_30day_Daily_Returns']
    
    # Enhance Trend Reversal Signal
    df['Short_Term_Momentum'] = df['close'].ewm(span=5).mean()
    df['Long_Term_Momentum'] = df['close'].ewm(span=20).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    df['Integrated_Final_Factor'] = df['Momentum_Reversal'] + df['Modified_Intraday_Momentum']
    
    # Refine Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Integrated_Final_Factor'].ewm(span=5).mean()
    
    return df['Final_Alpha_Factor'].dropna()
```
