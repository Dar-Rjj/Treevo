import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']
    df['Intraday_Price_Movement'] = (df['High_Low_Range'] + df['Close_Open_Diff']) / 2
    
    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum'] = df['Intraday_Price_Movement'] * df['volume']
    df['Volume_Adjusted_Momentum_EMA'] = df['Volume_Adjusted_Momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Absolute_Return_Diff'] = df['Daily_Return'].abs()
    df['Dynamic_Market_Volatility'] = df['Daily_Return'].rolling(window=30).std()
    df['Adjusted_Momentum'] = df['Volume_Adjusted_Momentum_EMA'] - df['Dynamic_Market_Volatility']
    
    # Incorporate Advanced Trend Reversal Signal
    df['Short_Term_Momentum'] = df['close'].rolling(window=5).mean()
    df['Long_Term_Momentum'] = df['close'].rolling(window=20).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    df['Reversal_Signal'] = np.sign(df['Momentum_Reversal'])
    df['Interim_Alpha_Factor'] = df['Adjusted_Momentum'] + df['Reversal_Signal']
    
    # Refine Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Interim_Alpha_Factor'].ewm(span=5, adjust=False).mean()
    
    return df['Final_Alpha_Factor']
