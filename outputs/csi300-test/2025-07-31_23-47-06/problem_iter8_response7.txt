```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Difference'] = df['close'] - df['open']
    df['Intraday_Price_Movement'] = (df['High_Low_Range'] + df['Close_Open_Difference']) / 2
    
    # Incorporate Volume Influence
    df['Volume_Adjusted_Momentum'] = df['Intraday_Price_Movement'] * df['volume']
    df['10_day_EMA_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    df['Daily_Return'] = df['close'].diff()
    df['Absolute_Return_Difference'] = df['Daily_Return'].abs()
    df['30_day_Stdev_Daily_Returns'] = df['Daily_Return'].rolling(window=30).std()
    df['Modified_Intraday_Momentum'] = df['Volume_Adjusted_Momentum'] - df['30_day_Stdev_Daily_Returns']
    
    # Enhance Trend Reversal Signal
    df['5_day_EMA_Close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['20_day_EMA_Close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Reversal'] = df['5_day_EMA_Close'] - df['20_day_EMA_Close']
    df['Integrated_Final_Factor'] = df['Momentum_Reversal'] + df['Modified_Intraday_Momentum']
    
    # Refine Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Integrated_Final_Factor'].ewm(span=5, adjust=False).mean()
    
    return df['Final_Alpha_Factor']
```
