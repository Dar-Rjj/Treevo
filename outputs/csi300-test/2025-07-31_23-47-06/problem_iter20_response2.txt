import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (close_open_diff * df['volume']).rolling(window=10).mean()
    volume_adjusted_momentum_smoothed = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    daily_return = df['close'].pct_change()
    absolute_daily_return = np.abs(daily_return)
    market_volatility = absolute_daily_return.rolling(window=30).std()
    modified_volume_adjusted_momentum = volume_adjusted_momentum_smoothed - market_volatility
    
    # Integrate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_signal = np.sign(momentum_reversal)
    
    # Combine Modified Momentum with Reversal Signal
    combined_alpha_factor = modified_volume_adjusted_momentum + reversal_signal
    
    # Integrate Non-Linear Transformation
    sqrt_transformed_factor = np.sqrt(np.abs(combined_alpha_factor))
    log_transformed_factor = np.log1p(np.abs(combined_alpha_factor))
    non_linear_combined_factor = sqrt_transformed_factor * log_transformed_factor * np.sign(combined_alpha_factor)
    
    # Refine Final Alpha Factor
    final_alpha_factor = non_linear_combined_factor.ewm(span=5, adjust=False).mean()
    
    return final_alpha_factor
