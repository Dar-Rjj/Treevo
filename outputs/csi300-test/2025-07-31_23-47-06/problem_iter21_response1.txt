```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday Price Indicators
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']

    # Incorporate Volume Influence
    df['volume_adjusted_momentum_high_low'] = df['high_low_range'] * df['volume']
    df['volume_adjusted_momentum_close_open'] = df['close_open_diff'] * df['volume']

    # Apply Adaptive Smoothing
    df['ema_volume_adjusted_momentum_high_low_10d'] = df['volume_adjusted_momentum_high_low'].ewm(span=10).mean()
    df['ema_volume_adjusted_momentum_close_open_10d'] = df['volume_adjusted_momentum_close_open'].ewm(span=10).mean()

    # Adjust for Dynamic Market Volatility
    df['daily_returns'] = df['close'].pct_change()
    df['absolute_daily_returns'] = df['daily_returns'].abs()
    df['dynamic_market_volatility'] = df['daily_returns'].rolling(window=10).std()

    df['modified_momentum_high_low'] = df['ema_volume_adjusted_momentum_high_low_10d'] / df['dynamic_market_volatility']
    df['modified_momentum_close_open'] = df['ema_volume_adjusted_momentum_close_open_10d'] / df['dynamic_market_volatility']

    # Integrate Non-Linear Transformations
    df['sqrt_modified_momentum_high_low'] = np.sqrt(df['modified_momentum_high_low'])
    df['log_modified_momentum_close_open'] = np.log(np.abs(df['modified_momentum_close_open']) + 1e-6)

    # Incorporate Early Reversal Signals
    df['short_term_momentum'] = df['close'].ewm(span=5).mean()
    df['long_term_momentum'] = df['close'].ewm(span=20).mean()
    df['momentum_reversal'] = df['short_term_momentum'] - df['long_term_momentum']
    df['reversal_points'] = np.sign(df['momentum_reversal'])

    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    df['smoothed_reversal_signal_5d'] = df['reversal_points'].ewm(span=5).mean()
    df['smoothed_reversal_signal_10d'] = df['reversal_points'].ewm(span=10).mean()
    df['combined_smoothed_reversal_signal'] = 0.5 * (df['smoothed_reversal_signal_5d'] + df['smoothed_reversal_signal_10d'])

    # Combine Smoothed Reversal Signal with Non-Linearly Transformed Momentum
    df['interim_alpha_factor'] = df['combined_smoothed_reversal_signal'] + df['sqrt_modified_momentum_high_low'] + df['log_modified_momentum_close_open']

    # Refine Final Alpha Factor
    df['final_alpha_factor_5d'] = df['interim_alpha_factor'].ewm(span=5).mean()
    df['final_alpha_factor_10d'] = df['interim_alpha_factor'].ewm(span=10).mean()
    df['final_alpha_factor'] = 0.5 * (df['final_alpha_factor_5d'] + df['final_alpha_factor_10d'])

    return df['final_alpha_factor']
```
