```python
import pandas as pd
import numpy as np

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate Intraday Price Movement
    high_low_range = data['high'] - data['low']
    close_open_diff = data['close'] - data['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = close_open_diff * data['volume']
    smoothed_momentum = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    daily_return = data['close'].pct_change()
    abs_daily_return = daily_return.abs()
    market_volatility = daily_return.rolling(window=30).std()
    adjusted_momentum = smoothed_momentum - market_volatility
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = data['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = data['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_signal = (momentum_reversal > 0).astype(int) - (momentum_reversal < 0).astype(int)
    
    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(adjusted_momentum)
    log_transformed_momentum = np.log(adjusted_momentum + 1e-6)  # Adding small constant to avoid log(0)
    
    # Integrate Reversal Signal into Final Factor
    interim_alpha_factor = sqrt_transformed_momentum + log_transformed_momentum + reversal_signal
    final_alpha_factor = interim_alpha_factor.ewm(span=5, adjust=False).mean()
    
    return final_alpha_factor
```
