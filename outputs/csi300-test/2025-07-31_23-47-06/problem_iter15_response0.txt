```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['volume_adjusted_momentum'] = df['volume'] * (df['high_low_range'] + df['close_open_diff'])
    df['vol_adj_mom_ema'] = df['volume_adjusted_momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    df['daily_return'] = df['close'].pct_change()
    df['abs_daily_return'] = df['daily_return'].abs()
    df['market_volatility'] = df['daily_return'].rolling(window=30).std()
    df['vol_adj_mom_normalized'] = df['vol_adj_mom_ema'] / df['market_volatility']
    
    # Incorporate Trend Reversal Signal
    df['short_term_mom'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_mom'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_reversal'] = df['short_term_mom'] - df['long_term_mom']
    df['reversal_signal'] = np.sign(df['momentum_reversal'])
    
    # Integrate Non-Linear Transformation
    df['sqrt_transformed_mom'] = np.sqrt(np.abs(df['vol_adj_mom_normalized']))
    df['log_transformed_mom'] = np.log(np.abs(df['vol_adj_mom_normalized']) + 1)
    
    # Enhance Reversal Signal with Adaptive Smoothing
    df['smoothed_reversal'] = df['reversal_signal'].ewm(span=5, adjust=False).mean()
    df['combined_signal'] = df['sqrt_transformed_mom'] + df['log_transformed_mom'] + df['smoothed_reversal']
    
    # Refine Final Alpha Factor
    df['final_alpha'] = df['combined_signal'].ewm(span=5, adjust=False).mean()
    
    return df['final_alpha']
```
