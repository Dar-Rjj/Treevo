```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Difference'] = df['close'] - df['open']

    # Incorporate Volume Influence
    df['Intraday_Movement'] = df['Close_Open_Difference'] + df['High_Low_Range']
    df['Volume_Adjusted_Momentum'] = df['Intraday_Movement'] * df['volume']
    df['EMA_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].ewm(span=10, adjust=False).mean()

    # Adjust for Market Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Adaptive_Volatility'] = df['Daily_Return'].ewm(com=30, adjust=False).std()
    df['Adjusted_Momentum'] = df['EMA_Volume_Adjusted_Momentum'] - df['Adaptive_Volatility']

    # Enhance Trend Reversal Signal
    df['Short_Term_Momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_Term_Momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Dynamic_Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']

    # Integrate Reversal Signal into Final Factor
    df['Final_Factor'] = df['Dynamic_Momentum_Reversal'] + df['Adjusted_Momentum']

    # Refine Final Alpha Factor
    df['Smoothed_Final_Factor'] = df['Final_Factor'].ewm(span=5, adjust=False).mean()

    return df['Smoothed_Final_Factor']
```
