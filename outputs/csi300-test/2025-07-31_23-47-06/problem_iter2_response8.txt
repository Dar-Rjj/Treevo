```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Momentum
    high_to_low_momentum = (df['high'] - df['low']) / df['low']
    
    # Calculate Short-Term Trend
    short_term_trend_5 = high_to_low_momentum.rolling(window=5).mean()
    short_term_trend_10 = high_to_low_momentum.rolling(window=10).mean()
    
    # Calculate Long-Term Trend
    long_term_trend_20 = high_to_low_momentum.rolling(window=20).mean()
    long_term_trend_50 = high_to_low_momentum.rolling(window=50).mean()
    
    # Compute Short-Term and Long-Term Trend Differences
    short_term_diff = short_term_trend_5 - short_term_trend_10
    long_term_diff = long_term_trend_20 - long_term_trend_50
    
    # Weight by Volume
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_weight = df['volume'] / volume_ma_10
    
    # Combine Trends
    combined_short_term = short_term_diff * volume_weight
    combined_long_term = long_term_diff * volume_weight
    
    # Final Alpha Factor
    alpha_factor = combined_short_term + combined_long_term
    
    return alpha_factor
```
