```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (high_low_range + close_open_diff) * df['volume']
    volume_adjusted_momentum_ema10 = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].pct_change()
    absolute_daily_return = daily_return.abs()
    dynamic_market_volatility = absolute_daily_return.rolling(window=20).std()
    modified_volume_adjusted_momentum = volume_adjusted_momentum_ema10 / dynamic_market_volatility
    
    # Integrate Non-Linear Transformation
    sqrt_transformed = np.sqrt(modified_volume_adjusted_momentum)
    log_transformed = np.log(modified_volume_adjusted_momentum)
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_signal = np.sign(momentum_reversal - momentum_reversal.shift(1))
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    reversal_signal_ema5 = reversal_signal.ewm(span=5, adjust=False).mean()
    reversal_signal_ema10 = reversal_signal.ewm(span=10, adjust=False).mean()
    smoothed_reversal_signal = 0.5 * (reversal_signal_ema5 + reversal_signal_ema10)
    
    # Create Interim Alpha Factor
    interim_alpha_factor = 0.5 * (sqrt_transformed + log_transformed) * smoothed_reversal_signal
    
    # Final Advanced Adaptive Smoothing
    final_alpha_factor_ema7 = interim_alpha_factor.ewm(span=7, adjust=False).mean()
    final_alpha_factor_ema14 = interim_alpha_factor.ewm(span=14, adjust=False).mean()
    final_alpha_factor = 0.5 * (final_alpha_factor_ema7 + final_alpha_factor_ema14)
    
    return final_alpha_factor
```
