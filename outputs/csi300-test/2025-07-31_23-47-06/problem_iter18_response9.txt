import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_high_low = df['high'] - df['low']
    intraday_close_open = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (intraday_close_open) * df['volume']
    volume_adjusted_momentum_smoothed = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    daily_return = df['close'].pct_change()
    dynamic_market_volatility = daily_return.rolling(window=20).std()
    modified_volume_adjusted_momentum = volume_adjusted_momentum_smoothed / dynamic_market_volatility
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_points = np.sign(momentum_reversal)
    
    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(modified_volume_adjusted_momentum.fillna(0))
    log_transformed_momentum = np.log(modified_volume_adjusted_momentum.fillna(1))
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    smoothed_reversal_signal_5 = momentum_reversal.ewm(span=5, adjust=False).mean()
    smoothed_reversal_signal_10 = momentum_reversal.ewm(span=10, adjust=False).mean()
    combined_smoothed_reversal_signal = (smoothed_reversal_signal_5 + smoothed_reversal_signal_10) / 2
    
    # Combine Smoothed Reversal Signal with Non-Linearly Transformed Momentum
    interim_alpha_factor = combined_smoothed_reversal_signal + sqrt_transformed_momentum + log_transformed_momentum
    
    # Final Advanced Adaptive Smoothing
    final_alpha_factor_5 = interim_alpha_factor.ewm(span=5, adjust=False).mean()
    final_alpha_factor_10 = interim_alpha_factor.ewm(span=10, adjust=False).mean()
    final_alpha_factor = (final_alpha_factor_5 + final_alpha_factor_10) / 2
    
    return final_alpha_factor
