```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['volume_weighted_intraday_movement'] = (df['high_low_range'] + df['close_open_diff']) * df['volume']
    df['volume_weighted_ema'] = df['volume_weighted_intraday_movement'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    df['daily_return'] = df['close'].pct_change()
    df['abs_daily_return'] = df['daily_return'].abs()
    df['dynamic_volatility'] = df['daily_return'].rolling(window=10).std()
    df['adjusted_volume_weighted_ema'] = df['volume_weighted_ema'] / df['dynamic_volatility']
    
    # Incorporate Trend Reversal Signal
    df['short_term_momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_reversal'] = df['short_term_momentum'] - df['long_term_momentum']
    df['reversal_signal'] = np.sign(df['momentum_reversal'])
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    df['reversal_signal_ema_5'] = df['reversal_signal'].ewm(span=5, adjust=False).mean()
    df['reversal_signal_ema_10'] = df['reversal_signal'].ewm(span=10, adjust=False).mean()
    df['smoothed_reversal_signal'] = 0.5 * df['reversal_signal_ema_5'] + 0.5 * df['reversal_signal_ema_10']
    df['combined_factor'] = df['adjusted_volume_weighted_ema'] + df['smoothed_reversal_signal']
    
    # Refine Final Alpha Factor
    df['interim_alpha_factor'] = df['combined_factor']
    df['final_alpha_ema_5'] = df['interim_alpha_factor'].ewm(span=5, adjust=False).mean()
    df['final_alpha_ema_10'] = df['interim_alpha_factor'].ewm(span=10, adjust=False).mean()
    df['final_alpha_factor'] = 0.5 * df['final_alpha_ema_5'] + 0.5 * df['final_alpha_ema_10']
    
    return df['final_alpha_factor']
```
