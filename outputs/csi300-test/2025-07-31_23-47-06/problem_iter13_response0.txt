import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['Intraday_Price_Movement'] = (df['High_Low_Range'] + df['Close_Open_Diff']) / 2
    df['Volume_Adjusted_Momentum'] = df['volume'] * df['Intraday_Price_Movement']
    df['EMA_10_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Market Volatility
    df['Daily_Return'] = df['close'].diff()
    df['Absolute_Daily_Return'] = df['Daily_Return'].abs()
    df['Market_Volatility'] = df['Daily_Return'].rolling(window=30).std()
    df['Modified_Volume_Adjusted_Momentum'] = df['EMA_10_Volume_Adjusted_Momentum'] - df['Market_Volatility']
    
    # Incorporate Trend Reversal Signal
    df['Short_Term_Momentum'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_Term_Momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Reversal'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    df['Reversal_Points'] = np.sign(df['Momentum_Reversal'])
    df['Smoothed_Reversal_Signal'] = df['Momentum_Reversal'].ewm(span=5, adjust=False).mean()
    
    # Integrate Non-Linear Transformation
    df['Square_Root_Transformation'] = np.sqrt(df['Modified_Volume_Adjusted_Momentum'])
    df['Logarithmic_Transformation'] = np.log(df['Modified_Volume_Adjusted_Momentum'] + 1)  # +1 to avoid log(0)
    
    # Enhance Reversal Signal with Adaptive Smoothing
    df['Combined_Momentum'] = df['Square_Root_Transformation'] + df['Logarithmic_Transformation']
    df['Interim_Alpha_Factor'] = df['Combined_Momentum'] + df['Smoothed_Reversal_Signal']
    df['Final_Alpha_Factor'] = df['Interim_Alpha_Factor'].ewm(span=5, adjust=False).mean()
    
    return df['Final_Alpha_Factor']
