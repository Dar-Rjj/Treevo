import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Incorporate Volume Influence
    df['volume_adjusted_momentum'] = df['volume'] * (df['high_low_range'] + df['close_open_diff'])
    df['ema_10_vol_adj_mom'] = df['volume_adjusted_momentum'].ewm(span=10).mean()
    
    # Adjust for Market Volatility
    df['daily_return'] = df['close'].pct_change()
    df['abs_daily_return'] = df['daily_return'].abs()
    df['market_volatility'] = df['daily_return'].rolling(window=30).std()
    df['modified_vol_adj_mom'] = df['ema_10_vol_adj_mom'] - df['market_volatility']
    
    # Incorporate Trend Reversal Signal
    df['short_term_momentum'] = df['close'].ewm(span=5).mean()
    df['long_term_momentum'] = df['close'].ewm(span=20).mean()
    df['momentum_reversal'] = df['short_term_momentum'] - df['long_term_momentum']
    df['reversal_signal'] = np.sign(df['momentum_reversal'])
    df['smoothed_reversal_signal'] = df['reversal_signal'].ewm(span=5).mean()
    
    # Integrate Non-Linear Transformation
    df['sqrt_transformed_mom'] = np.sqrt(np.abs(df['modified_vol_adj_mom']))
    df['log_transformed_mom'] = np.log(np.abs(df['modified_vol_adj_mom']) + 1)  # Adding 1 to avoid log(0)
    
    # Combine Smoothed Reversal Signal with Non-Linearly Transformed Momentum
    df['interim_alpha_factor'] = df['smoothed_reversal_signal'] + df['sqrt_transformed_mom'] + df['log_transformed_mom']
    
    # Final Adaptive Smoothing
    df['final_alpha_factor'] = df['interim_alpha_factor'].ewm(span=5).mean()
    
    return df['final_alpha_factor']
