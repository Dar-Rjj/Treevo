import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Incorporate Volume Influence
    volume_adjusted_momentum = (close_open_diff * df['volume'])
    volume_adjusted_momentum_ema = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    
    # Adjust for Dynamic Market Volatility
    daily_returns = df['close'].pct_change()
    abs_daily_returns = np.abs(daily_returns)
    dynamic_market_volatility = daily_returns.rolling(window=20).std()
    modified_volume_momentum = volume_adjusted_momentum / dynamic_market_volatility
    
    # Incorporate Trend Reversal Signal
    short_term_momentum = df['close'].ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()
    momentum_reversal = short_term_momentum - long_term_momentum
    reversal_points = np.sign(momentum_reversal)
    
    # Integrate Non-Linear Transformation
    sqrt_transformed_momentum = np.sqrt(modified_volume_momentum)
    log_transformed_momentum = np.log(modified_volume_momentum + 1)  # Adding 1 to avoid log(0)
    
    # Enhance Reversal Signal with Advanced Adaptive Smoothing
    smoothed_reversal_signal_5day = momentum_reversal.ewm(span=5, adjust=False).mean()
    smoothed_reversal_signal_10day = momentum_reversal.ewm(span=10, adjust=False).mean()
    combined_smoothed_reversal_signal = (smoothed_reversal_signal_5day + smoothed_reversal_signal_10day) / 2
    
    # Combine Smoothed Reversal Signal with Non-Linearly Transformed Momentum
    interim_alpha_factor = combined_smoothed_reversal_signal + (sqrt_transformed_momentum + log_transformed_momentum) / 2
    
    # Final Advanced Adaptive Smoothing
    final_alpha_factor_5day = interim_alpha_factor.ewm(span=5, adjust=False).mean()
    final_alpha_factor_10day = interim_alpha_factor.ewm(span=10, adjust=False).mean()
    final_alpha_factor = (final_alpha_factor_5day + final_alpha_factor_10day) / 2
    
    return final_alpha_factor
