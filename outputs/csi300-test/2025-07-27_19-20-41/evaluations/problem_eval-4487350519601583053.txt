import pandas as pd
import numpy as np

def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    std_price_range = hl_diff.rolling(window=10).std()
    sma_volume_sqrt = np.sqrt(df['volume']).rolling(window=5).mean()
    heuristics_matrix = std_price_range * sma_volume_sqrt
    return heuristics_matrix
