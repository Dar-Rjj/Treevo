import pandas as pd
import numpy as np

def heuristics_v2(df):
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    std_daily_return = df['close'].pct_change().rolling(window=10).std()
    roc_volume = df['volume'].pct_change(periods=5)
    heuristics_matrix = (ema_close * roc_volume) + (std_daily_return * (1 - roc_volume))
    return heuristics_matrix
