import pandas as pd

def heuristics_v2(df):
    close_prices = df['close']
    volumes = df['volume']
    avg_last_5_closes = close_prices.rolling(window=5).mean()
    std_dev_volume_10 = volumes.rolling(window=10).std()
    heuristics_matrix = (avg_last_5_closes - close_prices) / std_dev_volume_10
    return heuristics_matrix
