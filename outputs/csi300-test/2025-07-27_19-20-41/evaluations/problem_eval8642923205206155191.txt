import pandas as pd
import numpy as np

def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    volume_weighted_diff = hl_diff * np.log(df['volume'] + 1)
    weights = np.arange(1, 16)
    heuristics_matrix = volume_weighted_diff.rolling(window=15).apply(lambda x: (x * weights[:len(x)]).sum() / weights[:len(x)].sum(), raw=True)
    return heuristics_matrix
