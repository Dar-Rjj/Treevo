import pandas as pd
import numpy as np

def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    ewma_price_range = hl_diff.ewm(span=10).mean()
    cumsum_volume_ln = np.log(df['volume']).rolling(window=5).sum()
    heuristics_matrix = ewma_price_range * cumsum_volume_ln
    return heuristics_matrix
