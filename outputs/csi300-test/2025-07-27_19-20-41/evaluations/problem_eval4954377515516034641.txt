def heuristics_v2(df):
    # Calculate the price change for each day
    price_change = df['close'] - df['open']
    # Inverse of volume as weights
    inv_volume = 1 / df['volume']
    # Multiply price changes by their corresponding inverse volume weights
    weighted_changes = price_change * inv_volume
    # Compute a 20-day rolling sum of the weighted price changes
    heuristics_matrix = weighted_changes.rolling(window=20).sum()
    return heuristics_matrix
