import pandas as pd
import numpy as np

def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    ema_price_range = hl_diff.ewm(span=30, adjust=False).mean()
    cumsum_volume_cbrt = np.cbrt(df['volume']).rolling(window=15).sum()
    heuristics_matrix = ema_price_range * cumsum_volume_cbrt
    return heuristics_matrix
