import pandas as pd
import numpy as np

def heuristics_v2(df):
    sma_close = df['close'].rolling(window=50).mean()
    daily_returns = df['close'].pct_change()
    std_daily_returns = daily_returns.rolling(window=25).std()
    heuristics_matrix = sma_close / std_daily_returns
    return heuristics_matrix
