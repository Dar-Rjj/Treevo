import pandas as pd
import numpy as np

def heuristics_v2(df):
    pos_diff = df['close'].diff().clip(lower=0)
    neg_diff = -df['close'].diff().clip(upper=0)
    sum_diff = (pos_diff + neg_diff).rolling(window=10).sum()
    sma = df['close'].rolling(window=30).mean()
    heuristics_matrix = sum_diff / sma
    return heuristics_matrix
