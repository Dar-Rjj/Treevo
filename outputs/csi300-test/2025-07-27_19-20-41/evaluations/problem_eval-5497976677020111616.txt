def heuristics_v2(df):
    ema_close = df['close'].ewm(span=14, adjust=False).mean()
    std_daily_return = df['close'].pct_change().rolling(window=20).std()
    roc_volume = df['volume'].pct_change(periods=5)
    roc_close = df['close'].pct_change(periods=10)
    weight_roc_volume = roc_volume / (roc_volume + roc_close)
    heuristics_matrix = (ema_close * weight_roc_volume) + (std_daily_return * (1 - weight_roc_volume))
    return heuristics_matrix
