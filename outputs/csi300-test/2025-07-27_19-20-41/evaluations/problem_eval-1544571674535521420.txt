import pandas as pd
import numpy as np

def heuristics_v2(df):
    sma_close = df['close'].rolling(window=50).mean()
    std_daily_log_return = np.log(df['close']).diff().rolling(window=20).std()
    roc_volume = df['volume'].pct_change(periods=10)
    heuristics_matrix = (sma_close * roc_volume) + (std_daily_log_return * (1 - roc_volume))
    return heuristics_matrix
