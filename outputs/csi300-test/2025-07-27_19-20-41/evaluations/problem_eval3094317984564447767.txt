import pandas as pd
import numpy as np

def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    ema_price_range = hl_diff.ewm(span=20, adjust=False).mean()
    sum_log_volume = np.log(df['volume']).rolling(window=10).sum()
    heuristics_matrix = ema_price_range / sum_log_volume
    return heuristics_matrix
