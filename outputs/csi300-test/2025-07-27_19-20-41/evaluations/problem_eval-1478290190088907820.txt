import pandas as pd
import numpy as np

def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    sma_price_range = hl_diff.rolling(window=5).mean()
    cumsum_volume_sqrt = np.sqrt(df['volume']).rolling(window=10).sum()
    heuristics_matrix = sma_price_range * cumsum_volume_sqrt
    return heuristics_matrix
