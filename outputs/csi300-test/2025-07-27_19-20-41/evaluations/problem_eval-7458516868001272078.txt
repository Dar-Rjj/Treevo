def heuristics_v2(df):
    price_range = df['high'] - df['low']
    ema_price_range = price_range.ewm(span=20, adjust=False).mean()
    cum_volume_product = df['volume'].rolling(window=20).apply(lambda x: np.prod(x), raw=True)
    heuristics_matrix = ema_price_range / cum_volume_product
    return heuristics_matrix
