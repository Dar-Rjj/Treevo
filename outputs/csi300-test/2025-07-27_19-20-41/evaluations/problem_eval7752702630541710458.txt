def heuristics_v2(df):
    # Calculate the exponential moving average of the high-low difference
    ema_high_low_diff = (df['high'] - df['low']).ewm(span=20).mean()
    # Calculate the exponential moving average of the volume
    ema_volume = df['volume'].ewm(span=20).mean()
    # Generate the heuristic value
    heuristics_matrix = ema_high_low_diff / ema_volume
    return heuristics_matrix
