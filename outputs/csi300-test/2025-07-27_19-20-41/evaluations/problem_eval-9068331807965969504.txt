def heuristics_v2(df):
    momentum_30 = df['close'].pct_change(periods=30)
    mean_volume_10 = df['volume'].rolling(window=10).mean()
    std_volume_10 = df['volume'].rolling(window=10).std()
    coefficient_of_variation = std_volume_10 / mean_volume_10
    inverse_coefficient_of_variation = 1 / coefficient_of_variation
    heuristics_matrix = momentum_30 * inverse_coefficient_of_variation
    return heuristics_matrix
