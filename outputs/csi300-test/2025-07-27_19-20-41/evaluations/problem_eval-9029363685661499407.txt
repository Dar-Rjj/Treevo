import pandas as pd
import numpy as np

def heuristics_v2(df):
    rsi = 100 - (100 / (1 + (df['close'].diff(1).clip(lower=0).rolling(window=14).mean() / df['close'].diff(1).clip(upper=0).abs().rolling(window=14).mean())))
    sma_volume = df['volume'].rolling(window=20).mean()
    roc_daily_return = df['close'].pct_change().pct_change(periods=7)
    heuristics_matrix = (rsi * roc_daily_return) + (sma_volume * (1 - roc_daily_return))
    return heuristics_matrix
