def heuristics_v2(df):
    def calculate_trend(data, col_name):
        return (data[col_name] - data[col_name].shift(1)) / data[col_name].shift(1)

    trend_close = calculate_trend(df, 'close')
    volume_change = calculate_trend(df, 'volume')
    amount_change = calculate_trend(df, 'amount')
    
    heuristics_matrix = 0.5 * trend_close + 0.3 * volume_change + 0.2 * amount_change
    return heuristics_matrix
