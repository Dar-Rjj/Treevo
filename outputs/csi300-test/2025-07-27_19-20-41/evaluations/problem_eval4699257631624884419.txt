import pandas as pd
import numpy as np

def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    ema_price_range = hl_diff.ewm(span=10).mean()
    cumprod_volume_sqrt = np.sqrt(df['volume']).rolling(window=5).apply(np.prod, raw=True)
    heuristics_matrix = ema_price_range / cumprod_volume_sqrt
    return heuristics_matrix
