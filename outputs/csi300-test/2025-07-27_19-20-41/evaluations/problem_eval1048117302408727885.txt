def heuristics_v2(df):
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    ema_log_return = df['log_return'].ewm(span=10, adjust=False).mean()
    delta = df['close'].diff()
    avg_gain = delta.where(delta > 0, 0).rolling(window=28).mean()
    avg_loss = -delta.where(delta < 0, 0).rolling(window=28).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    heuristics_matrix = ema_log_return + rsi
    return heuristics_matrix
