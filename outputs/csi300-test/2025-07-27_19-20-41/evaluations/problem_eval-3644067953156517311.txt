def heuristics_v2(df):
    ma5 = df['close'].rolling(window=5).mean()
    ma20 = df['close'].rolling(window=20).mean()
    momentum_factor = ma5 - ma20
    volatility_factor = df['close'].pct_change().rolling(window=10).std()
    heuristics_matrix = momentum_factor + volatility_factor
    return heuristics_matrix
