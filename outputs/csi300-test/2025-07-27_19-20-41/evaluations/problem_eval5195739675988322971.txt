import pandas as pd
import numpy as np

def heuristics_v2(df):
    close_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_change = df['close'].diff()
    std_price_change = price_change.rolling(window=10).std()
    heuristics_matrix = close_ema / std_price_change
    return heuristics_matrix
