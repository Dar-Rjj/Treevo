def heuristics_v2(df):
    sma_close = df['close'].rolling(window=30).mean()
    std_daily_return = df['close'].pct_change().rolling(window=15).std()
    roc_volume = df['volume'].pct_change(periods=7)
    heuristics_matrix = (sma_close * roc_volume) + (std_daily_return * (1 - roc_volume))
    return heuristics_matrix
