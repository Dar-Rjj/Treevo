def heuristics_v2(df):
    heuristics_matrix = (df['close'] - df['low']) / (df['volume'].shift(1) + 1) + \
                        (df['high'] - df['open']) / (df['high'] + df['open']) - \
                        (df['amount'].shift(2) - df['amount'].shift(3)) / df['amount']
    return heuristics_matrix
