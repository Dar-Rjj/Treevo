def heuristics_v2(df):
    # Calculate the 50-day and 200-day simple moving averages of the 'close' column
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()

    # Calculate the 30-day exponential moving average of the 'volume' column
    ema_30_volume = df['volume'].ewm(span=30, adjust=False).mean()

    # Compute the heuristic: (50-day SMA - 200-day SMA) / 30-day EMA of volume
    heuristics_matrix = (sma_50 - sma_200) / ema_30_volume

    return heuristics_matrix
