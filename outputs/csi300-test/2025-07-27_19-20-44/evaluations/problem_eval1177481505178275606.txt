def heuristics_v2(df):
    short_window = 10
    long_window = 30
    rsi_period = 14
    
    # Compute short and long simple moving averages
    short_mavg = df['close'].rolling(window=short_window, min_periods=short_window).mean()
    long_mavg = df['close'].rolling(window=long_window, min_periods=long_window).mean()
    
    # Generate SMA crossover signal
    sma_crossover = short_mavg - long_mavg
    
    # Calculate RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=rsi_period, min_periods=rsi_period).mean()
    avg_loss = loss.rolling(window=rsi_period, min_periods=rsi_period).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate ROC for volume
    roc_volume = df['volume'].pct_change(rsi_period)
    
    # Fixed weights for combination
    w_sma = 0.5
    w_rsi = 0.3
    w_roc = 0.2
    
    # Calculate the heuristic factor
    heuristics_matrix = (w_sma * sma_crossover) + (w_rsi * rsi) + (w_roc * roc_volume)
    return heuristics_matrix
