def heuristics_v2(df):
    # Calculate the 5-day and 20-day exponential moving averages of the 'close' and 'high' columns respectively
    close_ema_short = df['close'].ewm(span=5, adjust=False).mean()
    high_ema_long = df['high'].ewm(span=20, adjust=False).mean()

    # Compute the heuristic: 20-day EMA of 'high' - 5-day EMA of 'close'
    heuristics_matrix = high_ema_long - close_ema_short

    return heuristics_matrix
