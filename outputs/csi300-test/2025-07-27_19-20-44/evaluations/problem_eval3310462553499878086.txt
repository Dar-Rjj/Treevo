def heuristics_v2(df):
    roc_close = df['close'].pct_change()
    rel_diff_high_low = (df['high'] - df['low']) / (df['high'] + df['low'])
    ewma_volume = df['volume'].ewm(span=10, adjust=False).mean()
    heuristics_matrix = 0.4 * roc_close + 0.3 * rel_diff_high_low + 0.3 * ewma_volume
    return heuristics_matrix
