def heuristics_v2(df):
    lookback_period = 20
    price_change = (df['close'] - df['open']).rolling(window=lookback_period).mean()
    volume_ratio = df['volume'] / df['volume'].shift(lookback_period)
    high_low_range = (df['high'] - df['low']) / df['close']
    
    # Calculate the heuristic factor
    heuristics_matrix = (price_change + volume_ratio + high_low_range) / 3
    return heuristics_matrix
