import pandas as pd

def heuristics_v2(df):
    lookback_period = 14
    delta = df['close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up = up.rolling(window=lookback_period).mean()
    roll_down = down.abs().rolling(window=lookback_period).mean()
    RS = roll_up / roll_down
    RSI = 100.0 - (100.0 / (1.0 + RS))
    price_change = df['close'].pct_change(lookback_period)
    volume_change = df['volume'].pct_change(lookback_period)
    heuristics_matrix = (price_change * volume_change) * RSI
    return heuristics_matrix
