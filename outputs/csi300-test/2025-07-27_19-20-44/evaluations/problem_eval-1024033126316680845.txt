def heuristics_v2(df):
    lookback_period = 20
    avg_price = (df['high'] + df['low']) / 2
    price_change = (df['close'] - avg_price) / avg_price
    amount_change = df['amount'].pct_change(lookback_period)
    
    # Calculate the heuristic factor
    heuristics_matrix = price_change + amount_change
    return heuristics_matrix
