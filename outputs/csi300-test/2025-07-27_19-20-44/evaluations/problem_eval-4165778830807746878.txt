def heuristics_v2(df):
    roc_close = df['close'].pct_change()
    high_open_ratio = df['high'] / df['open']
    vol_diff = df['volume'] - df['volume'].shift(1)
    
    w_roc_close = 0.4
    w_high_open = 0.3
    w_vol_diff = 0.3
    
    heuristics_matrix = (w_roc_close * roc_close) + (w_high_open * high_open_ratio) + (w_vol_diff * vol_diff)
    return heuristics_matrix
