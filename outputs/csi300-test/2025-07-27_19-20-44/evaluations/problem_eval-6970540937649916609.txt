def heuristics_v2(df):
    # Calculate price momentum over 5 days
    price_momentum = df['close'].pct_change(5)
    # Calculate volume change compared to the previous day
    volume_change = df['volume'].pct_change()
    # Combine price momentum and volume change for the alpha factor
    heuristics_matrix = (price_momentum + volume_change) / 2
    return heuristics_matrix
