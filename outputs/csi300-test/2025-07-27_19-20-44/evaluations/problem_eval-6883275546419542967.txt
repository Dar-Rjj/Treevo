def heuristics_v2(df):
    lookback_period = 20
    price_momentum = (df['close'] - df['close'].shift(lookback_period)) / df['close'].shift(lookback_period)
    high_low_range = (df['high'] - df['low']) / df['close']
    volatility = df['close'].rolling(window=lookback_period).std()
    
    # Construct the heuristic factor
    heuristics_matrix = price_momentum / (high_low_range + volatility + 1e-6)
    return heuristics_matrix
