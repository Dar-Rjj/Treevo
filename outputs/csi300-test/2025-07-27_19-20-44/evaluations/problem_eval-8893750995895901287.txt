def heuristics_v2(df):
    heuristics_matrix = (0.7 * (df['close'] - df['open']) / df['volume']) + \
                        0.2 * (df['high'] - df['low']) / (df['high'] + df['low']) - \
                        0.1 * (df['amount'].shift(1) - df['amount'].shift(2)) / df['amount']
    return heuristics_matrix
