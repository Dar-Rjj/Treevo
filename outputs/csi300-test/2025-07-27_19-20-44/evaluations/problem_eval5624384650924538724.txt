def heuristics_v2(df):
    # Calculate the 10-day exponential moving average of the 'volume' column
    volume_ema = df['volume'].ewm(span=10, adjust=False).mean()
    # Calculate the 50-day simple moving average of the 'close' column
    close_ma_50 = df['close'].rolling(window=50).mean()
    # Calculate the 20-day simple moving average of the 'high' column
    high_ma_20 = df['high'].rolling(window=20).mean()

    # Compute the heuristic: (10-day EMA of volume / 50-day average close) - 20-day average high
    heuristics_matrix = (volume_ema / close_ma_50) - high_ma_20

    return heuristics_matrix
