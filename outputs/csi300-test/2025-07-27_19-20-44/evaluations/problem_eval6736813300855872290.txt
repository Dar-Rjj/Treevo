def heuristics_v2(df):
    # Calculate price volatility
    price_volatility = (df['high'] - df['low']) / df['close'].rolling(window=5).std()
    
    # Calculate volume trend
    volume_trend = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Calculate simple moving average for 5 and 20 days
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the moving average convergence
    mac_convergence = (sma_5 - sma_20) / sma_20
    
    # Combine the factors
    heuristics_matrix = price_volatility * volume_trend + mac_convergence
    
    return heuristics_matrix
