def heuristics_v2(df):
    # Calculate the 5-day return of the 'close' column
    close_return = df['close'].pct_change(periods=5)
    
    # Compute the daily volume change as a percentage
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combine the two factors
    heuristics_matrix = close_return + volume_change

    return heuristics_matrix
