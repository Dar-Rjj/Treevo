def heuristics_v2(df):
    # Calculate the 10-day and 50-day EMA of the 'close' column
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=50, adjust=False).mean()

    # Compute the RSI based on the 'close' price
    delta = df['close'].diff()
    up = delta.clip(lower=0)
    down = -1 * delta.clip(upper=0)
    ema_up = up.ewm(com=13, adjust=False).mean()
    ema_down = down.ewm(com=13, adjust=False).mean()
    rs = ema_up / ema_down
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 50-day moving average of the 'amount' column
    amount_ma = df['amount'].rolling(window=50).mean()
    
    # Compute the heuristic: (EMA short - EMA long) + RSI + (today's amount / 50-day average amount)
    heuristics_matrix = (ema_short - ema_long) + rsi + (df['amount'] / amount_ma)

    return heuristics_matrix
