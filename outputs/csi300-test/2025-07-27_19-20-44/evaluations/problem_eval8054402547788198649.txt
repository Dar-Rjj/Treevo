def heuristics_v2(df):
    # Calculate recent price volatility
    recent_volatility = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Time-decayed moving average of price changes
    price_change_tdma = df['close'].ewm(span=10, adjust=False).mean().diff()
    
    # Exponentially weighted measure of volume
    ewm_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Construct the heuristics matrix
    heuristics_matrix = recent_volatility * price_change_tdma / ewm_volume
    return heuristics_matrix
