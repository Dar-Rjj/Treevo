def heuristics_v2(df):
    heuristics_matrix = (df['close'] - df['open']) / df['volume'] + \
                        (df['high'] - df['low']) / (df['high'] + df['low']) - \
                        (df['amount'].shift(1) - df['amount'].shift(2)) / df['amount'] + \
                        (df['close'].rolling(window=5).mean() - df['close']) / df['close'] + \
                        (df['volume'] - df['volume'].shift(1)) / df['volume']
    return heuristics_matrix
