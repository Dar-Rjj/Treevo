def heuristics_v2(df):
    # Calculate the 10-day rate of change for the 'close' column
    roc_close = df['close'].pct_change(periods=10)
    
    # Calculate the 30-day standard deviation of the 'close' column as a measure of volatility
    std_close = df['close'].rolling(window=30).std()
    
    # Calculate the 2-day rate of change for the 'volume' column
    roc_volume = df['volume'].pct_change(periods=2)
    
    # Compute the heuristic: (rate of change of close / standard deviation of close) + 2-day rate of change of volume
    heuristics_matrix = (roc_close / std_close) + roc_volume

    return heuristics_matrix
