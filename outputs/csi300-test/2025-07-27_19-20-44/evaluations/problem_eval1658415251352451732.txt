def heuristics_v2(df):
    lookback_period = 30
    price_momentum = (df['close'] - df['open'].shift(lookback_period)) / df['open'].shift(lookback_period)
    high_low_ratio_change = (df['high'] / df['low']).pct_change(lookback_period)
    volume_change = df['volume'].pct_change(lookback_period)
    
    # Fixed weights
    w_price = 0.6
    w_high_low = 0.2
    w_volume = 0.2
    
    # Calculate the heuristic factor
    heuristics_matrix = (w_price * price_momentum) + (w_high_low * high_low_ratio_change) + (w_volume * volume_change)
    return heuristics_matrix
