def heuristics_v2(df):
    positive_changes = df['close'].diff().apply(lambda x: x if x > 0 else 0).rolling(window=10).sum()
    negative_changes = df['close'].diff().apply(lambda x: x if x < 0 else 0).rolling(window=10).sum()
    pos_neg_ratio = positive_changes / (negative_changes.abs() + 1e-8)  # Add small constant to avoid division by zero
    volume_change = df['volume'].pct_change().fillna(0)
    heuristics_matrix = pos_neg_ratio + volume_change
    return heuristics_matrix
