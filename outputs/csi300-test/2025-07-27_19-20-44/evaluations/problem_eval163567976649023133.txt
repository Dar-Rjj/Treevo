def heuristics_v2(df):
    heuristics_matrix = (df['close'] - df['close'].shift(1)) / df['close'].shift(1) + \
                        (df['high'] - df['low']) / ((df['high'] + df['low']) / 2) - \
                        (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    return heuristics_matrix
