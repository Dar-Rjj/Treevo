def heuristics_v2(df):
    # Calculate the 20-day and 50-day moving averages of the 'close' column
    close_ma20 = df['close'].rolling(window=20).mean()
    close_ma50 = df['close'].rolling(window=50).mean()

    # Calculate the 10-day moving average of the 'volume' column
    volume_ma10 = df['volume'].rolling(window=10).mean()

    # Compute the heuristic: (20-day MA of close - 50-day MA of close) / 10-day MA of volume
    heuristics_matrix = (close_ma20 - close_ma50) / volume_ma10

    return heuristics_matrix
