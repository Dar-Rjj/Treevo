import pandas as pd

def heuristics_v2(df):
    def rsi(series, period=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    def roc(series, period=10):
        return (series.shift(-period) - series) / series

    rsi_close = rsi(df['close'], 14)
    roc_volume = roc(df['volume'], 10)
    heuristics_matrix = rsi_close + roc_volume
    return heuristics_matrix
