def heuristics_v2(df):
    heuristics_matrix = 0.5 * (df['close'] - df['open']) / df['volume'] + \
                        (df['high'] / df['low']) * (df['volume'].shift(1) / df['volume'].shift(2)) - \
                        (df['amount'].diff(periods=2) / df['amount'])
    return heuristics_matrix
