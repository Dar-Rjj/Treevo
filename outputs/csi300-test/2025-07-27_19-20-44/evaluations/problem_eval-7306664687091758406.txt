def heuristics_v2(df):
    lookback_short = 5
    lookback_long = 20
    price_momentum_short = (df['close'] - df['open'].shift(lookback_short)) / df['open'].shift(lookback_short)
    price_momentum_long = (df['close'] - df['open'].shift(lookback_long)) / df['open'].shift(lookback_long)
    high_low_ratio = df['high'] / df['low']
    amount_change = df['amount'].pct_change(lookback_short)
    
    # Adjusted weights
    w_price_short = 0.4
    w_price_long = 0.3
    w_amount = 0.2
    w_high_low = 0.1
    
    # Calculate the heuristic factor
    heuristics_matrix = (w_price_short * price_momentum_short) + (w_price_long * price_momentum_long) + (w_amount * amount_change) + (w_high_low * high_low_ratio)
    return heuristics_matrix
