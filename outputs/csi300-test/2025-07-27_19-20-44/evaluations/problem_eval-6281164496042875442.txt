def heuristics_v2(df):
    heuristics_matrix = (df['close'] - df['open']) / (df['high'] - df['low']) + \
                        (df['volume'].shift(1) - df['volume'].shift(2)) / df['volume'] - \
                        (df['amount'].shift(1) - df['amount'].shift(2)) / (df['amount'].shift(1) + df['amount'].shift(2))
    return heuristics_matrix
