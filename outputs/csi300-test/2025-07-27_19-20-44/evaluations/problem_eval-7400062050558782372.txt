def heuristics_v2(df):
    lookback_period = 15
    price_momentum = (df['close'] - df['open'].shift(lookback_period)) / df['open'].shift(lookback_period)
    avg_high_low = (df['high'] + df['low']) / 2
    avg_high_low_momentum = (avg_high_low - avg_high_low.shift(lookback_period)) / avg_high_low.shift(lookback_period)
    amount_change = df['amount'].pct_change(lookback_period)
    volume_change = df['volume'].pct_change(lookback_period)
    
    # Different weights
    w_price = 0.4
    w_avg_high_low = 0.3
    w_amount = 0.2
    w_volume = 0.1
    
    # Calculate the heuristic factor
    heuristics_matrix = (w_price * price_momentum) + (w_avg_high_low * avg_high_low_momentum) + (w_amount * amount_change) + (w_volume * volume_change)
    return heuristics_matrix
