def heuristics_v2(df):
    heuristics_matrix = 0.4 * (df['close'] - df['open']) / df['volume'] + \
                        0.3 * (df['high'] - df['low']) / (df['high'] + df['low']) + \
                        0.3 * (df['amount'].rolling(window=3).mean() - df['amount'].shift(1)) / df['amount']
    return heuristics_matrix
