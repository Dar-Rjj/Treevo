import pandas as pd
import numpy as np

def heuristics_v2(df):
    lookback_period = 20
    ema_return = df['close'].pct_change().ewm(span=lookback_period, adjust=False).mean()
    vol_std = df['volume'].rolling(window=lookback_period).std()
    heuristics_matrix = np.log(ema_return + 1) + np.log(vol_std + 1)
    return heuristics_matrix
