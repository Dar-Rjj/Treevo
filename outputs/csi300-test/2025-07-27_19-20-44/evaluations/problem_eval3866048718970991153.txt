def heuristics_v2(df):
    price_strength = (df['close'] - df['open']) / df['open']
    volatility = (df['high'] - df['low']) / df['close'].shift(1)
    liquidity_signal = df['amount'] / df['volume']
    
    # Custom aggregation
    heuristics_matrix = (price_strength * liquidity_signal) - volatility
    return heuristics_matrix
