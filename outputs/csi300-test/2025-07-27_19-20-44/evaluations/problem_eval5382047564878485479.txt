def heuristics_v2(df):
    heuristics_matrix = (0.4 * (df['close'] - df['open']) / df['open']) + \
                        0.3 * (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1) + \
                        0.3 * (df['high'] - df['low']) / (df['close'].shift(1))
    return heuristics_matrix
