def heuristics_v2(df):
    ema_close = df['close'].ewm(span=10).mean()
    ema_volume = df['volume'].ewm(span=10).mean()
    trading_intensity = (df['volume'] / df['amount']).rolling(window=5).mean()
    heuristics_matrix = 0.6 * (ema_close - df['close']) + 0.4 * (ema_volume * trading_intensity)
    return heuristics_matrix
