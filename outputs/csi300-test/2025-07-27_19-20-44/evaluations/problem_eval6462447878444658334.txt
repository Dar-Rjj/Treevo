def heuristics_v2(df):
    heuristics_matrix = (0.6 * (df['close'] - df['open']) / df['volume']) + \
                        0.2 * (df['high'] - df['low']) / (df['high'] + df['low']) - \
                        0.2 * (df['amount'] - df['amount'].shift(1)) / df['amount']
    return heuristics_matrix
