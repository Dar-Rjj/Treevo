def heuristics_v2(df):
    lookback_period = 10
    price_change = (df['close'] - df['close'].shift(lookback_period)) / df['close'].shift(lookback_period)
    amount_change = (df['amount'] - df['amount'].shift(lookback_period)) / df['amount'].shift(lookback_period)
    volume_change = (df['volume'] - df['volume'].shift(lookback_period)) / df['volume'].shift(lookback_period)
    
    w_price = 0.4
    w_amount = 0.35
    w_volume = 0.25
    
    heuristics_matrix = (w_price * price_change) + (w_amount * amount_change) + (w_volume * volume_change)
    return heuristics_matrix
