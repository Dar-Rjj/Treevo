import pandas as pd
import numpy as np

def heuristics_v2(df):
    short_window = 10
    long_window = 30
    very_long_window = 90
    
    short_ema = df['close'].ewm(span=short_window, adjust=False).mean()
    long_ema = df['close'].ewm(span=long_window, adjust=False).mean()
    very_long_ema = df['close'].ewm(span=very_long_window, adjust=False).mean()

    # Custom weight for emphasizing recent price movements
    momentum_factor = (2 * short_ema - long_ema) / 2
    
    # Calculate RSI over the long window as a measure of market condition
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=long_window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=long_window).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    heuristics_matrix = momentum_factor * (rsi / 50)
    return heuristics_matrix
