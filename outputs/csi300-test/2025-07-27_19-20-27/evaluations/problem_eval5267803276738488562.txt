def heuristics_v2(df):
    # Calculate On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).fillna(0).cumsum()
    
    # Calculate True Range (TR)
    tr1 = df['high'] - df['low']
    tr2 = np.abs(df['high'] - df['close'].shift())
    tr3 = np.abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate +DM and -DM
    dm_pos = df['high'].diff()
    dm_neg = -df['low'].diff()
    dm_pos[(dm_pos < 0) | (dm_neg > dm_pos)] = 0
    dm_neg[(dm_neg < 0) | (dm_pos > dm_neg)] = 0
    
    # Calculate smoothed +DM and -DM
    sma_dm_pos = dm_pos.rolling(window=14).mean()
    sma_dm_neg = dm_neg.rolling(window=14).mean()
    
    # Calculate DMI
    dmi_plus = (sma_dm_pos / true_range.rolling(window=14).mean()) * 100
    dmi_minus = (sma_dm_neg / true_range.rolling(window=14).mean()) * 100
    dmi = dmi_plus - dmi_minus
    
    # Combine OBV and DMI to form the heuristics matrix
    heuristics_matrix = (obv.pct_change(periods=14) * 100) + dmi
    
    return heuristics_matrix
