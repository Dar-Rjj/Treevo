def heuristics_v2(df):
    # Calculate 20-day simple moving average (SMA)
    sma = df['close'].rolling(window=20).mean()
    # Calculate 20-day standard deviation
    std = df['close'].rolling(window=20).std()
    # Calculate upper and lower Bollinger Bands
    upper_band = sma + 2 * std
    lower_band = sma - 2 * std
    # Calculate the Bollinger Bandwidth
    bollinger_bandwidth = (upper_band - lower_band) / sma
    
    # Calculate the Chaikin Money Flow (CMF)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combine Bollinger Bandwidth and CMF to form the heuristic factor
    heuristics_matrix = bollinger_bandwidth + cmf
    
    return heuristics_matrix
