def heuristics_v2(df):
    short_window = 10
    long_window = 30
    very_long_window = 90
    
    # Calculate exponential moving averages
    short_ema = df['close'].ewm(span=short_window, adjust=False).mean()
    long_ema = df['close'].ewm(span=long_window, adjust=False).mean()
    very_long_ema = df['close'].ewm(span=very_long_window, adjust=False).mean()

    # Calculate short-term and long-term volatility
    short_volatility = df['close'].rolling(window=short_window).std()
    long_volatility = df['close'].rolling(window=long_window).std()

    # Dynamic weighting based on the ratio of short-term to long-term volatility
    weight_factor = (short_volatility / long_volatility).ewm(alpha=0.15, adjust=False).mean()
    trend_factor = (weight_factor * short_ema + (1 - weight_factor) * long_ema)

    # Calculate the smoothed historical average true range
    tr = df['high'] - df['low']
    atr = tr.ewm(span=long_window, adjust=False).mean()

    heuristics_matrix = trend_factor / atr
    return heuristics_matrix
