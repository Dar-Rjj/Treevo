def heuristics_v2(df):
    # Calculate the 14-day Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day Rate of Change (ROC)
    roc = df['close'].pct_change(periods=10) * 100
    
    # Calculate the 30-day Moving Average (MA)
    ma_30 = df['close'].rolling(window=30).mean()
    ma_diff = df['close'] - ma_30
    
    # Combine RSI, ROC, and MA difference to form the heuristic factor
    heuristics_matrix = (rsi - 50) + roc + ma_diff
    
    return heuristics_matrix
