def heuristics_v2(df):
    sma_10 = df['close'].rolling(window=10).mean()
    vol_30 = df['volume'].rolling(window=30).mean()
    
    price_momentum = df['close'] - sma_10
    trading_activity = df['volume'] / vol_30
    
    heuristics_matrix = price_momentum * trading_activity
    return heuristics_matrix
