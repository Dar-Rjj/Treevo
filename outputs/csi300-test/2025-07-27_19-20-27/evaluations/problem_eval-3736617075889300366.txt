import pandas as pd
import numpy as np

def heuristics_v2(df):
    very_short_window = 2
    long_window = 60
    
    log_returns = np.log(df['close']).diff()
    
    very_short_term_log_return = log_returns.rolling(window=very_short_window).sum()
    long_term_volatility = log_returns.rolling(window=long_window).std()

    heuristics_matrix = (very_short_term_log_return) / long_term_volatility
    return heuristics_matrix
