import pandas as pd

def heuristics_v2(df):
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    very_long_ema = df['close'].ewm(span=60, adjust=False).mean()

    # Calculate the average true range (ATR)
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=20).mean()

    # Combine EMA differences and ATR
    ema_diff = (short_ema - long_ema) / 2 + (long_ema - very_long_ema) / 2
    heuristics_matrix = ema_diff / atr
    return heuristics_matrix
