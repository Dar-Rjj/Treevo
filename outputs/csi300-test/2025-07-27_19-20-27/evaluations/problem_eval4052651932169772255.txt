def heuristics_v2(df):
    short_window = 10
    medium_window = 25
    
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the signed volume
    df['signed_volume'] = df['volume'] * df['daily_return'].apply(lambda x: 1 if x > 0 else -1)
    
    # Cumulative sum of the signed volume over a short window
    cum_signed_vol = df['signed_volume'].rolling(window=short_window, min_periods=1).sum()
    
    # Adjust the cumulative sum by the ratio of the current close to its 25-day minimum
    adj_cum_signed_vol = cum_signed_vol / (df['close'] / df['close'].rolling(window=medium_window, min_periods=1).min())
    
    heuristics_matrix = adj_cum_signed_vol
    return heuristics_matrix
