import pandas as pd

def heuristics_v2(df):
    short_window = 5
    long_window = 20
    very_long_window = 60
    
    # Calculate rate of change for different windows
    roc_short = df['close'].pct_change(periods=short_window)
    roc_long = df['close'].pct_change(periods=long_window)
    roc_very_long = df['close'].pct_change(periods=very_long_window)

    # Combine ROCs with custom weights
    momentum_factor = (2 * roc_short - roc_long) / 2

    # Adjust momentum factor by volume ratio
    avg_volume = df['volume'].rolling(window=long_window).mean()
    volume_ratio = df['volume'] / avg_volume

    heuristics_matrix = momentum_factor * volume_ratio
    return heuristics_matrix
