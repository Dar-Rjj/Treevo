import pandas as pd

def heuristics_v3(df):
    ema_10 = df['close'].ewm(halflife=10).mean()
    high_30 = df['high'].rolling(window=30).max()
    low_30 = df['low'].rolling(window=30).min()
    high_low_ratio = high_30 / low_30
    heuristics_matrix = ema_10 * high_low_ratio
    return heuristics_matrix
