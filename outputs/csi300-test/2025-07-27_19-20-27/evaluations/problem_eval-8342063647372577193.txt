def heuristics_v2(df):
    # Calculate 5-day and 20-day exponential moving averages for close prices
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute the Accumulation/Distribution Line
    adl = (df['close'] - df['low']) - (df['high'] - df['close'])
    adl = adl * df['volume'] / (df['high'] - df['low'])
    adl = adl.cumsum()
    
    # Combine the EMA convergence and ADL
    heuristics_matrix = (ema_5 - ema_20) + adl
    
    return heuristics_matrix
