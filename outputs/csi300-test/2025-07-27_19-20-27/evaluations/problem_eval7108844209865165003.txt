import pandas as pd
import numpy as np

def heuristics_v2(df):
    short_window = 5
    long_window = 20
    very_long_window = 60
    
    short_ema = df['close'].ewm(span=short_window, adjust=False).mean()
    long_ema = df['close'].ewm(span=long_window, adjust=False).mean()
    very_long_ema = df['close'].ewm(span=very_long_window, adjust=False).mean()

    # Custom weight for emphasizing recent price movements
    trend_factor = (2 * short_ema - long_ema) / 2
    
    # Calculate average true range over the long window as a measure of volatility
    tr1 = df['high'] - df['low']
    tr2 = np.abs(df['high'] - df['close'].shift())
    tr3 = np.abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=long_window).mean()

    heuristics_matrix = trend_factor / atr
    return heuristics_matrix
