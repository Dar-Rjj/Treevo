def heuristics_v2(df):
    # Calculate the 9-day Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=9).mean()
    avg_loss = loss.rolling(window=9).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 5-day Rate of Change (ROC)
    roc = df['close'].pct_change(periods=5) * 100
    
    # Combine RSI and ROC to form the heuristic factor
    heuristics_matrix = (rsi - 50) + roc
    
    return heuristics_matrix
