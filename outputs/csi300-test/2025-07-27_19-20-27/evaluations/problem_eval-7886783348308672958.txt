def heuristics_v2(df):
    # Calculate the 5-day, 20-day, and 60-day percentage price changes
    pct_change_5 = df['close'].pct_change(periods=5) * 100
    pct_change_20 = df['close'].pct_change(periods=20) * 100
    pct_change_60 = df['close'].pct_change(periods=60) * 100
    
    # Calculate the volume-to-price ratio
    vpr = df['volume'] / df['close']
    
    # Combine the percentage changes with weights
    heuristics_matrix = (0.5 * pct_change_5 + 0.3 * pct_change_20 + 0.2 * pct_change_60) * vpr
    
    return heuristics_matrix
