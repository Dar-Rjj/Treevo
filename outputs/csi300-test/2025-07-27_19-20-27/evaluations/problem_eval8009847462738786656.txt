def heuristics_v2(df):
    # Calculate the 20-day cumulative return
    cum_ret = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate the 20-day average trading volume
    avg_vol = df['volume'].rolling(window=20).mean()
    
    # Combine cumulative return with average volume to form the heuristic factor
    heuristics_matrix = cum_ret * avg_vol
    
    return heuristics_matrix
