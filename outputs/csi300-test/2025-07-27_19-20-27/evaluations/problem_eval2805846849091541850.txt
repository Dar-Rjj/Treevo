def heuristics_v2(df):
    # Calculate the rate of change (ROC) for the close prices over 10 days
    roc = df['close'].pct_change(periods=10)
    
    # Compute the On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Combine the ROC and OBV
    heuristics_matrix = roc + obv
    
    return heuristics_matrix
