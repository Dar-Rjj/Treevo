import pandas as pd
import talib

def heuristics_v2(df):
    ema_short = df['close'].ewm(span=5, adjust=False).mean()
    ema_long = df['close'].ewm(span=20, adjust=False).mean()
    ema_very_long = df['close'].ewm(span=60, adjust=False).mean()

    # Custom weight emphasizing recent price movements
    trend_factor = (2 * ema_short - ema_long) / 2
    tr = talib.TRANGE(df['high'], df['low'], df['close'])
    atr = talib.SMA(tr, timeperiod=14)

    heuristics_matrix = trend_factor / atr
    return heuristics_matrix
