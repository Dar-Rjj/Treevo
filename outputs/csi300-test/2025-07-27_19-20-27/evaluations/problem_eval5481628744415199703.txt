import pandas as pd

def heuristics_v2(df):
    short_window = 10
    medium_window = 30
    very_long_window = 60
    
    # Calculate exponential moving averages
    short_ema = df['close'].ewm(span=short_window, adjust=False).mean()
    medium_ema = df['close'].ewm(span=medium_window, adjust=False).mean()
    very_long_ema = df['close'].ewm(span=very_long_window, adjust=False).mean()

    # Calculate short-term and very long-term volatility
    short_volatility = df['close'].rolling(window=short_window).std()
    very_long_volatility = df['close'].rolling(window=very_long_window).std()

    # Dynamic weighting based on the ratio of short-term to very long-term volatility
    weight_factor = short_volatility / very_long_volatility
    trend_factor = (weight_factor * short_ema + (1 - weight_factor) * medium_ema) / 2

    # Calculate the historical average true range
    tr = df['high'] - df['low']
    atr = tr.rolling(window=50).mean()

    heuristics_matrix = trend_factor / atr
    return heuristics_matrix
