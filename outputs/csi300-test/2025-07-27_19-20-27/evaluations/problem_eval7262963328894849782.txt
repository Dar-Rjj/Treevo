def heuristics_v2(df):
    short_window = 10
    long_window = 30
    very_long_window = 90
    
    # Calculate weighted moving averages with different windows
    short_wma = df['close'].rolling(window=short_window).mean()
    long_wma = df['close'].rolling(window=long_window).mean()
    very_long_wma = df['close'].rolling(window=very_long_window).mean()

    # Custom weight for emphasizing recent price movements
    momentum_factor = (short_wma - long_wma) / very_long_wma

    # Calculate the average true range over the very long window as a measure of volatility
    tr1 = df['high'] - df['low']
    tr2 = np.abs(df['high'] - df['close'].shift())
    tr3 = np.abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=very_long_window).mean()

    # Adjust the momentum factor by the normalized ATR
    normalized_atr = atr / df['close']
    heuristics_matrix = momentum_factor * normalized_atr
    return heuristics_matrix
