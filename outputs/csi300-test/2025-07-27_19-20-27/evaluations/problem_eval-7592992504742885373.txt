import pandas as pd

def heuristics_v2(df):
    short_window = 10
    long_window = 30
    ema_window = 20
    cmf_period = 20
    stoch_period = 14
    
    # Calculate Exponential Moving Averages
    ema_short = df['close'].ewm(span=short_window, adjust=False).mean()
    ema_long = df['close'].ewm(span=long_window, adjust=False).mean()
    
    # Calculate Chaikin Money Flow
    mf_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    mf_volume = mf_multiplier * df['volume']
    cmf = mf_volume.rolling(window=cmf_period).sum() / df['volume'].rolling(window=cmf_period).sum()
    
    # Calculate Stochastic Oscillator
    lowest_low = df['low'].rolling(window=stoch_period).min()
    highest_high = df['high'].rolling(window=stoch_period).max()
    k_percent = 100 * (df['close'] - lowest_low) / (highest_high - lowest_low)
    
    # Combine EMA, CMF, and Stochastic Oscillator
    heuristics_matrix = (ema_short / ema_long) * cmf * k_percent
    return heuristics_matrix
