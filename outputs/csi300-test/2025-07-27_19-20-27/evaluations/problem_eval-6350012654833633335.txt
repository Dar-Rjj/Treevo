import pandas as pd
import numpy as np

def heuristics_v2(df):
    short_window = 10
    medium_window = 30
    long_window = 50
    
    short_sma = df['close'].rolling(window=short_window).mean()
    long_sma = df['close'].rolling(window=long_window).mean()

    # Custom weight for emphasizing recent price movements
    trend_factor = (2 * short_sma - long_sma) / 2
    
    # Calculate standard deviation over the medium window as a measure of stability
    std_dev = df['close'].rolling(window=medium_window).std()

    heuristics_matrix = trend_factor / std_dev
    return heuristics_matrix
