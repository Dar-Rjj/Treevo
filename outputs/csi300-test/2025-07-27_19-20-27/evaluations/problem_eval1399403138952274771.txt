def heuristics_v2(df):
    # Calculate the 12-day and 26-day Exponential Moving Averages (EMA)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the MACD line
    macd_line = ema_12 - ema_26
    
    # Calculate the 9-day EMA of the MACD line (signal line)
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Calculate the MACD histogram
    macd_histogram = macd_line - signal_line
    
    # Calculate the On-Balance Volume (OBV)
    obv = (df['volume'] * (df['close'].diff() > 0).astype(int)).cumsum()
    
    # Combine MACD histogram and OBV to form the heuristic factor
    heuristics_matrix = macd_histogram + obv
    
    return heuristics_matrix
