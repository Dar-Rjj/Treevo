import pandas as pd

def heuristics_v2(df):
    short_window = 5
    long_window = 20
    
    # Calculate the percentage change in closing prices
    short_change = df['close'].pct_change(periods=short_window).fillna(0)
    
    # Calculate the standard deviation of the percentage changes as a measure of volatility
    short_volatility = df['close'].pct_change().rolling(window=short_window).std().fillna(0)
    
    # Calculate the normalized volume
    avg_volume = df['volume'].rolling(window=long_window).mean()
    volume_factor = df['volume'] / avg_volume
    
    # Combine the short-term price change, short-term volatility, and volume factor
    heuristics_matrix = (short_change * short_volatility) * volume_factor
    return heuristics_matrix
