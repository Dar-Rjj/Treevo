def heuristics_v2(df):
    short_window = 5
    long_window = 20
    very_long_window = 60
    
    short_mavg = df['close'].rolling(window=short_window).mean()
    long_mavg = df['close'].rolling(window=long_window).mean()
    very_long_mavg = df['close'].rolling(window=very_long_window).mean()

    # Custom weight for emphasizing recent price movements
    trend_factor = (short_mavg - very_long_mavg) / (df['high'] + df['low']) * 2
    volatility = df['close'].rolling(window=long_window).std()

    heuristics_matrix = trend_factor / volatility
    return heuristics_matrix
