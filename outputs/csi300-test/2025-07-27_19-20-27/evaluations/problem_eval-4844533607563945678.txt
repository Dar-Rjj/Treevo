def heuristics_v2(df):
    # Calculate 20-day Simple Moving Average (SMA) for close prices
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate 20-day standard deviation for close prices
    std_20 = df['close'].rolling(window=20).std()
    
    # Calculate the upper and lower Bollinger Bands
    upper_band = sma_20 + 2 * std_20
    lower_band = sma_20 - 2 * std_20
    
    # Compute the Bollinger Bandwidth
    bbw = (upper_band - lower_band) / sma_20
    
    # Calculate the On-Balance Volume (OBV)
    obv = ((df['close'] > df['close'].shift(1)) * df['volume']).cumsum() - ((df['close'] < df['close'].shift(1)) * df['volume']).cumsum()
    
    # Combine the Bollinger Bandwidth and OBV
    heuristics_matrix = bbw + obv
    
    return heuristics_matrix
