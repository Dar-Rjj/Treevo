import pandas as pd
import numpy as np

def heuristics_v2(df):
    short_window = 10
    long_window = 30
    very_long_window = 90
    
    short_wma = df['close'].rolling(window=short_window).mean()
    long_wma = df['close'].rolling(window=long_window).mean()
    very_long_wma = df['close'].rolling(window=very_long_window).mean()

    # Custom weight for emphasizing recent price movements
    trend_factor = (3 * short_wma - 2 * long_wma + very_long_wma) / 2
    
    # Calculate RSI over the long window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=long_window).mean()
    avg_loss = loss.rolling(window=long_window).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    heuristics_matrix = trend_factor * (rsi / 50)
    return heuristics_matrix
