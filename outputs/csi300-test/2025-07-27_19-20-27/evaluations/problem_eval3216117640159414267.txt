def heuristics_v2(df):
    short_window = 5
    long_window = 20
    very_long_window = 60
    
    # Calculate EMAs
    short_ema = df['close'].ewm(span=short_window, adjust=False).mean()
    long_ema = df['close'].ewm(span=long_window, adjust=False).mean()
    very_long_ema = df['close'].ewm(span=very_long_window, adjust=False).mean()

    # Calculate SMAs
    short_sma = df['close'].rolling(window=short_window).mean()
    long_sma = df['close'].rolling(window=long_window).mean()
    very_long_sma = df['close'].rolling(window=very_long_window).mean()

    # Custom momentum factor considering both EMA and SMA
    momentum_factor = (2 * short_ema - long_ema + 2 * short_sma - long_sma) / 4

    # Calculate the standard deviation of returns as a measure of volatility
    returns = df['close'].pct_change()
    std_dev = returns.rolling(window=long_window).std()

    # True range for mean true range
    tr1 = df['high'] - df['low']
    tr2 = np.abs(df['high'] - df['close'].shift())
    tr3 = np.abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=long_window).mean()

    # Volatility-adjusted heuristics matrix
    heuristics_matrix = momentum_factor * (std_dev / atr)
    return heuristics_matrix
