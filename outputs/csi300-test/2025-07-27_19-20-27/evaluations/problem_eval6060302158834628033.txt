def heuristics_v2(df):
    # Calculate the 14-day RSI for the close prices
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day percentage change in volume
    pct_change_volume = df['volume'].pct_change(10).fillna(0)
    
    # Composite factor: sum of RSI and percentage change in volume
    heuristics_matrix = rsi + pct_change_volume
    
    return heuristics_matrix
