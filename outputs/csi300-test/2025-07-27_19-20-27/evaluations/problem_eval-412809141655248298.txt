def heuristics_v2(df):
    # Calculate the 14-day Relative Strength Index (RSI) then smooth it over 10 days
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    smoothed_rsi = rsi.rolling(window=10).mean()
    
    # Calculate the 5-day Rate of Change (ROC)
    roc = df['close'].pct_change(periods=5) * 100
    
    # Combine smoothed RSI and ROC to form the heuristic factor
    heuristics_matrix = (smoothed_rsi - 50) + roc
    
    return heuristics_matrix
