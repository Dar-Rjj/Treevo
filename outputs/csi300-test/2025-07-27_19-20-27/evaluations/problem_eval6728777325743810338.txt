def heuristics_v2(df):
    # Calculate On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).fillna(0).cumsum()
    
    # Calculate Bollinger Bands
    sma = df['close'].rolling(window=20).mean()
    std = df['close'].rolling(window=20).std()
    upper_band = sma + 2 * std
    lower_band = sma - 2 * std
    
    # Calculate the percentage of close price within the Bollinger Bands
    bb_percentage = (df['close'] - lower_band) / (upper_band - lower_band) * 100
    
    # Combine OBV and BB percentage into the heuristics matrix
    heuristics_matrix = obv + bb_percentage
    
    return heuristics_matrix
