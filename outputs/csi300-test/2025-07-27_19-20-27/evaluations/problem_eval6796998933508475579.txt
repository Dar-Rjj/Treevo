def heuristics_v2(df):
    # Calculate the 20-day Exponential Moving Average (EMA)
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 5-day Rate of Change (ROC)
    roc = df['close'].pct_change(periods=5) * 100
    
    # Combine EMA and ROC to form the heuristic factor
    heuristics_matrix = (df['close'] - ema_20) + roc
    
    return heuristics_matrix
