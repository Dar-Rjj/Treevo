import pandas as pd
import numpy as np

def heuristics_v2(df):
    short_window = 10
    long_window = 30
    very_long_window = 90
    
    short_sma = df['close'].rolling(window=short_window).mean()
    long_sma = df['close'].rolling(window=long_window).mean()
    very_long_sma = df['close'].rolling(window=very_long_window).mean()

    # Custom weight for emphasizing recent price movements
    momentum_factor = (2 * short_sma - long_sma) / 2
    
    # Calculate standard deviation over the long window as a measure of volatility
    std_dev = df['close'].rolling(window=long_window).std()

    heuristics_matrix = momentum_factor / std_dev
    return heuristics_matrix
