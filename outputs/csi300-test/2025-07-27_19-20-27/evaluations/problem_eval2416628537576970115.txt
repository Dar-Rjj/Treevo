import pandas as pd

def heuristics_v2(df):
    short_window = 5
    long_window = 20
    rsi_period = 14

    # Calculate the relative strength index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=rsi_period).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=rsi_period).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate short-term and long-term volume change
    short_volume_change = df['volume'].pct_change(short_window).fillna(0)
    long_volume_change = df['volume'].pct_change(long_window).fillna(0)

    # Calculate the ratio of short-term to long-term volume change
    volume_ratio = short_volume_change / long_volume_change

    # Calculate the close price momentum
    momentum = df['close'].pct_change(short_window).fillna(0)

    # Scale the momentum by the RSI and volume ratio
    heuristics_matrix = momentum * rsi * volume_ratio
    return heuristics_matrix
