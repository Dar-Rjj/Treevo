def heuristics_v2(df):
    # Calculate 10-day and 30-day exponential moving averages for close prices
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the Modified Accumulation/Distribution Line
    adl = (2 * df['close'] - df['high'] - df['low']) / (df['high'] - df['low'])
    adl = adl * df['volume']
    adl = adl.cumsum()
    
    # Combine the EMA convergence and ADL
    heuristics_matrix = (ema_10 - ema_30) + adl
    
    return heuristics_matrix
