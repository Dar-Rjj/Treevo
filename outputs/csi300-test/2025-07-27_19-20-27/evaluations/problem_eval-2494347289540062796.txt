import pandas as pd

def heuristics_v2(df):
    short_window = 5
    medium_window = 20
    long_window = 60
    
    # Calculate moving averages
    short_ma = df['close'].rolling(window=short_window).mean()
    medium_ma = df['close'].rolling(window=medium_window).mean()
    long_ma = df['close'].rolling(window=long_window).mean()
    
    # Calculate the relative strength of closing prices to moving averages
    close_to_short_ma_strength = (df['close'] - short_ma) / short_ma
    close_to_medium_ma_strength = (df['close'] - medium_ma) / medium_ma
    close_to_long_ma_strength = (df['close'] - long_ma) / long_ma
    
    # Combine the relative strengths into a single measure
    combined_strength = (close_to_short_ma_strength + close_to_medium_ma_strength + close_to_long_ma_strength) / 3
    
    # Calculate the percentage change in volume
    volume_change = df['volume'].pct_change(periods=short_window).fillna(0)
    
    # Adjust the combined strength by the volume change
    heuristics_matrix = combined_strength * (1 + volume_change)
    
    return heuristics_matrix
