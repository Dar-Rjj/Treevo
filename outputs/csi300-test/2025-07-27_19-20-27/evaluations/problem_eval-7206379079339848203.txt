def heuristics_v2(df):
    short_window = 10
    long_window = 30
    very_long_window = 90
    
    short_sma = df['close'].rolling(window=short_window).mean()
    long_sma = df['close'].rolling(window=long_window).mean()
    very_long_sma = df['close'].rolling(window=very_long_window).mean()

    # Custom weight for emphasizing recent price movements
    trend_factor = (short_sma - long_sma + very_long_sma) / 3
    
    # Calculate average true range over the long window as a measure of volatility
    tr1 = df['high'] - df['low']
    tr2 = np.abs(df['high'] - df['close'].shift())
    tr3 = np.abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=long_window).mean()

    # Adjust the trend factor by the ratio of close price to ATR
    adjustment_factor = df['close'] / atr
    heuristics_matrix = trend_factor * adjustment_factor
    return heuristics_matrix
