def heuristics_v2(df):
    # Calculate the 14-day Average True Range (ATR)
    df['TrueRange'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    atr = df['TrueRange'].rolling(window=14).mean()
    
    # Compute the Positive Volume Index (PVI)
    pvi = 1000  # Starting point for PVI
    for i in range(1, len(df)):
        if df['volume'].iloc[i] > df['volume'].iloc[i-1]:
            pvi *= (df['close'].iloc[i] / df['close'].iloc[i-1]) * 100
        else:
            pvi = pvi
        pvi = round(pvi, 2)
    df['PVI'] = pvi
    
    # Combine ATR and PVI to form the heuristic factor
    heuristics_matrix = atr + df['PVI']
    
    return heuristics_matrix
