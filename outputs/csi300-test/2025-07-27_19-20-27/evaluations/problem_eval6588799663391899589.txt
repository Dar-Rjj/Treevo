import pandas as pd
import numpy as np

def heuristics_v2(df):
    short_window = 10
    long_window = 30
    
    short_sma = df['close'].rolling(window=short_window).mean()
    long_sma = df['close'].rolling(window=long_window).mean()
    
    # Custom momentum indicator based on the difference between short and long SMAs
    momentum_factor = short_sma - long_sma
    
    # Calculate the standard deviation of returns over the long window as a measure of volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=long_window).std()
    
    heuristics_matrix = momentum_factor / volatility
    return heuristics_matrix
