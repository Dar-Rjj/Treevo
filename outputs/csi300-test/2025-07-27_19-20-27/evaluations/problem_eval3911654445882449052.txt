import pandas as pd

def heuristics_v2(df):
    # Calculate 10-day EMA
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate daily price range
    daily_range = df['high'] - df['low']
    
    # Calculate 10-day EMA of the daily price range
    ema_daily_range = daily_range.ewm(span=10, adjust=False).mean()
    
    # Compute the ratio of EMA to its EMA of the daily price range
    heuristics_matrix = ema_10 / ema_daily_range
    
    return heuristics_matrix
