import pandas as pd
import numpy as np

def heuristics_v2(df):
    short_window = 5
    long_window = 20
    very_long_window = 60
    
    # Calculate the percentage change in closing prices
    short_change = df['close'].pct_change(periods=short_window).fillna(0)
    long_change = df['close'].pct_change(periods=long_window).fillna(0)
    
    # Calculate the standard deviation of the percentage changes as a measure of volatility
    short_volatility = df['close'].pct_change().rolling(window=short_window).std().fillna(0)
    long_volatility = df['close'].pct_change().rolling(window=long_window).std().fillna(0)

    # Dynamic weighting based on the ratio of short-term to long-term volatility
    weight_factor = short_volatility / (short_volatility + long_volatility)
    momentum_factor = (weight_factor * short_change + (1 - weight_factor) * long_change)

    # Combine momentum and volatility into a single factor
    heuristics_matrix = momentum_factor / (short_volatility + long_volatility)
    return heuristics_matrix
