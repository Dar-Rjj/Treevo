def heuristics_v2(df):
    short_window = 5
    long_window = 20
    very_long_window = 60
    
    short_sma = df['close'].rolling(window=short_window).mean()
    long_sma = df['close'].rolling(window=long_window).mean()
    very_long_sma = df['close'].rolling(window=very_long_window).mean()

    # Custom weight for emphasizing recent price movements
    trend_factor = (2 * short_sma - long_sma) / 2
    
    # Calculate standard deviation over the very long window as a measure of stability
    std_dev = df['close'].rolling(window=very_long_window).std()

    heuristics_matrix = trend_factor / std_dev
    return heuristics_matrix
