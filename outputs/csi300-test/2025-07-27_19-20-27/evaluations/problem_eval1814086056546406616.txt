def heuristics_v2(df):
    short_window = 5
    long_window = 20
    very_long_window = 60
    
    # Calculate the percentage change in closing prices
    short_change = df['close'].pct_change(periods=short_window).fillna(0)
    long_change = df['close'].pct_change(periods=long_window).fillna(0)
    
    # Momentum component
    momentum_comp = (short_change - long_change)
    
    # Volatility measures
    short_volatility = df['close'].pct_change().rolling(window=short_window).std().fillna(0)
    very_long_volatility = df['close'].pct_change().rolling(window=very_long_window).std().fillna(0)
    
    # Adjust volatility for a more comprehensive risk assessment
    volatility_adjustment = (short_volatility + very_long_volatility) / 2
    
    # Combine momentum and adjusted volatility
    heuristics_matrix = momentum_comp / volatility_adjustment
    return heuristics_matrix
