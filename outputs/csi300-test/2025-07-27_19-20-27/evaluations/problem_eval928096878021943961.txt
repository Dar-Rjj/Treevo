def heuristics_v2(df):
    # Calculate the 14-day Price Acceleration
    price_acceleration = (df['close'].diff(periods=2) - df['close'].diff()) / 2
    
    # Calculate the 20-day Exponential Moving Average
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 20-day Average True Range as a measure of volatility
    tr1 = df['high'] - df['low']
    tr2 = np.abs(df['high'] - df['close'].shift())
    tr3 = np.abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=20).mean()
    
    # Adjust the EMA by the ATR to reflect market volatility
    adjusted_ema = ema_20 / atr
    
    # Combine Price Acceleration and Adjusted EMA to form the heuristic factor
    heuristics_matrix = price_acceleration + adjusted_ema
    
    return heuristics_matrix
