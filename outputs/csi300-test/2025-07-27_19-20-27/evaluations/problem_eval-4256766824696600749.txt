import pandas as pd
import numpy as np

def heuristics_v2(df):
    short_window = 10
    long_window = 30
    very_long_window = 90
    
    short_ema = df['close'].ewm(span=short_window, adjust=False).mean()
    long_ema = df['close'].ewm(span=long_window, adjust=False).mean()
    very_long_ema = df['close'].ewm(span=very_long_window, adjust=False).mean()

    # Custom weight for emphasizing recent price movements
    trend_factor = (2 * short_ema - long_ema) / 2
    
    # Calculate the standard deviation of returns over the long window as a measure of volatility
    returns = df['close'].pct_change()
    std_dev = returns.rolling(window=long_window).std()

    heuristics_matrix = trend_factor / std_dev
    return heuristics_matrix
