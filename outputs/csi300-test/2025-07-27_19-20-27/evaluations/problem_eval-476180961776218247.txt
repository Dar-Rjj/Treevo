def heuristics_v2(df):
    # Calculate the 14-day Stochastic Oscillator
    low_14, high_14 = df['low'].rolling(window=14).min(), df['high'].rolling(window=14).max()
    k = 100 * (df['close'] - low_14) / (high_14 - low_14)
    
    # Calculate the 14-day Average True Range (ATR)
    tr = df[['high', 'low']].diff().abs().max(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Combine Stochastic Oscillator and ATR to form the heuristic factor
    heuristics_matrix = k + atr
    
    return heuristics_matrix
