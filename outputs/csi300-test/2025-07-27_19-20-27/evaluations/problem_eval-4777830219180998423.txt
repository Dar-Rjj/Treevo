def heuristics_v2(df):
    # Calculate 10-day and 30-day exponential moving averages for close prices
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the Modified Money Flow Index (MFI)
    mfi = ((df['close'] - df['low']) - (df['high'] - df['close'])) * df['volume']
    mfi = mfi.cumsum()
    
    # Combine the EMA convergence and MFI
    heuristics_matrix = (ema_10 - ema_30) + mfi
    
    return heuristics_matrix
