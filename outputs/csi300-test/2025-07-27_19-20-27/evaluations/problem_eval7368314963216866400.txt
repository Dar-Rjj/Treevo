import pandas as pd
import numpy as np

def heuristics_v2(df):
    short_window = 5
    long_window = 20
    very_long_window = 60
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    
    # Calculate differences from VWAP
    diff_short = df['close'] - vwap.rolling(window=short_window).mean()
    diff_long = df['close'] - vwap.rolling(window=long_window).mean()
    diff_very_long = df['close'] - vwap.rolling(window=very_long_window).mean()
    
    # Custom weight for emphasizing recent price movements
    momentum_factor = (2 * diff_short - diff_long) / 2
    
    # Calculate RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    
    avg_gain = gain.rolling(window=long_window).mean()
    avg_loss = loss.rolling(window=long_window).mean()
    
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Adjust momentum factor based on RSI
    heuristics_matrix = momentum_factor * rsi / 100
    return heuristics_matrix
