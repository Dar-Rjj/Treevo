import pandas as pd
import numpy as np

def heuristics_v2(df):
    short_window = 5
    long_window = 30
    
    # Calculate short and long simple moving averages
    short_sma = df['close'].rolling(window=short_window).mean()
    long_sma = df['close'].rolling(window=long_window).mean()
    
    # Compute the momentum factor
    momentum_factor = short_sma - long_sma
    
    # Calculate the standard deviation of daily returns as a measure of volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=long_window).std()
    
    # Adjust the momentum factor by the volatility
    heuristics_matrix = momentum_factor / volatility
    return heuristics_matrix
