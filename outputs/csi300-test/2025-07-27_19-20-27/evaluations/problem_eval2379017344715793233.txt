def heuristics_v2(df):
    # Calculate the 14-day Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day Rate of Change (ROC)
    roc = df['close'].pct_change(periods=10) * 100
    
    # Calculate the standard deviation of daily returns over the last 30 days for weighting
    std_dev = df['close'].pct_change().rolling(window=30).std()
    
    # Combine RSI and ROC, weighted by the inverse of the standard deviation
    heuristics_matrix = ((rsi - 50) + roc) / (std_dev + 1e-6)  # Adding a small constant to avoid division by zero
    
    return heuristics_matrix
