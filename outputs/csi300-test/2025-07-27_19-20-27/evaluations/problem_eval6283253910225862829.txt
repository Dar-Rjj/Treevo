def heuristics_v2(df):
    # Calculate the 12-day and 26-day Exponential Moving Averages (EMA)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the MACD line
    macd_line = ema_12 - ema_26
    
    # Calculate the 9-day EMA of the MACD Line to get the Signal Line
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Calculate the MACD Histogram
    macd_histogram = macd_line - signal_line
    
    # Calculate the 14-day Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_flow_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_flow_ratio))
    
    # Combine MACD Histogram and MFI to form the heuristic factor
    heuristics_matrix = macd_histogram + (mfi - 50)
    
    return heuristics_matrix
