import pandas as pd

def heuristics_v2(df):
    roc_window = 12
    std_window = 20
    
    # Calculate the rate of change (ROC) for the specified window
    roc = df['close'].pct_change(periods=roc_window)
    
    # Calculate the daily returns
    daily_returns = df['close'].pct_change()
    
    # Calculate the standard deviation of daily returns over the specified window
    std_dev = daily_returns.rolling(window=std_window).std()
    
    # Adjust ROC by the standard deviation
    heuristics_matrix = roc / std_dev
    return heuristics_matrix
