import pandas as pd
import numpy as np

def heuristics_v2(df):
    short_window = 10
    long_window = 30
    very_long_window = 90
    
    short_sma = df['close'].rolling(window=short_window).mean()
    long_sma = df['close'].rolling(window=long_window).mean()
    very_long_sma = df['close'].rolling(window=very_long_window).mean()

    # Custom weight for emphasizing recent price movements
    momentum_factor = (2 * short_sma - long_sma) / 2
    
    # Calculate RSI over the very long window as a measure of market condition
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=very_long_window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=very_long_window).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    heuristics_matrix = momentum_factor * rsi
    return heuristics_matrix
