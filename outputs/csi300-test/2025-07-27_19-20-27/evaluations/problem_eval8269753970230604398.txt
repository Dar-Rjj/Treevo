import pandas as pd

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    
    # Momentum heuristic: 12-day return
    momentum = close.pct_change(12)
    
    # Volatility heuristic: Standard deviation of the last 5 days' returns
    daily_return = close.pct_change()
    volatility = daily_return.rolling(window=5).std()
    
    # Trading activity heuristic: Change in volume
    volume_change = volume.pct_change()
    
    # Composite heuristic: Equal weighting of the three factors
    heuristics_matrix = (momentum + volatility + volume_change) / 3
    
    return heuristics_matrix
