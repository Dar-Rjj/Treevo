import pandas as pd
import numpy as np

def heuristics_v2(df):
    short_window = 7
    medium_window = 14
    long_window = 28
    very_long_window = 56
    
    short_ema = df['close'].ewm(span=short_window, adjust=False).mean()
    medium_ema = df['close'].ewm(span=medium_window, adjust=False).mean()
    long_ema = df['close'].ewm(span=long_window, adjust=False).mean()
    very_long_ema = df['close'].ewm(span=very_long_window, adjust=False).mean()

    # Custom weight for emphasizing recent price movements
    trend_factor = (3 * short_ema - 2 * medium_ema + long_ema) / 2
    
    # Calculate RSI over the long window as a measure of momentum
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=long_window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=long_window).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    heuristics_matrix = trend_factor * rsi
    return heuristics_matrix
