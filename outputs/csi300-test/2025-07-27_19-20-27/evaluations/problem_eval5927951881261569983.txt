def heuristics_v2(df):
    # Calculate the 12-day and 26-day Exponential Moving Averages (EMA)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the MACD line and signal line
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Calculate the 10-day Rate of Change (ROC)
    roc = df['close'].pct_change(periods=10) * 100
    
    # Combine MACD and ROC to form the heuristic factor
    heuristics_matrix = (macd_line - signal_line) + roc
    
    return heuristics_matrix
