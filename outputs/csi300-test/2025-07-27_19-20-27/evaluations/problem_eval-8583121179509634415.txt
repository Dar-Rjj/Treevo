import pandas as pd

def heuristics_v2(df):
    short_window = 5
    long_window = 20
    very_long_window = 60
    
    # Calculate simple moving averages
    short_sma = df['close'].rolling(window=short_window).mean()
    long_sma = df['close'].rolling(window=long_window).mean()
    very_long_sma = df['close'].rolling(window=very_long_window).mean()

    # Momentum score
    momentum_score = short_sma - long_sma

    # Relative Strength
    relative_strength = (df['close'] / very_long_sma) - 1

    # Volatility-adjusted momentum score
    short_volatility = df['close'].rolling(window=short_window).std()
    long_volatility = df['close'].rolling(window=long_window).std()
    volatility_adjustment = short_volatility / long_volatility
    adjusted_momentum_score = momentum_score * volatility_adjustment

    heuristics_matrix = adjusted_momentum_score * relative_strength
    return heuristics_matrix
