import pandas as pd
import numpy as np

def heuristics_v2(df):
    short_window = 10
    mid_window = 30
    long_window = 90
    
    # Calculate the volume-weighted percentage change in closing prices
    short_change = (df['close'] * df['volume']).pct_change(periods=short_window).fillna(0)
    mid_change = (df['close'] * df['volume']).pct_change(periods=mid_window).fillna(0)
    
    # Calculate the standard deviation of the volume-weighted percentage changes as a measure of volatility
    short_volatility = (df['close'] * df['volume']).pct_change().rolling(window=short_window).std().fillna(0)
    mid_volatility = (df['close'] * df['volume']).pct_change().rolling(window=mid_window).std().fillna(0)
    long_volatility = (df['close'] * df['volume']).pct_change().rolling(window=long_window).std().fillna(0)

    # Dynamic weighting based on the ratio of short-term to long-term volatility
    weight_factor = short_volatility / (short_volatility + mid_volatility)
    momentum_factor = (weight_factor * short_change + (1 - weight_factor) * mid_change)

    # Combine momentum and volatility into a single factor
    heuristics_matrix = momentum_factor / (short_volatility + mid_volatility + long_volatility)
    return heuristics_matrix
