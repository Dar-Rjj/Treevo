```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted combination of 5-day and 20-day returns
    df['weighted_momentum'] = (3 * df['5_day_return'] + df['20_day_return']) / 4
    
    # Intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine intraday price range and intraday return
    df['intraday_factor'] = df['intraday_range'] * df['intraday_return']
    
    # Volume trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust enhanced intraday and closing price momentum by volume trends
    df['adjusted_momentum'] = df['weighted_momentum'] * (df['volume_trend'] ** 0.75)
    
    # Daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Invert rolling standard deviation to reflect stability
    df['range_stability'] = 1 / df['daily_range'].rolling(window=5).std()
    
    # Calculate True Range
    df['tr_high_low'] = df['high'] - df['low']
    df['tr_high_close'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr_low_close'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr_high_low', 'tr_high_close', 'tr_low_close']].max(axis=1)
    
    # Volume-adjusted True Range
    df['v_adj_true_range'] = df['true_range'] * (df['volume_trend'] ** 0.5)
    
    # Combine factors
    df['combined_momentum'] = df['adjusted_momentum'] + df['v_adj_true_range']
    
    # Smooth combined momentum with a 3-day SMA
    df['smoothed_momentum'] = df['combined_momentum'].rolling(window=3).mean()
    
    # Apply EMA with a 5-day span to the combined momentum
    df['final_alpha_factor'] = df['smoothed_momentum'].ewm(span=5, adjust=False).mean()
    
    return df['final_alpha_factor']
```
