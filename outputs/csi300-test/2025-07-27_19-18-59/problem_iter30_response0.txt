import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted combination of 5-day and 20-day returns
    df['combined_return'] = (3 * df['5_day_return'] + df['20_day_return']) / 4
    
    # Intraday price range and return
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and return
    df['intraday_factor'] = df['intraday_range'] * df['intraday_return']
    
    # Determine volume trend and accumulate volume
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    df['accumulated_volume'] = df['volume'].rolling(window=5).sum()
    
    # Adjust combined intraday and closing price momentum
    df['adjusted_momentum'] = df['combined_return'] * (df['volume_trend'] ** 0.75) * (df['accumulated_volume'] / 5)
    
    # Calculate volume-adjusted true range
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - x[2]), abs(x[1] - x[2])), axis=1)
    df['volume_adjusted_true_range'] = df['true_range'] * (df['volume'] / df['volume'].shift(1)) ** 0.5
    
    # Measure high-low range stability
    df['daily_range'] = df['high'] - df['low']
    df['range_stability'] = 1 / df['daily_range'].rolling(window=5).std()
    
    # Incorporate directional bias (5-day ADX)
    df['dm_pos'] = df['high'].diff().clip(lower=0)
    df['dm_neg'] = df['low'].diff().clip(upper=0)
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - x[2]), abs(x[1] - x[2])), axis=1)
    df['dm_pos_smooth'] = df['dm_pos'].ewm(span=5, min_periods=5).mean()
    df['dm_neg_smooth'] = df['dm_neg'].ewm(span=5, min_periods=5).mean()
    df['atr'] = df['tr'].ewm(span=5, min_periods=5).mean()
    df['dx'] = (abs(df['dm_pos_smooth'] - df['dm_neg_smooth']) / (df['dm_pos_smooth'] + df['dm_neg_smooth'])) * 100
    df['adx'] = df['dx'].ewm(span=5, min_periods=5).mean()
    
    # Adjust stability by directional bias
    df['adjusted_stability'] = df['range_stability'] * df['adx']
    
    # Combine factors
    df['combined_factors'] = df['adjusted_momentum'] + df['volume_adjusted_true_range']
    
    # Smooth combined momentum
    df['smoothed_combined_factors'] = df['combined_factors'].rolling(window=3).mean()
    
    # Final alpha factor with weighted sum
    weights = [0.6, 0.3, 0.1]
    df['final_alpha_factor'] = (weights[0] * df['smoothed_combined_factors'] + 
                                weights[1] * df['adjusted_stability'] + 
                                weights[2] * df['adx'])
    
    return df['final_alpha_factor'].dropna()
