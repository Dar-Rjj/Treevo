```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Determine Volume Trends
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Adjust Intraday Return by Volume Trends
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_trend_ratio']
    
    # Measure and Integrate High-Low Range Stability
    df['daily_range'] = df['high'] - df['low']
    df['range_std_30d'] = df['daily_range'].rolling(window=30).std()
    df['stability_factor'] = 1 / df['range_std_30d']
    
    # Combine Adjusted Intraday Return and Volume-Adjusted Momentum
    df['combined_momentum'] = df['adjusted_intraday_return'] + (df['intraday_return'] * df['avg_volume_5d'])
    
    # Apply a 3-day Simple Moving Average (SMA) to the combined momentum
    df['final_alpha_factor'] = df['combined_momentum'] * df['stability_factor']
    df['final_alpha_factor'] = df['final_alpha_factor'].rolling(window=3).mean()
    
    return df['final_alpha_factor']
```
