import pandas as pd

def heuristics_v2(df, N=20, M=10, momentum_weight=0.7, stability_weight=0.3):
    # Calculate Volume-Adjusted Momentum
    df['SMA'] = df['close'].rolling(window=N).mean()
    df['Volume_Adjusted_Momentum'] = (df['close'] - df['SMA']) * df['volume'].rolling(window=N).mean()

    # Measure High-Low Range Stability
    df['Daily_Range'] = df['high'] - df['low']
    df['Range_Std'] = df['Daily_Range'].rolling(window=M).std()
    df['Stability'] = 1 / (1 + df['Range_Std'])

    # Combine Volume-Adjusted Momentum and Range Stability
    df['Alpha_Factor'] = (momentum_weight * df['Volume_Adjusted_Momentum']) + (stability_weight * df['Stability'])
    
    return df['Alpha_Factor'].dropna()
