import pandas as pd

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Adjust Combined Intraday and Closing Price Momentum by Intraday Return
    df['adjusted_momentum'] = df['combined_return'] * df['intraday_return']
    
    # Volume Trends
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Sum up volumes over the past 5 days
    df['volume_sum_5d'] = df['volume'].rolling(window=5).sum()
    
    # Adjust Combined Intraday and Closing Price Momentum by Volume Trends
    df['volume_adjusted_momentum'] = df['adjusted_momentum'] * df['volume_trend']
    
    # High-Low Range Volatility
    M = 10  # Number of days for rolling variance
    df['high_low_range'] = df['high'] - df['low']
    df['range_volatility'] = df['high_low_range'].rolling(window=M).var()
    
    # Final Alpha Factor
    predefined_weight_momentum = 0.7
    predefined_weight_volatility = 0.3
    df['pre_final_alpha'] = (df['volume_adjusted_momentum'] * predefined_weight_momentum) + (df['range_volatility'] * predefined_weight_volatility)
    
    # Smooth Final Alpha Factor with 3-day SMA
    df['final_alpha'] = df['pre_final_alpha'].rolling(window=3).mean()
    
    return df['final_alpha'].dropna()
