import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_range = df['high'] - df['low']
    intraday_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = intraday_range * intraday_return
    
    # Integrate Volume Activity
    volume_trend_ratio = df['volume'] / df['volume'].shift(1)
    accumulated_volume = df['volume'].rolling(window=5).sum()
    volume_adjusted_momentum = intraday_momentum * (accumulated_volume / df['volume'])
    
    # Incorporate Smoothing for Stability
    smoothed_momentum = volume_adjusted_momentum.rolling(window=3).mean()
    
    # Measure High-Low Range Stability
    daily_range = df['high'] - df['low']
    range_stability = 1 / daily_range.rolling(window=10).std()
    
    # Combine Adjusted Momentum and Range Stability
    alpha_factor = 0.7 * smoothed_momentum + 0.3 * range_stability
    
    return alpha_factor
