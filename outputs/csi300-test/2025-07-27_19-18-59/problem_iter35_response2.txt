```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute intraday price range and return
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and intraday return
    df['intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Determine volume trends
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_trend_5d'] = df['volume'].rolling(window=5).sum()
    
    # Adjust combined momentum by volume trends
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume_trend_ratio']
    
    # Measure high-low range volatility
    M = 10  # Number of days for rolling variance
    df['high_low_volatility'] = df['high'] - df['low']
    df['high_low_volatility'] = df['high_low_volatility'].rolling(window=M).var()
    
    # Calculate volume-adjusted momentum
    N = 20  # Number of days for SMA
    df['sma_close'] = df['close'].rolling(window=N).mean()
    df['volume_adjusted_sma'] = (df['close'] - df['sma_close']) * df['volume'].rolling(window=N).mean()
    
    # Combine factors
    df['combined_momentum'] = df['volume_adjusted_momentum'] + df['volume_adjusted_sma']
    
    # Smooth combined momentum
    df['smoothed_combined_momentum'] = df['combined_momentum'].rolling(window=3).mean()
    
    # Incorporate weighted sum with predefined weights
    weight_momentum = 0.7  # Weight for smoothed combined momentum
    weight_volatility = 0.3  # Weight for high-low range volatility
    df['final_alpha_factor'] = (weight_momentum * df['smoothed_combined_momentum'] +
                                weight_volatility * df['high_low_volatility'])
    
    return df['final_alpha_factor']
```
