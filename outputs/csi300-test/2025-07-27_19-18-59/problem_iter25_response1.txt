import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day Returns
    df['5d_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20d_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted Combination of 5-day and 20-day Returns
    df['weighted_returns'] = (3 * df['5d_return'] + df['20d_return']) / 4
    
    # Combine Intraday Price Range and Intraday Return
    df['intraday_momentum'] = (df['high'] - df['low']) * ((df['close'] - df['open']) / df['open'])
    
    # Determine Volume Trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Adjusted Momentum: Combined Intraday and Closing Price Momentum * (Volume Trend ** 0.75)
    df['adjusted_momentum'] = df['intraday_momentum'] * (df['volume_trend'] ** 0.75)
    
    # Calculate True Range
    df['true_range'] = df[['high' - 'low', 
                           abs(df['high'] - df['close'].shift(1)), 
                           abs(df['low'] - df['close'].shift(1))]].max(axis=1)
    
    # Volume-Adjusted True Range
    df['vol_adjusted_tr'] = df['true_range'] * (df['volume_trend'] ** 0.5)
    
    # Measure and Integrate High-Low Range Stability
    M = 10  # Number of days for rolling standard deviation
    df['daily_range'] = df['high'] - df['low']
    df['range_stability'] = 1 / df['daily_range'].rolling(window=M).std()
    
    # Combine Adjusted Momentum and Volume-Adjusted True Range
    df['combined_momentum'] = df['adjusted_momentum'] + df['vol_adjusted_tr']
    
    # Apply Exponential Moving Average (EMA) with a 5-day span to the combined momentum for final alpha factor
    df['final_alpha_factor'] = df['combined_momentum'].ewm(span=5, adjust=False).mean()
    
    return df['final_alpha_factor']

# Example usage:
