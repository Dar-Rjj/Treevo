```python
import pandas as pd

def heuristics_v2(df, M=10):
    # Compute 5-day and 20-day Returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Average 5-day and 20-day Returns
    df['avg_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Combine with Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) * ((df['close'] - df['open']) / df['open'])
    df['combined_momentum'] = df['avg_return'] + df['intraday_return']
    
    # Determine Volume Trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust Combined Momentum
    df['adjusted_momentum'] = df['combined_momentum'] * df['volume_trend']
    
    # Calculate True Range
    df['true_range'] = df.apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift(1)), abs(x['low'] - x['close'].shift(1))), axis=1)
    
    # Measure and Integrate High-Low Range Stability
    df['daily_range'] = df['high'] - df['low']
    df['range_stability'] = df['daily_range'].rolling(window=M).std()
    
    # Combine Adjusted Momentum and True Range
    df['combined_factor'] = df['adjusted_momentum'] + df['true_range']
    
    # Apply Weighted Moving Average to the combined factor for final alpha factor
    df['alpha_factor'] = (df['adjusted_momentum'] * 0.5 + 
                          df['true_range'] * 0.3 + 
                          df['range_stability'] * 0.2)
    
    return df['alpha_factor'].dropna()
```
