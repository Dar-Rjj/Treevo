import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine intraday price range and intraday return
    df['intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Determine volume trends
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_trend_5d'] = df['volume'].rolling(window=5).sum()
    
    # Adjust combined return by volume trend
    df['adjusted_combined_momentum'] = df['combined_return'] * df['volume_trend_ratio']
    
    # Compute rolling variance of high-low range over M days
    M = 20
    df['high_low_range_volatility'] = df['high'] - df['low']
    df['high_low_range_volatility'] = df['high_low_range_volatility'].rolling(window=M).var()
    
    # Combine factors
    df['combined_momentum'] = df['adjusted_combined_momentum'] + df['intraday_momentum']
    
    # Apply 5-day Exponential Moving Average (EMA) to smooth the combined momentum
    df['smoothed_combined_momentum'] = df['combined_momentum'].ewm(span=5, adjust=False).mean()
    
    # Incorporate weighted sum with predefined weights
    weight_momentum = 0.7
    weight_volatility = 0.3
    df['final_alpha_factor'] = (weight_momentum * df['smoothed_combined_momentum']) + (weight_volatility * df['high_low_range_volatility'])
    
    # Return the final alpha factor
    return df['final_alpha_factor'].dropna()
