import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day Returns
    df['combined_close_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Price Range and Intraday Return
    df['intraday_range_return'] = (df['high'] - df['low']) * df['intraday_return']
    
    # Determine Volume Trends
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Accumulate Volume Over Multiple Days
    df['volume_5d_sum'] = df['volume'].rolling(window=5).sum()
    
    # Adjust Combined Intraday and Closing Price Momentum by Volume Trends
    df['adjusted_momentum'] = df['combined_close_return'] * df['volume_trend']
    
    # Compute Rolling Standard Deviation of High-Low Range over M days
    M = 10  # Number of days for rolling standard deviation
    df['high_low_std'] = df['high'] - df['low']
    df['high_low_std'] = df['high_low_std'].rolling(window=M).std()
    
    # Compute Exponential Moving Average (EMA) of Close Prices over N days
    N = 10  # Number of days for EMA
    df['ema_close'] = df['close'].ewm(span=N, adjust=False).mean()
    
    # Subtract EMA from Current Close Price
    df['price_diff_from_ema'] = df['close'] - df['ema_close']
    
    # Multiply by Average Volume over N days
    df['avg_volume_n_days'] = df['volume'].rolling(window=N).mean()
    df['volume_adjusted_price_diff'] = df['price_diff_from_ema'] * df['avg_volume_n_days']
    
    # Add the two adjusted momentums
    df['combined_momentum'] = df['adjusted_momentum'] + df['volume_adjusted_price_diff']
    
    # Apply a 5-day Exponential Moving Average (EMA) to the combined momentum
    df['final_alpha_factor'] = df['combined_momentum'].ewm(span=5, adjust=False).mean()
    
    # Drop NaN values
    df.dropna(inplace=True)
    
    # Return the final alpha factor
    return df['final_alpha_factor']
