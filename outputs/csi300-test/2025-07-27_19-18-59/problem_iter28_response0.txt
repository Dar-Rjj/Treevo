```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['weighted_momentum'] = (3 * df['5_day_return'] + df['20_day_return']) / 4
    
    # Integrate Volume Activity
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    df['adjusted_momentum'] = df['weighted_momentum'] * (df['volume_trend'] ** 0.75)
    
    # Calculate Volume-Adjusted True Range
    df['high_low_range'] = df['high'] - df['low']
    df['high_prev_close'] = (df['high'] - df['close'].shift(1)).abs()
    df['low_prev_close'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['high_low_range', 'high_prev_close', 'low_prev_close']].max(axis=1)
    df['volume_adjusted_true_range'] = df['true_range'] * (df['volume_trend'] ** 0.5)
    
    # Measure and Integrate High-Low Range Volatility
    M = 20
    df['range_volatility'] = df['high_low_range'].rolling(window=M).std()
    df['range_stability'] = 1 / (df['range_volatility'] + 1e-6)  # Add a small constant to avoid division by zero
    
    # Combine all components
    df['final_factor'] = df['adjusted_momentum'] * df['range_stability'] / df['volume_adjusted_true_range']
    
    # Smooth Final Alpha Factor
    df['smoothed_final_factor'] = df['final_factor'].ewm(span=5, adjust=False).mean()
    
    return df['smoothed_final_factor']
```
