```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Price Range and Intraday Return
    df['intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Determine Volume Trends
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_trend_5day'] = df['volume'].rolling(window=5).sum()
    
    # Adjust Intraday Momentum by Volume Trends
    df['adjusted_intraday_momentum'] = df['intraday_momentum'] * df['volume_trend_ratio']
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low', 'close']].join(df['close'].shift(1)).apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close.shift.1']), abs(x['low'] - x['close.shift.1'])), axis=1
    )
    
    # Average the True Ranges
    df['avg_true_range'] = df['true_range'].rolling(window=5).mean()
    
    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=5).mean()
    df['volume_adjusted_true_range'] = df['avg_true_range'] * (df['volume'] / avg_volume)
    
    # Measure and Integrate High-Low Range Stability
    df['daily_range'] = df['high'] - df['low']
    df['range_stability'] = df['daily_range'].rolling(window=5).std()
    
    # Combine Adjusted Intraday Momentum and Volume-Adjusted True Range
    df['combined_momentum'] = df['adjusted_intraday_momentum'] + df['volume_adjusted_true_range']
    
    # Apply a 7-day Exponential Moving Average (EMA) to the combined momentum
    df['ema_combined_momentum'] = df['combined_momentum'].ewm(span=7, adjust=False).mean()
    
    # Introduce a Weighted Momentum Component
    df['weight'] = df['volume_trend_ratio'].apply(lambda x: 1 if x >= 1 else 0.5)
    df['weighted_ema_momentum'] = df['ema_combined_momentum'] * df['weight']
    
    # Normalize the Weighted Sum
    final_alpha_factor = (df['weighted_ema_momentum'] - df['weighted_ema_momentum'].mean()) / df['weighted_ema_momentum'].std()
    
    return final_alpha_factor
```
