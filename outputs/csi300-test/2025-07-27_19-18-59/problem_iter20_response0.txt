```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Synthesized Intraday and Closing Price Momentum
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['weighted_returns'] = (3 * df['5_day_return'] + df['20_day_return']) / 4
    df['intraday_price_range'] = df['high'] - df['low']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['combined_momentum'] = df['intraday_price_range'] * df['intraday_return']

    # Integrate Volume Activity
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    df['accumulated_volume'] = df['volume'].rolling(window=5).sum()
    df['adjusted_combined_momentum'] = df['combined_momentum'] * (df['volume_trend'] ** 0.75)

    # Calculate Volume-Weighted True Range
    df['true_range'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    df['volume_weighted_true_range'] = df['true_range'] * (df['volume'] / df['volume'].shift(1)) ** 0.5

    # Measure and Integrate High-Low Range Volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=20).std()
    df['range_stability'] = 1 / (1 + df['range_volatility'])

    # Calculate Final Alpha Factor
    df['smoothed_combined_momentum'] = df['adjusted_combined_momentum'].rolling(window=3).mean()
    df['ema_combined_momentum'] = df['adjusted_combined_momentum'].ewm(span=5, adjust=False).mean()
    weight_smoothed = 0.6
    weight_stability = 0.4
    df['final_alpha_factor'] = (weight_smoothed * df['smoothed_combined_momentum'] +
                                weight_stability * df['range_stability'])

    return df['final_alpha_factor']
```
