import pandas as pd

def heuristics_v2(df, M=20, weight1=0.7, weight2=0.3):
    # Calculate Combined Intraday and Closing Price Momentum
    df['5d_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20d_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['combined_momentum'] = (df['5d_return'] + df['20d_return']) / 2
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_price_range'] = (df['high'] - df['low']) * df['intraday_return']
    
    # Integrate Volume Activity
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    df['volume_sum_5d'] = df['volume'].rolling(window=5).sum()
    df['volume_adjusted_momentum'] = df['combined_momentum'] * df['volume_trend']
    
    # Measure and Integrate High-Low Range Volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=M).var()
    
    # Combine Factors
    df['total_momentum'] = df['volume_adjusted_momentum'] + df['intraday_price_range']
    df['smoothed_momentum'] = df['total_momentum'].rolling(window=3).mean()
    
    # Incorporate Weighted Sum with Predefined Weights
    df['weighted_sum'] = (df['smoothed_momentum'] * weight1) + (df['range_volatility'] * weight2)
    
    return df['weighted_sum']
