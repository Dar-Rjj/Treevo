```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = df['close'].pct_change(periods=5)
    df['20_day_return'] = df['close'].pct_change(periods=20)
    
    # Combine 5-day and 20-day returns
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine intraday price range and intraday return
    df['intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Determine volume trends
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Accumulate volume over multiple days
    df['accumulated_volume'] = df['volume'].rolling(window=5).sum()
    
    # Adjust combined return by volume trend
    df['adjusted_momentum'] = df['combined_return'] * df['volume_trend']
    
    # Apply a 3-day Simple Moving Average (SMA) to the adjusted momentum
    df['smoothed_momentum'] = df['adjusted_momentum'].rolling(window=3).mean()
    
    # Calculate the rolling standard deviation of daily range over M days
    df['daily_range'] = df['high'] - df['low']
    df['range_stability'] = df['daily_range'].rolling(window=20).std().apply(lambda x: 1 / (1 + x))
    
    # Compute weighted sum of the adjusted momentum and range stability
    df['final_alpha_factor'] = (df['smoothed_momentum'] * 0.7) + (df['range_stability'] * 0.3)
    
    return df['final_alpha_factor'].dropna()
```
