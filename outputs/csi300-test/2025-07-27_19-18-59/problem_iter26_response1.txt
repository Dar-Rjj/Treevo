import pandas as pd

def heuristics_v2(df, N=5, M=20):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and intraday return
    df['intraday_combined'] = df['intraday_range'] * df['intraday_return']
    
    # Integrate volume activity
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_5_days_sum'] = df['volume'].rolling(window=5).sum()
    
    # Adjust combined intraday and closing price momentum by volume trends
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume_trend_ratio']
    
    # Measure and integrate high-low range volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=M).var()
    
    # Calculate volume-adjusted momentum
    df['sma_close_N'] = df['close'].rolling(window=N).mean()
    df['sma_volume_N'] = df['volume'].rolling(window=N).mean()
    df['volume_adjusted_price_momentum'] = (df['close'] - df['sma_close_N']) * df['sma_volume_N']
    
    # Combine factors
    df['combined_momentum'] = df['volume_adjusted_momentum'] + df['volume_adjusted_price_momentum']
    
    # Smooth combined momentum
    df['smoothed_combined_momentum'] = df['combined_momentum'].rolling(window=3).mean()
    
    # Incorporate weighted sum with predefined weights
    weight_momentum = 0.7
    weight_volatility = 0.3
    df['final_alpha_factor'] = (weight_momentum * df['smoothed_combined_momentum'] + 
                                weight_volatility * df['range_volatility'])
    
    return df['final_alpha_factor'].dropna()
