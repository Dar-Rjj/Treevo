```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['price_momentum'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute 5-day and 20-day volume trends
    df['5_day_vol_trend'] = df['volume'] - df['volume'].shift(5)
    df['20_day_vol_trend'] = df['volume'] - df['volume'].shift(20)
    
    # Combine 5-day and 20-day volume trends
    df['volume_momentum'] = (df['5_day_vol_trend'] + df['20_day_vol_trend']) / 2
    
    # Weighted average of price and volume momentum
    df['weighted_avg_momentum'] = (df['price_momentum'] * 0.7) + (df['volume_momentum'] * 0.3)
    
    # Compute Volume-Adjusted Momentum
    N = 20
    df['ema_close'] = df['close'].ewm(span=N, adjust=False).mean()
    df['volume_adjusted_momentum'] = (df['close'] - df['ema_close']) * df['volume'].rolling(window=N).mean()
    
    # Measure High-Low Range Stability
    M = 10
    df['daily_range'] = df['high'] - df['low']
    df['rolling_median_range'] = df['daily_range'].rolling(window=M).median()
    df['high_low_stability'] = 1 / (df['rolling_median_range'] + 1e-6)  # Add a small constant to avoid division by zero
    
    # Combine all factors
    df['alpha_factor'] = (df['weighted_avg_momentum'] * 0.6) + \
                         (df['volume_adjusted_momentum'] * 0.2) + \
                         (df['high_low_stability'] * 0.2)
    
    return df['alpha_factor'].dropna()
```
