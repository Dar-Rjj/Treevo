```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['Intraday_Price_Range'] = df['High'] - df['Low']
    df['Intraday_Return'] = (df['Close'] - df['Open']) / df['Open']
    df['Intraday_Momentum'] = df['Intraday_Price_Range'] * df['Intraday_Return']

    # Integrate Volume Activity
    df['Volume_Trend'] = df['Volume'] / df['Volume'].shift(1)
    df['Volume_Sum_5D'] = df['Volume'].rolling(window=5).sum()

    # Calculate Price Momentum
    df['5D_Return'] = (df['Close'] / df['Close'].shift(5)) - 1
    df['20D_Return'] = (df['Close'] / df['Close'].shift(20)) - 1
    df['Combined_Price_Momentum'] = (df['5D_Return'] + df['20D_Return']) / 2

    # Calculate Volume Momentum
    df['5D_Volume_Trend'] = df['Volume'] - df['Volume'].shift(5)
    df['20D_Volume_Trend'] = df['Volume'] - df['Volume'].shift(20)
    df['Combined_Volume_Momentum'] = (df['5D_Volume_Trend'] + df['20D_Volume_Trend']) / 2

    # Adjust Intraday Momentum by Volume Trends
    df['Adjusted_Intraday_Momentum'] = df['Intraday_Momentum'] * (df['Volume_Trend'] / df['Volume_Trend'].mean())

    # Merge Intraday and Multi-Day Momentum
    df['Merged_Momentum'] = (df['Adjusted_Intraday_Momentum'] + df['Combined_Price_Momentum']) / 2

    # Weighted Average of Merged Momentum and Volume Momentum
    df['Alpha_Factor'] = (df['Merged_Momentum'] * 0.6) + (df['Combined_Volume_Momentum'] * 0.4)

    return df['Alpha_Factor']
```
