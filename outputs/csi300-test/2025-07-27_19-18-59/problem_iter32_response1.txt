```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-day and 20-day returns
    df['5_day_return'] = df['close'].pct_change(5)
    df['20_day_return'] = df['close'].pct_change(20)
    
    # Weighted combination of 5-day and 20-day returns
    df['weighted_returns'] = (3 * df['5_day_return'] + df['20_day_return']) / 4
    
    # Compute intraday price range and return
    df['intraday_price_range'] = df['high'] - df['low']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_momentum'] = df['intraday_price_range'] * df['intraday_return']
    
    # Combine intraday and closing price momentum
    df['combined_momentum'] = df['weighted_returns'] + df['intraday_momentum']
    
    # Determine volume trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust combined momentum by volume trend
    df['volume_adjusted_momentum'] = df['combined_momentum'] * (df['volume_trend'] ** 0.75)
    
    # Sum up volumes over the past 5 days
    df['5_day_volume_sum'] = df['volume'].rolling(window=5).sum()
    
    # Calculate true range
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x[0] - x[1], abs(x[0] - df['close'].shift(1)), abs(x[1] - df['close'].shift(1))), axis=1)
    
    # Volume-adjusted true range
    df['volume_adjusted_true_range'] = df['true_range'] * (df['volume_trend'] ** 0.5)
    
    # Measure high-low range volatility
    df['daily_range'] = df['high'] - df['low']
    M = 20
    df['range_volatility'] = df['daily_range'].rolling(window=M).std().fillna(0)
    df['range_stability'] = 1 / (df['range_volatility'] + 1e-6)  # Add a small constant to avoid division by zero
    
    # Combine factors
    df['combined_factors'] = df['volume_adjusted_momentum'] + df['volume_adjusted_true_range'] + df['range_stability']
    
    # Smooth combined factors with 5-day EMA
    df['smoothed_combined_factors'] = df['combined_factors'].ewm(span=5, adjust=False).mean()
    
    # Incorporate weighted sum with predefined weights
    weight_momentum = 0.7
    weight_volatility = 0.3
    df['final_alpha_factor'] = (weight_momentum * df['smoothed_combined_factors'] +
                                weight_volatility * df['range_stability'])
    
    return df['final_alpha_factor']
```
