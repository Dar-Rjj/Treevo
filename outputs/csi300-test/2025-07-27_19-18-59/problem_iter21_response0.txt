```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday and Closing Price Momentum
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['weighted_return'] = (4 * df['5_day_return'] + df['20_day_return']) / 5
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['enhanced_momentum'] = (df['high'] - df['low']) * df['intraday_return']
    
    # Integrate Volume Activity
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    df['adjusted_momentum'] = df['enhanced_momentum'] * (df['volume_trend'] ** 0.8)
    
    # Calculate Volume-Adjusted True Range
    df['true_range'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x[0] - x[1], abs(x[0] - x[2].shift(1)), abs(x[1] - x[2].shift(1))), axis=1
    )
    df['volume_adjusted_true_range'] = df['true_range'] * (df['volume_trend'] ** 0.6)
    
    # Measure and Integrate High-Low Range Volatility
    M = 10  # number of days for rolling standard deviation
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=M).std()
    df['combined_momentum'] = df['adjusted_momentum'] + df['volume_adjusted_true_range']
    
    # Invert Rolling Standard Deviation to Reflect Stability
    df['stability_factor'] = 1 / (df['range_volatility'] + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Apply Exponential Moving Average (EMA) with a 7-day span to the combined momentum
    df['alpha_factor'] = df['combined_momentum'] * df['stability_factor']
    df['alpha_factor'] = df['alpha_factor'].ewm(span=7, adjust=False).mean()
    
    return df['alpha_factor']
```
