import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and intraday return
    df['intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Volume trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Accumulate volume over the past 5 days
    df['volume_5_days'] = df['volume'].rolling(window=5).sum()
    
    # Adjust combined intraday and closing price momentum by volume trends
    df['adjusted_momentum'] = df['combined_return'] * df['volume_trend']
    
    # Measure high-low range volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=5).std()
    
    # Calculate volume-adjusted true range
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - x[2]), abs(x[1] - x[2])), axis=1)
    df['volume_adjusted_true_range'] = df['true_range'] * (df['volume'] / df['volume'].shift(1)) ** 0.5
    
    # Combine factors
    df['combined_momentum'] = df['adjusted_momentum'] + df['intraday_momentum']
    
    # Smooth combined momentum with a 5-day EMA
    df['alpha_factor'] = df['combined_momentum'].ewm(span=5, adjust=False).mean()
    
    return df['alpha_factor']
