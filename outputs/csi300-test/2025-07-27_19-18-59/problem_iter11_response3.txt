import pandas as pd

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = df['close'].pct_change(5)
    df['20_day_return'] = df['close'].pct_change(20)
    
    # Combine 5-day and 20-day returns
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine intraday price range and intraday return
    df['intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Integrate volume activity
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_trend'] = df['volume'].rolling(window=5).sum() / df['volume'].shift(1).rolling(window=5).sum()
    
    # Adjust combined momentum by volume trends
    df['adjusted_momentum'] = df['combined_return'] * df['volume_trend_ratio']
    
    # Measure and integrate high-low range volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=5).var()
    
    # Combine adjusted intraday and volume-adjusted momentum
    df['combined_momentum'] = df['adjusted_momentum'] + df['intraday_momentum']
    
    # Smooth combined momentum with a 3-day SMA
    df['smoothed_momentum'] = df['combined_momentum'].rolling(window=3).mean()
    
    # Incorporate weighted sum with predefined weights
    weight_momentum = 0.7
    weight_volatility = 0.3
    df['final_alpha_factor'] = (weight_momentum * df['smoothed_momentum']) + (weight_volatility * df['range_volatility'])
    
    return df['final_alpha_factor'].dropna()
