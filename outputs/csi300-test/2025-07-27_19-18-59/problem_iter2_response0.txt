import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Price Range and Intraday Return
    intraday_momentum = intraday_range * intraday_return
    
    # Determine Volume Trends
    volume_trend = df['volume'] / df['volume'].shift(1)
    volume_rolling_sum = df['volume'].rolling(window=5).sum()
    
    # Adjust Intraday Momentum by Volume Trends
    volume_adjusted_momentum = intraday_momentum * volume_trend
    
    # Compute Simple Moving Average (SMA) of Close Prices over N days
    close_sma_3 = df['close'].rolling(window=3).mean()
    
    # Subtract the SMA of Close Prices from Current Close Price
    close_diff = df['close'] - close_sma_3
    
    # Multiply by Average Volume over N days
    avg_volume_3 = df['volume'].rolling(window=3).mean()
    volume_adjusted_close_diff = close_diff * avg_volume_3
    
    # Measure and Integrate High-Low Range Stability
    daily_range = df['high'] - df['low']
    range_stability = daily_range.rolling(window=3).std().transform(lambda x: 1 / x)
    
    # Combine Adjusted Intraday Momentum and Volume-Adjusted Momentum
    combined_momentum = volume_adjusted_momentum + volume_adjusted_close_diff
    
    # Apply a 3-day Simple Moving Average (SMA) to the combined momentum
    final_alpha_factor = combined_momentum * range_stability
    final_alpha_factor = final_alpha_factor.rolling(window=3).mean()
    
    return final_alpha_factor
