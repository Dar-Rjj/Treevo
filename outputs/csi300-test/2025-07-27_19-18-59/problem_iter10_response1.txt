```python
import pandas as pd

def heuristics_v2(df, N=20, M=10, momentum_weight=0.7, stability_weight=0.3):
    # Calculate Volume-Adjusted Momentum
    sma_close = df['close'].rolling(window=N).mean()
    raw_momentum = df['close'] - sma_close
    avg_volume = df['volume'].rolling(window=N).mean()
    volume_adjusted_momentum = raw_momentum * avg_volume
    
    # Measure High-Low Range Stability
    daily_range = df['high'] - df['low']
    range_stability = (daily_range.rolling(window=M).std()).transform(lambda x: 1 / x)
    
    # Combine Factors
    factor_values = (volume_adjusted_momentum * momentum_weight) + (range_stability * stability_weight)
    
    return factor_values
```
