```python
import pandas as pd
import numpy as np

def heuristics_v2(df, M=30):
    # Compute 5-day and 20-day Returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day Returns
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Price Range and Intraday Return
    df['intraday_momentum'] = (df['high'] - df['low']) * df['intraday_return']
    
    # Integrate Volume Activity
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    df['adjusted_momentum'] = (df['combined_return'] + df['intraday_momentum']) * df['volume_trend']
    
    # Calculate True Range
    df['true_range'] = df[['high' - 'low', abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1))]].max(axis=1)
    
    # Measure High-Low Range Stability
    df['daily_range'] = df['high'] - df['low']
    df['range_stability'] = 1 / df['daily_range'].rolling(window=M).std()
    
    # Combine Adjusted Momentum and True Range
    df['combined_factor'] = df['adjusted_momentum'] + df['true_range']
    
    # Apply 3-day SMA to the combined factor for final alpha factor
    df['alpha_factor'] = df['combined_factor'] * df['range_stability']
    df['final_alpha_factor'] = df['alpha_factor'].rolling(window=3).mean()
    
    return df['final_alpha_factor']
```
