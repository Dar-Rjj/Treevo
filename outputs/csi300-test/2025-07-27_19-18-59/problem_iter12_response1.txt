import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['Intraday_Price_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    df['Intraday_Momentum'] = df['Intraday_Price_Range'] * df['Intraday_Return']
    
    # Determine Volume Trends
    df['Volume_Trend'] = df['volume'] / df['volume'].shift(1)
    df['Accumulated_Volume'] = df['volume'].rolling(window=5).sum()
    df['Volume_Trend_Ratio'] = df['Volume_Trend'] / (df['Accumulated_Volume'] / 5)
    
    # Adjust Intraday Momentum by Volume Trends
    df['Adjusted_Intraday_Momentum'] = df['Intraday_Momentum'] * df['Volume_Trend_Ratio']
    
    # Calculate True Range
    df['True_Range'] = pd.concat([
        (df['high'] - df['low']),
        abs(df['high'] - df['close'].shift(1)),
        abs(df['low'] - df['close'].shift(1))
    ], axis=1).max(axis=1)
    
    # Average the True Ranges and Adjust by Volume
    df['Average_True_Range'] = df['True_Range'].rolling(window=5).mean()
    avg_volume = df['volume'].rolling(window=5).mean()
    df['Volume_Adjusted_True_Range'] = df['Average_True_Range'] * (df['volume'] / avg_volume)
    
    # Measure and Integrate High-Low Range Stability
    df['Daily_Range'] = df['high'] - df['low']
    df['Range_Stability'] = df['Daily_Range'].rolling(window=7).std().apply(lambda x: 1/x if x != 0 else 0)
    
    # Combine Adjusted Intraday Momentum and Volume-Adjusted True Range
    df['Combined_Momentum'] = df['Adjusted_Intraday_Momentum'] + df['Volume_Adjusted_True_Range']
    
    # Apply a 7-day Exponential Moving Average (EMA) to the combined momentum
    df['EMA_Combined_Momentum'] = df['Combined_Momentum'].ewm(span=7, adjust=False).mean()
    
    # Introduce a Weighted Momentum Component
    df['Weighted_Momentum'] = df['EMA_Combined_Momentum'] * df['Volume_Trend']
    df['Normalized_Weighted_Momentum'] = df['Weighted_Momentum'] / df['Weighted_Momentum'].rolling(window=7).mean()
    
    # Final Alpha Factor
    alpha_factor = df['Normalized_Weighted_Momentum'] * df['Range_Stability']
    
    return alpha_factor.dropna()
