import pandas as pd

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Price Range and Intraday Return
    df['intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Determine Volume Trends
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['accumulated_volume'] = df['volume'].rolling(window=5).sum()
    
    # Adjust Combined Intraday and Closing Price Momentum by Volume Trends
    df['adjusted_momentum'] = df['combined_return'] * df['volume_trend_ratio']
    
    # Calculate True Range
    df['true_range_1'] = df['high'] - df['low']
    df['true_range_2'] = (df['high'] - df['close'].shift(1)).abs()
    df['true_range_3'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['true_range_1', 'true_range_2', 'true_range_3']].max(axis=1)
    
    # Average the True Ranges
    df['average_true_range'] = df['true_range'].rolling(window=5).mean()
    
    # Measure and Integrate High-Low Range Stability
    df['daily_range'] = df['high'] - df['low']
    df['range_stability'] = df['daily_range'].rolling(window=5).std().replace(0, 1)  # Avoid division by zero
    df['stability_factor'] = 1 / df['range_stability']
    
    # Combine Adjusted Combined Intraday and Closing Price Momentum and Volume-Adjusted True Range
    df['combined_momentum'] = df['adjusted_momentum'] + df['average_true_range']
    
    # Apply a 3-day Simple Moving Average (SMA) to the combined momentum
    df['final_alpha_factor'] = df['combined_momentum'].rolling(window=3).mean() * df['stability_factor']
    
    return df['final_alpha_factor'].dropna()
