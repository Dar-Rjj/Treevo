```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted combination of 5-day and 20-day returns
    df['weighted_returns'] = (3 * df['5_day_return'] + df['20_day_return']) / 4
    
    # Intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine intraday price range and intraday return
    df['enhanced_intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Determine volume trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Accumulate volume over multiple days
    df['accumulated_volume'] = df['volume'].rolling(window=5).sum()
    
    # Adjust enhanced intraday and closing price momentum by volume trends
    df['adjusted_momentum'] = df['weighted_returns'] * (df['volume_trend'] ** 0.75)
    
    # Measure high-low range volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=5).std()
    df['range_stability'] = 1 / df['range_volatility'].replace([np.inf, -np.inf], np.nan).fillna(0)
    
    # Calculate volume-adjusted true range
    df['true_range'] = df[['high', 'low', 'close']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    df['volume_adjusted_true_range'] = df['true_range'] * (df['volume_trend'] ** 0.5)
    
    # Combine factors
    df['combined_momentum'] = df['adjusted_momentum'] + df['volume_adjusted_true_range']
    
    # Smooth combined momentum with a 3-day SMA
    df['smoothed_combined_momentum'] = df['combined_momentum'].rolling(window=3).mean()
    
    # Apply EMA with a 5-day span to the combined momentum
    df['ema_combined_momentum'] = df['combined_momentum'].ewm(span=5, adjust=False).mean()
    
    # Calculate weighted sum
    df['final_alpha_factor'] = 0.5 * df['smoothed_combined_momentum'] + 0.3 * df['range_stability'] + 0.2 * df['ema_combined_momentum']
    
    return df['final_alpha_factor'].dropna()
```
