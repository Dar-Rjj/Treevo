import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted combination of 5-day and 20-day returns
    df['weighted_return'] = (3 * df['5_day_return'] + df['20_day_return']) / 4
    
    # Compute intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and intraday return
    df['intraday_range'] = df['high'] - df['low']
    df['combined_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Determine volume trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust combined momentum by volume trends
    df['adjusted_combined_momentum'] = df['combined_momentum'] * (df['volume_trend'] ** 0.75)
    
    # Measure and integrate high-low range volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = 1 / df['daily_range'].rolling(window=20).std()
    
    # Calculate volume-adjusted true range
    df['true_range'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift(1)), abs(x['low'] - x['close'].shift(1))),
        axis=1
    )
    df['volume_adjusted_true_range'] = df['true_range'] * (df['volume'] / df['volume'].shift(1)) ** 0.5
    
    # Combine factors
    df['combined_factors'] = df['adjusted_combined_momentum'] + df['volume_adjusted_true_range']
    
    # Smooth combined momentum with a 5-day SMA
    df['sma_5'] = df['combined_factors'].rolling(window=5).mean()
    
    # Apply EMA with a 10-day span to the combined momentum
    df['final_alpha_factor'] = df['sma_5'].ewm(span=10, adjust=False).mean()
    
    return df['final_alpha_factor'].dropna()
