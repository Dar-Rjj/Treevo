```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['price_momentum'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute 5-day and 20-day volume trends
    df['5_day_volume_trend'] = df['volume'] - df['volume'].shift(5)
    df['20_day_volume_trend'] = df['volume'] - df['volume'].shift(20)
    
    # Combine 5-day and 20-day volume trends
    df['volume_momentum'] = (df['5_day_volume_trend'] + df['20_day_volume_trend']) / 2
    
    # Compute 5-day and 20-day price volatilities
    df['5_day_price_volatility'] = df['close'].rolling(window=5).std()
    df['20_day_price_volatility'] = df['close'].rolling(window=20).std()
    
    # Combine 5-day and 20-day price volatilities
    df['price_volatility'] = (df['5_day_price_volatility'] + df['20_day_price_volatility']) / 2
    
    # Compute 5-day and 20-day volume volatilities
    df['5_day_volume_volatility'] = df['volume'].rolling(window=5).std()
    df['20_day_volume_volatility'] = df['volume'].rolling(window=20).std()
    
    # Combine 5-day and 20-day volume volatilities
    df['volume_volatility'] = (df['5_day_volume_volatility'] + df['20_day_volume_volatility']) / 2
    
    # Determine weights
    df['price_weight'] = 1 / (1 + (df['price_volatility'] / (df['price_volatility'] + df['volume_volatility'])))
    df['volume_weight'] = 1 / (1 + (df['volume_volatility'] / (df['price_volatility'] + df['volume_volatility'])))
    
    # Multiply and sum to get the final factor
    df['alpha_factor'] = (df['price_momentum'] * df['price_weight']) + (df['volume_momentum'] * df['volume_weight'])
    
    return df['alpha_factor']
```
