import pandas as pd
import numpy as np

def heuristics_v2(df, M=20):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['combined_returns'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and intraday return
    df['intraday_momentum'] = (df['high'] - df['low']) * df['intraday_return']
    
    # Integrate volume activity
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_adjusted_momentum'] = df['combined_returns'] * df['volume_trend_ratio']
    
    # Measure high-low range volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=M).var()
    
    # Smooth combined momentum with a 3-day SMA
    df['smoothed_momentum'] = df['volume_adjusted_momentum'].rolling(window=3).mean()
    
    # Final alpha factor: weighted sum of smoothed momentum and high-low range volatility
    weight_momentum = 0.7
    weight_volatility = 0.3
    df['alpha_factor'] = (weight_momentum * df['smoothed_momentum']) + (weight_volatility * df['range_volatility'])
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
