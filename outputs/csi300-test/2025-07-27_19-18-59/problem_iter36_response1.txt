```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['combined_momentum'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine intraday price range and intraday return
    df['intraday_volatility'] = df['intraday_range'] * df['intraday_return']
    
    # Determine volume trends
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    df['accumulated_volume'] = df['volume'].rolling(window=5).sum()
    
    # Adjust combined momentum by volume trend
    df['adjusted_momentum'] = df['combined_momentum'] * df['volume_trend']
    
    # Compute high-low range volatility
    df['range_volatility'] = df['high'] - df['low']
    df['rolling_std'] = df['range_volatility'].rolling(window=30).std()
    
    # Combine adjusted momentum and high-low range volatility
    df['combined_factor'] = df['adjusted_momentum'] + df['intraday_volatility']
    
    # Smooth combined factor with 7-day EMA
    df['smoothed_momentum'] = df['combined_factor'].ewm(span=7).mean()
    
    # Define weights for the weighted sum
    weight_momentum = 0.6
    weight_volatility = 0.4
    
    # Compute weighted sum
    df['final_alpha_factor'] = (weight_momentum * df['smoothed_momentum']) + (weight_volatility * df['rolling_std'])
    
    return df['final_alpha_factor']
```
