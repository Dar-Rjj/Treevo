import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-day and 20-day returns
    df['5d_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20d_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['combined_close_momentum'] = (df['5d_return'] + df['20d_return']) / 2
    
    # Compute intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Price Range and Intraday Return
    df['intraday_price_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Integrate Volume Activity
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_trend_sum'] = df['volume'].rolling(window=5).sum()
    
    # Adjust Combined Intraday and Closing Price Momentum by Volume Trends
    df['adjusted_momentum'] = (df['combined_close_momentum'] + df['intraday_price_momentum']) * df['volume_trend_ratio']
    
    # Measure and Integrate High-Low Range Volatility
    df['daily_range'] = df['high'] - df['low']
    M = 20
    df['high_low_volatility'] = df['daily_range'].rolling(window=M).std()
    
    # Adjust Combined Intraday and Closing Price Momentum by High-Low Range Volatility
    df['volatility_adjusted_momentum'] = df['adjusted_momentum'] / df['high_low_volatility']
    
    # Combine Factors
    df['combined_momentum'] = df['adjusted_momentum'] + df['volatility_adjusted_momentum']
    
    # Smooth Combined Momentum
    df['smoothed_momentum'] = df['combined_momentum'].ewm(span=3, adjust=False).mean()
    
    # Incorporate Weighted Sum with Predefined Weights
    weight_momentum = 0.7
    weight_volatility = 0.3
    df['final_alpha_factor'] = (weight_momentum * df['smoothed_momentum']) + (weight_volatility * df['high_low_volatility'])
    
    return df['final_alpha_factor'].dropna()
