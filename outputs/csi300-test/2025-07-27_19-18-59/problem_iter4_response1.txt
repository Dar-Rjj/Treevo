```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Momentum
    df['intraday_price_range'] = df['high'] - df['low']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_momentum'] = df['intraday_price_range'] * df['intraday_return']
    
    # Volume Activity
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    df['accumulate_volume'] = df['volume'].rolling(window=5).sum()
    
    # Adjusted Momentum
    df['adjusted_momentum'] = df['intraday_momentum'] * df['volume_trend']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_momentum'].rolling(window=3).mean()
    
    return df['final_alpha_factor']
```
