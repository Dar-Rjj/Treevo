import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()
    
    # Calculate 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted combination of 5-day and 20-day returns
    df['weighted_return'] = (2 * df['5_day_return'] + df['20_day_return']) / 3
    
    # Compute intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and intraday return
    df['intraday_momentum'] = (df['high'] - df['low']) * df['intraday_return']
    
    # Determine volume trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust dynamic intraday and closing price momentum
    df['adjusted_momentum'] = df['weighted_return'] + df['intraday_momentum'] * np.sqrt(df['volume_trend'])
    
    # Calculate true range
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Volume-weighted true range
    df['vol_weighted_tr'] = df['true_range'] * df['volume_trend']
    
    # Measure and integrate high-low range stability
    M = 5  # Number of days for rolling standard deviation
    df['daily_range'] = df['high'] - df['low']
    df['range_std'] = df['daily_range'].rolling(window=M).std()
    df['range_stability'] = 1 / (1 + df['range_std'])
    
    # Combine adjusted momentum and volume-weighted true range
    df['combined_momentum'] = df['adjusted_momentum'] + df['vol_weighted_tr']
    
    # Apply exponential moving average (EMA) with a 3-day span to the combined momentum
    df['final_alpha_factor'] = df['combined_momentum'].ewm(span=3, adjust=False).mean() * df['range_stability']
    
    return df['final_alpha_factor'].dropna()
