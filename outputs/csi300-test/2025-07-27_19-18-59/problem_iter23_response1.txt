```python
import pandas as pd
import numpy as np

def heuristics_v2(df, M=5, momentum_weight=0.7, volatility_weight=0.3):
    # Compute 5-day and 20-day returns
    df['5d_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20d_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day Returns
    df['combined_long_term_momentum'] = (df['5d_return'] + df['20d_return']) / 2
    
    # Compute intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Price Range and Intraday Return
    df['combined_intraday'] = df['intraday_range'] * df['intraday_return']
    
    # Integrate Volume Activity: Determine Volume Trends
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['past_5d_volume'] = df['volume'].rolling(window=5).sum()
    
    # Adjust Combined Intraday and Closing Price Momentum by Volume Trends
    df['volume_adjusted_momentum'] = df['combined_long_term_momentum'] * df['volume_trend_ratio']
    
    # Measure High-Low Range Volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=M).var()
    
    # Combine Factors
    df['combined_momentum'] = df['volume_adjusted_momentum'] + df['combined_intraday']
    
    # Smooth Combined Momentum with 3-day SMA
    df['smoothed_combined_momentum'] = df['combined_momentum'].rolling(window=3).mean()
    
    # Incorporate Weighted Sum
    df['final_alpha_factor'] = (momentum_weight * df['smoothed_combined_momentum']) + (volatility_weight * df['range_volatility'])
    
    return df['final_alpha_factor'].dropna()
```
