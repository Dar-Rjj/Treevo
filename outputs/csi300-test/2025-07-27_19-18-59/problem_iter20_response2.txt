```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted combination of 5-day and 20-day returns
    df['weighted_returns'] = (2 * df['5_day_return'] + df['20_day_return']) / 3
    
    # Intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and intraday return
    df['intraday_price_range'] = df['high'] - df['low']
    df['combined_intraday_momentum'] = df['intraday_price_range'] * df['intraday_return']
    
    # Volume trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    df['accumulated_volume'] = df['volume'].rolling(window=5).sum()
    df['volume_trend_ratio'] = df['volume'] / df['accumulated_volume'].shift(1)
    
    # Adjust combined intraday and closing price momentum by volume trend
    df['adjusted_momentum'] = df['weighted_returns'] + df['combined_intraday_momentum']
    df['volume_adjusted_momentum'] = df['adjusted_momentum'] * df['volume_trend_ratio']
    
    # True Range
    df['true_range'] = np.maximum(
        df['high'] - df['low'],
        np.maximum(
            np.abs(df['high'] - df['close'].shift(1)),
            np.abs(df['low'] - df['close'].shift(1))
        )
    )
    
    # Volume-Weighted True Range
    df['volume_weighted_true_range'] = df['true_range'] * df['volume_trend']
    
    # High-Low Range Stability
    M = 10  # Number of days for rolling standard deviation
    df['daily_range'] = df['high'] - df['low']
    df['range_stability'] = df['daily_range'].rolling(window=M).std()
    df['range_stability_inv'] = 1 / df['range_stability']
    
    # Combine adjusted momentum and volume-weighted true range
    df['combined_momentum'] = df['volume_adjusted_momentum'] + df['volume_weighted_true_range']
    
    # Apply Exponential Moving Average (EMA) with a 3-day span
    df['smoothed_momentum'] = df['combined_momentum'].ewm(span=3, adjust=False).mean()
    
    # Final Alpha Factor: Weighted sum of smoothed combined momentum and high-low range stability
    alpha_factor = 0.7 * df['smoothed_momentum'] + 0.3 * df['range_stability_inv']
    
    return alpha_factor
```
