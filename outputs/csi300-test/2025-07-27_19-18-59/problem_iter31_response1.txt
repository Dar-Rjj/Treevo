```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day Returns
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Determine Volume Trend Ratio
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust Combined Intraday and Closing Price Momentum by Volume Trends
    df['adjusted_momentum'] = df['combined_return'] * df['volume_trend_ratio']
    
    # Compute Rolling Variance over M days for High-Low Range
    M = 10  # Number of days for rolling variance
    df['high_low_range'] = df['high'] - df['low']
    df['rolling_variance'] = df['high_low_range'].rolling(window=M).var()
    
    # Normalize the rolling variance to make it comparable with the adjusted momentum
    df['normalized_rolling_variance'] = (df['rolling_variance'] - df['rolling_variance'].mean()) / df['rolling_variance'].std()
    
    # Combine adjusted momentum and high-low range volatility
    df['combined_factor'] = df['adjusted_momentum'] - df['normalized_rolling_variance']
    
    # Apply a 3-day Simple Moving Average (SMA) to the combined factor
    df['final_alpha_factor'] = df['combined_factor'].rolling(window=3).mean()
    
    return df['final_alpha_factor'].dropna()
```
