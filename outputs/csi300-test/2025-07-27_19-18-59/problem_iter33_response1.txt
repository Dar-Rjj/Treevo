```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = df['close'].pct_change(5)
    df['20_day_return'] = df['close'].pct_change(20)
    
    # Combine 5-day and 20-day returns
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine intraday price range and intraday return
    df['intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Determine volume trends
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    df['volume_trend_5d'] = df['volume'].rolling(window=5).mean()
    
    # Adjust combined intraday and closing price momentum by volume trends
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume_trend']
    
    # Measure high-low range volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=5).std()
    
    # Compute Exponential Moving Average (EMA) of Close Prices over N days
    df['ema_close'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate volume-adjusted momentum
    df['volume_adjusted_momentum_ema'] = (df['close'] - df['ema_close']) * df['volume_trend_5d']
    
    # Combine factors
    df['combined_momentum'] = df['volume_adjusted_momentum'] + df['volume_adjusted_momentum_ema']
    
    # Smooth combined momentum
    df['smoothed_combined_momentum'] = df['combined_momentum'].ewm(span=5, adjust=False).mean()
    
    # Incorporate weighted sum with predefined weights
    weight_momentum = 0.7
    weight_volatility = 0.3
    df['final_alpha_factor'] = (weight_momentum * df['smoothed_combined_momentum'] +
                                weight_volatility * df['range_volatility'])
    
    return df['final_alpha_factor']
```
