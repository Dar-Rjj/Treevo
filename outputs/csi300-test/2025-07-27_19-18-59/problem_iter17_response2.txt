```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted combination of 5-day and 20-day returns
    df['weighted_returns'] = (2 * df['5_day_return'] + df['20_day_return']) / 3
    
    # Compute intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and intraday return
    df['intraday_range_return'] = (df['high'] - df['low']) * df['intraday_return']
    
    # Determine volume trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Accumulate volume over multiple days
    df['volume_5_days'] = df['volume'].rolling(window=5).sum()
    
    # Adjust combined intraday and closing price momentum by volume trend
    df['adjusted_momentum'] = df['weighted_returns'] + df['intraday_range_return']
    df['adjusted_momentum'] = df['adjusted_momentum'] * (df['volume'] / df['volume_5_days'])
    
    # Integrate Volume-Weighted True Range
    df['true_range'] = df[['high' - 'low', abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1))]].max(axis=1)
    df['volume_weighted_true_range'] = df['true_range'] * (df['volume'] / df['volume'].shift(1))
    
    # Measure high-low range stability
    df['daily_range'] = df['high'] - df['low']
    df['range_std'] = df['daily_range'].rolling(window=20).std()
    df['range_stability'] = 1 / df['range_std']
    
    # Combine adjusted momentum and volume-weighted true range
    df['combined_momentum'] = df['adjusted_momentum'] + df['volume_weighted_true_range']
    
    # Apply exponential moving average (EMA) with a 3-day span
    df['smoothed_combined_momentum'] = df['combined_momentum'].ewm(span=3, adjust=False).mean()
    
    # Incorporate weighted sum with predefined weights
    alpha_factor = 0.7 * df['smoothed_combined_momentum'] + 0.3 * df['range_stability']
    
    return alpha_factor.dropna()
```
