```python
import pandas as pd

def heuristics_v2(df, N=20, M=20, momentum_weight=0.7, stability_weight=0.3):
    # Calculate Volume-Adjusted Momentum
    sma_close = df['close'].rolling(window=N).mean()
    momentum = (df['close'] - sma_close) * df['volume'].rolling(window=N).mean()
    
    # Measure High-Low Range Stability
    daily_range = df['high'] - df['low']
    range_stability = 1 / daily_range.rolling(window=M).std()
    
    # Normalize Volume-Adjusted Momentum
    norm_momentum = momentum / momentum.abs().max()
    
    # Normalize Range Stability
    norm_range_stability = range_stability / range_stability.abs().max()
    
    # Compute Weighted Sum
    alpha_factor = (norm_momentum * momentum_weight) + (norm_range_stability * stability_weight)
    
    return alpha_factor
```
