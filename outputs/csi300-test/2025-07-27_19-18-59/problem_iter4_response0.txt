```python
import pandas as pd

def heuristics_v2(df, N=20, M=10, weight_momentum=0.7, weight_stability=0.3):
    # Calculate Volume-Adjusted Momentum
    sma_close = df['close'].rolling(window=N).mean()
    momentum = (df['close'] - sma_close) * df['volume'].rolling(window=N).mean()
    
    # Measure High-Low Range Stability
    daily_range = df['high'] - df['low']
    range_stdev = daily_range.rolling(window=M).std()
    stability = 1 / (range_stdev + 1e-6)  # Add a small constant to avoid division by zero
    
    # Combine Volume-Adjusted Momentum and Range Stability
    factor = (weight_momentum * momentum) + (weight_stability * stability)
    
    return factor
```
