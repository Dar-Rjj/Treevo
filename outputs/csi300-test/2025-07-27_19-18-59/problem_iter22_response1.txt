import pandas as pd
import numpy as np

def heuristics_v2(df, M=10):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['combined_returns'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine intraday range and intraday return
    df['intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Combined intraday and closing price momentum
    df['combined_momentum'] = (df['combined_returns'] + df['intraday_momentum']) / 2
    
    # Volume trends
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_trend_sum'] = df['volume'].rolling(window=5).sum()
    df['volume_trend_adjusted'] = df['volume_trend_ratio'] * df['volume_trend_sum']
    
    # Adjust combined momentum by volume trend
    df['volume_adjusted_momentum'] = df['combined_momentum'] * df['volume_trend_adjusted']
    
    # High-Low range volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=M).var()
    
    # Integrate high-low range volatility
    df['final_momentum'] = df['volume_adjusted_momentum'] + df['range_volatility']
    
    # Smooth combined momentum with a 3-day SMA
    df['alpha_factor'] = df['final_momentum'].rolling(window=3).mean()
    
    return df['alpha_factor']
