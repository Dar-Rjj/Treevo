import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['Previous Close'] = df['close'].shift(1)
    df['True Range 1'] = df['high'] - df['low']
    df['True Range 2'] = (df['high'] - df['Previous Close']).abs()
    df['True Range 3'] = (df['low'] - df['Previous Close']).abs()
    df['True Range'] = df[['True Range 1', 'True Range 2', 'True Range 3']].max(axis=1)

    # Calculate Weighted Average True Range
    df['Weight'] = df['amount']
    df['Weighted True Range'] = df['True Range'] * df['Weight']

    # Sum the weighted true ranges and divide by total weight
    total_weight = df['Weight'].sum()
    total_weighted_true_range = df['Weighted True Range'].sum()

    # Adjust by volume
    df['Volume Multiplier'] = df['volume']
    df['Adjusted Weighted True Range'] = df['Weighted True Range'] * df['Volume Multiplier']

    # Sum the adjusted weighted true ranges and divide by total weight
    total_adjusted_weighted_true_range = df['Adjusted Weighted True Range'].sum()
    
    # Final factor value
    factor_value = total_adjusted_weighted_true_range / total_weight

    # Return the factor values as a pandas Series
    return pd.Series(factor_value, index=df.index, name='Weighted ATR with Volume and Amount')
