```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted combination of 5-day and 20-day returns
    df['weighted_returns'] = (3 * df['5_day_return'] + df['20_day_return']) / 4
    
    # Compute intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and intraday return
    df['intraday_price_range_momentum'] = (df['high'] - df['low']) * df['intraday_return']
    
    # Determine volume trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Accumulate volume over multiple days
    df['volume_sum_5d'] = df['volume'].rolling(window=5).sum()
    df['smoothed_volume'] = df['volume_sum_5d'] / 5
    
    # Adjust enhanced intraday and closing price momentum by volume trend
    df['adjusted_momentum'] = df['weighted_returns'] * (df['volume_trend'] ** 0.75)
    
    # Calculate Volume-Adjusted True Range
    df['truerange'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    df['prev_close'] = df['close'].shift(1)
    df['true_high_low'] = np.maximum(df['high'] - df['low'], 
                                      np.abs(df['high'] - df['prev_close']), 
                                      np.abs(df['low'] - df['prev_close']))
    df['volume_adjusted_truerange'] = df['true_high_low'] * (df['volume_trend'] ** 0.5)
    
    # Measure high-low range stability
    df['daily_range'] = df['high'] - df['low']
    df['range_stability'] = df['daily_range'].rolling(window=20).std()
    df['inverted_stability'] = 1 / (df['range_stability'] + 1e-6)  # Add small constant to avoid division by zero
    
    # Combine adjusted momentum and volume-adjusted true range
    df['combined_momentum'] = df['adjusted_momentum'] + df['volume_adjusted_truerange']
    
    # Apply Exponential Moving Average (EMA) with a 5-day span to the combined momentum
    df['smoothed_combined_momentum'] = df['combined_momentum'].ewm(span=5, adjust=False).mean()
    
    # Incorporate weighted sum with predefined weights
    alpha_factor = 0.7 * df['smoothed_combined_momentum'] + 0.3 * df['inverted_stability']
    
    return alpha_factor
```
