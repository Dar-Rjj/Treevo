```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['combined_price_momentum'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and intraday return
    df['intraday_adjusted'] = df['intraday_range'] * df['intraday_return']
    
    # Integrate volume activity
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_5d_sum'] = df['volume'].rolling(window=5).sum()
    df['price_momentum_volume_adjusted'] = df['combined_price_momentum'] * df['volume_trend_ratio']

    # Measure high-low range volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=5).std()

    # Smooth combined momentum with 3-day SMA
    df['smoothed_momentum'] = df['price_momentum_volume_adjusted'].rolling(window=3).mean()

    # Define weights for the final alpha factor
    momentum_weight = 0.6
    volatility_weight = 0.4

    # Calculate final alpha factor
    df['final_alpha_factor'] = (df['smoothed_momentum'] * momentum_weight) + (df['range_volatility'] * volatility_weight)

    return df['final_alpha_factor']
```
