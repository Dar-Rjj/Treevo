import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum and Volume-Adjusted Momentum
    df['intraday_price_range'] = df['high'] - df['low']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_momentum'] = df['intraday_price_range'] * df['intraday_return']
    
    # Determine Volume Trends
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_5d_sum'] = df['volume'].rolling(window=5).sum()
    
    # Adjust Intraday Momentum by Volume Trends
    df['adjusted_intraday_momentum'] = df['intraday_momentum'] * df['volume_trend_ratio']
    
    # Compute Simple Moving Average (SMA) of Close Prices over 3 days
    df['sma_close_3d'] = df['close'].rolling(window=3).mean()
    df['close_sma_diff'] = df['close'] - df['sma_close_3d']
    df['avg_volume_3d'] = df['volume'].rolling(window=3).mean()
    df['volume_adjusted_close_sma_diff'] = df['close_sma_diff'] * df['avg_volume_3d']
    
    # Measure and Integrate High-Low Range Stability
    df['daily_range'] = df['high'] - df['low']
    df['range_stability'] = 1 / df['daily_range'].rolling(window=5).std()
    
    # Combine Adjusted Intraday and Volume-Adjusted Momentum
    df['combined_momentum'] = df['adjusted_intraday_momentum'] + df['volume_adjusted_close_sma_diff']
    
    # Smooth Combined Momentum
    df['smoothed_combined_momentum'] = df['combined_momentum'].rolling(window=3).mean()
    
    # Incorporate Weighted Sum with Predefined Weights
    weight_momentum = 0.7
    weight_stability = 0.3
    df['final_alpha_factor'] = (weight_momentum * df['smoothed_combined_momentum'] +
                                weight_stability * df['range_stability'])
    
    return df['final_alpha_factor']
