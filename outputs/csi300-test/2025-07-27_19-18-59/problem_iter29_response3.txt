```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted combination of 5-day and 20-day returns
    df['weighted_returns'] = (3 * df['5_day_return'] + df['20_day_return']) / 4
    
    # Intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and intraday return
    df['combined_intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Volume trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust combined intraday and closing price momentum by volume trends
    df['adjusted_combined_momentum'] = df['weighted_returns'] + df['combined_intraday_momentum'] * (df['volume_trend'] ** 0.75)
    
    # Sum up volumes over the past 5 days
    df['sum_5_day_volume'] = df['volume'].rolling(window=5).sum()
    
    # Calculate True Range
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Volume-Adjusted True Range
    df['volume_adjusted_tr'] = df['true_range'] * (df['volume_trend'] ** 0.5)
    
    # Measure high-low range volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=5).std()
    df['range_stability'] = 1 / df['range_volatility']
    
    # Combine factors
    df['combined_factor'] = df['adjusted_combined_momentum'] + df['volume_adjusted_tr'] + df['range_stability']
    
    # Smooth combined factor with 5-day EMA
    df['alpha_factor'] = df['combined_factor'].ewm(span=5, adjust=False).mean()
    
    # Drop NaN values
    df.dropna(inplace=True)
    
    return df['alpha_factor']
```
