import pandas as pd

def heuristics_v2(df, N=20, M=30, momentum_weight=0.6, stability_weight=0.4):
    # Calculate Volume-Adjusted Momentum
    df['SMA_Close'] = df['close'].rolling(window=N).mean()
    df['Momentum'] = (df['close'] - df['SMA_Close']) * df['volume'].rolling(window=N).mean()
    
    # Measure High-Low Range Stability
    df['Daily_Range'] = df['high'] - df['low']
    df['Range_Std'] = df['Daily_Range'].rolling(window=M).std()
    df['Stability'] = 1 / (df['Range_Std'] + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Combine Factors
    df['Alpha_Factor'] = (momentum_weight * df['Momentum']) + (stability_weight * df['Stability'])
    
    return df['Alpha_Factor']
