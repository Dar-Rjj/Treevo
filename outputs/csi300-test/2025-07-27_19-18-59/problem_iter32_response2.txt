```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute intraday price range and return
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and return
    df['intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Integrate volume activity
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_trend_sum'] = df['volume'].rolling(window=5).sum()
    df['volume_adjusted_momentum'] = (df['combined_return'] + df['intraday_momentum']) * df['volume_trend_ratio']
    
    # Measure and integrate high-low range volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=20).var()
    
    # Calculate volume-adjusted momentum
    df['ema_close_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['volume_adjusted_momentum_2'] = (df['close'] - df['ema_close_20']) * df['volume_trend_sum'] / 20
    
    # Combine factors
    df['combined_momentum'] = df['volume_adjusted_momentum'] + df['volume_adjusted_momentum_2']
    
    # Smooth combined momentum
    df['smoothed_combined_momentum'] = df['combined_momentum'].ewm(span=5, adjust=False).mean()
    
    # Incorporate weighted sum with predefined weights
    weight_momentum = 0.7
    weight_volatility = 0.3
    df['final_alpha_factor'] = (weight_momentum * df['smoothed_combined_momentum']) + (weight_volatility * df['range_volatility'])
    
    # Return the final alpha factor
    return df['final_alpha_factor']
```
