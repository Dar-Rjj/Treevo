```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted combination of 5-day and 20-day returns
    df['weighted_return'] = (2 * df['5_day_return'] + df['20_day_return']) / 3
    
    # Compute intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and intraday return
    df['combined_momentum'] = (df['high'] - df['low']) * df['intraday_return']
    
    # Determine volume trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust dynamic intraday and closing price momentum by volume trend
    df['adjusted_momentum'] = df['combined_momentum'] * np.sqrt(df['volume_trend'])
    
    # Calculate true range
    df['true_range'] = (df[['high', 'low']]
                        .apply(lambda x: max(x['high'] - x['low'], 
                                            abs(x['high'] - df['close'].shift(1)), 
                                            abs(x['low'] - df['close'].shift(1))), axis=1))
    
    # Weight true range by volume
    df['volume_weighted_true_range'] = df['true_range'] * (df['volume'] / df['volume'].shift(1))
    
    # Measure high-low range stability
    rolling_std = df['high'] - df['low']
    df['range_stability'] = 1 / rolling_std.rolling(window=20).std()
    
    # Apply EMA to the combined momentum for final alpha factor
    df['final_momentum'] = df['adjusted_momentum'].ewm(span=7).mean()
    
    # Accumulate volume over multiple days
    df['volume_5_days'] = df['volume'].rolling(window=5).sum()
    
    # Adjust intraday momentum by volume trends
    df['volume_trend_ratio'] = df['volume_5_days'] / df['volume'].shift(5)
    df['adjusted_final_momentum'] = df['final_momentum'] * df['volume_trend_ratio']
    
    # Combine adjusted intraday momentum and volume-adjusted true range
    df['combined_adjusted_momentum'] = df['adjusted_final_momentum'] + df['volume_weighted_true_range']
    
    # Apply a 7-day EMA to the combined momentum
    df['ema_combined_momentum'] = df['combined_adjusted_momentum'].ewm(span=7).mean()
    
    # Introduce a weighted momentum component based on volume trend
    df['weight'] = df['volume_trend'].apply(lambda x: 1 if x > 1 else 0.5)
    df['weighted_momentum'] = df['weight'] * df['ema_combined_momentum']
    
    # Normalize the weighted sum
    df['normalized_weighted_momentum'] = df['weighted_momentum'] / df['weighted_momentum'].max()
    
    # Final alpha factor
    alpha_factor = df['normalized_weighted_momentum']
    
    return alpha_factor
```
