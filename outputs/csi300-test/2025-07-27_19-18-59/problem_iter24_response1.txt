import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted combination of 5-day and 20-day returns
    df['weighted_returns'] = (2 * df['5_day_return'] + df['20_day_return']) / 3
    
    # Intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine intraday price range and intraday return
    df['intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Combine intraday and closing price momentum
    df['combined_momentum'] = df['weighted_returns'] + df['intraday_momentum']
    
    # Volume trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust combined momentum by volume trend
    df['adjusted_momentum'] = df['combined_momentum'] * np.sqrt(df['volume_trend'])
    
    # Calculate True Range
    df['true_range'] = df[['high' - 'low', abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1))]].max(axis=1)
    
    # Volume-weighted True Range
    df['vwt_true_range'] = df['true_range'] * (df['volume'] / df['volume'].shift(1))
    
    # High-Low Range Stability
    df['daily_range'] = df['high'] - df['low']
    df['range_stability'] = 1 / df['daily_range'].rolling(window=10).std()
    
    # Directional Bias (5-day ADX)
    df['tr'] = df[['high' - 'low', abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1))]].max(axis=1)
    df['dm_pos'] = np.where(df['high'] > df['high'].shift(1), df['high'] - df['high'].shift(1), 0)
    df['dm_neg'] = np.where(df['low'].shift(1) > df['low'], df['low'].shift(1) - df['low'], 0)
    df['di_pos'] = 100 * (df['dm_pos'].rolling(window=5).sum() / df['tr'].rolling(window=5).sum())
    df['di_neg'] = 100 * (df['dm_neg'].rolling(window=5).sum() / df['tr'].rolling(window=5).sum())
    df['adx'] = 100 * (abs(df['di_pos'] - df['di_neg']) / (df['di_pos'] + df['di_neg']))
    
    # Adjust stability by directional bias
    df['adjusted_stability'] = df['range_stability'] * df['adx']
    
    # Combine factors
    df['combined_factors'] = df['adjusted_momentum'] + df['vwt_true_range']
    
    # Smooth combined momentum
    df['smoothed_momentum'] = df['combined_factors'].ewm(span=3).mean()
    
    # Final alpha factor with weighted sum
    df['final_alpha_factor'] = 0.7 * df['smoothed_momentum'] + 0.3 * df['adjusted_stability']
    
    return df['final_alpha_factor']
