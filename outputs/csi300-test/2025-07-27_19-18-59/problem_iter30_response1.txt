```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday price range and intraday return
    df['intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Combine intraday and closing price momentum
    df['combined_momentum'] = df['combined_return'] + df['intraday_momentum']
    
    # Determine volume trends
    df['volume_trend_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_trend_5d'] = df['volume'].rolling(window=5).sum()
    
    # Adjust combined momentum by volume trends
    df['volume_adjusted_momentum'] = df['combined_momentum'] * df['volume_trend_ratio']
    
    # Measure high-low range volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=5).var()
    
    # Calculate volume-adjusted momentum
    n_days = 5
    df['ema_close'] = df['close'].ewm(span=n_days, adjust=False).mean()
    df['volume_adjusted_momentum'] = (df['close'] - df['ema_close']) * df['volume'].rolling(window=n_days).mean()
    
    # Combine factors
    df['combined_factor'] = df['volume_adjusted_momentum'] + df['range_volatility']
    
    # Smooth combined momentum
    df['smoothed_combined_momentum'] = df['combined_factor'].ewm(span=5, adjust=False).mean()
    
    # Incorporate weighted sum with predefined weights
    weight_momentum = 0.7
    weight_volatility = 0.3
    df['final_alpha_factor'] = (weight_momentum * df['smoothed_combined_momentum']) + (weight_volatility * df['range_volatility'])
    
    return df['final_alpha_factor']
```
