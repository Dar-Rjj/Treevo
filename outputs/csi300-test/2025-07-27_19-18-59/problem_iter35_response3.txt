import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute intraday price range and return
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Integrate volume activity
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_trend'] = df['volume'].rolling(window=5).sum()
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume_ratio']
    
    # Measure high-low range volatility
    M = 10  # Number of days for rolling variance
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=M).var()
    
    # Calculate volume-adjusted momentum using EMA
    N = 10  # Number of days for EMA
    df['ema_close'] = df['close'].ewm(span=N, adjust=False).mean()
    df['volume_adjusted_ema_momentum'] = (df['close'] - df['ema_close']) * df['volume'].rolling(window=N).mean()
    
    # Combine factors
    df['combined_momentum'] = df['volume_adjusted_momentum'] + df['volume_adjusted_ema_momentum']
    
    # Smooth combined momentum with 3-day SMA
    df['smoothed_combined_momentum'] = df['combined_momentum'].rolling(window=3).mean()
    
    # Incorporate weighted sum with predefined weights
    weight_momentum = 0.7  # Weight for smoothed combined momentum
    weight_volatility = 0.3  # Weight for high-low range volatility
    df['final_alpha_factor'] = (weight_momentum * df['smoothed_combined_momentum']) + (weight_volatility * df['range_volatility'])
    
    # Return the final alpha factor
    return df['final_alpha_factor']
