```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()
    
    # Calculate 5-day and 20-day Returns
    df['5d_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20d_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted Combination of 5-day and 20-day Returns
    df['weighted_returns'] = (3 * df['5d_return'] + 2 * df['20d_return']) / 5
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Price Range and Intraday Return
    df['combined_momentum'] = (df['high'] - df['low']) * df['intraday_return']
    
    # Determine Volume Trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust Enhanced Intraday and Closing Price Momentum
    df['adjusted_momentum'] = df['combined_momentum'] * (df['volume_trend'] ** 0.7)
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    
    # Find Maximum of (High - Low), |High - Previous Close|, |Low - Previous Close|
    df['true_range'] = np.maximum.reduce([
        df['high'] - df['low'],
        abs(df['high'] - df['close'].shift(1)),
        abs(df['low'] - df['close'].shift(1))
    ])
    
    # Weight by Volume
    df['volume_adjusted_true_range'] = df['true_range'] * (df['volume_trend'] ** 0.5)
    
    # Calculate Rolling Standard Deviation of Daily Range over M days (let's assume M=14)
    rolling_std = df['high'] - df['low']
    df['rolling_std_14'] = rolling_std.rolling(window=14).std()
    
    # Combine Adjusted Momentum and Volume-Adjusted True Range
    df['combined_factor'] = df['adjusted_momentum'] + df['volume_adjusted_true_range']
    
    # Invert Rolling Standard Deviation to Reflect Stability
    df['stability_factor'] = 1 / (df['rolling_std_14'] + 1e-6)
    
    # Apply Exponential Moving Average (EMA) with a 14-day span to the combined momentum for final alpha factor
    df['final_alpha'] = df['combined_factor'].ewm(span=14).mean() * df['stability_factor']
    
    # Compute Short-Term Volatility: EMA of Absolute 5-day Return over 14 days
    df['short_term_vol'] = df['5d_return'].abs().ewm(span=14).mean()
    
    # Compute Long-Term Volatility: EMA of Absolute 20-day Return over 28 days
    df['long_term_vol'] = df['20d_return'].abs().ewm(span=28).mean()
    
    # Calculate Dynamic Weights
    df['dynamic_weights'] = df['short_term_vol'] / (df['short_term_vol'] + df['long_term_vol'])
    
    # Final Alpha Factor: (Dynamic Weights * Adjusted Momentum) + ((1 - Dynamic Weights) * Volume-Adjusted True Range)
    df['final_alpha'] = (df['dynamic_weights'] * df['adjusted_momentum']) + ((1 - df['dynamic_weights']) * df['volume_adjusted_true_range'])
    
    # Drop NaN values
    df = df.dropna()
    
    return df['final_alpha']
```
