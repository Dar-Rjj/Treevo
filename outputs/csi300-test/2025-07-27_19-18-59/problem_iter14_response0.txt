```python
import pandas as pd
import numpy as np
from scipy.stats import skew

def heuristics_v2(df, N=20, M=5):
    # Calculate Volume-Weighted Momentum
    close_ema = df['close'].ewm(span=N, adjust=False).mean()
    volume_weighted_momentum = (df['close'] - close_ema) * df['volume'].rolling(window=N).sum()
    
    # Calculate Combined Intraday and Closing Price Momentum
    five_day_return = (df['close'] / df['close'].shift(5)) - 1
    twenty_day_return = (df['close'] / df['close'].shift(20)) - 1
    combined_returns = (five_day_return + twenty_day_return) / 2
    
    intraday_range = df['high'] - df['low']
    intraday_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = intraday_range * intraday_return
    
    # Integrate Volume Activity
    volume_trend = df['volume'] / df['volume'].shift(1)
    accumulated_volume = df['volume'].rolling(window=5).sum()
    adjusted_intraday_closing_momentum = combined_returns * volume_trend
    
    # Measure High-Low Range Skewness and Volatility
    daily_range = df['high'] - df['low']
    rolling_skewness = daily_range.rolling(window=M).apply(skew, raw=False)
    rolling_variance = daily_range.rolling(window=M).var()
    
    # Combine Volume-Adjusted Momentum and High-Low Range Dynamics
    weights = {'w1': 0.4, 'w2': 0.3, 'w3': 0.3}
    combined_momentum = (volume_weighted_momentum * weights['w1']) + (adjusted_intraday_closing_momentum * weights['w2'])
    combined_factor = (combined_momentum * weights['w3']) + (rolling_skewness * (1 - weights['w3'])) + rolling_variance
    
    return combined_factor
```
