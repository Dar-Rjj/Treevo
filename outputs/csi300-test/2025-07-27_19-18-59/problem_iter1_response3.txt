import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate Intraday Price Range
    data['intraday_range'] = data['high'] - data['low']
    
    # Compute Intraday Return
    data['intraday_return'] = (data['close'] - data['open']) / data['open']
    
    # Combine Intraday Price Range and Intraday Return
    data['intraday_momentum'] = data['intraday_range'] * data['intraday_return']
    
    # Determine Volume Trends
    data['volume_trend'] = data['volume'] / data['volume'].shift(1)
    
    # Accumulate Volume Over Multiple Days
    data['volume_sum_5d'] = data['volume'].rolling(window=5).sum()
    
    # Merge Intraday Momentum and Volume Trends
    data['adjusted_momentum'] = data['intraday_momentum'] * data['volume_trend']
    
    # Incorporate Smoothing for Stability
    data['alpha_factor'] = data['adjusted_momentum'].rolling(window=3).mean()
    
    return data['alpha_factor']
