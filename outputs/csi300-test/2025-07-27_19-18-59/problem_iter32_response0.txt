import pandas as pd

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day returns
    df['combined_return'] = (df['5_day_return'] + df['20_day_return']) / 2
    
    # Compute intraday price range and return
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Price Range and Intraday Return
    df['intraday_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Volume Trends
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_5_days'] = df['volume'].rolling(window=5).sum()
    
    # Adjust Combined Intraday and Closing Price Momentum by Volume Trends
    df['adjusted_combined_momentum'] = df['combined_return'] * df['volume_ratio']
    
    # High-Low Range Volatility
    df['range_volatility'] = df['intraday_range'].rolling(window=30).var()
    
    # Volume-Adjusted Momentum
    df['sma_close_20'] = df['close'].rolling(window=20).mean()
    df['volume_adjusted_momentum'] = (df['close'] - df['sma_close_20']) * df['volume'].rolling(window=20).mean()
    
    # Combine Factors
    df['combined_momentum'] = df['adjusted_combined_momentum'] + df['intraday_momentum'] + df['volume_adjusted_momentum']
    
    # Smooth Combined Momentum
    df['smoothed_combined_momentum'] = df['combined_momentum'].rolling(window=3).mean()
    
    # Final Alpha Factor with Weighted Sum
    weight_combined_momentum = 0.7
    weight_volatility = 0.3
    df['alpha_factor'] = (weight_combined_momentum * df['smoothed_combined_momentum']) + (weight_volatility * df['range_volatility'])
    
    return df['alpha_factor'].dropna()
