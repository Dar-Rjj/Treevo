```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Price Range and Intraday Return
    intraday_momentum = intraday_price_range * intraday_return
    
    # Determine Volume Trends
    volume_trend_ratio = df['volume'] / df['volume'].shift(1)
    accumulated_volume = df['volume'].rolling(window=5).sum()
    
    # Adjust Intraday Momentum by Volume Trends
    adjusted_intraday_momentum = intraday_momentum * volume_trend_ratio
    
    # Calculate True Range
    true_range = pd.concat([
        df['high'] - df['low'],
        (df['high'] - df['close'].shift(1)).abs(),
        (df['low'] - df['close'].shift(1)).abs()
    ], axis=1).max(axis=1)
    
    # Average the True Ranges
    average_true_range = true_range.rolling(window=5).mean()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    volume_factor = df['volume'] / average_volume
    volume_adjusted_true_range = average_true_range / (df['volume'] * volume_factor)
    
    # Measure and Integrate High-Low Range Stability
    rolling_std_daily_range = (df['high'] - df['low']).rolling(window=5).std()
    
    # Combine Adjusted Intraday Momentum and Volume-Adjusted True Range
    combined_momentum = adjusted_intraday_momentum + volume_adjusted_true_range
    
    # Apply a 7-day Exponential Moving Average (EMA) to the combined momentum
    ema_combined_momentum = combined_momentum.ewm(span=7).mean()
    
    # Assign Weights Based on Volume Trend
    weights = np.where(volume_trend_ratio > 1, 1, 0.5)
    
    # Calculate Weighted Sum of Combined Momentum
    weighted_sum = (ema_combined_momentum * weights).rolling(window=7).sum()
    
    # Normalize the Weighted Sum
    normalized_weighted_sum = weighted_sum / weights.rolling(window=7).sum()
    
    # Final Alpha Factor
    alpha_factor = normalized_weighted_sum / rolling_std_daily_range
    
    return alpha_factor
```
