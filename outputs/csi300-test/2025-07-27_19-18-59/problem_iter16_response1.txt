```python
import pandas as pd

def heuristics_v2(df, N=20, M=10, momentum_weight=0.7, stability_weight=0.3):
    # Calculate Volume-Adjusted Momentum
    df['SMA'] = df['close'].rolling(window=N).mean()
    df['Momentum'] = (df['close'] - df['SMA']) * df['volume'].rolling(window=N).mean()
    
    # Measure High-Low Range Stability
    df['DailyRange'] = df['high'] - df['low']
    df['Stability'] = df['DailyRange'].rolling(window=M).std()
    
    # Combine Factors
    df['AlphaFactor'] = (momentum_weight * df['Momentum']) - (stability_weight * df['Stability'])
    
    return df['AlphaFactor']
```
