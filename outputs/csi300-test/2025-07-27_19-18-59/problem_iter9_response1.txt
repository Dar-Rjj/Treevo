```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Price Range and Intraday Return
    intraday_momentum = intraday_range * intraday_return
    
    # Determine Volume Trends
    volume_trend = df['volume'] / df['volume'].shift(1)
    volume_trend_smoothed = df['volume'].rolling(window=5).sum() / 5
    
    # Adjust Intraday Momentum by Volume Trends
    adjusted_intraday_momentum = intraday_momentum * volume_trend_smoothed
    
    # Calculate True Range
    true_range = pd.concat([
        df['high'] - df['low'],
        (df['high'] - df['close'].shift(1)).abs(),
        (df['low'] - df['close'].shift(1)).abs()
    ], axis=1).max(axis=1)
    
    # Average the True Ranges
    average_true_range = true_range.rolling(window=5).mean()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    volume_adjusted_true_range = average_true_range * (df['volume'] / average_volume)
    
    # Measure High-Low Range Stability
    daily_range = df['high'] - df['low']
    rolling_std = daily_range.rolling(window=5).std()
    stability_factor = 1 / (rolling_std + 1e-6)  # Add a small constant to avoid division by zero
    
    # Combine Adjusted Intraday Momentum and Volume-Adjusted True Range
    combined_momentum = adjusted_intraday_momentum + volume_adjusted_true_range
    
    # Apply a 7-day Exponential Moving Average (EMA) to the combined momentum
    alpha_factor = combined_momentum * stability_factor
    alpha_factor_ema = alpha_factor.ewm(span=7, adjust=False).mean()
    
    return alpha_factor_ema
```
