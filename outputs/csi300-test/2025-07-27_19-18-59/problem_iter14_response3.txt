```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the daily range
    df['range'] = (df['high'] - df['low']) / df['open']
    
    # Calculate the relative close position in the range
    df['rel_close_pos'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the normalized volume
    df['norm_volume'] = (df['volume'] - df['volume'].mean()) / df['volume'].std()
    
    # Calculate the factor by combining the above metrics
    df['factor'] = df['return'].shift(1) * df['range'] * df['rel_close_pos'] * df['norm_volume']

    return df['factor'].dropna()
```
