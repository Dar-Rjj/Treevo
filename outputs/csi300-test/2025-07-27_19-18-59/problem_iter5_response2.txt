```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range: High - Low
    intraday_price_range = df['high'] - df['low']
    
    # Compute Intraday Return: (Close - Open) / Open
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine: Intraday Price Range * Intraday Return
    intraday_momentum = intraday_price_range * intraday_return
    
    # Integrate Volume Activity
    # Volume Trend: Volume(t) / Volume(t-1)
    volume_trend = df['volume'] / df['volume'].shift(1)
    
    # Accumulate Volume: Sum of Volume over past 5 days
    accumulate_volume = df['volume'].rolling(window=5).sum()
    
    # Adjusted Momentum: Intraday Momentum * Volume Trend
    adjusted_momentum = intraday_momentum * volume_trend
    
    # Final Intraday Momentum: 3-day SMA of Adjusted Momentum
    final_intraday_momentum = adjusted_momentum.rolling(window=3).mean()
    
    # Compute 5-day Return
    five_day_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Compute 20-day Return
    twenty_day_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine 5-day and 20-day Returns
    combined_price_return = (five_day_return + twenty_day_return) / 2
    
    # Compute 5-day Volume Trend
    five_day_volume_trend = df['volume'] - df['volume'].shift(5)
    
    # Compute 20-day Volume Trend
    twenty_day_volume_trend = df['volume'] - df['volume'].shift(20)
    
    # Combine 5-day and 20-day Volume Trends
    combined_volume_trend = (five_day_volume_trend + twenty_day_volume_trend) / 2
    
    # Weighted Average of Intraday and Long-Term Momentum
    weighted_average_momentum = (final_intraday_momentum * 0.6) + (combined_price_return * 0.4)
    
    # Measure High-Low Range Stability
    # Determine Daily Range
    daily_range = df['high'] - df['low']
    
    # Compute Rolling Standard Deviation over M days
    rolling_std_daily_range = daily_range.rolling(window=20).std()
    
    # Invert to Reflect Stability
    stability = 1 / rolling_std_daily_range
    
    # Combine All Factors
    # Compute Weighted Sum
    alpha_factor = (weighted_average_momentum * 0.7) + (stability * 0.3)
    
    return alpha_factor
```
