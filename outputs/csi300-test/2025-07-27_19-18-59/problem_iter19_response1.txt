```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Price Range and Intraday Return
    intraday_momentum = intraday_price_range * intraday_return
    
    # Determine Volume Trend
    volume_trend = df['volume'] / df['volume'].shift(1)
    volume_trend.fillna(1, inplace=True)  # Handle the first day's volume trend
    
    # Adjust Intraday Momentum by Volume Trend
    adjusted_intraday_momentum = intraday_momentum * volume_trend
    
    # Calculate True Range
    true_range = pd.concat([
        df['high'] - df['low'],
        abs(df['high'] - df['close'].shift(1)),
        abs(df['low'] - df['close'].shift(1))
    ], axis=1).max(axis=1)
    
    # Average True Ranges
    average_true_range = true_range.rolling(window=14).mean()
    
    # Adjust True Range by Volume
    volume_adjusted_true_range = average_true_range / df['volume']
    
    # Measure High-Low Range Stability
    daily_range = df['high'] - df['low']
    range_stability = 1 / daily_range.rolling(window=14).std()
    range_stability.fillna(method='bfill', inplace=True)  # Handle NaNs in the beginning
    
    # Smooth Combined Momentum
    combined_momentum = adjusted_intraday_momentum + volume_adjusted_true_range
    smoothed_combined_momentum = combined_momentum.ewm(span=7, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = smoothed_combined_momentum * range_stability
    
    return final_alpha_factor
```
