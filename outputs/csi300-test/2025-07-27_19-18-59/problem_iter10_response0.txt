```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20, momentum_weight=0.7, volatility_weight=0.3):
    # Calculate Volume-Weighted Momentum
    close_ema = df['close'].ewm(span=N, adjust=False).mean()
    momentum = (df['close'] - close_ema) * df['volume'].rolling(window=N).sum()
    
    # Measure High-Low Range Volatility
    daily_range = df['high'] - df['low']
    range_volatility = daily_range.rolling(window=M).var()
    
    # Combine Volume-Weighted Momentum and Range Volatility
    combined_factor = (momentum * momentum_weight) + (range_volatility * volatility_weight)
    
    return combined_factor
```
