```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5d_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20d_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted combination of 5-day and 20-day returns
    df['weighted_return'] = (3 * df['5d_return'] + 2 * df['20d_return']) / 5
    
    # Intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine intraday price range and intraday return
    df['combined_momentum'] = df['intraday_range'] * df['intraday_return']
    
    # Determine volume trend
    df['volume_trend'] = df['volume'] / df['volume'].shift(1)
    
    # Accumulate volume over multiple days
    df['volume_5d_sum'] = df['volume'].rolling(window=5).sum()
    
    # Adjust combined momentum by volume trend
    df['adjusted_momentum'] = df['combined_momentum'] * (df['volume_trend'] ** 0.75)
    
    # Calculate true range
    df['true_range'] = df[['high' - 'low', abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1))]].max(axis=1)
    
    # Volume-adjusted true range
    df['vol_adjusted_tr'] = df['true_range'] * (df['volume_trend'] ** 0.5)
    
    # High-low range volatility
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_range'].rolling(window=14).std()
    df['range_stability'] = 1 / (df['range_volatility'] + 1e-6)
    
    # Short-term and long-term volatility
    df['5d_abs_return'] = abs(df['5d_return'])
    df['20d_abs_return'] = abs(df['20d_return'])
    df['short_term_vol'] = df['5d_abs_return'].ewm(span=14, adjust=False).mean()
    df['long_term_vol'] = df['20d_abs_return'].ewm(span=28, adjust=False).mean()
    
    # Dynamic weights
    df['dynamic_weights'] = df['short_term_vol'] / (df['short_term_vol'] + df['long_term_vol'])
    
    # Smooth combined momentum
    df['smoothed_momentum'] = df['adjusted_momentum'].rolling(window=3).mean()
    
    # EMA of the smoothed momentum
    df['ema_momentum'] = df['smoothed_momentum'].ewm(span=5, adjust=False).mean()
    
    # Weighted sum
    df['weighted_sum'] = 0.7 * df['ema_momentum'] + 0.3 * df['range_stability']
    
    # Final alpha factor
    df['final_alpha'] = (df['dynamic_weights'] * df['ema_momentum']) + ((1 - df['dynamic_weights']) * df['vol_adjusted_tr'])
    
    return df['final_alpha']
```
