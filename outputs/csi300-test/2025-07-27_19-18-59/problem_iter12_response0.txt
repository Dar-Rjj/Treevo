```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=20, momentum_weight=0.7, skewness_weight=0.3):
    # Calculate Volume-Weighted Momentum
    close_ema = df['close'].ewm(span=N, adjust=False).mean()
    momentum = (df['close'] - close_ema) * df['volume'].rolling(window=N).sum()
    
    # Measure High-Low Range Skewness
    daily_range = df['high'] - df['low']
    range_skewness = daily_range.rolling(window=M).skew()
    
    # Combine Volume-Weighted Momentum and Range Skewness
    combined_factor = (momentum * momentum_weight) + (range_skewness * skewness_weight)
    
    return combined_factor
```
