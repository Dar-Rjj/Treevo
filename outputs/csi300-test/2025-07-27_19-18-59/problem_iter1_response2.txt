import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['Previous Close'] = df['close'].shift(1)
    df['High_Low'] = df['high'] - df['low']
    df['High_Close'] = (df['high'] - df['Previous Close']).abs()
    df['Low_Close'] = (df['low'] - df['Previous Close']).abs()
    df['True Range'] = df[['High_Low', 'High_Close', 'Low_Close']].max(axis=1)

    # Average the True Ranges over a period, for example 14 days
    df['Average True Range'] = df['True Range'].rolling(window=14).mean()

    # Normalize by Daily Volume
    df['Average Volume'] = df['volume'].rolling(window=14).mean()
    df['Volume Factor'] = df['volume'] / df['Average Volume']

    # Adjust the ATR by Volume
    df['Adjusted ATR'] = df['Average True Range'] * df['Volume Factor']

    # Return the final factor
    return df['Adjusted ATR']
