import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Compute Logarithmic Transformation
    df['log_momentum'] = np.log(1 + df['close_return'])
    
    # Determine Volume Change and Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine with Price Momentum
    df['volume_adjusted_momentum'] = df['log_momentum'] * df['volume_ratio']
    
    # Calculate Intraday Price Movement
    df['intraday_movement'] = df['high'] - df['low']
    
    # Integrate Past Movements
    N = 5  # Number of days for moving sum
    df['aggregated_intraday_movements'] = df['intraday_movement'].rolling(window=N).sum()
    df['average_close_over_N_days'] = df['close'].rolling(window=N).mean()
    df['integrated_movement_ratio'] = df['aggregated_intraday_movements'] / df['average_close_over_N_days']
    
    # Calculate Daily Price Volatility
    df['daily_volatility'] = (df['high'] - df['low']).rolling(window=N).mean()
    
    # Incorporate Trading Amount
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['amount_volume_weighted_momentum'] = df['volume_adjusted_momentum'] * df['amount_ratio']
    
    # Combine Factors
    df['alpha_factor'] = df['daily_volatility'] - df['amount_volume_weighted_momentum']
    df['alpha_factor'] *= df['integrated_movement_ratio']
    
    # Interpretable Adjustment
    df['open_close_return'] = (df['close'] - df['open']) / df['open']
    df['adjusted_alpha_factor'] = df['alpha_factor']
    df.loc[df['open_close_return'] > 0, 'adjusted_alpha_factor'] *= (1 + df['open_close_return'])
    df.loc[df['open_close_return'] < 0, 'adjusted_alpha_factor'] *= (1 - np.abs(df['open_close_return']))
    
    return df['adjusted_alpha_factor'].dropna()
