import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Volume-Weighted Exponential Momentum with Open Price
    ema_open = df['open'].ewm(span=10, adjust=False).mean()  # Span N days
    volume_weighted_ema = ema_open * df['volume']
    
    # Intraday Return with Close Price
    intraday_return = df['close'].pct_change()
    
    # Adjusted Open-Close Price Movement
    open_close_movement = (df['close'] - df['open']) / df['open']
    adjusted_open_close_movement = open_close_movement * (df['close'] - df['open'])
    
    # Combine Factors
    combined_factor = (
        intraday_price_range 
        - volume_weighted_ema 
        + intraday_return
    )
    
    # Adjust by Adjusted Open-Close Movement
    positive_adjustment = (1 + open_close_movement)
    negative_adjustment = (1 - abs(open_close_movement))
    adjustment = np.where(adjusted_open_close_movement > 0, positive_adjustment, negative_adjustment)
    combined_factor *= adjustment
    
    # Dynamic Volume Change Adjustment
    volume_ratio = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_increase_adjustment = 1 + volume_ratio
    volume_decrease_adjustment = 1 - abs(volume_ratio)
    volume_adjustment = np.where(volume_ratio > 0, volume_increase_adjustment, volume_decrease_adjustment)
    combined_factor *= volume_adjustment
    
    return pd.Series(combined_factor, index=df.index, name='alpha_factor')
