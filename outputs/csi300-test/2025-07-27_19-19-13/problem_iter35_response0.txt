```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Logarithmic daily return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Volume change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Volume ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Volume-weighted log momentum
    df['volume_weighted_log_momentum'] = df['log_return'] * df['volume_ratio']
    
    # True range
    df['true_range'] = df['high'] - df['low']
    
    # High-low to close difference
    df['high_low_close_diff'] = (df[['high', 'close']].max(axis=1) - 
                                 df[['low', 'close']].min(axis=1).shift(1))
    
    # Intraday return
    df['intraday_return'] = df['close'] - df['open']
    
    # Volume-weighted intraday return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Amount change
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    # Amount ratio
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    
    # Amount and volume-weighted price momentum
    df['amount_volume_weighted_price_momentum'] = df['volume_weighted_log_momentum'] * df['amount_ratio']
    
    # Aggregated movements (sum over N days, e.g., 30 days)
    N = 30
    df['aggregated_movements'] = df['volume_weighted_intraday_return'].rolling(window=N).sum()
    
    # Simple Moving Average of close prices over 30 days
    df['SMA_30_close'] = df['close'].rolling(window=N).mean()
    
    # Movement ratio
    df['movement_ratio'] = df['aggregated_movements'] / df['SMA_30_close']
    
    # Open-to-close return
    df['open_to_close_return'] = df['close'] - df['open']
    
    # Adjust alpha factor based on open-to-close return
    def adjust_alpha_factor(row):
        if row['open_to_close_return'] > 0:
            return row['alpha_factor'] * (1 + 0.7 * row['open_to_close_return'])
        else:
            return row['alpha_factor'] * (1 - 0.7 * abs(row['open_to_close_return']))
    
    # Combine all components
    df['alpha_factor'] = (df['amount_volume_weighted_price_momentum'] + 
                          df['true_range'] + 
                          df['volume_weighted_intraday_return'] + 
                          df['movement_ratio'])
    
    # Apply the adjustment
    df['adjusted_alpha_factor'] = df.apply(adjust_alpha_factor, axis=1)
    
    return df['adjusted_alpha_factor']
```
