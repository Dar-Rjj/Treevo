```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Price Momentum
    df['close_to_close_return'] = df['close'].pct_change()
    df['log_momentum'] = np.log(1 + df['close_to_close_return'])

    # Weight by Volume Trend
    df['volume_change'] = df['volume'].pct_change()
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_weighted_momentum'] = df['log_momentum'] * df['volume_ratio']

    # Calculate Daily Price Metrics
    df['daily_price_volatility'] = (df['high'] - df['low']) / df['close']
    df['daily_price_range'] = (df['high'] - df['low'])
    
    # Incorporate Trading Amount
    df['amount_change'] = df['amount'].pct_change()
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['amount_weighted_momentum'] = df['volume_weighted_momentum'] * df['amount_ratio']

    # Combine Factors
    df['alpha_factor'] = (df['daily_price_volatility'] + 
                          df['daily_price_range']) * df['amount_weighted_momentum']

    # Interpretable Adjustment
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['alpha_factor_adjusted'] = np.where(
        df['open_to_close_return'] > 0,
        df['alpha_factor'] * (1 + 0.5 * df['open_to_close_return']),
        df['alpha_factor'] * (1 - 0.5 * np.abs(df['open_to_close_return']))
    )

    return df['alpha_factor_adjusted'].dropna()
```
