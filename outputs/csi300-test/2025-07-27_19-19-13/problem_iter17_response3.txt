```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Price Momentum
    df['close_to_close_return'] = df['close'].pct_change()
    df['log_momentum'] = np.log(1 + df['close_to_close_return'])

    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']

    # Calculate Daily Intraday Volatility
    df['daily_intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Determine Volume Change and Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)

    # Weight by Volume Trend
    df['vol_weighted_log_momentum'] = df['log_momentum'] * df['volume_ratio']
    df['vol_weighted_intraday_return'] = df['intraday_return'] * df['volume_ratio']

    # Incorporate Trading Amount
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['amt_vol_weighted_log_momentum'] = df['vol_weighted_log_momentum'] * df['amount_ratio']
    df['amt_vol_weighted_intraday_return'] = df['vol_weighted_intraday_return'] * df['amount_ratio']

    # Combine and Adjust Factors
    df['alpha_factor'] = df['amt_vol_weighted_log_momentum'] + df['amt_vol_weighted_intraday_return'] + df['daily_intraday_volatility']
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['alpha_factor'] = df['alpha_factor'] * (1 + 0.5 * df['open_to_close_return']).where(df['open_to_close_return'] > 0, 
                                                                                           1 - 0.5 * abs(df['open_to_close_return']))

    # Volume-Weighted Momentum with Open Price
    df['open_momentum'] = df['open'].pct_change()
    df['vol_weighted_open_momentum'] = df['open_momentum'] * df['volume_ratio']

    # Intraday Return with Close Price
    df['close_intraday_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Open-Close Price Movement
    df['open_close_movement'] = df['close'] - df['open']

    # Final Adjustment
    df['alpha_factor'] -= df['vol_weighted_open_momentum']
    df['alpha_factor'] *= df['close_intraday_return']
    df['alpha_factor'] = df['alpha_factor'] * (1 + df['open_close_movement']).where(df['open_close_movement'] > 0, 
                                                                                   1 - abs(df['open_close_movement']))

    # Volume Change Adjustment
    df['volume_change_adj'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['alpha_factor'] *= (1 + df['volume_change_adj']).where(df['volume_change_adj'] > 0, 
                                                              1 - abs(df['volume_change_adj']))

    return df['alpha_factor']
```
