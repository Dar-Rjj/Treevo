```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return and Logarithmic Transformation
    close_to_close_return = df['close'].pct_change()
    log_close_to_close_return = np.log(1 + close_to_close_return)

    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']

    # Calculate Daily Intraday Volatility
    high_low_diff = df['high'] - df['low']
    high_close_diff = df['high'] - df['close'].shift(1)
    daily_intraday_volatility = (high_low_diff + high_close_diff) / 2

    # Determine Volume Change and Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)

    # Weight by Volume Trend
    volume_weighted_log_momentum = log_close_to_close_return * volume_ratio
    volume_weighted_intraday_return = intraday_return * volume_ratio

    # Incorporate Trading Amount
    amount_ratio = df['amount'] / df['amount'].shift(1)
    amount_volume_weighted_log_momentum = volume_weighted_log_momentum * amount_ratio
    amount_volume_weighted_intraday_return = volume_weighted_intraday_return * amount_ratio

    # Combine and Adjust Factors
    alpha_factor = (
        amount_volume_weighted_log_momentum +
        amount_volume_weighted_intraday_return +
        daily_intraday_volatility
    )

    # Calculate Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']

    # Conditionally Adjust Alpha Factor based on Open-to-Close Return
    alpha_factor = np.where(
        open_to_close_return > 0,
        alpha_factor * (1 + 0.5 * open_to_close_return),
        alpha_factor * (1 - 0.5 * np.abs(open_to_close_return))
    )

    # Introduce High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    alpha_factor = np.where(
        high_low_ratio > 1.05,
        alpha_factor * (1 + 0.2 * (high_low_ratio - 1)),
        np.where(
            high_low_ratio < 0.95,
            alpha_factor * (1 - 0.2 * (1 - high_low_ratio)),
            alpha_factor
        )
    )

    return pd.Series(alpha_factor, index=df.index, name='alpha_factor')
```
