```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return using High and Low prices
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Adjust for Volume Impact
    adjusted_intraday_return = intraday_return / df['volume']
    
    # Aggregate Past Movements using Moving Sum
    moving_sum = adjusted_intraday_return.rolling(window=5).sum()
    
    # Calculate Open-Close Price Movement
    open_close_movement = (df['close'] - df['open']) / df['open']
    
    # Combine Factors
    combined_factor = (moving_sum + open_close_movement + (df['high'] - df['low']) / df['low']) / 3
    
    return combined_factor
```
