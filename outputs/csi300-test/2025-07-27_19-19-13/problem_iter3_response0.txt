```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    df['price_diff_high_low'] = df['high'] - df['low']
    df['price_diff_high_prev_close'] = df['high'] - df['close'].shift(1)
    df['intraday_volatility'] = (df[['price_diff_high_low', 'price_diff_high_prev_close']].std(axis=1) * 100)

    # Calculate Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Open-Close Price Movement
    df['open_close_movement'] = (df['close'] - df['open']) / df['open']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Combine Factors
    weighted_intraday_return = df['intraday_return'] * 0.5 + df['open_close_movement'] * 0.5
    alpha_factor = (weighted_intraday_return +
                    df['intraday_volatility'] * 0.3 +
                    df['volume_change'] * 0.2 +
                    df['high_low_range'] * 0.1)
    
    return alpha_factor
```
