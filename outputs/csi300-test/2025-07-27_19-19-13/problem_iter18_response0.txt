```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20):
    # Calculate Raw Price Momentum
    df['close_to_close_return'] = df['close'].pct_change()
    df['log_return'] = np.log(1 + df['close_to_close_return'])

    # Weight by Volume Trend
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_weighted_momentum'] = df['log_return'] * df['volume_ratio']

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)

    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_ratio']

    # Combine Factors
    df['combined_factors'] = (
        df['volume_weighted_momentum'] +
        df['intraday_volatility'] +
        df['weighted_intraday_return']
    )

    # Incorporate Trading Amount
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['combined_factors'] *= df['amount_ratio']

    # Integrate Past Movements
    df['intraday_movements'] = (df['high'] - df['low'])
    df['aggregated_movements'] = df['intraday_movements'].rolling(window=N).sum()
    df['average_close'] = df['close'].rolling(window=N).mean()
    df['integrated_movements'] = df['aggregated_movements'] / df['average_close']

    # Calculate Daily Price Volatility
    df['daily_price_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['average_daily_volatility'] = df['daily_price_volatility'].rolling(window=N).mean()

    # Calculate Volume-Weighted Momentum
    df['shifted_close'] = df['close'].shift(N)
    df['volume_weighted_momentum'] = df['shifted_close'] * df['volume'] / df['volume'].rolling(window=N).mean()

    # Combine Additional Factors
    df['additional_factors'] = (
        df['combined_factors'] +
        df['average_daily_volatility'] -
        df['volume_weighted_momentum']
    )

    # Interpretable Adjustment
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['adjusted_alpha_factor'] = df['additional_factors']

    df.loc[df['open_to_close_return'] > 0, 'adjusted_alpha_factor'] *= (1 + 0.7 * df['open_to_close_return'])
    df.loc[df['open_to_close_return'] < 0, 'adjusted_alpha_factor'] *= (1 - 0.7 * abs(df['open_to_close_return']))

    return df['adjusted_alpha_factor']
```
