```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_to_close_return'] = df['close'].pct_change()
    # Apply logarithmic transformation
    df['log_close_to_close_return'] = np.log(1 + df['close_to_close_return'])

    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']

    # Calculate Daily Intraday Volatility
    df['price_diff_1'] = df['high'] - df['low']
    df['price_diff_2'] = df['high'] - df['close'].shift(1)
    df['daily_intraday_volatility'] = (df['price_diff_1'] + df['price_diff_2']) / 2

    # Determine Volume Change and Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)

    # Weight by Volume Trend
    df['volume_weighted_log_momentum'] = df['log_close_to_close_return'] * df['volume_ratio']
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume_ratio']

    # Incorporate Trading Amount
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['amount_weighted_log_momentum'] = df['volume_weighted_log_momentum'] * df['amount_ratio']
    df['amount_weighted_intraday_return'] = df['volume_weighted_intraday_return'] * df['amount_ratio']

    # Combine and Adjust Factors
    df['alpha_factor'] = (df['amount_weighted_log_momentum'] 
                          + df['amount_weighted_intraday_return'] 
                          + df['daily_intraday_volatility'])

    # Calculate Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']

    # Conditionally Adjust Alpha Factor
    df['alpha_factor'] = np.where(
        df['open_to_close_return'] > 0,
        df['alpha_factor'] * (1 + 0.5 * df['open_to_close_return']),
        df['alpha_factor'] * (1 - 0.5 * abs(df['open_to_close_return']))
    )

    return df['alpha_factor']
```
