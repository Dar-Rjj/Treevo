```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10):
    # Calculate Intraday Price Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Volume-Weighted Momentum with Open Price
    momentum_open = df['open'] / df['open'].shift(N) - 1
    df['volume_weighted_momentum'] = momentum_open * df['volume']
    
    # Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Adjust for Volume Impact
    df['avg_volume_past_N'] = df['volume'].rolling(window=N).mean()
    df['adjusted_intraday_return'] = df['intraday_return'] / df['avg_volume_past_N']
    
    # Aggregate Past Movements
    df['exp_moving_sum'] = df['adjusted_intraday_return'].ewm(span=N, adjust=False).sum()
    
    # Open-Close Price Movement
    df['open_close_movement'] = df['close'] - df['open']
    
    # Combine Factors
    combined_factor = df['intraday_volatility'] - df['volume_weighted_momentum']
    combined_factor *= df['adjusted_intraday_return']
    combined_factor = np.where(df['open_close_movement'] > 0, 
                               combined_factor * (1 + df['open_close_movement']), 
                               combined_factor * (1 - np.abs(df['open_close_movement'])))
    df['combined_factors'] = combined_factor
    
    # Volume Change Adjustment
    df['volume_ratio'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_adjustment = np.where(df['volume_ratio'] > 0, 
                                 (1 + df['volume_ratio']), 
                                 (1 - np.abs(df['volume_ratio'])))
    df['volume_adjusted_combined'] = df['combined_factors'] * volume_adjustment
    
    # Integrate Past Movements
    df['moving_sum_adjusted'] = df['adjusted_intraday_return'].rolling(window=N).sum()
    df['ratio_aggregated_movements'] = df['moving_sum_adjusted'] / df['close'].rolling(window=N).mean()
    
    # Calculate Daily Price Volatility
    df['daily_price_volatility'] = (df['high'] - df['low']).rolling(window=N).mean() / df['close'].rolling(window=N).mean()
    
    # Combine Additional Factors
    final_factor = df['daily_price_volatility'] - df['volume_weighted_momentum']
    final_factor *= df['ratio_aggregated_movements']
    
    # Interpretable Adjustment
    open_to_close_return = df['close'] / df['open'] - 1
    final_factor = np.where(open_to_close_return > 0, 
                            final_factor * (1 + open_to_close_return), 
                            final_factor * (1 - np.abs(open_to_close_return)))
    
    return pd.Series(final_factor, index=df.index, name='heuristics_v2')
```
