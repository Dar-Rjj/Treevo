```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Logarithmic Transformation
    df['log_momentum'] = np.log(1 + df['close_return'])
    
    # Determine Volume Change and Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine with Price Momentum
    df['volume_weighted_momentum'] = df['log_momentum'] * df['volume_ratio']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) + (df['high'] - df['close'].shift(1))
    
    # Weighted Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_ratio']
    
    # Combine Factors
    df['combined_factor'] = df['volume_weighted_momentum'] + df['intraday_volatility'] + df['weighted_intraday_return']
    
    # Incorporate Trading Amount
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['amount_weighted_factor'] = df['combined_factor'] * df['amount_ratio']
    
    # Interpretable Adjustment
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['alpha_factor'] = df['amount_weighted_factor']
    
    # Conditionally Adjust Alpha Factor
    df.loc[df['open_to_close_return'] > 0, 'alpha_factor'] *= (1 + 0.7 * df['open_to_close_return'])
    df.loc[df['open_to_close_return'] <= 0, 'alpha_factor'] *= (1 - 0.7 * abs(df['open_to_close_return']))
    
    return df['alpha_factor']
```
