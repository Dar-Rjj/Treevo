import pandas as pd

def heuristics_v2(df, shift_n=1):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Volume-Weighted Momentum with Open Price
    momentum_open = df['open'] - df['open'].shift(shift_n)
    volume_weighted_momentum = momentum_open * df['volume']
    
    # Intraday Return with Close Price
    intraday_return_close = df['close'] / df['close'].shift(1) - 1
    
    # Open-Close Price Movement
    open_close_movement = df['close'] - df['open']
    
    # Combine Factors
    factor = (intraday_price_range
              - volume_weighted_momentum
              * intraday_return_close)
    
    # Adjust by Open-Close Movement
    factor = factor * (1 + (df['close'] - df['open']).apply(lambda x: 1 + x if x > 0 else 1 - abs(x)))
    
    # Volume Change Adjustment
    volume_ratio = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_adjustment = (1 + volume_ratio).where(volume_ratio > 0, 1 - abs(volume_ratio))
    
    # Final Alpha Factor
    final_factor = factor * volume_adjustment
    
    return final_factor
