```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Compute Logarithmic Transformation
    df['log_momentum'] = np.log(1 + df['close_return'])
    
    # Determine Volume Change and Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine with Price Momentum
    df['volume_weighted_momentum'] = df['log_momentum'] * df['volume_ratio']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume'].pct_change()
    
    # Combine Factors
    df['combined_factor'] = (
        df['volume_weighted_momentum'] + 
        df['intraday_volatility'] + 
        df['weighted_intraday_return']
    )
    
    # Incorporate Trading Amount
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['combined_factor'] = df['combined_factor'] * df['amount_ratio']
    
    # Integrate Past Movements
    n_days = 10  # Define the number of days for the moving sum
    df['moving_sum_intraday'] = df['intraday_return'].rolling(window=n_days).sum()
    df['average_close_n_days'] = df['close'].rolling(window=n_days).mean()
    df['past_movements_ratio'] = df['moving_sum_intraday'] / df['average_close_n_days']
    
    df['combined_factor'] = df['combined_factor'] * df['past_movements_ratio']
    
    # Interpretable Adjustment
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['adjusted_alpha'] = np.where(
        df['open_to_close_return'] > 0,
        df['combined_factor'] * (1 + 0.7 * df['open_to_close_return']),
        df['combined_factor'] * (1 - 0.7 * np.abs(df['open_to_close_return']))
    )
    
    return df['adjusted_alpha']
```
