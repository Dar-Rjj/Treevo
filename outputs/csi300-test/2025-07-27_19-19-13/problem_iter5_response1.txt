```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday Price Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=N).mean()
    
    # Calculate Volume-Weighted Momentum
    momentum = df['close'].pct_change(N) * df['volume']
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Open-Close Price Movement
    open_close_movement = df['close'] - df['open']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine Factors
    alpha_factor = (intraday_volatility - momentum) * intraday_return
    
    # Adjust by Open-Close Price Movement and High-Low Range
    open_to_close_return = open_close_movement / df['open']
    adjustment_factor = open_to_close_return.apply(lambda x: 1 + x if x > 0 else 1 - abs(x))
    alpha_factor *= adjustment_factor
    
    # Interpretable Adjustment
    volume_change = df['volume'].pct_change()
    volume_adjustment = volume_change.apply(lambda x: 1 + x if x > 0 else 1 - abs(x))
    alpha_factor *= volume_adjustment
    
    return alpha_factor
```
