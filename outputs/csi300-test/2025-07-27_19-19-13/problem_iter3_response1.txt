```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Volatility
    daily_volatility = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Calculate Volume-Weighted Momentum
    momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    max_volume_period = df['volume'].rolling(window=20).max()
    volume_weighted_momentum = (momentum * df['volume']) / max_volume_period
    
    # Calculate Price Trend Strength using Exponential Moving Average
    ema = df['close'].ewm(span=20, adjust=False).mean()
    trend_strength = df['close'] / ema
    
    # Combine Factors
    combined_factor = daily_volatility - volume_weighted_momentum
    combined_factor *= trend_strength
    
    # Interpretable Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    adjustment = 1 + 2 * open_to_close_return
    adjustment = np.where(open_to_close_return > 0, adjustment, 1 - 2 * np.abs(open_to_close_return))
    final_alpha_factor = combined_factor * adjustment
    
    return final_alpha_factor
```
