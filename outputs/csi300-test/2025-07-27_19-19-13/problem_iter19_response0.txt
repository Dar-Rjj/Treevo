import pandas as pd
import numpy as np

def heuristics_v2(df, N=10):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Raw Price Momentum
    df['close_to_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['log_return'] = np.log(1 + df['close_to_close_return'])
    
    # Determine Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust for Volume Impact
    df['avg_volume_N_days'] = df['volume'].rolling(window=N).mean()
    df['adjusted_intraday_return'] = df['intraday_return'] / df['avg_volume_N_days']
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Combine Factors
    df['combined_factors'] = df['weighted_intraday_return'] + df['log_return'] + df['volume_ratio']
    
    # Incorporate Trading Amount
    df['amount_change'] = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    df['amount_weighted_intraday_return'] = df['weighted_intraday_return'] * df['amount_change']
    
    # Final Factor
    df['final_factor'] = (df['weighted_intraday_return'] * df['volume_ratio']) + df['log_return'] + df['intraday_return']
    
    # Integrate Past Movements
    df['moving_sum_adjusted_intraday'] = df['adjusted_intraday_return'].rolling(window=N).sum()
    df['avg_close_N_days'] = df['close'].rolling(window=N).mean()
    df['integrated_movements'] = df['moving_sum_adjusted_intraday'] / df['avg_close_N_days']
    
    # Aggregate Past Movements
    df['exp_moving_sum_intraday'] = df['adjusted_intraday_return'].ewm(span=N).mean()
    
    # Calculate Open-Close Price Movement
    df['open_close_movement'] = (df['close'] - df['open']) / df['open']
    
    # Combine Additional Factors
    df['additional_factors'] = df['final_factor'] + df['open_close_movement'] - df['log_return'] * df['volume_ratio']
    
    # Interpretable Adjustment
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['interpretable_adj_factor'] = df['final_factor']
    df.loc[df['open_to_close_return'] > 0, 'interpretable_adj_factor'] *= (1 + df['open_to_close_return'])
    df.loc[df['open_to_close_return'] < 0, 'interpretable_adj_factor'] *= (1 - abs(df['open_to_close_return']))
    
    return df['interpretable_adj_factor']
