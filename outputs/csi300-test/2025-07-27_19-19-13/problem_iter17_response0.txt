```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range and Volatility
    intraday_range = df['high'] - df['low']
    prev_close = df['close'].shift(1)
    intraday_volatility = (df['high'] + df['low']) / 2 - prev_close

    # Volume-Weighted Exponential Momentum with Open Price
    span = 10  # Span N days
    ema_open = df['open'].ewm(span=span, adjust=False).mean()
    volume_weighted_ema_open = ema_open * df['volume']

    # Intraday Return with Close Price
    intraday_return = df['close'] - df['close'].shift(1)

    # Adjusted Open-Close Price Movement
    price_movement = (df['close'] - df['open']) / df['open']
    adjusted_price_movement = df['close'] * price_movement

    # Combine Factors
    combined_factor = (
        intraday_range + intraday_volatility -
        volume_weighted_ema_open
    ) * intraday_return * adjusted_price_movement

    # Apply Open-Close movement adjustment
    open_close_ratio = (df['close'] - df['open']) / df['open']
    combined_factor = combined_factor * np.where(
        open_close_ratio > 0,
        1 + open_close_ratio,
        1 - np.abs(open_close_ratio)
    )

    # Dynamic Volume Change Adjustment
    volume_ratio = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_adjustment = np.where(
        volume_ratio > 0,
        1 + volume_ratio,
        1 - np.abs(volume_ratio)
    )
    combined_factor = combined_factor * volume_adjustment

    return combined_factor
```
