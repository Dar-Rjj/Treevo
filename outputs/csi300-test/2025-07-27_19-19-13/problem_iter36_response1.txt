import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate logarithmic return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate volume ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Volume-weighted log momentum
    df['volume_weighted_log_momentum'] = df['log_return'] * df['volume_ratio']
    
    # Calculate intraday return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    
    # Calculate intraday volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Weighted intraday return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_ratio']
    
    # Combined intraday factor
    df['combined_intraday_factor'] = df['intraday_volatility'] + df['weighted_intraday_return']
    
    # Calculate amount change
    df['amount_change'] = df['amount'] / df['amount'].shift(1)
    
    # Amount-adjusted momentum
    df['amount_adjusted_momentum'] = df['volume_weighted_log_momentum'] * df['amount_change']
    
    # Moving sum on combined intraday factor (N days, e.g., 30 days)
    N = 30
    df['aggregated_movements'] = df['combined_intraday_factor'].rolling(window=N).sum()
    
    # Compute the ratio of aggregated movements to average close over N days
    df['SMA_30_close'] = df['close'].rolling(window=N).mean()
    df['movement_ratio'] = df['aggregated_movements'] / df['SMA_30_close']
    
    # Calculate open-to-close return
    df['open_to_close_return'] = df['close'] - df['open']
    
    # Conditionally adjust alpha factor
    df['adjusted_factor'] = np.where(
        df['open_to_close_return'] > 0,
        df['movement_ratio'] * (1 + 0.5 * df['open_to_close_return'] / df['close']),
        df['movement_ratio'] * (1 - 0.5 * abs(df['open_to_close_return']) / df['close'])
    )
    
    # Return the adjusted factor
    return df['adjusted_factor'].dropna()
