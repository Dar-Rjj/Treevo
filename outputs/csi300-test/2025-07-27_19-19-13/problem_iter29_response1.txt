```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Logarithmic daily return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Volume change and volume ratio
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Volume-weighted log momentum
    df['volume_weighted_log_momentum'] = df['log_return'] * df['volume_ratio']
    
    # True range
    df['true_range'] = df['high'] - df['low']
    
    # High-low to close difference
    df['high_low_close_diff'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    
    # Intraday return
    df['intraday_return'] = df['close'] - df['open']
    
    # Volume-weighted intraday return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Amount change and amount ratio
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    
    # Amount and volume-weighted price momentum
    df['amount_volume_weighted_price_momentum'] = df['volume_weighted_log_momentum'] * df['amount_ratio']
    
    # Open-to-close return
    df['open_to_close_return'] = df['close'] / df['open'] - 1
    
    # Conditionally adjust alpha factor
    df['alpha_factor'] = df['amount_volume_weighted_price_momentum']
    df.loc[df['open_to_close_return'] > 0, 'alpha_factor'] *= (1 + 0.7 * df['open_to_close_return'])
    df.loc[df['open_to_close_return'] <= 0, 'alpha_factor'] *= (1 - 0.7 * abs(df['open_to_close_return']))
    
    # Synthesize final alpha factor
    df['final_alpha_factor'] = (
        df['alpha_factor'] + 
        df['true_range'] + 
        df['volume_weighted_intraday_return']
    )
    
    return df['final_alpha_factor'].dropna()
```
