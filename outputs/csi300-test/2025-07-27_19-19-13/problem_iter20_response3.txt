```python
import pandas as pd
import numpy as np

def heuristics_v2(df, span=20):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Volume-Weighted Exponential Momentum with Open Price
    ema_open = df['open'].ewm(span=span, adjust=False).mean()
    volume_weighted_ema_open = ema_open * df['volume']
    
    # Intraday Return with Close Price
    intraday_return = df['close'].pct_change()
    
    # Adjusted Open-Close Price Movement
    open_close_movement = (df['close'] - df['open']) / df['open']
    
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df[['high', 'low']].mean(axis=1)
    
    # Volume-Weighted Momentum
    close_t = df['close']
    close_t_n = df['close'].shift(span)
    volume_weighted_momentum = (close_t / close_t_n) * df['volume']
    
    # Combine Factors
    combined_factors = (
        intraday_price_range
        - volume_weighted_ema_open
        * intraday_return
        * (1 + np.sign(open_close_movement) * open_close_movement)
        + intraday_volatility
        - volume_weighted_momentum
    )
    
    # Final Adjustment
    combined_factors = combined_factors * (1 + np.abs((df['close'] - df['open']) / df['open']))
    
    # Dynamic Volume Change Adjustment
    volume_ratio = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    combined_factors = combined_factors * (1 + np.sign(volume_ratio) * volume_ratio)
    
    return combined_factors
```
