```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change and percentage change from the previous day
    df['price_change'] = df['close'].diff()
    df['pct_change'] = df['close'].pct_change()

    # Calculate the accumulation/distribution line
    df['adl'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['adl'] *= df['volume']
    df['adl'] = df['adl'].cumsum()

    # Calculate the relative strength index (RSI)
    df['gain'] = df['price_change'].apply(lambda x: x if x > 0 else 0)
    df['loss'] = df['price_change'].apply(lambda x: abs(x) if x < 0 else 0)
    avg_gain = df['gain'].rolling(window=14).mean()
    avg_loss = df['loss'].rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))

    # Calculate the money flow index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    df['mfi'] = 100 - (100 / (1 + money_ratio))

    # Combine the factors into a single alpha factor
    alpha_factor = 0.5 * (df['adl'] - df['adl'].shift(1)) + 0.3 * (df['rsi'] - 50) + 0.2 * (df['mfi'] - 50)

    return alpha_factor
```
