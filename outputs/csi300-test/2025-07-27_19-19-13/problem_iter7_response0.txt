```python
import pandas as pd

def heuristics_v2(df, n_days=10):
    # Calculate Daily Price Volatility
    daily_volatility = (df['high'] - df['low']).rolling(window=n_days).mean()

    # Calculate Volume-Weighted Momentum
    close_shifted = df['close'].shift(n_days)
    max_volume = df['volume'].rolling(window=n_days).max()
    volume_weighted_momentum = (df['close'] - close_shifted) * df['volume'] / max_volume

    # Combine Factors
    alpha_factor = daily_volatility - volume_weighted_momentum

    # Interpretable Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    
    def adjust_alpha_factor(adjustment_series, alpha_factor_series):
        adjusted_alpha_factor = []
        for i in range(len(alpha_factor_series)):
            if adjustment_series[i] > 0:
                adjusted_alpha_factor.append(alpha_factor_series[i] * (1 + 2 * adjustment_series[i]))
            else:
                adjusted_alpha_factor.append(alpha_factor_series[i] * (1 - 2 * abs(adjustment_series[i])))
        return pd.Series(adjusted_alpha_factor, index=alpha_factor_series.index)

    final_alpha_factor = adjust_alpha_factor(open_to_close_return, alpha_factor)

    return final_alpha_factor
```
