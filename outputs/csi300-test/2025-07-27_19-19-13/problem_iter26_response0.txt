import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate moving averages for closing prices
    df['MA_50_close'] = df['close'].rolling(window=50).mean()
    df['MA_200_close'] = df['close'].rolling(window=200).mean()
    
    # Calculate the ratio of today's volume to the average volume over a period (SMA_30_volume)
    df['SMA_30_volume'] = df['volume'].rolling(window=30).mean()
    df['volume_surge'] = df['volume'] / df['SMA_30_volume']
    
    # Price change following high/low volume days
    df['price_change_high_volume'] = (df['close'] - df['open']).where(df['volume'] > df['SMA_30_volume'] * 1.5, 0)
    
    # True range
    df['true_range'] = df['high'] - df['low']
    
    # Relative price movement
    df['relative_price_movement'] = df['close'] - (df['high'] + df['low']) / 2
    
    # Volume-weighted daily return
    df['volume_weighted_daily_return'] = df['daily_return'] * (df['volume'] / df['SMA_30_volume'])
    
    # Momentum-adjusted volume
    df['momentum_adjusted_volume'] = df['volume'] * (df['close'] / df['close'].shift(30) - 1)

    # Combine multiple factors into a single alpha factor
    df['alpha_factor'] = (
        df['volume_weighted_daily_return'] +
        df['momentum_adjusted_volume'] +
        (df['MA_50_close'] - df['MA_200_close']) / df['close'] +
        df['price_change_high_volume'] +
        df['relative_price_movement']
    )

    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
