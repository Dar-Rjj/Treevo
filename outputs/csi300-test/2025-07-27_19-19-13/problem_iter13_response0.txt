```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Volume-Weighted Momentum
    momentum = (df['close'] / df['close'].shift(20)) * df['volume']
    
    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Open-Close Price Movement
    open_close_movement = (df['close'] - df['open']) / df['open']
    
    # Combine Factors
    factor = (intraday_volatility - momentum) * intraday_return
    factor *= (1 + open_close_movement.abs()) if open_close_movement > 0 else (1 - open_close_movement.abs())
    
    # Volume Change Adjustment
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    factor *= (1 + volume_change) if volume_change > 0 else (1 - volume_change.abs())
    
    return factor
```
