```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Compute Logarithmic Transformation
    df['log_momentum'] = np.log(1 + df['close_return'])
    
    # Determine Volume Change
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine with Price Momentum
    df['volume_weighted_momentum'] = df['log_momentum'] * df['volume_ratio']
    
    # Calculate Daily Price Volatility
    df['daily_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate Amount Change
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    
    # Combine with Volume-Weighted Price Momentum
    df['amount_volume_weighted_momentum'] = df['volume_weighted_momentum'] * df['amount_ratio']
    
    # Add Daily Price Volatility
    df['combined_factor'] = (df['amount_volume_weighted_momentum'] + df['daily_volatility']) * df['amount_volume_weighted_momentum']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Conditionally Adjust Alpha Factor
    df['alpha_factor'] = df.apply(lambda row: row['combined_factor'] * (1 + row['open_to_close_return']) if row['open_to_close_return'] > 0 else row['combined_factor'] * (1 - abs(row['open_to_close_return'])), axis=1)
    
    return df['alpha_factor']
```
