```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Price Momentum
    close_to_close_return = df['close'].pct_change()
    log_return = np.log(1 + close_to_close_return)
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Daily Intraday Volatility
    high_low_diff = df['high'] - df['low']
    high_prev_close_diff = df['high'] - df['close'].shift(1)
    daily_intraday_volatility = 0.5 * (high_low_diff + high_prev_close_diff) / df['close'].shift(1)
    
    # Determine Volume Change and Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Weight by Volume Trend
    volume_weighted_log_return = log_return * volume_ratio
    volume_weighted_intraday_return = intraday_return * volume_ratio
    
    # Incorporate Trading Amount
    amount_ratio = df['amount'] / df['amount'].shift(1)
    volume_amount_weighted_log_return = volume_weighted_log_return * amount_ratio
    volume_amount_weighted_intraday_return = volume_weighted_intraday_return * amount_ratio
    
    # Combine and Adjust Factors
    alpha_factor = volume_amount_weighted_log_return + volume_amount_weighted_intraday_return + daily_intraday_volatility
    
    open_to_close_return = (df['close'] - df['open']) / df['open']
    alpha_factor = np.where(open_to_close_return > 0, alpha_factor * (1 + 0.5 * open_to_close_return), 
                            alpha_factor * (1 - 0.5 * np.abs(open_to_close_return)))
    
    high_low_ratio = df['high'] / df['low']
    alpha_factor = np.where(high_low_ratio > 1.05, alpha_factor * (1 + 0.2 * (high_low_ratio - 1)),
                            np.where(high_low_ratio < 0.95, alpha_factor * (1 - 0.2 * (1 - high_low_ratio)), alpha_factor))
    
    # Incorporate Opening Gap
    opening_gap = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    alpha_factor = np.where(opening_gap > 0, alpha_factor * (1 + 0.3 * opening_gap),
                            alpha_factor * (1 - 0.3 * np.abs(opening_gap)))
    
    return alpha_factor
```
