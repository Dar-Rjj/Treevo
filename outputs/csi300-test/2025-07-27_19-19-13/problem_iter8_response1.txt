```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Raw Price Momentum
    close_to_close_return = df['close'].pct_change()
    log_return = np.log(1 + close_to_close_return)
    
    # Calculate Intraday Volatility
    high_low_diff = df['high'] - df['low']
    high_prev_close_diff = df['high'] - df['close'].shift(1)
    intraday_volatility = (high_low_diff + high_prev_close_diff) / 2
    
    # Determine Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_change
    
    # Combine Factors
    combined_factor = (weighted_intraday_return * volume_ratio 
                       + intraday_volatility 
                       + log_return)
    
    # Incorporate Trading Amount
    amount_change = df['amount'] - df['amount'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    amount_weighted_return = weighted_intraday_return * amount_ratio
    
    # Final Factor
    alpha_factor = (amount_weighted_return * volume_ratio
                    + intraday_volatility
                    + log_return)
    
    # Interpretable Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    adjustment = 1 + open_to_close_return
    adjustment = np.where(open_to_close_return > 0, 1 + open_to_close_return, 1 - np.abs(open_to_close_return))
    alpha_factor *= adjustment
    
    return alpha_factor
```
