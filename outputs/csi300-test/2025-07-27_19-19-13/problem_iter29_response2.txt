```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Logarithmic Transformation
    df['log_momentum'] = np.log1p(df['close_return'])
    
    # Volume Change and Ratio
    df['volume_change'] = df['volume'].diff()
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine with Price Momentum
    df['vol_weighted_mom'] = df['log_momentum'] * df['volume_ratio']
    
    # Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs() + (df['high'] - df['close'].shift(1)).abs()
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Combine Factors
    df['combined_factor'] = df['intraday_volatility'] + df['weighted_intraday_return'] * df['vol_weighted_mom']
    
    # Incorporate Trading Amount
    df['amount_change'] = df['amount'].diff()
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['combined_factor'] *= df['amount_ratio']
    
    # Integrate Past Movements
    N = 5  # Number of days for moving sum
    df['aggregated_movements'] = df['weighted_intraday_return'].rolling(window=N).sum()
    df['average_close'] = df['close'].rolling(window=N).mean()
    df['past_movements_ratio'] = df['aggregated_movements'] / df['average_close']
    df['combined_factor'] *= df['past_movements_ratio']
    
    # Interpretable Adjustment
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['adjusted_alpha'] = df['combined_factor']
    df.loc[df['open_to_close_return'] > 0, 'adjusted_alpha'] *= (1 + 0.7 * df['open_to_close_return'])
    df.loc[df['open_to_close_return'] < 0, 'adjusted_alpha'] *= (1 - 0.7 * df['open_to_close_return'].abs())
    
    return df['adjusted_alpha'].dropna()
```
