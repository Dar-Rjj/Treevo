```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Logarithmic Transformation
    df['log_momentum'] = np.log(1 + df['close_return'])
    
    # Determine Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine with Price Momentum
    df['vol_weighted_momentum'] = df['log_momentum'] * df['volume_ratio']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) + (df['high'] - df['close'].shift(1))
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Combine Factors
    df['combined_factor'] = (df['intraday_volatility'] + 
                             df['weighted_intraday_return'] * 
                             df['vol_weighted_momentum'])
    
    # Incorporate Trading Amount
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['adjusted_combined_factor'] = df['combined_factor'] * df['amount_ratio']
    
    # Integrate Past Movements
    N = 5
    df['moving_sum_intraday'] = df['weighted_intraday_return'].rolling(window=N).sum()
    df['avg_close_N_days'] = df['close'].rolling(window=N).mean()
    df['integrated_movement'] = df['moving_sum_intraday'] / df['avg_close_N_days']
    
    # Interpretable Adjustment
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['alpha_factor'] = df['adjusted_combined_factor'] * df['integrated_movement']
    
    # Conditional Adjustment
    df['alpha_factor'] = np.where(
        df['open_to_close_return'] > 0,
        df['alpha_factor'] * (1 + 0.7 * df['open_to_close_return']),
        df['alpha_factor'] * (1 - 0.7 * abs(df['open_to_close_return']))
    )
    
    return df['alpha_factor']
```
