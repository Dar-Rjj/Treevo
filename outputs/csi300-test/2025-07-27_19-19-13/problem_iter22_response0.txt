```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['close']
    
    # Calculate Raw Price Momentum
    close_to_close_return = (df['close'].pct_change()).fillna(0)
    log_return = np.log(1 + close_to_close_return)
    
    # Determine Volume Change
    volume_change = (df['volume'].pct_change()).fillna(0)
    
    # Calculate Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1).fillna(1)
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_change
    
    # Combine Factors
    combined_factors = weighted_intraday_return + log_return + volume_ratio
    
    # Incorporate Trading Amount
    amount_change = (df['amount'].pct_change()).fillna(0)
    combined_factors *= amount_change
    
    # Integrate Past Movements
    n_days = 5
    moving_sum_weighted_intraday = weighted_intraday_return.rolling(window=n_days, min_periods=1).sum()
    avg_close = df['close'].rolling(window=n_days, min_periods=1).mean()
    aggregated_movements = moving_sum_weighted_intraday / avg_close
    
    # Calculate Open-Close Price Movement
    open_close_movement = (df['close'] - df['open']) / df['open']
    
    # Combine Additional Factors
    final_factor = combined_factors + open_close_movement - close_to_close_return * volume_ratio
    
    # Interpretable Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    adjusted_factor = final_factor.copy()
    adjusted_factor[final_factor > 0] *= (1 + open_to_close_return)
    adjusted_factor[final_factor < 0] *= (1 - np.abs(open_to_close_return))
    
    return adjusted_factor
```
