```python
import pandas as pd

def heuristics_v2(df, n_days=5):
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2)
    
    # Volume-Weighted Momentum
    volume_weighted_momentum = (df['close'] / df['close'].shift(n_days)) * df['volume']
    
    # Open-Close Price Movement
    open_close_movement = (df['close'] - df['open']) / df['open']
    
    # Adjust Open-Close Price Movement
    adjusted_open_close = open_close_movement.apply(lambda x: (1 + abs(x)) if x > 0 else (1 - abs(x)))
    open_close_factor = open_close_movement * adjusted_open_close
    
    # Volume Change Adjustment
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_volume_change = volume_change.apply(lambda x: (1 + x) if x > 0 else (1 - abs(x)))
    
    # Combine Factors
    alpha_factor = (intraday_volatility 
                    - volume_weighted_momentum 
                    * open_close_factor 
                    * adjusted_volume_change)
    
    return alpha_factor
```
