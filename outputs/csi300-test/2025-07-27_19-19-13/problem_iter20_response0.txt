```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Volume-Weighted Exponential Momentum with Open Price
    span = 5  # Example span, can be adjusted
    ema_open = df['open'].ewm(span=span).mean()
    volume_weighted_ema_open = ema_open * df['volume']
    
    # Raw Price Momentum and Logarithmic Transformation
    close_to_close_return = df['close'].pct_change()
    log_return = np.log(1 + close_to_close_return)
    
    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Daily Intraday Volatility
    high_low_diff = df['high'] - df['low']
    high_prev_close_diff = df['high'] - df['close'].shift(1)
    daily_intraday_volatility = (high_low_diff + high_prev_close_diff) / 2
    
    # Determine Volume Change and Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Weight by Volume Trend
    volume_weighted_log_momentum = log_return * volume_ratio
    volume_weighted_intraday_return = intraday_return * volume_ratio
    
    # Incorporate Trading Amount
    amount_ratio = df['amount'] / df['amount'].shift(1)
    volume_amount_weighted_log_momentum = volume_weighted_log_momentum * amount_ratio
    volume_amount_weighted_intraday_return = volume_weighted_intraday_return * amount_ratio
    
    # Combine and Adjust Factors
    combined_factors = (
        volume_amount_weighted_log_momentum +
        volume_amount_weighted_intraday_return +
        daily_intraday_volatility
    )
    
    open_to_close_return = (df['close'] - df['open']) / df['open']
    combined_factors = combined_factors * np.where(
        open_to_close_return > 0,
        1 + 0.5 * open_to_close_return,
        1 - 0.5 * np.abs(open_to_close_return)
    )
    
    # Adjusted Open-Close Price Movement
    adjusted_open_close_movement = (df['close'] - df['open']) / df['open']
    
    # Final Combination
    final_factor = (
        intraday_range -
        volume_weighted_ema_open * df['intraday_return'] * np.where(
            adjusted_open_close_movement > 0,
            1 + adjusted_open_close_movement,
            1 - np.abs(adjusted_open_close_movement)
        )
    )
    
    # Dynamic Volume Change Adjustment
    volume_change_adjustment = np.where(
        df['volume'] > df['volume'].shift(1),
        1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1),
        1 - np.abs((df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1))
    )
    final_factor = final_factor * volume_change_adjustment
    
    return final_factor
```
