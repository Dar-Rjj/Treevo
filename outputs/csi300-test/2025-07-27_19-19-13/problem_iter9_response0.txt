```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'] / df['close'].shift(1)
    
    # Compute Logarithmic Transformation of the return
    df['log_momentum'] = np.log(df['close_return'])
    
    # Determine Volume Change and Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine with Price Momentum
    df['volume_weighted_momentum'] = df['log_momentum'] * df['volume_ratio']
    
    # Calculate Daily Price Volatility
    df['daily_volatility'] = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Incorporate Trading Amount
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    
    # Combine with Volume-Weighted Price Momentum
    df['amount_volume_weighted_momentum'] = df['volume_weighted_momentum'] * df['amount_ratio']
    
    # Add Daily Price Volatility
    df['factor'] = df['amount_volume_weighted_momentum'] + df['daily_volatility']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Conditionally Adjust Alpha Factor
    df['alpha_factor'] = df.apply(lambda row: row['factor'] * (1 + 0.5 * row['open_to_close_return']) 
                                  if row['open_to_close_return'] > 0 else row['factor'] * (1 - 0.5 * abs(row['open_to_close_return'])), axis=1)
    
    return df['alpha_factor'].dropna()
```
