```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday Price Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Volume-Weighted Momentum with Open Price
    df['momentum_open'] = (df['open'] / df['open'].shift(n) - 1) * df['volume']
    
    # Intraday Return with Close Price
    df['intraday_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Open-Close Price Movement
    df['open_close_movement'] = df['close'] - df['open']
    
    # Combine Factors
    combined_factor = (df['intraday_volatility'] 
                       - df['momentum_open'] 
                       * df['intraday_return'])
    combined_factor = combined_factor.where(df['open_close_movement'] > 0,
                                            combined_factor * (1 + df['open_close_movement']))
    combined_factor = combined_factor.where(df['open_close_movement'] <= 0,
                                            combined_factor * (1 - abs(df['open_close_movement'])))
    
    # Volume Change Adjustment
    volume_ratio = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_factor = combined_factor.where(volume_ratio > 0,
                                            combined_factor * (1 + volume_ratio))
    adjusted_factor = adjusted_factor.where(volume_ratio <= 0,
                                            combined_factor * (1 - abs(volume_ratio)))
    
    return adjusted_factor
```
