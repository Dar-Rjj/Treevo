import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Logarithmic daily return
    df['log_daily_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Volume change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Volume ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Volume-weighted log momentum
    df['volume_weighted_log_momentum'] = df['log_daily_return'] * df['volume_ratio']
    
    # True range
    df['true_range'] = df['high'] - df['low']
    
    # High-low to close difference
    df['high_low_close_diff'] = np.maximum(df['high'], df['close'].shift(1)) - np.minimum(df['low'], df['close'].shift(1))
    
    # Intraday return
    df['intraday_return'] = df['close'] - df['open']
    
    # Volume-weighted intraday return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Amount change
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    # Amount ratio
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    
    # Amount and volume-weighted price momentum
    df['amount_volume_weighted_price_momentum'] = df['volume_weighted_log_momentum'] * df['amount_ratio']
    
    # Aggregated movements (N days, e.g., 30 days)
    N = 30
    df['aggregated_movements'] = df['volume_weighted_intraday_return'].rolling(window=N).sum()
    
    # Simple moving average of close over N days
    df['SMA_30_close'] = df['close'].rolling(window=N).mean()
    
    # Movement ratio
    df['movement_ratio'] = df['aggregated_movements'] / df['SMA_30_close']
    
    # Open-to-close return
    df['open_to_close_return'] = df['close'] - df['open']
    
    # Conditional adjustment
    df['adjusted_alpha_factor'] = df.apply(
        lambda row: row['amount_volume_weighted_price_momentum'] + row['true_range'] + row['volume_weighted_intraday_return'] + row['movement_ratio'],
        axis=1
    )
    df['final_alpha_factor'] = df.apply(
        lambda row: row['adjusted_alpha_factor'] * (1 + 0.7 * row['open_to_close_return']) if row['open_to_close_return'] > 0 else row['adjusted_alpha_factor'] * (1 - 0.7 * abs(row['open_to_close_return'])),
        axis=1
    )
    
    return df['final_alpha_factor']
