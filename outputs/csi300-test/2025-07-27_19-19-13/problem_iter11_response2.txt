```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=5):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Raw Price Momentum
    close_to_close_return = (df['close'] / df['close'].shift(1)) - 1
    log_return = np.log(1 + close_to_close_return)
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) + (df['high'] - df['close'].shift(1))
    
    # Determine Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_change
    
    # Combine Factors
    combined_factor = (weighted_intraday_return + intraday_volatility + log_return) * volume_ratio
    
    # Incorporate Trading Amount
    amount_change = df['amount'] - df['amount'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    combined_factor *= amount_ratio
    
    # Integrate Past Movements
    adjusted_intraday_movements = combined_factor.rolling(window=N).sum()
    average_close = df['close'].rolling(window=N).mean()
    integrated_movement = adjusted_intraday_movements / average_close
    
    # Calculate Daily Price Volatility
    daily_price_volatility = (df['high'] - df['low']).rolling(window=N).mean()
    
    # Calculate Volume-Weighted Momentum
    shifted_close = df['close'].shift(N)
    volume_weighted_momentum = shifted_close * df['volume'] / df['volume'].rolling(window=N).mean()
    
    # Combine Additional Factors
    final_factor = (integrated_movement + daily_price_volatility - volume_weighted_momentum)
    
    # Interpretable Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    adjustment = (final_factor > 0) * (1 + open_to_close_return) + (final_factor < 0) * (1 - np.abs(open_to_close_return))
    final_factor *= adjustment
    
    return final_factor
```
