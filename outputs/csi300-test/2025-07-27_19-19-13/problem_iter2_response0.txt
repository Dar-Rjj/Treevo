```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Logarithmic Transformation
    df['log_momentum'] = np.log(1 + df['close_return'])
    
    # Determine Volume Change and Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine with Price Momentum
    df['vol_weighted_momentum'] = df['log_momentum'] * df['volume_ratio']
    
    # Calculate Daily Price Volatility
    df['daily_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Average Volatility over Period
    df['avg_daily_volatility'] = df['daily_volatility'].rolling(window=5).mean()
    
    # Combine Factors
    df['alpha_factor'] = (df['vol_weighted_momentum'] + df['avg_daily_volatility']) * df['vol_weighted_momentum']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Conditionally Adjust Alpha Factor
    df['adjusted_alpha_factor'] = df['alpha_factor']
    df.loc[df['open_to_close_return'] > 0, 'adjusted_alpha_factor'] *= (1 + df['open_to_close_return'])
    df.loc[df['open_to_close_return'] < 0, 'adjusted_alpha_factor'] *= (1 - np.abs(df['open_to_close_return']))
    
    return df['adjusted_alpha_factor']
```
