```python
import pandas as pd
import numpy as np

def heuristics_v2(df, span=20):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Volume-Weighted Exponential Momentum with Open Price
    ema_open = df['open'].ewm(span=span, adjust=False).mean()
    volume_weighted_ema_open = ema_open * df['volume']
    
    # Intraday Return with Close Price
    intraday_return = df['close'].pct_change()
    
    # Adjusted Open-Close Price Movement
    open_close_movement = (df['close'] - df['open']) / df['open']
    adjusted_open_close = df['open'] + (df['close'] - df['open']) * (1 + open_close_movement)
    
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close'].rolling(window=span).mean()
    
    # Volume-Weighted Momentum
    close_t = df['close']
    close_t_minus_n = df['close'].shift(span)
    volume_weighted_momentum = (close_t / close_t_minus_n) * df['volume']
    
    # Raw Price Momentum
    close_to_close_return = df['close'].pct_change()
    log_momentum = np.log(1 + close_to_close_return)
    
    # Weight by Volume Trend
    volume_t = df['volume']
    volume_t_minus_1 = df['volume'].shift(1)
    volume_ratio = volume_t / volume_t_minus_1
    weighted_log_momentum = log_momentum * volume_ratio
    
    # Weighted Intraday Return
    volume_change = volume_t - volume_t_minus_1
    weighted_intraday_return = intraday_return * volume_change
    
    # Combine Factors
    alpha_factor = (
        intraday_price_range 
        - volume_weighted_ema_open
        + intraday_return
        + intraday_volatility
        - volume_weighted_momentum
        + weighted_intraday_return
        + weighted_log_momentum
    )
    
    # Final Adjustment
    open_close = df['close'] - df['open']
    alpha_factor = alpha_factor * (1 + np.abs(open_close)) if open_close > 0 else alpha_factor * (1 - np.abs(open_close))
    
    # Dynamic Volume Change Adjustment
    volume_ratio_dynamic = (volume_t - volume_t_minus_1) / volume_t_minus_1
    alpha_factor = alpha_factor * (1 + volume_ratio_dynamic) if volume_t > volume_t_minus_1 else alpha_factor * (1 - np.abs(volume_ratio_dynamic))
    
    # Incorporate Trading Amount
    amount_t = df['amount']
    amount_t_minus_1 = df['amount'].shift(1)
    amount_ratio = amount_t / amount_t_minus_1
    alpha_factor = alpha_factor * amount_ratio
    
    # Integrate Past Movements
    moving_sum_adjusted = alpha_factor.rolling(window=span).sum()
    avg_close = df['close'].rolling(window=span).mean()
    integrated_alpha = moving_sum_adjusted / avg_close
    
    # Interpretable Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    integrated_alpha = integrated_alpha * (1 + 0.7 * open_to_close_return) if open_to_close_return > 0 else integrated_alpha * (1 - 0.7 * np.abs(open_to_close_return))
    
    return integrated_alpha
```
