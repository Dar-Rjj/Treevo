```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_to_close_return'] = df['close'].pct_change()
    
    # Compute Logarithmic Transformation
    df['log_close_to_close_return'] = np.log(1 + df['close_to_close_return'])
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Daily Intraday Volatility
    df['daily_intraday_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Determine Volume Change and Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Weight by Volume Trend
    df['volume_weighted_log_momentum'] = df['log_close_to_close_return'] * df['volume_ratio']
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume_ratio']
    
    # Incorporate Trading Amount
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    
    # Combine with Volume-Weighted Factors
    df['volume_and_amount_weighted_log_momentum'] = df['volume_weighted_log_momentum'] * df['amount_ratio']
    df['volume_and_amount_weighted_intraday_return'] = df['volume_weighted_intraday_return'] * df['amount_ratio']
    
    # Combine and Adjust Factors
    df['alpha_factor'] = (df['volume_and_amount_weighted_log_momentum'] +
                          df['volume_and_amount_weighted_intraday_return'] +
                          df['daily_intraday_volatility'])
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Conditionally Adjust Alpha Factor
    df['alpha_factor'] = np.where(df['open_to_close_return'] > 0, 
                                  df['alpha_factor'] * (1 + 0.5 * df['open_to_close_return']),
                                  df['alpha_factor'] * (1 - 0.5 * abs(df['open_to_close_return'])))
    
    # Introduce High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Adjust Alpha Factor based on High-Low Ratio
    df['alpha_factor'] = np.where(df['high_low_ratio'] > 1.05, 
                                  df['alpha_factor'] * (1 + 0.2 * (df['high_low_ratio'] - 1)),
                                  np.where(df['high_low_ratio'] < 0.95,
                                           df['alpha_factor'] * (1 - 0.2 * (1 - df['high_low_ratio'])),
                                           df['alpha_factor']))
    
    # Incorporate Opening Gap
    df['opening_gap'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Adjust Alpha Factor based on Opening Gap
    df['alpha_factor'] = np.where(df['opening_gap'] > 0, 
                                  df['alpha_factor'] * (1 + 0.3 * df['opening_gap']),
                                  df['alpha_factor'] * (1 - 0.3 * abs(df['opening_gap'])))
    
    # Incorporate Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Adjust Alpha Factor based on Price Range
    price_range_adj = 0.1 * (df['price_range'] / df['close'])
    df['alpha_factor'] = np.where(df['price_range'] > df['close'] * 0.1, 
                                  df['alpha_factor'] * (1 + price_range_adj),
                                  df['alpha_factor'] * (1 - price_range_adj))
    
    return df['alpha_factor']
```
