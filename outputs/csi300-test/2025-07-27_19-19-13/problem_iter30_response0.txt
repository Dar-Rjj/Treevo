```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Volume-Weighted Exponential Momentum with Close Price
    ema_close = df['close'].ewm(span=5).mean()
    volume_weighted_ema = ema_close * df['volume']

    # Intraday Return with Open Price
    intraday_return_open = df['open'].pct_change()

    # Adjusted Open-Close Price Movement
    adjusted_open_close_movement = (df['close'] - df['open']) / df['open']
    adjusted_open_close = df['open'] * (1 + adjusted_open_close_movement)

    # Combine Factors
    combined_factor = intraday_price_range - volume_weighted_ema
    combined_factor = combined_factor * intraday_return_open
    combined_factor = combined_factor.where(adjusted_open_close_movement > 0, 
                                           combined_factor * (1 - abs(adjusted_open_close_movement)),
                                           combined_factor * (1 + adjusted_open_close_movement))

    # Dynamic Volume Change Adjustment
    volume_ratio = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    dynamic_volume_adjustment = combined_factor.where(volume_ratio > 0,
                                                      combined_factor * (1 + volume_ratio),
                                                      combined_factor * (1 - abs(volume_ratio)))

    # Calculate Raw Price Momentum
    raw_price_momentum = np.log((df['close'] - df['close'].shift(1)) / df['close'].shift(1))

    # Determine Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Calculate Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)

    # Weighted Intraday Return
    weighted_intraday_return = intraday_return_open * volume_change

    # Incorporate Trading Amount
    amount_change = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    weighted_intraday_amount = weighted_intraday_return * amount_change

    # Final Factor
    final_factor = weighted_intraday_amount * volume_ratio + raw_price_momentum

    # Integrate Past Movements
    past_movements = final_factor.rolling(window=5).sum() / df['close'].rolling(window=5).mean()

    # Enhanced Volume-Weighted Momentum
    enhanced_ema_close = df['close'].ewm(span=10).mean()
    enhanced_volume_weighted_ema = enhanced_ema_close * df['volume']

    # Short-Term Price Reversal
    short_term_reversal = np.log((df['close'] - df['open'].shift(-1)) / df['open'].shift(-1))

    # Volume-Adjusted Reversal
    volume_adjusted_reversal = short_term_reversal * volume_change

    # Adjusted Final Factor
    adjusted_final_factor = past_movements + enhanced_volume_weighted_ema + volume_adjusted_reversal
    adjusted_final_factor = adjusted_final_factor * (1 + intraday_price_range)

    # Intraday Volatility Factor
    intraday_volatility = (df['high'] - df['low']) / (df['high'] + df['low'])
    log_intraday_volatility = np.log(intraday_volatility)

    # Volume-Weighted Intraday Volatility
    volume_weighted_intraday_volatility = log_intraday_volatility * df['volume']

    # Integrated Final Factor
    integrated_final_factor = adjusted_final_factor + volume_weighted_intraday_volatility
    integrated_final_factor = integrated_final_factor * (1 + dynamic_volume_adjustment)

    return integrated_final_factor
```
