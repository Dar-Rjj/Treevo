```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10):
    # Calculate Daily Price Volatility
    df['price_volatility'] = df['high'] - df['low']
    df['avg_price_volatility'] = df['price_volatility'].rolling(window=n_days).mean()

    # Calculate Volume-Weighted Momentum
    df['momentum'] = df['close'] / df['close'].shift(n_days) - 1
    df['volume_weighted_momentum'] = df['momentum'] * df['volume']

    # Calculate Price Trend Strength using Moving Average
    df['moving_avg'] = df['close'].rolling(window=n_days).mean()
    df['trend_strength'] = df['close'] - df['moving_avg']

    # Combine Factors
    df['alpha_factor'] = (df['avg_price_volatility'] 
                          - df['volume_weighted_momentum'] 
                          * df['trend_strength'])

    # Interpretable Adjustment
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Conditionally Adjust Alpha Factor
    df.loc[df['open_to_close_return'] > 0, 'adjusted_alpha_factor'] = (
        df['alpha_factor'] * (1 + df['open_to_close_return'])
    )
    df.loc[df['open_to_close_return'] <= 0, 'adjusted_alpha_factor'] = (
        df['alpha_factor'] * (1 - abs(df['open_to_close_return']))
    )

    return df['adjusted_alpha_factor']
```
