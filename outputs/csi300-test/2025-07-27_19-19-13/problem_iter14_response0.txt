```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    
    # Adjust for Volume Impact
    df['avg_volume'] = df['volume'].rolling(window=n_days).mean()
    df['adjusted_intraday_return'] = df['intraday_return'] / df['avg_volume']
    
    # Aggregate Past Movements
    df['exp_moving_sum'] = df['adjusted_intraday_return'].ewm(span=n_days).sum()
    
    # Calculate Intraday Volatility
    df['intraday_volatility_1'] = (df['high'] - df['low']) / df['close']
    df['intraday_volatility_2'] = (df['high'] - df['close'].shift(1)) / df['close'].shift(1)
    df['intraday_volatility'] = (df['intraday_volatility_1'] + df['intraday_volatility_2']) / 2
    
    # Calculate Open-Close Price Movement
    df['open_close_movement'] = df['close'] - df['open']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Raw Price Momentum
    df['close_to_close_return'] = df['close'].pct_change()
    df['log_return'] = np.log(1 + df['close_to_close_return'])
    
    # Determine Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Combine Factors
    df['combined_factors'] = (df['adjusted_intraday_return'] + 
                              df['intraday_volatility'] + 
                              df['open_close_movement'] + 
                              df['high_low_range'] + 
                              df['log_return'] + 
                              df['volume_ratio'])
    
    # Final Factor
    df['final_factor'] = (df['weighted_intraday_return'] * df['volume_ratio'] + 
                          df['intraday_volatility'] + 
                          df['log_return'] + 
                          df['open_close_movement'])
    
    # Integrate Past Movements
    df['aggregated_movements'] = df['adjusted_intraday_return'].rolling(window=n_days).sum()
    df['avg_close'] = df['close'].rolling(window=n_days).mean()
    df['integrated_movements'] = df['aggregated_movements'] / df['avg_close']
    
    # Calculate Daily Price Volatility
    df['daily_price_volatility'] = (df['high'] - df['low']).rolling(window=n_days).mean() / df['close']
    
    # Calculate Volume-Weighted Momentum
    df['shifted_close'] = df['close'].shift(n_days)
    df['volume_weighted_momentum'] = (df['shifted_close'] * df['volume']) / df['volume'].rolling(window=n_days).mean()
    
    # Combine Additional Factors
    df['additional_factors'] = (df['daily_price_volatility'] - 
                                df['volume_weighted_momentum'] + 
                                df['final_factor'])
    
    # Interpretable Adjustment
    df['open_to_close_return'] = df['close'] / df['open'] - 1
    df['alpha_factor'] = df['additional_factors']
    df.loc[df['alpha_factor'] > 0, 'alpha_factor'] *= (1 + df['open_to_close_return'])
    df.loc[df['alpha_factor'] < 0, 'alpha_factor'] *= (1 - abs(df['open_to_close_return']))
    
    return df['alpha_factor']
```
