```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return_high = df['high'] / df['open'].shift(1) - 1
    intraday_return_low = df['low'] / df['open'].shift(1) - 1
    
    # Combine High and Low intraday returns
    intraday_return = (intraday_return_high + intraday_return_low) / 2
    
    # Calculate Intraday Volatility
    price_diff_1 = df['high'] - df['low']
    price_diff_2 = df['high'] - df['close'].shift(1)
    
    # Combine the two price differences
    intraday_volatility = (price_diff_1 + price_diff_2) / 2
    
    # Calculate Volume Change
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    
    # Combine Factors
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_change
    
    # Final factor
    factor = intraday_return + intraday_volatility + volume_change * weighted_intraday_return
    
    return factor
```
