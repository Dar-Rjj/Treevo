```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10, avg_vol_window=30):
    # Calculate Daily Price Volatility
    df['daily_volatility'] = (df['high'] - df['low']) / 2
    df['avg_daily_volatility'] = df['daily_volatility'].rolling(window=n_days).mean()
    
    # Calculate Volume-Weighted Momentum
    df['momentum'] = df['close'] - df['close'].shift(n_days)
    df['vol_normalized'] = df['volume'] / df['volume'].rolling(window=avg_vol_window).mean()
    df['vol_weighted_momentum'] = df['momentum'] * df['vol_normalized']
    
    # Calculate Price Trend Strength
    df['price_trend_strength'] = df['close'].rolling(window=n_days).mean()
    
    # Combine Factors
    df['alpha_factor'] = df['avg_daily_volatility'] - df['vol_weighted_momentum'] * df['price_trend_strength']
    
    # Interpretable Adjustment
    df['open_close_return'] = (df['close'] - df['open']) / df['open']
    df['alpha_factor_adjusted'] = df.apply(
        lambda x: x['alpha_factor'] * (1 + x['open_close_return']) if x['open_close_return'] > 0 else x['alpha_factor'] * (1 - abs(x['open_close_return'])), axis=1
    )
    
    return df['alpha_factor_adjusted'].dropna()
```
