```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Calculate Intraday Return using High and Low prices
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Adjust for Volume Impact by dividing by the average volume of past N days
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['adjusted_intraday_return'] = df['intraday_return'] / df['avg_volume']

    # Aggregate Past Movements using an Exponential Moving Sum
    df['exp_moving_sum'] = df['adjusted_intraday_return'].ewm(span=n).sum()

    # Calculate Intraday Volatility using (High - Low) / Close and (High - Previous Close) / Previous Close
    df['intraday_volatility_1'] = (df['high'] - df['low']) / df['close']
    df['prev_close'] = df['close'].shift(1)
    df['intraday_volatility_2'] = (df['high'] - df['prev_close']) / df['prev_close']
    df['intraday_volatility'] = (df['intraday_volatility_1'] + df['intraday_volatility_2']) / 2

    # Calculate Open-Close Price Movement
    df['open_close_movement'] = (df['close'] - df['open']) / df['open']

    # Calculate High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['low']

    # Combine Factors
    df['alpha_factor'] = (
        df['exp_moving_sum'] +
        df['intraday_volatility'] +
        df['open_close_movement'] +
        df['high_low_range']
    ) / 4

    return df['alpha_factor']
```
