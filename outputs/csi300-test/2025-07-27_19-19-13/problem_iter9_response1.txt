```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Price Momentum
    df['Close_Return'] = df['close'].pct_change()
    df['Log_Momentum'] = np.log(1 + df['Close_Return']).fillna(0)
    
    # Weight by Volume Trend
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1).fillna(1)
    df['Volume_Adjusted_Momentum'] = df['Log_Momentum'] * df['Volume_Ratio']
    
    # Calculate Daily Price Range
    df['Price_Range'] = df['high'] - df['low']
    df['Avg_Price_Range'] = df['Price_Range'].rolling(window=5).mean().fillna(0)
    
    # Incorporate Trading Amount
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1).fillna(1)
    df['Vol_Amt_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'] * df['Amount_Ratio']
    
    # Combine Factors
    df['Alpha_Factor'] = (df['Avg_Price_Range'] + df['Vol_Amt_Adjusted_Momentum'])
    
    # Interpretable Adjustment
    df['Open_Close_Return'] = (df['close'] - df['open']) / df['open']
    df['Alpha_Factor'] = np.where(
        df['Open_Close_Return'] > 0,
        df['Alpha_Factor'] * (1 + df['Open_Close_Return']),
        df['Alpha_Factor'] * (1 - np.abs(df['Open_Close_Return']))
    )
    
    return df['Alpha_Factor']
```
