```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Price Momentum
    df['close_return'] = df['close'].pct_change()
    df['log_momentum'] = np.log(1 + df['close_return'])
    
    # Weight by Volume Trend
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_weighted_momentum'] = df['log_momentum'] * df['volume_ratio']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Intraday Volatility
    df['price_diff_1'] = df['high'] - df['low']
    df['price_diff_2'] = df['high'] - df['close'].shift(1)
    df['intraday_volatility'] = (df['price_diff_1'] + df['price_diff_2']) / 2
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * (df['volume'] - df['volume'].shift(1))
    
    # Combine Factors
    df['combined_factor'] = (df['intraday_volatility'] + 
                             df['weighted_intraday_return'] * 
                             df['volume_weighted_momentum'])
    
    # Incorporate Trading Amount
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['amount_weighted_momentum'] = df['combined_factor'] * df['amount_ratio']
    
    # Integrate Past Movements
    N = 5  # Number of days for moving sum
    df['adjusted_intraday_movements'] = df['intraday_volatility'] + df['weighted_intraday_return']
    df['moving_sum'] = df['adjusted_intraday_movements'].rolling(window=N).sum()
    df['avg_close'] = df['close'].rolling(window=N).mean()
    df['past_movements_ratio'] = df['moving_sum'] / df['avg_close']
    
    # Interpretable Adjustment
    df['open_close_return'] = (df['close'] - df['open']) / df['open']
    df['open_price_return'] = (df['open'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Conditionally Adjust Alpha Factor
    alpha_factor = df['amount_weighted_momentum'] * df['past_movements_ratio']
    alpha_factor = np.where(
        df['open_close_return'] > 0,
        alpha_factor * (1 + 0.7 * df['open_close_return']),
        alpha_factor * (1 - 0.7 * np.abs(df['open_close_return']))
    )
    
    alpha_factor = np.where(
        df['open_price_return'] > 0,
        alpha_factor * (1 + 0.5 * df['open_price_return']),
        alpha_factor * (1 - 0.5 * np.abs(df['open_price_return']))
    )
    
    return pd.Series(alpha_factor, index=df.index, name='alpha_factor')
```
