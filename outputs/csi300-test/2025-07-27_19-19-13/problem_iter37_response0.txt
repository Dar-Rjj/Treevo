```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Price Momentum
    close_to_close_return = df['close'].pct_change()
    log_momentum = np.log(1 + close_to_close_return)
    
    # Weight by Volume Trend
    volume_ratio = df['volume'].pct_change() + 1
    volume_weighted_momentum = log_momentum * volume_ratio
    
    # Calculate Intraday Volatility
    max_price_diff = (df[['high', 'low', 'open']].max(axis=1) - 
                      df[['high', 'low', 'open']].min(axis=1))
    
    # Weighted Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    weighted_intraday_return = intraday_return * volume_ratio
    
    # Combine Factors
    combined_factor = (volume_weighted_momentum + 
                       max_price_diff + 
                       weighted_intraday_return)
    
    # Incorporate Trading Amount
    amount_ratio = df['amount'].pct_change() + 1
    combined_factor *= amount_ratio
    
    # Interpretable Adjustment
    open_to_close_return = (df['close'] / df['open']) - 1
    adjustment_multiplier = 1 + 0.8 * open_to_close_return
    adjustment_multiplier[open_to_close_return < 0] = 1 - 0.8 * abs(open_to_close_return[open_to_close_return < 0])
    
    alpha_factor = combined_factor * adjustment_multiplier
    
    return alpha_factor
```
