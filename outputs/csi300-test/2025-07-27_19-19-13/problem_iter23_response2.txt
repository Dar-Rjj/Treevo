import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['close']
    
    # Calculate Raw Price Momentum
    close_to_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    log_momentum = np.log(1 + close_to_close_return)
    
    # Determine Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_change
    
    # Combine Factors
    combined_factors = weighted_intraday_return + log_momentum + volume_ratio
    
    # Incorporate Trading Amount
    amount_change = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    combined_factors *= amount_change
    
    # Integrate Past Movements
    n_days = 5
    aggregated_movements = weighted_intraday_return.rolling(window=n_days).sum()
    average_close = df['close'].rolling(window=n_days).mean()
    integrated_past_movements = aggregated_movements / average_close
    
    # Calculate Open-Close Price Movement
    open_close_movement = (df['close'] - df['open']) / df['open']
    
    # Combine Additional Factors
    alpha_factor = combined_factors + open_close_movement - log_momentum * volume_ratio
    
    # Interpretable Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    alpha_factor = np.where(
        open_to_close_return > 0,
        alpha_factor * (1 + open_to_close_return),
        alpha_factor * (1 - np.abs(open_to_close_return))
    )
    
    # Volume-Weighted Exponential Momentum with Open Price
    exp_momentum_open = df['open'].ewm(span=n_days).mean() * df['volume']
    
    # Daily Intraday Volatility
    price_diff_high_low = df['high'] - df['low']
    price_diff_high_prev_close = df['high'] - df['close'].shift(1)
    daily_intraday_volatility = price_diff_high_low + price_diff_high_prev_close
    
    # Weight by Volume Trend
    volume_weighted_log_momentum = log_momentum * volume_ratio
    volume_weighted_intraday_return = intraday_return * volume_ratio
    
    # Incorporate Trading Amount
    amount_ratio = df['amount'] / df['amount'].shift(1)
    volume_amount_weighted_log_momentum = volume_weighted_log_momentum * amount_ratio
    volume_amount_weighted_intraday_return = volume_weighted_intraday_return * amount_ratio
    
    # Combine and Adjust Factors
    alpha_factor = (
        volume_amount_weighted_log_momentum +
        volume_amount_weighted_intraday_return +
        daily_intraday_volatility
    )
    alpha_factor = np.where(
        open_to_close_return > 0,
        alpha_factor * (1 + 0.5 * open_to_close_return),
        alpha_factor * (1 - 0.5 * np.abs(open_to_close_return))
    )
    
    # Adjusted Open-Close Price Movement
    adjusted_open_close_movement = (df['close'] - df['open']) / df['open'] * (df['close'] - df['open']) / df['open']
    
    # Final Combination
    alpha_factor = (
        intraday_return -
        exp_momentum_open +
        intraday_return * df['close']
    )
    alpha_factor = np.where(
        adjusted_open_close_movement > 0,
        alpha_factor * (1 + (df['close'] - df['open']) / df['open']),
        alpha_factor * (1 - np.abs((df['close'] - df['open']) / df['open']))
    )
    
    # Dynamic Volume Change Adjustment
    volume_ratio_dynamic = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    alpha_factor = np.where(
        volume_ratio_dynamic > 0,
        alpha_factor * (1 + volume_ratio),
        alpha_factor * (1 - np.abs(volume_ratio))
    )
    
    return pd.Series(alpha_factor, index=df.index)
