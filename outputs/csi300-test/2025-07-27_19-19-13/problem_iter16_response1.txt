```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Intraday Volatility: (High price - Low price) / Average over Period
    intraday_volatility = (df['high'] - df['low']) / df[['open', 'close']].mean(axis=1).rolling(window=N).mean()

    # Volume-Weighted Momentum: (Close price t / Close price t-N) * Volume
    momentum = (df['close'] / df['close'].shift(N)) * df['volume']

    # Intraday Return: (High price - Low price) / Low price
    intraday_return = (df['high'] - df['low']) / df['low']

    # Open-Close Price Movement: (Close price - Open price) / Open price
    open_close_movement = (df['close'] - df['open']) / df['open']

    # Adjust Intraday Return by Open-Close Movement
    open_close_adjustment = (1 + open_close_movement.abs()) if open_close_movement > 0 else (1 - open_close_movement.abs())
    intraday_return_adjusted = intraday_return * open_close_adjustment

    # Volume Change: (Today's Volume - Previous Day's Volume) / Previous Day's Volume
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Volume Change Adjustment
    volume_change_adjustment = (1 + volume_change) if volume_change > 0 else (1 - volume_change.abs())

    # Combine Factors
    factor = (
        intraday_volatility
        - momentum
        * intraday_return_adjusted
        * volume_change_adjustment
    )

    return factor
```
