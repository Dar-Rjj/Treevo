```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Compute Logarithmic Transformation
    df['log_return'] = np.log(1 + df['close_return'])
    
    # Determine Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    
    # Combine with Price Momentum
    df['volume_weighted_momentum'] = df['log_return'] * df['volume_change']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) + (df['high'] - df['close'].shift(1))
    
    # Weighted Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Combine Factors
    df['combined_factors'] = (df['volume_weighted_momentum'] 
                              + df['intraday_volatility'] 
                              + df['weighted_intraday_return'])
    
    # Incorporate Trading Amount
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['adjusted_factors'] = df['combined_factors'] * df['amount_ratio']
    
    # Interpretable Adjustment
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Conditionally Adjust Alpha Factor
    df['alpha_factor'] = df['adjusted_factors']
    df.loc[df['open_to_close_return'] > 0, 'alpha_factor'] *= (1 + 0.8 * df['open_to_close_return'])
    df.loc[df['open_to_close_return'] <= 0, 'alpha_factor'] *= (1 - 0.8 * abs(df['open_to_close_return']))
    
    return df['alpha_factor'].dropna()
```
