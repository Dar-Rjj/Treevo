```python
import pandas as pd
import numpy as np

def heuristics_v2(df, span=10):
    # Calculate Intraday Price Range
    df['Intraday_Price_Range'] = df['high'] - df['low']
    
    # Volume-Weighted Exponential Momentum with Close Price
    df['EMA_Close'] = df['close'].ewm(span=span, adjust=False).mean()
    df['Volume_Weighted_EMA'] = df['EMA_Close'] * df['volume']
    
    # Intraday Return with Open Price
    df['Intraday_Return_Open'] = (df['open'] / df['open'].shift(1)) - 1
    
    # Adjusted Open-Close Price Movement
    df['Open_Close_Movement'] = (df['close'] - df['open']) / df['open']
    df['Adjusted_Open_Close_Movement'] = df['Open_Close_Movement'] * df['open']
    
    # Combine Factors
    df['Combined_Factor'] = (
        df['Intraday_Price_Range'] 
        - df['Volume_Weighted_EMA'] 
        * df['Intraday_Return_Open']
        * df['Adjusted_Open_Close_Movement'].apply(lambda x: 1 + x if x > 0 else 1 - abs(x))
    )
    
    # Dynamic Volume Change Adjustment
    df['Volume_Ratio'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Dynamic_Volume_Adjustment'] = df['Volume_Ratio'].apply(lambda x: 1 + x if x > 0 else 1 - abs(x))
    df['Combined_Factor'] *= df['Dynamic_Volume_Adjustment']
    
    # Calculate Raw Price Momentum
    df['Raw_Price_Momentum'] = np.log((df['close'] / df['close'].shift(1)).fillna(1))
    
    # Weighted Intraday Return
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Weighted_Intraday_Return'] = df['Intraday_Return_Open'] * df['Volume_Change']
    
    # Incorporate Trading Amount
    df['Amount_Change'] = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    df['Volume_Weighted_Intraday_Return'] = df['Weighted_Intraday_Return'] * df['Amount_Change']
    
    # Final Factor
    df['Final_Factor'] = (df['Volume_Weighted_Intraday_Return'] * df['Volume_Ratio'] + df['Raw_Price_Momentum'])
    
    # Integrate Past Movements
    df['Moving_Sum'] = df['Final_Factor'].rolling(window=span).sum()
    df['Average_Close'] = df['close'].rolling(window=span).mean()
    df['Alpha_Factor'] = df['Moving_Sum'] / df['Average_Close']
    
    return df['Alpha_Factor']
```
