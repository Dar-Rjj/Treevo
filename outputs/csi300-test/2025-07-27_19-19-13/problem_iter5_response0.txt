```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return_high'] = (df['high'] - df['open']) / df['open']
    df['intraday_return_low'] = (df['low'] - df['open']) / df['open']
    
    # Adjust for Volume Impact
    df['adjusted_intraday_return'] = (df['intraday_return_high'] + df['intraday_return_low']) / 2
    df['volume_impact_adjusted'] = df['adjusted_intraday_return'] / df['volume']
    
    # Aggregate Past Movements
    df['moving_sum'] = df['volume_impact_adjusted'].rolling(window=10).sum()
    df['relative_strength'] = df['moving_sum'] / df['close'].rolling(window=10).mean()
    
    # Calculate Intraday Volatility
    df['price_diff_1'] = df['high'] - df['low']
    df['price_diff_2'] = df['high'] - df['close'].shift(1)
    df['intraday_volatility'] = (df['price_diff_1'] + df['price_diff_2']) / 2
    
    # Calculate Open-Close Price Movement
    df['open_close_movement'] = (df['close'] - df['open']) / df['open']
    
    # Calculate High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['low']
    
    # Combine Factors
    weight = 0.5  # Example weight, can be adjusted
    df['weighted_adjusted_intraday_return'] = df['adjusted_intraday_return'] * weight
    df['combined_factor'] = (df['adjusted_intraday_return'] + 
                             df['intraday_volatility'] + 
                             df['open_close_movement'] + 
                             df['high_low_range'] * df['weighted_adjusted_intraday_return'])
    
    return df['combined_factor']
```
