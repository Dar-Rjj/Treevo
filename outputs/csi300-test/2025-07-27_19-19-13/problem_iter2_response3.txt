```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday Price Movement
    intraday_movement = df['high'] - df['low']
    
    # Adjust for Volume Impact
    volume_weighted_intraday = intraday_movement / df['volume']
    
    # Integrate Past Movements
    aggregated_movements = volume_weighted_intraday.rolling(window=n).sum()
    avg_close = df['close'].rolling(window=n).mean()
    relative_strength = aggregated_movements / avg_close
    
    # Calculate Daily Price Volatility
    daily_volatility = (df['high'] - df['low']).rolling(window=n).mean()
    
    # Calculate Volume-Weighted Momentum
    momentum = df['close'] / df['close'].shift(n)
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_normalized_momentum = (momentum * df['volume']) / avg_volume
    
    # Combine Factors
    alpha_factor = (relative_strength + daily_volatility) - volume_normalized_momentum
    
    # Interpretable Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    adjustment = 1 + open_to_close_return
    adjustment[adjustment < 0] = 1 - (-open_to_close_return[open_to_close_return < 0])
    alpha_factor = alpha_factor * adjustment
    
    return alpha_factor
```
