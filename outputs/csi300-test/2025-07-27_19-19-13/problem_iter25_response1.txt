import pandas as pd
import numpy as np

def heuristics_v2(df, N=5):
    # Calculate Intraday Price Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Volume-Weighted Momentum with Open Price
    df['Momentum_Open'] = (df['open'] / df['open'].shift(N)) * df['volume']
    
    # Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2)
    
    # Adjust for Volume Impact
    df['Avg_Volume'] = df['volume'].rolling(window=N).mean()
    df['Adjusted_Intraday_Return'] = df['Intraday_Return'] / df['Avg_Volume']
    
    # Aggregate Past Movements
    df['Exponential_Moving_Sum'] = df['Adjusted_Intraday_Return'].ewm(span=N, adjust=False).sum()
    
    # Open-Close Price Movement
    df['Open_Close_Movement'] = df['close'] - df['open']
    
    # Combine Factors
    combined_factor = (
        df['Intraday_Volatility'] 
        - df['Momentum_Open'] 
        * df['Adjusted_Intraday_Return'] 
        * (1 + df['Open_Close_Movement']) if df['Open_Close_Movement'] > 0 else 1 - abs(df['Open_Close_Movement'])
    )
    
    # Volume Change Adjustment
    df['Volume_Ratio'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_adjustment = (1 + df['Volume_Ratio']) if df['Volume_Ratio'] > 0 else (1 - abs(df['Volume_Ratio']))
    combined_factor *= volume_adjustment
    
    return pd.Series(combined_factor, index=df.index, name='Enhanced_Alpha_Factor')
