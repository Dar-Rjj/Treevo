```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    df['intraday_volatility_high_low'] = (df['high'] - df['low'])
    df['intraday_volatility_high_close'] = (df['high'] - df['close'].shift(1)).abs()
    df['intraday_volatility'] = (df['intraday_volatility_high_low'] + df['intraday_volatility_high_close']) / 2
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Raw Price Momentum and apply log to return
    df['raw_price_momentum'] = np.log(df['close'] / df['close'].shift(1))
    
    # Determine Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Final Factor
    df['final_factor'] = (
        df['weighted_intraday_return'] * df['volume_ratio'] +
        df['intraday_volatility'] +
        df['raw_price_momentum']
    )
    
    return df['final_factor'].dropna()
```
