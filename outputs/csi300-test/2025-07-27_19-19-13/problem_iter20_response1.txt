```python
import pandas as pd
import numpy as np

def heuristics_v2(df, span=20):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Volume-Weighted Exponential Momentum with Open Price
    ema_open = df['open'].ewm(span=span, adjust=False).mean()
    volume_weighted_ema_open = ema_open * df['volume']

    # Intraday Return with Close Price
    intraday_return = df['close'].pct_change()

    # Adjusted Open-Close Price Movement
    open_close_movement = (df['close'] - df['open']) / df['open']
    adjusted_open_close = (df['close'] - df['open']) * open_close_movement
    
    # Combine Factors
    combined_factors = (intraday_price_range 
                        - volume_weighted_ema_open 
                        * intraday_return 
                        * (1 + open_close_movement if open_close_movement > 0 else 1 - np.abs(open_close_movement)))
    
    # Dynamic Volume Change Adjustment
    volume_ratio = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    dynamic_volume_adjustment = (1 + volume_ratio) if volume_ratio > 0 else (1 - np.abs(volume_ratio))
    combined_factors *= dynamic_volume_adjustment
    
    # Calculate Raw Price Momentum
    raw_price_momentum = np.log((df['close'] - df['close'].shift(1)) / df['close'].shift(1))
    
    # Determine Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_change
    
    # Incorporate Trading Amount
    amount_change = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    combined_with_amount = weighted_intraday_return * amount_change
    
    # Final Factor
    final_factor = combined_with_amount * volume_ratio + raw_price_momentum
    
    # Integrate Past Movements
    moving_sum = final_factor.rolling(window=span).sum()
    average_close = df['close'].rolling(window=span).mean()
    integrated_movements = moving_sum / average_close
    
    return integrated_movements
```
