```python
import pandas as pd

def heuristics_v2(df, n=1):
    # Calculate Intraday Price Volatility
    intraday_volatility = (df['high'] - df['low']) / df[['open', 'close']].mean(axis=1)
    intraday_volatility_avg = intraday_volatility.rolling(window=n).mean()

    # Volume-Weighted Momentum
    momentum = df['close'].pct_change(periods=n) * df['volume']

    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']

    # Open-Close Price Movement
    open_close_movement = df['close'] - df['open']

    # Combine Factors
    factor = (
        intraday_volatility_avg
        - momentum
        + intraday_return
        + open_close_movement.apply(lambda x: (1 + x) if x > 0 else (1 - abs(x)))
    )

    # Volume Change Adjustment
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    factor = factor * volume_change.apply(lambda x: (1 + x) if x > 0 else (1 - abs(x)))

    return factor
```
