```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20):
    # Calculate Raw Price Momentum
    df['close_return'] = df['close'].pct_change()
    df['log_momentum'] = np.log(1 + df['close_return'])
    
    # Weight by Volume Trend
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_weighted_momentum'] = df['log_momentum'] * df['volume_ratio']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) + (df['high'] - df['close'].shift(1))
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * (df['volume'] - df['volume'].shift(1))
    
    # Combine Factors
    df['combined_factors'] = (
        df['intraday_volatility'] + 
        df['weighted_intraday_return'] + 
        df['volume_weighted_momentum']
    )
    
    # Incorporate Trading Amount
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['combined_factors'] *= df['amount_ratio']
    
    # Integrate Past Movements
    df['adjusted_intraday_movements'] = df['combined_factors'].rolling(window=N).sum()
    df['avg_close'] = df['close'].rolling(window=N).mean()
    df['integrated_past_movements'] = df['adjusted_intraday_movements'] / df['avg_close']
    
    # Calculate Daily Price Volatility
    df['daily_volatility'] = (df['high'] - df['low']).rolling(window=N).mean()
    
    # Calculate Volume-Weighted Momentum
    df['shifted_close'] = df['close'].shift(N)
    df['vol_weighted_momentum'] = (df['shifted_close'] * df['volume']) / df['volume'].rolling(window=N).mean()
    
    # Combine Additional Factors
    df['additional_factors'] = (
        df['integrated_past_movements'] + 
        df['daily_volatility'] - 
        df['vol_weighted_momentum']
    )
    
    # Interpretable Adjustment
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['alpha_factor'] = df['additional_factors']
    df.loc[df['open_to_close_return'] > 0, 'alpha_factor'] *= (1 + 0.5 * df['open_to_close_return'])
    df.loc[df['open_to_close_return'] < 0, 'alpha_factor'] *= (1 - 0.5 * abs(df['open_to_close_return']))
    
    return df['alpha_factor'].dropna()
```
