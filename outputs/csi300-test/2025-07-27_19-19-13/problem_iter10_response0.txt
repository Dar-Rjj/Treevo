```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return_high'] = (df['high'] - df['open']) / df['open']
    df['intraday_return_low'] = (df['low'] - df['open']) / df['open']
    
    # Adjust for Volume Impact
    df['adjusted_intraday_return'] = (df['intraday_return_high'] + df['intraday_return_low']) / 2 / df['volume']

    # Aggregate Past Movements
    df['moving_sum_adjusted_return'] = df['adjusted_intraday_return'].rolling(window=5).sum()

    # Calculate Intraday Volatility
    df['intraday_volatility_high_low'] = (df['high'] - df['low']) / df['open']
    df['intraday_volatility_high_close'] = (df['high'] - df['close'].shift(1)) / df['open']
    
    # Combine intraday volatilities
    df['intraday_volatility'] = (df['intraday_volatility_high_low'] + df['intraday_volatility_high_close']) / 2
    
    # Calculate Open-Close Price Movement
    df['open_close_movement'] = (df['close'] - df['open']) / df['open']
    
    # Calculate High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['open']
    
    # Combine Factors
    df['alpha_factor'] = (
        df['moving_sum_adjusted_return'] +
        df['intraday_volatility'] +
        df['open_close_movement'] +
        df['high_low_range']
    ) / 4
    
    return df['alpha_factor']
```
