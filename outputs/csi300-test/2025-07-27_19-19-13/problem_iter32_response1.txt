import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return and logarithmic return
    df['daily_return'] = df['close'].pct_change()
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Volume change and volume ratio
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Volume-weighted log momentum
    df['volume_weighted_log_momentum'] = df['log_return'] * df['volume_ratio']
    
    # Intraday return and intraday volatility
    df['intraday_return'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['intraday_volatility'] = df['high'] - df['low']
    
    # High-Low to Close Difference
    df['high_low_to_close_diff'] = np.maximum(df['high'], df['close'].shift(1)) - np.minimum(df['low'], df['close'].shift(1))
    
    # Volume-weighted intraday return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Combined intraday factor
    df['combined_intraday_factor'] = df['intraday_volatility'] + df['volume_weighted_intraday_return']
    
    # Amount change and amount ratio
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    
    # Amount and volume-weighted price momentum
    df['amount_and_volume_weighted_price_momentum'] = df['volume_weighted_log_momentum'] * df['amount_ratio']
    
    # Aggregated movements (N Days, e.g., 30 days)
    N = 30
    df['aggregated_movements'] = df['combined_intraday_factor'].rolling(window=N).sum()
    
    # Moving average close over N Days
    df['SMA_30_close'] = df['close'].rolling(window=N).mean()
    
    # Movement ratio
    df['movement_ratio'] = df['aggregated_movements'] / df['SMA_30_close']
    
    # Open-to-close return
    df['open_to_close_return'] = df['close'] - df['open']
    
    # Conditionally adjust alpha factor
    df['adjusted_factor'] = np.where(
        df['open_to_close_return'] > 0,
        df['movement_ratio'] * (1 + 0.7 * df['open_to_close_return']),
        df['movement_ratio'] * (1 - 0.7 * np.abs(df['open_to_close_return']))
    )
    
    # Final alpha factor
    df['final_alpha_factor'] = (
        df['amount_and_volume_weighted_price_momentum'] +
        df['combined_intraday_factor'] +
        df['movement_ratio']
    )
    
    # Return the final alpha factor
    return df['final_alpha_factor']
