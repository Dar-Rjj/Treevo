```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10, alpha=0.3):
    # Calculate Intraday Return using High and Low prices
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Adjust for Volume Impact: Divide Intraday Return by Average Volume of Past N Days
    avg_volume_n_days = df['volume'].rolling(window=n_days).mean()
    adjusted_intraday_return = intraday_return / avg_volume_n_days
    
    # Aggregate Past Movements: Use Exponential Moving Sum on Adjusted Intraday Returns
    ema_adjusted_return = adjusted_intraday_return.ewm(alpha=alpha, adjust=False).sum()

    # Calculate Intraday Volatility: Use (High - Low) / Close and (High - Previous Close) / Previous Close
    intraday_volatility_1 = (df['high'] - df['low']) / df['close']
    intraday_volatility_2 = (df['high'] - df['close'].shift(1)) / df['close'].shift(1)
    intraday_volatility = (intraday_volatility_1 + intraday_volatility_2) / 2

    # Calculate Open-Close Price Movement
    open_close_movement = (df['close'] - df['open']) / df['open']

    # Calculate High-Low Range
    high_low_range = (df['high'] - df['low']) / df['low']

    # Combine Factors
    factor = (
        ema_adjusted_return +
        intraday_volatility +
        open_close_movement +
        high_low_range
    )
    
    return factor
```
