```python
import pandas as pd

def heuristics_v2(df, N=20):
    # Calculate Intraday Price Movement
    intraday_movement = df['high'] - df['low']
    
    # Adjust for Volume Impact
    volume_adjusted_movement = intraday_movement / df['volume']
    
    # Integrate Past Movements
    aggregated_movements = volume_adjusted_movement.rolling(window=N).sum()
    avg_close = df['close'].rolling(window=N).mean()
    ratio = aggregated_movements / avg_close
    
    # Calculate Daily Price Volatility
    daily_volatility = (df['high'] - df['low']).rolling(window=N).mean()
    
    # Calculate Volume-Weighted Momentum
    shifted_close = df['close'].shift(N)
    volume_weighted_momentum = (shifted_close * df['volume']) / df['volume'].rolling(window=N).mean()
    
    # Combine Factors
    factor = ratio + daily_volatility - volume_weighted_momentum
    
    # Interpretable Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    condition = open_to_close_return > 0
    factor = factor.where(condition, factor * (1 - open_to_close_return.abs()))
    factor = factor.where(~condition, factor * (1 + open_to_close_return))
    
    return factor
```
