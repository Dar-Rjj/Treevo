```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Price Momentum
    df['close_return'] = df['close'].pct_change()
    df['log_return'] = np.log(1 + df['close_return'])
    
    # Weight by Volume Trend
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_weighted_momentum'] = df['log_return'] * df['volume_ratio']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_ratio']
    
    # Combine Factors
    df['combined_factor'] = (df['volume_weighted_momentum'] +
                             df['intraday_volatility'] +
                             df['weighted_intraday_return'])
    
    # Incorporate Trading Amount
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['amount_adjusted_factor'] = df['combined_factor'] * df['amount_ratio']
    
    # Integrate Past Movements
    N = 5  # Number of days for moving sum
    df['adjusted_intraday_movements'] = (df['intraday_volatility'] + 
                                         df['weighted_intraday_return']).rolling(window=N).sum()
    df['average_close'] = df['close'].rolling(window=N).mean()
    df['past_movements_ratio'] = df['adjusted_intraday_movements'] / df['average_close']
    
    # Interpretable Adjustment
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['open_price_return'] = df['open'].pct_change()
    
    # Conditionally Adjust Alpha Factor
    alpha_factor = df['past_movements_ratio'] * df['amount_adjusted_factor']
    alpha_factor = np.where(
        df['open_to_close_return'] > 0,
        alpha_factor * (1 + 0.7 * df['open_to_close_return']),
        alpha_factor * (1 - 0.7 * abs(df['open_to_close_return']))
    )
    
    alpha_factor = np.where(
        df['open_price_return'] > 0,
        alpha_factor * (1 + 0.5 * df['open_price_return']),
        alpha_factor * (1 - 0.5 * abs(df['open_price_return']))
    )
    
    return pd.Series(alpha_factor, index=df.index, name='alpha_factor')
```
