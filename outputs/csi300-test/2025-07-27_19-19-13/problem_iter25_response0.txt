```python
import pandas as pd
import numpy as np

def heuristics_v2(df, span=10, n_days=10):
    # Calculate Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Volume-Weighted Exponential Momentum with Open Price
    df['ewm_open'] = df['open'].ewm(span=span).mean()
    df['volume_weighted_ewm_open'] = df['ewm_open'] * df['volume']
    
    # Intraday Return with Close Price
    df['intraday_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Adjusted Open-Close Price Movement
    df['price_movement_adjustment'] = (df['close'] - df['open']) / df['open']
    df['adjusted_open_close'] = (df['close'] - df['open']) * df['price_movement_adjustment']
    
    # Combine Factors
    combined_factors = (
        df['intraday_price_range']
        - df['volume_weighted_ewm_open']
        ) * df['intraday_return']
    
    # Adjust by Adjusted Open-Close Movement
    adjustment_factor = np.where(
        df['price_movement_adjustment'] > 0,
        1 + df['price_movement_adjustment'],
        1 - np.abs(df['price_movement_adjustment'])
    )
    combined_factors *= adjustment_factor
    
    # Dynamic Volume Change Adjustment
    volume_ratio = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    combined_factors = np.where(
        volume_ratio > 0,
        combined_factors * (1 + volume_ratio),
        combined_factors * (1 - np.abs(volume_ratio))
    )
    
    # Final Factor
    raw_price_momentum = df['close'] / df['close'].shift(span) - 1
    adjusted_intraday_movements = combined_factors
    aggregated_movements = adjusted_intraday_movements.rolling(window=n_days).sum()
    average_close_n_days = df['close'].rolling(window=n_days).mean()
    
    final_factor = (
        combined_factors * (1 + volume_ratio)
        + raw_price_momentum
        + aggregated_movements / average_close_n_days
    )
    
    return final_factor
```
