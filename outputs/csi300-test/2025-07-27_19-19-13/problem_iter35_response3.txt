```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the Exponential Moving Average of the money flow
    ema_money_flow = money_flow.ewm(span=14, adjust=False).mean()
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Heuristic: The alpha factor is the normalized difference between the EMA of money flow and its rolling mean
    # First, calculate the rolling mean and standard deviation of the EMA of money flow
    rolling_mean = ema_money_flow.rolling(window=10).mean()
    rolling_std = ema_money_flow.rolling(window=10).std()
    
    # Normalize the EMA of money flow
    normalized_ema_money_flow = (ema_money_flow - rolling_mean) / rolling_std
    
    # Calculate the final alpha factor
    alpha_factor = normalized_ema_money_flow * daily_return.shift(-1)
    
    # Drop NaN values
    alpha_factor = alpha_factor.dropna()
    
    return alpha_factor
```
