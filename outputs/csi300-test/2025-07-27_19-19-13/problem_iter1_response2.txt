```python
import pandas as pd

def heuristics_v2(df, n_days=10):
    # Calculate Intraday Price Movement
    intraday_movement = df['high'] - df['low']
    
    # Adjust for Volume Impact
    volume_adjusted_movement = intraday_movement / df['volume']
    
    # Integrate Past Movements: Aggregate Last N Days Movements
    moving_sum = volume_adjusted_movement.rolling(window=n_days).sum()
    
    # Calculate Relative Strength: Use Close price
    avg_close = df['close'].rolling(window=n_days).mean()
    
    # Compute Ratio: Divide Aggregated Movements by Average Close
    relative_strength = moving_sum / avg_close
    
    return relative_strength
```
