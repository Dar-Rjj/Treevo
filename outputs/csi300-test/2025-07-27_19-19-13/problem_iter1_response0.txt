```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'] / df['close'].shift(1)
    
    # Logarithmic Transformation of the return
    df['log_return'] = np.log(df['close_return'])
    
    # Determine Volume Change
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine Log Momentum with Volume Ratio
    df['alpha_factor'] = df['log_return'] * df['volume_ratio']
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
```
