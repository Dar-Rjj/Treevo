```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Close Return
    df['close_to_close_return'] = df['close'].pct_change()
    
    # Compute Logarithmic Transformation
    df['log_close_to_close_return'] = np.log(1 + df['close_to_close_return'])
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate Daily Intraday Volatility
    df['daily_intraday_volatility'] = np.maximum(df['high'] - df['low'], df['high'] - df['close'].shift(1))
    
    # Determine Volume Change
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Weight by Volume Trend
    df['volume_weighted_log_momentum'] = df['log_close_to_close_return'] * df['volume_ratio']
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume_ratio']
    
    # Incorporate Trading Amount
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['amount_volume_weighted_log_momentum'] = df['volume_weighted_log_momentum'] * df['amount_ratio']
    df['amount_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'] * df['amount_ratio']
    
    # Combine and Adjust Factors
    alpha_factor = (df['amount_volume_weighted_log_momentum'] + 
                    df['amount_volume_weighted_intraday_return'] + 
                    df['daily_intraday_volatility'])
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Conditionally Adjust Alpha Factor based on Open-to-Close Return
    alpha_factor = alpha_factor * (1 + 0.5 * df['open_to_close_return'] * (df['open_to_close_return'] > 0) -
                                  0.5 * np.abs(df['open_to_close_return']) * (df['open_to_close_return'] < 0))
    
    # Introduce High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    alpha_factor = alpha_factor * (1 + 0.2 * (df['high_low_ratio'] - 1) * (df['high_low_ratio'] > 1.05) -
                                  0.2 * (1 - df['high_low_ratio']) * (df['high_low_ratio'] < 0.95))
    
    # Incorporate Opening Gap
    df['opening_gap'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    alpha_factor = alpha_factor * (1 + 0.3 * df['opening_gap'] * (df['opening_gap'] > 0) -
                                  0.3 * np.abs(df['opening_gap']) * (df['opening_gap'] < 0))
    
    # Incorporate Price Range
    df['price_range'] = df['high'] - df['low']
    price_range_adjustment = 0.1 * (df['price_range'] / df['close'])
    alpha_factor = alpha_factor * (1 + price_range_adjustment * (df['price_range'] > 0.1 * df['close']) -
                                  price_range_adjustment * (df['price_range'] < 0.1 * df['close']))
    
    # Incorporate Overnight Volatility
    df['overnight_volatility'] = np.abs(df['open'] - df['close'].shift(1))
    overnight_volatility_adjustment = 0.1 * df['overnight_volatility']
    alpha_factor = alpha_factor * (1 + overnight_volatility_adjustment * (df['overnight_volatility'] > 0.1 * df['close'].shift(1)) -
                                  overnight_volatility_adjustment * (df['overnight_volatility'] < 0.1 * df['close'].shift(1)))
    
    return alpha_factor
```
