```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Average High-Low Range over a Lookback Period (e.g., 14 days)
    lookback_window = 14
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=lookback_window).mean()
    
    # Calculate Deviation from Average High-Low Range
    df['deviation_from_avg'] = df['high_low_range'] - df['avg_high_low_range']
    
    # Identify Positive and Negative Deviations
    def momentum_score(deviation):
        if deviation > 0:
            return 1
        else:
            return -1
    
    df['momentum_score'] = df['deviation_from_avg'].apply(momentum_score)
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Weight by Close-to-Open Return
    df['weighted_momentum_score'] = df['momentum_score'] * df['close_to_open_return']
    
    # Final Alpha Factor: Weighted Momentum Score
    alpha_factor = df['weighted_momentum_score']
    
    return alpha_factor
```
