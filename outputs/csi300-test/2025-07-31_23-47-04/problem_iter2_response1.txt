```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    long_term_price_change = df['close'].pct_change(20)
    short_term_price_change = df['close'].pct_change(5)

    # Calculate Volume Ratio
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume_20_days

    # Calculate Volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()

    # Adjust Price Momentum by Volume Ratio
    adjusted_long_term_momentum = long_term_price_change * volume_ratio
    adjusted_short_term_momentum = short_term_price_change * volume_ratio

    # Factor Calculation
    factor = adjusted_short_term_momentum - adjusted_long_term_momentum

    # Final Factor Adjustment
    final_factor = factor * volatility

    return final_factor
```
