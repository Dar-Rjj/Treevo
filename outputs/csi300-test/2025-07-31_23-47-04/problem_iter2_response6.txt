```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Returns
    df['log_return'] = np.log(df['close']).diff()
    
    # Calculate 2-Period and 5-Period Log Returns
    df['log_return_2d'] = df['log_return'].rolling(window=2).sum()
    df['log_return_5d'] = df['log_return'].rolling(window=5).sum()
    
    # Calculate Realized Volatility (Standard Deviation over a Lookback Period of 20 days)
    df['realized_volatility'] = df['log_return'].rolling(window=20).std()
    
    # Adjust Returns by Realized Volatility
    df['log_return_2d_adjusted'] = df['log_return_2d'] / df['realized_volatility']
    df['log_return_5d_adjusted'] = df['log_return_5d'] / df['realized_volatility']
    
    # Calculate Average Volume over the last 3 days
    df['average_volume'] = df['volume'].rolling(window=3).mean()
    
    # Final Alpha Factor: Adjusted Returns Weighted by Volume
    df['alpha_factor'] = (df['log_return_2d_adjusted'] * df['average_volume']) + \
                         (df['log_return_5d_adjusted'] * df['average_volume'])
    
    return df['alpha_factor'].dropna()
```
