```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['High-Low_Spread'] = df['high'] - df['low']
    df['Close-Open_Spread'] = df['close'] - df['open']
    df['Intraday_Momentum'] = df['High-Low_Spread'] - df['Close-Open_Spread']

    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = df['amount'] / df['volume']

    # Determine Volume Synchronization
    df['Log_Volume_Change'] = np.log(df['volume']) - np.log(df['volume'].shift(1))
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['Volume_Synchronization'] = df['Log_Volume_Change'] * df['Log_Return']

    # Calculate Intraday Volatility
    df['Abs_High-Low_Spread'] = abs(df['high'] - df['low'])
    df['Abs_Close-Open_Spread'] = abs(df['close'] - df['open'])
    df['Intraday_Volatility'] = (df['Abs_High-Low_Spread'] + df['Abs_Close-Open_Spread']) / 2

    # Combine Indicators
    df['Combined_Indicator'] = (df['Intraday_Momentum'] * df['Volume_Synchronization'] + 
                                df['VWAP']) / (df['Intraday_Volatility'] + 1e-6)

    # Ensure Positive Values
    df['Combined_Indicator'] = df['Combined_Indicator'].apply(lambda x: max(x, 0))

    return df['Combined_Indicator']
```
