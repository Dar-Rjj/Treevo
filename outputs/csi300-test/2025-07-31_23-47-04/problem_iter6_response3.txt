import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Subtract Close to Open Spread
    close_open_spread = df['close'] - df['open']
    intraday_momentum = high_low_spread - close_open_spread
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Determine Volume Synchronization
    log_volume_change = np.log(df['volume']).diff()
    log_return = np.log(df['close']).diff()
    volume_synchronization = log_volume_change * log_return
    
    # Integrate Price and Volume Dynamics
    integrated_indicator = intraday_momentum * volume_synchronization + vwap
    
    # Combine Integrated Indicator with Recent Trend
    short_term_price_change = df['close'] - df['close'].shift(5)
    final_alpha_factor = integrated_indicator * short_term_price_change
    
    return final_alpha_factor
