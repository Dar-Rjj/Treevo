```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_momentum = high_low_spread - close_open_spread
    
    # Calculate Volume Weighted Average Price
    volume_weighted_avg_price = df['amount'] / df['volume']
    
    # Determine Volume Synchronization
    log_volume_change = np.log(df['volume']) - np.log(df['volume'].shift(1))
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    volume_synchronization = log_volume_change * log_return
    
    # Calculate Rolling Volatility
    daily_returns = df['close'].pct_change()
    rolling_volatility = daily_returns.rolling(window=20).std()
    
    # Adjust Indicators for Volatility
    intraday_momentum_adjusted = intraday_momentum / rolling_volatility
    volume_synchronization_adjusted = volume_synchronization / rolling_volatility
    vw_ap_adjusted = volume_weighted_avg_price / rolling_volatility
    
    # Combine Adjusted Indicators
    factor_value = (intraday_momentum_adjusted * volume_synchronization_adjusted) + vw_ap_adjusted
    
    return factor_value
```
