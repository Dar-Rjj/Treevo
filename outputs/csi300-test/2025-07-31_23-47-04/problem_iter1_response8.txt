```python
def heuristics_v2(df):
    # Calculate Close Price Change
    df['close_change'] = df['close'].diff()
    
    # Calculate 2-Day Close Price Acceleration
    df['close_acceleration'] = df['close'].diff().diff()
    
    # Calculate Average Volume
    df['avg_volume'] = df['volume'].rolling(window=3).mean()
    
    # Adjust by Average Volume
    df['adjusted_close_acceleration'] = df['close_change'] / df['avg_volume']
    
    return df['adjusted_close_acceleration']
```
