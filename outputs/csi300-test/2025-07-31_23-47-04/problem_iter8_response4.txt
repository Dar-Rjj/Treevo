import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()

    # Calculate Intraday Momentum
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_momentum = high_low_spread - close_open_spread

    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']

    # Determine Volume Synchronization
    log_volume_change = np.log(df['volume'] / df['volume'].shift(1))
    log_return = np.log(df['close'] / df['close'].shift(1))

    # Integrate Price and Volume Dynamics
    integrated_momentum = intraday_momentum * log_return
    integrated_indicator = integrated_momentum + vwap

    # Enhance Factor with Intraday Price Trend
    intraday_trend = (df['high'] - df['low']) / df['low']
    combined_indicator = intraday_trend * integrated_indicator

    # Incorporate Volatility Measures
    historical_volatility = df['close'].rolling(window=5).std()
    final_alpha_factor = combined_indicator / historical_volatility

    return final_alpha_factor
