```python
def heuristics_v2(data):
    import pandas as pd
    
    # Calculate Daily Range
    data['daily_range'] = data['high'] - data['low']
    
    # Momentum with Respect to Open
    data['open_momentum'] = (data['open'] - data['open'].shift(1)) / data['open'].shift(1)
    
    # Combine with Daily Range
    data['alpha_factor'] = (data['daily_range'] / data['open'].shift(1)) * data['open_momentum']
    
    return data['alpha_factor']
```
