import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Subtract Close to Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Intraday Momentum
    intraday_momentum = high_low_spread - close_open_spread
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Compute Logarithmic Volume Change
    log_volume_change = np.log(df['volume']) - np.log(df['volume'].shift(1))
    
    # Calculate Logarithmic Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Determine Volume Synchronization
    volume_sync = log_volume_change * log_return
    
    # Integrate Price and Volume Dynamics
    integrated_indicator = intraday_momentum * log_return + vwap
    
    # Enhance Factor with Intraday Price Trend
    intraday_trend = (df['high'] - df['low']) / df['low']
    enhanced_factor = (integrated_indicator + vwap) * intraday_trend
    
    # Final Alpha Factor
    final_alpha_factor = enhanced_factor
    
    return final_alpha_factor
