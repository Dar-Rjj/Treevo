```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_momentum = high_low_spread - close_open_spread
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Determine Volume Synchronization
    df['log_volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Integrate Price and Volume Dynamics
    integrated_indicator = intraday_momentum * df['log_return']
    combined_indicator = integrated_indicator + vwap
    
    # Enhance Factor with Intraday and Relative Strength
    refined_intraday_trend = (df['high'] - df['low']) / df['low']
    rolling_avg_close = df['close'].rolling(window=20).mean()
    relative_strength = df['close'] / rolling_avg_close
    
    # Final Alpha Factor
    final_alpha_factor = refined_intraday_trend * combined_indicator * relative_strength
    
    return final_alpha_factor
```
