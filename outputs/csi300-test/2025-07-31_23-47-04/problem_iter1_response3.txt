```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return using High and Low prices
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Average Volume over the past 20 days
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Calculate Relative Volume
    df['relative_volume'] = df['volume'] / df['avg_volume_20']
    
    # Calculate Close-to-Previous Close Return
    df['prev_close'] = df['close'].shift(1)
    df['close_to_prev_close_return'] = (df['close'] - df['prev_close']) / df['prev_close']
    
    # Final Factor Calculation
    df['factor_value'] = df['intraday_return'] * df['relative_volume'] * df['close_to_prev_close_return']
    
    return df['factor_value'].dropna()
```
