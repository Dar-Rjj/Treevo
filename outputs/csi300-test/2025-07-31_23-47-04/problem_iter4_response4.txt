```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Subtract Close to Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Dynamic Intraday Momentum
    intraday_momentum = high_low_spread - close_open_spread
    
    # Apply Dynamic Lookback Period
    short_term_momentum = intraday_momentum.rolling(window=5).mean()
    long_term_momentum = intraday_momentum.rolling(window=20).mean()
    dynamic_momentum = (short_term_momentum + long_term_momentum) / 2
    
    # Calculate Intraday Volatility (Average True Range)
    true_range = np.maximum(np.maximum(df['high'] - df['low'], np.abs(df['high'] - df['close'].shift(1))), 
                            np.abs(df['low'] - df['close'].shift(1)))
    atr = true_range.rolling(window=14).mean()
    volatility_adjusted_momentum = dynamic_momentum / atr
    
    # Calculate Volume Weighted Average Price
    vwap = df['amount'] / df['volume']
    
    # Determine Volume Synchronization
    log_volume_change = np.log(df['volume'] / df['volume'].shift(1))
    log_return = np.log(df['close'] / df['close'].shift(1))
    volume_synchronization = log_volume_change * log_return
    
    # Combine Indicators
    combined_indicator = volatility_adjusted_momentum * volume_synchronization + vwap
    
    # Integrate Multiple Price and Volume Derivatives
    long_term_price_velocity = (df['close'] - df['close'].shift(20)) / 20
    short_term_price_velocity = (df['close'] - df['close'].shift(5)) / 5
    price_velocity = (long_term_price_velocity + short_term_price_velocity) / 2
    
    long_term_volume_velocity = (df['volume'] - df['volume'].shift(20)) / 20
    short_term_volume_velocity = (df['volume'] - df['volume'].shift(5)) / 5
    volume_velocity = (long_term_volume_velocity + short_term_volume_velocity) / 2
    
    # Final Factor Calculation
    final_factor = combined_indicator + price_velocity + volume_velocity
    
    return final_factor
```
