```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate 10-Day Average of High-Low Spread
    df['10_day_avg_high_low_spread'] = df['high_low_spread'].rolling(window=10).mean()
    
    # Calculate 50-Day Average of High-Low Spread
    df['50_day_avg_high_low_spread'] = df['high_low_spread'].rolling(window=50).mean()
    
    # Calculate Difference Between Averages
    df['diff_10_50_avg_high_low_spread'] = df['10_day_avg_high_low_spread'] - df['50_day_avg_high_low_spread']
    
    # Calculate 200-Day Average of High-Low Spread
    df['200_day_avg_high_low_spread'] = df['high_low_spread'].rolling(window=200).mean()
    
    # Calculate Long-Term Momentum Score
    df['long_term_momentum_score'] = df['diff_10_50_avg_high_low_spread'] * df['200_day_avg_high_low_spread']
    
    # Adjust for Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['long_term_momentum_score'] * df['volume']
    
    return df['volume_weighted_momentum']
```
