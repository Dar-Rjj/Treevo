```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Historical Volatility (Standard Deviation over a Fixed Window, e.g., 20 days)
    df['volatility'] = df['daily_return'].rolling(window=20).std()
    
    # Adjust Lookback Period Based on Volatility
    df['lookback_period'] = np.where(df['volatility'] > df['volatility'].mean(), 10, 30)
    
    # Calculate Average Volume over the Dynamic Lookback Period
    df['average_volume'] = df['volume'].rolling(window=df['lookback_period']).mean()
    
    # Calculate Relative Volume
    df['relative_volume'] = df['volume'] / df['average_volume']
    
    # Calculate Close-to-Previous Close Return
    df['close_to_prev_close_return'] = df['close'].pct_change()
    
    # Combine Intraday Return, Relative Volume, and Price Trend
    df['combined_factor'] = df['intraday_return'] * df['relative_volume'] * df['close_to_prev_close_return']
    
    # Calculate Price Change Velocity
    df['price_velocity'] = df['close'].pct_change(periods=5) / df['close'].rolling(window=5).mean()
    
    # Calculate Volume Spike
    df['volume_spike'] = np.where(df['volume'] > 2 * df['volume'].shift(1), 1, 0)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factor'] * df['price_velocity'] * df['volume_spike']
    
    return df['alpha_factor']
```
