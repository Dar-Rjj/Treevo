```python
import pandas as pd

def heuristics_v2(df, lookback_period=14):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Separate Positive and Negative Changes
    df['positive_change'] = df['price_change'].apply(lambda x: x if x > 0 else 0)
    df['negative_change'] = df['price_change'].apply(lambda x: abs(x) if x < 0 else 0)
    
    # Weight Each Change by Volume
    df['weighted_positive_change'] = df['positive_change'] * df['volume']
    df['weighted_negative_change'] = df['negative_change'] * df['volume']
    
    # Calculate Average Gain and Loss Over Lookback Period
    avg_gain = df['weighted_positive_change'].rolling(window=lookback_period).mean() / df['volume'].rolling(window=lookback_period).mean()
    avg_loss = (df['weighted_negative_change'].rolling(window=lookback_period).mean() / df['volume'].rolling(window=lookback_period).mean()).abs()
    
    # Compute RSIVA Factor
    rs = avg_gain / avg_loss
    rsiva = 100 - (100 / (1 + rs))
    
    # Clamp values between 0 and 100
    rsiva = rsiva.clip(lower=0, upper=100)
    
    return rsiva
```
