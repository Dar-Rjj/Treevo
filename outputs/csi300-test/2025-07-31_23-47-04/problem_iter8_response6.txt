import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Compute Close-Open Spread
    close_open_spread = df['close'] - df['open']

    # Calculate Intraday Momentum
    intraday_momentum = high_low_spread - close_open_spread

    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']

    # Compute Logarithmic Volume Change
    log_volume_change = np.log(df['volume'] / df['volume'].shift(1))

    # Calculate Logarithmic Return
    log_return = np.log(df['close'] / df['close'].shift(1))

    # Combine Intraday Momentum and Volume Synchronization
    integrated_momentum = intraday_momentum * log_return

    # Combine Integrated Momentum and VWAP
    combined_indicator = integrated_momentum + vwap

    # Calculate Intraday Trend
    intraday_trend = (df['high'] - df['low']) / df['low']

    # Final Alpha Factor
    final_alpha_factor = intraday_trend * combined_indicator

    return final_alpha_factor
