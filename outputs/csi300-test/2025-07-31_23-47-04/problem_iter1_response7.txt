```python
def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Momentum with Respect to Open
    open_momentum = df['open'] - df['open'].shift(1)
    
    # Combine with Daily Range
    factor = (daily_range / df['open'].shift(1)) * open_momentum
    
    return factor
```
