import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic Intraday Momentum
    df['High-Low_Spread'] = df['high'] - df['low']
    df['Close-Open_Spread'] = df['close'] - df['open']
    df['Intraday_Momentum'] = df['High-Low_Spread'] - df['Close-Open_Spread']
    df['Intraday_Momentum_5d'] = df['Intraday_Momentum'].rolling(window=5).mean()
    df['Intraday_Momentum_20d'] = df['Intraday_Momentum'].rolling(window=20).mean()
    df['Dynamic_Intraday_Momentum'] = (df['Intraday_Momentum_5d'] + df['Intraday_Momentum_20d']) / 2

    # Calculate Intraday Volatility
    df['True_Range'] = df[['high', 'low']].subtract(df['close'].shift(1), axis=0).abs().max(axis=1)
    df['Average_True_Range'] = df['True_Range'].rolling(window=14).mean()
    df['Intraday_Volatility_Adjusted'] = df['Dynamic_Intraday_Momentum'] / df['Average_True_Range']

    # Calculate Volume Weighted Average Price
    df['VWAP'] = df['amount'] / df['volume']

    # Determine Volume Synchronization
    df['Log_Volume_Change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))

    # Combine Indicators
    df['Volume_Synchronization'] = df['Log_Volume_Change'] * df['Log_Return']
    df['Combined_Factor'] = df['Intraday_Volatility_Adjusted'] * df['Volume_Synchronization'] + df['VWAP']

    # Integrate Multiple Price and Volume Derivatives
    df['Long-term_Price_Velocity'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    df['Short-term_Price_Velocity'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['Price_Velocity'] = (df['Long-term_Price_Velocity'] + df['Short-term_Price_Velocity']) / 2

    df['Long-term_Volume_Velocity'] = (df['volume'] - df['volume'].shift(20)) / df['volume'].shift(20)
    df['Short-term_Volume_Velocity'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    df['Volume_Velocity'] = (df['Long-term_Volume_Velocity'] + df['Short-term_Volume_Velocity']) / 2

    # Final Factor Calculation
    df['Final_Factor'] = df['Combined_Factor'] + df['Price_Velocity'] + df['Volume_Velocity']

    return df['Final_Factor'].dropna()
