```python
import pandas as pd

def heuristics_v2(df):
    # Compute Long-term Price Change
    long_term_change = (df['close'] / df['close'].shift(20)) - 1
    
    # Compute Short-term Price Change
    short_term_change = (df['close'] / df['close'].shift(5)) - 1
    
    # Compute Average Volume over 20 days
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    
    # Compute Today's Volume
    today_volume = df['volume']
    
    # Calculate Volume Ratio
    volume_ratio = today_volume / avg_volume_20
    
    # Adjust Price Momentum by Volume Ratio
    adjusted_long_term_change = long_term_change * volume_ratio
    adjusted_short_term_change = short_term_change * volume_ratio
    
    # Factor Calculation
    factor = adjusted_short_term_change - adjusted_long_term_change
    
    return factor
```
