```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily price-volume product using close price and volume
    df['price_volume'] = df['close'] * df['volume']
    
    # Calculate the 10-day and 20-day sum of the price-volume product
    df['10_day_pv_sum'] = df['price_volume'].rolling(window=10).sum()
    df['20_day_pv_sum'] = df['price_volume'].rolling(window=20).sum()
    
    # Subtract the 10-day sum from the 20-day sum
    df['momentum_change'] = df['20_day_pv_sum'] - df['10_day_pv_sum']
    
    # Standardize the momentum by dividing by the 20-day sum
    df['cumulative_price_volume_momentum'] = df['momentum_change'] / df['20_day_pv_sum']
    
    # Return the alpha factor as a pandas Series
    return df['cumulative_price_volume_momentum'].dropna()
```
