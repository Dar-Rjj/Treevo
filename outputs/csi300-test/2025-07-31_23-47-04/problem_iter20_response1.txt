```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Intraday Momentum
    intraday_momentum = high_low_spread - close_open_spread
    
    # Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Logarithmic Volume Change
    log_volume_change = np.log(df['volume'] / df['volume'].shift(1))
    
    # Logarithmic Return
    log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Integrated Indicator
    integrated_indicator = intraday_momentum * log_return
    
    # Combined Indicator
    combined_indicator = integrated_indicator + vwap
    
    # Intraday Trend
    intraday_trend = (df['high'] - df['low']) / df['low']
    
    # Relative Strength
    rolling_avg_close = df['close'].rolling(window=30).mean()
    relative_strength = df['close'] / rolling_avg_close
    
    # Advanced Realized Volatility
    advanced_realized_volatility = np.sqrt(((np.log(df['close'] / df['close'].shift(1)))**2).rolling(window=20).mean())
    
    # Volume Volatility
    volume_volatility = np.sqrt(((df['volume'] - df['volume'].shift(1))**2).rolling(window=20).mean())
    
    # Close-to-Open Ratio
    close_to_open_ratio = df['close'] / df['open'].shift(-1)
    
    # Final Alpha Factor
    final_alpha_factor = intraday_trend * combined_indicator * relative_strength * advanced_realized_volatility * volume_volatility * close_to_open_ratio
    
    return final_alpha_factor
```
