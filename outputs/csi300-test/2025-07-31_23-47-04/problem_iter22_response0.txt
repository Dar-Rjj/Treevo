```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_momentum = high_low_spread - close_open_spread
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Determine Volume Synchronization
    log_volume_change = np.log(df['volume'] / df['volume'].shift(1))
    log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Integrate Price and Volume Dynamics
    integrated_indicator = intraday_momentum * log_return
    combined_indicator = integrated_indicator + vwap
    
    # Enhance Factor with Short-Term Intraday and Relative Strength
    short_term_intraday_trend = (df['high'] - df['low']) / df['low']
    rolling_avg_close = df['close'].rolling(window=10).mean()
    short_term_relative_strength = df['close'] / rolling_avg_close
    
    # Incorporate Realized Volatility
    realized_volatility = (df['close'].diff().fillna(0) ** 2).rolling(window=10).mean().apply(np.sqrt)
    
    # Incorporate Liquidity
    avg_daily_volume = df['volume'].rolling(window=10).mean()
    
    # Incorporate Market Sentiment
    close_to_open_ratio = df['close'] / df['open'].shift(-1)
    
    # Final Alpha Factor
    final_alpha_factor = (
        short_term_intraday_trend *
        combined_indicator *
        short_term_relative_strength *
        realized_volatility *
        avg_daily_volume *
        close_to_open_ratio
    )
    
    return final_alpha_factor
```
