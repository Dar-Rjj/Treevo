import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Intraday Momentum
    intraday_momentum = high_low_spread - close_open_spread
    
    # Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Logarithmic Volume Change
    log_volume_change = np.log(df['volume']).diff()
    
    # Logarithmic Return
    log_return = np.log(df['close']).diff()
    
    # Integrated Indicator
    integrated_indicator = intraday_momentum * log_return
    
    # Combined Indicator
    combined_indicator = integrated_indicator + vwap
    
    # Intraday Trend
    intraday_trend = (df['high'] - df['low']) / df['low']
    
    # Final Alpha Factor
    final_alpha_factor = intraday_trend * combined_indicator
    
    return final_alpha_factor
