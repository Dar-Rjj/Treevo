```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']

    # Subtract Close to Open Spread
    df['Close_Open_Spread'] = df['close'] - df['open']
    
    # Calculate Intraday Momentum
    df['Intraday_Momentum'] = df['High_Low_Spread'] - df['Close_Open_Spread']

    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = df['amount'] / df['volume']

    # Compute Logarithmic Volume Change
    df['Log_Volume_Change'] = np.log(df['volume'] / df['volume'].shift(1))

    # Calculate Logarithmic Return
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))

    # Combine Intraday Momentum and Volume Synchronization
    df['Integrated_Momentum'] = df['Intraday_Momentum'] * df['Log_Return']

    # Add VWAP
    df['Combined_Indicator'] = df['Integrated_Momentum'] + df['VWAP']

    # Calculate Intraday Trend
    df['Intraday_Trend'] = (df['high'] - df['low']) / df['low']

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Intraday_Trend'] * df['Combined_Indicator']

    return df['Final_Alpha_Factor']
```
