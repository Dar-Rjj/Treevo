```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving averages of the closing price
    ma_5d = df['close'].rolling(window=5).mean()
    ma_20d = df['close'].rolling(window=20).mean()
    
    # Calculate the 5-day and 20-day rolling standard deviations of the closing price as measures of volatility
    vol_5d = df['close'].rolling(window=5).std()
    vol_20d = df['close'].rolling(window=20).std()
    
    # Calculate the 5-day and 20-day high-low ranges
    high_low_diff_5d = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    high_low_diff_20d = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate the 5-day and 20-day change in volume
    volume_change_5d = df['volume'].pct_change(periods=5)
    volume_change_20d = df['volume'].pct_change(periods=20)
    
    # Calculate the 10-day and 30-day relative strength index (RSI)
    delta = df['close'].diff()
    gain_10d = (delta.where(delta > 0, 0)).rolling(window=10).mean()
    loss_10d = (-delta.where(delta < 0, 0)).rolling(window=10).mean()
    rs_10d = gain_10d / loss_10d
    rsi_10d = 100 - (100 / (1 + rs_10d))
    
    gain_30d = (delta.where(delta > 0, 0)).rolling(window=30).mean()
    loss_30d = (-delta.where(delta < 0, 0)).rolling(window=30).mean()
    rs_30d = gain_30d / loss_30d
    rsi_30d = 100 - (100 / (1 + rs_30d))
    
    # Create a factor that considers the relationship between the current close price and the 5-day and 20-day moving averages,
    # adjusted by the 5-day and 20-day high-low ranges and 5-day and 20-day volatility
    factor = (df['close'] - ma_5d) / (high_low_diff_5d + vol_5d + 1e-7) + (df['close'] - ma_20d) / (high_low_diff_20d + vol_20d + 1e-7)
    
    # Incorporate the 5-day and 20-day volume changes and 10-day and 30-day RSI into the factor
    factor += volume_change_5d + volume_change_20d + rsi_10d + rsi_30d
    
    return factor
```
