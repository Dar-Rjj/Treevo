```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive moving average window based on volatility
    vol_10d = df['close'].rolling(window=10).std()
    ma_window_short = (vol_10d / vol_10d.mean() * 5).round().astype(int).clip(lower=3, upper=7)
    ma_window_medium = (vol_10d / vol_10d.mean() * 10).round().astype(int).clip(lower=5, upper=15)
    ma_window_long = (vol_10d / vol_10d.mean() * 20).round().astype(int).clip(lower=10, upper=30)

    ma_short = df['close'].rolling(window=ma_window_short, min_periods=1).mean()
    ma_medium = df['close'].rolling(window=ma_window_medium, min_periods=1).mean()
    ma_long = df['close'].rolling(window=ma_window_long, min_periods=1).mean()

    # Adaptive high-low range
    high_low_diff_short = (df['high'] - df['low']).rolling(window=ma_window_short, min_periods=1).mean()
    high_low_diff_medium = (df['high'] - df['low']).rolling(window=ma_window_medium, min_periods=1).mean()
    high_low_diff_long = (df['high'] - df['low']).rolling(window=ma_window_long, min_periods=1).mean()

    # Adaptive volume change
    volume_change_short = df['volume'].pct_change(periods=ma_window_short)
    volume_change_medium = df['volume'].pct_change(periods=ma_window_medium)
    volume_change_long = df['volume'].pct_change(periods=ma_window_long)

    # Adaptive RSI
    delta = df['close'].diff()
    gain_short = (delta.where(delta > 0, 0)).rolling(window=ma_window_short*2, min_periods=1).mean()
    loss_short = (-delta.where(delta < 0, 0)).rolling(window=ma_window_short*2, min_periods=1).mean()
    rs_short = gain_short / loss_short.replace(0, 1e-7)
    rsi_short = 100 - (100 / (1 + rs_short))

    gain_medium = (delta.where(delta > 0, 0)).rolling(window=ma_window_medium*2, min_periods=1).mean()
    loss_medium = (-delta.where(delta < 0, 0)).rolling(window=ma_window_medium*2, min_periods=1).mean()
    rs_medium = gain_medium / loss_medium.replace(0, 1e-7)
    rsi_medium = 100 - (100 / (1 + rs_medium))

    gain_long = (delta.where(delta > 0, 0)).rolling(window=ma_window_long*2, min_periods=1).mean()
    loss_long = (-delta.where(delta < 0, 0)).rolling(window=ma_window_long*2, min_periods=1).mean()
    rs_long = gain_long / loss_long.replace(0, 1e-7)
    rsi_long = 100 - (100 / (1 + rs_long))

    # Price deviation and normalize with dynamic volatility and range
    price_deviation_short = df['close'] - ma_short
    price_deviation_medium = df['close'] - ma_medium
    price_deviation_long = df['close'] - ma_long

    normalized_price_deviation_short = price_deviation_short / (high_low_diff_short + vol_10d + 1e-7)
    normalized_price_deviation_medium = price_deviation_medium / (high_low_diff_medium + vol_10d + 1e-7)
    normalized_price_deviation_long = price_deviation_long / (high_low_diff_long + vol_10d + 1e-7)

    # Combine the factors
    factor = (normalized_price_deviation_short + normalized_price_deviation_medium + normalized_price_deviation_long +
              volume_change_short + volume_change_medium + volume_change_long +
              rsi_short + rsi_medium + rsi_long)

    return factor
```
