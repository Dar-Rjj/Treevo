```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day moving average of the closing price
    ma_5d = df['close'].rolling(window=5).mean()
    
    # Calculate the 5-day rolling standard deviation of the closing price as a measure of volatility
    vol_5d = df['close'].rolling(window=5).std()
    
    # Calculate the 5-day high-low range
    high_low_diff_5d = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    
    # Calculate the 5-day change in volume
    volume_change_5d = df['volume'].pct_change(periods=5)
    
    # Calculate the 10-day relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=10).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=10).mean()
    rs = gain / loss
    rsi_10d = 100 - (100 / (1 + rs))
    
    # Create a factor that considers the relationship between the current close price and the 5-day moving average,
    # adjusted by the 5-day high-low range and 5-day volatility
    factor = (df['close'] - ma_5d) / (high_low_diff_5d + vol_5d + 1e-7)
    
    # Incorporate the 5-day volume change and 10-day RSI into the factor
    factor += volume_change_5d + rsi_10d
    
    return factor
```
