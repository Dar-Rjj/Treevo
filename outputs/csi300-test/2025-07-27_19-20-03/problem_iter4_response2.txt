```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the relative change in amount and volume
    amount_change = df['amount'].pct_change().fillna(0)
    volume_change = df['volume'].pct_change().fillna(0)

    # Calculate the ratio of close to open, adjusted by the change in amount and volume
    factor_value = (df['close'] / df['open']) * (1 + amount_change) * (1 + volume_change)

    # Incorporate short-term momentum
    short_momentum = df['close'].pct_change(periods=5).fillna(0)
    factor_value *= (1 + short_momentum)

    # Incorporate long-term momentum
    long_momentum = df['close'].pct_change(periods=20).fillna(0)
    factor_value *= (1 + 0.5 * long_momentum)

    # Incorporate short-term volatility
    short_volatility = df['close'].rolling(window=20).std().fillna(0)
    factor_value /= (1 + short_volatility)

    # Incorporate long-term volatility
    long_volatility = df['close'].rolling(window=50).std().fillna(0)
    factor_value /= (1 + 0.5 * long_volatility)

    # Incorporate relative strength with adaptive windows
    rs_short = df['close'].rolling(window=20).mean() / df['close'].rolling(window=50).mean()
    rs_long = df['close'].rolling(window=50).mean() / df['close'].rolling(window=100).mean()
    factor_value *= (rs_short + rs_long) / 2

    return factor_value
```
