```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate multi-period momentum
    short_momentum = df['close'].pct_change(periods=5).fillna(0)
    medium_momentum = df['close'].pct_change(periods=20).fillna(0)
    long_momentum = df['close'].pct_change(periods=60).fillna(0)

    # Incorporate adaptive volatility
    vol_5d = df['close'].rolling(window=5).std().fillna(0)
    vol_20d = df['close'].rolling(window=20).std().fillna(0)
    vol_60d = df['close'].rolling(window=60).std().fillna(0)
    adaptive_volatility = (vol_5d + vol_20d + vol_60d) / 3

    # Calculate the 5-day and 20-day high-low ranges
    high_low_diff_5d = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    high_low_diff_20d = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()

    # Calculate the 5-day and 20-day change in volume
    volume_change_5d = df['volume'].pct_change(periods=5).fillna(0)
    volume_change_20d = df['volume'].pct_change(periods=20).fillna(0)

    # Calculate the 10-day and 20-day relative strength index (RSI)
    delta = df['close'].diff()
    gain_10d = (delta.where(delta > 0, 0)).rolling(window=10).mean()
    loss_10d = (-delta.where(delta < 0, 0)).rolling(window=10).mean()
    rs_10d = gain_10d / loss_10d
    rsi_10d = 100 - (100 / (1 + rs_10d))

    gain_20d = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss_20d = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs_20d = gain_20d / loss_20d
    rsi_20d = 100 - (100 / (1 + rs_20d))

    # Create a factor that considers the relationship between the current close price and the 5-day and 20-day moving averages,
    # adjusted by the 5-day and 20-day high-low ranges and adaptive volatility
    ma_5d = df['close'].rolling(window=5).mean()
    ma_20d = df['close'].rolling(window=20).mean()
    price_deviation_5d = df['close'] - ma_5d
    price_deviation_20d = df['close'] - ma_20d
    normalized_price_deviation_5d = price_deviation_5d / (high_low_diff_5d + adaptive_volatility + 1e-7)
    normalized_price_deviation_20d = price_deviation_20d / (high_low_diff_20d + adaptive_volatility + 1e-7)

    # Incorporate the 5-day and 20-day volume changes and RSI with dynamic windows into the factor
    dynamic_rsi = (rsi_10d + rsi_20d) / 2
    factor = normalized_price_deviation_5d + normalized_price_deviation_20d + volume_change_5d + volume_change_20d + dynamic_rsi + short_momentum + medium_momentum + long_momentum

    return factor
```
