```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive moving average window based on 10-day volatility
    vol_10d = df['close'].rolling(window=10).std()
    ma_window = (vol_10d / vol_10d.mean() * 5).round().astype(int).clip(lower=3, upper=7)
    ma = df['close'].rolling(window=ma_window, min_periods=1).mean()

    # Adaptive high-low range
    high_low_diff = (df['high'] - df['low']).rolling(window=ma_window, min_periods=1).mean()

    # Adaptive volume change
    volume_change = df['volume'].pct_change(periods=ma_window).fillna(0)

    # Adaptive RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=ma_window*2, min_periods=1).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=ma_window*2, min_periods=1).mean()
    rs = gain / (loss.replace(0, 1e-7))
    rsi = 100 - (100 / (1 + rs))

    # Momentum indicators
    short_momentum = df['close'].pct_change(periods=5).fillna(0)
    medium_momentum = df['close'].pct_change(periods=20).fillna(0)
    long_momentum = df['close'].pct_change(periods=60).fillna(0)

    # Price deviation normalized with adaptive high-low range and 10-day volatility
    price_deviation = df['close'] - ma
    normalized_price_deviation = price_deviation / (high_low_diff + vol_10d + 1e-7)

    # Combine the factors
    factor = (
        normalized_price_deviation 
        + volume_change 
        + rsi 
        + short_momentum 
        + medium_momentum 
        + long_momentum
    )

    return factor
```
