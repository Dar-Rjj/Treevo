```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the relative change in amount and volume
    amount_change = df['amount'].pct_change().fillna(0)
    volume_change = df['volume'].pct_change().fillna(0)

    # Calculate the ratio of close to open, adjusted by the change in amount and volume
    price_ratio = df['close'] / df['open']

    # Calculate the adjusted high-low range
    hl_range = (df['high'] - df['low']) / df['open']

    # Combine the price ratio, amount change, volume change, and high-low range
    factor_value = (price_ratio * (1 + amount_change) * (1 + volume_change) * (1 + hl_range))
    
    return factor_value
```
