```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving averages of the close price for momentum
    ma_5d = df['close'].rolling(window=5).mean()
    ma_20d = df['close'].rolling(window=20).mean()
    
    # Calculate the 5-day and 20-day rolling standard deviations for volatility
    std_5d = df['close'].rolling(window=5).std()
    std_20d = df['close'].rolling(window=20).std()
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ratio of close to open, adjusted by the change in amount and volume
    amount_change = df['amount'].pct_change().fillna(0)
    volume_change = df['volume'].pct_change().fillna(0)
    price_ratio = (df['close'] / df['open']) * (1 + amount_change) * (1 + volume_change)
    
    # Combine the factors into a final alpha factor
    factor_value = (ma_5d - ma_20d) / std_20d * rsi * price_ratio
    
    return factor_value
```
