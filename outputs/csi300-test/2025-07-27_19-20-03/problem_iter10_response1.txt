```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate multi-period momentum
    short_momentum = df['close'].pct_change(periods=5).fillna(0)
    medium_momentum = df['close'].pct_change(periods=20).fillna(0)
    
    # Incorporate adaptive volatility
    vol_5d = df['close'].rolling(window=5).std().fillna(0)
    vol_20d = df['close'].rolling(window=20).std().fillna(0)
    adaptive_volatility = (vol_5d + vol_20d) / 2

    # Calculate the 5-day high-low range
    high_low_diff_5d = (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()).fillna(0)

    # Calculate the 5-day change in volume
    volume_change_5d = df['volume'].pct_change(periods=5).fillna(0)

    # Calculate the 10-day and 20-day relative strength index (RSI)
    delta = df['close'].diff()
    gain_10d = (delta.where(delta > 0, 0)).rolling(window=10).mean().fillna(0)
    loss_10d = (-delta.where(delta < 0, 0)).rolling(window=10).mean().fillna(0)
    rs_10d = gain_10d / loss_10d.replace(0, 1e-7)
    rsi_10d = 100 - (100 / (1 + rs_10d))

    # Create a factor that considers the relationship between the current close price and the 5-day moving average,
    # adjusted by the 5-day high-low range and adaptive volatility
    ma_5d = df['close'].rolling(window=5).mean().fillna(0)
    price_deviation = df['close'] - ma_5d
    normalized_price_deviation = price_deviation / (high_low_diff_5d + adaptive_volatility + 1e-7)

    # Incorporate the 5-day volume change and RSI into the factor
    dynamic_rsi = rsi_10d
    factor = normalized_price_deviation + volume_change_5d + dynamic_rsi + short_momentum + medium_momentum

    return factor
```
