```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day moving average of the closing price for a shorter-term trend
    ma_3d = df['close'].rolling(window=3).mean()
    
    # Calculate the 3-day rolling standard deviation of the closing price as a measure of short-term volatility
    vol_3d = df['close'].rolling(window=3).std()
    
    # Calculate the 3-day high-low range to capture short-term price movement
    high_low_diff_3d = df['high'].rolling(window=3).max() - df['low'].rolling(window=3).min()
    
    # Calculate the 3-day change in volume to capture short-term trading activity
    volume_change_3d = df['volume'].pct_change(periods=3).fillna(0)
    
    # Calculate the 5-day relative strength index (RSI) for a more dynamic momentum indicator
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=5).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=5).mean()
    rs = gain / loss
    rsi_5d = 100 - (100 / (1 + rs))
    
    # Create a factor that considers the relationship between the current close price and the 3-day moving average,
    # adjusted by the 3-day high-low range and 3-day volatility
    factor = (df['close'] - ma_3d) / (high_low_diff_3d + vol_3d + 1e-7)
    
    # Incorporate the 3-day volume change and 5-day RSI into the factor
    factor += volume_change_3d + rsi_5d
    
    return factor
```
