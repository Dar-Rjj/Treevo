```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices over a 5-day period
    high_low_diff_5d = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    
    # Calculate the average of the closing price over the same 5-day period
    avg_close_5d = df['close'].rolling(window=5).mean()
    
    # Create a factor that considers the relationship between the current close price
    # and the 5-day average, adjusted by the 5-day high-low range
    factor = (df['close'] - avg_close_5d) / (high_low_diff_5d + 1e-7)
    
    return factor
```
