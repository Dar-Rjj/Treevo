```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive window for moving average based on 5-day to 10-day range, depending on recent volatility
    vol_5d = df['close'].rolling(window=5).std()
    ma_window = (vol_5d / df['close'].rolling(window=5).std().mean() * 5 + 5).astype(int)
    ma_adaptive = df['close'].rolling(window=ma_window, min_periods=5).mean()

    # 10-day high-low range
    high_low_diff_10d = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()

    # 10-day change in volume
    volume_change_10d = df['volume'].pct_change(periods=10)

    # 14-day relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_14d = 100 - (100 / (1 + rs))

    # Price deviation from adaptive moving average, normalized by 10-day high-low range and 5-day volatility
    price_deviation = df['close'] - ma_adaptive
    normalized_price_deviation = price_deviation / (high_low_diff_10d + vol_5d + 1e-7)

    # Short-term momentum
    short_momentum = df['close'].pct_change(periods=5).fillna(0)
    normalized_short_momentum = short_momentum / (df['close'].rolling(window=5).std() + 1e-7)

    # Long-term momentum
    long_momentum = df['close'].pct_change(periods=20).fillna(0)
    normalized_long_momentum = long_momentum / (df['close'].rolling(window=20).std() + 1e-7)

    # Incorporate the 10-day volume change, 14-day RSI, and both short-term and long-term momentum into the factor
    factor = normalized_price_deviation + volume_change_10d + rsi_14d + normalized_short_momentum + 0.5 * normalized_long_momentum

    return factor
```
