```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, capturing the momentum
    df['momentum'] = df['close'].diff()
    
    # Calculate the 10-day moving average of the closing price to smooth out short-term fluctuations
    df['ma_10'] = df['close'].rolling(window=10).mean()

    # The 'amount' can be seen as a proxy for dollar volume. We calculate the ratio between the amount 
    # and rolling sum (5 days) of the amount to capture recent transaction activity relative to the past
    df['amount_ratio'] = df['amount'] / df['amount'].rolling(window=5).sum()
    
    # Combine the factors: momentum, moving average deviation, and amount ratio
    # The intuition is that stocks with positive momentum, above their moving average, and high recent transaction activity
    # might have a higher chance of increasing in value in the near future.
    alpha_factor = df['momentum'] + (df['close'] - df['ma_10']) + df['amount_ratio']

    return alpha_factor
```
