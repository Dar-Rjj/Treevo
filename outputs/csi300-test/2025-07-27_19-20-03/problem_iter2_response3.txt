def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the relative change in amount and volume
    amount_change = df['amount'].pct_change().fillna(0)
    volume_change = df['volume'].pct_change().fillna(0)

    # Calculate the ratio of close to open, adjusted by the change in amount and volume
    factor_value = (df['close'] / df['open']) * (1 + amount_change) * (1 + volume_change)

    # Incorporate momentum
    momentum = df['close'].pct_change(periods=5).fillna(0)
    factor_value *= (1 + momentum)

    # Incorporate volatility
    volatility = df['close'].rolling(window=20).std().fillna(0)
    factor_value /= (1 + volatility)

    # Incorporate relative strength
    rs = df['close'].rolling(window=20).mean() / df['close'].rolling(window=50).mean()
    factor_value *= rs

    return factor_value
