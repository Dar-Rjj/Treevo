```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the normalized moving average of the volume over a 10-day window
    volume_moving_avg = df['volume'].rolling(window=10).mean() / df['volume']
    
    # Calculate the alpha factor by combining the price difference with the normalized volume
    alpha_factor = (diff_close_open * volume_moving_avg)
    
    return alpha_factor
```
