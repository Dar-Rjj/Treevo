```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate 5-day and 10-day volatility
    vol_5d = df['close'].rolling(window=5).std()
    vol_10d = df['close'].rolling(window=10).std()

    # Determine the adaptive window for moving average based on 5-day to 10-day range, depending on recent volatility
    ma_window = (vol_5d / vol_10d * 5 + 5).astype(int)
    ma_adaptive = df['close'].rolling(window=ma_window, min_periods=5).mean()

    # Calculate the 10-day high-low range
    high_low_diff_10d = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()

    # Calculate the 10-day change in volume
    volume_change_10d = df['volume'].pct_change(periods=10)

    # Calculate the 14-day relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_14d = 100 - (100 / (1 + rs))

    # Create a factor that considers the relationship between the current close price and the adaptive moving average,
    # adjusted by the 10-day high-low range and 5-day volatility
    price_deviation = df['close'] - ma_adaptive
    normalized_price_deviation = price_deviation / (high_low_diff_10d + vol_5d + 1e-7)

    # Incorporate short-term momentum
    short_momentum = df['close'].pct_change(periods=5).fillna(0)

    # Incorporate long-term momentum
    long_momentum = df['close'].pct_change(periods=20).fillna(0)

    # Combine all factors
    factor = normalized_price_deviation + volume_change_10d + rsi_14d + short_momentum + 0.5 * long_momentum

    return factor
```
