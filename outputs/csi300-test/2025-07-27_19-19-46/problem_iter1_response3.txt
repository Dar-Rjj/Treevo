```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the total range of the day (high - low)
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume
    money_flow_ratio = df['amount'] / df['volume']
    
    # Compute the weighted average of close-open difference and money flow ratio
    factor_values = (close_open_diff * money_flow_ratio) / daily_range
    
    return factor_values
```
