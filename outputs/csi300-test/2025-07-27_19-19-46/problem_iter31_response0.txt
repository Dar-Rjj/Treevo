```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate adaptive EMA for short and long terms
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate GARCH(1,1) volatility
    returns = df['close'].pct_change().dropna()
    omega = np.var(returns)
    alpha = 0.06
    beta = 0.92
    garch_volatility = np.sqrt(omega + alpha * returns**2 + beta * returns.shift(1)**2)
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Calculate the relative positioning of the close price with respect to the EMAs
    relative_positioning = (df['close'] - ema_short) / (ema_long - ema_short)
    
    # Combine GARCH volatility, money flow ratio, daily range, and relative positioning
    factor_values = (garch_volatility * money_flow_ratio * relative_positioning) / daily_range
    
    return factor_values
```
