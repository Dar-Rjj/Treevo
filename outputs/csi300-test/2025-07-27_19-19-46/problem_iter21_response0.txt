```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the Exponential Moving Average (EMA) for trend identification
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=50, adjust=False).mean()
    
    # Compute the difference between the EMAs to capture the trend strength
    trend_strength = ema_short - ema_long
    
    # Calculate the Average True Range (ATR) for volatility
    tr = df[['high', 'low']].diff().abs().max(axis=1)
    atr = tr.rolling(window=14).mean().fillna(1e-7)
    
    # Calculate the volume-weighted momentum
    volume_weighted_momentum = (df['close'].diff(1) * df['volume']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Combine trend strength, ATR, volume-weighted momentum, and money flow ratio
    factor_values = (trend_strength * volume_weighted_momentum * money_flow_ratio) / atr
    
    return factor_values
```
