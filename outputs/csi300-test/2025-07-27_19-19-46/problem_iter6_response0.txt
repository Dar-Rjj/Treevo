```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the ratio of the volume to the amount, which can be seen as a measure related to the liquidity
    dollar_vol_ratio = df['volume'] / (df['amount'] + 1e-7)
    
    # Calculate the relative strength, which is the difference between close and open divided by the open price
    rel_strength = (df['close'] - df['open']) / (df['open'] + 1e-7)
    
    # Calculate the percentage change in daily range (high - low) over the closing price
    daily_range_change = (df['high'] - df['low']) / (df['close'] + 1e-7)
    
    # Calculate the rate of change in the close price over a 5-day period
    roc_5 = df['close'].pct_change(5).fillna(0)
    
    # Calculate the average true range over 5 days to capture volatility
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    atr_5 = tr.rolling(window=5).mean().fillna(0)
    
    # Calculate the 20-day exponential moving average to capture a medium-term trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean().fillna(0)
    
    # Calculate the 50-day exponential moving average to capture a longer-term trend
    ema_50 = df['close'].ewm(span=50, adjust=False).mean().fillna(0)
    
    # Calculate the difference between the current close price and the 20-day EMA
    diff_close_ema_20 = (df['close'] - ema_20) / (ema_20 + 1e-7)
    
    # Calculate the difference between the current close price and the 50-day EMA
    diff_close_ema_50 = (df['close'] - ema_50) / (ema_50 + 1e-7)
    
    # Combine the factors using their product, incorporating momentum, liquidity, and volatility
    factor_values = dollar_vol_ratio * rel_strength * daily_range_change * roc_5 * atr_5 * diff_close_ema_20 * diff_close_ema_50
    
    return factor_values
```
