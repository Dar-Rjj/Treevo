```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate Exponential Moving Averages (EMA) for different periods
    ema_short = df['close'].ewm(span=12, adjust=False).mean()
    ema_long = df['close'].ewm(span=26, adjust=False).mean()
    
    # Compute the MACD line and signal line
    macd_line = ema_short - ema_long
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Calculate the RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Bollinger Bands
    sma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    upper_band = sma_20 + 2 * std_20
    lower_band = sma_20 - 2 * std_20
    
    # Calculate the percentage of the close price relative to the Bollinger Bands
    bb_percentage = (df['close'] - lower_band) / (upper_band - lower_band)
    
    # Calculate the volume-weighted close price
    vwap = (df['amount'] / df['volume']).fillna(1e-7)
    
    # Compute the log returns for smoother momentum
    log_returns = np.log(df['close'] / df['close'].shift(1)).fillna(1e-7)
    
    # Combine MACD, RSI, Bollinger Bands, VWAP, and log returns
    factor_values = (macd_line - signal_line) * rsi * bb_percentage * vwap / log_returns
    
    return factor_values
```
