```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day, 20-day, and 60-day simple moving averages (SMA) of the close price for momentum
    sma_5 = df['close'].rolling(window=5).mean().fillna(1e-7)
    sma_20 = df['close'].rolling(window=20).mean().fillna(1e-7)
    sma_60 = df['close'].rolling(window=60).mean().fillna(1e-7)
    
    # Relative position to SMAs as a momentum indicator
    relative_position_sma_5 = (df['close'] - sma_5) / sma_5
    relative_position_sma_20 = (df['close'] - sma_20) / sma_20
    relative_position_sma_60 = (df['close'] - sma_60) / sma_60
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Compute a factor that combines relative position to SMAs, money flow ratio, and daily range
    factor_values = (relative_position_sma_5 + relative_position_sma_20 + relative_position_sma_60) * money_flow_ratio / daily_range
    
    return factor_values
```
