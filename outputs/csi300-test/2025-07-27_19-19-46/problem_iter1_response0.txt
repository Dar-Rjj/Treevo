```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price, capturing the momentum
    momentum = df['close'].diff(1)
    # Calculate the money flow index (MFI) as a measure of buying and selling pressure
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow[df['close'] > df['close'].shift(1)].fillna(0)
    negative_money_flow = raw_money_flow[df['close'] < df['close'].shift(1)].fillna(0)
    mfi_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + mfi_ratio))
    # Combine the MFI with momentum to create a factor that captures both trend and market sentiment
    factor = momentum * mfi
    return factor
```
