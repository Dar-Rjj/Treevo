```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price for momentum
    momentum = df['close'].diff(1)
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Compute simple moving averages for short and long terms
    sma_short = df['close'].rolling(window=10).mean()
    sma_long = df['close'].rolling(window=50).mean()
    
    # Calculate the relative positioning of the close price with respect to the SMAs
    relative_positioning = (df['close'] - sma_short) / (sma_long - sma_short)
    
    # Calculate the volume-weighted price change to capture the impact of volume on price movement
    volume_weighted_price_change = (df['close'] - df['open']) * df['volume']
    
    # Calculate the average true range (ATR) over a 14-day period to measure market volatility
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Calculate the relative strength index (RSI) over a 14-day period to measure the strength of the trend
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine momentum, money flow ratio, relative positioning, volume-weighted price change, and RSI
    factor_values = (momentum * money_flow_ratio * relative_positioning * volume_weighted_price_change * rsi) / (daily_range * atr)
    
    return factor_values
```
