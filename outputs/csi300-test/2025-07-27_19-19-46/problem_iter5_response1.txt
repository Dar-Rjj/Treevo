```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving average of close prices for momentum
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the momentum as the difference between the 5-day and 20-day moving averages
    momentum = ma_5 - ma_20
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Compute a factor that combines momentum, money flow ratio, and daily range
    factor_values = (momentum * money_flow_ratio) / daily_range
    
    # Add a simple 5-day percentage change in volume to incorporate liquidity
    volume_change = df['volume'].pct_change(5).fillna(1e-7)
    
    # Combine the original factor with the volume change
    factor_values = factor_values * volume_change
    
    return factor_values
```
