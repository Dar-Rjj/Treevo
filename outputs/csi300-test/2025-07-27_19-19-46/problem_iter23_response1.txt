```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the Exponential Moving Average (EMA) for short and long terms
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the Average True Range (ATR) to capture volatility
    tr = df['high'] - df['low']
    atr = tr.rolling(window=14).mean().fillna(1e-7)
    
    # Calculate the Simple Moving Average (SMA) for additional trend signals
    sma_short = df['close'].rolling(window=10).mean()
    sma_long = df['close'].rolling(window=50).mean()
    
    # Calculate the volume-weighted close price
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Compute the relative positioning of the close price with respect to the EMAs
    relative_positioning_ema = (df['close'] - ema_short) / (ema_long - ema_short)
    
    # Combine momentum, money flow ratio, ATR, and relative positioning
    factor_values = (df['close'].diff(1) * money_flow_ratio * relative_positioning_ema) / atr
    
    return factor_values
```
