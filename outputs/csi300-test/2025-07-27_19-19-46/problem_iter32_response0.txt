```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    import pandas as pd
    
    # Calculate the difference between today's and yesterday's close price for momentum
    momentum = df['close'].diff(1).fillna(0)
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Compute simple moving averages for short and long terms
    sma_short = df['close'].rolling(window=10).mean()
    sma_long = df['close'].rolling(window=50).mean()
    
    # Calculate the relative positioning of the close price with respect to the SMAs
    relative_positioning = (df['close'] - sma_short) / (sma_long - sma_short).fillna(1e-7)
    
    # Calculate the Relative Strength Index (RSI) to measure the speed and change of price movements
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss.fillna(1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the Moving Average Convergence Divergence (MACD)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_12 - ema_26
    signal = macd.rolling(window=9).mean()
    macd_diff = macd - signal
    
    # Combine momentum, money flow ratio, relative positioning, RSI, and MACD
    factor_values = (momentum * 0.2 + money_flow_ratio * 0.3 + relative_positioning * 0.2 + rsi * 0.15 + macd_diff * 0.15) / daily_range
    
    return factor_values
```
