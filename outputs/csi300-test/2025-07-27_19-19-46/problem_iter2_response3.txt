```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day moving average of the close price for a short-term trend
    short_moving_avg = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the close price for a long-term trend
    long_moving_avg = df['close'].rolling(window=20).mean()
    
    # Calculate the volatility as the standard deviation of the close price over the last 20 days
    volatility = df['close'].rolling(window=20).std()
    
    # Calculate the ratio of the volume to the amount, which can be seen as a normalized dollar volume
    dollar_vol_ratio = (df['volume'] * df['close']) / (df['amount'] + 1e-7)
    
    # Calculate the relative strength, which is the difference between close and open divided by close
    rel_strength = (df['close'] - df['open']) / (df['close'] + 1e-7)
    
    # Combine the two factors using their product to create a factor that could indicate momentum combined with liquidity
    liquidity_momentum_factor = dollar_vol_ratio * rel_strength
    
    # Calculate the moving average convergence divergence (MACD) to identify the direction and strength of the trend
    macd = short_moving_avg - long_moving_avg
    
    # Combine the MACD, volatility, and liquidity-momentum factor to create a more robust and dynamic alpha factor
    factor = (macd / (volatility + 1e-7)) * liquidity_momentum_factor
    
    return factor
```
