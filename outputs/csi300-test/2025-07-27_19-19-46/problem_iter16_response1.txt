```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price for momentum
    momentum = df['close'].diff(1)
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the volume-weighted price change
    volume_weighted_price_change = (df['close'] - df['open']) * df['volume']
    
    # Compute simple moving averages for short, medium, and long terms
    sma_short = df['close'].rolling(window=10).mean()
    sma_medium = df['close'].rolling(window=30).mean()
    sma_long = df['close'].rolling(window=50).mean()
    
    # Calculate the relative positioning of the close price with respect to the SMAs
    relative_positioning_short_medium = (df['close'] - sma_short) / (sma_medium - sma_short).fillna(1e-7)
    relative_positioning_medium_long = (df['close'] - sma_medium) / (sma_long - sma_medium).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Calculate the Chaikin Oscillator for additional volume-based trend information
    chaikin_oscillator = (df['amount'].ewm(span=3, adjust=False).mean() - df['amount'].ewm(span=10, adjust=False).mean())
    
    # Combine momentum, volume-weighted price change, daily range, relative positioning, money flow ratio, and Chaikin Oscillator
    factor_values = (momentum + volume_weighted_price_change) * (relative_positioning_short_medium + relative_positioning_medium_long) * money_flow_ratio * chaikin_oscillator / daily_range
    
    return factor_values
```
