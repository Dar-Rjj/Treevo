```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price for momentum
    momentum = df['close'].diff(1)
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Compute a 5-day simple moving average of the close price
    sma_5 = df['close'].rolling(window=5).mean()
    
    # Compute a 20-day simple moving average of the close price
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the difference between the 5-day and 20-day moving averages to capture trend
    trend = sma_5 - sma_20
    
    # Calculate the average daily range over the last 5 days
    avg_daily_range_5 = daily_range.rolling(window=5).mean().fillna(1e-7)
    
    # Calculate the average money flow ratio over the last 5 days
    avg_money_flow_ratio_5 = money_flow_ratio.rolling(window=5).mean().fillna(1e-7)
    
    # Compute a factor that combines momentum, trend, average money flow ratio, and average daily range
    factor_values = (momentum + trend) * (avg_money_flow_ratio_5 / avg_daily_range_5)
    
    return factor_values
```
