```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, representing the range of the day
    daily_range = df['high'] - df['low']
    
    # Calculate the buying pressure by comparing the close to the low
    buying_pressure = (df['close'] - df['low']) / (df['high'] - df['low'] + 1e-7)
    
    # Calculate the selling pressure by comparing the open to the high
    selling_pressure = (df['high'] - df['open']) / (df['high'] - df['low'] + 1e-7)
    
    # Calculate the money flow ratio, which is the amount of money involved in the transaction
    money_flow_ratio = df['amount'] / (df['volume'] + 1e-7)
    
    # Combine the buying pressure, selling pressure, and money flow ratio to create a composite factor
    factor = (buying_pressure - selling_pressure) * money_flow_ratio
    
    return factor
```
