```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price for momentum
    momentum = df['close'].diff(1)
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Compute adaptive simple moving averages for short and long terms
    sma_short = df['close'].rolling(window=10, min_periods=5).mean()
    sma_long = df['close'].rolling(window=50, min_periods=25).mean()
    
    # Calculate the relative positioning of the close price with respect to the SMAs
    relative_positioning = (df['close'] - sma_short) / (sma_long - sma_short).fillna(1e-7)
    
    # Calculate the Average True Range (ATR) for a more robust volatility measure
    true_range = df[['high', 'low']].diff(axis=1).max(axis=1).abs() + df['close'].shift(1).fillna(1e-7)
    atr = true_range.rolling(window=14, min_periods=7).mean()
    
    # Combine momentum, money flow ratio, relative positioning, and ATR
    factor_values = (momentum * money_flow_ratio * relative_positioning) / (daily_range * atr)
    
    return factor_values
```
