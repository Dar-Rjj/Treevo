```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price for momentum
    momentum = df['close'].diff(1)
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Compute adaptive weighted moving averages for short and long terms
    wma_short = df['close'].ewm(span=10, adjust=False).mean()
    wma_long = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the relative positioning of the close price with respect to the WMA
    relative_positioning = (df['close'] - wma_short) / (wma_long - wma_short)
    
    # Dynamic volatility measure using the standard deviation of the last 10 days
    dynamic_volatility = df['close'].rolling(window=10).std().fillna(1e-7)
    
    # Combine momentum, money flow ratio, relative positioning, and dynamic volatility
    factor_values = (momentum * money_flow_ratio * relative_positioning) / (daily_range * dynamic_volatility)
    
    return factor_values
```
