```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price for momentum
    momentum = df['close'].diff(1)
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Combine momentum, money flow ratio, and daily range
    factor_values = (momentum * money_flow_ratio) / daily_range
    
    return factor_values
```
