```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price for momentum
    momentum = df['close'].diff(1)
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Calculate the volume-weighted price change to capture the impact of volume on price movement
    volume_weighted_price_change = (df['close'] - df['open']) * df['volume']
    
    # Compute exponential moving averages for short and long terms to add trend-following components
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the relative positioning of the close price with respect to the EMAs
    relative_positioning = (df['close'] - ema_short) / (ema_long - ema_short)
    
    # Combine momentum, money flow ratio, volume-weighted price change, and relative positioning
    factor_values = (momentum * money_flow_ratio + volume_weighted_price_change) / daily_range * relative_positioning
    
    return factor_values
```
