```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price for momentum
    momentum = df['close'].diff(1)
    
    # Calculate the exponential moving average (EMA) of the close price for trend analysis
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=50, adjust=False).mean()
    
    # Compute the ATR (Average True Range) as a measure of volatility
    tr = df[['high', 'low']].diff(axis=1).iloc[:, 0].abs() + df['high'] - df['low']
    atr = tr.rolling(window=14).mean().fillna(1e-7)
    
    # Calculate the volume-weighted money flow ratio
    money_flow_ratio = (df['amount'] / df['volume']).ewm(span=10, adjust=False).mean()
    
    # Calculate the relative positioning of the close price with respect to the EMAs
    relative_positioning = (df['close'] - ema_short) / (ema_long - ema_short)
    
    # Combine momentum, money flow ratio, ATR, and relative positioning
    factor_values = (momentum * money_flow_ratio * relative_positioning) / atr
    
    return factor_values
```
