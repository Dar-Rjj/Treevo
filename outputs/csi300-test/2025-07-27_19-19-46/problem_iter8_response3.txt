```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price for momentum
    momentum = df['close'].diff(1)
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Calculate the 50-day simple moving average as a trend-following indicator
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the 200-day simple moving average as an even longer-term trend-following indicator
    sma_200 = df['close'].rolling(window=200).mean()
    
    # Calculate the relative position of the close price with respect to the 50-day and 200-day SMAs
    rel_pos_sma_50 = (df['close'] - sma_50) / (sma_50 + 1e-7)
    rel_pos_sma_200 = (df['close'] - sma_200) / (sma_200 + 1e-7)
    
    # Calculate the 20-day Bollinger Bands percentage to capture volatility and trend
    std_20 = df['close'].rolling(window=20).std()
    bollinger_high = sma_50 + 2 * std_20
    bollinger_low = sma_50 - 2 * std_20
    bollinger_pct = (df['close'] - bollinger_low) / (bollinger_high - bollinger_low + 1e-7)
    
    # Calculate the RSI (Relative Strength Index) over the last 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Non-linear combination of factors
    factor_values = (momentum ** 2 * money_flow_ratio * daily_range ** 0.5 * rel_pos_sma_50 * rel_pos_sma_200 * bollinger_pct * rsi) ** 0.5
    
    return factor_values
```
