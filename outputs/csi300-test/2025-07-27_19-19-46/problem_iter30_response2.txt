```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price for momentum
    momentum = df['close'].diff(1)
    
    # Calculate the total range of the day (high - low) to capture volatility, avoiding division by zero
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Compute exponential moving averages for short and long terms
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the relative positioning of the close price with respect to the EMAs
    relative_positioning = (df['close'] - ema_short) / (ema_long - ema_short).fillna(1e-7)
    
    # Integrate sector-specific trends (assuming 'sector' is a column in the DataFrame)
    # For simplicity, we use the mean of the closing prices of the same sector
    sector_trend = df.groupby('sector')['close'].transform('mean').fillna(1e-7)
    
    # Incorporate liquidity metrics (volume as a proxy for liquidity)
    liquidity_metric = df['volume'] / df['volume'].rolling(window=20).mean().fillna(1e-7)
    
    # Calculate the rate of change in the close price over a 14-day period to capture trend shifts
    roc = df['close'].pct_change(periods=14).fillna(0)
    
    # Combine momentum, money flow ratio, daily range, relative positioning, sector trend, and liquidity metric
    factor_values = (momentum * money_flow_ratio * relative_positioning * liquidity_metric * (1 + roc)) / (daily_range * sector_trend)
    
    return factor_values
```
