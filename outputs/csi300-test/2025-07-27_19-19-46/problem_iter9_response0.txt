```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price for momentum
    momentum = df['close'].diff(1)
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Compute multiple simple moving averages (SMA) for relative positioning
    sma_5 = df['close'].rolling(window=5).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Combine SMAs to create a relative positioning factor
    relative_positioning = (sma_5 - sma_10) / (sma_10 - sma_20)
    
    # Non-linear combination of factors for enhanced stability and interpretability
    factor_values = (momentum * money_flow_ratio) / daily_range + relative_positioning
    
    return factor_values
```
