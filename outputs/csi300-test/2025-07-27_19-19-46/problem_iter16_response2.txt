```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price for momentum
    momentum = df['close'].diff(1)
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the volume-weighted price change
    volume_weighted_price_change = (df['close'] - df['open']) * df['volume']
    
    # Compute adaptive moving averages for short and long terms using a rolling standard deviation
    std_short = df['close'].rolling(window=10).std()
    std_long = df['close'].rolling(window=50).std()
    sma_short = df['close'].ewm(span=10, adjust=False).mean()
    sma_long = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the relative positioning of the close price with respect to the SMAs
    relative_positioning = (df['close'] - sma_short) / (sma_long - sma_short).fillna(1e-7)
    
    # Calculate the liquidity factor as the ratio of amount to daily range
    liquidity = df['amount'] / daily_range.fillna(1e-7)
    
    # Combine momentum, volume-weighted price change, daily range, relative positioning, and liquidity
    factor_values = (momentum + volume_weighted_price_change) * relative_positioning * liquidity / daily_range
    
    return factor_values
```
