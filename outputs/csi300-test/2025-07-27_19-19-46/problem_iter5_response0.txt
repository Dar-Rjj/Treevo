```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price for momentum
    momentum = df['close'].diff(1)
    
    # Calculate the total range of the day (high - low) to capture volatility
    daily_range = (df['high'] - df['low']).fillna(1e-7)
    
    # Calculate the money flow ratio by dividing the amount by volume to gauge market sentiment
    money_flow_ratio = df['amount'] / df['volume']
    
    # Compute a moving average of the close price over 5 days to capture a short-term trend
    close_ma_5 = df['close'].rolling(window=5).mean()
    
    # Compute a moving average of the close price over 20 days to capture a longer-term trend
    close_ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the percentage change in the close price over 5 days
    close_pct_change_5 = df['close'].pct_change(5)
    
    # Calculate the percentage change in the close price over 20 days
    close_pct_change_20 = df['close'].pct_change(20)
    
    # Combine the momentum, money flow ratio, daily range, and percentage changes
    factor_values = (momentum * money_flow_ratio / daily_range) + close_pct_change_5 + close_pct_change_20
    
    return factor_values
```
