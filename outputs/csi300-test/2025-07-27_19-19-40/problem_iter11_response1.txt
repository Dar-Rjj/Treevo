```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 10 and 30 periods to capture short and long-term trend information
    df['short_rolling_mean_close'] = df['close'].rolling(window=10).mean()
    df['long_rolling_mean_close'] = df['close'].rolling(window=30).mean()
    
    # Calculate a rolling standard deviation of volume over 10 and 30 periods to understand short and long-term volatility in trading activity
    df['short_rolling_std_volume'] = df['volume'].rolling(window=10).std()
    df['long_rolling_std_volume'] = df['volume'].rolling(window=30).std()
    
    # A momentum indicator using the difference between the current close and the average close over the past 10 and 30 periods, scaled by the relative volume
    short_term_momentum = (df['close'] - df['short_rolling_mean_close']) * df['volume'] / (df['short_rolling_std_volume'] + 1e-7)
    long_term_momentum = (df['close'] - df['long_rolling_mean_close']) * df['volume'] / (df['long_rolling_std_volume'] + 1e-7)
    
    # RSI calculation for an additional signal
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the short and long-term momentum signals with RSI
    alpha_factor = (short_term_momentum + long_term_momentum) * rsi
    
    return alpha_factor
```
