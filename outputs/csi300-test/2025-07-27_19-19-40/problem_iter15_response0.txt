```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Short-term trend using 5 period moving average
    df['short_moving_avg'] = df['close'].rolling(window=5).mean()
    # Long-term trend using 20 period moving average
    df['long_moving_avg'] = df['close'].rolling(window=20).mean()
    
    # Calculate the relative volume as a ratio of current volume to the 10-day average volume
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['relative_volume'] = df['volume'] / (df['avg_volume_10d'] + 1e-7)
    
    # Dynamic indicator using the difference between short and long moving averages
    df['trend_diff'] = df['short_moving_avg'] - df['long_moving_avg']
    
    # Final alpha factor: Trend difference scaled by relative volume
    alpha_factor = df['trend_diff'] * df['relative_volume']
    return alpha_factor
```
