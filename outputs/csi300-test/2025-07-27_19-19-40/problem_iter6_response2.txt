```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between current close and the previous day's close for momentum
    df['close_diff'] = df['close'] - df['close'].shift(1)
    
    # Calculate the 5-day moving average of the closing prices for a short-term trend
    df['5_day_avg_close'] = df['close'].rolling(window=5).mean()
    
    # Calculate the ratio of the current volume to the 5-day average volume for volume shock
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / (df['5_day_avg_volume'] + 1e-7)
    
    # Calculate the relative strength compared to the 5-day moving average
    df['rel_strength'] = df['close'] / df['5_day_avg_close'] - 1
    
    # Additional factor: difference between high and low for volatility
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the 20-day moving average of the closing prices for a medium-term trend
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    
    # Calculate the relative strength compared to the 20-day moving average
    df['rel_strength_20'] = df['close'] / df['20_day_avg_close'] - 1
    
    # Alpha factor: combining short-term and medium-term trends, momentum, volume shock, and intraday volatility
    alpha_factor = (df['close_diff'] * df['volume_ratio'] * df['rel_strength'] * df['rel_strength_20']) * df['high_low_diff']
    
    return alpha_factor
```
