```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 10 periods to capture short-term trend
    df['rolling_mean_close_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate a rolling standard deviation of volume over 5 periods to capture short-term volatility in trading activity
    df['rolling_std_volume_5'] = df['volume'].rolling(window=5).std()
    
    # Calculate the relative change in close price as the difference between the current close and the 10-period rolling mean
    df['relative_change_close'] = df['close'] - df['rolling_mean_close_10']
    
    # Adjust the relative change in close price by the 5-period rolling standard deviation of volume
    alpha_factor = df['relative_change_close'] * df['volume'] / (df['rolling_std_volume_5'] + 1e-7)
    
    return alpha_factor
```
