```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 30 periods to capture longer-term trend information
    df['rolling_mean_30_close'] = df['close'].rolling(window=30).mean()
    
    # Calculate a rolling standard deviation of volume over 30 periods to understand longer-term volatility in trading activity
    df['rolling_std_30_volume'] = df['volume'].rolling(window=30).std()
    
    # Calculate the difference between the current close and the average close over the past 30 periods
    df['close_diff_30'] = df['close'] - df['rolling_mean_30_close']
    
    # Calculate the ratio of the current volume to the 10-day average volume for volume shock
    df['rolling_mean_10_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio_10'] = df['volume'] / (df['rolling_mean_10_volume'] + 1e-7)
    
    # Calculate the relative strength compared to the 30-day moving average
    df['rel_strength_30'] = df['close'] / (df['rolling_mean_30_close'] + 1e-7) - 1
    
    # Calculate the difference between high and low for intraday volatility
    df['high_low_diff'] = df['high'] - df['low']
    
    # Alpha factor: combining longer-term trends, momentum, volume shock, and intraday volatility
    alpha_factor = (df['close_diff_30'] * df['volume_ratio_10'] * df['rel_strength_30']) * df['high_low_diff']
    
    return alpha_factor
```
