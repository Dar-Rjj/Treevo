```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate 10-period and 30-period Exponential Moving Averages (EMA) of close prices for short-term and medium-term trend signals
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()

    # Relative Strength Index (RSI) over a 14-day period to capture overbought/oversold conditions
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))

    # Calculate the MACD line and signal line
    df['macd_line'] = df['ema_10'] - df['ema_30']
    df['signal_line'] = df['macd_line'].ewm(span=9, adjust=False).mean()

    # Calculate the difference between current close and 10-period EMA, scaled by the RSI
    close_ema_diff = (df['close'] - df['ema_10']) * df['rsi']

    # Use the difference in MACD and signal lines as a momentum indicator
    macd_signal_diff = df['macd_line'] - df['signal_line']

    # Combine the momentum indicators with the relative volume
    relative_volume = df['volume'] / df['volume'].rolling(window=10).mean()
    alpha_factor = (close_ema_diff + macd_signal_diff) * relative_volume

    return alpha_factor
```
