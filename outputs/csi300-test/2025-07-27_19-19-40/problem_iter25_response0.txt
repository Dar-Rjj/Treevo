```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the rolling mean and standard deviation of close prices over 10 and 20 periods to capture trend and volatility information
    df['rolling_mean_close_10'] = df['close'].rolling(window=10).mean()
    df['rolling_std_close_20'] = df['close'].rolling(window=20).std()

    # A momentum indicator using the difference between the current close and the average close over the past 10 periods, adjusted by the current volume
    momentum_factor = (df['close'] - df['rolling_mean_close_10']) * df['volume']

    # A mean reversion factor using the z-score over the last 20 periods, scaled by the inverse of the current volume to favor less liquid stocks
    mean_reversion_factor = ((df['close'] - df['close'].rolling(window=20).mean()) / df['rolling_std_close_20']) / df['volume']

    # Combine the momentum and mean reversion factors, with a focus on both trend continuation and potential reversals
    alpha_factor = momentum_factor + mean_reversion_factor

    return alpha_factor
```
