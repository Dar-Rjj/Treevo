```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 10 periods to capture long-term trend information
    df['rolling_mean_close_10'] = df['close'].rolling(window=10).mean()
    # Calculate a rolling mean of close prices over 5 periods to capture short-term trend information
    df['rolling_mean_close_5'] = df['close'].rolling(window=5).mean()
    # Calculate a rolling standard deviation of volume over 10 periods to understand volatility in trading activity
    df['rolling_std_volume'] = df['volume'].rolling(window=10).std()
    
    # Calculate the Relative Strength Index (RSI) for the last 14 periods
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # A momentum indicator using the difference between the current close and the average close over the past 10 periods,
    # adjusted by the RSI and scaled by the relative volume
    alpha_factor = ((df['close'] - df['rolling_mean_close_10']) * rsi) * df['volume'] / (df['rolling_std_volume'] + 1e-7)
    
    return alpha_factor
```
