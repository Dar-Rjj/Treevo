```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 5 periods to capture short-term trend information
    df['rolling_mean_close'] = df['close'].rolling(window=5).mean()
    # Calculate a rolling standard deviation of volume over 5 periods to understand short-term volatility in trading activity
    df['rolling_std_volume'] = df['volume'].rolling(window=5).std()
    # Calculate the percentage change in close price to capture price momentum
    df['price_change'] = df['close'].pct_change()
    # A momentum indicator using the difference between the current close and the average close over the past 5 periods, scaled by the relative volume
    alpha_factor = (df['close'] - df['rolling_mean_close']) * df['price_change'] * df['volume'] / (df['rolling_std_volume'] + 1e-7)
    return alpha_factor
```
