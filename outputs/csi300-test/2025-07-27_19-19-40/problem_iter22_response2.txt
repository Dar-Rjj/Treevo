```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 5 periods to capture short-term trend information
    df['rolling_mean_close'] = df['close'].rolling(window=5).mean()
    # A simple mean reversion factor using the difference between the current close and the rolling mean
    alpha_factor = df['close'] - df['rolling_mean_close']
    return alpha_factor
```
