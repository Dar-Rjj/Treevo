```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 60 periods for an even longer-term trend
    df['rolling_mean_close_60'] = df['close'].rolling(window=60).mean()
    
    # Calculate a rolling standard deviation of close prices over 60 periods to measure long-term volatility
    df['rolling_std_close_60'] = df['close'].rolling(window=60).std()
    
    # Calculate a rolling mean of volume over 30 periods to capture the average trading activity
    df['rolling_mean_volume_30'] = df['volume'].rolling(window=30).mean()
    
    # Calculate the relative volume as the ratio of today's volume to the 30-period rolling mean volume
    df['relative_volume_30'] = df['volume'] / (df['rolling_mean_volume_30'] + 1e-7)
    
    # Calculate the difference between the current close and the 60-period rolling mean, 
    # adjusted by the 60-period rolling standard deviation, and scaled by the relative volume
    alpha_factor = ((df['close'] - df['rolling_mean_close_60']) / (df['rolling_std_close_60'] + 1e-7)) * df['relative_volume_30']
    
    return alpha_factor
```
