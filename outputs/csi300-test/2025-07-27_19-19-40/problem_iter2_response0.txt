```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 30-day moving average of the closing price for a longer-term trend
    ma_30 = df['close'].rolling(window=30).mean()
    
    # Calculate the relative strength compared to the 30-day moving average
    relative_strength = df['close'] / ma_30
    
    # Calculate the 10-day standard deviation of the high prices for volatility
    std_high = df['high'].rolling(window=10).std()
    
    # Calculate the 5-day average volume
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate the volume anomaly as the ratio of current volume to the 5-day average volume
    volume_anomaly = df['volume'] / (avg_volume + 1e-7)
    
    # Heuristic factor: (Relative Strength * Volatility) * Volume Anomaly
    factor_values = (relative_strength * std_high) * volume_anomaly
    
    return factor_values
```
