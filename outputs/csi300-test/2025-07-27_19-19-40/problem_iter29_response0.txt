```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate Relative Strength Index (RSI) with a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    RS = gain / loss
    RSI = 100 - (100 / (1 + RS))
    
    # Calculate Moving Average Convergence Divergence (MACD) with 12 and 26 periods
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    MACD = short_ema - long_ema
    
    # Volume-Weighted Price Change
    volume_weighted_price_change = (df['close'] - df['open']) * df['volume']
    
    # Combine the factors using adaptive weights based on volatility
    volatility = df['close'].rolling(window=10).std()
    rsi_weight = 1 / (volatility + 1e-8)
    macd_weight = 1 / (volatility + 1e-8)
    vwpc_weight = 1 / (volatility + 1e-8)
    
    alpha_factor = (rsi_weight * RSI + macd_weight * MACD + vwpc_weight * volume_weighted_price_change) / (rsi_weight + macd_weight + vwpc_weight)
    
    return alpha_factor
```
