```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 10 periods to capture trend information
    df['rolling_mean_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate a rolling standard deviation of close prices over 10 periods to understand price volatility
    df['rolling_std_close'] = df['close'].rolling(window=10).std()
    
    # A momentum indicator using the difference between the current close and the average close over the past 10 periods
    df['momentum'] = df['close'] - df['rolling_mean_close']
    
    # A volatility-adjusted momentum factor, scaled by the rolling standard deviation of close prices
    alpha_factor = df['momentum'] / (df['rolling_std_close'] + 1e-7)
    
    return alpha_factor
```
