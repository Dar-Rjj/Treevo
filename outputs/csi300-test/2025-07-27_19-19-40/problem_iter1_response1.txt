```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices, adjusted by volume
    price_diff = (df['close'] - df['open']) * df['volume']
    
    # Weighted moving average of the price difference
    wma = price_diff.rolling(window=5).mean() * 0.4 + \
          price_diff.rolling(window=10).mean() * 0.3 + \
          price_diff.rolling(window=20).mean() * 0.2 + \
          price_diff.rolling(window=60).mean() * 0.1
    
    # Calculate a relative strength compared to the moving average of the close price
    close_wma = df['close'].rolling(window=20).mean()
    rel_strength = df['close'] / close_wma - 1
    
    # Final factor combining the weighted moving average and relative strength
    factor = wma * rel_strength
    return factor
```
