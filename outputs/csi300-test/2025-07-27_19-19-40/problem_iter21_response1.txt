```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling median of close prices over 15 periods to capture trend information
    df['rolling_median_close'] = df['close'].rolling(window=15).median()
    # A volatility indicator using the standard deviation of the close prices over the past 15 periods
    df['rolling_std_close'] = df['close'].rolling(window=15).std()
    # A momentum-based factor using the difference between the current close and the rolling median, scaled by the current volume
    # This is further normalized by the rolling standard deviation to account for volatility
    alpha_factor = ((df['close'] - df['rolling_median_close']) / df['rolling_std_close']) * df['volume']
    return alpha_factor
```
