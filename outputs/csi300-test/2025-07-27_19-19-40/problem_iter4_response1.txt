```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 10 periods to capture trend information
    df['rolling_mean_close'] = df['close'].rolling(window=10).mean()
    # A momentum indicator using the difference between the current close and the average close over the past 10 periods
    alpha_factor = df['close'] - df['rolling_mean_close']
    return alpha_factor
```
