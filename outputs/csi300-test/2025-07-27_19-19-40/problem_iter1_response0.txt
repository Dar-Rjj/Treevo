```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between current close and the previous day's close
    df['close_diff'] = df['close'] - df['close'].shift(1)
    
    # Calculate the average of the last 5 day's closing prices
    df['5_day_avg_close'] = df['close'].rolling(window=5).mean()
    
    # Calculate the ratio of the current volume to the 5-day average volume
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / (df['5_day_avg_volume'] + 1e-7)
    
    # Alpha factor: a combination of momentum and volume shock
    alpha_factor = df['close_diff'] * df['volume_ratio']
    
    return alpha_factor
```
