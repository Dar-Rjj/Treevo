```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 20 periods for long-term trend
    df['long_trend'] = df['close'].rolling(window=20).mean()
    # Calculate a rolling standard deviation of volume over 10 periods to understand recent volatility
    df['recent_volatility'] = df['volume'].rolling(window=10).std()
    
    # Calculate the momentum of the close price relative to the long-term trend
    df['momentum'] = df['close'] - df['long_trend']
    
    # Calculate the relative strength of the volume, normalized by recent volatility
    df['relative_strength'] = (df['volume'] - df['volume'].shift(1)) / (df['recent_volatility'] + 1e-7)
    
    # Combine the momentum and relative strength to create the alpha factor
    alpha_factor = df['momentum'] * df['relative_strength']
    
    return alpha_factor
```
