```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 20 and 5 periods to capture long and short-term trend information
    df['rolling_mean_close_20'] = df['close'].rolling(window=20).mean()
    df['rolling_mean_close_5'] = df['close'].rolling(window=5).mean()

    # Calculate a rolling standard deviation of volume over 20 and 5 periods to understand long and short-term volatility in trading activity
    df['rolling_std_volume_20'] = df['volume'].rolling(window=20).std()
    df['rolling_std_volume_5'] = df['volume'].rolling(window=5).std()

    # Calculate the difference between the current close and the average close over the past 20 and 5 periods
    df['close_diff_20'] = df['close'] - df['rolling_mean_close_20']
    df['close_diff_5'] = df['close'] - df['rolling_mean_close_5']

    # Calculate the relative volume, scaling by the long and short-term standard deviation of volume
    df['relative_volume_20'] = df['volume'] / (df['rolling_std_volume_20'] + 1e-7)
    df['relative_volume_5'] = df['volume'] / (df['rolling_std_volume_5'] + 1e-7)

    # A momentum indicator using the difference in close prices, scaled by the relative volume
    alpha_factor = (df['close_diff_20'] * df['relative_volume_20']) + (df['close_diff_5'] * df['relative_volume_5'])

    # Incorporate market microstructure: ratio of amount to volume
    df['amount_to_volume_ratio'] = df['amount'] / (df['volume'] + 1e-7)

    # Adjust the alpha factor with the amount to volume ratio, which might capture changes in average transaction size
    alpha_factor_adjusted = alpha_factor * df['amount_to_volume_ratio']

    return alpha_factor_adjusted
```
