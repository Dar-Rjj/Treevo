```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the exponential moving average (EMA) of close prices over 5 periods to capture short-term trend information
    df['ema_close'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the standard deviation of daily returns over 20 days as a measure of volatility
    df['volatility'] = df['daily_return'].rolling(window=20).std()
    
    # A momentum indicator using the difference between the current close and the EMA, scaled by the volume and adjusted for volatility
    alpha_factor = (df['close'] - df['ema_close']) * df['volume'] / (df['volatility'] + 1e-7)
    
    return alpha_factor
```
