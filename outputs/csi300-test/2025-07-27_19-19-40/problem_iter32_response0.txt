```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling median of close prices over 20 periods to capture trend information
    df['rolling_median_close'] = df['close'].rolling(window=20).median()
    
    # A momentum indicator using the difference between the current close and the median close over the past 20 periods, scaled by the current volume
    alpha_factor = (df['close'] - df['rolling_median_close']) * df['volume']
    
    # Add a liquidity factor based on the ratio of the current volume to the average volume over the past 10 periods
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    liquidity_factor = df['volume'] / df['avg_volume_10']
    
    # Combine the momentum and liquidity factors
    alpha_factor = alpha_factor * liquidity_factor
    
    return alpha_factor
```
