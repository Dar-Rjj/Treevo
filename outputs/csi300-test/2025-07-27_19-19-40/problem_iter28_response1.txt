```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 10 periods to capture trend information
    df['rolling_mean_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate the momentum by taking the difference between the current close and the rolling mean of close
    df['momentum'] = df['close'] - df['rolling_mean_close']
    
    # Calculate the volume-weighted momentum to emphasize the strength of the trend
    alpha_factor = df['momentum'] * df['volume']
    
    return alpha_factor
```
