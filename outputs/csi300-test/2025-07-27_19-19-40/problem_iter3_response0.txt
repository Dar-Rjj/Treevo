```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 20 periods to capture longer-term trend information
    df['rolling_mean_close_20'] = df['close'].rolling(window=20).mean()
    # Calculate a rolling standard deviation of volume over 20 periods to understand longer-term volatility in trading activity
    df['rolling_std_volume_20'] = df['volume'].rolling(window=20).std()
    # A momentum indicator using the difference between the current close and the average close over the past 20 periods, scaled by the relative volume
    alpha_factor = (df['close'] - df['rolling_mean_close_20']) * df['volume'] / (df['rolling_std_volume_20'] + 1e-7)
    return alpha_factor
```
