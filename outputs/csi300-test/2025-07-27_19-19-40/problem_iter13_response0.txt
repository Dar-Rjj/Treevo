def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Long-term trend: Calculate a rolling mean of close prices over 30 periods
    df['long_trend'] = df['close'].rolling(window=30).mean()
    # Short-term trend: Calculate a rolling mean of close prices over 10 periods
    df['short_trend'] = df['close'].rolling(window=10).mean()
    # Adjust for volatility: Calculate a rolling standard deviation of volume over 10 periods
    df['volatility'] = df['volume'].rolling(window=10).std()
    # Relative strength index (RSI) to capture the momentum
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    # Combine long and short trends, adjust for volatility, and scale by relative volume
    alpha_factor = ((df['close'] - df['long_trend']) * (df['close'] - df['short_trend'])) * df['volume'] / (df['volatility'] + 1e-7) * df['rsi']
    return alpha_factor
