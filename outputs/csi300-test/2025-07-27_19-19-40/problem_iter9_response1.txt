```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 20 periods for a longer-term trend
    df['rolling_mean_close_20'] = df['close'].rolling(window=20).mean()
    # Calculate a rolling standard deviation of close prices over 20 periods to measure long-term volatility
    df['rolling_std_close_20'] = df['close'].rolling(window=20).std()
    # Calculate a rolling mean of volume over 10 periods to capture the average trading activity
    df['rolling_mean_volume_10'] = df['volume'].rolling(window=10).mean()
    # An alpha factor combining the difference between the current close and the 20-period rolling mean, 
    # adjusted by the 20-period rolling standard deviation, and scaled by the relative volume
    alpha_factor = ((df['close'] - df['rolling_mean_close_20']) / (df['rolling_std_close_20'] + 1e-7)) * (df['volume'] / df['rolling_mean_volume_10'])
    return alpha_factor
```
