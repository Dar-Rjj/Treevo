```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    daily_return = df['close'].diff()
    # Calculate the average volume over 5 days
    avg_volume = df['volume'].rolling(window=5).mean()
    # Calculate the standard deviation of high prices over the last 10 days
    std_high = df['high'].rolling(window=10).std()
    # Heuristic factor: (Daily return / Avg Volume) * Std(High, 10)
    factor_values = (daily_return / avg_volume) * std_high
    return factor_values
```
