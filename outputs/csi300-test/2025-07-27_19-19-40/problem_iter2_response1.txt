def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 10 periods to capture trend information
    df['rolling_mean_close'] = df['close'].rolling(window=10).mean()
    # Calculate a rolling standard deviation of close prices over 10 periods to understand price volatility
    df['rolling_std_close'] = df['close'].rolling(window=10).std()
    # Calculate a rolling standard deviation of volume over 10 periods to understand volatility in trading activity
    df['rolling_std_volume'] = df['volume'].rolling(window=10).std()
    # Momentum indicator using the difference between the current close and the average close over the past 10 periods, scaled by the relative volume
    momentum = (df['close'] - df['rolling_mean_close']) * df['volume'] / (df['rolling_std_volume'] + 1e-7)
    # Trend-following factor using the ratio of the current close to the rolling mean close, adjusted by the rolling standard deviation of close prices
    trend_following = (df['close'] / df['rolling_mean_close']) * (df['rolling_std_close'] + 1e-7)
    # Combine momentum and trend-following factors
    alpha_factor = momentum * trend_following
    return alpha_factor
