```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a 15-period rolling mean of close prices to capture medium-term trend
    df['rolling_mean_close_15'] = df['close'].rolling(window=15).mean()
    
    # Calculate a 15-period rolling standard deviation of volume to understand medium-term volatility in trading activity
    df['rolling_std_volume_15'] = df['volume'].rolling(window=15).std()
    
    # Calculate the relative strength index (RSI) over a 14-day period to gauge the strength of recent price action
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    df['rsi_14'] = rsi
    
    # A momentum indicator using the difference between the current close and the 15-period average close, 
    # adjusted by the 15-period rolling standard deviation of volume and the RSI
    alpha_factor = (df['close'] - df['rolling_mean_close_15']) * df['volume'] / (df['rolling_std_volume_15'] + 1e-7) * df['rsi_14']
    
    return alpha_factor
```
