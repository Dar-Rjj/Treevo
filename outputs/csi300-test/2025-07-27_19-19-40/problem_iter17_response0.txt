def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day Exponential Moving Average (EMA) of the close prices for a smoother trend representation
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    # Calculate the relative volume as the ratio of today's volume to the average volume over the past 10 days
    df['relative_volume'] = df['volume'] / df['volume'].rolling(window=10).mean()
    # Calculate the price change per volume, a measure of how much the price changes with each unit of volume traded
    df['price_change_per_volume'] = (df['close'] - df['open']) / df['volume']
    # Alpha factor combining EMA, relative volume, and price change per volume for a more responsive and interpretable signal
    alpha_factor = (df['close'] - df['ema_10']) * df['relative_volume'] * df['price_change_per_volume']
    return alpha_factor
