```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum factor using the difference between current close and 20-periods rolling mean of close, scaled by volume divided by 20-periods rolling standard deviation of volume
    alpha_factor = (df['close'] - df['close'].rolling(window=20).mean()) * df['volume'] / (df['volume'].rolling(window=20).std() + 1e-7)
    return alpha_factor
```
