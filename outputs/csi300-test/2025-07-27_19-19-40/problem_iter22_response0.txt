```txt
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 5 periods to capture the short-term trend
    df['rolling_mean_close'] = df['close'].rolling(window=5).mean()
    
    # Calculate a rolling standard deviation of volume over 5 periods to understand the volatility in trading activity
    df['rolling_std_volume'] = df['volume'].rolling(window=5).std()
    
    # A momentum indicator using the difference between the current close and the average close over the past 5 periods, scaled by the relative volume
    alpha_factor = (df['close'] - df['rolling_mean_close']) * df['volume'] / (df['rolling_std_volume'] + 1e-7)
    
    return alpha_factor
```
