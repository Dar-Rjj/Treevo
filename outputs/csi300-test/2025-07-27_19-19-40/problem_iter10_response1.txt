def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 20 periods to capture medium-term trend information
    df['rolling_mean_close_20'] = df['close'].rolling(window=20).mean()
    # Calculate a rolling standard deviation of close prices over 20 periods to understand volatility
    df['rolling_std_close_20'] = df['close'].rolling(window=20).std()
    # Calculate a rolling mean of close prices over 50 periods to capture long-term trend information
    df['rolling_mean_close_50'] = df['close'].rolling(window=50).mean()
    # Calculate a rolling mean of volume over 20 periods to understand the average trading activity
    df['rolling_mean_volume_20'] = df['volume'].rolling(window=20).mean()
    # Calculate the relative volume as the ratio of today's volume to the average volume
    df['relative_volume'] = df['volume'] / (df['rolling_mean_volume_20'] + 1e-7)
    # Calculate the relative strength index (RSI) for the close price over a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    df['rsi'] = rsi
    # Calculate the moving average convergence divergence (MACD)
    df['ema_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['macd'] = df['ema_12'] - df['ema_26']
    df['signal'] = df['macd'].ewm(span=9, adjust=False).mean()
    # A novel alpha factor combining the RSI, MACD, and the difference between the current close and the 20-period rolling mean, scaled by the rolling standard deviation
    alpha_factor = ((df['close'] - df['rolling_mean_close_20']) * (100 - df['rsi']) / (df['rolling_std_close_20'] + 1e-7)) + (df['macd'] - df['signal']) * df['relative_volume']
    return alpha_factor
