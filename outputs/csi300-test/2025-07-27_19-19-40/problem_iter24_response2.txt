```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Short-term rolling mean of close prices over 5 periods
    df['short_term_rolling_mean_close'] = df['close'].rolling(window=5).mean()
    # Moderate-term rolling mean of close prices over 10 periods
    df['moderate_term_rolling_mean_close'] = df['close'].rolling(window=10).mean()
    # Long-term rolling mean of close prices over 20 periods
    df['long_term_rolling_mean_close'] = df['close'].rolling(window=20).mean()
    
    # Calculate the ratio of the short-term to moderate-term rolling means to capture mean reversion
    df['mean_reversion_ratio_short_moderate'] = df['short_term_rolling_mean_close'] / df['moderate_term_rolling_mean_close']
    # Calculate the ratio of the moderate-term to long-term rolling means to capture another level of mean reversion
    df['mean_reversion_ratio_moderate_long'] = df['moderate_term_rolling_mean_close'] / df['long_term_rolling_mean_close']
    
    # Momentum indicator using the difference between the current close and the short-term rolling mean, scaled by the current volume
    momentum_factor = (df['close'] - df['short_term_rolling_mean_close']) * df['volume']
    
    # Combine the mean reversion ratios and momentum factor
    alpha_factor = df['mean_reversion_ratio_short_moderate'] + df['mean_reversion_ratio_moderate_long'] + momentum_factor
    
    return alpha_factor
```
