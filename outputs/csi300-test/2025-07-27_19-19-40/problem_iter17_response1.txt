def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the Exponential Moving Average (EMA) of close prices over 10 periods for a more responsive trend indicator
    df['ema_close'] = df['close'].ewm(span=10, adjust=False).mean()
    # Calculate the rolling standard deviation of close prices over 10 periods to capture volatility
    df['rolling_std_close'] = df['close'].rolling(window=10).std()
    # Relative volume is the ratio of today's volume to the average volume over the past 10 periods
    df['average_volume'] = df['volume'].rolling(window=10).mean()
    df['relative_volume'] = df['volume'] / df['average_volume']
    # A novel alpha factor combining EMA, relative volume, and volatility
    alpha_factor = (df['close'] - df['ema_close']) * df['relative_volume'] / (df['rolling_std_close'] + 1e-7)
    return alpha_factor
