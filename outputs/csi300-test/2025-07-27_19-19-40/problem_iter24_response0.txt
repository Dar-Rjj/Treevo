```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 10 periods to capture trend information
    df['rolling_mean_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate a rolling standard deviation of close prices over 10 periods to capture volatility
    df['rolling_std_close'] = df['close'].rolling(window=10).std()
    
    # A momentum and volatility adjusted factor using the difference between the current close and the rolling mean, 
    # scaled by the current volume and normalized by the rolling standard deviation
    alpha_factor = (df['close'] - df['rolling_mean_close']) * df['volume'] / (df['rolling_std_close'] + 1e-6)
    
    return alpha_factor
```
