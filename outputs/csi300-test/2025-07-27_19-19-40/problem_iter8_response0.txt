```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the rolling mean of close prices over 10 periods to capture the trend
    df['rolling_mean_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate the rolling standard deviation of close prices over 10 periods to capture volatility
    df['rolling_std_close'] = df['close'].rolling(window=10).std()
    
    # Calculate the rolling mean of volume over 10 periods to understand the average trading activity
    df['rolling_mean_volume'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the relative volume as the ratio of today's volume to the average volume
    df['relative_volume'] = df['volume'] / (df['rolling_mean_volume'] + 1e-7)
    
    # A momentum indicator using the difference between the current close and the average close, scaled by the relative volume and adjusted for volatility
    alpha_factor = (df['close'] - df['rolling_mean_close']) * df['relative_volume'] / (df['rolling_std_close'] + 1e-7)
    
    return alpha_factor
```
