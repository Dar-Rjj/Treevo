```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling median of close prices over 15 periods to capture trend information with less sensitivity to outliers
    df['rolling_median_close'] = df['close'].rolling(window=15).median()
    # A momentum indicator using the difference between the current close and the rolling median close, scaled by the square root of the current volume for non-linear scaling
    alpha_factor = (df['close'] - df['rolling_median_close']) * np.sqrt(df['volume'])
    return alpha_factor
```
