```python
import pandas as pd
from sklearn.ensemble import RandomForestRegressor
from vaderSentiment.vaderSentiment import SentimentIntensityAnalyzer

def heuristics_v2(df: pd.DataFrame, sentiment_data: pd.Series) -> pd.Series:
    # Multi-period RSI
    def rsi(series, periods=14):
        delta = series.diff()
        gain = (delta.where(delta > 0, 0)).fillna(0)
        loss = (-delta.where(delta < 0, 0)).fillna(0)
        
        avg_gain = gain.rolling(window=periods, min_periods=1).mean()
        avg_loss = loss.rolling(window=periods, min_periods=1).mean()
        
        rs = avg_gain / avg_loss
        return 100 - (100 / (1 + rs))
    
    # Dynamic MACD
    def dynamic_macd(series, fast=12, slow=26, signal=9):
        ema_fast = series.ewm(span=fast, adjust=False).mean()
        ema_slow = series.ewm(span=slow, adjust=False).mean()
        macd_line = ema_fast - ema_slow
        signal_line = macd_line.ewm(span=signal, adjust=False).mean()
        return macd_line - signal_line
    
    # Weighted VWAP
    def weighted_vwap(high, low, close, volume, window=10):
        typical_price = (high + low + close) / 3
        vwap = (typical_price * volume).rolling(window=window).sum() / volume.rolling(window=window).sum()
        return vwap
    
    # Market Regime Detection
    def detect_market_regime(series, window=50):
        rolling_mean = series.rolling(window=window).mean()
        rolling_std = series.rolling(window=window).std()
        regime = (series > rolling_mean + 2 * rolling_std).astype(int)
        return regime
    
    # Sentiment Analysis
    analyzer = SentimentIntensityAnalyzer()
    sentiment_scores = sentiment_data.apply(lambda x: analyzer.polarity_scores(x)['compound'])
    
    # Calculate the indicators
    df['rsi_14'] = rsi(df['close'])
    df['macd_hist'] = dynamic_macd(df['close'])
    df['vwap_10'] = weighted_vwap(df['high'], df['low'], df['close'], df['volume'])
    df['market_regime'] = detect_market_regime(df['close'])
    
    # Machine Learning for Adaptive Weights
    X = df[['rsi_14', 'macd_hist', 'vwap_10', 'market_regime']]
    y = df['close'].shift(-1) - df['close']
    model = RandomForestRegressor(n_estimators=100, random_state=42)
    model.fit(X.iloc[:-1], y.dropna())
    adaptive_weights = model.predict(X)
    
    # Alpha factor
    alpha_factor = (df['rsi_14'] * adaptive_weights + 
                    df['macd_hist'] * adaptive_weights + 
                    df['vwap_10'] * adaptive_weights + 
                    sentiment_scores * adaptive_weights)
    
    return alpha_factor
```
