```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the rolling mean and standard deviation of close prices over 10 and 20 periods to capture trend and volatility information
    df['rolling_mean_close_10'] = df['close'].rolling(window=10).mean()
    df['rolling_std_close_20'] = df['close'].rolling(window=20).std()

    # A momentum indicator using the difference between the current close and the average close over the past 10 periods, adjusted by the current volume
    momentum_factor = (df['close'] - df['rolling_mean_close_10']) * df['volume']

    # A mean reversion factor using the z-score over the last 20 periods, scaled by the inverse of the current volume to favor less liquid stocks
    mean_reversion_factor = ((df['close'] - df['close'].rolling(window=20).mean()) / df['rolling_std_close_20']) / df['volume']

    # Calculate the average true range (ATR) over a 14-period window as a measure of market volatility
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x['high'], x['close'].shift(1)) - min(x['low'], x['close'].shift(1)), axis=1)
    df['atr_14'] = df['tr'].rolling(window=14).mean()

    # A factor that incorporates the ATR, which can help in identifying potential breakouts or reversals
    atr_factor = (df['close'] - df['close'].shift(1)) / df['atr_14']

    # Incorporate the amount traded as a liquidity measure, which can be useful in identifying significant trading activity
    liquidity_factor = df['amount'] / df['volume']

    # Combine the momentum, mean reversion, ATR, and liquidity factors, with a focus on both trend continuation and potential reversals
    alpha_factor = momentum_factor + mean_reversion_factor + atr_factor + liquidity_factor

    return alpha_factor
```
