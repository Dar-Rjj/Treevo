```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 30 periods for an even longer-term trend
    df['rolling_mean_close_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate a rolling standard deviation of close prices over 30 periods to measure long-term volatility
    df['rolling_std_close_30'] = df['close'].rolling(window=30).std()
    
    # Calculate a rolling mean of volume over 15 periods to capture the average trading activity
    df['rolling_mean_volume_15'] = df['volume'].rolling(window=15).mean()
    
    # Calculate the relative volume as the ratio of today's volume to the 15-period rolling mean volume
    df['relative_volume_15'] = df['volume'] / (df['rolling_mean_volume_15'] + 1e-7)
    
    # An alpha factor combining the difference between the current close and the 30-period rolling mean, 
    # adjusted by the 30-period rolling standard deviation, and scaled by the relative volume
    alpha_factor = ((df['close'] - df['rolling_mean_close_30']) / (df['rolling_std_close_30'] + 1e-7)) * df['relative_volume_15']
    
    return alpha_factor
```
