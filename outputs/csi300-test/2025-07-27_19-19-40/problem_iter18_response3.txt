```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the Relative Strength Index (RSI) over a 14-period window to measure the speed and change of price movements
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the Moving Average Convergence Divergence (MACD) using 12 and 26 periods for the fast and slow EMAs, respectively
    ema_fast = df['close'].ewm(span=12, adjust=False).mean()
    ema_slow = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_fast - ema_slow
    
    # Calculate the rolling mean of close prices over 10 periods to capture trend information
    df['rolling_mean_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate the rolling standard deviation of volume over 10 periods to understand volatility in trading activity
    df['rolling_std_volume'] = df['volume'].rolling(window=10).std()
    
    # A composite alpha factor combining RSI, MACD, and the momentum indicator
    alpha_factor = (df['close'] - df['rolling_mean_close']) * df['volume'] / (df['rolling_std_volume'] + 1e-7) + rsi + macd
    
    return alpha_factor
```
