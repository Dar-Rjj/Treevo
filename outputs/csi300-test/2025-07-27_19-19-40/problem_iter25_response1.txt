```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the rolling mean of close prices over 10 and 30 periods to capture short and long-term trends
    df['short_rolling_mean_close'] = df['close'].rolling(window=10).mean()
    df['long_rolling_mean_close'] = df['close'].rolling(window=30).mean()
    
    # Calculate the momentum factor as the difference between the short and long rolling means, scaled by the current volume
    momentum_factor = (df['short_rolling_mean_close'] - df['long_rolling_mean_close']) * df['volume']
    
    # Calculate the standard deviation of the close prices over a 10-period window to measure volatility
    df['volatility'] = df['close'].rolling(window=10).std()
    
    # Calculate the mean reversion factor as the negative of the difference between the current close and the long-term rolling mean, divided by the volatility
    mean_reversion_factor = -(df['close'] - df['long_rolling_mean_close']) / df['volatility']
    
    # Combine the momentum and mean reversion factors
    alpha_factor = momentum_factor + mean_reversion_factor
    
    return alpha_factor
```
