```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close for momentum signal
    df['price_diff'] = df['close'].diff(1)
    # Calculate the volume change to identify volume shocks
    df['volume_change'] = df['volume'].diff(1)
    # Generate a factor that combines price movement with volume changes, capturing both momentum and trading activity spikes
    alpha_factor = df['price_diff'] * (df['volume_change'] + 1e-7)  # Adding a small constant to avoid division by zero
    return alpha_factor
```
