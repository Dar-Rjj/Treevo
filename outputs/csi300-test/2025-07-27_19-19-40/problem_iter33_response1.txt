def heuristics_v2(df: pd.DataFrame) -> pd.Series:

    # Multi-period RSI with adaptive parameters
    def rsi(series, periods=14):
        delta = series.diff()
        gain = (delta.where(delta > 0, 0)).fillna(0)
        loss = (-delta.where(delta < 0, 0)).fillna(0)
        
        avg_gain = gain.rolling(window=periods, min_periods=1).mean()
        avg_loss = loss.rolling(window=periods, min_periods=1).mean()
        
        rs = avg_gain / avg_loss
        return 100 - (100 / (1 + rs))
    
    # Dynamic MACD with adaptive parameters
    def dynamic_macd(series, fast=12, slow=26, signal=9):
        ema_fast = series.ewm(span=fast, adjust=False).mean()
        ema_slow = series.ewm(span=slow, adjust=False).mean()
        macd_line = ema_fast - ema_slow
        signal_line = macd_line.ewm(span=signal, adjust=False).mean()
        return macd_line - signal_line
    
    # Weighted VWAP with adaptive parameters
    def weighted_vwap(high, low, close, volume, window=10):
        typical_price = (high + low + close) / 3
        vwap = (typical_price * volume).rolling(window=window).sum() / volume.rolling(window=window).sum()
        return vwap
    
    # Calculate the indicators
    df['rsi_14'] = rsi(df['close'])
    df['macd_hist'] = dynamic_macd(df['close'])
    df['vwap_10'] = weighted_vwap(df['high'], df['low'], df['close'], df['volume'])

    # Adaptive weights for volatility and liquidity
    df['volatility'] = df['close'].rolling(window=10).std()
    df['liquidity'] = df['volume'].rolling(window=10).mean()
    total_volatility = df['volatility'].sum()
    total_liquidity = df['liquidity'].sum()
    volatility_weight = df['volatility'] / total_volatility
    liquidity_weight = df['liquidity'] / total_liquidity
    adaptive_weight = (volatility_weight + liquidity_weight) / 2

    # Additional factor: Volume-Weighted Momentum
    def volume_weighted_momentum(close, volume, window=10):
        price_change = close.pct_change(window)
        weighted_momentum = (price_change * volume).rolling(window=window).mean()
        return weighted_momentum

    df['vwm_10'] = volume_weighted_momentum(df['close'], df['volume'])

    # Combine factors
    alpha_factor = (df['rsi_14'] * adaptive_weight +
                    df['macd_hist'] * adaptive_weight +
                    df['vwap_10'] * adaptive_weight +
                    df['vwm_10'] * adaptive_weight)

    return alpha_factor
