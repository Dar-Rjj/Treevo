```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the short-term momentum as the difference between current close and close 1 period ago
    df['momentum'] = df['close'].diff(1)
    
    # Integrate price changes over a 5-period window for smoothness
    df['price_change_5'] = df['close'].pct_change(periods=5)
    
    # Calculate relative volume as the ratio of today's volume to the average volume over the last 10 days
    df['relative_volume'] = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Combine the momentum, price change, and relative volume into a composite alpha factor
    alpha_factor = df['momentum'] + (df['price_change_5'] * df['relative_volume'])
    
    return alpha_factor
```
