```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean and standard deviation of close prices over 20 periods to capture trend and volatility information
    df['rolling_mean_close'] = df['close'].rolling(window=20).mean()
    df['rolling_std_close'] = df['close'].rolling(window=20).std()
    
    # Smooth the volume using an exponentially weighted moving average (EWMA) with a span of 10
    df['ewma_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # A combined momentum and volatility factor: the difference between the current close and the rolling mean,
    # scaled by the smoothed volume and adjusted for volatility
    alpha_factor = ((df['close'] - df['rolling_mean_close']) / df['rolling_std_close']) * df['ewma_volume']
    
    return alpha_factor
```
