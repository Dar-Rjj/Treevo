```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 20 periods for longer-term trend
    df['rolling_mean_close_20'] = df['close'].rolling(window=20).mean()
    # Calculate a rolling standard deviation of volume over 20 periods to capture long-term volatility
    df['rolling_std_volume_20'] = df['volume'].rolling(window=20).std()
    # Calculate the difference between the current close and the average close over the past 20 periods
    df['close_diff_20'] = df['close'] - df['rolling_mean_close_20']
    # Calculate the relative volume, scaling by the long-term standard deviation of volume
    df['relative_volume_20'] = df['volume'] / (df['rolling_std_volume_20'] + 1e-7)
    # Alpha factor: momentum captured by the difference in close prices, scaled by relative volume
    alpha_factor = df['close_diff_20'] * df['relative_volume_20']
    return alpha_factor
```
