```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a simple moving average (SMA) of close prices over 5 periods to capture short-term trend information
    df['sma_close'] = df['close'].rolling(window=5).mean()
    
    # A momentum indicator using the difference between the current close and the SMA, scaled by the volume
    alpha_factor = (df['close'] - df['sma_close']) * df['volume']
    
    return alpha_factor
```
