```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the rolling mean and standard deviation of close prices over 10 and 20 periods
    df['rolling_mean_close_10'] = df['close'].rolling(window=10).mean()
    df['rolling_std_close_20'] = df['close'].rolling(window=20).std()

    # Momentum factor using the difference between the current close and the average close over the past 10 periods, adjusted by the current volume
    momentum_factor = (df['close'] - df['rolling_mean_close_10']) * df['volume']

    # Mean reversion factor using the z-score over the last 20 periods, scaled by the inverse of the current volume
    mean_reversion_factor = ((df['close'] - df['close'].rolling(window=20).mean()) / df['rolling_std_close_20']) / df['volume']

    # Calculate the Relative Strength Index (RSI) over a 14-period window
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the MACD and its signal line
    exp1 = df['close'].ewm(span=12, adjust=False).mean()
    exp2 = df['close'].ewm(span=26, adjust=False).mean()
    macd = exp1 - exp2
    signal = macd.ewm(span=9, adjust=False).mean()

    # MACD histogram
    macd_hist = macd - signal

    # Combine the momentum, mean reversion, RSI, and MACD factors
    alpha_factor = 0.3 * momentum_factor + 0.2 * mean_reversion_factor + 0.2 * (rsi - 50) + 0.3 * macd_hist

    return alpha_factor
```
