```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 10 periods to capture the short-term trend
    df['short_rolling_mean_close'] = df['close'].rolling(window=10).mean()
    # Calculate a rolling mean of close prices over 60 periods to capture the long-term trend
    df['long_rolling_mean_close'] = df['close'].rolling(window=60).mean()
    # Calculate a rolling standard deviation of volume over 10 periods to understand volatility in trading activity
    df['rolling_std_volume'] = df['volume'].rolling(window=10).std()
    # A momentum indicator using the difference between the current close and the average close over the past 10 and 60 periods, scaled by the relative volume
    alpha_factor = ((df['close'] - df['short_rolling_mean_close']) + (df['close'] - df['long_rolling_mean_close'])) * df['volume'] / (df['rolling_std_volume'] + 1e-7)
    return alpha_factor
```
