```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 5 periods to capture short-term trend information
    df['rolling_mean_close'] = df['close'].rolling(window=5).mean()
    # A mean reversion indicator using the difference between the current close and the average close over the past 5 periods, scaled by the current volume
    alpha_factor = (df['rolling_mean_close'] - df['close']) * df['volume']
    return alpha_factor
```
