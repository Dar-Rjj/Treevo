```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate Relative Strength Index (RSI) with a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    RS = gain / loss
    RSI = 100 - (100 / (1 + RS))
    
    # Calculate Moving Average Convergence Divergence (MACD) with 12 and 26 periods
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    MACD_line = short_ema - long_ema
    signal_line = MACD_line.ewm(span=9, adjust=False).mean()
    MACD_histogram = MACD_line - signal_line
    
    # Volume-Weighted Price Change from open to close
    volume_weighted_price_change = (df['close'] - df['open']) * df['volume']
    
    # Calculate 10-day rolling volatility
    volatility = df['close'].rolling(window=10).std()
    
    # Calculate 10-day rolling liquidity
    liquidity = df['volume'].rolling(window=10).mean()
    
    # Adaptive weights based on inverse of volatility and liquidity
    rsi_weight = 1 / (volatility + 1e-8)
    macd_weight = 1 / (volatility + 1e-8)
    vwpc_weight = 1 / (liquidity + 1e-8)
    
    # Additional market signals
    # Momentum factor: 10-day return
    momentum = df['close'].pct_change(10)
    
    # Money Flow Index (MFI) with a 14-day period
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    MFI = 100 - (100 / (1 + (positive_money_flow_sum / negative_money_flow_sum)))
    
    # Adaptive weights for new factors
    momentum_weight = 1 / (volatility + 1e-8)
    mfi_weight = 1 / (volatility + 1e-8)
    
    # Combine the factors using adaptive weights
    alpha_factor = (
        (rsi_weight * RSI) +
        (macd_weight * MACD_histogram) +
        (vwpc_weight * volume_weighted_price_change) +
        (momentum_weight * momentum) +
        (mfi_weight * MFI)
    ) / (rsi_weight + macd_weight + vwpc_weight + momentum_weight + mfi_weight)
    
    return alpha_factor
```
