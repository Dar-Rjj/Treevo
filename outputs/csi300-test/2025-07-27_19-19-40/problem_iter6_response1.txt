```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between current close and previous close to capture short-term momentum
    df['price_change'] = df['close'].diff()
    
    # Calculate a rolling mean of price changes over 3 periods to smooth out noise
    df['rolling_mean_price_change'] = df['price_change'].rolling(window=3).mean()
    
    # Calculate a rolling standard deviation of volume over 3 periods to understand volatility in trading activity
    df['rolling_std_volume'] = df['volume'].rolling(window=3).std()
    
    # Calculate a rolling standard deviation of price changes over 3 periods to understand price volatility
    df['rolling_std_price_change'] = df['price_change'].rolling(window=3).std()
    
    # A momentum indicator using the smoothed price change, scaled by the relative volume and adjusted for price volatility
    alpha_factor = (df['price_change'] - df['rolling_mean_price_change']) * df['volume'] / (df['rolling_std_volume'] + 1e-7) / (df['rolling_std_price_change'] + 1e-7)
    
    return alpha_factor
```
