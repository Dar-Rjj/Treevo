```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate rolling means of close prices over different windows to capture trends
    df['rolling_mean_close_5'] = df['close'].rolling(window=5).mean()
    df['rolling_mean_close_10'] = df['close'].rolling(window=10).mean()
    df['rolling_mean_close_20'] = df['close'].rolling(window=20).mean()

    # Calculate rolling standard deviations of volume over different windows to understand volatility
    df['rolling_std_volume_5'] = df['volume'].rolling(window=5).std()
    df['rolling_std_volume_10'] = df['volume'].rolling(window=10).std()
    df['rolling_std_volume_20'] = df['volume'].rolling(window=20).std()

    # Momentum indicators using the difference between the current close and the average close, scaled by the relative volume
    alpha_factor_5 = (df['close'] - df['rolling_mean_close_5']) * df['volume'] / (df['rolling_std_volume_5'] + 1e-7)
    alpha_factor_10 = (df['close'] - df['rolling_mean_close_10']) * df['volume'] / (df['rolling_std_volume_10'] + 1e-7)
    alpha_factor_20 = (df['close'] - df['rolling_mean_close_20']) * df['volume'] / (df['rolling_std_volume_20'] + 1e-7)

    # Combine the different momentum indicators using a non-linear transformation
    combined_momentum = (alpha_factor_5 ** 2 + alpha_factor_10 ** 2 + alpha_factor_20 ** 2) ** 0.5

    # Price change ratios over the same adaptive windows
    df['price_change_ratio_5'] = df['close'] / df['close'].shift(5) - 1
    df['price_change_ratio_10'] = df['close'] / df['close'].shift(10) - 1
    df['price_change_ratio_20'] = df['close'] / df['close'].shift(20) - 1

    # Average price change ratio
    average_price_change_ratio = (df['price_change_ratio_5'] + df['price_change_ratio_10'] + df['price_change_ratio_20']) / 3

    # Combine the combined momentum with the average price change ratio
    alpha_factor = combined_momentum + average_price_change_ratio

    return alpha_factor
```
