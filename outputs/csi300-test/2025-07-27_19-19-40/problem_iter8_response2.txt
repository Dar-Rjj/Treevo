```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of close prices over 10 periods to capture the trend
    df['rolling_mean_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate a rolling standard deviation of close prices over 10 periods to capture volatility
    df['rolling_std_close'] = df['close'].rolling(window=10).std()
    
    # Calculate a rolling standard deviation of volume over 10 periods to understand volatility in trading activity
    df['rolling_std_volume'] = df['volume'].rolling(window=10).std()
    
    # A momentum indicator using the difference between the current close and the average close over the past 10 periods
    df['momentum_indicator'] = df['close'] - df['rolling_mean_close']
    
    # Scale the momentum indicator by the relative volume (current volume divided by the average volume over the past 10 periods)
    df['relative_volume'] = df['volume'] / (df['rolling_std_volume'] + 1e-7)
    
    # Combine the scaled momentum with the rolling standard deviation of close prices to form the alpha factor
    alpha_factor = df['momentum_indicator'] * df['relative_volume'] + df['rolling_std_close']
    
    return alpha_factor
```
