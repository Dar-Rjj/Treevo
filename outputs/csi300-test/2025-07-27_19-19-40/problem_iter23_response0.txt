```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the short-term (5-day) and moderate (20-day) rolling means of close prices to capture different trends
    df['short_rolling_mean_close'] = df['close'].rolling(window=5).mean()
    df['moderate_rolling_mean_close'] = df['close'].rolling(window=20).mean()

    # A momentum indicator using the difference between the current close and the short-term mean, scaled by the current volume
    short_momentum = (df['close'] - df['short_rolling_mean_close']) * df['volume']
    
    # A momentum indicator using the difference between the current close and the moderate mean, scaled by the current volume
    moderate_momentum = (df['close'] - df['moderate_rolling_mean_close']) * df['volume']

    # Mean reversion factor: the ratio of the current close to the moderate rolling mean
    mean_reversion_ratio = df['close'] / df['moderate_rolling_mean_close']

    # Alpha factor combining short-term and moderate momentum, with a focus on mean reversion
    alpha_factor = 0.5 * short_momentum + 0.5 * moderate_momentum - mean_reversion_ratio

    return alpha_factor
```
