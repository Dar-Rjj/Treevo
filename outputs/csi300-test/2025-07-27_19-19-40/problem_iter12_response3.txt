```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a 5-period rolling mean of close prices to capture short-term trend
    df['rolling_mean_close_5'] = df['close'].rolling(window=5).mean()
    
    # Calculate the 5-period rolling standard deviation of volume to understand short-term volatility in trading activity
    df['rolling_std_volume_5'] = df['volume'].rolling(window=5).std()
    
    # A momentum indicator using the difference between the current close and the 5-period average close, 
    # adjusted by the 5-period rolling standard deviation of volume
    alpha_factor = (df['close'] - df['rolling_mean_close_5']) / (df['rolling_std_volume_5'] + 1e-7)
    
    return alpha_factor
```
