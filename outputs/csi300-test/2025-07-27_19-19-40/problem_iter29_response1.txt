```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the Relative Strength Index (RSI) for a 14-day period to gauge overbought/oversold conditions
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the Moving Average Convergence Divergence (MACD) to identify trend momentum and potential reversals
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.rolling(window=9).mean()
    macd_histogram = macd_line - signal_line
    
    # Volume-weighted price change to integrate volume and price movement
    vwpc = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Combine RSI, MACD, and Volume-Weighted Price Change with adaptive weights
    alpha_factor = 0.5 * rsi + 0.3 * macd_histogram + 0.2 * vwpc

    return alpha_factor
```
