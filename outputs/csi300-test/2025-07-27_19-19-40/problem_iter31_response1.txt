def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    import pandas as pd
    
    # Calculate the 14-period RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 12 and 26 period EMAs for MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_histogram = macd_line - signal_line
    
    # Calculate the weighted VWAP
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Dynamic weighting based on volatility and liquidity
    volatility = df['close'].rolling(window=10).std()
    liquidity = df['volume'].rolling(window=10).mean()
    weight_rsi = volatility / (volatility + liquidity)
    weight_macd = liquidity / (volatility + liquidity)
    weight_vwap = 1 - (weight_rsi + weight_macd)
    
    # Combine the indicators into a single alpha factor
    alpha_factor = (rsi * weight_rsi) + (macd_histogram * weight_macd) + (vwap * weight_vwap)
    
    return alpha_factor
