```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = (df['high'] - df['low']) * df['volume']
    
    # Incorporate Historical Price Change
    price_change = df['close'].pct_change()
    
    # Smooth the Resulting Series
    ema_alpha = 0.1
    sma_window = 5
    smoothed_price_change = price_change.ewm(alpha=ema_alpha, adjust=False).mean().rolling(window=sma_window).mean()
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    adjusted_factor = smoothed_price_change * (1 + np.sign(volume_change) * 0.1)
    
    # Adjust for Amount Trend
    amount_change = df['amount'].pct_change()
    final_factor = adjusted_factor * (1 + np.sign(volume_change + amount_change) * 0.1)
    
    # Combine and Smooth Final Factor
    final_factor_smoothed = final_factor.rolling(window=sma_window).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = price_change * volume_trend_ratio
    final_momentum = intraday_momentum + long_term_momentum
    
    # Combine Momentum and Volatility
    combined_momentum = final_momentum - high_low_spread.rolling(window=sma_window).mean()
    
    # Final Alpha Factor
    alpha_factor = combined_momentum * vwap + volume_weighted_high_low_range.rolling(window=5).sum() + final_factor_smoothed
    return alpha_factor
```
