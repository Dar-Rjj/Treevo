```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the closing price within the daily range
    df['daily_range'] = df['high'] - df['low']
    df['close_pos'] = (df['close'] - df['low']) / df['daily_range']

    # Calculate the money flow by multiplying the typical price by the volume
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']

    # Calculate a simple moving average of the money flow
    df['mf_sma'] = df['money_flow'].rolling(window=14).mean()

    # Compute the factor value as the ratio of the close position to the smoothed money flow
    df['factor'] = df['close_pos'] / df['mf_sma']

    # Return the factor as a pandas Series
    return df['factor']
```
