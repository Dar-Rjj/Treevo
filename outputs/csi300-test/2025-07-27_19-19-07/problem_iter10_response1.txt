```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Return
    close_to_prev_close_return = df['close'].pct_change()
    
    # Multiply High-Low Range by Close-to-Previous Close Return
    range_return_product = high_low_range * close_to_prev_close_return
    
    # Smooth the Resulting Series
    smoothed_series = range_return_product.rolling(window=5).mean()
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    adjusted_factor = smoothed_series * (1 + volume_change)
    
    # Combine and Smooth Final Factor
    final_factor_smoothed = adjusted_factor.rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Aggregate High-Low Price Bandwidth
    high_low_bandwidth_aggregated = high_low_range.rolling(window=5).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    momentum_component = (intraday_return + open_to_close_return) * close_to_prev_close_return * volume_trend_ratio
    
    # Compute Volatility Component
    volatility_intraday = intraday_return.rolling(window=5).std()
    volatility_open_to_close = open_to_close_return.rolling(window=5).std()
    total_volatility = (volatility_intraday + volatility_open_to_close) / 2
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = momentum_component - total_volatility
    
    # Combine and Smooth Combined Momentum and Volatility
    final_combined_smoothed = combined_momentum_volatility.rolling(window=5).mean()
    
    # Final Alpha Factor
    alpha_factor = final_combined_smoothed * vwap + high_low_bandwidth_aggregated
    
    return alpha_factor
```
