```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    price_change = df['close'].pct_change()
    
    # Smooth the Resulting Series
    ema_alpha = 0.5
    sma_window = 10
    smoothed_series = price_change.ewm(alpha=ema_alpha).mean().rolling(window=sma_window).mean()
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    adjusted_smoothed_series = smoothed_series * (1 + np.sign(volume_change) * np.abs(volume_change))
    
    # Adjust for Amount Trend
    amount_change = df['amount'].pct_change()
    final_factor = adjusted_smoothed_series * (1 + np.sign(amount_change) * np.abs(amount_change))
    
    # Combine and Smooth Final Factor
    final_factor = final_factor.rolling(window=sma_window).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_volume = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume / long_term_volume
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    adjusted_close_price_change = price_change * volume_trend_ratio
    long_term_momentum = adjusted_close_price_change
    intraday_momentum = intraday_return + open_close_return
    final_momentum = long_term_momentum + intraday_momentum
    
    # Compute Volatility Component
    intraday_volatility = intraday_return.rolling(window=5).std()
    open_close_volatility = open_close_return.rolling(window=5).std()
    volatility_component = intraday_volatility + open_close_volatility
    
    # Combine Momentum and Volatility
    combined_momentum = final_momentum - 0.5 * volatility_component
    
    # Final Alpha Factor
    alpha_factor = combined_momentum * vwap + high_low_spread.rolling(window=5).sum() + final_factor.rolling(window=sma_window).mean()
    
    return alpha_factor
```
