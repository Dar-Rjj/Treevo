```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['Volume_Weighted_High_Low'] = (df['high'] - df['low']) * df['volume']
    
    # Incorporate Historical Price Change
    df['Close_to_Close_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['High_Low_Price_Momentum'] = df['Volume_Weighted_High_Low'] * df['Close_to_Close_Return']
    
    # Adjust for Volume and Amount Trend
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Amount_Change'] = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    
    # Final Factor Calculation
    def adjust_factor(row):
        if row['Volume_Change'] > 0 and row['Amount_Change'] > 0:
            return row['High_Low_Price_Momentum'] * 1.5
        elif row['Volume_Change'] > 0 or row['Amount_Change'] > 0:
            return row['High_Low_Price_Momentum'] * 1.2
        else:
            return row['High_Low_Price_Momentum'] * 0.8
    
    df['Final_Factor'] = df.apply(adjust_factor, axis=1)
    
    # Smooth the Resulting Series
    df['EMA_Smooth'] = df['Final_Factor'].ewm(span=10, adjust=False).mean()
    df['SMA_Smooth'] = df['Final_Factor'].rolling(window=20).mean()
    
    # Calculate Intraday and Open-to-Close Returns
    df['Intraday_Return'] = df['high'] - df['low']
    df['Open_to_Close_Return'] = df['open'] - df['close']
    
    # Compute Short-Term and Long-Term Volume Trends
    df['Short_Term_Volume_Trend'] = df['volume'].rolling(window=5).mean()
    df['Long_Term_Volume_Trend'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    df['Volume_Trend_Ratio'] = df['Short_Term_Volume_Trend'] / df['Long_Term_Volume_Trend']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Momentum and Volatility
    df['Intraday_Momentum'] = df['Intraday_Return'] + df['Open_to_Close_Return']
    df['Long_Term_Momentum'] = df['Close_to_Close_Return'] * df['Volume_Trend_Ratio']
    df['Weighted_Volatility'] = df['Intraday_Return'].std() + df['Open_to_Close_Return'].std()
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Intraday_Momentum'] + df['Long_Term_Momentum'] - df['Weighted_Volatility']) * df['VWAP'] + df['High_Low_Price_Momentum'] + df['SMA_Smooth']
    
    return df['Alpha_Factor']
```
