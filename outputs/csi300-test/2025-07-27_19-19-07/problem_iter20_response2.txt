```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['Volume_Weighted_HL_Range'] = (df['high'] - df['low']) * df['volume']
    
    # Calculate Close-to-Previous Close Return
    df['Close_to_Prev_Close_Return'] = df['close'].pct_change()
    
    # Multiply Volume-Weighted High-Low Range by Close-to-Previous Close Return
    df['Momentum_Adjusted'] = df['Volume_Weighted_HL_Range'] * df['Close_to_Prev_Close_Return']
    
    # Smooth the Resulting Series using Exponential Moving Average
    df['Smoothed_Momentum'] = df['Momentum_Adjusted'].ewm(alpha=0.1).mean()
    
    # Adjust for Volume Trend
    df['Volume_Change'] = df['volume'].pct_change()
    df['Volume_Adjusted_Factor'] = df['Smoothed_Momentum'] * df['Volume_Change']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['Open_to_Close_Return'] = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend (last 5 days)
    df['Short_Term_Volume_Trend'] = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend (last 20 days)
    df['Long_Term_Volume_Trend'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    df['Volume_Trend_Ratio'] = df['Short_Term_Volume_Trend'] / df['Long_Term_Volume_Trend']
    
    # Calculate High-Low Price Bandwidth
    df['HL_Price_Bandwidth'] = df['high'] - df['low']
    
    # Aggregate High-Low Price Bandwidth over last 5 days
    df['Aggregated_HL_Bandwidth'] = df['HL_Price_Bandwidth'].rolling(window=5).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df[['high', 'low', 'close']].sum(axis=1) / 3) * df['volume']
    df['VWAP'] = df['VWAP'].cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    df['Intraday_Momentum'] = df['Intraday_Return'] + df['Open_to_Close_Return']
    df['Long_Term_Momentum'] = df['close'].pct_change() * df['Volume_Trend_Ratio']
    
    # Combine Momentum and Volatility
    df['Combined_Momentum'] = df['Intraday_Momentum'] + df['Long_Term_Momentum']
    
    # Smooth Combined Momentum and Volatility
    df['Smoothed_Combined_Momentum'] = df['Combined_Momentum'].rolling(window=10).mean()
    
    # Incorporate the Intraday Price Reversal into the Final Alpha Factor
    df['Intraday_Price_Reversal'] = (df['high'] - df['low']) - df['Intraday_Return']
    df['Final_Alpha_Factor'] = df['Smoothed_Combined_Momentum'] * df['VWAP'] + df['Aggregated_HL_Bandwidth'] * df['Intraday_Price_Reversal']
    
    return df['Final_Alpha_Factor']
```
