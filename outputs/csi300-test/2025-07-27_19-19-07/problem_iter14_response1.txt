```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']

    # Incorporate Historical Price Change
    close_to_close_return = df['close'].pct_change()
    price_change_factor = high_low_spread * close_to_close_return

    # Smooth the Resulting Series
    ema_alpha = 0.3  # Specific alpha for EMA
    sma_window = 5  # Specific window size for SMA
    smoothed_price_change_factor = price_change_factor.ewm(alpha=ema_alpha).mean().rolling(window=sma_window).mean()

    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    volume_adjusted_factor = smoothed_price_change_factor * (1 + volume_change)

    # Combine and Smooth Final Factor
    final_smoothed_factor = volume_adjusted_factor.rolling(window=sma_window).mean()

    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']

    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']

    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()

    # Calculate Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()

    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend

    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    vwap = vwap / df['volume']

    # Compute Final Momentum Component
    adjusted_close_price_change = df['close'].pct_change() * volume_trend_ratio
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = adjusted_close_price_change
    final_momentum = intraday_momentum + long_term_momentum

    # Compute Volatility Component
    intraday_volatility = intraday_return.rolling(window=sma_window).std()
    open_to_close_volatility = open_to_close_return.rolling(window=sma_window).std()
    total_volatility = (intraday_volatility + open_to_close_volatility) / 2

    # Combine Momentum and Volatility
    combined_momentum = final_momentum - total_volatility

    # Aggregate High-Low Price Bandwidth
    high_low_bandwidth_aggregated = high_low_spread.rolling(window=5).sum()

    # Add Smoothed and Volume-Adjusted Momentum Factor
    final_alpha_factor = combined_momentum * vwap + high_low_bandwidth_aggregated + final_smoothed_factor

    return final_alpha_factor
```
