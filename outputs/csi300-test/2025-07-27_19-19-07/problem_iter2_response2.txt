```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Close Price Change
    df['close_change'] = df['close'].diff()
    
    # Compute Short-Term Volume Trend
    df['short_term_vol_avg'] = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    df['long_term_vol_avg'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    df['vol_trend_ratio'] = df['short_term_vol_avg'] / df['long_term_vol_avg']
    
    # Calculate High-Low Price Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    df['close_to_close_return'] = df['close'].pct_change()
    df['price_momentum'] = df['high_low_spread'] * df['close_to_close_return']
    
    # Combine Momentum and Volume Factors
    df['combined_factor'] = (df['close_change'] * df['vol_trend_ratio']) + df['price_momentum']
    
    # Adjust for Volume Change
    df['volume_change'] = df['volume'].pct_change()
    df['final_factor'] = df['combined_factor'] * df['volume_change'].apply(lambda x: 1 + abs(x) if x > 0 else 1 - abs(x))
    
    return df['final_factor']
```
