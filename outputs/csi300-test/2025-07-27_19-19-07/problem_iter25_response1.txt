```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['vol_weighted_high_low_range'] = (df['high'] - df['low']) * df['volume']

    # Incorporate Historical Price Change
    df['close_to_close_return'] = df['close'].pct_change()
    df['price_adj_vol_weighted_range'] = df['vol_weighted_high_low_range'] * df['close_to_close_return']

    # Adjust for Volume Trend
    df['volume_change'] = df['volume'].pct_change()
    df['vol_adj_factor'] = np.where(df['volume_change'] > 0, 1.5, 0.5)
    df['price_adj_vol_weighted_range'] *= df['vol_adj_factor']

    # Smooth the Resulting Series
    df['ema_smoothed'] = df['price_adj_vol_weighted_range'].ewm(span=5).mean()
    df['sma_smoothed'] = df['price_adj_vol_weighted_range'].rolling(window=5).mean()

    # Adjust for Amount Trend
    df['amount_change'] = df['amount'].pct_change()
    df['amount_adj_factor'] = np.where((df['volume_change'] > 0) & (df['amount_change'] > 0), 2,
                                       np.where((df['volume_change'] > 0) | (df['amount_change'] > 0), 1, 0.5))
    df['final_smoothed'] = df['sma_smoothed'] * df['amount_adj_factor']

    # Combine and Smooth Final Factor
    df['final_factor'] = df['final_smoothed'].rolling(window=5).mean()

    # Calculate Intraday and Open-to-Close Returns
    df['intraday_return'] = df['high'] - df['low']
    df['open_to_close_return'] = df['open'] - df['close']

    # Compute Short-Term and Long-Term Volume Trends
    df['short_term_volume_trend'] = df['volume'].rolling(window=5).mean()
    df['long_term_volume_trend'] = df['volume'].rolling(window=20).mean()

    # Compute Volume Trend Ratio
    df['volume_trend_ratio'] = df['short_term_volume_trend'] / df['long_term_volume_trend']

    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()

    # Compute Final Momentum Component
    df['intraday_momentum'] = df['intraday_return'] + df['open_to_close_return']
    df['long_term_momentum'] = df['close_to_close_return'] * df['volume_trend_ratio']

    # Compute Volatility Component
    df['intraday_volatility'] = df['intraday_return'].rolling(window=5).std()
    df['open_to_close_volatility'] = df['open_to_close_return'].rolling(window=5).std()

    # Combine Momentum and Volatility
    df['combined_momentum'] = df['intraday_momentum'] + df['long_term_momentum']
    df['weighted_volatility'] = (df['intraday_volatility'] + df['open_to_close_volatility']) / 2
    df['momentum_adjusted'] = df['combined_momentum'] - df['weighted_volatility']

    # Final Alpha Factor
    df['alpha_factor'] = (df['momentum_adjusted'] * df['vwap']) + df['vol_weighted_high_low_range'] + df['final_factor']

    return df['alpha_factor'].dropna()
```
