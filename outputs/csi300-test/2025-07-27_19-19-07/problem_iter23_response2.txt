```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['vol_weighted_high_low_range'] = (df['high'] - df['low']) * df['volume']
    
    # Calculate Close-to-Previous Close Return
    df['close_to_prev_close_return'] = df['close'].pct_change()
    
    # Multiply Volume-Weighted High-Low Range by Close-to-Previous Close Return
    df['initial_factor'] = df['vol_weighted_high_low_range'] * df['close_to_prev_close_return']
    
    # Smooth the Resulting Series using Exponential Moving Average
    alpha = 0.1  # Specific Alpha for EMA
    df['smoothed_series'] = df['initial_factor'].ewm(alpha=alpha).mean()
    
    # Adjust for Volume Trend
    df['volume_change'] = df['volume'].pct_change()
    df['adjusted_factor'] = df['smoothed_series'] * df['volume_change']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    df['short_term_volume_trend'] = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    df['long_term_volume_trend'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    df['volume_trend_ratio'] = df['short_term_volume_trend'] / df['long_term_volume_trend']
    
    # Calculate High-Low Price Bandwidth
    df['high_low_bandwidth'] = df['high'] - df['low']
    
    # Aggregate High-Low Price Bandwidth over last 5 days
    df['aggregated_bandwidth'] = df['high_low_bandwidth'].rolling(window=5).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume'] / df['volume'].rolling(window=1).sum()
    
    # Compute Final Momentum Component
    df['intraday_momentum'] = df['intraday_return'] + df['open_to_close_return']
    df['long_term_momentum'] = df['close'].pct_change() * df['volume_trend_ratio']
    
    # Combine Momentum Components
    df['combined_momentum'] = df['intraday_momentum'] + df['long_term_momentum']
    
    # Combine and Smooth Final Factor
    window_size = 10  # Specific Window Size for SMA
    df['final_factor_smoothed'] = df['combined_momentum'].rolling(window=window_size).mean()
    
    # Incorporate the Intraday Price Reversal into the Final Alpha Factor
    df['intraday_price_reversal'] = df['high'] - df['low'] - df['intraday_return']
    df['final_alpha_factor'] = df['final_factor_smoothed'] * df['vwap'] + df['aggregated_bandwidth'] * df['intraday_price_reversal']
    
    return df['final_alpha_factor']
```
