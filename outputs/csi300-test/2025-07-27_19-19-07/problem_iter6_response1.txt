```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Multiply High-Low Spread by Close-to-Close Return
    high_low_momentum = high_low_spread * close_to_close_return
    
    # Calculate Volume Change
    volume_change = df['volume'].pct_change()
    
    # Adjust for Volume Trend
    adjusted_high_low_momentum = high_low_momentum.where(volume_change > 0, high_low_momentum / abs(volume_change))
    
    # Calculate Amount Change
    amount_change = df['amount'].pct_change()
    
    # Final Factor Calculation based on Volume and Amount Changes
    final_factor = adjusted_high_low_momentum.copy()
    positive_volume_amount = (volume_change > 0) & (amount_change > 0)
    negative_volume_amount = (volume_change < 0) & (amount_change < 0)
    mixed_trend = ((volume_change > 0) & (amount_change < 0)) | ((volume_change < 0) & (amount_change > 0))
    
    final_factor[positive_volume_amount] *= 1.5  # Amplify the factor
    final_factor[negative_volume_amount] /= 1.5  # Dampen the factor
    final_factor[mixed_trend] *= 1.2  # Moderate adjustment
    
    return final_factor
```
