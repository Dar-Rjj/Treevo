```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Multiply High-Low Spread by Close-to-Close Return
    high_low_price_momentum = high_low_spread * close_to_close_return
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    volume_adjusted_momentum = high_low_price_momentum * (1 + volume_change)
    
    # Calculate Cumulative Sum of Weighted Returns over a Moving Window
    window_size = 20
    cumulative_sum = volume_adjusted_momentum.rolling(window=window_size).sum()
    
    # Smooth the Resulting Series with a Simple Moving Average
    smoothed_series = cumulative_sum.rolling(window=window_size).mean()
    
    # Adjust for Amount Trend
    amount_change = df['amount'].pct_change()
    
    # Final Factor Calculation
    final_factor = smoothed_series
    for i in range(1, len(final_factor)):
        if volume_change.iloc[i] > 0 and amount_change.iloc[i] > 0:
            final_factor.iloc[i] *= 1.5
        elif volume_change.iloc[i] > 0 or amount_change.iloc[i] > 0:
            final_factor.iloc[i] *= 1.25
        else:
            final_factor.iloc[i] *= 0.75
    
    # Combine and Smooth Final Factor
    final_factor_smoothed = final_factor.rolling(window=window_size).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    final_momentum_component = intraday_return + open_to_close_return * volume_trend_ratio
    
    # Compute Volatility Component
    volatility_component = (intraday_return.std() + open_to_close_return.std()) / 2
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = final_momentum_component - volatility_component
    
    # Calculate Intraday Volume-Price Interaction
    intraday_volume_price_interaction = intraday_return * df['volume']
    
    # Adjust Final Alpha Factor
    final_alpha_factor = (combined_momentum_volatility * vwap +
                          high_low_spread.rolling(window=window_size).mean() +
                          final_factor_smoothed +
                          intraday_volume_price_interaction)
    
    return final_alpha_factor
```
