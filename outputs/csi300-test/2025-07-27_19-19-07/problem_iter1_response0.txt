```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap_price'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    total_volume = df['volume'].sum()
    vwap = df['vwap_price'].sum() / total_volume
    
    # Compute Final Alpha Factor
    intraday_momentum = df['intraday_return'] + df['open_to_close_return']
    
    # Volatility Component
    std_intraday_return = df['intraday_return'].std()
    std_open_to_close_return = df['open_to_close_return'].std()
    volatility = (std_intraday_return + std_open_to_close_return) / 2
    
    # Combine Components
    alpha_factor = (intraday_momentum * vwap) - (volatility * vwap)
    
    return alpha_factor
```
