```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Close Price Change
    df['close_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Short-Term Volume Trend
    short_term_vol_avg = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    long_term_vol_avg = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    df['vol_trend_ratio'] = short_term_vol_avg / long_term_vol_avg
    
    # Calculate High-Low Price Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    df['close_to_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['price_momentum'] = df['high_low_spread'] * df['close_to_close_return']
    
    # Combine Momentum and Volume Factors
    df['combined_factor'] = df['close_change'] * df['vol_trend_ratio'] + df['price_momentum']
    
    # Adjust for Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['final_factor'] = df['combined_factor'] * (df['volume_change'] > 0).astype(int) * 2 - df['combined_factor'] * (df['volume_change'] <= 0).astype(int)
    
    return df['final_factor']
```
