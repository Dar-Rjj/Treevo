```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    price_change_pct = df['close'].pct_change()
    
    # Smooth the Resulting Series
    smoothed_price_change = price_change_pct.ewm(alpha=0.1).mean()
    
    # Adjust for Volume Trend
    volume_change_pct = df['volume'].pct_change()
    adjusted_smoothed_price_change = smoothed_price_change + (volume_change_pct * 0.5)
    
    # Adjust for Amount Trend
    amount_change_pct = df['amount'].pct_change()
    adjusted_factor = adjusted_smoothed_price_change + (amount_change_pct * 0.5)
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = smoothed_price_change * volume_trend_ratio
    final_momentum = intraday_momentum + long_term_momentum
    
    # Combine Momentum and Volatility
    combined_momentum = final_momentum
    
    # Calculate VWAP
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Final Alpha Factor
    high_low_bandwidth = high_low_spread.rolling(window=5).mean()
    final_alpha_factor = (combined_momentum * vwap) + high_low_bandwidth
    
    return final_alpha_factor
```
