```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    price_change = df['close'].pct_change()
    
    # Smooth the Resulting Series
    ema_alpha = 0.1  # Specific Alpha for EMA
    sma_window = 5  # Specific Window Size for SMA
    smoothed_price_change = price_change.ewm(alpha=ema_alpha).mean().rolling(window=sma_window).mean()
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    adjusted_factor = smoothed_price_change * np.sign(volume_change)
    
    # Combine and Smooth Final Factor
    final_smoothed_factor = adjusted_factor.rolling(window=sma_window).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = price_change * volume_trend_ratio
    final_momentum = intraday_momentum + long_term_momentum
    
    # Compute Volatility Component
    intraday_volatility = intraday_return.rolling(window=5).std()
    open_to_close_volatility = open_to_close_return.rolling(window=5).std()
    total_volatility = intraday_volatility + open_to_close_volatility
    
    # Combine Momentum and Volatility
    combined_momentum = final_momentum - total_volatility
    
    # Introduce New Intraday Strength Indicator
    intraday_strength = (df['high'] - df['low']) / (df['high'] + df['low'])
    smoothed_intraday_strength = intraday_strength.ewm(alpha=ema_alpha).mean()
    combined_momentum += smoothed_intraday_strength
    
    # Final Alpha Factor
    alpha_factor = (combined_momentum * vwap) + high_low_spread.rolling(window=5).sum() + final_smoothed_factor
    
    # Incorporate Recent Volume Spikes
    recent_volume_spike = df['volume'] > df['volume'].rolling(window=5).mean() * 1.5
    alpha_factor[recent_volume_spike] *= 1.5  # Amplify factor on volume spikes
    
    return alpha_factor
```
