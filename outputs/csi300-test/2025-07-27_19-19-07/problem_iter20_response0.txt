```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Compute the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the daily range
    daily_range = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate the volume weighted price
    vwp = (df['amount'] / df['volume']).fillna(0)
    
    # Calculate the price momentum
    momentum = df['close'] / df['close'].shift(20) - 1
    
    # Calculate the volume change
    volume_change = df['volume'].pct_change().fillna(0)
    
    # Combine the factors into a single alpha factor
    alpha_factor = (
        daily_return * 0.3 +
        daily_range * 0.2 +
        (vwp - df['close'].shift(1)) / df['close'].shift(1) * 0.2 +
        momentum * 0.2 +
        volume_change * 0.1
    )
    
    return alpha_factor
```
