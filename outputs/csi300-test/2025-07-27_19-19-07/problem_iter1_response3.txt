```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Close-to-Close Return
    close_to_close_return = df['close'].pct_change()

    # Multiply High-Low Spread by Close-to-Close Return
    high_low_momentum = high_low_spread * close_to_close_return

    # Calculate Volume Change
    volume_change = df['volume'].pct_change()

    # Final Factor Calculation
    # If Volume Change is positive, amplify the factor; otherwise, dampen
    final_factor = high_low_momentum * volume_change.abs() * (1 if volume_change > 0 else -1)

    return final_factor
```
