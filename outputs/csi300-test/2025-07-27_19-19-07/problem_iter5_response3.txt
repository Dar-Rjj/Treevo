```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Return
    close_to_prev_close_return = df['close'].pct_change()
    
    # Multiply High-Low Range by Close-to-Previous Close Return
    weighted_returns = high_low_range * close_to_prev_close_return
    
    # Calculate Cumulative Sum of Weighted Returns
    cum_sum_weighted_returns = weighted_returns.rolling(window=10).sum()
    
    # Smooth the Resulting Series
    smoothed_series = cum_sum_weighted_returns.rolling(window=5).mean()
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    adjusted_series = smoothed_series * (1 + volume_change)
    
    # Combine and Smooth Final Factor
    final_factor_smoothed = adjusted_series.rolling(window=3).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate High-Low Price Bandwidth
    high_low_bandwidth = df['high'] - df['low']
    
    # Aggregate High-Low Price Bandwidth
    aggregated_high_low_bandwidth = high_low_bandwidth.rolling(window=5).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    total_volume = df['volume'].rolling(window=5).sum()
    vwap = vwap.rolling(window=5).sum() / total_volume
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = df['close'].pct_change(periods=20) * volume_trend_ratio
    combined_momentum = intraday_momentum + long_term_momentum
    
    # Compute Volatility Component
    intraday_volatility = intraday_return.rolling(window=5).std()
    open_close_volatility = open_to_close_return.rolling(window=5).std()
    volatility = intraday_volatility + open_close_volatility
    
    # Combine Momentum and Volatility
    combined_factor = combined_momentum - 0.5 * volatility  # Weighted Volatility
    
    # Final Alpha Factor
    final_alpha_factor = (final_factor_smoothed + combined_factor).rolling(window=3).mean() * vwap + aggregated_high_low_bandwidth
    
    return final_alpha_factor
```
