import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    df['close_to_close_return'] = df['close'].pct_change()
    
    # Adjust for Historical Price Change
    df['price_momentum'] = df['high_low_spread'] * df['close_to_close_return']
    
    # Adjust for Volume Trend
    df['volume_change'] = df['volume'].pct_change()
    
    # Adjust for Amount Trend
    df['amount_change'] = df['amount'].pct_change()
    
    # Smoothing factor using a moving average of past 5 days' Volume and Amount Changes
    df['avg_volume_change'] = df['volume_change'].rolling(window=5).mean()
    df['avg_amount_change'] = df['amount_change'].rolling(window=5).mean()
    
    # Final Factor Calculation
    def adjust_factor(row):
        if row['volume_change'] > 0 and row['amount_change'] > 0:
            return row['price_momentum'] * (1 + row['avg_volume_change'] + row['avg_amount_change'])
        elif row['volume_change'] > 0 or row['amount_change'] > 0:
            return row['price_momentum'] * (1 + 0.5 * (row['avg_volume_change'] + row['avg_amount_change']))
        else:
            return row['price_momentum'] * (1 - 0.5 * (row['avg_volume_change'] + row['avg_amount_change']))

    df['alpha_factor'] = df.apply(adjust_factor, axis=1)
    
    return df['alpha_factor']
