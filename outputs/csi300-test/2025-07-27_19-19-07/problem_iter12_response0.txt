```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index, and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series with (date) as index, representing the calculated factor values.
    """

    # Calculate the relative strength of the closing price
    df['rel_strength_close'] = (df['close'] - df['close'].rolling(window=20).min()) / \
                               (df['close'].rolling(window=20).max() - df['close'].rolling(window=20).min())
                               
    # Calculate the money flow index
    df['money_flow'] = (df['amount'] * (df['close'] - df['open'])) / (df['volume'] * (df['high'] - df['low']))
    df['money_flow_index'] = df['money_flow'].rolling(window=14).mean()
    
    # Combine the two factors to create a composite factor
    df['composite_factor'] = df['rel_strength_close'] + 2 * df['money_flow_index']
    
    return df['composite_factor']
```
