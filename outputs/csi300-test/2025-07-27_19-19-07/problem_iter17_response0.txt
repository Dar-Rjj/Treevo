```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series representing the alpha factor values, indexed by date.
    """
    # Calculate the relative price change from open to close
    rel_price_change = (df['close'] - df['open']) / df['open']
    
    # Calculate the relative volume change from the previous day
    rel_vol_change = df['volume'].pct_change()
    
    # Calculate the amount to volume ratio
    amt_to_vol_ratio = df['amount'] / df['volume']
    
    # Combine the factors into a single alpha factor
    alpha_factor = rel_price_change * rel_vol_change * amt_to_vol_ratio
    
    return alpha_factor
```
