```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Multiply High-Low Spread by Close-to-Close Return
    high_low_momentum = high_low_spread * close_to_close_return
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    high_low_volume_adjusted = high_low_momentum * (1 + volume_change)
    
    # Calculate Cumulative Sum of Weighted Returns over a Moving Window
    cum_sum_window = 5
    cum_sum_high_low_vol = high_low_volume_adjusted.rolling(window=cum_sum_window).sum()
    
    # Smooth the Resulting Series with Simple Moving Average
    smooth_window = 10
    smoothed_cum_sum = cum_sum_high_low_vol.rolling(window=smooth_window).mean()
    
    # Adjust for Amount Trend
    amount_change = df['amount'].pct_change()
    
    # Final Factor Calculation
    final_factor = smoothed_cum_sum
    final_factor = final_factor * (1 + volume_change) * (1 + amount_change)
    
    # Combine and Smooth Final Factor
    final_factor_smoothed = final_factor.rolling(window=smooth_window).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    adjusted_close_price_change = df['close'].pct_change() * volume_trend_ratio
    long_term_momentum = adjusted_close_price_change
    
    # Combine Momentum Components
    combined_momentum = intraday_momentum + long_term_momentum
    
    # Subtract Weighted Volatility
    weighted_volatility = high_low_spread * volume_trend_ratio
    combined_momentum -= weighted_volatility
    
    # Final Alpha Factor
    alpha_factor = (combined_momentum * vwap + high_low_spread + final_factor_smoothed)
    
    return alpha_factor
```
