```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']

    # Incorporate Historical Price Change
    price_change = df['close'].pct_change()
    ema_price_change = price_change.ewm(alpha=0.1).mean()

    # Smooth the Resulting Series
    sma_ema_price_change = ema_price_change.rolling(window=5).mean()

    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    
    # Adjust for Amount Trend
    amount_change = df['amount'].pct_change()
    trend_adjustment = (volume_change + amount_change) / 2
    trend_adjusted_factor = sma_ema_price_change * trend_adjustment

    # Combine and Smooth Final Factor
    final_factor = trend_adjusted_factor.rolling(window=5).mean()

    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']

    # Calculate Open-to-Close Return
    open_close_return = df['close'] - df['open']

    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()

    # Compute Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()

    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend

    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()

    # Compute Final Momentum Component
    adjusted_close_change = df['close'].pct_change().shift(-1) * volume_trend_ratio
    final_momentum = intraday_return + open_close_return + adjusted_close_change

    # Introduce New Intraday Strength Indicator
    intraday_strength = (df['high'] - df['low']) / (df['high'] + df['low'])
    ema_intraday_strength = intraday_strength.ewm(alpha=0.1).mean()
    combined_momentum = final_momentum + ema_intraday_strength

    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = (df['high'] - df['low']) * df['volume']
    volume_weighted_momentum = volume_weighted_high_low_range * price_change

    # Smooth the Resulting Series
    smoothed_volume_weighted_momentum = volume_weighted_momentum.ewm(alpha=0.1).mean()

    # Final Alpha Factor
    high_low_bandwidth = (df['high'] - df['low']).rolling(window=5).sum()
    final_alpha_factor = combined_momentum * vwap + high_low_bandwidth + smoothed_volume_weighted_momentum.rolling(window=5).mean()

    # Consider Market Sentiment
    if 'sentiment_score' in df.columns:
        final_alpha_factor = final_alpha_factor * df['sentiment_score']

    return final_alpha_factor
```
