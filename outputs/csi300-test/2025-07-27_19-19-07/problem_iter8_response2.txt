```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    vol_weighted_high_low = (df['high'] - df['low']) * df['volume']
    
    # Calculate Close-to-Previous Close Return
    close_to_prev_close_return = df['close'].pct_change()
    
    # Multiply Volume-Weighted High-Low Range by Close-to-Previous Close Return
    momentum_factor = vol_weighted_high_low * close_to_prev_close_return
    
    # Smooth the Resulting Series with Exponential Moving Average
    alpha_ema = 0.5  # Specific Alpha for EMA
    smoothed_momentum = momentum_factor.ewm(alpha=alpha_ema).mean()
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    adjusted_momentum = smoothed_momentum * np.sign(volume_change)
    
    # Combine and Smooth Final Factor with Simple Moving Average
    window_size = 10  # Specific Window Size for SMA
    final_smoothed_momentum = adjusted_momentum.rolling(window=window_size).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = df['close'].pct_change().shift(-1) * volume_trend_ratio
    combined_momentum = intraday_momentum + long_term_momentum
    
    # Compute Volatility Component
    intraday_volatility = df['high'] - df['low']
    open_to_close_volatility = df['open'] - df['close']
    total_volatility = (intraday_volatility.std() + open_to_close_volatility.std()) / 2
    
    # Combine Momentum and Volatility
    momentum_and_volatility = combined_momentum - total_volatility
    
    # Final Alpha Factor
    high_low_bandwidth = (df['high'] - df['low']).rolling(window=window_size).mean()
    final_alpha_factor = (momentum_and_volatility * vwap + high_low_bandwidth + final_smoothed_momentum).fillna(0)
    
    return final_alpha_factor
```
