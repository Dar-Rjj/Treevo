import pandas as pd

def heuristics_v2(df):
    """
    This function generates a novel and interpretable alpha factor based on the given DataFrame.
    The DataFrame is expected to have (date) as index, and columns: open, high, low, close, amount, volume.
    
    :param df: pandas DataFrame with market features
    :return: pandas Series representing the factor values
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the average true range over a 14-day period
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['avg_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the price momentum as the difference between the current closing price and the 30-day moving average
    df['30_day_ma'] = df['close'].rolling(window=30).mean()
    df['momentum'] = df['close'] - df['30_day_ma']
    
    # Calculate the relative strength compared to the total traded amount
    df['relative_strength'] = df['close'] / df['amount']
    
    # Combine these factors into a single alpha factor
    df['alpha_factor'] = (df['daily_return'] * 0.2 
                          + (df['avg_true_range'] / df['close']) * 0.3
                          + (df['momentum'] / df['close']) * 0.2
                          + (df['relative_strength'] / df['close']) * 0.3)
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
