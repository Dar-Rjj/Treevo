```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Multiply High-Low Spread by Close-to-Close Return
    high_low_momentum = high_low_spread * close_to_close_return
    
    # Calculate Volume Change
    volume_change = df['volume'].pct_change()
    
    # Adjust for Volume Trend
    high_low_volume_adjusted = high_low_momentum * (1 + np.sign(volume_change) * 0.5)
    
    # Calculate Cumulative Sum of Weighted Returns
    cumulative_sum = high_low_volume_adjusted.rolling(window=20).sum()
    
    # Smooth the Resulting Series
    smoothed_series = cumulative_sum.rolling(window=10).mean()
    
    # Calculate Amount Change
    amount_change = df['amount'].pct_change()
    
    # Final Factor Calculation based on Volume and Amount Changes
    final_factor = smoothed_series * (1 + np.sign(volume_change) * 0.5) * (1 + np.sign(amount_change) * 0.5)
    
    # Apply Simple Moving Average to Final Factor
    final_factor_smoothed = final_factor.rolling(window=10).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = close_to_close_return * volume_trend_ratio
    combined_momentum = intraday_momentum + long_term_momentum
    
    # Compute Volatility Component
    intraday_volatility = intraday_return.rolling(window=20).std()
    open_to_close_volatility = open_to_close_return.rolling(window=20).std()
    total_volatility = (intraday_volatility + open_to_close_volatility) / 2
    
    # Combine Momentum and Volatility
    momentum_adjusted = combined_momentum - total_volatility
    
    # Final Alpha Factor
    alpha_factor = momentum_adjusted * vwap + high_low_spread + final_factor_smoothed
    
    return alpha_factor
```
