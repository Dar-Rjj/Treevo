```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    price_change = df['close'].pct_change()
    
    # Smooth the Resulting Series
    smoothed_price_change = price_change.ewm(alpha=0.2).mean().rolling(window=5).mean()
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    volume_adjusted_factor = smoothed_price_change + (volume_change / 100)
    
    # Adjust for Amount Trend
    amount_change = df['amount'].pct_change()
    combined_volume_amount_change = volume_change + amount_change
    if combined_volume_amount_change > 0:
        final_factor = volume_adjusted_factor * 1.5
    elif combined_volume_amount_change < 0:
        final_factor = volume_adjusted_factor * 0.5
    else:
        final_factor = volume_adjusted_factor
    
    # Combine and Smooth Final Factor
    final_factor = final_factor.rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_close_return = df['close'] - df['open']
    
    # Compute Short-Term and Long-Term Volume Trends
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume'] / df['volume'].sum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_close_return
    long_term_momentum = (df['close'].pct_change() * volume_trend_ratio).fillna(0)
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = (df['high'] - df['low']) * df['volume']
    
    # Multiply Volume-Weighted High-Low Range by Close-to-Previous Close Return
    momentum_factor = volume_weighted_high_low_range * df['close'].pct_change()
    momentum_factor = momentum_factor.ewm(alpha=0.2).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (intraday_momentum + long_term_momentum) * vwap
    final_alpha_factor += high_low_spread.rolling(window=5).sum()
    final_alpha_factor += momentum_factor.rolling(window=5).mean()
    
    # Introduce New Intraday Strength Indicator
    intraday_strength = (df['high'] - df['low']) / (df['high'] + df['low'])
    intraday_strength = intraday_strength.ewm(alpha=0.2).mean()
    final_alpha_factor += intraday_strength
    
    # Compute Volatility Component
    volatility_component = np.sqrt((intraday_return ** 2).rolling(window=5).mean() + 
                                   (open_close_return ** 2).rolling(window=5).mean())
    
    # Combine Momentum and Volatility
    combined_momentum = intraday_momentum + long_term_momentum
    final_alpha_factor = combined_momentum - (volatility_component * 0.5)
    
    # Introduce New Factor: Momentum-Volatility Ratio
    momentum_volatility_ratio = combined_momentum / volatility_component
    final_alpha_factor *= momentum_volatility_ratio
    
    return final_alpha_factor
```
