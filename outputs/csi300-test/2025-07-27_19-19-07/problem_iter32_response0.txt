import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    price_change_pct = df['close'].pct_change()
    
    # Smooth the Resulting Series
    ema_alpha = 0.1
    sma_window = 5
    smoothed_price_change = price_change_pct.ewm(alpha=ema_alpha).mean().rolling(window=sma_window).mean()
    
    # Adjust for Volume Trend
    volume_change_pct = df['volume'].pct_change()
    volume_adjusted_factor = smoothed_price_change + (volume_change_pct / 100)
    
    # Adjust for Amount Trend
    amount_change_pct = df['amount'].pct_change()
    combined_trend_adjustment = (volume_change_pct > 0) & (amount_change_pct > 0)
    combined_trend_adjustment = combined_trend_adjustment.astype(int) * 2 - 1
    adjusted_factor = volume_adjusted_factor * combined_trend_adjustment
    
    # Combine and Smooth Final Factor
    final_smoothed_factor = adjusted_factor.rolling(window=sma_window).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = (df['close'].pct_change().shift(-1) * volume_trend_ratio).fillna(0)
    final_momentum = intraday_momentum + long_term_momentum
    
    # Introduce New Intraday Strength Indicator
    intraday_strength = (df['high'] - df['low']) / (df['high'] + df['low'])
    ema_intraday_strength = intraday_strength.ewm(alpha=ema_alpha).mean()
    combined_momentum = final_momentum + ema_intraday_strength
    
    # Combine Momentum and Volatility
    combined_momentum_and_volatility = combined_momentum - high_low_spread.rolling(window=sma_window).std()
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = (df['high'] - df['low']) * df['volume']
    
    # Multiply Volume-Weighted High-Low Range by Close-to-Previous Close Return
    close_to_previous_close_return = df['close'].pct_change()
    multiplied_factor = volume_weighted_high_low_range * close_to_previous_close_return
    
    # Smooth the Resulting Series
    smoothed_multiplied_factor = multiplied_factor.ewm(alpha=ema_alpha).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (combined_momentum_and_volatility * vwap 
                          + high_low_spread.rolling(window=5).mean() 
                          + smoothed_multiplied_factor.rolling(window=sma_window).mean())
    
    # Introduce a New Intraday Price Reversal Component
    intraday_price_reversal = (df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)) - intraday_return
    final_alpha_factor += intraday_price_reversal * smoothed_multiplied_factor.rolling(window=sma_window).mean()
    
    # Consider Market Sentiment
    if 'news_sentiment' in df.columns:
        news_sentiment_score = df['news_sentiment']
        final_alpha_factor *= (1 + news_sentiment_score / 100)
    
    return final_alpha_factor
