import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the price range
    df['range'] = df['high'] - df['low']
    
    # Calculate the normalized volume
    df['norm_volume'] = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Calculate the weighted average of the daily return and the normalized volume
    df['weighted_return'] = df['return'] * df['norm_volume']
    
    # Calculate the moving average of the weighted return
    df['ma_weighted_return'] = df['weighted_return'].rolling(window=5).mean()
    
    # Calculate the final heuristic factor
    df['heuristic_factor'] = df['ma_weighted_return'] * df['range']
    
    # Return the heuristic factor as a Series
    return df['heuristic_factor']
