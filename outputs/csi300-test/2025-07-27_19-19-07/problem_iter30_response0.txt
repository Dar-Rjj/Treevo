```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    price_change = df['close'].pct_change()
    
    # Smooth the Resulting Series
    smoothed_price_change = price_change.ewm(alpha=0.1).mean()
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    
    # Adjust for Amount Trend
    amount_change = df['amount'].pct_change()
    
    # Combine volume and amount changes
    combined_volume_amount_change = (volume_change + amount_change) / 2
    
    # Adjust factor based on combined volume and amount changes
    adjusted_factor = smoothed_price_change * (1 + combined_volume_amount_change)
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Compute Short-Term Volume Trend
    short_term_vol_trend = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_vol_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_vol_trend / long_term_vol_trend
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = adjusted_factor * volume_trend_ratio
    
    # Combine Momentum and Volatility
    combined_momentum = intraday_momentum + long_term_momentum
    
    # Integrate Price Bandwidth
    price_bandwidth = high_low_spread.rolling(window=5).mean()
    integrated_momentum = combined_momentum + price_bandwidth
    
    # Calculate VWAP
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Final Alpha Factor
    final_alpha_factor = integrated_momentum * vwap
    final_alpha_factor = final_alpha_factor.rolling(window=5).mean()
    
    return final_alpha_factor
```
