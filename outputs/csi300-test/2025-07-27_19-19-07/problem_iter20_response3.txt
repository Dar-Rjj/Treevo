```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    high_low_range = df['high'] - df['low']
    volume_weighted_high_low_range = high_low_range * df['volume']

    # Calculate Close-to-Previous Close Return
    close_to_prev_close_return = df['close'].pct_change()

    # Multiply Volume-Weighted High-Low Range by Close-to-Previous Close Return
    factor = volume_weighted_high_low_range * close_to_prev_close_return

    # Smooth the Resulting Series
    alpha = 0.3  # Specific alpha for EMA
    smoothed_factor = factor.ewm(alpha=alpha).mean()

    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    adjusted_factor = smoothed_factor * np.sign(volume_change)

    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']

    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']

    # Compute Short-Term Volume Trend (last 5 days)
    short_term_volume_trend = df['volume'].rolling(window=5).mean()

    # Compute Long-Term Volume Trend (last 20 days)
    long_term_volume_trend = df['volume'].rolling(window=20).mean()

    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend

    # Calculate High-Low Price Bandwidth
    high_low_price_bandwidth = df['high'] - df['low']

    # Aggregate High-Low Price Bandwidth over last 5 days
    aggregated_high_low_bandwidth = high_low_price_bandwidth.rolling(window=5).sum()

    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']) / df['volume'].rolling(window=1).sum()

    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = (df['close'] - df['close'].shift(20)) * volume_trend_ratio
    final_momentum = intraday_momentum + long_term_momentum

    # Compute Volatility Component
    intraday_volatility = intraday_return.rolling(window=5).std()
    open_to_close_volatility = open_to_close_return.rolling(window=5).std()
    combined_volatility = intraday_volatility + open_to_close_volatility

    # Combine Momentum and Volatility
    combined_momentum_volatility = final_momentum - combined_volatility

    # Final Alpha Factor
    final_factor = combined_momentum_volatility.rolling(window=5).mean() * vwap + aggregated_high_low_bandwidth

    # Introduce a New Intraday Price Reversal Component
    intraday_price_reversal = df['high'] - df['low'] - intraday_return

    # Incorporate the Intraday Price Reversal into the Final Alpha Factor
    final_factor_with_reversal = final_factor * intraday_price_reversal

    # Add Smoothed and Volume-Adjusted Momentum Factor
    final_alpha_factor = final_factor_with_reversal.rolling(window=5).mean()

    return final_alpha_factor
```
