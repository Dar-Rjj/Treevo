import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Multiply High-Low Spread by Close-to-Close Return
    high_low_momentum = high_low_spread * close_to_close_return
    
    # Calculate Volume Change
    volume_change = df['volume'].pct_change()
    
    # Adjust for Volume Trend
    high_low_momentum_volume_adjusted = high_low_momentum * (1 + volume_change)
    
    # Calculate Amount Change
    amount_change = df['amount'].pct_change()
    
    # Final Factor Calculation based on Volume and Amount Changes
    final_factor = high_low_momentum_volume_adjusted
    final_factor = final_factor * (1 + amount_change) * (1 + volume_change)
    
    # Smoothing factor using a moving average of past 5 days' Volume and Amount Changes
    smoothing_factor = (df['volume'] + df['amount']).rolling(window=5).mean().pct_change()
    final_factor *= (1 + smoothing_factor)
    
    # Calculate Cumulative Sum of Weighted Returns over a Moving Window
    weighted_returns = high_low_momentum.rolling(window=5).sum()
    
    # Smooth the Resulting Series
    smoothed_series = weighted_returns.rolling(window=5).mean()
    
    # Combine and Smooth Final Factor
    final_smoothed_factor = final_factor.rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = (df['close'].pct_change() * volume_trend_ratio)
    combined_momentum = intraday_momentum + long_term_momentum
    
    # Compute Volatility Component
    volatility = intraday_return.std() + open_to_close_return.std()
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = combined_momentum - (volatility * 0.5)
    
    # Final Alpha Factor
    alpha_factor = (combined_momentum_volatility * vwap + high_low_spread + final_smoothed_factor)
    
    return alpha_factor
