```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    price_change = df['close'].pct_change()
    
    # Smooth the Resulting Series
    smoothed_price_change = price_change.ewm(alpha=0.5).mean().rolling(window=5).mean()
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    
    # Adjust for Amount Trend
    amount_change = df['amount'].pct_change()
    
    # Combine and adjust based on volume and amount changes
    adjustment_factor = 1 + (volume_change > 0) * (amount_change > 0) * 0.5 + \
                        (volume_change < 0) * (amount_change < 0) * (-0.5)
    
    adjusted_smoothed_price_change = smoothed_price_change * adjustment_factor
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_close_return = df['close'] - df['open']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = (intraday_return + open_close_return) / 2
    long_term_momentum = price_change * volume_trend_ratio
    
    # Combine Momentum and Volatility
    combined_momentum = intraday_momentum + long_term_momentum
    
    # Final Alpha Factor
    final_alpha_factor = (combined_momentum * vwap + 
                          high_low_spread.rolling(window=5).sum() + 
                          adjusted_smoothed_price_change.rolling(window=5).mean())
    
    return final_alpha_factor
```
