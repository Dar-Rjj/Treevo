```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['vol_weighted_hl_range'] = (df['high'] - df['low']) * df['volume']
    
    # Calculate Close-to-Previous Close Return
    df['close_to_prev_close_return'] = df['close'].pct_change()
    
    # Multiply Volume-Weighted High-Low Range by Close-to-Previous Close Return
    df['initial_factor'] = df['vol_weighted_hl_range'] * df['close_to_prev_close_return']
    
    # Smooth the Resulting Series
    df['smoothed_factor'] = df['initial_factor'].ewm(alpha=0.1, adjust=False).mean()
    
    # Adjust for Volume Trend
    df['volume_change'] = df['volume'].pct_change()
    df['adjusted_factor'] = df['smoothed_factor'] * np.sign(df['volume_change'])
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    df['short_term_vol_trend'] = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    df['long_term_vol_trend'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    df['volume_trend_ratio'] = df['short_term_vol_trend'] / df['long_term_vol_trend']
    
    # Calculate High-Low Price Bandwidth
    df['hl_bandwidth'] = df['high'] - df['low']
    
    # Aggregate High-Low Price Bandwidth
    df['agg_hl_bandwidth'] = df['hl_bandwidth'].rolling(window=5).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['vwap'] = df['vwap'].cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    df['intraday_momentum'] = df['intraday_return'] + df['open_to_close_return']
    df['adjusted_close_price_change'] = df['close'].pct_change()
    df['long_term_momentum'] = df['adjusted_close_price_change'] * df['volume_trend_ratio']
    
    # Combine Momentum and Volatility
    df['combined_momentum'] = df['intraday_momentum'] + df['long_term_momentum']
    
    # Combine and Smooth Final Factor
    df['final_factor'] = df['combined_momentum'].rolling(window=10).mean()
    
    # Multiply by VWAP and Add Aggregated High-Low Price Bandwidth
    df['final_alpha_factor'] = (df['final_factor'] * df['vwap']) + df['agg_hl_bandwidth']
    
    return df['final_alpha_factor']
```
