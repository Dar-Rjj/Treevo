import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the positive and negative money flow
    pos_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    neg_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Calculate the ratio of positive to negative money flow
    pos_neg_ratio = pos_money_flow.rolling(window=14).sum() / neg_money_flow.rolling(window=14).sum()
    
    # Calculate the money flow index
    mfi = 100 - (100 / (1 + pos_neg_ratio))
    
    # Calculate the weighted moving average of the close price
    wma_close = df['close'].rolling(window=14).apply(lambda prices: (prices * range(1, len(prices) + 1)).sum() / sum(range(1, len(prices) + 1)))
    
    # Calculate the alpha factor
    alpha_factor = (mfi - 50) * (wma_close - df['close'])
    
    return alpha_factor
