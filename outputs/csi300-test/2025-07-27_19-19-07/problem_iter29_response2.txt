```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    close_change_pct = df['close'].pct_change()
    
    # Smooth the Resulting Series
    ema_alpha = 0.5
    sma_window = 5
    smoothed_close_change = close_change_pct.ewm(alpha=ema_alpha).mean().rolling(window=sma_window).mean()
    
    # Adjust for Volume Trend
    volume_change_pct = df['volume'].pct_change()
    adjusted_factor = smoothed_close_change + volume_change_pct
    
    # Adjust for Amount Trend
    amount_change_pct = df['amount'].pct_change()
    combined_change = (volume_change_pct > 0) & (amount_change_pct > 0)
    mixed_change = (volume_change_pct > 0) ^ (amount_change_pct > 0)
    negative_change = (volume_change_pct < 0) & (amount_change_pct < 0)
    adjustment_factor = combined_change * 1.5 + mixed_change * 0.5 + negative_change * -0.5
    adjusted_factor = adjusted_factor * adjustment_factor
    
    # Combine and Smooth Final Factor
    final_smoothed_factor = adjusted_factor.rolling(window=sma_window).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = close_change_pct * volume_trend_ratio
    final_momentum = intraday_momentum + long_term_momentum
    
    # Combine Momentum and Volatility
    combined_momentum = final_momentum - high_low_spread.rolling(window=sma_window).std()
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = (df['high'] - df['low']) * df['volume']
    
    # Multiply Volume-Weighted High-Low Range by Close-to-Previous Close Return
    momentum_factor = volume_weighted_high_low_range * close_change_pct
    
    # Smooth the Resulting Series
    smoothed_momentum_factor = momentum_factor.ewm(alpha=ema_alpha).mean().rolling(window=sma_window).mean()
    
    # Final Alpha Factor
    aggregated_high_low_bandwidth = high_low_spread.rolling(window=5).sum()
    final_alpha_factor = (combined_momentum * vwap) + aggregated_high_low_bandwidth + smoothed_momentum_factor
    
    # Introduce a New Intraday Price Reversal Component
    intraday_price_reversal = (df['high'] - df['low']) - intraday_return
    final_alpha_factor = final_alpha_factor * (intraday_price_reversal * final_smoothed_factor)
    
    # Consider Market Sentiment
    if 'sentiment_score' in df.columns:
        sentiment_adjustment = df['sentiment_score'].fillna(0)
        final_alpha_factor = final_alpha_factor * (1 + sentiment_adjustment)
    
    return final_alpha_factor
```
