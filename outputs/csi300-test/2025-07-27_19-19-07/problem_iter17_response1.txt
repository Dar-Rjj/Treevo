```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['vol_weighted_high_low'] = (df['high'] - df['low']) * df['volume']
    
    # Calculate Close-to-Previous Close Return
    df['close_to_prev_close_return'] = df['close'].pct_change()
    
    # Multiply Volume-Weighted High-Low Range by Close-to-Previous Close Return
    df['vol_weighted_momentum'] = df['vol_weighted_high_low'] * df['close_to_prev_close_return']
    
    # Smooth the Resulting Series
    df['ema_smoothed'] = df['vol_weighted_momentum'].ewm(alpha=0.1, adjust=False).mean()
    df['sma_smoothed'] = df['vol_weighted_momentum'].rolling(window=5).mean()
    
    # Adjust for Volume Trend
    df['volume_change'] = df['volume'].pct_change()
    df['volume_adjusted_momentum'] = df['sma_smoothed'] * df['volume_change']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    df['short_term_volume_trend'] = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    df['long_term_volume_trend'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    df['volume_trend_ratio'] = df['short_term_volume_trend'] / df['long_term_volume_trend']
    
    # Calculate High-Low Price Bandwidth
    df['high_low_bandwidth'] = df['high'] - df['low']
    
    # Aggregate High-Low Price Bandwidth
    df['agg_high_low_bandwidth'] = df['high_low_bandwidth'].rolling(window=5).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    df['intraday_momentum'] = df['intraday_return'] + df['open_to_close_return']
    df['long_term_momentum'] = df['close'].pct_change(periods=20) * df['volume_trend_ratio']
    
    # Compute Volatility Component
    df['intraday_volatility'] = df['intraday_return'].rolling(window=5).std()
    df['open_to_close_volatility'] = df['open_to_close_return'].rolling(window=5).std()
    df['total_volatility'] = df['intraday_volatility'] + df['open_to_close_volatility']
    
    # Combine Momentum and Volatility
    df['combined_momentum'] = df['intraday_momentum'] + df['long_term_momentum']
    df['combined_factor'] = df['combined_momentum'] - df['total_volatility']
    
    # Combine and Smooth Final Factor
    df['final_factor'] = df['combined_factor'] * df['vwap'] + df['agg_high_low_bandwidth'] + df['volume_adjusted_momentum']
    df['smoothed_final_factor'] = df['final_factor'].rolling(window=5).mean()
    
    return df['smoothed_final_factor']
```
