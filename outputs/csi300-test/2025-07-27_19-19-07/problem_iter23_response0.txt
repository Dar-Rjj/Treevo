```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['vol_weighted_high_low'] = (df['high'] - df['low']) * df['volume']
    
    # Incorporate Historical Price Change
    df['close_to_close_return'] = df['close'].pct_change()
    df['momentum_factor'] = df['vol_weighted_high_low'] * df['close_to_close_return']
    
    # Adjust for Volume and Amount Trend
    df['volume_change'] = df['volume'].pct_change()
    df['amount_change'] = df['amount'].pct_change()
    
    def adjust_factor(row):
        if row['volume_change'] > 0 and row['amount_change'] > 0:
            return row['momentum_factor'] * 1.5
        elif row['volume_change'] > 0 or row['amount_change'] > 0:
            return row['momentum_factor'] * 1.2
        else:
            return row['momentum_factor'] * 0.8
    
    df['adjusted_momentum_factor'] = df.apply(adjust_factor, axis=1)
    
    # Smooth the Resulting Series
    df['ema_smoothed'] = df['adjusted_momentum_factor'].ewm(span=5, adjust=False).mean()
    df['sma_smoothed'] = df['adjusted_momentum_factor'].rolling(window=10).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['open'] - df['close']
    
    # Compute Short-Term and Long-Term Volume Trends
    df['short_term_vol_trend'] = df['volume'].rolling(window=5).mean()
    df['long_term_vol_trend'] = df['volume'].rolling(window=20).mean()
    df['vol_trend_ratio'] = df['short_term_vol_trend'] / df['long_term_vol_trend']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['total_volume'] = df['volume'].rolling(window=1).sum()
    df['vwap'] = df['vwap'] / df['total_volume']
    
    # Compute Final Momentum Component
    df['intraday_momentum'] = df['intraday_return'] + df['open_to_close_return']
    df['long_term_momentum'] = df['close_to_close_return'] * df['vol_trend_ratio']
    df['combined_momentum'] = df['intraday_momentum'] + df['long_term_momentum']
    
    # Compute Volatility Component
    df['std_intraday'] = df['intraday_return'].rolling(window=10).std()
    df['std_open_to_close'] = df['open_to_close_return'].rolling(window=10).std()
    df['weighted_volatility'] = (df['std_intraday'] + df['std_open_to_close']) / 2
    
    df['final_momentum'] = df['combined_momentum'] - df['weighted_volatility']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['final_momentum'] * df['vwap'] + 
                          df['vol_weighted_high_low'] + 
                          df['sma_smoothed'])
    
    return df['alpha_factor']
```
