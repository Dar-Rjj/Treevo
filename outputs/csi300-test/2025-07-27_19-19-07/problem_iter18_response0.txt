```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    close_to_close_return = df['close'].pct_change()
    high_low_price_momentum = high_low_spread * close_to_close_return
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    adjusted_high_low_momentum = high_low_price_momentum * np.where(volume_change > 0, 1.5, 0.5)
    
    # Calculate Cumulative Sum of Weighted Returns over a Moving Window
    cumulative_sum = adjusted_high_low_momentum.rolling(window=10).sum()
    
    # Smooth the Resulting Series with Exponential Moving Average
    smoothed_cumulative_sum = cumulative_sum.ewm(alpha=0.1).mean()
    
    # Adjust for Amount Trend
    amount_change = df['amount'].pct_change()
    volume_amount_adjustment = np.where((volume_change > 0) & (amount_change > 0), 1.5,
                                        np.where((volume_change > 0) ^ (amount_change > 0), 1.0, 0.5))
    final_factor = smoothed_cumulative_sum * volume_amount_adjustment
    
    # Combine and Smooth Final Factor
    combined_smoothed_final_factor = final_factor.ewm(alpha=0.1).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = close_to_close_return * volume_trend_ratio
    combined_momentum = intraday_momentum + long_term_momentum
    
    # Compute Volatility Component
    intraday_volatility = intraday_return.rolling(window=10).std()
    open_to_close_volatility = open_to_close_return.rolling(window=10).std()
    total_volatility = intraday_volatility + open_to_close_volatility
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = combined_momentum - 0.5 * total_volatility
    
    # Final Alpha Factor
    final_alpha_factor = combined_momentum_volatility * vwap + high_low_spread + combined_smoothed_final_factor
    
    return final_alpha_factor
```
