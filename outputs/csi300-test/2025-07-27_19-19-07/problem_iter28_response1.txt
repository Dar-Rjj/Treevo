```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']

    # Incorporate Historical Price Change
    price_change = df['close'].pct_change()

    # Smooth the Resulting Series
    alpha = 0.1  # smoothing factor
    smoothed_price_change = price_change.ewm(alpha=alpha).mean()

    # Adjust for Volume and Amount Trends
    volume_change = df['volume'].pct_change()
    amount_change = df['amount'].pct_change()
    
    # Combine volume and amount changes
    combined_change = (volume_change + amount_change) / 2
    adjustment_factor = np.where(combined_change > 0, 1.5, np.where(combined_change < 0, 0.5, 1))
    final_smoothed_series = smoothed_price_change * adjustment_factor

    # Apply Simple Moving Average with a specific window size
    window_size = 5
    final_smoothed_series_sma = final_smoothed_series.rolling(window=window_size).mean()

    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']

    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']

    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()

    # Compute Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()

    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend

    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()

    # Compute Final Momentum Component
    intraday_momentum = (intraday_return + open_to_close_return) / 2
    adjusted_close_price_change = df['close'].pct_change().shift(-1)
    long_term_momentum = adjusted_close_price_change * volume_trend_ratio
    final_momentum = intraday_momentum + long_term_momentum

    # Combine Momentum and Volatility
    combined_momentum_volatility = final_momentum

    # Final Alpha Factor
    final_alpha_factor = (combined_momentum_volatility * vwap +
                          high_low_spread.rolling(window=5).mean() +
                          final_smoothed_series_sma)

    return final_alpha_factor
```
