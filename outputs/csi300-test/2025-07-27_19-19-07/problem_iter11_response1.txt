```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Multiply High-Low Spread by Close-to-Close Return
    high_low_momentum = high_low_spread * close_to_close_return
    
    # Calculate Volume Change
    volume_change = df['volume'].pct_change()
    
    # Adjust for Volume Trend
    volume_adjusted_factor = high_low_momentum * (1 + volume_change)
    
    # Calculate Amount Change
    amount_change = df['amount'].pct_change()
    
    # Final Factor Calculation
    final_factor = volume_adjusted_factor.copy()
    final_factor[volume_change > 0] *= (1 + amount_change[volume_change > 0])
    final_factor[volume_change <= 0] *= (1 - amount_change[volume_change <= 0])
    
    # Introduce a smoothing factor using a moving average of past 5 days' Volume and Amount Changes
    smoothing_factor = (df['volume'].rolling(window=5).mean().pct_change() + 
                        df['amount'].rolling(window=5).mean().pct_change()) / 2
    final_factor *= (1 + smoothing_factor)
    
    # Smooth the Resulting Series
    smoothed_final_factor = final_factor.rolling(window=5).mean()
    
    # Combine and Smooth Final Factor
    combined_smoothed_factor = smoothed_final_factor.rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = close_to_close_return * volume_trend_ratio
    momentum_component = intraday_momentum + long_term_momentum
    
    # Compute Volatility Component
    volatility_component = intraday_return.rolling(window=5).std() + open_to_close_return.rolling(window=5).std()
    
    # Combine Momentum and Volatility
    combined_momentum = momentum_component - 0.5 * volatility_component
    
    # Final Alpha Factor
    alpha_factor = combined_momentum * vwap + high_low_spread + combined_smoothed_factor
    
    return alpha_factor
```
