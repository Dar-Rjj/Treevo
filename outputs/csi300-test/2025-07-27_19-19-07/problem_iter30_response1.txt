import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given DataFrame.
    The function takes a pandas DataFrame with (open, high, low, close, amount, volume) and returns
    a pandas Series representing the factor values.
    
    :param df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
               and index as date.
    :return: pandas Series with the calculated alpha factor and date as index.
    """
    # Calculate the price change from the previous day
    df['price_change'] = df['close'].pct_change()
    
    # Calculate the volume change from the previous day
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate the average true range over 14 days
    df['atr_14'] = df['high'] - df['low']
    df['atr_14'] = df['atr_14'].rolling(window=14).mean()
    
    # Calculate the money flow ratio
    df['mf_ratio'] = df['amount'] / df['volume']
    
    # Heuristic alpha factor
    df['alpha_factor'] = (
        (df['price_change'] * 0.5) + 
        (df['volume_change'] * 0.3) + 
        (df['atr_14'] * 0.1) + 
        (df['mf_ratio'] * 0.1)
    )
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
