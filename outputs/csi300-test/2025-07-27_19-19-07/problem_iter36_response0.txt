```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['High_Low_Range'] = (df['high'] - df['low']) * df['volume']
    
    # Incorporate Historical Price Change
    df['Close_to_Close_Return'] = df['close'].pct_change()
    df['Price_Momentum'] = df['High_Low_Range'] * df['Close_to_Close_Return']
    
    # Adjust for Volume and Amount Trend
    df['Volume_Change'] = df['volume'].pct_change()
    df['Amount_Change'] = df['amount'].pct_change()
    
    # Final Factor Calculation
    conditions = [
        (df['Volume_Change'] > 0) & (df['Amount_Change'] > 0),
        (df['Volume_Change'] > 0) ^ (df['Amount_Change'] > 0),
        (df['Volume_Change'] < 0) & (df['Amount_Change'] < 0)
    ]
    choices = [1.5, 1.2, 0.8]
    df['Adjusted_Factor'] = np.select(conditions, choices, default=1.0) * df['Price_Momentum']
    
    # Smooth the Resulting Series
    decay_factor = 0.95
    ema_smoothed = df['Adjusted_Factor'].ewm(alpha=decay_factor).mean()
    sma_window_size = 5
    sma_smoothed = df['Adjusted_Fctor'].rolling(window=sma_window_size).mean()
    df['Smoothed_Factor'] = (ema_smoothed + sma_smoothed) / 2
    
    # Calculate Intraday and Open-to-Close Returns
    df['Intraday_Return'] = df['high'] - df['low']
    df['Open_to_Close_Return'] = df['open'] - df['close']
    
    # Compute Short-Term and Long-Term Volume Trends
    short_term_vol_trend = df['volume'].rolling(window=5).mean()
    long_term_vol_trend = df['volume'].rolling(window=20).mean()
    df['Volume_Trend_Ratio'] = short_term_vol_trend / long_term_vol_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    df['VWAP'] = vwap
    
    # Combine Momentum and Volatility
    df['Intraday_Momentum'] = df['Intraday_Return'] + df['Open_to_Close_Return']
    df['Long_Term_Momentum'] = (df['close'].pct_change(20)) * df['Volume_Trend_Ratio']
    df['Volatility'] = (df['Intraday_Return'].std() + df['Open_to_Close_Return'].std()) / 2
    df['Combined_Momentum'] = df['Intraday_Momentum'] + df['Long_Term_Momentum'] - df['Volatility']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Combined_Momentum'] * df['VWAP']) + df['Smoothed_Factor']
    
    # Introduce a New Component: Price and Volume Synchronization
    price_volume_corr = df[['close', 'volume']].rolling(window=20).corr().unstack().iloc[::2]['close']['volume']
    df['Price_Volume_Correlation'] = price_volume_corr
    
    conditions = [
        (df['Price_Volume_Correlation'] > 0.5),
        (df['Price_Volume_Correlation'] < -0.5)
    ]
    choices = [1.5, 0.5]
    correlation_adjustment = np.select(conditions, choices, default=1.0)
    df['Final_Alpha_Factor'] = df['Alpha_Factor'] * correlation_adjustment
    
    return df['Final_Alpha_Factor']
```
