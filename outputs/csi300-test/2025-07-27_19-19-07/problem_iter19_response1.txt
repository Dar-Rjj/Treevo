```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    price_change = (df['close'] / df['close'].shift(1) - 1) * 100
    
    # Smooth the Resulting Series
    ema_alpha = 0.5
    sma_window = 10
    smoothed_price_change = price_change.ewm(alpha=ema_alpha).mean().rolling(window=sma_window).mean()
    
    # Adjust for Volume Trend
    volume_change = (df['volume'] / df['volume'].shift(1) - 1) * 100
    volume_adjusted_factor = smoothed_price_change * (1 + volume_change / 100)
    
    # Combine and Smooth Final Factor
    final_factor_smoothed = volume_adjusted_factor.rolling(window=sma_window).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).sum() / 5
    
    # Compute Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).sum() / 20
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = (intraday_return + open_close_return) / 2
    long_term_momentum = (df['close'] / df['close'].shift(1) - 1) * volume_trend_ratio
    final_momentum = intraday_momentum + long_term_momentum
    
    # Compute Volatility Component
    volatility = (intraday_return.std() + open_close_return.std()) / 2
    
    # Combine Momentum and Volatility
    combined_momentum = final_momentum - 0.5 * volatility
    
    # Final Alpha Factor
    alpha_factor = (combined_momentum * vwap) + high_low_spread.rolling(window=5).sum() + final_factor_smoothed
    
    return alpha_factor
```
