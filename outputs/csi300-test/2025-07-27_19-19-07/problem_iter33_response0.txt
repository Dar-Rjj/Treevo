import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']

    # Incorporate Historical Price Change
    close_change = df['close'].pct_change()

    # Smooth the Resulting Series
    alpha = 0.1  # Smoothing factor
    smoothed_series = close_change.ewm(alpha=alpha).mean()

    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    adjusted_smoothed_series = smoothed_series * (1 + volume_change)

    # Adjust for Amount Trend
    amount_change = df['amount'].pct_change()
    adjusted_smoothed_series *= (1 + amount_change)

    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']

    # Calculate Open-to-Close Return
    open_close_return = df['close'] - df['open']

    # Compute Short-Term Volume Trend
    short_term_vol_trend = df['volume'].rolling(window=5).mean()

    # Compute Long-Term Volume Trend
    long_term_vol_trend = df['volume'].rolling(window=20).mean()

    # Compute Volume Trend Ratio
    vol_trend_ratio = short_term_vol_trend / long_term_vol_trend

    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_close_return
    long_term_momentum = close_change * vol_trend_ratio
    combined_momentum = intraday_momentum + long_term_momentum

    # Integrate Price Bandwidth
    high_low_bandwidth = (df['high'] - df['low']).rolling(window=5).mean()
    integrated_momentum = combined_momentum + high_low_bandwidth

    # Incorporate Volume-Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    final_factor = integrated_momentum * vwap

    # Apply Simple Moving Average with a specific window size
    window_size = 10
    final_alpha_factor = final_factor.rolling(window=window_size).mean()

    return final_alpha_factor
