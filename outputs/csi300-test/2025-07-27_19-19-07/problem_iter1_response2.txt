import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Close Price Change
    df['close_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Short-Term Volume Trend
    short_term_vol = df['volume'].rolling(window=5).mean()
    df['short_term_vol_trend'] = short_term_vol
    
    # Calculate Long-Term Volume Trend
    long_term_vol = df['volume'].rolling(window=20).mean()
    df['long_term_vol_trend'] = long_term_vol
    
    # Compute Volume Trend Ratio
    df['vol_trend_ratio'] = df['short_term_vol_trend'] / df['long_term_vol_trend']
    
    # Calculate High-Low Price Bandwidth
    df['high_low_bandwidth'] = df['high'] - df['low']
    
    # Aggregate High-Low Price Bandwidth over last 5 days
    df['agg_high_low_bandwidth'] = df['high_low_bandwidth'].rolling(window=5).sum()
    
    # Combine Factors
    df['alpha_factor'] = (df['close_change'] * df['vol_trend_ratio']) + df['agg_high_low_bandwidth']
    
    return df['alpha_factor']
