```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']

    # Incorporate Historical Price Change
    close_price_change = df['close'].pct_change()

    # Smooth the Resulting Series
    ema_alpha = 0.1  # Exponential Moving Average Alpha
    sma_window_1 = 5  # Simple Moving Average Window Size
    smoothed_series = (close_price_change.ewm(alpha=ema_alpha).mean() + 
                       close_price_change.rolling(window=sma_window_1).mean()) / 2

    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    adjusted_smoothed_series = smoothed_series * (1 + volume_change)

    # Combine and Smooth Final Factor
    sma_window_2 = 10  # Simple Moving Average Window Size for final smoothing
    combined_smoothed_factor = adjusted_smoothed_series.rolling(window=sma_window_2).mean()

    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']

    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']

    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).sum() / 5

    # Compute Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).sum() / 20

    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend

    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()

    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = close_price_change * volume_trend_ratio
    final_momentum = intraday_momentum + long_term_momentum

    # Compute Volatility Component
    volatility = (intraday_return.std() + open_to_close_return.std()) / 2

    # Combine Momentum and Volatility
    combined_momentum = final_momentum - 0.5 * volatility

    # Final Alpha Factor
    alpha_factor = (combined_momentum * vwap +
                    high_low_spread.rolling(window=5).sum() +
                    combined_smoothed_factor.rolling(window=sma_window_2).mean())

    # Incorporate Recent Volume Spikes
    recent_volume_spike = df['volume'] > df['volume'].rolling(window=5).mean() * 1.5
    alpha_factor[recent_volume_spike] *= 1.1

    return alpha_factor
```
