```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    df['HighLowSpread'] = df['high'] - df['low']
    
    # Calculate Percentage Price Change
    df['PriceChangePct'] = (df['close'].shift(-1) - df['close']) / df['close']
    
    # Apply Exponential Moving Average on Percentage Price Change
    df['EMA_PriceChange'] = df['PriceChangePct'].ewm(alpha=0.3, adjust=False).mean()
    
    # Adjust for Volume Trend
    df['VolumeChangePct'] = (df['volume'].shift(-1) - df['volume']) / df['volume']
    df['EMA_VolumeChange'] = df['VolumeChangePct'].ewm(alpha=0.3, adjust=False).mean()
    
    # Adjust for Amount Trend
    df['AmountChangePct'] = (df['amount'].shift(-1) - df['amount']) / df['amount']
    df['EMA_AmountChange'] = df['AmountChangePct'].ewm(alpha=0.3, adjust=False).mean()
    
    # Combine volume and amount trend adjustments
    df['TrendAdjustment'] = (df['EMA_VolumeChange'] + df['EMA_AmountChange']) / 2
    
    # Calculate Intraday Return
    df['IntradayReturn'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['OpenCloseReturn'] = df['close'] - df['open']
    
    # Compute Short-Term Volume Trend
    df['ShortTermVolume'] = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    df['LongTermVolume'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    df['VolumeTrendRatio'] = df['ShortTermVolume'] / df['LongTermVolume']
    
    # Compute Final Momentum Component
    df['IntradayMomentum'] = df['IntradayReturn'] + df['OpenCloseReturn']
    df['LongTermMomentum'] = df['EMA_PriceChange'] * df['VolumeTrendRatio']
    
    # Combine Momentum
    df['CombinedMomentum'] = df['IntradayMomentum'] + df['LongTermMomentum']
    
    # Integrate Price Bandwidth
    df['PriceBandwidth'] = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    df['MomentumWithBandwidth'] = df['CombinedMomentum'] + df['PriceBandwidth']
    
    # Calculate VWAP (Volume-Weighted Average Price)
    df['VWAP'] = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = (df['MomentumWithBandwidth'] * df['VWAP']) * df['TrendAdjustment']
    
    # Apply Simple Moving Average with a specific window size
    df['FinalAlphaFactor'] = df['FinalAlphaFactor'].rolling(window=10).mean()
    
    return df['FinalAlphaFactor'].dropna()
```
