```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    high_low_range = (df['high'] - df['low']) * df['volume']
    
    # Incorporate Historical Price Change
    close_to_close_return = df['close'].pct_change()
    volume_weighted_high_low_range = high_low_range * close_to_close_return
    
    # Smooth the Resulting Series
    ema_alpha = 0.1  # Specific alpha for EMA
    sma_window = 10  # Specific window size for SMA
    smoothed_ema = volume_weighted_high_low_range.ewm(alpha=ema_alpha, adjust=False).mean()
    smoothed_sma = volume_weighted_high_low_range.rolling(window=sma_window).mean()
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    adjusted_factor = smoothed_sma * (1 + volume_change)
    
    # Introduce New Intraday Strength Indicator
    intraday_strength = (df['high'] - df['low']) / (df['high'] + df['low'])
    intraday_strength_ema = intraday_strength.ewm(alpha=ema_alpha, adjust=False).mean()
    combined_momentum = adjusted_factor + intraday_strength_ema
    
    # Combine and Smooth Final Factor
    final_factor_smoothed = combined_momentum.rolling(window=sma_window).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    adjusted_close_price_change = close_to_close_return * volume_trend_ratio
    long_term_momentum = adjusted_close_price_change
    final_momentum = intraday_momentum + long_term_momentum
    
    # Compute Volatility Component
    std_intraday_returns = intraday_return.rolling(window=sma_window).std()
    std_open_to_close_returns = open_to_close_return.rolling(window=sma_window).std()
    volatility_component = std_intraday_returns + std_open_to_close_returns
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = final_momentum - 0.5 * volatility_component  # Weighted by 0.5
    
    # Final Alpha Factor
    final_alpha_factor = combined_momentum_volatility * vwap + high_low_range.rolling(window=5).sum() + final_factor_smoothed
    momentum_volatility_ratio = combined_momentum / volatility_component
    final_alpha_factor += momentum_volatility_ratio
    
    return final_alpha_factor
```
