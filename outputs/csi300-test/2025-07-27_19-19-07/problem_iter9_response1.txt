```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Return
    close_return = df['close'].pct_change()
    
    # Multiply High-Low Range by Close-to-Previous Close Return
    weighted_return = high_low_range * close_return
    
    # Calculate Cumulative Sum of Weighted Returns over a moving window (e.g., 10 days)
    cumulative_sum = weighted_return.rolling(window=10).sum()
    
    # Smooth the resulting series using an Exponential Moving Average (e.g., 5 days)
    smoothed_series = cumulative_sum.ewm(span=5).mean()
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    volume_adjusted_factor = smoothed_series * volume_change.apply(lambda x: abs(x) if x > 0 else 1)
    
    # Combine and Smooth Final Factor using an Exponential Moving Average (e.g., 5 days)
    final_smoothed_factor = volume_adjusted_factor.ewm(span=5).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    long_term_momentum = df['close'].pct_change().rolling(window=20).mean() * volume_trend_ratio
    intraday_momentum = intraday_return + open_close_return
    combined_momentum = intraday_momentum + long_term_momentum
    
    # Compute Volatility Component
    intraday_volatility = intraday_return.rolling(window=20).std()
    open_close_volatility = open_close_return.rolling(window=20).std()
    total_volatility = intraday_volatility + open_close_volatility
    
    # Combine Momentum and Volatility
    combined_momentum_and_volatility = combined_momentum - total_volatility
    
    # Final Alpha Factor
    final_alpha_factor = combined_momentum_and_volatility * vwap + high_low_range + final_smoothed_factor
    
    return final_alpha_factor
```
