```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    close_to_close_return = df['close'].pct_change()
    
    # Multiply High-Low Spread by Close-to-Close Return
    high_low_price_momentum = high_low_spread * close_to_close_return
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    volume_adjustment = volume_change.apply(lambda x: 1 + x if x > 0 else 1 / (1 + abs(x)))
    
    # Adjust for Amount Trend
    amount_change = df['amount'].pct_change()
    amount_adjustment = amount_change.apply(lambda x: 1 + x if x > 0 else 1 / (1 + abs(x)))
    
    # Final Factor Calculation
    final_factor = high_low_price_momentum * volume_adjustment * amount_adjustment
    
    return final_factor
```
