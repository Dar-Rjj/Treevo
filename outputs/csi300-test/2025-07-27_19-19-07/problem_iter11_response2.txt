```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Return
    close_to_prev_close_return = df['close'].pct_change()
    
    # Multiply High-Low Range by Close-to-Previous Close Return
    range_return_product = high_low_range * close_to_prev_close_return
    
    # Smooth the Resulting Series
    ema_alpha = 0.1
    ema_smoothed = range_return_product.ewm(alpha=ema_alpha).mean()
    sma_window = 5
    sma_smoothed = range_return_product.rolling(window=sma_window).mean()
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    volume_adjusted_factor = sma_smoothed * volume_change.apply(lambda x: x if x > 0 else 0)
    
    # Combine and Smooth Final Factor
    final_sma_window = 10
    final_smoothed_factor = volume_adjusted_factor.rolling(window=final_sma_window).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    adjusted_close_price_change = df['close'].pct_change()
    long_term_momentum = adjusted_close_price_change * volume_trend_ratio
    intraday_momentum = intraday_return + open_to_close_return
    
    # Compute Volatility Component
    intraday_volatility = intraday_return.rolling(window=5).std()
    open_to_close_volatility = open_to_close_return.rolling(window=5).std()
    total_volatility = (intraday_volatility + open_to_close_volatility) / 2
    
    # Combine Momentum and Volatility
    combined_momentum = intraday_momentum + long_term_momentum
    weighted_volatility = total_volatility * 0.5  # Weight of 0.5 for volatility
    combined_momentum_volatility = combined_momentum - weighted_volatility
    
    # Final Alpha Factor
    high_low_bandwidth_sum = high_low_range.rolling(window=5).sum()
    alpha_factor = (combined_momentum_volatility * vwap + high_low_bandwidth_sum + final_smoothed_factor).dropna()
    
    return alpha_factor
```
