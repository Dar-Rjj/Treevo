```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Percentage Change in Close Price
    close_change = df['close'].pct_change()
    
    # Smooth the Close Price Change
    smoothed_close_change = close_change.ewm(alpha=0.1, adjust=False).mean()
    
    # Calculate Volume Change
    volume_change = df['volume'].pct_change()
    
    # Calculate Amount Change
    amount_change = df['amount'].pct_change()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Compute Short-Term Volume Trend
    short_term_vol_trend = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_vol_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    vol_trend_ratio = short_term_vol_trend / long_term_vol_trend
    
    # Compute Final Momentum Component
    intraday_momentum = (intraday_return + open_to_close_return) / 2
    long_term_momentum = smoothed_close_change * vol_trend_ratio
    
    # Combine Momentum Components
    combined_momentum = intraday_momentum + long_term_momentum
    
    # Calculate VWAP
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Aggregated High-Low Price Bandwidth over last 5 days
    high_low_bandwidth = high_low_spread.rolling(window=5).mean()
    
    # Final Alpha Factor
    final_alpha_factor = combined_momentum * vwap + high_low_bandwidth
    
    return final_alpha_factor
```
