import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between today's Open and Close
    open_close_diff = df['open'] - df['close']
    
    # Calculate the percentage change from Open to Close
    open_close_pct_change = (df['close'] - df['open']) / df['open']
    
    # Calculate the daily range (High - Low)
    daily_range = df['high'] - df['low']
    
    # Calculate the change in volume compared to the previous day
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate the relative change in volume
    rel_volume_change = volume_change / df['volume'].shift(1)
    
    # Calculate the ratio of amount to volume as a proxy for average price
    avg_price_proxy = df['amount'] / df['volume']
    
    # Calculate the change in this ratio over time
    avg_price_proxy_change = avg_price_proxy - avg_price_proxy.shift(1)
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['open'] + df['high'] - df['low']) / df['volume']
    
    # Calculate the Simple Moving Average of Close prices (n_days=5)
    sma_close = df['close'].rolling(window=5).mean()
    
    # Calculate the daily return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate the volatility as a measure of price movement (n_days=5)
    volatility = daily_return.rolling(window=5).std()
    
    # Combine multiple factors into a single interpretable factor
    combined_factor = (df['open'] - df['close']) * (df['volume'] - df['volume'].shift(1)) / df['volume']
    
    # Return the combined factor as the final alpha factor
    return combined_factor
