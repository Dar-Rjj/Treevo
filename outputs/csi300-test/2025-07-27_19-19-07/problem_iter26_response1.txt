import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-day Money Flow Index (MFI)
    typical_price_rolling = typical_price.rolling(window=14).mean()
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0).rolling(window=14).sum()
    negative_money_flow = -money_flow.where(typical_price < typical_price.shift(1), 0).rolling(window=14).sum()
    money_ratio = positive_money_flow / negative_money_flow
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Calculate the 14-day Average True Range (ATR)
    true_range = df[['high', 'low']].diff(axis=1).iloc[:, 0].abs().combine(df['close'].diff().abs(), max, axis=0)
    atr = true_range.rolling(window=14).mean()
    
    # Combine MFI and ATR to create a novel alpha factor
    factor = (mfi / atr) - 1
    
    return factor
