```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['Vol_Weighted_High_Low_Range'] = (df['high'] - df['low']) * df['volume']
    
    # Incorporate Historical Price Change
    df['Close_to_Close_Return'] = df['close'].pct_change()
    df['Weighted_Price_Change'] = df['Vol_Weighted_High_Low_Range'] * df['Close_to_Close_Return']
    
    # Smooth the Resulting Series
    df['Smoothed_Series_EMA'] = df['Weighted_Price_Change'].ewm(alpha=0.1).mean()
    df['Smoothed_Series_SMA'] = df['Weighted_Price_Change'].rolling(window=5).mean()
    
    # Adjust for Volume Trend
    df['Volume_Change'] = df['volume'].pct_change()
    df['Volume_Adjusted_Factor'] = df['Smoothed_Series_EMA'] * (1 + df['Volume_Change'])
    
    # Introduce New Intraday Strength Indicator
    df['Intraday_Strength'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    df['EMA_Intraday_Strength'] = df['Intraday_Strength'].ewm(alpha=0.1).mean()
    
    # Combine and Smooth Final Factor
    df['Combined_Momentum'] = df['Volume_Adjusted_Factor'] + df['EMA_Intraday_Strength']
    df['SMA_Combined_Momentum'] = df['Combined_Momentum'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    df['Intraday_Return'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['Open_to_Close_Return'] = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    df['Short_Term_Volume_Trend'] = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    df['Long_Term_Volume_Trend'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    df['Volume_Trend_Ratio'] = df['Short_Term_Volume_Trend'] / df['Long_Term_Volume_Trend']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df[['high', 'low', 'close']].sum(axis=1) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    df['Intraday_Momentum'] = df['Intraday_Return'] + df['Open_to_Close_Return']
    df['Long_Term_Momentum'] = df['Close_to_Close_Return'] * df['Volume_Trend_Ratio']
    
    # Compute Volatility Component
    df['Std_Intraday_Returns'] = df['Intraday_Return'].rolling(window=5).std()
    df['Std_Open_to_Close_Returns'] = df['Open_to_Close_Return'].rolling(window=5).std()
    
    # Combine Momentum and Volatility
    df['Combined_Momentum_Components'] = df['Intraday_Momentum'] + df['Long_Term_Momentum']
    df['Momentum_Minus_Weighted_Volatility'] = df['Combined_Momentum_Components'] - (df['Std_Intraday_Returns'] + df['Std_Open_to_Close_Returns']) * 0.5
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Momentum_Minus_Weighted_Volatility'] * df['VWAP'] 
                                 + df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min() 
                                 + df['SMA_Combined_Momentum'])
    df['Momentum_Volatility_Ratio'] = df['Combined_Momentum_Components'] / (df['Std_Intraday_Returns'] + df['Std_Open_to_Close_Returns'])
    
    return df['Final_Alpha_Factor'].dropna()
```
