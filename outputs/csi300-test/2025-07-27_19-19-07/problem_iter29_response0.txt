import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['vol_weighted_high_low'] = (df['high'] - df['low']) * df['volume']
    
    # Incorporate Historical Price Change
    df['close_to_close_return'] = df['close'].pct_change()
    df['price_change_factor'] = df['vol_weighted_high_low'] * df['close_to_close_return']
    
    # Adjust for Volume and Amount Trend
    df['volume_change'] = df['volume'].pct_change()
    df['amount_change'] = df['amount'].pct_change()
    
    # Final Factor Calculation
    def adjust_factor(row):
        if row['volume_change'] > 0 and row['amount_change'] > 0:
            return row['price_change_factor'] * 1.5
        elif row['volume_change'] > 0 or row['amount_change'] > 0:
            return row['price_change_factor'] * 1.2
        else:
            return row['price_change_factor'] * 0.8
    
    df['final_factor'] = df.apply(adjust_factor, axis=1)
    
    # Smooth the Resulting Series
    df['final_factor_smoothed_ema'] = df['final_factor'].ewm(span=7, adjust=False).mean()
    df['final_factor_smoothed_sma'] = df['final_factor'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    df['short_term_vol_trend'] = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    df['long_term_vol_trend'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    df['vol_trend_ratio'] = df['short_term_vol_trend'] / df['long_term_vol_trend']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df[['high', 'low', 'close']].sum(axis=1) / 3) * df['volume'] / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    df['intraday_momentum'] = df['intraday_return'] + df['open_to_close_return']
    df['long_term_momentum'] = df['close_to_close_return'] * df['vol_trend_ratio']
    df['combined_momentum'] = df['intraday_momentum'] + df['long_term_momentum']
    
    # Combine Momentum and Volatility
    df['momentum_volatility'] = df['combined_momentum'] * df['vwap']
    
    # Add Aggregated High-Low Price Bandwidth over last 5 days
    df['high_low_bandwidth'] = df['high'] - df['low']
    df['aggregated_bandwidth'] = df['high_low_bandwidth'].rolling(window=5).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['momentum_volatility'] + df['aggregated_bandwidth']
    
    # Introduce a Time-of-Day Seasonality Adjustment
    df.index = pd.to_datetime(df.index)
    df['hour'] = df.index.hour
    df['seasonality_adjustment'] = np.where((df['hour'] >= 9) & (df['hour'] < 16), 1.2, 1.0)
    df['alpha_factor'] = df['alpha_factor'] * df['seasonality_adjustment']
    
    # Consider Market Sentiment
    # Assuming `sentiment_score` is available in the DataFrame
    # df['alpha_factor'] = df['alpha_factor'] * df['sentiment_score']
    
    return df['alpha_factor']
