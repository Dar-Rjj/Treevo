import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    price_change = df['close'].pct_change()
    
    # Smooth the Resulting Series
    smoothed_series = price_change.ewm(alpha=0.1).mean().rolling(window=5).mean()
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    
    # Adjust for Amount Trend
    amount_change = df['amount'].pct_change()
    
    # Combine volume and amount changes
    combined_trend = volume_change + amount_change
    adjusted_factor = smoothed_series * np.sign(combined_trend)
    
    # Combine and Smooth Final Factor
    final_factor = adjusted_factor.rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_close_return = df['close'] - df['open']
    
    # Compute Short-Term Volume Trend
    short_term_vol_trend = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_vol_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_vol_trend / long_term_vol_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_close_return
    long_term_momentum = (df['close'].pct_change()).shift(-1) * volume_trend_ratio
    final_momentum = intraday_momentum + long_term_momentum
    
    # Introduce New Intraday Strength Indicator
    intraday_strength = (df['high'] - df['low']) / (df['high'] + df['low'])
    smoothed_intraday_strength = intraday_strength.ewm(alpha=0.1).mean()
    combined_momentum = final_momentum + smoothed_intraday_strength
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = combined_momentum - (df['high'] - df['low']).rolling(window=5).std()
    
    # Calculate Volume-Weighted High-Low Range
    vol_weighted_high_low_range = (df['high'] - df['low']) * df['volume']
    
    # Multiply Volume-Weighted High-Low Range by Close-to-Previous Close Return
    vol_weighted_factor = vol_weighted_high_low_range * df['close'].pct_change()
    
    # Smooth the Resulting Series
    smoothed_final_factor = vol_weighted_factor.ewm(alpha=0.1).mean()
    
    # Integrate Price Bandwidth
    price_bandwidth = (df['high'] - df['low']).rolling(window=5).sum()
    integrated_bandwidth = combined_momentum + price_bandwidth
    
    # Incorporate Volume-Weighted Average Price (VWAP)
    vwap_5_days = vwap.rolling(window=5).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (integrated_bandwidth * vwap_5_days) + price_bandwidth + smoothed_final_factor.rolling(window=5).mean()
    
    # Incorporate the Intraday Price Reversal into the Final Alpha Factor
    intraday_reversal = (df['high'] - df['low']) - intraday_return
    final_alpha_factor += intraday_reversal * smoothed_final_factor
    
    # Consider Market Sentiment
    # Assuming 'news_sentiment' is a column in the DataFrame
    if 'news_sentiment' in df.columns:
        final_alpha_factor *= df['news_sentiment']
    
    return final_alpha_factor
