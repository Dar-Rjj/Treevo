```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) * df['volume']
    
    # Incorporate Historical Price Change
    df['close_to_close_return'] = df['close'].pct_change()
    df['hl_weighted_return'] = df['high_low_range'] * df['close_to_close_return']
    
    # Adjust for Volume Trend
    df['volume_change'] = df['volume'].pct_change()
    df['volume_adjusted_factor'] = df['hl_weighted_return'] * (1 + np.sign(df['volume_change']))
    
    # Smooth the Resulting Series
    df['ema_smoothed'] = df['volume_adjusted_factor'].ewm(alpha=0.1).mean()
    df['sma_smoothed'] = df['volume_adjusted_factor'].rolling(window=5).mean()
    
    # Adjust for Amount Trend
    df['amount_change'] = df['amount'].pct_change()
    df['final_factor'] = df['sma_smoothed']
    
    df['final_factor'] = np.where(
        (df['volume_change'] > 0) & (df['amount_change'] > 0),
        df['final_factor'] * 1.2,
        np.where(
            (df['volume_change'] > 0) | (df['amount_change'] > 0),
            df['final_factor'] * 1.1,
            df['final_factor'] * 0.9
        )
    )
    
    # Combine and Smooth Final Factor
    df['final_factor_sma'] = df['final_factor'].rolling(window=10).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    df['short_term_volume'] = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    df['long_term_volume'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    df['volume_trend_ratio'] = df['short_term_volume'] / df['long_term_volume']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    df['intraday_momentum'] = df['intraday_return'] + df['open_to_close_return']
    df['adjusted_close_return'] = df['close'].pct_change()
    df['long_term_momentum'] = df['adjusted_close_return'] * df['volume_trend_ratio']
    
    # Compute Volatility Component
    df['intraday_volatility'] = df['intraday_return'].std()
    df['open_to_close_volatility'] = df['open_to_close_return'].std()
    
    # Combine Momentum and Volatility
    df['combined_momentum'] = df['intraday_momentum'] + df['long_term_momentum']
    df['weighted_volatility'] = 0.5 * (df['intraday_volatility'] + df['open_to_close_volatility'])
    df['final_momentum'] = df['combined_momentum'] - df['weighted_volatility']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['final_momentum'] * df['vwap'] + 
                          df['high_low_range'] + 
                          df['final_factor_sma'])
    
    return df['alpha_factor']
```
