```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Historical Price Change
    price_change_pct = df['close'].pct_change()
    
    # Smooth the Resulting Series
    ealpha = 0.2  # Example alpha for EMA
    ema_smoothed = price_change_pct.ewm(alpha=ealpha, adjust=False).mean()
    sma_window = 5  # Example window size for SMA
    sma_smoothed = ema_smoothed.rolling(window=sma_window).mean()
    
    # Adjust for Volume Trend
    volume_change_pct = df['volume'].pct_change()
    volume_trend_adjustment = (volume_change_pct > 0) * 1 + (volume_change_pct < 0) * -1
    
    # Adjust for Amount Trend
    amount_change_pct = df['amount'].pct_change()
    combined_trend_adjustment = (volume_change_pct > 0) & (amount_change_pct > 0) * 1.5 + \
                                (volume_change_pct > 0) & (amount_change_pct < 0) * 0.5 + \
                                (volume_change_pct < 0) & (amount_change_pct > 0) * 0.5 + \
                                (volume_change_pct < 0) & (amount_change_pct < 0) * -1.5
    
    adjusted_factor = sma_smoothed * combined_trend_adjustment
    
    # Combine and Smooth Final Factor
    final_sma_window = 10  # Example window size for final SMA
    final_factor = adjusted_factor.rolling(window=final_sma_window).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Compute Short-Term Volume Trend
    short_term_volume = df['volume'].rolling(window=5).mean()
    
    # Compute Long-Term Volume Trend
    long_term_volume = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume / long_term_volume
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = adjusted_factor * volume_trend_ratio
    final_momentum = intraday_momentum + long_term_momentum
    
    # Combine Momentum and Volatility
    combined_momentum = final_momentum - high_low_spread
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low = (df['high'] - df['low']) * df['volume']
    
    # Multiply Volume-Weighted High-Low Range by Close-to-Previous Close Return
    volume_weighted_momentum = volume_weighted_high_low * price_change_pct
    
    # Smooth the Resulting Series
    smoothed_final_factor = volume_weighted_momentum.ewm(alpha=ealpha, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha = combined_momentum * vwap + high_low_spread.rolling(window=5).mean() + smoothed_final_factor.rolling(window=final_sma_window).mean()
    
    # Introduce a New Intraday Price Reversal Component
    intraday_reversal = (df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)) - intraday_return
    final_alpha *= smoothed_final_factor + intraday_reversal
    
    # Consider Market Sentiment
    if 'sentiment' in df.columns:
        final_alpha *= df['sentiment']
    
    return final_alpha
```
