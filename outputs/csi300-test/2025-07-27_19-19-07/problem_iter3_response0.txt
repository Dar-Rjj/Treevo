```python
import pandas as pd

def heuristics_v2(df, window=20, alpha=0.2):
    """
    Calculate the Volume-Weighted High-Low Range Momentum Factor.
    
    Parameters:
    - df: pandas DataFrame with columns (open, high, low, close, amount, volume)
    - window: int, the size of the moving window for cumulative sum
    - alpha: float, the smoothing factor for the exponential moving average
    
    Returns:
    - pandas Series with the calculated factor values
    """
    
    # Calculate Volume-Weighted High-Low Range
    vol_weighted_high_low_range = (df['high'] - df['low']) * df['volume']
    
    # Calculate Close-to-Previous Close Return
    close_to_prev_close_return = df['close'].pct_change()
    
    # Multiply Volume-Weighted High-Low Range by Close-to-Previous Close Return
    weighted_returns = vol_weighted_high_low_range * close_to_prev_close_return
    
    # Calculate Cumulative Sum of Weighted Returns over a Moving Window
    cum_sum_weighted_returns = weighted_returns.rolling(window=window).sum()
    
    # Smooth the Resulting Series using Exponential Moving Average
    smoothed_series = cum_sum_weighted_returns.ewm(alpha=alpha).mean()
    
    return smoothed_series
```
