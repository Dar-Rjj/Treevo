```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Return
    close_to_prev_close_return = df['close'].pct_change()
    
    # Multiply High-Low Range by Close-to-Previous Close Return
    multiplied_values = high_low_range * close_to_prev_close_return
    
    # Calculate Cumulative Sum of Weighted Returns over a Moving Window
    cumulative_sum = multiplied_values.rolling(window=5).sum()
    
    # Smooth the Resulting Series using Simple Moving Average
    smoothed_series = cumulative_sum.rolling(window=5).mean()
    
    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    adjusted_factor = smoothed_series * (1 + volume_change)
    
    # Combine and Smooth Final Factor
    final_factor = adjusted_factor.rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']
    
    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = df['close'].pct_change(periods=20) * volume_trend_ratio
    combined_momentum = intraday_momentum + long_term_momentum
    
    # Compute Volatility Component
    intraday_volatility = intraday_return.rolling(window=20).std()
    open_to_close_volatility = open_to_close_return.rolling(window=20).std()
    total_volatility = intraday_volatility + open_to_close_volatility
    
    # Combine Momentum and Volatility
    combined_momentum_adjusted = combined_momentum - total_volatility
    
    # Final Alpha Factor
    final_alpha_factor = combined_momentum_adjusted * vwap + high_low_range.rolling(window=5).sum() + final_factor
    
    return final_alpha_factor
```
