```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    high_low_range = df['high'] - df['low']
    volume_weighted_high_low_range = high_low_range * df['volume']

    # Calculate Close-to-Previous Close Return
    close_to_prev_close_return = df['close'].pct_change()

    # Multiply Volume-Weighted High-Low Range by Close-to-Previous Close Return
    weighted_momentum = volume_weighted_high_low_range * close_to_prev_close_return

    # Smooth the Resulting Series
    alpha = 0.3  # Exponential Moving Average smoothing factor
    smoothed_series = weighted_momentum.ewm(alpha=alpha).mean()

    # Adjust for Volume Trend
    volume_change = df['volume'].pct_change()
    adjusted_factor = smoothed_series * np.sign(volume_change)

    # Combine and Smooth Final Factor
    window_size = 5
    final_smoothed_factor = adjusted_factor.rolling(window=window_size).mean()

    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']

    # Calculate Open-to-Close Return
    open_to_close_return = df['open'] - df['close']

    # Compute Short-Term Volume Trend
    short_term_volume_trend = df['volume'].rolling(window=5).mean()

    # Calculate Long-Term Volume Trend
    long_term_volume_trend = df['volume'].rolling(window=20).mean()

    # Compute Volume Trend Ratio
    volume_trend_ratio = short_term_volume_trend / long_term_volume_trend

    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    vwap = vwap.cumsum() / df['volume'].cumsum()

    # Compute Final Momentum Component
    intraday_momentum = intraday_return + open_to_close_return
    long_term_momentum = df['close'].pct_change() * volume_trend_ratio

    # Compute Volatility Component
    intraday_volatility = intraday_return.rolling(window=window_size).std()
    open_close_volatility = open_to_close_return.rolling(window=window_size).std()
    combined_volatility = (intraday_volatility + open_close_volatility) / 2

    # Combine Momentum and Volatility
    combined_momentum = intraday_momentum + long_term_momentum
    adjusted_combined_momentum = combined_momentum - combined_volatility

    # Aggregated High-Low Price Bandwidth
    high_low_bandwidth = (df['high'] - df['low']).rolling(window=window_size).sum()

    # Final Alpha Factor
    final_alpha = (adjusted_combined_momentum * vwap) + high_low_bandwidth + final_smoothed_factor

    return final_alpha
```
