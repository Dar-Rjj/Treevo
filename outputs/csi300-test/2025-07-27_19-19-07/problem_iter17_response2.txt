```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series representing the factor values, indexed by date.
    """
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the price change from the previous day
    price_change = df['close'].diff(1)
    
    # Calculate the volume change from the previous day
    volume_change = df['volume'].diff(1)
    
    # Calculate the amount change from the previous day
    amount_change = df['amount'].diff(1)
    
    # Factor 1: Price momentum adjusted by volume
    factor_1 = price_change / df['close'].shift(1) * volume_change
    
    # Factor 2: Average price to close ratio
    factor_2 = avg_price / df['close']
    
    # Factor 3: Amount change to volume change ratio
    factor_3 = amount_change / volume_change
    
    # Combine the factors into a single alpha factor
    alpha_factor = factor_1 * factor_2 * factor_3
    
    return alpha_factor
```
