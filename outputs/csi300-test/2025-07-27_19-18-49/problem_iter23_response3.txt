import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=30, L=180):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n)
    open_momentum = df['open'] / df['open'].shift(n)
    
    # Adjust Log Return by Open Price
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    open_adjusted_log_return = daily_log_return - np.log(df['open']) + np.log(df['close'].shift(1))
    
    # Multiply Open-Adjusted Log Return by Volume
    volume_adjusted_log_return = open_adjusted_log_return * df['volume']
    
    # Weight by Recent Trading Activity
    avg_volume_m = df['volume'].rolling(window=m).mean()
    long_term_avg_volume = df['volume'].rolling(window=L).mean()
    weight = (avg_volume_m > long_term_avg_volume).astype(int)
    
    # Detect Volume Shocks
    daily_return = df['close'] / df['close'].shift(1)
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_volume_change_30 = volume_change.rolling(window=30).mean()
    sudden_volume_increase = (volume_change > 2.5 * avg_volume_change_30).astype(int)
    
    # Combine Indicators
    momentum_sum = close_momentum + open_momentum
    adjusted_momentum = momentum_sum * sudden_volume_increase * weight
    
    # 5-Day Price and Volume Momentum
    price_momentum_5 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volume_momentum_5 = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # 5-Day Price Oscillation
    price_oscillation_5 = df['high'] - df['low']
    
    # Final Alpha Factor
    cumulative_sum = (volume_adjusted_log_return * weight).rolling(window=n).sum()
    final_alpha_factor = (
        cumulative_sum 
        * price_momentum_5 
        * volume_momentum_5 
        * price_oscillation_5
    )
    
    return final_alpha_factor
