```python
import pandas as pd

def heuristics_v2(df):
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()
    
    # Calculate Integrated Price Momentum
    # (Close - Open) / Open + (Close - Close 1 day ago) / Close 1 day ago
    integrated_momentum_daily = ((df['close'] - df['open']) / df['open']) + \
                                ((df['close'] - df['close'].shift(1)) / df['close'].shift(1))
    
    # Aggregate over the last 5 days: Sum of daily integrated momentums / 5
    integrated_momentum = integrated_momentum_daily.rolling(window=5).sum() / 5
    
    # Calculate Volume Momentum
    # (Volume - Volume 1 day ago) / Volume 1 day ago
    volume_change_daily = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Aggregate over the last 5 days: Sum of daily volume changes / 5
    volume_momentum = volume_change_daily.rolling(window=5).sum() / 5
    
    # Combine Momentum Factors
    # Multiply Integrated Price Momentum by Volume Momentum
    combined_momentum = integrated_momentum * volume_momentum
    
    return combined_momentum
```
