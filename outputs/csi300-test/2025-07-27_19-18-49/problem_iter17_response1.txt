```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, N=20, m=10, L=60):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n) - 1
    open_momentum = df['open'] / df['open'].shift(n) - 1
    volume_momentum = df['volume'] / df['volume'].shift(n) - 1
    high_momentum = df['high'] / df['high'].shift(n) - 1
    low_momentum = df['low'] / df['low'].shift(n) - 1
    
    # Calculate Daily Log Return
    log_return = np.log(df['close']).diff()
    
    # Adjust Log Return by Open Price
    open_adjusted_log_return = log_return - np.log(df['open']) + np.log(df['close'].shift(1))
    
    # Incorporate Price Range Momentum
    price_range = df['high'] - df['low']
    price_range_momentum = price_range / price_range.shift(n) - 1
    
    # Multiply Open-Adjusted Log Return by Volume
    volume_adjusted_log_return = open_adjusted_log_return * df['volume']
    
    # Sum Over N Days
    sum_N_days = volume_adjusted_log_return.rolling(window=N).sum()
    
    # Weight by Recent Trading Activity
    avg_volume_short = df['volume'].rolling(window=m).mean()
    avg_volume_long = df['volume'].rolling(window=L).mean()
    weight = avg_volume_short / avg_volume_long
    
    # Detect Volume Shocks
    daily_return = df['close'].pct_change()
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_30_day_vol_change = volume_change.rolling(window=30).mean()
    threshold = 2 * avg_30_day_vol_change
    volume_shock = (volume_change > threshold).astype(int)
    
    # Combine Indicators
    combined_momentum = (
        close_momentum * weight +
        open_momentum * weight +
        high_momentum * weight +
        low_momentum * weight +
        price_range_momentum * weight
    )
    
    final_alpha_factor = combined_momentum * sum_N_days * volume_shock
    
    return final_alpha_factor
```
