```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=30, L=100, volume_shock_threshold=2.0, gap_percentage=0.02):
    # Momentum Components
    close_momentum = df['close'] / df['close'].shift(n) - 1
    open_momentum = df['open'] / df['open'].shift(n) - 1
    volume_momentum = df['volume'] / df['volume'].shift(n)
    
    # Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Open-Adjusted Daily Log Return
    open_adjusted_log_return = daily_log_return - (np.log(df['open']) - np.log(df['close'].shift(1)))
    
    # Multiply Open-Adjusted Log Return by Volume
    open_adjusted_volume = open_adjusted_log_return * df['volume']
    
    # Sum Over N Days
    cumulative_sum = open_adjusted_volume.rolling(window=n).sum()
    
    # Weight by Recent Trading Activity
    recent_average_volume = df['volume'].rolling(window=m).mean()
    long_term_average_volume = df['volume'].rolling(window=L).mean()
    weight = recent_average_volume / long_term_average_volume
    
    # Detect Volume Shocks
    daily_return = df['close'] / df['close'].shift(1) - 1
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_30_day_volume_change = volume_change.rolling(window=30).mean()
    volume_shock = (volume_change > volume_shock_threshold * avg_30_day_volume_change).astype(int)
    
    # Detect Price Gaps
    gap_size = df['open'] - df['close'].shift(1)
    significant_gap = (gap_size.abs() > gap_percentage * df['close'].shift(1)).astype(int)
    
    # Combine Indicators
    combined_momentum = (close_momentum + open_momentum + volume_momentum) * weight
    alpha_factor = (combined_momentum * volume_shock * significant_gap).shift(1)
    
    return alpha_factor
```
