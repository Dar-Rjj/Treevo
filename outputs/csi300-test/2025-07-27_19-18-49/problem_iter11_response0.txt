```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as [open, high, low, close, amount, volume].
    
    Returns:
    pd.Series: A Series with index as date representing the factor values.
    """
    # Calculate the relative range of the stock price for the day
    df['range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the relative amount traded compared to the average amount over the last 10 days
    df['amount_10d_avg'] = df['amount'].rolling(window=10).mean()
    df['rel_amount'] = df['amount'] / df['amount_10d_avg']
    
    # Calculate the relative volume traded compared to the average volume over the last 10 days
    df['volume_10d_avg'] = df['volume'].rolling(window=10).mean()
    df['rel_volume'] = df['volume'] / df['volume_10d_avg']
    
    # Calculate the price change from the previous day
    df['price_change'] = df['close'].pct_change()
    
    # Combine the factors into a single score
    df['alpha_factor'] = (df['range'] + df['rel_amount'] + df['rel_volume'] + df['price_change']) / 4
    
    return df['alpha_factor'].dropna()
```
