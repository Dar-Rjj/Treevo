import pandas as pd
import numpy as np

def heuristics_v2(df, n_days_momentum=20, m_days_recent_vol=10, L_days_long_vol=60, N_days_cumsum=50):
    # Calculate Daily Log Return
    df['log_close'] = np.log(df['close'])
    df['daily_log_return'] = df['log_close'].diff()

    # Adjust Log Return by Open Price
    df['log_open'] = np.log(df['open'])
    df['adjusted_log_return'] = df['daily_log_return'] - (df['log_open'] - df['log_close'].shift(1))

    # Calculate Momentum Components
    df['close_momentum'] = df['close'] / df['close'].shift(n_days_momentum)
    df['open_momentum'] = df['open'] / df['open'].shift(n_days_momentum)
    df['volume_momentum'] = df['volume'] / df['volume'].shift(n_days_momentum)

    # Weight by Recent Trading Activity
    recent_average_volume = df['volume'].rolling(window=m_days_recent_vol).mean()
    long_term_average_volume = df['volume'].rolling(window=L_days_long_vol).mean()
    volume_weight = recent_average_volume / long_term_average_volume

    # Combine Momentum and Volume Weights
    df['weighted_close_momentum'] = df['close_momentum'] * volume_weight
    df['weighted_open_momentum'] = df['open_momentum'] * volume_weight
    df['combined_momentum'] = df['weighted_close_momentum'] + df['weighted_open_momentum']

    # Cumulative Sum of Volume-Weighted Adjusted Returns
    df['volume_weighted_adjusted_return'] = df['adjusted_log_return'] * df['volume']
    df['cumulative_volume_weighted_adjusted_return'] = df['volume_weighted_adjusted_return'].rolling(window=N_days_cumsum).sum()

    # Final Factor
    factor = df['cumulative_volume_weighted_adjusted_return'] + df['combined_momentum']

    return factor.dropna()
