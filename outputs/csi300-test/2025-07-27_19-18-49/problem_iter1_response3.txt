```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily price momentum
    df['price_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate price momentum over the last 5 days
    df['price_momentum_5d'] = df['price_momentum'].rolling(window=5).sum() / 5
    
    # Calculate daily volume momentum
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Aggregate volume momentum over the last 5 days
    df['volume_momentum_5d'] = df['volume_change'].rolling(window=5).sum() / 5
    
    # Combine Price and Volume Momentum
    df['alpha_factor'] = df['price_momentum_5d'] * df['volume_momentum_5d']
    
    return df['alpha_factor']
```
