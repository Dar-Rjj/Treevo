import pandas as pd
import numpy as np

def heuristics_v2(df, m=5, L=30, N=20):
    # Momentum Components
    df['Close_Momentum'] = (df['close'] / df['close'].shift(5)) - 1
    df['Open_Momentum'] = (df['open'] / df['open'].shift(5)) - 1
    df['Volume_Momentum'] = (df['volume'] / df['volume'].shift(5)) - 1
    df['High_Momentum'] = (df['high'] / df['high'].shift(5)) - 1
    df['Low_Momentum'] = (df['low'] / df['low'].shift(5)) - 1
    
    # Daily Log Return
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust Log Return by Open Price
    df['Open_Adjusted_Log_Return'] = df['Log_Return'] - np.log(df['open']) + np.log(df['close'].shift(1))
    
    # Multiply Open-Adjusted Log Return by Volume
    df['Volume_Adjusted_Log_Return'] = df['Open_Adjusted_Log_Return'] * df['volume']
    
    # Weight by Recent Trading Activity
    df['Avg_Volume_m'] = df['volume'].rolling(window=m).mean()
    df['Avg_Volume_L'] = df['volume'].rolling(window=L).mean()
    df['Weight'] = df['Avg_Volume_m'] / df['Avg_Volume_L']
    
    # Price Range and Momentum
    df['Price_Range'] = df['high'] - df['low']
    df['Price_Range_Momentum'] = (df['Price_Range'] / df['Price_Range'].shift(5)) - 1
    
    # 5-Day Price Oscillation
    df['Daily_Price_Range'] = df['high'] - df['low']
    df['5_Day_Price_Oscillation'] = df['Daily_Price_Range'].rolling(window=5).sum() / 5
    
    # 5-Day Volume Trend
    df['5_Day_Volume_Trend'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    df['Smoothed_Volume_Trend'] = df['5_Day_Volume_Trend'].rolling(window=5).mean()
    
    # Detect Volume Shocks
    df['Daily_Return'] = df['close'].pct_change()
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    avg_vol_change_30 = df['Volume_Change'].rolling(window=30).mean()
    df['Volume_Shock'] = (df['Volume_Change'] > 2 * avg_vol_change_30).astype(int)
    
    # Integrate Price, Volume, and Close Momentum with Open-Price Integration
    df['Integrated_Momentum'] = (df['close'] - df['open']) / df['open'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5_Day_Integrated_Momentum'] = df['Integrated_Momentum'].rolling(window=5).sum() / 5
    
    # Combine Indicators
    df['Combined_Factor'] = (
        df['Close_Momentum'] * df['Weight'] +
        df['Open_Momentum'] * df['Weight'] +
        df['High_Momentum'] * df['Weight'] +
        df['Low_Momentum'] * df['Weight'] +
        df['Price_Range_Momentum'] * df['Weight'] +
        df['5_Day_Price_Oscillation'] * df['Weight'] +
        df['Smoothed_Volume_Trend'] * df['Weight']
    ) * df['Volume_Shock']
    
    # Final Alpha Factor
    df['Volume_Weighted_Adjusted_Returns'] = df['Volume_Adjusted_Log_Return'] * df['Weight']
    df['Cumulative_Sum'] = df['Volume_Weighted_Adjusted_Returns'].rolling(window=N).sum()
    df['Final_Alpha_Factor'] = df['Combined_Factor'] + df['Cumulative_Sum']
    
    return df['Final_Alpha_Factor']
