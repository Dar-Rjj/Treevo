```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, m=5, L=100):
    # Calculate Momentum Components
    df['Close_Momentum'] = df['close'].pct_change(5)
    df['Open_Momentum'] = df['open'].pct_change(5)
    df['Volume_Momentum'] = df['volume'].pct_change(5)
    df['High_Momentum'] = df['high'].pct_change(5)
    df['Low_Momentum'] = df['low'].pct_change(5)
    
    # Calculate Daily Log Return
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust Log Return by Open Price
    df['Open_Adjusted_Log_Return'] = df['Log_Return'] - np.log(df['open']) + np.log(df['close'].shift(1))
    
    # Multiply Open-Adjusted Log Return by Volume
    df['Volume_Adjusted_Return'] = df['Open_Adjusted_Log_Return'] * df['volume']
    
    # Sum Over N Days
    df['Sum_N_Days'] = df['Volume_Adjusted_Return'].rolling(window=N).sum()
    
    # Weight by Recent Trading Activity
    df['Recent_Volume_Avg'] = df['volume'].rolling(window=m).mean()
    df['Long_Term_Volume_Avg'] = df['volume'].rolling(window=L).mean()
    df['Volume_Weight'] = df['Recent_Volume_Avg'] / df['Long_Term_Volume_Avg']
    
    # Incorporate Price Range Momentum
    df['Price_Range'] = df['high'] - df['low']
    df['Price_Range_Momentum'] = df['Price_Range'].pct_change(5)
    
    # Calculate 5-Day Price Oscillation
    df['Price_Oscillation'] = (df['high'] - df['low']).rolling(window=5).sum() / 5
    
    # Calculate 5-Day Volume Trend
    df['Volume_Trend'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Detect Volume Shocks
    df['Daily_Return'] = df['close'].pct_change()
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    avg_vol_change = df['Volume_Change'].rolling(window=30).mean()
    df['Volume_Shock'] = (df['Volume_Change'] > 2 * avg_vol_change).astype(int)
    
    # Combine Momentum and Volume Weights
    df['Momentum_Weight'] = (df['Close_Momentum'] + df['Open_Momentum'] + df['High_Momentum'] + df['Low_Momentum'] + df['Price_Range_Momentum'] + df['Price_Oscillation'] + df['Volume_Trend']) * df['Volume_Weight']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Sum_N_Days'] * df['Volume_Shock']
    
    return df['Alpha_Factor']
```
