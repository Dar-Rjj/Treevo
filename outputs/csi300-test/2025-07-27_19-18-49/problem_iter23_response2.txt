import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=10, L=60):
    # Calculate Daily Log Return
    df['log_close'] = np.log(df['close'])
    df['daily_log_return'] = df['log_close'] - df['log_close'].shift(1)
    
    # Adjust Log Return by Open Price
    df['log_open'] = np.log(df['open'])
    df['open_adjusted_log_return'] = df['daily_log_return'] - df['log_open'] + df['log_close'].shift(1)
    
    # Calculate Momentum Components
    df['close_momentum'] = df['close'] / df['close'].shift(n)
    df['open_momentum'] = df['open'] / df['open'].shift(n)
    df['volume_momentum'] = df['volume'] / df['volume'].shift(n)
    df['high_momentum'] = df['high'] / df['high'].shift(n)
    df['low_momentum'] = df['low'] / df['low'].shift(n)
    
    # Weight by Recent Trading Activity
    df['avg_vol_m_days'] = df['volume'].rolling(window=m).mean()
    df['long_term_avg_vol'] = df['volume'].rolling(window=L).mean()
    df['weight'] = np.where(df['avg_vol_m_days'] > df['long_term_avg_vol'], 1, 0.5)
    
    # Incorporate Price Range Momentum
    df['price_range'] = df['high'] - df['low']
    df['price_range_momentum'] = df['price_range'] / df['price_range'].shift(n)
    
    # Integrate Price Momentum
    df['integrated_price_momentum'] = ((df['close'] - df['open']) / df['open'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)).rolling(window=n).sum() / n
    
    # Calculate 5-Day Price Oscillation
    df['price_oscillation'] = (df['high'] - df['low']).rolling(window=n).sum() / n
    
    # Calculate 5-Day Volume Trend
    df['volume_trend'] = (df['volume'] - df['volume'].shift(n)) / df['volume'].shift(n)
    df['smoothed_volume_trend'] = df['volume_trend'].rolling(window=n).mean()
    
    # Detect Volume Shocks
    df['daily_return'] = df['close'] / df['close'].shift(1)
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['sudden_volume_increase'] = np.where(df['volume_change'] > 2.5 * df['volume_change'].rolling(window=30).mean(), 1, 0)
    
    # Combine Indicators
    df['momentum_sum'] = df['close_momentum'] + df['open_momentum'] + df['volume_momentum'] + df['high_momentum'] + df['low_momentum']
    df['volume_shock_indicator'] = df['sudden_volume_increase']
    df['weighted_momentum'] = df['momentum_sum'] * df['volume_shock_indicator'] * df['weight']
    df['final_factor'] = df['weighted_momentum'] * df['integrated_price_momentum'] * df['price_oscillation'] * df['smoothed_volume_trend']
    
    # Cumulative Sum of Volume-Weighted Adjusted Returns
    df['vol_weighted_adj_return'] = df['volume'] * df['open_adjusted_log_return'] * df['weight']
    df['cumulative_sum'] = df['vol_weighted_adj_return'].cumsum()
    
    return df['final_factor']
