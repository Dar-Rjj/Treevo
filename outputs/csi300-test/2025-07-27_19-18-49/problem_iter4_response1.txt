```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['price_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volume Momentum
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Aggregate Averages
    df['average_price_momentum'] = df['price_momentum'].rolling(window=5).mean()
    df['average_volume_momentum'] = df['volume_momentum'].rolling(window=5).mean()
    
    # Combine Averages
    df['alpha_factor'] = df['average_price_momentum'] * df['average_volume_momentum']
    
    return df['alpha_factor'].dropna()
```
