```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum Components
    df['Close_Momentum'] = df['close'] / df['close'].shift(10)
    df['Open_Momentum'] = df['open'] / df['open'].shift(10)
    df['Volume_Momentum'] = df['volume'] / df['volume'].shift(10)
    df['High_Momentum'] = df['high'] / df['high'].shift(10)
    df['Low_Momentum'] = df['low'] / df['low'].shift(10)

    # Calculate Daily Log Return
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))

    # Adjust Log Return by Open Price
    df['Open_Adjusted_Log_Return'] = (np.log(df['close']) - np.log(df['open'])) + np.log(df['close'].shift(1))

    # Multiply Open-Adjusted Log Return by Volume
    df['Volume_Adjusted_Log_Return'] = df['Open_Adjusted_Log_Return'] * df['volume']

    # Weight by Recent Trading Activity
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Long_Term_Avg_Volume'] = df['volume'].expanding().mean()
    df['Volume_Ratio'] = df['Avg_Volume_20'] / df['Long_Term_Avg_Volume']

    # Incorporate Price Range Momentum
    df['Price_Range_Momentum'] = (df['high'] - df['low']) / (df['high'].shift(10) - df['low'].shift(10))

    # Calculate 10-Day Price Oscillation
    df['Daily_Price_Range'] = df['high'] - df['low']
    df['10_Day_Price_Oscillation'] = df['Daily_Price_Range'].rolling(window=10).sum() / 10

    # Calculate 10-Day Volume Trend
    df['10_Day_Volume_Trend'] = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10)

    # Detect Volume Shocks
    df['Daily_Return'] = df['close'] / df['close'].shift(1) - 1
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    volume_shock_threshold = df['Volume_Change'].quantile(0.95)
    df['Volume_Shock'] = (df['Volume_Change'] > volume_shock_threshold).astype(int)

    # Integrate Price, Volume, and Close Momentum with Open-Price Integration
    df['Integrated_Momentum'] = (df['close'] - df['open']) / df['open'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Indicators
    df['Combined_Indicators'] = (
        df['Close_Momentum'] * df['Volume_Ratio'] +
        df['Open_Momentum'] * df['Volume_Ratio'] +
        df['High_Momentum'] * df['Volume_Ratio'] +
        df['Low_Momentum'] * df['Volume_Ratio'] +
        df['Price_Range_Momentum'] * df['Volume_Ratio'] +
        df['10_Day_Price_Oscillation'] * df['Volume_Ratio'] +
        df['10_Day_Volume_Trend'] * df['Volume_Ratio'] +
        df['Volume_Shock']
    )

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Volume_Adjusted_Log_Return'].cumsum()

    return df['Final_Alpha_Factor']
```
