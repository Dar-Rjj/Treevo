```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=10, L=60, N=20):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n) - 1
    open_momentum = df['open'] / df['open'].shift(n) - 1
    volume_momentum = df['volume'] / df['volume'].shift(n) - 1
    
    # Calculate Daily Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust Log Return by Open Price
    adjusted_log_return = log_return - (np.log(df['open']) - np.log(df['close'].shift(1)))
    
    # Multiply Open-Adjusted Log Return by Volume
    volume_weighted_adjusted_return = adjusted_log_return * df['volume']
    
    # Weight by Recent Trading Activity
    recent_volume_avg = df['volume'].rolling(window=m).mean()
    long_term_volume_avg = df['volume'].rolling(window=L).mean()
    trading_activity_weight = recent_volume_avg / long_term_volume_avg
    
    # Detect Volume Shocks
    daily_return = df['close'] / df['close'].shift(1) - 1
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_shock_threshold = 2 * volume_change.rolling(window=30).mean()
    volume_shock = (volume_change > volume_shock_threshold).astype(int)
    
    # Integrate Price, Volume, and Close Momentum with Open-Price Integration
    integrated_momentum = ((df['close'] - df['open']) / df['open'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)).rolling(window=n).mean()
    
    # Combine Indicators
    combined_momentum = (close_momentum * trading_activity_weight + open_momentum * trading_activity_weight + volume_momentum) * volume_shock
    combined_momentum *= integrated_momentum
    combined_momentum *= (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_momentum *= (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    combined_momentum *= (df['high'] - df['low']) / df['close']
    
    # Cumulative Sum of Volume-Weighted Adjusted Returns
    cumulative_volume_weighted_returns = volume_weighted_adjusted_return.rolling(window=N).sum()
    
    # Final Alpha Factor
    alpha_factor = combined_momentum * cumulative_volume_weighted_returns
    
    return alpha_factor
```
