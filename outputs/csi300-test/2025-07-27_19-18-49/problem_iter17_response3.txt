```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, L=100, N=50):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n) - 1
    open_momentum = df['open'] / df['open'].shift(n) - 1
    volume_momentum = df['volume'] / df['volume'].shift(n) - 1
    
    # Calculate Daily Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust Log Return by Open Price
    open_adjusted_log_return = (np.log(df['close']) - np.log(df['open'])) + np.log(df['close'].shift(1))
    
    # Multiply Open-Adjusted Log Return by Volume
    volume_weighted_adj_log_return = open_adjusted_log_return * df['volume']
    
    # Weight by Recent Trading Activity
    recent_vol_avg = df['volume'].rolling(window=m).mean()
    long_term_vol_avg = df['volume'].rolling(window=L).mean()
    trading_activity_weight = recent_vol_avg / long_term_vol_avg
    
    # Detect Volume Shocks
    daily_return = df['close'] / df['close'].shift(1) - 1
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_shock_threshold = 2 * volume_change.rolling(window=30).mean()
    volume_shock = (volume_change > volume_shock_threshold).astype(int)
    
    # Combine Indicators
    combined_momentum = (close_momentum * trading_activity_weight + 
                         open_momentum * trading_activity_weight + 
                         volume_momentum) * volume_shock
    
    # Cumulative Sum of Volume-Weighted Adjusted Returns
    cum_sum_vol_weighted_returns = (volume_weighted_adj_log_return * combined_momentum).rolling(window=N).sum()
    
    return cum_sum_vol_weighted_returns
```
