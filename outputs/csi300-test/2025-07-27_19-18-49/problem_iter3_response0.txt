```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Integrated Price Momentum
    df['close_1d_ago'] = df['close'].shift(1)
    df['open_today'] = df['open']
    df['integrated_momentum_daily'] = ((df['close'] - df['open_today']) / df['open_today']) + \
                                      ((df['close'] - df['close_1d_ago']) / df['close_1d_ago'])
    
    df['integrated_momentum_5d'] = df['integrated_momentum_daily'].rolling(window=5).sum() / 5
    
    # Calculate Volume Momentum
    df['volume_1d_ago'] = df['volume'].shift(1)
    df['volume_change_daily'] = (df['volume'] - df['volume_1d_ago']) / df['volume_1d_ago']
    
    df['volume_momentum_5d'] = df['volume_change_daily'].rolling(window=5).sum() / 5
    
    # Combine Integrated Price and Volume Momentum
    df['alpha_factor'] = df['integrated_momentum_5d'] * df['volume_momentum_5d']
    
    return df['alpha_factor'].dropna()
```
