```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily integrated momentum
    df['integrated_momentum'] = (df['close'] - df['open']) / df['open'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate: Sum of daily integrated momentums over last 5 days / 5
    df['integrated_momentum_5d'] = df['integrated_momentum'].rolling(window=5).mean()

    # Calculate 5-day price momentum
    df['price_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate: Sum of daily price momentums over last 5 days / 5
    df['price_momentum_5d'] = df['price_momentum'].rolling(window=5).mean()

    # Calculate 5-day volume momentum
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Aggregate: Sum of daily volume changes over last 5 days / 5
    df['volume_momentum_5d'] = df['volume_change'].rolling(window=5).mean()

    # Calculate 5-day price oscillation
    df['price_range'] = df['high'] - df['low']
    
    # Aggregate: Sum of daily price ranges over last 5 days / 5
    df['price_oscillation_5d'] = df['price_range'].rolling(window=5).mean()

    # Combine all factors
    df['alpha_factor'] = (df['integrated_momentum_5d'] * 
                          df['price_momentum_5d'] * 
                          df['volume_momentum_5d'] * 
                          df['price_oscillation_5d'])

    return df['alpha_factor']
```
