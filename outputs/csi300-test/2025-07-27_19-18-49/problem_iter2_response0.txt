```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, L=60, N=50):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']).diff()
    
    # Calculate Open-Adjusted Daily Log Return
    df['open_adjusted_log_return'] = df['log_return'] - np.log(df['open']).diff() + np.log(df['close']).shift(1)
    
    # Calculate Momentum Components
    df['close_momentum'] = (np.log(df['close']) - np.log(df['close'].shift(n)))
    df['open_momentum'] = (np.log(df['open']) - np.log(df['open'].shift(n)))
    df['volume_momentum'] = (df['volume'] / df['volume'].shift(n))
    
    # Weight by Recent Trading Activity
    df['avg_volume_m'] = df['volume'].rolling(window=m).mean()
    df['avg_volume_L'] = df['volume'].rolling(window=L).mean()
    df['volume_weight'] = df['avg_volume_m'] / df['avg_volume_L']
    
    # Combine Momentum and Volume Weights
    df['weighted_close_momentum'] = df['close_momentum'] * df['volume_weight']
    df['weighted_open_momentum'] = df['open_momentum'] * df['volume_weight']
    df['total_momentum'] = df['weighted_close_momentum'] + df['weighted_open_momentum'] + df['volume_momentum']
    
    # Multiply Open-Adjusted Log Return by Volume
    df['open_adj_vol_weighted_return'] = df['open_adjusted_log_return'] * df['volume']
    
    # Cumulative Sum of Open-Adjusted Volume Weighted Returns
    df['cumulative_sum'] = df['open_adj_vol_weighted_return'].rolling(window=N).sum()
    
    # Final Alpha Factor
    alpha_factor = df['cumulative_sum']
    
    return alpha_factor
```
