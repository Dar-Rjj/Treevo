```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=50, m=10, L=100):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(5)
    open_momentum = df['open'] / df['open'].shift(5)
    volume_momentum = df['volume'] / df['volume'].shift(5)
    high_momentum = df['high'] / df['high'].shift(5)
    low_momentum = df['low'] / df['low'].shift(5)

    # Calculate Daily Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))

    # Adjust Log Return by Open Price
    open_adjusted_log_return = log_return - (np.log(df['open']) - np.log(df['close'].shift(1)))

    # Multiply Open-Adjusted Log Return by Volume
    volume_weighted_log_return = open_adjusted_log_return * df['volume']

    # Sum Over N Days
    cumulative_sum = volume_weighted_log_return.rolling(window=N).sum()

    # Weight by Recent Trading Activity
    recent_avg_volume = df['volume'].rolling(window=m).mean()
    long_term_avg_volume = df['volume'].rolling(window=L).mean()
    volume_weight = recent_avg_volume / long_term_avg_volume

    # Incorporate Price Range Momentum
    price_range = df['high'] - df['low']
    price_range_momentum = price_range / price_range.shift(5)

    # Calculate 5-Day Price Oscillation
    price_oscillation = (df['high'] - df['low']).rolling(window=5).sum() / 5

    # Calculate 5-Day Volume Trend
    volume_trend = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    smoothed_volume_trend = volume_trend.rolling(window=5).mean()

    # Detect Volume Shocks
    daily_return = df['close'] / df['close'].shift(1) - 1
    volume_change = df['volume'] - df['volume'].shift(1)
    average_vol_change = volume_change.rolling(window=30).mean()
    volume_shock = (volume_change > 2 * average_vol_change).astype(int)

    # Integrate Price, Volume, and Close Momentum with Open-Price Integration
    integrated_momentum = ((df['close'] - df['open']) / df['open'] + 
                           (df['close'] - df['close'].shift(1)) / df['close'].shift(1))
    aggregated_momentum = integrated_momentum.rolling(window=5).sum() / 5

    # Combine Indicators
    combined_indicators = (
        close_momentum * volume_weight +
        open_momentum * volume_weight +
        high_momentum * volume_weight +
        low_momentum * volume_weight +
        price_range_momentum * volume_weight +
        price_oscillation * volume_weight +
        smoothed_volume_trend * volume_weight
    ) * volume_shock

    # Final Alpha Factor
    final_alpha_factor = cumulative_sum * combined_indicators

    return final_alpha_factor
```
