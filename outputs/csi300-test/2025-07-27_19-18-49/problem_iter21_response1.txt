```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=10, L=60, N=30):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n)
    open_momentum = df['open'] / df['open'].shift(n)

    # Adjust Log Return by Open Price
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    open_adjusted_log_return = daily_log_return - np.log(df['open']) + np.log(df['close'].shift(1))

    # Multiply Open-Adjusted Log Return by Volume
    volume_weighted_log_return = df['volume'] * open_adjusted_log_return

    # Weight by Recent Trading Activity
    avg_volume_last_m_days = df['volume'].rolling(m).mean()
    long_term_avg_volume = df['volume'].rolling(L).mean()
    trading_activity_weight = (avg_volume_last_m_days > long_term_avg_volume).astype(int)

    # Detect Volume Shocks
    daily_return = df['close'] / df['close'].shift(1)
    volume_change = df['volume'] - df['volume'].shift(1)
    average_30_day_vol_change = volume_change.rolling(N).mean()
    volume_shock_indicator = (volume_change > 2.5 * average_30_day_vol_change).astype(int)

    # Combine Indicators
    momentum_sum = close_momentum + open_momentum
    adjusted_momentum = momentum_sum * volume_shock_indicator * trading_activity_weight

    # Cumulative Sum of Volume-Weighted Adjusted Returns
    cumulative_sum = (volume_weighted_log_return * trading_activity_weight).cumsum()

    # Alpha Factor
    alpha_factor = cumulative_sum * adjusted_momentum

    return alpha_factor
```
