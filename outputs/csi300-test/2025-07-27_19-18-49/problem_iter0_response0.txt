```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Volume Weighted Close-to-Open Return.
    
    Parameters:
    - df: pandas DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    
    Returns:
    - pandas Series indexed by (date) representing the factor values
    """
    # Calculate the close-to-open return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight the return by volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    return volume_weighted_return
```
