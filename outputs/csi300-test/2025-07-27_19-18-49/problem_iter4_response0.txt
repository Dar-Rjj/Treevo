```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20):
    # Compute Logarithm of Close Price
    log_close = np.log(df['close'])
    
    # Compute Daily Log Return
    log_return = log_close - log_close.shift(1)
    
    # Adjust Log Return by Open Price
    adjusted_log_return = log_return - np.log(df['open']) + np.log(df['close'].shift(1))
    
    # Multiply Adjusted Log Return by Volume
    volume_weighted_adjusted_return = adjusted_log_return * df['volume']
    
    # Cumulative Sum Over N Days
    cumulative_volume_weighted_momentum = volume_weighted_adjusted_return.rolling(window=N).sum()
    
    return cumulative_volume_weighted_momentum
```
