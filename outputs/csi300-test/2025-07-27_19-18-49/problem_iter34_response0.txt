```python
import pandas as pd
import numpy as np

def heuristics_v2(df, m=10, L=60, N=20):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(5)
    open_momentum = df['open'] / df['open'].shift(5)
    high_momentum = df['high'] / df['high'].shift(5)
    low_momentum = df['low'] / df['low'].shift(5)

    # Calculate Daily Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))

    # Adjust Log Return by Open Price
    open_adjusted_log_return = (np.log(df['close']) - np.log(df['open'])) + (np.log(df['close'].shift(1)) - np.log(df['open'].shift(1)))

    # Multiply Open-Adjusted Log Return by Volume
    volume_weighted_log_return = open_adjusted_log_return * df['volume']

    # Weight by Recent Trading Activity
    recent_avg_volume = df['volume'].rolling(m).mean()
    long_term_avg_volume = df['volume'].rolling(L).mean()
    weight = recent_avg_volume / long_term_avg_volume

    # Incorporate Price Range Momentum
    price_range = df['high'] - df['low']
    price_range_momentum = price_range / price_range.shift(5)

    # Calculate 5-Day Price Oscillation
    price_oscillation = (df['high'] - df['low']).rolling(5).sum() / 5

    # Detect Volume Shocks
    daily_return = df['close'] / df['close'].shift(1) - 1
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_shock_threshold = 2 * volume_change.rolling(30).mean()
    volume_shock = (volume_change > volume_shock_threshold).astype(int)

    # Integrate Price, Volume, and Close Momentum with Open-Price Integration
    integrated_momentum = (df['close'] - df['open']) / df['open'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    integrated_momentum_5_day = integrated_momentum.rolling(5).sum() / 5

    # Combine Indicators
    combined_indicators = (close_momentum * weight + 
                           open_momentum * weight + 
                           high_momentum * weight + 
                           low_momentum * weight + 
                           price_range_momentum * weight + 
                           price_oscillation * weight) * volume_shock

    # Final Alpha Factor
    alpha_factor = (combined_indicators * volume_weighted_log_return).cumsum().rolling(N).sum()

    return alpha_factor
```
