```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, L=50):
    # Integrated Close-Price Momentum
    close_momentum = (df['close'] - df['open']) / df['open'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    close_momentum[close_momentum.isna()] = 0  # Handle NaNs
    
    # Integrated Open-Price Momentum
    open_momentum = (df['open'] - df['open'].shift(n)) / df['open'].shift(n)
    open_momentum[open_momentum.isna()] = 0  # Handle NaNs
    
    # Integrated Volume Momentum
    volume_momentum = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_momentum[volume_momentum.isna()] = 0  # Handle NaNs
    
    # Aggregate Momentum Over N Days
    close_momentum_sum = close_momentum.rolling(window=n).sum()
    open_momentum_sum = open_momentum.rolling(window=n).sum()
    volume_momentum_sum = volume_momentum.rolling(window=n).sum()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Open-Adjusted Daily Log Return
    open_adjusted_log_return = daily_log_return - np.log(df['open']) + np.log(df['close'].shift(1))
    
    # Multiply Open-Adjusted Log Return by Volume
    volume_weighted_return = open_adjusted_log_return * df['volume']
    
    # Sum Over N Days
    volume_weighted_return_sum = volume_weighted_return.rolling(window=n).sum()
    
    # Weight by Recent Trading Activity
    avg_volume_m = df['volume'].rolling(window=m).mean()
    avg_volume_L = df['volume'].rolling(window=L).mean()
    volume_weight = avg_volume_m / avg_volume_L
    
    # Combine Integrated Momentum and Volume Weights
    combined_factor = (close_momentum_sum * volume_weight +
                       open_momentum_sum * volume_weight +
                       volume_momentum_sum * volume_weight +
                       volume_weighted_return_sum)
    
    return combined_factor
```
