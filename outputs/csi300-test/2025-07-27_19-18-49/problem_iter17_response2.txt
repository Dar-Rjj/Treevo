```python
import pandas as pd
import numpy as np

def heuristics_v2(df, m=10, L=100, N=20):
    # Calculate Daily Log Return
    df['log_close'] = np.log(df['close'])
    df['log_return'] = df['log_close'] - df['log_close'].shift(1)
    
    # Adjust Log Return by Open Price
    df['log_open'] = np.log(df['open'])
    df['adj_log_return'] = df['log_return'] - (df['log_open'] - df['log_close'].shift(1))
    
    # Calculate Momentum Components
    df['close_mom'] = df['close'] / df['close'].shift(5)
    df['open_mom'] = df['open'] / df['open'].shift(5)
    df['volume_mom'] = df['volume'] / df['volume'].shift(5)
    df['high_mom'] = df['high'] / df['high'].shift(5)
    df['low_mom'] = df['low'] / df['low'].shift(5)
    
    # Price Range Momentum
    df['price_range'] = df['high'] - df['low']
    df['price_range_mom'] = df['price_range'] / df['price_range'].shift(5)
    
    # Weight by Recent Trading Activity
    df['avg_volume_m'] = df['volume'].rolling(window=m).mean()
    df['avg_volume_L'] = df['volume'].rolling(window=L).mean()
    df['volume_weight'] = df['avg_volume_m'] / df['avg_volume_L']
    
    # Integrated Price Momentum
    df['integrated_price_mom'] = ((df['close'] - df['open']) / df['open'] + 
                                  (df['close'] - df['close'].shift(1)) / df['close'].shift(1))
    df['integrated_price_mom_5'] = df['integrated_price_mom'].rolling(window=5).mean()
    
    # 5-Day Price Oscillation
    df['price_oscillation_5'] = df['high'] - df['low']
    df['price_oscillation_5'] = df['price_oscillation_5'].rolling(window=5).mean()
    
    # 5-Day Volume Trend
    df['volume_trend_5'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    df['smoothed_volume_trend_5'] = df['volume_trend_5'].rolling(window=5).mean()
    
    # Combine Momentum and Volume Weights
    df['weighted_close_mom'] = df['close_mom'] * df['volume_weight']
    df['weighted_open_mom'] = df['open_mom'] * df['volume_weight']
    df['weighted_high_mom'] = df['high_mom'] * df['volume_weight']
    df['weighted_low_mom'] = df['low_mom'] * df['volume_weight']
    df['weighted_price_range_mom'] = df['price_range_mom'] * df['volume_weight']
    df['weighted_integrated_price_mom'] = df['integrated_price_mom_5'] * df['volume_weight']
    df['weighted_price_oscillation_5'] = df['price_oscillation_5'] * df['volume_weight']
    df['weighted_volume_trend_5'] = df['smoothed_volume_trend_5'] * df['volume_weight']
    
    # Cumulative Sum of Volume-Weighted Adjusted Returns
    df['vol_weighted_adj_log_return'] = df['adj_log_return'] * df['volume']
    df['cum_vol_weighted_adj_log_return'] = df['vol_weighted_adj_log_return'].rolling(window=N).sum()
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['weighted_close_mom'] + df['weighted_open_mom'] + 
                          df['weighted_high_mom'] + df['weighted_low_mom'] + 
                          df['weighted_price_range_mom'] + df['weighted_integrated_price_mom'] + 
                          df['weighted_price_oscillation_5'] + df['weighted_volume_trend_5'] + 
                          df['cum_vol_weighted_adj_log_return'])
    
    return df['alpha_factor'].dropna()
```
