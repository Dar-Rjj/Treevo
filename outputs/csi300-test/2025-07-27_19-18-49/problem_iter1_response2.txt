```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['Price_Momentum'] = df['close'] / df['SMA_10']
    
    # Detect Volume Shocks
    df['Daily_Return'] = df['close'].pct_change()
    df['Volume_Change'] = df['volume'].diff()
    
    avg_vol_change_30 = df['Volume_Change'].rolling(window=30).mean().abs()
    df['Volume_Shock'] = (df['Volume_Change'] > 2 * avg_vol_change_30).astype(int)
    
    # Combine Indicators
    df['Alpha_Factor'] = df['Price_Momentum'] * df['Volume_Shock']
    
    return df['Alpha_Factor']
```
