```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Integrated Price Momentum
    df['integrated_momentum'] = (df['close'] - df['open']) / df['open'] + (df['close'].shift(-1) - df['close']) / df['close']
    df['integrated_momentum_5d'] = df['integrated_momentum'].rolling(window=5).mean()
    
    # Calculate 5-Day Volume Momentum
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['volume_momentum_5d'] = df['volume_change'].rolling(window=5).mean()
    
    # Calculate 5-Day Price Oscillation
    df['price_range'] = df['high'] - df['low']
    df['price_oscillation_5d'] = df['price_range'].rolling(window=5).mean()
    
    # Combine Momentum and Oscillation Factors
    df['alpha_factor'] = (df['integrated_momentum_5d'] * 
                          df['volume_momentum_5d'] * 
                          df['price_oscillation_5d'])
    
    return df['alpha_factor']
```
