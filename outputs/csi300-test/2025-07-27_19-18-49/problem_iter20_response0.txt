import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=5, L=30, N=20):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust Log Return by Open Price
    df['adj_log_return'] = df['log_return'] - (np.log(df['open']) - np.log(df['close'].shift(1)))
    
    # Calculate Momentum Components
    df['close_mom'] = df['close'] / df['close'].shift(n) - 1
    df['open_mom'] = df['open'] / df['open'].shift(n) - 1
    df['volume_mom'] = df['volume'] / df['volume'].shift(n) - 1
    df['high_mom'] = df['high'] / df['high'].shift(n) - 1
    df['low_mom'] = df['low'] / df['low'].shift(n) - 1
    
    # Weight by Recent Trading Activity
    df['avg_vol_m'] = df['volume'].rolling(window=m).mean()
    df['long_term_avg_vol_L'] = df['volume'].rolling(window=L).mean()
    df['vol_weight'] = df['avg_vol_m'] / df['long_term_avg_vol_L']
    
    # Incorporate Price Range Momentum
    df['price_range'] = df['high'] - df['low']
    df['price_range_mom'] = df['price_range'] / df['price_range'].shift(n) - 1
    
    # Integrate Price Momentum
    df['integrated_mom'] = ((df['close'] - df['open']) / df['open'] + 
                            (df['close'] - df['close'].shift(1)) / df['close'].shift(1))
    df['integrated_mom_5'] = df['integrated_mom'].rolling(window=5).sum() / 5
    
    # Calculate 5-Day Price Oscillation
    df['price_oscillation'] = df['high'] - df['low']
    df['price_oscillation_5'] = df['price_oscillation'].rolling(window=5).sum() / 5
    
    # Calculate 5-Day Volume Trend
    df['volume_trend'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    df['volume_trend_smoothed'] = df['volume_trend'].rolling(window=5).mean()
    
    # Detect Volume Shocks
    df['daily_return'] = df['close'].pct_change()
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    avg_vol_change_30 = df['volume_change'].rolling(window=30).mean()
    df['volume_shock'] = (df['volume_change'] > 2 * avg_vol_change_30).astype(int)
    
    # Combine Indicators
    df['combined_factor'] = (df['close_mom'] * df['vol_weight'] +
                             df['open_mom'] * df['vol_weight'] +
                             df['high_mom'] * df['vol_weight'] +
                             df['low_mom'] * df['vol_weight'] +
                             df['price_range_mom'] * df['vol_weight'] +
                             df['integrated_mom_5'] * df['vol_weight'] +
                             df['price_oscillation_5'] * df['vol_weight'] +
                             df['volume_trend_smoothed'] * df['vol_weight'])
    df['combined_factor'] *= df['volume_shock']
    
    # Cumulative Sum of Volume-Weighted Adjusted Returns
    df['cum_sum_factor'] = (df['adj_log_return'] * df['volume']).rolling(window=N).sum()
    
    return df['combined_factor']
