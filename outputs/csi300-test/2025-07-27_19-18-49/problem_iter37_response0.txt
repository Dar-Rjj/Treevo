```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=10, L=60, K=10, N=20):
    # Momentum Components
    close_momentum = df['close'] / df['close'].shift(n) - 1
    open_momentum = df['open'] / df['open'].shift(n) - 1
    volume_momentum = df['volume'] / df['volume'].shift(n) - 1
    
    # Adjust Daily Log Return by Open Price Change
    log_close = np.log(df['close'])
    log_prev_close = np.log(df['close'].shift(1))
    log_open = np.log(df['open'])
    adjusted_log_return = (log_close - log_prev_close - (log_open - log_prev_close))
    
    # Multiply Open-Adjusted Log Return by Volume
    volume_weighted_return = adjusted_log_return * df['volume']
    
    # Weight by Recent Trading Activity
    short_term_avg_volume = df['volume'].rolling(window=m).mean()
    long_term_avg_volume = df['volume'].rolling(window=L).mean()
    volume_weight = short_term_avg_volume / long_term_avg_volume
    
    # Integrate High-Low Range Indicator
    high_low_range = df['high'] - df['low']
    avg_high_low_range = high_low_range.rolling(window=K).mean()
    high_low_indicator = (high_low_range > avg_high_low_range).astype(int)
    
    # Calculate 5-Day Price Oscillation
    price_oscillation = (df['high'] - df['low']).rolling(window=5).sum() / 5
    
    # Detect Volume Shocks
    daily_return = df['close'].pct_change()
    volume_change = df['volume'].diff()
    vol_shock_threshold = 2 * volume_change.rolling(window=30).mean()
    volume_shock = (volume_change > vol_shock_threshold).astype(int)
    
    # Detect Volatility
    rolling_std = daily_return.rolling(window=20).std()
    volatility_threshold = 2 * rolling_std.rolling(window=20).mean()
    high_volatility = (rolling_std > volatility_threshold).astype(int)
    
    # Combine Indicators
    momentum_sum = close_momentum + open_momentum + volume_momentum
    combined_factor = momentum_sum * volume_shock * high_low_indicator * high_volatility
    final_alpha = (volume_weighted_return * combined_factor).cumsum().rolling(window=N).mean()
    
    return final_alpha
```
