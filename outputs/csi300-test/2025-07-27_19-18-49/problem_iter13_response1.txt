```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, sum_n=20, volume_shock_threshold=2, gap_percentage=0.02):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n) - 1
    open_momentum = df['open'] / df['open'].shift(n) - 1
    volume_momentum = df['volume'] / df['volume'].shift(n) - 1
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Open-Adjusted Daily Log Return
    open_adjusted_log_return = daily_log_return - np.log(df['open']) + np.log(df['close'].shift(1))
    
    # Multiply Open-Adjusted Log Return by Volume
    volume_weighted_log_return = open_adjusted_log_return * df['volume']
    
    # Sum Over N Days
    sum_volume_weighted_log_return = volume_weighted_log_return.rolling(window=sum_n).sum()
    
    # Detect Volume Shocks
    volume_change = df['volume'] - df['volume'].shift(1)
    average_30_day_volume_change = volume_change.rolling(window=30).mean()
    volume_shock = (volume_change > volume_shock_threshold * average_30_day_volume_change).astype(int)
    
    # Detect Price Gaps
    gap_size = df['open'] - df['close'].shift(1)
    price_gap = (np.abs(gap_size) > gap_percentage * df['close'].shift(1)).astype(int)
    
    # Combine Indicators
    momentum_sum = (close_momentum + open_momentum + volume_momentum) * (volume_shock + price_gap)
    
    # Final Alpha Factor
    alpha_factor = sum_volume_weighted_log_return * momentum_sum
    
    return alpha_factor
```
