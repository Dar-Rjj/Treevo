```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, L=50):
    # Calculate Daily Log Return
    df['log_close'] = np.log(df['close'])
    df['daily_log_return'] = df['log_close'] - df['log_close'].shift(1)
    
    # Calculate Open-Adjusted Daily Log Return
    df['open_adjusted_log_return'] = df['daily_log_return'] - (np.log(df['open']) - np.log(df['close'].shift(1)))
    
    # Calculate Momentum Components
    df['close_momentum'] = df['close'] / df['close'].shift(n) - 1
    df['open_momentum'] = df['open'] / df['open'].shift(n) - 1
    df['volume_momentum'] = df['volume'] / df['volume'].shift(n) - 1
    
    # Weight by Recent Trading Activity
    df['avg_volume_m'] = df['volume'].rolling(window=m).mean()
    df['avg_volume_L'] = df['volume'].rolling(window=L).mean()
    df['volume_weight'] = df['avg_volume_m'] / df['avg_volume_L']
    
    # Combine Momentum and Volume Weights
    df['weighted_close_momentum'] = df['close_momentum'] * df['volume_weight']
    df['weighted_open_momentum'] = df['open_momentum'] * df['volume_weight']
    df['combined_momentum'] = df['weighted_close_momentum'] + df['weighted_open_momentum'] + df['volume_momentum']
    
    # Cumulative Sum of Open-Adjusted Volume Weighted Returns
    df['volume_weighted_log_return'] = df['open_adjusted_log_return'] * df['volume']
    df['cumulative_volume_weighted_return'] = df['volume_weighted_log_return'].rolling(window=n).sum()
    
    # Introduce High-Low Price Spread Indicator
    df['high_low_spread'] = (df['high'] - df['low']) / df['close']
    
    # Incorporate High-Low Price Spread into Momentum
    df['high_low_weighted_momentum'] = df['high_low_spread'] * df['close_momentum']
    df['final_momentum'] = df['combined_momentum'] + df['high_low_weighted_momentum']
    
    # Final Alpha Factor
    alpha_factor = df['cumulative_volume_weighted_return'] + df['final_momentum']
    
    return alpha_factor
```
