```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=5, L=60, K=10):
    # Calculate Dynamic Momentum Components
    close_momentum = df['close'] / df['close'].shift(n)
    open_momentum = df['open'] / df['open'].shift(n)
    volume_momentum = df['volume'] / df['volume'].shift(n)
    adjusted_momentum = (close_momentum + open_momentum) * volume_momentum
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust Daily Log Return by Open Price Change
    open_adjusted_log_return = daily_log_return - np.log(df['open']) + np.log(df['close'].shift(1))
    
    # Multiply Open-Adjusted Log Return by Volume
    open_adjusted_volume_log_return = open_adjusted_log_return * df['volume']
    
    # Sum Over N Days
    open_adjusted_vol_sum = open_adjusted_volume_log_return.rolling(window=n).sum()
    
    # Weight by Recent Trading Activity
    avg_volume_m = df['volume'].rolling(window=m).mean()
    avg_volume_L = df['volume'].rolling(window=L).mean()
    volume_weight = avg_volume_m / avg_volume_L
    
    # Detect Volume Shocks
    daily_return = df['close'] / df['close'].shift(1)
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_30_day_volume_change = df['volume'].diff().rolling(window=30).mean()
    volume_shock = (volume_change > 3 * avg_30_day_volume_change).astype(int)
    
    # Integrate High-Low Range Indicator
    high_low_range = df['high'] - df['low']
    avg_high_low_range_K = high_low_range.rolling(window=K).mean()
    high_low_indicator = (high_low_range > avg_high_low_range_K).astype(int)
    
    # Incorporate Dynamic Price Range Momentum
    price_range = df['high'] - df['low']
    price_range_momentum = price_range / price_range.shift(n)
    
    # Integrate Dynamic Price Momentum
    dynamic_price_momentum = ((df['close'] - df['open']) / df['open'] + 
                              (df['close'] - df['close'].shift(1)) / df['close'].shift(1))
    dynamic_price_momentum = dynamic_price_momentum.rolling(window=5).sum() / 5
    
    # Calculate 5-Day Dynamic Price Oscillation
    daily_price_range = df['high'] - df['low']
    price_oscillation = daily_price_range.rolling(window=5).sum() / 5
    
    # Calculate 5-Day Dynamic Volume Trend
    volume_trend = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    smoothed_volume_trend = volume_trend.rolling(window=5).mean()
    
    # Combine Indicators
    momentum_sum = (close_momentum + open_momentum + volume_momentum + price_range_momentum)
    momentum_weighted = momentum_sum * volume_weight
    momentum_with_shock = momentum_weighted * (1.5 * volume_shock + (1 - volume_shock))
    final_factor = (momentum_with_shock * high_low_indicator * 
                    dynamic_price_momentum * price_oscillation * smoothed_volume_trend)
    
    return final_factor
```
