```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, vol_shock_threshold_multiplier=2):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n) - 1
    open_momentum = df['open'] / df['open'].shift(n) - 1
    volume_momentum = df['volume'] / df['volume'].shift(n) - 1

    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))

    # Calculate Open-Adjusted Daily Log Return
    open_adjusted_log_return = daily_log_return - np.log(df['open']) + np.log(df['close'].shift(1))

    # Multiply Open-Adjusted Log Return by Volume
    open_adjusted_vol_log_return = open_adjusted_log_return * df['volume']

    # Sum Over N Days
    open_adjusted_vol_log_return_sum = open_adjusted_vol_log_return.rolling(window=n).sum()

    # Detect Volume Shocks
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volume_change = df['volume'] - df['volume'].shift(1)
    
    avg_volume_change = volume_change.rolling(window=30).mean()
    volume_shock = (volume_change > vol_shock_threshold_multiplier * avg_volume_change).astype(int)

    # Combine Indicators
    combined_momentum = (close_momentum + open_momentum + volume_momentum) / 3
    alpha_factor = combined_momentum * open_adjusted_vol_log_return_sum * volume_shock

    return alpha_factor
```
