```python
import pandas as pd

def heuristics_v2(df, n=20, m=5, L=60, vol_shock_threshold=2.0):
    # Calculate Momentum Components
    close_price_momentum = df['close'].pct_change(periods=n)
    open_price_momentum = df['open'].pct_change(periods=n)
    volume_momentum = df['volume'].pct_change(periods=n)

    # Weight by Recent Trading Activity
    avg_volume_m_days = df['volume'].rolling(window=m).mean()
    long_term_avg_volume = df['volume'].rolling(window=L).mean()
    recent_activity_weight = avg_volume_m_days / long_term_avg_volume

    # Detect Volume Shocks
    daily_return = df['close'].pct_change()
    volume_change = df['volume'].diff()
    avg_vol_change_30d = volume_change.rolling(window=30).mean()
    volume_shock_indicator = (volume_change > vol_shock_threshold * avg_vol_change_30d).astype(int)

    # Combine Indicators
    combined_momentum = (
        close_price_momentum * recent_activity_weight +
        open_price_momentum * recent_activity_weight +
        volume_momentum * recent_activity_weight
    )
    alpha_factor = combined_momentum * volume_shock_indicator

    return alpha_factor
```
