```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, L=50, volume_shock_threshold=2.0):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n)
    open_momentum = df['open'] / df['open'].shift(n)
    volume_momentum = df['volume'] / df['volume'].shift(n)

    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Open-Adjusted Daily Log Return
    open_adjusted_log_return = daily_log_return - np.log(df['open']) + np.log(df['close'].shift(1))

    # Multiply Open-Adjusted Log Return by Volume
    weighted_log_return = open_adjusted_log_return * df['volume']

    # Sum Over N Days
    cumulative_sum = weighted_log_return.rolling(window=n).sum()

    # Weight by Recent Trading Activity
    recent_activity_avg_vol = df['volume'].rolling(window=m).mean()
    long_term_avg_vol = df['volume'].rolling(window=L).mean()
    activity_weight = (recent_activity_avg_vol / long_term_avg_vol).fillna(1)

    # Detect Volume Shocks
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_volume_change = volume_change.rolling(window=30).mean()
    volume_shock = (volume_change > volume_shock_threshold * avg_volume_change).astype(int)

    # Introduce High-Low Price Spread Indicator
    high_low_spread = (df['high'] - df['low']) / df['close']
    normalized_high_low_spread = high_low_spread * close_momentum

    # Incorporate High-Low Price Spread into Momentum
    combined_momentum = (close_momentum + open_momentum + volume_momentum) * activity_weight

    # Combine Indicators
    final_alpha_factor = (combined_momentum + normalized_high_low_spread) * volume_shock

    return final_alpha_factor
```
