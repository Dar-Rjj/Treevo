```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, k=30):
    # Calculate Close-Price Momentum
    close_momentum = df['close'] - df['close'].shift(n)
    
    # Calculate Open-Price Momentum
    open_momentum = df['open'] - df['open'].shift(n)
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'] - df['volume'].shift(n)
    
    # Adjust Daily Log Return by Open Price Change
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    open_adjusted_log_return = (log_return - np.log(df['open'])) + np.log(df['close'].shift(1))
    
    # Multiply Open-Adjusted Log Return by Volume
    open_adjusted_vol_log_return = open_adjusted_log_return * df['volume']
    
    # Sum Over N Days
    summed_factor = open_adjusted_vol_log_return.rolling(window=n).sum()
    
    # Detect Volume Spikes
    daily_return = df['close'] / df['close'].shift(1) - 1
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_volume_change = volume_change.rolling(window=30).mean()
    volume_spike = (volume_change > 3 * avg_volume_change).astype(int)
    
    # Integrate High-Low Range Indicator
    high_low_range = df['high'] - df['low']
    avg_high_low_range = high_low_range.rolling(window=k).mean()
    high_low_indicator = (high_low_range > avg_high_low_range).astype(int)
    
    # Combine Indicators
    alpha_factor = (close_momentum + open_momentum + volume_momentum + summed_factor) * volume_spike * high_low_indicator
    
    return alpha_factor
```
