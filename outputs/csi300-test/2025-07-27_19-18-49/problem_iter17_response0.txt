import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=10, L=100, N=20):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n) - 1
    open_momentum = df['open'] / df['open'].shift(n) - 1
    volume_momentum = df['volume'] / df['volume'].shift(n) - 1
    
    # Calculate Daily Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust Log Return by Open Price
    adjusted_log_return = (np.log(df['close']) - np.log(df['open'])) + np.log(df['close'].shift(1))
    
    # Multiply Open-Adjusted Log Return by Volume
    volume_weighted_return = adjusted_log_return * df['volume']
    
    # Weight by Recent Trading Activity
    recent_avg_volume = df['volume'].rolling(window=m).mean()
    long_term_avg_volume = df['volume'].rolling(window=L).mean()
    trading_activity_weight = recent_avg_volume / long_term_avg_volume
    
    # Detect Volume Shocks
    daily_return = df['close'].pct_change()
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_volume_change_30d = volume_change.rolling(window=30).mean()
    volume_shock = (volume_change > 2 * avg_volume_change_30d).astype(int)
    
    # Combine Indicators
    combined_momentum = (close_momentum * trading_activity_weight +
                         open_momentum * trading_activity_weight +
                         volume_momentum * trading_activity_weight) * volume_shock
    
    # Cumulative Sum of Volume-Weighted Adjusted Returns
    cumulative_sum = (volume_weighted_return * combined_momentum).rolling(window=N).sum()
    
    return cumulative_sum
