```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, L=60, N=20):
    # Calculate Daily Log Return
    df['log_close'] = np.log(df['close'])
    df['daily_log_return'] = df['log_close'] - df['log_close'].shift(1)
    
    # Calculate Open-Adjusted Daily Log Return
    df['open_adjusted_return'] = (df['daily_log_return'] 
                                  - np.log(df['open']) 
                                  + np.log(df['close'].shift(1)))
    
    # Calculate Momentum Components
    df['close_momentum'] = np.log(df['close']) - np.log(df['close'].shift(n))
    df['open_momentum'] = np.log(df['open']) - np.log(df['open'].shift(n))
    df['volume_momentum'] = df['volume'] / df['volume'].shift(n) - 1
    
    # Weight by Recent Trading Activity
    df['avg_volume_m'] = df['volume'].rolling(window=m).mean()
    df['avg_volume_L'] = df['volume'].rolling(window=L).mean()
    df['recent_activity_weight'] = df['avg_volume_m'] / df['avg_volume_L']
    
    # Combine Momentum and Volume Weights
    df['weighted_close_momentum'] = df['close_momentum'] * df['recent_activity_weight']
    df['weighted_open_momentum'] = df['open_momentum'] * df['recent_activity_weight']
    df['weighted_volume_momentum'] = df['volume_momentum'] * df['recent_activity_weight']
    
    # Cumulative Sum of Open-Adjusted Volume Weighted Returns
    df['open_adjusted_volume_weighted_return'] = df['open_adjusted_return'] * df['volume']
    df['cumulative_adjusted_return'] = df['open_adjusted_volume_weighted_return'].rolling(window=N).sum()
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['weighted_close_momentum'] 
                          + df['weighted_open_momentum'] 
                          + df['weighted_volume_momentum'] 
                          + df['cumulative_adjusted_return'])
    
    return df['alpha_factor']
```
