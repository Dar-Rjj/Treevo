```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Momentum Components
    close_momentum = df['close'].pct_change(n)
    open_momentum = df['open'].pct_change(n)
    volume_momentum = df['volume'].pct_change(n)
    high_low_range_momentum = (df['high'] - df['low']).pct_change(n)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Open-Adjusted Daily Log Return
    open_adjusted_log_return = daily_log_return - np.log(df['open']) + np.log(df['close'].shift(1))
    
    # Multiply Open-Adjusted Log Return by Volume
    open_adjusted_volume_log_return = open_adjusted_log_return * df['volume']
    
    # Sum Over N Days
    sum_n_days = open_adjusted_volume_log_return.rolling(window=n).sum()
    
    # Detect Volume Shocks
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_30_day_vol_change = volume_change.rolling(window=30).mean()
    vol_shock_threshold = 2 * avg_30_day_vol_change
    vol_shock_indicator = (volume_change > vol_shock_threshold).astype(int)
    
    # Detect High-Low Range Shocks
    high_low_range = df['high'] - df['low']
    high_low_range_change = high_low_range - high_low_range.shift(1)
    avg_30_day_high_low_range_change = high_low_range_change.rolling(window=30).mean()
    high_low_range_shock_threshold = 2 * avg_30_day_high_low_range_change
    high_low_range_shock_indicator = (high_low_range_change > high_low_range_shock_threshold).astype(int)
    
    # Combine Indicators
    combined_momentum = 0.25 * (close_momentum + open_momentum + volume_momentum + high_low_range_momentum)
    combined_momentum_with_shocks = combined_momentum * vol_shock_indicator * high_low_range_shock_indicator
    
    # Output the final Alpha Factor
    alpha_factor = sum_n_days * combined_momentum_with_shocks
    return alpha_factor
```
