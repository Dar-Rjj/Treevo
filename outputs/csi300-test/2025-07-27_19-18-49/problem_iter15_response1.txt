```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=5, L=60):
    # Calculate Daily Log Return
    df['log_close'] = np.log(df['close'])
    df['log_return'] = df['log_close'] - df['log_close'].shift(1)
    
    # Adjust Log Return by Open Price
    df['log_open'] = np.log(df['open'])
    df['adjusted_log_return'] = df['log_return'] - df['log_open'] + df['log_close'].shift(1)
    
    # Calculate Momentum Components
    df['close_momentum'] = df['close'] / df['close'].shift(n) - 1
    df['open_momentum'] = df['open'] / df['open'].shift(n) - 1
    df['high_momentum'] = df['high'] / df['high'].shift(n) - 1
    df['low_momentum'] = df['low'] / df['low'].shift(n) - 1
    df['volume_momentum'] = df['volume'] / df['volume'].shift(n) - 1
    
    # Price Range Momentum
    df['price_range'] = df['high'] - df['low']
    df['price_range_momentum'] = df['price_range'] / df['price_range'].shift(n) - 1
    
    # Weight by Recent Trading Activity
    df['avg_vol_m'] = df['volume'].rolling(window=m).mean()
    df['long_term_avg_vol'] = df['volume'].rolling(window=L).mean()
    df['weight'] = df['avg_vol_m'] / df['long_term_avg_vol']
    
    # Combine Momentum and Volume Weights
    df['weighted_close_momentum'] = df['close_momentum'] * df['weight']
    df['weighted_open_momentum'] = df['open_momentum'] * df['weight']
    df['weighted_high_momentum'] = df['high_momentum'] * df['weight']
    df['weighted_low_momentum'] = df['low_momentum'] * df['weight']
    df['weighted_price_range_momentum'] = df['price_range_momentum'] * df['weight']
    
    df['momentum_factor'] = (df['weighted_close_momentum'] + 
                             df['weighted_open_momentum'] + 
                             df['weighted_high_momentum'] + 
                             df['weighted_low_momentum'] + 
                             df['weighted_price_range_momentum'])
    
    # Cumulative Sum of Volume-Weighted Adjusted Returns
    df['vol_weighted_adj_return'] = df['adjusted_log_return'] * df['volume']
    df['cumulative_vol_weighted_adj_return'] = df['vol_weighted_adj_return'].rolling(window=n).sum()
    
    return df['momentum_factor']
```
