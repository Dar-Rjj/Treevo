```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['Close_1d_ago'] = df['close'].shift(1)
    df['Daily_Price_Momentum'] = (df['close'] - df['Close_1d_ago']) / df['Close_1d_ago']
    df['5Day_Price_Momentum'] = df['Daily_Price_Momentum'].rolling(window=5).sum() / 5
    
    # Calculate 5-Day Volume Momentum
    df['Volume_1d_ago'] = df['volume'].shift(1)
    df['Daily_Volume_Change'] = (df['volume'] - df['Volume_1d_ago']) / df['Volume_1d_ago']
    df['5Day_Volume_Momentum'] = df['Daily_Volume_Change'].rolling(window=5).sum() / 5
    
    # Calculate 5-Day Price Oscillation
    df['High_1d_ago'] = df['high'].shift(1)
    df['Low_1d_ago'] = df['low'].shift(1)
    df['Daily_Price_Range'] = df['high'] - df['low']
    df['5Day_Price_Oscillation'] = df['Daily_Price_Range'].rolling(window=5).sum() / 5
    
    # Combine Price, Volume, and Price Oscillation
    df['Alpha_Factor'] = (df['5Day_Price_Momentum'] * 
                          df['5Day_Volume_Momentum'] * 
                          df['5Day_Price_Oscillation'])
    
    return df['Alpha_Factor']
```
