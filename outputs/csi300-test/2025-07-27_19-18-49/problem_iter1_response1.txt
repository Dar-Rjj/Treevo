```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Compute Logarithm of Close Price
    log_close = np.log(df['close'])
    
    # Subtract Logarithm of Previous Day's Close Price to get Daily Log Return
    daily_log_return = log_close - log_close.shift(1)
    
    # Calculate the Open-Adjusted Daily Log Return
    log_open = np.log(df['open'])
    open_adjusted_log_return = daily_log_return - (log_open - log_close.shift(1))
    
    # Multiply Adjusted Log Return by Volume
    volume_weighted_return = open_adjusted_log_return * df['volume']
    
    # Sum Over N Days: Cumulative Sum of Open-Adjusted Volume Weighted Returns
    cumulative_volume_weighted_return = volume_weighted_return.rolling(window=n).sum()
    
    return cumulative_volume_weighted_return
```
