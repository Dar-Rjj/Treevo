```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, N=20, m=10, L=60, K=10):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n) - 1
    open_momentum = df['open'] / df['open'].shift(n) - 1
    volume_momentum = df['volume'] / df['volume'].shift(n) - 1
    
    # Adjust Daily Log Return by Open Price Change
    log_close = np.log(df['close'])
    log_prev_close = log_close.shift(1)
    log_open = np.log(df['open'])
    adjusted_log_return = (log_close - log_prev_close - log_open + log_prev_close)
    
    # Multiply Open-Adjusted Log Return by Volume
    open_adjusted_volume = adjusted_log_return * df['volume']
    
    # Sum Over N Days
    cumulative_sum = open_adjusted_volume.rolling(window=N).sum()
    
    # Weight by Recent Trading Activity
    avg_volume_recent = df['volume'].rolling(window=m).mean()
    avg_volume_long_term = df['volume'].rolling(window=L).mean()
    activity_weight = avg_volume_recent / avg_volume_long_term
    
    # Detect Volume Shocks and Spikes
    daily_return = df['close'] / df['close'].shift(1) - 1
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_vol_change_30 = volume_change.rolling(window=30).mean()
    volume_shock_spike = np.where(volume_change > 3 * avg_vol_change_30, 1.5, 
                                  np.where(volume_change > avg_vol_change_30, 1, 0))
    
    # Integrate High-Low Range Indicator
    high_low_range = df['high'] - df['low']
    avg_high_low_range = high_low_range.rolling(window=K).mean()
    high_low_indicator = np.where(high_low_range > avg_high_low_range, 1, 0)
    
    # Combine Indicators
    momentum_sum = close_momentum + open_momentum + volume_momentum
    combined_factor = (momentum_sum * volume_shock_spike * high_low_indicator * activity_weight).fillna(0)
    
    return combined_factor
```
