```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, L=50, K=30):
    # Calculate Close-Price Momentum
    close_momentum = df['close'] - df['close'].shift(n)
    
    # Calculate Open-Price Momentum
    open_momentum = df['open'] - df['open'].shift(n)
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'] - df['volume'].shift(n)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust Daily Log Return by Open Price Change
    adjusted_log_return = (np.log(df['close']) - np.log(df['open'])) + np.log(df['close'].shift(1))
    
    # Multiply Open-Adjusted Log Return by Volume
    log_return_volume = adjusted_log_return * df['volume']
    
    # Sum Over N Days
    sum_n_days = log_return_volume.rolling(window=n).sum()
    
    # Weight by Recent Trading Activity
    avg_volume_m = df['volume'].rolling(window=m).mean()
    long_term_avg_volume_L = df['volume'].rolling(window=L).mean()
    trading_activity_weight = avg_volume_m / long_term_avg_volume_L
    
    # Detect Volume Spikes
    daily_return = df['close'] / df['close'].shift(1) - 1
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_vol_change_30 = volume_change.rolling(window=30).mean()
    volume_spike = (volume_change > 3 * avg_vol_change_30).astype(int)
    
    # Integrate High-Low Range Indicator
    high_low_range = df['high'] - df['low']
    avg_high_low_range_K = high_low_range.rolling(window=K).mean()
    high_low_indicator = (high_low_range > avg_high_low_range_K).astype(int)
    
    # Combine Indicators
    momentum_sum = (close_momentum + open_momentum + volume_momentum) * trading_activity_weight
    alpha_factor = momentum_sum * volume_spike * high_low_indicator
    
    return alpha_factor
```
