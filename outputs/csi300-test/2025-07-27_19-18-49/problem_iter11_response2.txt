```python
import pandas as pd
import numpy as np

def heuristics_v2(df, m=10, L=250, N=20):
    # Calculate Daily Log Return
    df['log_close'] = np.log(df['close'])
    df['log_return'] = df['log_close'] - df['log_close'].shift(1)
    
    # Adjust Log Return by Open Price
    df['log_open'] = np.log(df['open'])
    df['adjusted_log_return'] = df['log_return'] - (df['log_open'] - df['log_close'].shift(1))
    
    # Calculate 5-Day Integrated Price Momentum
    df['integrated_momentum'] = ((df['close'] - df['open']) / df['open'] + 
                                 (df['close'] - df['close'].shift(1)) / df['close'].shift(1))
    df['5_day_integrated_momentum'] = df['integrated_momentum'].rolling(window=5).sum() / 5
    
    # Calculate 5-Day Volume Momentum
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['5_day_volume_momentum'] = df['volume_change'].rolling(window=5).sum() / 5
    
    # Calculate 5-Day Price Oscillation
    df['price_range'] = df['high'] - df['low']
    df['5_day_price_oscillation'] = df['price_range'].rolling(window=5).sum() / 5
    
    # Weight by Recent Trading Activity
    recent_avg_volume = df['volume'].rolling(window=m).mean()
    long_term_avg_volume = df['volume'].rolling(window=L).mean()
    df['volume_weight'] = recent_avg_volume / long_term_avg_volume
    
    # Combine Integrated Price, Volume, and Price Oscillation
    df['combined_factor'] = (df['5_day_integrated_momentum'] * df['5_day_volume_momentum'] * 
                             df['5_day_price_oscillation'])
    
    # Volume-Weighted Adjusted Returns
    df['weighted_adjusted_return'] = df['adjusted_log_return'] * df['volume']
    df['cumulative_weighted_adjusted_return'] = df['weighted_adjusted_return'].rolling(window=N).sum()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['cumulative_weighted_adjusted_return'] * df['combined_factor']
    
    return df['alpha_factor']
```
