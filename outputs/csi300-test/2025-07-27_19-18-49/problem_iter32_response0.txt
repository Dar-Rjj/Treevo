```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=10, L=50, K=10, N=20):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n) - 1
    open_momentum = df['open'] / df['open'].shift(n) - 1
    volume_momentum = df['volume'] / df['volume'].shift(n) - 1
    high_momentum = df['high'] / df['high'].shift(n) - 1
    low_momentum = df['low'] / df['low'].shift(n) - 1
    
    # Adjust Daily Log Return by Open Price Change
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    open_adj_log_return = log_return - (np.log(df['open']) - np.log(df['close'].shift(1)))
    
    # Multiply Open-Adjusted Log Return by Volume
    vol_weighted_adj_log_return = open_adj_log_return * df['volume']
    
    # Weight by Recent Trading Activity
    avg_volume_m_days = df['volume'].rolling(window=m).mean()
    long_term_avg_volume = df['volume'].rolling(window=L).mean()
    trading_activity_weight = avg_volume_m_days / long_term_avg_volume
    
    # Integrate High-Low Range Indicator
    high_low_range = df['high'] - df['low']
    avg_high_low_range_k_days = high_low_range.rolling(window=K).mean()
    high_low_range_indicator = (high_low_range > avg_high_low_range_k_days).astype(int)
    
    # Incorporate Price Range Momentum
    price_range = df['high'] - df['low']
    price_range_momentum = price_range / price_range.shift(n) - 1
    
    # Calculate 5-Day Price Oscillation
    price_oscillation = (df['high'] - df['low']).rolling(window=5).sum() / 5
    
    # Detect Volume Shocks
    daily_return = df['close'] / df['close'].shift(1) - 1
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_30_day_vol_change = volume_change.rolling(window=30).mean()
    vol_shock = (volume_change > 2 * avg_30_day_vol_change).astype(int)
    
    # Detect Volatility
    daily_returns = df['close'] / df['close'].shift(1) - 1
    rolling_std_dev = daily_returns.rolling(window=20).std()
    volatility_threshold = 2 * rolling_std_dev.mean()
    high_volatility = (rolling_std_dev > volatility_threshold).astype(int)
    
    # Combine Indicators
    momentum_sum = (close_momentum + open_momentum + volume_momentum + high_momentum + low_momentum + price_range_momentum)
    momentum_sum *= vol_shock  # Volume Shock presence
    momentum_sum *= high_low_range_indicator  # High-Low Range Indicator
    momentum_sum *= high_volatility  # Adjust for Volatility
    
    # Final Alpha Factor
    final_alpha_factor = (vol_weighted_adj_log_return * trading_activity_weight * momentum_sum).cumsum().rolling(window=N).sum()
    
    return final_alpha_factor
```
