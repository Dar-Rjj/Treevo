```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=50, m=10, L=100):
    """
    Generate a novel alpha factor based on the provided thought tree.
    
    Parameters:
    df (pd.DataFrame): Dataframe with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    n (int): Number of days for momentum and sum over N days
    m (int): Number of days for recent average volume
    L (int): Number of days for long-term average volume
    
    Returns:
    pd.Series: Alpha factor values
    """

    # Calculate Close-Price Momentum
    close_momentum = df['close'] / df['close'].shift(n) - 1

    # Calculate Open-Price Momentum
    open_momentum = df['open'] / df['open'].shift(n) - 1

    # Calculate Volume Momentum
    volume_momentum = df['volume'] / df['volume'].shift(n) - 1

    # Calculate Daily Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))

    # Adjust Daily Log Return by Open Price Change
    open_adjusted_log_return = (log_return - np.log(df['open']) + np.log(df['close'].shift(1)))

    # Multiply Open-Adjusted Log Return by Volume
    open_adjusted_volume_return = open_adjusted_log_return * df['volume']

    # Sum Over N Days
    sum_over_n_days = open_adjusted_volume_return.rolling(window=n).sum()

    # Weight by Recent Trading Activity
    avg_volume_m = df['volume'].rolling(window=m).mean()
    avg_volume_L = df['volume'].rolling(window=L).mean()
    recent_activity_weight = avg_volume_m / avg_volume_L

    # Detect Volume Spikes
    daily_return = df['close'] / df['close'].shift(1) - 1
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_30d_volume_change = abs(volume_change).rolling(window=30).mean()
    volume_spike = (volume_change > 3 * avg_30d_volume_change).astype(int)

    # Combine Indicators
    combined_momentum = (close_momentum + open_momentum) * recent_activity_weight
    final_alpha_factor = (combined_momentum + sum_over_n_days) * volume_spike

    return final_alpha_factor
```
