```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['Price_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    price_momentum = df['Price_Momentum'].rolling(window=5).sum() / 5
    
    # Calculate Volume Momentum
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_momentum = df['Volume_Change'].rolling(window=5).sum() / 5
    
    # Introduce Price Oscillation
    df['Price_Range'] = df['high'] - df['low']
    price_oscillation = df['Price_Range'].rolling(window=5).sum() / 5
    
    # Incorporate Amount Factor
    df['Amount_Change'] = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    amount_factor = df['Amount_Change'].rolling(window=5).sum() / 5
    
    # Combine Price, Volume, Price Oscillation, and Amount
    alpha_factor = price_momentum * volume_momentum * price_oscillation * amount_factor
    
    return alpha_factor
```
