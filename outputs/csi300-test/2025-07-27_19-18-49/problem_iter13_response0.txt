import pandas as pd

def heuristics_v2(df):
    # Calculate Integrated Price Momentum
    df['integrated_price_momentum'] = (df['close'] - df['open']) / df['open'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['integrated_price_momentum_5d'] = df['integrated_price_momentum'].rolling(window=5).mean()

    # Calculate 5-Day Price Momentum
    df['price_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['price_momentum_5d'] = df['price_momentum'].rolling(window=5).mean()

    # Calculate 5-Day Volume Momentum
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['volume_momentum_5d'] = df['volume_momentum'].rolling(window=5).mean()

    # Calculate 5-Day Price Oscillation
    df['price_oscillation'] = df['high'] - df['low']
    df['price_oscillation_5d'] = df['price_oscillation'].rolling(window=5).mean()

    # Combine All Factors
    df['combined_factor'] = (df['integrated_price_momentum_5d'] * 
                             df['price_momentum_5d'] * 
                             df['volume_momentum_5d'] * 
                             df['price_oscillation_5d'])

    return df['combined_factor']
