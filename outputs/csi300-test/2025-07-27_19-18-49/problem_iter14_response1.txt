```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, L=30, N=20):
    # Momentum Components
    close_momentum = df['close'] / df['close'].shift(n)
    open_momentum = df['open'] / df['open'].shift(n)
    volume_momentum = df['volume'] / df['volume'].shift(n)
    
    # Daily Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust Log Return by Open Price
    adjusted_log_return = (log_return - (np.log(df['open']) - np.log(df['close'].shift(1))))
    
    # Multiply Open-Adjusted Log Return by Volume
    volume_weighted_adjusted_return = adjusted_log_return * df['volume']
    
    # Weight by Recent Trading Activity
    recent_avg_volume = df['volume'].rolling(window=m).mean()
    long_term_avg_volume = df['volume'].rolling(window=L).mean()
    weight = recent_avg_volume / long_term_avg_volume
    
    # Detect Volume Shocks
    daily_return = (df['close'] / df['close'].shift(1)) - 1
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_30_day_vol_change = volume_change.rolling(window=30).mean()
    volume_shock = (volume_change > 2 * avg_30_day_vol_change).astype(int)
    
    # Combine Indicators
    combined_momentum = (close_momentum + open_momentum + volume_momentum) * weight
    combined_momentum_with_volume_shock = combined_momentum * volume_shock
    
    # 5-Day Price Oscillation
    price_oscillation = (df['high'] - df['low']).rolling(window=5).sum() / 5
    
    # Integrate Price Oscillation
    integrated_factor = combined_momentum_with_volume_shock * price_oscillation
    
    # Cumulative Sum of Volume-Weighted Adjusted Returns
    cumulative_sum = (volume_weighted_adjusted_return * weight).rolling(window=N).sum()
    
    # Final Alpha Factor
    alpha_factor = integrated_factor * cumulative_sum
    
    return alpha_factor
```
