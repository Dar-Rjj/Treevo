import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, N=20, volume_shock_threshold_multiplier=3.0):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n) - 1
    open_momentum = df['open'] / df['open'].shift(n) - 1
    volume_momentum = df['volume'] / df['volume'].shift(n) - 1
    
    # Calculate Daily Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Open-Adjusted Daily Log Return
    open_adjusted_log_return = log_return - np.log(df['open']) + np.log(df['close'].shift(1))
    
    # Multiply Open-Adjusted Log Return by Volume
    open_adjusted_volume_log_return = open_adjusted_log_return * df['volume']
    
    # Sum Over N Days
    sum_over_n_days = open_adjusted_volume_log_return.rolling(window=N).sum()
    
    # Detect Volume Shocks
    daily_return = df['close'].pct_change()
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_30_day_volume_change = volume_change.rolling(window=30).mean().abs()
    volume_shock = (volume_change > volume_shock_threshold_multiplier * avg_30_day_volume_change).astype(int)
    
    # Combine Indicators
    momentum_sum = close_momentum + open_momentum + volume_momentum
    volume_shock_factor = volume_shock * 1.5
    final_alpha_factor = momentum_sum * volume_shock_factor
    
    return final_alpha_factor
