```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['Close_1_day_ago'] = df['close'].shift(1)
    df['Daily_Momentum'] = (df['close'] - df['Close_1_day_ago']) / df['Close_1_day_ago']
    df['Momentum_Aggregated'] = df['Daily_Momentum'].rolling(window=5).mean()
    
    # Calculate Volume Momentum
    df['Volume_1_day_ago'] = df['volume'].shift(1)
    df['Daily_Volume_Change'] = (df['volume'] - df['Volume_1_day_ago']) / df['Volume_1_day_ago']
    df['Volume_Momentum_Aggregated'] = df['Daily_Volume_Change'].rolling(window=5).mean()
    
    # Introduce Price Oscillation
    df['Price_Range'] = df['high'] - df['low']
    df['Price_Oscillation_Aggregated'] = df['Price_Range'].rolling(window=5).mean()
    
    # Combine Price, Volume, and Price Oscillation
    df['Alpha_Factor'] = df['Momentum_Aggregated'] * df['Volume_Momentum_Aggregated'] * df['Price_Oscillation_Aggregated']
    
    return df['Alpha_Factor']
```
