```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=20, L=100):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n)
    open_momentum = df['open'] / df['open'].shift(n)
    volume_momentum = df['volume'] / df['volume'].shift(n)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Open-Adjusted Daily Log Return
    open_adjusted_log_return = daily_log_return - (np.log(df['open']) - np.log(df['close'].shift(1)))
    
    # Multiply Open-Adjusted Log Return by Volume
    open_adjusted_volume_log_return = open_adjusted_log_return * df['volume']
    
    # Sum Over N Days
    cum_sum = open_adjusted_volume_log_return.rolling(window=n).sum()
    
    # Weight by Recent Trading Activity
    avg_volume = df['volume'].rolling(window=m).mean()
    long_term_avg_volume = df['volume'].rolling(window=L).mean()
    weight = avg_volume / long_term_avg_volume
    
    # Detect Volume Spikes
    daily_return = df['close'].pct_change()
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_volume_change = volume_change.rolling(window=30).mean()
    volume_spike = (volume_change > 3 * avg_volume_change).astype(int)
    
    # Combine Indicators
    close_mom_weighted = close_momentum * weight
    open_mom_weighted = open_momentum * weight
    total_momentum = (close_mom_weighted + open_mom_weighted + volume_momentum) / 3
    alpha_factor = total_momentum * cum_sum * (1 + volume_spike)
    
    return alpha_factor
```
