```python
import pandas as pd

def heuristics_v2(df):
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()

    # Calculate daily integrated momentum
    df['integrated_momentum'] = (df['close'] - df['open']) / df['open'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Aggregate over the last 5 days for integrated momentum
    df['integrated_momentum_5d'] = df['integrated_momentum'].rolling(window=5).sum() / 5

    # Calculate daily volume change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Aggregate over the last 5 days for volume momentum
    df['volume_momentum_5d'] = df['volume_change'].rolling(window=5).sum() / 5

    # Calculate high-low band
    df['high_low_band'] = (df['high'] - df['low']) / df['low']

    # Aggregate over the last 5 days for high-low band momentum
    df['high_low_band_5d'] = df['high_low_band'].rolling(window=5).sum() / 5

    # Combine momentum factors
    df['combined_momentum'] = df['integrated_momentum_5d'] * df['volume_momentum_5d'] * df['high_low_band_5d']

    # Return the combined momentum factor
    return df['combined_momentum'].dropna()
```
