import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, L=50, K=10):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n)
    open_momentum = df['open'] / df['open'].shift(n)
    volume_momentum = df['volume'] / df['volume'].shift(n)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust Daily Log Return by Open Price Change
    adjusted_log_return = (np.log(df['close']) - np.log(df['open'])) + np.log(df['close'].shift(1))
    
    # Multiply Open-Adjusted Log Return by Volume
    vol_adjusted_return = adjusted_log_return * df['volume']
    
    # Sum Over N Days
    cum_sum_vol_adjusted_return = vol_adjusted_return.rolling(window=n).sum()
    
    # Weight by Recent Trading Activity
    avg_volume_m = df['volume'].rolling(window=m).mean()
    avg_volume_L = df['volume'].rolling(window=L).mean()
    volume_weight = avg_volume_m / avg_volume_L
    
    # Detect Volume Shocks
    daily_return = df['close'] / df['close'].shift(1)
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_30_day_vol_change = volume_change.rolling(window=30).mean()
    volume_shock = (volume_change > 3 * avg_30_day_vol_change).astype(int)
    
    # Integrate High-Low Range Indicator
    high_low_range = df['high'] - df['low']
    avg_high_low_range_K = high_low_range.rolling(window=K).mean()
    high_low_indicator = (high_low_range > avg_high_low_range_K).astype(int)
    
    # Incorporate Price Range Momentum
    price_range = df['high'] - df['low']
    price_range_momentum = price_range / price_range.shift(n)
    
    # Integrate Price Momentum
    integrated_price_momentum = ((df['close'] - df['open']) / df['open'] + 
                                 (df['close'] - df['close'].shift(1)) / df['close'].shift(1)).rolling(window=5).sum() / 5
    
    # Calculate 5-Day Price Oscillation
    daily_price_range = df['high'] - df['low']
    price_oscillation = daily_price_range.rolling(window=5).sum() / 5
    
    # Calculate 5-Day Volume Trend
    volume_trend = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    smoothed_volume_trend = volume_trend.rolling(window=5).mean()
    
    # Combine Indicators
    combined_momentum = (close_momentum + open_momentum + volume_momentum + price_range_momentum) * volume_weight
    combined_momentum *= (1.5 * volume_shock + (1 - volume_shock))
    combined_momentum *= high_low_indicator
    combined_momentum += integrated_price_momentum
    combined_momentum *= price_oscillation
    combined_momentum *= smoothed_volume_trend
    
    return combined_momentum.dropna()
