```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['price_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5_day_price_momentum'] = df['price_momentum'].rolling(window=5).mean()
    
    # Calculate 5-Day Volume Momentum
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['5_day_volume_momentum'] = df['volume_change'].rolling(window=5).mean()
    
    # Calculate 5-Day Price Range
    df['price_range'] = df['high'] - df['low']
    df['5_day_price_range'] = df['price_range'].rolling(window=5).mean()
    
    # Combine Price, Volume, and Price Range
    df['alpha_factor'] = df['5_day_price_momentum'] * df['5_day_volume_momentum'] * df['5_day_price_range']
    
    return df['alpha_factor'].dropna()
```
