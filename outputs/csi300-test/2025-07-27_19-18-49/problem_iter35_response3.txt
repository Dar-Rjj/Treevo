```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, k=10, m=5, l=60):
    # Calculate Close-Price Momentum
    close_momentum = df['close'].pct_change(n)
    
    # Calculate Open-Price Momentum
    open_momentum = df['open'].pct_change(n)
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'].pct_change(n)
    
    # Adjust Daily Log Return by Open Price Change
    log_return_open_adj = (np.log(df['close']) - np.log(df['close'].shift(1)) - 
                           np.log(df['open']) + np.log(df['close'].shift(1)))
    
    # Multiply Open-Adjusted Log Return by Volume
    open_adj_vol_weighted = log_return_open_adj * df['volume']
    
    # Sum Over N Days
    sum_n_days = open_adj_vol_weighted.rolling(window=n).sum()
    
    # Detect Volume Shocks and Spikes
    daily_return = df['close'].pct_change()
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_vol_change_30 = volume_change.rolling(window=30).mean()
    vol_shock_spike = (volume_change > 3 * avg_vol_change_30).astype(float) * 1.5
    vol_shock_spike[(volume_change > avg_vol_change_30) & (volume_change <= 3 * avg_vol_change_30)] = 1
    
    # Integrate High-Low Range Indicator
    high_low_range = df['high'] - df['low']
    avg_high_low_range = high_low_range.rolling(window=k).mean()
    high_low_indicator = (high_low_range > avg_high_low_range).astype(int)
    
    # Weight by Recent Trading Activity
    avg_volume_m = df['volume'].rolling(window=m).mean()
    avg_volume_l = df['volume'].rolling(window=l).mean()
    weight = avg_volume_m / avg_volume_l
    
    # Incorporate Dynamic Price Range Momentum
    price_range = df['high'] - df['low']
    price_range_momentum = price_range.pct_change(n)
    
    # Integrate Dynamic Price Momentum
    dynamic_price_momentum = ((df['close'] - df['open']) / df['open'] + 
                             (df['close'] - df['close'].shift(1)) / df['close'].shift(1))
    dynamic_price_momentum = dynamic_price_momentum.rolling(window=5).mean()
    
    # Calculate 5-Day Dynamic Price Oscillation
    dynamic_price_oscillation = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate 5-Day Dynamic Volume Trend
    dynamic_volume_trend = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    dynamic_volume_trend = dynamic_volume_trend.rolling(window=5).mean()
    
    # Combine Indicators
    momentum_sum = (close_momentum + open_momentum + volume_momentum + price_range_momentum)
    factor = (momentum_sum * vol_shock_spike * high_low_indicator * weight * 
              dynamic_price_momentum * dynamic_price_oscillation * dynamic_volume_trend)
    
    return factor
```
