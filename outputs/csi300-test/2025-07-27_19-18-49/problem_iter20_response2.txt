```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=10, L=60, N=20):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(n)
    open_momentum = df['open'] / df['open'].shift(n)
    volume_momentum = df['volume'] / df['volume'].shift(n)
    
    # Calculate Daily Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust Log Return by Open Price
    adjusted_log_return = (np.log(df['close']) - np.log(df['open'])) + (np.log(df['close'].shift(1)) - np.log(df['open'].shift(1)))
    
    # Multiply Open-Adjusted Log Return by Volume
    volume_weighted_adjusted_log_return = adjusted_log_return * df['volume']
    
    # Weight by Recent Trading Activity
    recent_average_volume = df['volume'].rolling(window=m).mean()
    long_term_average_volume = df['volume'].rolling(window=L).mean()
    trading_activity_weight = recent_average_volume / long_term_average_volume
    
    # Detect Volume Shocks
    daily_return = df['close'].pct_change()
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_shock_threshold = 2 * volume_change.rolling(window=30).mean()
    volume_shock_indicator = (volume_change > volume_shock_threshold).astype(int)
    
    # Integrate Price, Volume, and Close Momentum with Open-Price Integration
    integrated_momentum = ((df['close'] - df['open']) / df['open']) + ((df['close'] - df['close'].shift(1)) / df['close'].shift(1))
    aggregated_integrated_momentum = integrated_momentum.rolling(window=n).sum() / n
    
    # Combine Indicators
    combined_momentum = (close_momentum * trading_activity_weight) + (open_momentum * trading_activity_weight) + volume_momentum
    combined_momentum_with_shock = combined_momentum * volume_shock_indicator
    combined_momentum_with_all_factors = combined_momentum_with_shock * aggregated_integrated_momentum * daily_return * (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1) * (df['high'] - df['low'])
    
    # Cumulative Sum of Volume-Weighted Adjusted Returns
    cumulative_sum = (volume_weighted_adjusted_log_return * combined_momentum_with_all_factors).rolling(window=N).sum()
    
    return cumulative_sum
```
