import pandas as pd

def heuristics_v2(data):
    # Calculate 5-Day Integrated Price Momentum
    data['integrated_momentum_daily'] = ((data['close'] - data['open']) / data['open']) + \
                                         ((data['close'] - data['close'].shift(1)) / data['close'].shift(1))
    data['integrated_momentum_5d'] = data['integrated_momentum_daily'].rolling(window=5).sum() / 5
    
    # Calculate 5-Day Volume Momentum
    data['volume_change_daily'] = (data['volume'] - data['volume'].shift(1)) / data['volume'].shift(1)
    data['volume_momentum_5d'] = data['volume_change_daily'].rolling(window=5).sum() / 5
    
    # Calculate 5-Day Price Oscillation
    data['price_range_daily'] = data['high'] - data['low']
    data['price_oscillation_5d'] = data['price_range_daily'].rolling(window=5).sum() / 5
    
    # Combine Integrated Price, Volume, and Price Oscillation
    data['alpha_factor'] = data['integrated_momentum_5d'] * data['volume_momentum_5d'] * data['price_oscillation_5d']
    
    return data['alpha_factor']
