```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, k=30):
    # Calculate Close-Price Momentum
    close_momentum = df['close'].pct_change(n)
    
    # Calculate Open-Price Momentum
    open_momentum = df['open'].pct_change(n)
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'].pct_change(n)
    
    # Adjust Daily Log Return by Open Price Change
    log_return_adj = np.log(df['close']) - np.log(df['close'].shift(1)) - (np.log(df['open']) - np.log(df['close'].shift(1)))
    
    # Multiply Open-Adjusted Log Return by Volume
    vol_adj_log_return = log_return_adj * df['volume']
    
    # Sum Over N Days
    sum_n_days = vol_adj_log_return.rolling(window=n).sum()
    
    # Detect Volume Spikes
    daily_return = df['close'] / df['close'].shift(1) - 1
    vol_change = df['volume'] - df['volume'].shift(1)
    avg_vol_change = vol_change.rolling(window=k).mean()
    spike_threshold = 3 * avg_vol_change
    volume_spike = (vol_change > spike_threshold).astype(int)
    
    # Integrate High-Low Range Indicator
    high_low_range = df['high'] - df['low']
    avg_high_low_range = high_low_range.rolling(window=k).mean()
    high_low_indicator = (high_low_range > avg_high_low_range).astype(int)
    
    # Combine Indicators
    combined_momentum = close_momentum + open_momentum + volume_momentum
    combined_momentum *= volume_spike
    combined_momentum *= high_low_indicator
    
    final_alpha_factor = combined_momentum * sum_n_days
    return final_alpha_factor
```
