```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, L=30):
    # Calculate Momentum Components
    close_momentum = df['close'].pct_change(n)
    open_momentum = df['open'].pct_change(n)
    volume_momentum = df['volume'].pct_change(n)

    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']).diff()

    # Calculate Open-Adjusted Daily Log Return
    open_adjusted_log_return = daily_log_return - np.log(df['open']/df['close'].shift(1))

    # Multiply Open-Adjusted Log Return by Volume
    adjusted_log_return_volume = open_adjusted_log_return * df['volume']

    # Sum Over N Days
    sum_n_days = adjusted_log_return_volume.rolling(window=n).sum()

    # Weight by Recent Trading Activity
    avg_volume_recent = df['volume'].rolling(window=m).mean()
    long_term_avg_volume = df['volume'].rolling(window=L).mean()
    recent_activity_weight = avg_volume_recent / long_term_avg_volume

    # Detect Volume Shocks
    daily_return = df['close'].pct_change()
    volume_change = df['volume'].diff()
    threshold = 2 * volume_change.rolling(window=30).std()
    volume_shock = (volume_change > threshold).astype(int)

    # Combine Indicators
    close_momentum_weighted = close_momentum * recent_activity_weight
    open_momentum_weighted = open_momentum * recent_activity_weight
    momentum_sum = close_momentum + open_momentum + volume_momentum
    alpha_factor = (momentum_sum * sum_n_days) * volume_shock

    return alpha_factor
```
