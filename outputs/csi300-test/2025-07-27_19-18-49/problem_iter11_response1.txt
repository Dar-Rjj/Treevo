```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Integrated Price Momentum
    df['integrated_momentum'] = (df['close'] - df['open']) / df['open'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    integrated_price_momentum = df['integrated_momentum'].rolling(window=5).mean()
    
    # Calculate 5-Day Price Momentum
    daily_price_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    price_momentum = daily_price_momentum.rolling(window=5).mean()
    
    # Calculate 5-Day Volume Momentum
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_momentum = daily_volume_change.rolling(window=5).mean()
    
    # Calculate 5-Day Price Oscillation
    daily_price_range = df['high'] - df['low']
    price_oscillation = daily_price_range.rolling(window=5).mean()
    
    # Combine All Momentum and Oscillation Factors
    alpha_factor = integrated_price_momentum * price_momentum * volume_momentum * price_oscillation
    
    return alpha_factor
```
