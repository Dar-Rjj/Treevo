```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Integrated Price Momentum
    df['Integrated_Momentum'] = (df['close'] - df['open']) / df['open'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Sum of daily integrated momentums over last 5 days / 5
    df['Integrated_Momentum_5d'] = df['Integrated_Momentum'].rolling(window=5).mean()
    
    # 5-Day Price Momentum
    df['Price_Momentum_5d'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # 5-Day Volume Momentum
    df['Volume_Momentum_5d'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # 5-Day Price Oscillation
    df['Price_Oscillation_5d'] = df['high'] - df['low']
    
    # Combine factors
    df['Alpha_Factor'] = (df['Integrated_Momentum_5d'] 
                          * df['Price_Momentum_5d'] 
                          * df['Volume_Momentum_5d'] 
                          * df['Price_Oscillation_5d'])
    
    return df['Alpha_Factor'].dropna()
```
