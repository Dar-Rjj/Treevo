```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the daily range
    df['range'] = df['high'] - df['low']
    
    # Calculate the average true range over a 14-day period
    df['atr_14'] = df['range'].rolling(window=14).mean()
    
    # Calculate the relative strength based on the amount and volume
    df['rs'] = df['amount'] / df['volume']
    
    # Calculate the rate of change in the relative strength over a 14-day period
    df['roc_rs_14'] = df['rs'].pct_change(periods=14)
    
    # Calculate the weighted moving average of close prices using ATR as weights
    df['wma_close_atr'] = (df['close'] * df['atr_14']).rolling(window=14).sum() / df['atr_14'].rolling(window=14).sum()
    
    # Calculate the novel alpha factor
    df['alpha_factor'] = (df['wma_close_atr'] - df['close']) / df['atr_14'] + df['roc_rs_14']
    
    return df['alpha_factor']
```
