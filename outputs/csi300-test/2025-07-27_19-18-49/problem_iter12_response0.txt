```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    price_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    avg_price_momentum = price_momentum.rolling(window=5).mean()

    # Calculate 5-Day Volume Momentum
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    avg_volume_momentum = volume_change.rolling(window=5).mean()

    # Calculate 5-Day Price Range
    price_range = df['high'] - df['low']
    avg_price_range = price_range.rolling(window=5).mean()

    # Calculate 5-Day Relative Strength Index (RSI)
    daily_price_change = df['close'] - df['close'].shift(1)
    
    gain = daily_price_change.where(daily_price_change > 0, 0)
    loss = -daily_price_change.where(daily_price_change < 0, 0)
    
    avg_gain = gain.rolling(window=5).mean()
    avg_loss = loss.rolling(window=5).mean()
    
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Combine Price, Volume, Price Range, and RSI
    alpha_factor = (avg_price_momentum * avg_volume_momentum * avg_price_range * rsi)

    return alpha_factor
```
