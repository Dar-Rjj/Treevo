```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Integrated Price Momentum
    df['integrated_momentum'] = (df['close'] - df['open']) / df['open'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    integrated_price_momentum = df['integrated_momentum'].rolling(window=5).sum() / 5
    
    # Calculate 5-Day Price Momentum
    df['price_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    price_momentum_5d = df['price_momentum'].rolling(window=5).sum() / 5
    
    # Calculate 5-Day Volume Momentum
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_momentum_5d = df['volume_change'].rolling(window=5).sum() / 5
    
    # Calculate 5-Day Price Oscillation
    df['price_range'] = df['high'] - df['low']
    price_oscillation_5d = df['price_range'].rolling(window=5).sum() / 5
    
    # Combine All Momentum and Oscillation Factors
    alpha_factor = (integrated_price_momentum * 
                    price_momentum_5d * 
                    volume_momentum_5d * 
                    price_oscillation_5d)
    
    return alpha_factor
```
