import pandas as pd
import numpy as np

def heuristics_v2(df, m=5, L=20):
    # Calculate Momentum Components
    close_momentum = df['close'] / df['close'].shift(5)
    open_momentum = df['open'] / df['open'].shift(5)
    volume_momentum = df['volume'] / df['volume'].shift(5)
    high_momentum = df['high'] / df['high'].shift(5)
    low_momentum = df['low'] / df['low'].shift(5)

    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))

    # Adjust Log Return by Open Price
    open_adjusted_log_return = (np.log(df['close']) - np.log(df['open'])) + np.log(df['close'].shift(1))

    # Multiply Open-Adjusted Log Return by Volume
    volume_weighted_adjusted_return = open_adjusted_log_return * df['volume']

    # Weight by Recent Trading Activity
    avg_volume_m = df['volume'].rolling(window=m).mean()
    avg_volume_L = df['volume'].rolling(window=L).mean()
    recent_activity_weight = avg_volume_m / avg_volume_L

    # Incorporate Price Range Momentum
    price_range_momentum = (df['high'] - df['low']) / (df['high'].shift(5) - df['low'].shift(5))

    # Calculate 5-Day Price Oscillation
    price_oscillation = (df['high'] - df['low']).rolling(window=5).sum() / 5

    # Calculate 5-Day Volume Trend
    volume_trend = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)

    # Detect Volume Shocks
    daily_return = df['close'] / df['close'].shift(1) - 1
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_shock = (volume_change > 2 * volume_change.rolling(window=5).std()).astype(int)

    # Integrate Price, Volume, and Close Momentum with Open-Price Integration
    integrated_momentum = (df['close'] - df['open']) / df['open'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Indicators
    combined_indicators = (
        close_momentum * recent_activity_weight +
        open_momentum * recent_activity_weight +
        high_momentum * recent_activity_weight +
        low_momentum * recent_activity_weight +
        price_range_momentum * recent_activity_weight +
        price_oscillation * recent_activity_weight +
        volume_trend * recent_activity_weight +
        volume_shock
    )

    # Final Alpha Factor
    final_alpha_factor = (volume_weighted_adjusted_return * recent_activity_weight).cumsum()

    return final_alpha_factor
