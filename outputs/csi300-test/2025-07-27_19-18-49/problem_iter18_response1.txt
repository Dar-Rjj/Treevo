```python
import pandas as pd

def heuristics_v2(df):
    df = df.copy()
    
    # Calculate Integrated Price Momentum
    integrated_price_momentum = (df['close'] - df['open']) / df['open'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    integrated_price_momentum = integrated_price_momentum.rolling(window=5).sum().fillna(0)

    # Calculate 5-Day Price Momentum
    price_momentum_5d = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    price_momentum_5d = price_momentum_5d.rolling(window=5).sum().fillna(0)

    # Calculate 5-Day Volume Momentum
    volume_momentum_5d = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_momentum_5d = volume_momentum_5d.rolling(window=5).sum().fillna(0)

    # Calculate 5-Day Price Oscillation
    price_oscillation_5d = (df['high'] - df['low']) / df['close']
    price_oscillation_5d = price_oscillation_5d.rolling(window=5).sum().fillna(0)

    # Calculate 5-Day Volume Trend
    volume_trend_5d = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    volume_trend_5d = volume_trend_5d.rolling(window=5).mean().fillna(0)

    # Combine All Factors
    combined_factor = (integrated_price_momentum * price_momentum_5d * 
                       volume_momentum_5d * price_oscillation_5d * 
                       volume_trend_5d)

    return combined_factor
```
