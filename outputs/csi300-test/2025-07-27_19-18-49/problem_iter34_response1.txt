```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, k=10):
    # Calculate Close-Open Price Momentum
    df['close_open_momentum'] = df['close'] - df['open'].shift(n)
    
    # Adjust Daily Log Return by Open Price
    df['log_return_close'] = np.log(df['close'])
    df['log_return_open'] = np.log(df['open'])
    df['prev_log_return_close'] = df['log_return_close'].shift(1)
    df['adjusted_log_return'] = df['log_return_close'] - df['log_return_open'] + df['prev_log_return_close']
    
    # Multiply Open-Adjusted Log Return by Volume
    df['volume_adjusted_return'] = df['volume'] * df['adjusted_log_return']
    
    # Sum Over N Days
    df['sum_volume_adjusted_return'] = df['volume_adjusted_return'].rolling(window=n).sum()
    
    # Detect Volume Shocks and Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    avg_volume_change = df['volume_change'].rolling(window=30).mean()
    std_volume_change = df['volume_change'].rolling(window=30).std()
    threshold = 3 * std_volume_change
    df['volume_shock'] = (df['volume_change'] > threshold).astype(float)
    df['volume_spike'] = (df['volume_change'] > 0.5 * threshold).astype(float)
    df['volume_factor'] = 1.5 * df['volume_shock'] + df['volume_spike']
    
    # Integrate High-Low Range Indicator
    df['high_low_range'] = df['high'] - df['low']
    avg_high_low_range = df['high_low_range'].rolling(window=k).mean()
    df['high_low_indicator'] = (df['high_low_range'] > avg_high_low_range).astype(int)
    
    # Combine Indicators
    df['alpha_factor'] = df['sum_volume_adjusted_return'] * df['volume_factor'] * df['high_low_indicator']
    
    return df['alpha_factor'].dropna()
```
