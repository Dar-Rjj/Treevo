import pandas as pd

def heuristics_v2(df):
    # Calculate the 1-day and 5-day lags for close, open, high, low, and volume
    df['close_1d'] = df['close'].shift(1)
    df['open_1d'] = df['open'].shift(1)
    df['volume_1d'] = df['volume'].shift(1)
    df['high_1d'] = df['high'].shift(1)
    df['low_1d'] = df['low'].shift(1)
    df['volume_5d'] = df['volume'].shift(5)
    
    # Integrated Price Momentum
    df['integrated_price_momentum'] = ((df['close'] - df['open']) / df['open']) + \
                                      ((df['close'] - df['close_1d']) / df['close_1d'])
    df['integrated_price_momentum_5d'] = df['integrated_price_momentum'].rolling(window=5).mean()
    
    # 5-Day Price Momentum
    df['price_momentum'] = (df['close'] - df['close_1d']) / df['close_1d']
    df['price_momentum_5d'] = df['price_momentum'].rolling(window=5).mean()
    
    # 5-Day Volume Momentum
    df['volume_momentum'] = (df['volume'] - df['volume_1d']) / df['volume_1d']
    df['volume_momentum_5d'] = df['volume_momentum'].rolling(window=5).mean()

    # 5-Day Price Oscillation
    df['price_oscillation'] = df['high'] - df['low']
    df['price_oscillation_5d'] = df['price_oscillation'].rolling(window=5).mean()

    # 5-Day Volume Trend
    df['volume_trend'] = (df['volume'] - df['volume_5d']) / df['volume_5d']
    df['volume_trend_5d'] = df['volume_trend'].rolling(window=5).mean()
    
    # Combine all factors
    factor = (df['integrated_price_momentum_5d'] * 
              df['price_momentum_5d'] * 
              df['volume_momentum_5d'] * 
              df['price_oscillation_5d'] * 
              df['volume_trend_5d'])
    
    return factor.dropna()
