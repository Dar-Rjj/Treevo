```python
import pandas as pd
import numpy as np

def heuristics_v2(df, m=20, L=252, N=20):
    # Calculate Daily Log Return
    df['log_close'] = np.log(df['close'])
    df['log_return'] = df['log_close'] - df['log_close'].shift(1)
    
    # Adjust Log Return by Open Price
    df['log_open'] = np.log(df['open'])
    df['adj_log_return'] = df['log_return'] - (df['log_open'] - df['log_close'].shift(1))
    
    # Calculate 5-Day Integrated Price Momentum
    df['integrated_momentum'] = ((df['close'] - df['open']) / df['open']) + \
                                ((df['close'] - df['close'].shift(1)) / df['close'].shift(1))
    df['5_day_integrated_momentum'] = df['integrated_momentum'].rolling(window=5).mean()
    
    # Calculate 5-Day Volume Momentum
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['5_day_volume_momentum'] = df['volume_change'].rolling(window=5).mean()
    
    # Calculate 5-Day Price Oscillation
    df['price_range'] = df['high'] - df['low']
    df['5_day_price_oscillation'] = df['price_range'].rolling(window=5).mean()
    
    # Weight by Recent Trading Activity
    df['avg_vol_m_days'] = df['volume'].rolling(window=m).mean()
    df['long_term_avg_vol'] = df['volume'].rolling(window=L).mean()
    df['vol_weight'] = df['avg_vol_m_days'] / df['long_term_avg_vol']
    
    # Combine Integrated Price, Volume, and Price Oscillation
    df['combined_momentum_oscillation'] = (df['5_day_integrated_momentum'] * 
                                           df['5_day_volume_momentum'] * 
                                           df['5_day_price_oscillation'])
    
    # Volume-Weighted Adjusted Returns
    df['vol_weighted_adj_return'] = df['adj_log_return'] * df['volume']
    df['cumulative_vol_weighted_adj_return'] = df['vol_weighted_adj_return'].rolling(window=N).sum()
    
    # Calculate 5-Day Open-Adjusted Momentum
    df['open_adjusted_momentum'] = (df['close'] - df['open']) / df['open']
    df['5_day_open_adjusted_momentum'] = df['open_adjusted_momentum'].rolling(window=5).mean()
    
    # Combine All Components
    df['final_component'] = (df['combined_momentum_oscillation'] * 
                             df['5_day_open_adjusted_momentum'])
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['cumulative_vol_weighted_adj_return'] * 
                          df['final_component'] * 
                          df['vol_weight'])
    
    return df['alpha_factor']
```
