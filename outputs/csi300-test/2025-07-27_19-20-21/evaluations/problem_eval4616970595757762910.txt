def heuristics_v2(df):
    # Calculate the 14-period Rate of Change (ROC)
    roc_14 = df['close'].pct_change(periods=14) * 100
    
    # Calculate the 14-period Average True Range (ATR)
    tr = pd.DataFrame({
        'high_low': df['high'] - df['low'],
        'high_close': (df['high'] - df['close'].shift()).abs(),
        'low_close': (df['low'] - df['close'].shift()).abs()
    }).max(axis=1)
    atr_14 = tr.rolling(window=14).mean()
    
    # Combine ROC and ATR into a single heuristic matrix
    heuristics_matrix = 0.5 * roc_14 + 0.5 * (1 / (1 + atr_14))
    
    return heuristics_matrix
