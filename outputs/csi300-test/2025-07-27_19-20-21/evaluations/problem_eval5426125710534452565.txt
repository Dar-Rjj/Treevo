def heuristics_v2(df):
    # Calculate the 14-period RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 14-period ATR
    tr = pd.DataFrame(index=df.index, columns=['tr'])
    tr['tr'] = df[['high', 'low']].apply(lambda x: x[0] - x[1], axis=1)
    tr['tr'] = tr['tr'].combine(df['close'].diff().abs(), max, fill_value=0)
    atr = tr['tr'].rolling(window=14).mean()
    
    # Combine RSI and ATR into a single heuristic matrix
    heuristics_matrix = 0.7 * rsi + 0.3 * (1 - atr / atr.max())
    
    return heuristics_matrix
