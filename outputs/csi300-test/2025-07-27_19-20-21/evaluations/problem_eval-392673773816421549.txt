import pandas as pd

def heuristics_v2(df):
    # Calculate short term and long term simple moving averages
    short_mavg = df['close'].rolling(window=10).mean()
    long_mavg = df['close'].rolling(window=50).mean()
    
    # Calculate standard deviation as a measure of volatility
    std_dev = df['close'].rolling(window=20).std()
    
    # Alpha factor formula: (Short MA - Long MA) / Standard Deviation
    heuristics_matrix = (short_mavg - long_mavg) / std_dev
    
    return heuristics_matrix
