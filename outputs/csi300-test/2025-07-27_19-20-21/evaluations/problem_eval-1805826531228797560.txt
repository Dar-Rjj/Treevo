def heuristics_v3(df):
    # Calculate the 14-day average of (high + low)/2
    avg_price_14 = (df['high'] + df['low']) / 2
    avg_price_14 = avg_price_14.rolling(window=14).mean()
    
    # Calculate the ratio of close to the 14-day average price
    close_to_avg_ratio = df['close'] / avg_price_14
    
    # Calculate the 10-day and 30-day EMAs
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # Generate the EMA crossover signal
    ema_crossover = ema_10 - ema_30
    
    # Calculate the standard deviation of daily returns over 25 days
    daily_returns = df['close'].pct_change()
    std_dev_25 = daily_returns.rolling(window=25).std()
    
    # Combine the factors into a single heuristics matrix with weights
    heuristics_matrix = 0.4 * close_to_avg_ratio + 0.3 * ema_crossover - 0.3 * std_dev_25
    
    return heuristics_matrix
