def heuristics_v2(df):
    # Calculate the 14-period Rate of Change (ROC)
    roc = df['close'].pct_change(periods=14) * 100
    
    # Calculate the 14-period Average True Range (ATR)
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Combine ROC and ATR into a single heuristic matrix
    heuristics_matrix = 0.5 * roc + 0.5 * (1 / atr)
    
    return heuristics_matrix
