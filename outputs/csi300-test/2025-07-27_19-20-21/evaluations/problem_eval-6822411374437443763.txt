def heuristics_v2(df):
    # Calculate the 10-period EMA and 30-period EMA
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the Percentage Price Oscillator (PPO)
    ppo_line = ((ema_10 - ema_30) / ema_30) * 100
    
    # Calculate Bollinger Bands
    sma_30 = df['close'].rolling(window=30).mean()
    std_30 = df['close'].rolling(window=30).std()
    upper_band = sma_30 + (2.5 * std_30)
    lower_band = sma_30 - (2.5 * std_30)
    
    # Calculate the %B indicator (position within Bollinger Bands)
    b_percentage = (df['close'] - lower_band) / (upper_band - lower_band)
    
    # Combine PPO and %B into a single heuristic matrix
    heuristics_matrix = 0.6 * ppo_line + 0.4 * (1 - b_percentage)
    
    return heuristics_matrix
