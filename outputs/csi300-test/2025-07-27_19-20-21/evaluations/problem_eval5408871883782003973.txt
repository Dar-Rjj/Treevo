def heuristics_v2(df):
    # Calculate the 14-period RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 14-period ATR
    tr = pd.DataFrame({
        'high-low': df['high'] - df['low'],
        'high-close_prev': (df['high'] - df['close'].shift()).abs(),
        'low-close_prev': (df['low'] - df['close'].shift()).abs()
    }).max(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Combine RSI and ATR into a single heuristic matrix
    heuristics_matrix = 0.5 * rsi + 0.5 * (1 - atr / atr.max())
    
    return heuristics_matrix
