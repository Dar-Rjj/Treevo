def heuristics_v2(df):
    # Calculate the RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    RS = gain / loss
    rsi_line = 100 - (100 / (1 + RS))
    
    # Calculate the Chaikin Money Flow (CMF)
    money_flow_multiplier = ((df['close']-df['low']) - (df['high']-df['close'])) / (df['high']-df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf_line = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combine RSI and CMF into a single heuristic matrix
    heuristics_matrix = 0.5 * rsi_line + 0.5 * cmf_line
    
    return heuristics_matrix
