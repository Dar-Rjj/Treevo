def heuristics_v2(df):
    # Calculate the 20-day simple moving average of the close price
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the ratio of the current closing price to the 20-day SMA
    close_to_sma = df['close'] / sma_20
    
    # Calculate the 20-day exponential moving average of the volume
    vol_ema_20 = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate the difference between the current volume and the 20-day EMA of the volume
    vol_diff = df['volume'] - vol_ema_20
    
    # Combine the factors into a single heuristics matrix
    heuristics_matrix = close_to_sma * (1 + vol_diff)
    
    return heuristics_matrix
