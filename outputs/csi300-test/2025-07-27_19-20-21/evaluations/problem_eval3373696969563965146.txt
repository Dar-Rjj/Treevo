def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate MFI
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0).rolling(window=14).sum()
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0).rolling(window=14).sum()
    mfi = 100 * (positive_money_flow / (positive_money_flow + negative_money_flow))
    
    # Calculate ROC
    roc = ((df['close'] - df['close'].shift(14)) / df['close'].shift(14)) * 100
    
    # Add Volatility Factor
    log_returns = np.log(df['close']).diff()
    volatility = log_returns.rolling(window=14).std()
    
    # Weighted Moving Average
    wma_weights = np.arange(1,15)[::-1]
    wma = df['close'].rolling(window=14).apply(lambda x: (x * wma_weights).sum() / wma_weights.sum(), raw=True)
    wma_factor = (df['close'] - wma) / wma
    
    # Combine RSI, MFI, ROC, Volatility, and WMA into a single heuristic matrix
    heuristics_matrix = 0.45 * rsi + 0.3 * mfi + 0.1 * roc + 0.1 * volatility + 0.05 * wma_factor
    
    return heuristics_matrix
