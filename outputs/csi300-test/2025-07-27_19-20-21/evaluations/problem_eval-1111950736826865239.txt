def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=10).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=10).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate MACD
    ema_9 = df['close'].ewm(span=9, adjust=False).mean()
    ema_21 = df['close'].ewm(span=21, adjust=False).mean()
    macd_line = ema_9 - ema_21
    signal_line = macd_line.ewm(span=5, adjust=False).mean()
    macd_hist = macd_line - signal_line
    
    # Calculate Chaikin Money Flow
    mfm = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    mf_volume = mfm * df['volume']
    cmf = mf_volume.rolling(window=15).sum() / df['volume'].rolling(window=15).sum()
    
    # Combine RSI, MACD, and CMF into a single heuristic matrix
    heuristics_matrix = 0.5 * rsi + 0.2 * macd_hist + 0.3 * cmf
    
    return heuristics_matrix
