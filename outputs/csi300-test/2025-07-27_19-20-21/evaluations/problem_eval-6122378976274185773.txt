def heuristics_v2(df):
    ema_15 = df['close'].ewm(span=15, adjust=False).mean()
    ema_45 = df['close'].ewm(span=45, adjust=False).mean()
    ppo_line = ((ema_15 - ema_45) / ema_45) * 100
    sma_40 = df['close'].rolling(window=40).mean()
    std_40 = df['close'].rolling(window=40).std()
    upper_band = sma_40 + (2.0 * std_40)
    lower_band = sma_40 - (2.0 * std_40)
    b_percentage = (df['close'] - lower_band) / (upper_band - lower_band)
    heuristics_matrix = 0.7 * ppo_line + 0.3 * (1 - b_percentage)
    return heuristics_matrix
