def heuristics_v2(df):
    # Calculate the 14-period Momentum
    momentum = df['close'].diff(14)
    
    # Calculate the 14-period Average True Range (ATR)
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Combine Momentum and ATR into a single heuristic matrix
    heuristics_matrix = (momentum / atr)
    
    return heuristics_matrix
