import pandas as pd

def heuristics_v2(df):
    # Calculate Chaikin Money Flow
    mfv = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    mfv *= df['volume']
    cmf_20 = mfv.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate Percentage Price Oscillator
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    ppo = (ema_12 - ema_26) / ema_26 * 100
    
    # Calculate 10-day standard deviation
    std_10 = df['close'].rolling(window=10).std()
    
    # Combine the factors into a single heuristics matrix
    heuristics_matrix = cmf_20 + ppo - std_10
    
    return heuristics_matrix
