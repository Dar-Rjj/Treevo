def heuristics_v2(df):
    # Calculate Williams %R
    highest_high = df['high'].rolling(window=14).max()
    lowest_low = df['low'].rolling(window=14).min()
    williams_r = -100 * (highest_high - df['close']) / (highest_high - lowest_low)
    
    # Calculate On-Balance Volume (OBV)
    obv = (df['volume'] * ((df['close'] > df['close'].shift(1)).astype(int) - (df['close'] < df['close'].shift(1)).astype(int))).cumsum()
    
    # Calculate Aroon Oscillator
    aroon_up = 100 * df['high'].rolling(window=25).apply(lambda x: (x.argmax() + 1) / 25)
    aroon_down = 100 * df['low'].rolling(window=25).apply(lambda x: (x.argmin() + 1) / 25)
    aroon_oscillator = aroon_up - aroon_down
    
    # Calculate z-scores for each indicator
    williams_r_z = (williams_r - williams_r.mean()) / williams_r.std()
    obv_z = (obv - obv.mean()) / obv.std()
    aroon_oscillator_z = (aroon_oscillator - aroon_oscillator.mean()) / aroon_oscillator.std()
    
    # Combine z-scores into a single heuristic matrix
    heuristics_matrix = 0.4 * williams_r_z + 0.3 * obv_z + 0.3 * aroon_oscillator_z
    
    return heuristics_matrix
