import pandas as pd

def heuristics_v2(df):
    short_window = 12
    long_window = 26
    signal_smoothing = 9
    exp1 = df['close'].ewm(span=short_window, adjust=False).mean()
    exp2 = df['close'].ewm(span=long_window, adjust=False).mean()
    macd = exp1 - exp2
    signal = macd.ewm(span=signal_smoothing, adjust=False).mean()
    heuristics_matrix = macd - signal
    return heuristics_matrix
