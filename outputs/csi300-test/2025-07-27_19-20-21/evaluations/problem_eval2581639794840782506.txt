def heuristics_v2(df):
    # Calculate the 14-period Rate of Change (ROC)
    roc = df['close'].pct_change(periods=14) * 100
    
    # Calculate the 14-period Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine ROC and RSI into a single heuristic matrix
    heuristics_matrix = 0.5 * roc + 0.5 * rsi
    
    return heuristics_matrix
