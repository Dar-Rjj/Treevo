import pandas as pd

def heuristics_v2(df):
    roc = df['close'].pct_change(periods=10)
    tr = pd.DataFrame(index=df.index, columns=['tr'])
    tr['tr'] = df[['high', 'low']].diff().abs().max(axis=1)
    atr = tr['tr'].rolling(window=14).mean()
    dm_plus = (df['high'].diff(1)).apply(lambda x: max(x, 0))
    dm_minus = (-df['low'].diff(1)).apply(lambda x: max(x, 0))
    di_plus = (dm_plus / atr).rolling(window=14).mean() * 100
    di_minus = (dm_minus / atr).rolling(window=14).mean() * 100
    adx = 100 * (abs(di_plus - di_minus) / (di_plus + di_minus)).rolling(window=14).mean()
    heuristics_matrix = roc * adx
    return heuristics_matrix
