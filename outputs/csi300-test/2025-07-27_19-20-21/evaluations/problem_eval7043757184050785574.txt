def heuristics_v2(df):
    # Calculate the 14-period RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 14-period Stochastic Oscillator
    lowest_14 = df['low'].rolling(window=14).min()
    highest_14 = df['high'].rolling(window=14).max()
    k_line = ((df['close'] - lowest_14) / (highest_14 - lowest_14)) * 100
    
    # Combine RSI and %K into a single heuristic matrix
    heuristics_matrix = 0.5 * rsi + 0.5 * (100 - k_line)
    
    return heuristics_matrix
