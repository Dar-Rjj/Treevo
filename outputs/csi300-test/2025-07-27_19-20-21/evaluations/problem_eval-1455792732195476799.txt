def heuristics_v2(df):
    # Calculate the 14-period Stochastic Oscillator
    low_14 = df['low'].rolling(window=14).min()
    high_14 = df['high'].rolling(window=14).max()
    k_percent = 100 * ((df['close'] - low_14) / (high_14 - low_14))
    
    # Calculate the 12-day Rate of Change (ROC)
    roc_12 = ((df['close'] / df['close'].shift(12)) - 1) * 100
    
    # Combine the Stochastic Oscillator (%K) and ROC into a single heuristic matrix
    heuristics_matrix = 0.5 * k_percent + 0.5 * roc_12
    
    return heuristics_matrix
