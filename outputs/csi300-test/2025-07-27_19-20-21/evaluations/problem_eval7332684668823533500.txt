def heuristics_v2(df):
    # Calculate the 20-period EMA and 50-period EMA
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the Percentage Price Oscillator (PPO) with the new EMAs
    ppo_line = ((ema_20 - ema_50) / ema_50) * 100
    
    # Calculate Bollinger Bands with a 30-day window and 3 standard deviations
    sma_30 = df['close'].rolling(window=30).mean()
    std_30 = df['close'].rolling(window=30).std()
    upper_band = sma_30 + (3 * std_30)
    lower_band = sma_30 - (3 * std_30)
    
    # Calculate the %B indicator (position within Bollinger Bands)
    b_percentage = (df['close'] - lower_band) / (upper_band - lower_band)
    
    # Combine PPO and %B into a single heuristic matrix
    heuristics_matrix = 0.5 * ppo_line + 0.5 * (1 - b_percentage)
    
    return heuristics_matrix
