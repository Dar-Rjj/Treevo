def heuristics_v2(df):
    # Calculate Williams %R
    highest_high = df['high'].rolling(window=10).max()
    lowest_low = df['low'].rolling(window=10).min()
    williams_r = -100 * (highest_high - df['close']) / (highest_high - lowest_low)
    
    # Calculate On-Balance Volume (OBV)
    obv = (df['volume'] * ((df['close'] > df['close'].shift(1)).astype(int) - (df['close'] < df['close'].shift(1)).astype(int))).cumsum()
    
    # Calculate Aroon Oscillator
    aroon_up = 100 * df['high'].rolling(window=20).apply(lambda x: (x.argmax() + 1) / 20)
    aroon_down = 100 * df['low'].rolling(window=20).apply(lambda x: (x.argmin() + 1) / 20)
    aroon_oscillator = aroon_up - aroon_down
    
    # Combine Williams %R, OBV, and Aroon Oscillator into a single heuristic matrix
    heuristics_matrix = 0.5 * williams_r + 0.2 * obv + 0.3 * aroon_oscillator
    
    return heuristics_matrix
