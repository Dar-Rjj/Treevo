def heuristics_v2(df):
    # Calculate Bollinger Bands
    sma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    upper_band = sma_20 + 2 * std_20
    lower_band = sma_20 - 2 * std_20
    bollinger_width = (upper_band - lower_band) / sma_20
    bollinger_width_change = bollinger_width.pct_change().fillna(0)
    
    # Calculate Modified MACD
    ema_15 = df['close'].ewm(span=15, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    macd_mod = ema_15 - ema_30
    signal_mod = macd_mod.rolling(window=9).mean()
    
    # Combine the factors into a single heuristics matrix
    heuristics_matrix = bollinger_width_change * macd_mod - signal_mod
    
    return heuristics_matrix
