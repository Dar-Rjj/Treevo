def heuristics_v2(df):
    # Calculate Stochastic Oscillator
    highest_high = df['high'].rolling(window=14).max()
    lowest_low = df['low'].rolling(window=14).min()
    k_line = 100 * (df['close'] - lowest_low) / (highest_high - lowest_low)
    
    # Calculate Chaikin Money Flow (CMF)
    mf_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    mf_volume = mf_multiplier * df['volume']
    cmf = mf_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate Average Directional Index (ADX)
    plus_dm = df['high'].diff()
    minus_dm = df['low'].diff().abs()
    plus_dm[plus_dm < 0] = 0
    minus_dm[minus_dm < 0] = 0
    tr = pd.DataFrame({
        'high-low': df['high'] - df['low'],
        'high-close': abs(df['high'] - df['close'].shift(1)),
        'low-close': abs(df['low'] - df['close'].shift(1))
    }).max(axis=1)
    atr = tr.rolling(window=14).mean()
    plus_di = 100 * (plus_dm.rolling(window=14).mean() / atr)
    minus_di = 100 * (minus_dm.rolling(window=14).mean() / atr)
    dx = 100 * (abs(plus_di - minus_di) / (plus_di + minus_di))
    adx = dx.rolling(window=14).mean()
    
    # Combine Stochastic Oscillator, CMF, and ADX into a single heuristic matrix
    heuristics_matrix = 0.4 * k_line + 0.3 * cmf + 0.3 * adx
    
    return heuristics_matrix
