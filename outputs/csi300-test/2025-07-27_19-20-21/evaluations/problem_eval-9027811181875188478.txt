def heuristics_v2(df):
    # Calculate the 12-period EMA and 26-period EMA
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the Percentage Price Oscillator (PPO)
    ppo_line = ((ema_12 - ema_26) / ema_26) * 100
    
    # Calculate Bollinger Bands
    sma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    upper_band = sma_20 + (2 * std_20)
    lower_band = sma_20 - (2 * std_20)
    
    # Calculate the %B indicator (position within Bollinger Bands)
    b_percentage = (df['close'] - lower_band) / (upper_band - lower_band)

    # Calculate the Rate of Change (ROC) over 14 periods
    roc = df['close'].pct_change(14) * 100

    # Calculate the Average True Range (ATR) over 14 periods
    tr = df[['high', 'low']].diff().abs().max(axis=1)
    atr = tr.rolling(window=14).mean()

    # Combine PPO, %B, ROC, and ATR into a single heuristic matrix
    heuristics_matrix = 0.3 * ppo_line + 0.3 * (1 - b_percentage) + 0.2 * roc + 0.2 * (1 / atr)
    
    return heuristics_matrix
