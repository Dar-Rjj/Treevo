def heuristics_v2(df):
    # Calculate the 12-period RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=12).mean()
    avg_loss = loss.rolling(window=12).mean()
    rs = avg_gain / avg_loss
    rsi_12 = 100 - (100 / (1 + rs))

    # Calculate the Chaikin Money Flow over 20 periods
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf_20 = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Combine RSI and CMF into a single heuristic matrix
    heuristics_matrix = 0.5 * rsi_12 + 0.5 * cmf_20

    return heuristics_matrix
