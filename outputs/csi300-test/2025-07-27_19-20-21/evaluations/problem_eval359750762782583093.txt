import pandas as pd

def heuristics_v3(df):
    # Calculate the Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the Moving Average Convergence Divergence (MACD) line
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    
    # Calculate the KDJ indicator
    low_min = df['low'].rolling(window=9).min()
    high_max = df['high'].rolling(window=9).max()
    rsv = (df['close'] - low_min) / (high_max - low_min) * 100
    k_value = rsv.rolling(window=3).mean()
    d_value = k_value.rolling(window=3).mean()
    j_value = 3 * k_value - 2 * d_value
    
    # Combine the factors into a single heuristics matrix
    heuristics_matrix = 0.4 * rsi + 0.4 * macd_line + 0.2 * j_value
    
    return heuristics_matrix
