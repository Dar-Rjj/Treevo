def heuristics_v2(df):
    # Calculate the 50-period ROC
    roc_50 = df['close'].pct_change(periods=50)
    
    # Calculate the RSI over a 14-period window
    delta = df['close'].diff(1)
    up = delta.clip(lower=0)
    down = -delta.clip(upper=0)
    ema_up = up.ewm(com=13, adjust=False).mean()
    ema_down = down.ewm(com=13, adjust=False).mean()
    rs = ema_up / ema_down
    rsi = 100 - (100 / (1 + rs))
    
    # Combine ROC and RSI into a single heuristic matrix
    heuristics_matrix = 0.7 * roc_50 + 0.3 * (100 - rsi)
    
    return heuristics_matrix
