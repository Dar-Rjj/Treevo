def heuristics_v2(df):
    # Calculate Donchian Channel
    donchian_high = df['high'].rolling(window=20).max()
    donchian_low = df['low'].rolling(window=20).min()
    donchian_channel = (donchian_high + donchian_low) / 2
    
    # Calculate Rate of Change (ROC)
    roc = (df['close'] - df['close'].shift(12)) / df['close'].shift(12) * 100
    
    # Calculate Accumulation/Distribution Line (ADL)
    mfm = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    adl = (mfm * df['volume']).cumsum()
    
    # Combine Donchian Channel, ROC, and ADL into a single heuristic matrix
    heuristics_matrix = 0.4 * donchian_channel + 0.3 * roc + 0.3 * adl
    
    return heuristics_matrix
