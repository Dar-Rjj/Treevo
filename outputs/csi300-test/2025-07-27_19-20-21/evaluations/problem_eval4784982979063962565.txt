def heuristics_v2(df):
    # Calculate Rate of Change (ROC)
    roc = (df['close'] - df['close'].shift(12)) / df['close'].shift(12) * 100
    
    # Calculate On-Balance Volume (OBV)
    obv = df['volume'].copy()
    obv[df['close'] < df['close'].shift(1)] -= df['volume']
    obv[df['close'] == df['close'].shift(1)] = 0
    obv[df['close'] > df['close'].shift(1)] += df['volume']
    obv = obv.cumsum()
    
    # Calculate Aroon Oscillator
    aroon_up = df['high'].rolling(window=25).apply(lambda x: (x.argmax() + 1) / 25 * 100, raw=True)
    aroon_down = df['low'].rolling(window=25).apply(lambda x: (x.argmin() + 1) / 25 * 100, raw=True)
    aroon_osc = aroon_up - aroon_down
    
    # Combine ROC, OBV, and Aroon Oscillator into a single heuristic matrix
    heuristics_matrix = 0.4 * roc + 0.3 * obv + 0.3 * aroon_osc
    
    return heuristics_matrix
