def heuristics_v2(df):
    # Calculate the 14-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 14-day ATR
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Normalize ATR to range [0, 1] for combination
    atr_normalized = (atr - atr.min()) / (atr.max() - atr.min())
    
    # Combine RSI and normalized ATR into a single heuristic matrix
    heuristics_matrix = 0.7 * rsi + 0.3 * (1 - atr_normalized)
    
    return heuristics_matrix
