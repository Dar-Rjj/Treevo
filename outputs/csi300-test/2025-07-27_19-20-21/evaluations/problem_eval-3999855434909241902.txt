def heuristics_v2(df):
    # Calculate the 9-period EMA and 21-period EMA
    ema_9 = df['close'].ewm(span=9, adjust=False).mean()
    ema_21 = df['close'].ewm(span=21, adjust=False).mean()
    
    # Calculate the EMA crossover signal
    ema_crossover = ema_9 - ema_21
    
    # Calculate the 14-day RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine EMA crossover and RSI into a single heuristic matrix
    heuristics_matrix = 0.5 * ema_crossover + 0.5 * (100 - rsi)
    
    return heuristics_matrix
