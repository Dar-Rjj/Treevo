def heuristics_v2(df):
    # Calculate Bollinger Bands
    sma = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    upper_band = sma + 2 * std_20
    lower_band = sma - 2 * std_20
    bb_diff = (df['close'] - lower_band) / (upper_band - lower_band)
    
    # Calculate On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Calculate custom relative strength measure
    short_sma = df['close'].rolling(window=10).mean()
    long_sma = df['close'].rolling(window=50).mean()
    rel_strength = (short_sma - long_sma) / long_sma
    
    # Combine the factors into a single heuristic matrix
    heuristics_matrix = 0.5 * bb_diff + 0.3 * obv + 0.2 * rel_strength
    
    return heuristics_matrix
