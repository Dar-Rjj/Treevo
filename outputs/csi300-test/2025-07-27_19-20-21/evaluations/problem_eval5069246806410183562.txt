def heuristics_v2(df):
    # Calculate exponential moving averages with different windows
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate RSI over 14 periods
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Bollinger Bands
    bb_middle = df['close'].rolling(window=20).mean()
    bb_std = df['close'].rolling(window=20).std()
    bb_upper = bb_middle + 2 * bb_std
    bb_lower = bb_middle - 2 * bb_std
    bb_diff = (df['close'] - bb_middle) / (bb_upper - bb_lower)
    
    # Combine the factors into a single heuristics matrix with new weights
    heuristics_matrix = 0.5 * (ema_short - ema_long) + 0.3 * rsi + 0.2 * bb_diff
    
    return heuristics_matrix
