def heuristics_v2(df):
    # Calculate Bollinger Bands
    sma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    upper_band = sma_20 + 2 * std_20
    lower_band = sma_20 - 2 * std_20
    bb_width = (upper_band - lower_band) / sma_20
    
    # Calculate the Price Rate of Change over 20 periods
    roc_20 = df['close'].pct_change(periods=20)
    
    # Calculate On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Combine the factors into a single heuristics matrix with weights
    heuristics_matrix = 0.5 * bb_width + 0.3 * roc_20 + 0.2 * obv
    
    return heuristics_matrix
