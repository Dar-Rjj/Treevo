def heuristics_v2(df):
    # Calculate Stochastic Oscillator
    lowest_low = df['low'].rolling(window=14).min()
    highest_high = df['high'].rolling(window=14).max()
    k_line = 100 * (df['close'] - lowest_low) / (highest_high - lowest_low)
    
    # Calculate ADX
    tr = pd.DataFrame(index=df.index)
    tr['h-l'] = df['high'] - df['low']
    tr['h-pc'] = abs(df['high'] - df['close'].shift(1))
    tr['l-pc'] = abs(df['low'] - df['close'].shift(1))
    tr['tr'] = tr.max(axis=1)
    dm_pos = (df['high'] - df['high'].shift(1)).apply(lambda x: max(x, 0)) if df['high'].diff() > df['low'].diff() else 0
    dm_neg = (df['low'].shift(1) - df['low']).apply(lambda x: max(x, 0)) if df['low'].diff() > df['high'].diff() else 0
    atr = tr['tr'].rolling(window=14).mean()
    di_pos = 100 * (dm_pos.rolling(window=14).mean() / atr)
    di_neg = 100 * (dm_neg.rolling(window=14).mean() / atr)
    dx = 100 * (abs(di_pos - di_neg) / (di_pos + di_neg))
    adx = dx.ewm(alpha=1/14).mean()
    
    # Calculate Elder's Force Index
    force_index_13 = df['close'].diff(1) * df['volume'].diff(1)
    efi = force_index_13.rolling(window=13).sum()
    
    # Combine Stochastic Oscillator, ADX, and EFI into a single heuristic matrix
    heuristics_matrix = 0.4 * k_line + 0.3 * adx + 0.3 * efi
    
    return heuristics_matrix
