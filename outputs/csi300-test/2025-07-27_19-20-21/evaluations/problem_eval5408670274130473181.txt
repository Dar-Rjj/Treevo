import pandas as pd

def heuristics_v2(df):
    # Calculate Bollinger Bands
    sma_15 = df['close'].rolling(window=15).mean()
    std_15 = df['close'].rolling(window=15).std()
    upper_band = sma_15 + 2 * std_15
    lower_band = sma_15 - 2 * std_15
    bollinger_width = (upper_band - lower_band) / sma_15
    
    # Calculate MACD
    ema_9 = df['close'].ewm(span=9, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_9 - ema_26
    signal = macd.rolling(window=5).mean()
    
    # Combine the factors into a single heuristics matrix
    heuristics_matrix = bollinger_width * macd - signal
    
    return heuristics_matrix
