def heuristics_v2(df):
    # Calculate the ratio of the close to the max high over the past 50 days
    ratio_close_max_high = df['close'] / df['high'].rolling(window=50).max()
    
    # Calculate the ratio of the close to the min low over the past 50 days
    ratio_close_min_low = df['close'] / df['low'].rolling(window=50).min()
    
    # Calculate the difference between the two ratios
    close_ratio_diff = ratio_close_max_high - ratio_close_min_low
    
    # Volume rate of change over 20 periods
    vol_roc_20 = df['volume'].pct_change(periods=20)
    
    # Heuristics matrix combining the close ratio difference and volume ROC
    heuristics_matrix = close_ratio_diff + vol_roc_20
    
    return heuristics_matrix
