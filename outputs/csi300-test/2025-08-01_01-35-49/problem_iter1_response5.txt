```python
import pandas as pd

def heuristics_v2(df, alpha=0.2):
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Compute Price Efficiency
    df['close_momentum'] = df['close'].diff()
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Aggregate Volume and Price Efficiency
    df['aggregate_efficiency'] = (df['volume_change'] * df['close_momentum']) * df['high_low_ratio']
    
    # Apply Exponential Smoothing
    df['alpha_factor'] = df['aggregate_efficiency'].ewm(alpha=alpha).mean()
    
    return df['alpha_factor']
```
