```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_spread = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Calculate Volume Dynamics
    volume_roc = df['volume'] / df['volume'].shift(1)
    
    intraday_momentum_volume = (high_low_spread + close_open_diff) * volume_roc
    
    # Incorporate Price Trend
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    trend_strength = ma_5 - ma_20
    
    # Integrate Short-Term and Long-Term Trends with Intraday Dynamics
    integrated_factor = trend_strength * intraday_momentum_volume
    
    return integrated_factor
```
