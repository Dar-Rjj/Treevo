```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine Intraday Return and High-Low Range
    combined_factor = intraday_return * high_low_range
    combined_factor_ema = combined_factor.ewm(span=14).mean()
    
    # Apply Volume Weighting
    volume_weighted_factor = combined_factor_ema * df['volume']
    
    # Incorporate Previous Day's Closing Gap
    closing_gap = df['open'] - df['close'].shift(1)
    gap_adjusted_factor = volume_weighted_factor + closing_gap
    
    # Integrate Long-Term Momentum
    long_term_return = df['close'] - df['close'].shift(50)
    normalized_long_term_return = long_term_return / high_low_range
    
    # Include Enhanced Dynamic Volatility Component
    rolling_std = intraday_return.rolling(window=20).std()
    atr = (df['high'] - df['low']).ewm(span=14).mean()
    combined_volatility = (rolling_std + atr) / 2
    adjusted_volatility = combined_volatility * df['volume']
    
    # Sector-Specific Adjustments
    # Assuming sector data is available in the DataFrame as 'sector_close'
    sector_average_return = df.groupby('sector')['close'].transform(lambda x: x.pct_change().rolling(window=30).mean())
    individual_stock_return = df['close'].pct_change()
    sector_adjustment = individual_stock_return - sector_average_return
    
    # Seasonality Adjustment
    historical_seasonal_return = df['close'].pct_change().groupby([df.index.month, df.index.day]).transform('mean')
    
    # Market Sentiment Indicator
    # Assuming sentiment scores are available in the DataFrame as 'sentiment_score'
    daily_sentiment_score = df['sentiment_score']
    
    # Final Factor Calculation
    final_factor = (
        gap_adjusted_factor +
        normalized_long_term_return +
        adjusted_volatility +
        sector_adjustment +
        historical_seasonal_return +
        daily_sentiment_score
    )
    
    # Apply Non-Linear Transformation
    final_factor = np.log(1 + final_factor)
    
    return final_factor
```
