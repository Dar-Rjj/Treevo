```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_high_low_spread = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']

    # Calculate Volume Thrust
    volume_rate_of_change = df['volume'] / df['volume'].shift(1)

    # Integrate Intraday Momentum and Volume Thrust
    intraday_momentum_volume_thrust = (intraday_high_low_spread + close_open_diff) * volume_rate_of_change

    # Incorporate Price Trend
    five_day_ma = df['close'].rolling(window=5).mean()
    twenty_day_ma = df['close'].rolling(window=20).mean()

    price_trend = five_day_ma - twenty_day_ma

    # Final Integration
    final_factor = intraday_momentum_volume_thrust * price_trend

    return final_factor
```
