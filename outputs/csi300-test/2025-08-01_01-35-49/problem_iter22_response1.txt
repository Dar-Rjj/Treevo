```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine Intraday Return and High-Low Range
    combined_factor = intraday_return * high_low_range
    smoothed_factor = combined_factor.ewm(span=14, adjust=False).mean()
    
    # Apply Volume Weighting
    volume_weighted_factor = smoothed_factor * df['volume']
    
    # Incorporate Previous Day's Closing Gap
    prev_day_close_gap = df['open'].diff()
    prev_day_close_gap.iloc[0] = 0  # Handle the first row
    volume_weighted_factor_with_gap = volume_weighted_factor + prev_day_close_gap
    
    # Integrate Long-Term Momentum
    long_term_return = df['close'] - df['close'].shift(50)
    normalized_long_term_return = long_term_return / high_low_range
    
    # Include Enhanced Dynamic Volatility Component
    rolling_std = df['close'].pct_change().rolling(window=20).std()
    atr = (df['high'] - df['low']).rolling(window=14).mean()
    combined_volatility = (rolling_std + atr) / 2
    adjusted_volatility = combined_volatility * df['volume']
    
    # Additional Volatility Measures
    rolling_variance = df['close'].pct_change().rolling(window=20).var()
    historical_volatility = df['close'].pct_change().rolling(window=30).std()
    enhanced_combined_volatility = (rolling_std + atr + rolling_variance + historical_volatility) / 4
    enhanced_adjusted_volatility = enhanced_combined_volatility * df['volume']
    
    # Final Factor Calculation
    final_factor = volume_weighted_factor_with_gap + normalized_long_term_return + enhanced_adjusted_volatility
    
    # Apply Non-Linear Transformation
    non_linear_transformed_factor = np.log(1 + final_factor**2)
    
    # Refine Non-Linear Transformations
    sigmoid_transformed_factor = 1 / (1 + np.exp(-final_factor))
    softmax_transformed_factor = np.exp(final_factor) / np.exp(final_factor).sum()
    
    # Final Alpha Factor
    alpha_factor = non_linear_transformed_factor + sigmoid_transformed_factor + softmax_transformed_factor
    
    return alpha_factor
```
