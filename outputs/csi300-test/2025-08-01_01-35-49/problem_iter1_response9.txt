```python
import pandas as pd

def heuristics_v2(df):
    # Compute Price Difference
    price_diff = df['high'] - df['low']
    
    # Calculate Intraday Range Growth
    intraday_range_growth = price_diff / df['open']
    
    # Calculate Volume Weighted Average
    volume_weighted_avg = intraday_range_growth * df['volume']
    
    # Adjust by Close-to-Open Spread
    close_to_open_spread = (df['close'] - df['open']) / df['open']
    
    # Final factor value
    factor_values = volume_weighted_avg + close_to_open_spread
    
    return factor_values
```
