```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    volume_weighted_intraday_return_10d = df['volume_weighted_intraday_return'].rolling(window=10).sum()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Weighted High-Low Range
    df['weighted_high_low_range'] = df['high_low_range'] * df['amount']
    
    # Combine Volume-Weighted Intraday Return and Weighted High-Low Range
    df['enhanced_intraday_momentum'] = volume_weighted_intraday_return_10d + df['weighted_high_low_range']
    
    return df['enhanced_intraday_momentum']
```
