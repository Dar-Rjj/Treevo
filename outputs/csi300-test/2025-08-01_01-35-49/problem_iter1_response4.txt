```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].diff()
    
    # Compute Upward and Downward Movements
    df['upward_movement'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['downward_movement'] = df['daily_return'].apply(lambda x: -x if x < 0 else 0)
    
    # Smoothed Upward and Downward Movements
    window = 14
    df['smoothed_up'] = df['upward_movement'].rolling(window=window).mean()
    df['smoothed_down'] = df['downward_movement'].rolling(window=window).mean()
    
    # Calculate Relative Strength
    df['relative_strength'] = df['smoothed_up'] / df['smoothed_down']
    
    # Calculate RSI
    df['RSI'] = (1 + df['relative_strength']) / 2 * 100
    
    # Adjust RSI by Volume and Amount
    # Compute Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Amount Weighted Average Price (AWAP)
    df['AWAP'] = (df[['high', 'low', 'close']].mean(axis=1) * df['amount']).cumsum() / df['amount'].cumsum()
    
    # Calculate VWAP Factor
    df['VWAP_factor'] = (df['VWAP'] - df['VWAP'].shift(1)) / df['VWAP'].shift(1)
    
    # Calculate AWAP Factor
    df['AWAP_factor'] = (df['AWAP'] - df['AWAP'].shift(1)) / df['AWAP'].shift(1)
    
    # Combine VWAP and AWAP Factors
    df['combined_factors'] = df['VWAP_factor'] + df['AWAP_factor']
    
    # Multiply Combined Factors with RSI
    df['adjusted_RSI'] = df['RSI'] * df['combined_factors']
    
    return df['adjusted_RSI']
```
