```python
import pandas as pd
import numpy as np

def heuristics_v2(df, ema_period=10, long_term_period=50):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Combine Intraday Return and High-Low Range
    combined_factor = intraday_return * intraday_range
    
    # Smooth using Exponential Moving Average (EMA)
    smoothed_factor = combined_factor.ewm(span=ema_period, adjust=False).mean()
    
    # Apply Volume Weighting
    volume_weighted_factor = smoothed_factor * df['volume']
    
    # Incorporate Previous Day's Closing Gap
    closing_gap = (df['open'].shift(-1) - df['close']).shift(1)
    volume_weighted_factor_with_gap = volume_weighted_factor + closing_gap
    
    # Integrate Long-Term Momentum
    long_term_return = (df['close'] - df['close'].shift(long_term_period)) / df['close'].shift(long_term_period)
    normalized_long_term_return = long_term_return / intraday_range
    
    # Final Factor Calculation
    final_factor = volume_weighted_factor_with_gap + normalized_long_term_return
    
    # Apply Non-Linear Transformation
    final_factor = np.log1p(final_factor)
    
    return final_factor
```
