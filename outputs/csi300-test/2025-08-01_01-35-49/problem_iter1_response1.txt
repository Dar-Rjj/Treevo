```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjust by Previous Close
    prev_close = df['close'].shift(1)
    adj_intraday_range = intraday_range / prev_close
    
    # Consider Volume Momentum
    volume_diff = (df['volume'] - df['volume'].shift(1)).abs()
    volume_momentum = volume_diff / df['volume'].shift(1)
    
    # Combine Intraday Volatility and Volume Momentum
    alpha_factor = adj_intraday_range * volume_momentum
    
    return alpha_factor
```
