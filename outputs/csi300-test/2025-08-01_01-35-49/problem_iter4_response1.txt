```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['low']
    
    # Combine Intraday Return and High-Low Range
    df['combined_factor'] = df['intraday_return'] * df['high_low_range']
    
    # Smooth using Exponential Moving Average (EMA) with a period of 5
    df['smoothed_combined_factor'] = df['combined_factor'].ewm(span=5, adjust=False).mean()
    
    # Apply Volume Weighting
    df['volume_weighted_factor'] = df['smoothed_combined_factor'] * df['volume']
    
    # Incorporate Historical Volatility
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['historical_volatility'] = df['log_returns'].rolling(window=20).std()
    df['normalized_volatility'] = df['historical_volatility'] / df['historical_volatility'].rolling(window=20).max()
    
    # Integrate All Components
    df['integrated_factor'] = df['volume_weighted_factor'] * df['normalized_volatility']
    
    # Final Smoothing with a 5-Day Simple Moving Average
    df['final_alpha_factor'] = df['integrated_factor'].rolling(window=5).mean()
    
    return df['final_alpha_factor']
```
