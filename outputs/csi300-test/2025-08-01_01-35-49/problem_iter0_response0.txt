def heuristics_v2(data):
    """
    Calculate a volume weighted close-to-open return factor.
    
    :param data: A pandas DataFrame with (open, high, low, close, amount, volume) prices and features.
    :return: A pandas Series representing the factor values.
    """
    # Calculate Close-to-Open Return
    close_to_open_return = data['open'] / data['close'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * data['volume']
    
    return volume_weighted_return
