```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Compute Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Price Efficiency
    df['price_efficiency'] = df['intraday_return'] / df['high_low_range']
    
    # Aggregate Volume and Price Efficiency
    df['vol_price_efficiency'] = df['volume_change'] * df['price_efficiency']
    
    # Apply Simple Moving Average with a lookback period of 20 days
    lookback_period = 20
    df['alpha_factor'] = df['vol_price_efficiency'].rolling(window=lookback_period).mean()
    
    return df['alpha_factor']
```
