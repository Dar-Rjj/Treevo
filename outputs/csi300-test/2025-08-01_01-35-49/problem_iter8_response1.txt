```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Intraday High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Combine Intraday Return and High-Low Range
    df['combined_factor'] = df['intraday_return'] * df['high_low_range']
    
    # Smooth using Exponential Moving Average (EMA) with 14 days
    df['smoothed_combined_factor'] = df['combined_factor'].ewm(span=14, adjust=False).mean()
    
    # Apply Volume Weighting
    df['volume_weighted_factor'] = df['smoothed_combined_factor'] * df['volume']
    
    # Incorporate Previous Day's Closing Gap
    df['prev_day_close'] = df['close'].shift(1)
    df['closing_gap'] = df['open'] - df['prev_day_close']
    df['volume_weighted_closing_gap'] = df['volume_weighted_factor'] + df['closing_gap']
    
    # Integrate Long-Term Momentum
    df['long_term_return'] = df['close'] - df['close'].shift(50)
    df['normalized_long_term_return'] = df['long_term_return'] / df['high_low_range']
    
    # Include Adaptive Volatility Component
    df['intraday_volatility'] = df['intraday_return'].rolling(window=20).std()
    df['atr'] = df['high_low_range'].ewm(span=14, adjust=False).mean()
    df['combined_volatility'] = (df['intraday_volatility'] + df['atr']) / 2
    
    # Adjust Volatility Component with Volume
    df['volume_adjusted_volatility'] = df['combined_volatility'] * df['volume']
    
    # Incorporate Sentiment Analysis (Assuming a sentiment score column is available)
    df['sentiment_score'] = df.get('sentiment_score', 1.0)  # Default to 1.0 if no sentiment score
    df['sentiment_adjusted_volatility'] = df['volume_adjusted_volatility'] * df['sentiment_score']
    
    # Final Factor Calculation
    df['final_factor'] = (
        df['volume_weighted_closing_gap'] +
        df['normalized_long_term_return'] +
        df['sentiment_adjusted_volatility']
    )
    
    # Apply Non-Linear Transformation
    df['alpha_factor'] = np.log(1 + df['final_factor'])
    
    return df['alpha_factor']
```
