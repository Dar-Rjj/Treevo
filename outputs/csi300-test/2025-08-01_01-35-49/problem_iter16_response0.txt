```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine Intraday Return and High-Low Range
    combined_factor = intraday_return * high_low_range
    
    # Adaptive EMA Period based on recent volatility
    recent_volatility = df['close'].pct_change().rolling(window=5).std()
    ema_period = 5 + (14 - 5) * (recent_volatility / recent_volatility.rolling(window=30).max())
    ema_period = ema_period.round().astype(int)
    
    # Smooth using Exponential Moving Average (EMA)
    smoothed_combined_factor = combined_factor.ewm(span=ema_period, adjust=False).mean()
    
    # Apply Volume Weighting
    volume_weighted_factor = smoothed_combined_factor * df['volume']
    
    # Incorporate Previous Day's Closing Gap
    previous_day_close_gap = df['open'] - df['close'].shift(1)
    volume_weighted_with_gap = volume_weighted_factor + previous_day_close_gap
    
    # Integrate Long-Term Momentum
    long_term_return = df['close'] - df['close'].shift(50)
    adjusted_long_term_return = long_term_return / high_low_range
    
    # Include Dynamic Volatility Component
    adaptive_rolling_window = 10 + (20 - 10) * (recent_volatility / recent_volatility.rolling(window=30).max())
    adaptive_rolling_window = adaptive_rolling_window.round().astype(int)
    rolling_std_dev = intraday_return.rolling(window=adaptive_rolling_window).std()
    
    # Calculate Average True Range (ATR)
    adaptive_atr_window = 7 + (14 - 7) * (recent_volatility / recent_volatility.rolling(window=30).max())
    adaptive_atr_window = adaptive_atr_window.round().astype(int)
    tr = np.maximum(np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))), abs(df['low'] - df['close'].shift(1)))
    atr = tr.rolling(window=adaptive_atr_window).mean()
    
    # Adjust Volatility Component with Volume
    volume_adjusted_volatility = (rolling_std_dev + atr) * df['volume']
    
    # Final Factor Calculation
    final_factor = volume_weighted_with_gap + adjusted_long_term_return + volume_adjusted_volatility
    
    # Apply Non-Linear Transformation
    final_factor = np.log(1 + final_factor)
    
    return final_factor
```
