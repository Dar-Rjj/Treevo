```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Close to Open Difference
    df['close_open_diff'] = df['close'] - df['open']

    # Calculate Volume Rate of Change
    df['volume_rate_of_change'] = df['volume'] / df['volume'].shift(1)

    # Integrate Intraday Momentum and Volume Thrust
    df['integrated_momentum_thrust'] = (df['intraday_high_low_spread'] + 
                                         df['close_open_diff']) * df['volume_rate_of_change']

    # Calculate 5-Day Moving Average
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate 20-Day Moving Average
    df['20_day_ma'] = df['close'].rolling(window=20).mean()

    # Subtract 20-Day MA from 5-Day MA
    df['ma_diff'] = df['5_day_ma'] - df['20_day_ma']

    # Final Integration
    df['alpha_factor'] = df['integrated_momentum_thrust'] * df['ma_diff']

    return df['alpha_factor'].copy()
```
