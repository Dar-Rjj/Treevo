```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Enhanced Intraday Momentum
    volume_enhancement = df['volume'] * intraday_return
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Add Weighted High-Low Range
    amount_weighted_range = df['amount'] * high_low_range
    
    # Combine the components to form the final factor
    enhanced_momentum = volume_enhancement + amount_weighted_range
    
    return enhanced_momentum
```
