```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close to Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Calculate Volume Rate of Change
    volume_roc = df['volume'].pct_change()
    
    # Combine Intraday Momentum components
    intraday_momentum = (high_low_spread + close_open_diff) / 2
    
    # Integrate Intraday Momentum and Volume Thrust
    factor_values = intraday_momentum * volume_roc
    
    return factor_values
```
