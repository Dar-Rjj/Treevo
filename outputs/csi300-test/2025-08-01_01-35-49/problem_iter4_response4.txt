```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = (intraday_return * df['volume']).rolling(window=10).sum()
    
    # Calculate Weighted High-Low Range
    high_low_range = (df['high'] - df['low']) / df['open']
    weighted_high_low_range = high_low_range * df['amount']
    
    # Calculate Multi-Day Momentum
    multi_day_momentum = df['close'].pct_change(periods=5)
    
    # Combine Volume-Weighted Intraday Return, Weighted High-Low Range, and Multi-Day Momentum
    combined_factor = volume_weighted_intraday_return + weighted_high_low_range + multi_day_momentum
    
    return combined_factor
```
