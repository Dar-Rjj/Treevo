```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['momentum'] = df['close'] - df['close'].shift(20)
    
    # Measure Daily Price Range
    df['daily_range'] = df['high'] - df['low']
    
    # Compute Average Daily Range over 20 Days
    df['avg_20_day_range'] = df['daily_range'].rolling(window=20).mean()
    
    # Compute Expansion Ratio
    df['expansion_ratio'] = df['daily_range'] / df['avg_20_day_range']
    
    # Adjust Momentum by Expansion Ratio
    df['adjusted_momentum'] = df['momentum'] * df['expansion_ratio']
    
    # Return the adjusted momentum factor
    return df['adjusted_momentum'].dropna()
```
