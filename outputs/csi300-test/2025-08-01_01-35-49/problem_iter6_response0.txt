import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Intraday High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Combine Intraday Return and High-Low Range
    df['combined_factor'] = df['intraday_return'] * df['high_low_range']
    
    # Smooth using Exponential Moving Average (EMA) with period 14
    df['smoothed_combined_factor'] = df['combined_factor'].ewm(span=14, adjust=False).mean()
    
    # Apply Volume Weighting
    df['volume_weighted_factor'] = df['smoothed_combined_factor'] * df['volume']
    
    # Incorporate Previous Day's Closing Gap
    df['previous_close_gap'] = df['open'] - df['close'].shift(1)
    df['volume_weighted_with_gap'] = df['volume_weighted_factor'] + df['previous_close_gap']
    
    # Integrate Long-Term Momentum
    df['long_term_return'] = df['close'] - df['close'].shift(50)
    df['normalized_long_term_return'] = df['long_term_return'] / df['high_low_range']
    
    # Include Volatility Component
    df['intraday_returns_rolling_std'] = df['intraday_return'].rolling(window=20).std()
    df['volume_adjusted_volatility'] = df['intraday_returns_rolling_std'] * df['volume']
    
    # Incorporate Market Sentiment (for simplicity, assume a constant sentiment score of 0.1)
    sentiment_score = 0.1
    df['market_sentiment_adjusted_factor'] = df['volume_weighted_with_gap'] * (1 + sentiment_score)
    
    # Final Factor Calculation
    df['final_factor'] = (
        df['volume_weighted_with_gap'] +
        df['normalized_long_term_return'] +
        df['volume_adjusted_volatility'] +
        df['market_sentiment_adjusted_factor']
    )
    
    # Apply Non-Linear Transformation
    df['alpha_factor'] = np.log1p(df['final_factor'])
    
    return df['alpha_factor']
