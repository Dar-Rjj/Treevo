```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Price Difference
    df['price_diff'] = df['high'] - df['low']
    
    # Calculate Intraday Range Growth
    df['intraday_range_growth'] = df['price_diff'] / df['open']
    
    # Calculate Close-to-Open Spread
    df['close_to_open_spread'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Weighted Average
    df['volume_weighted_avg'] = df['volume'] * df['intraday_range_growth']
    
    # Adjust by Close-to-Open Spread
    df['adjusted_volume_weighted_avg'] = df['volume_weighted_avg'] * df['close_to_open_spread']
    
    # Calculate Rolling Standard Deviation of Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    df['volatility'] = df['daily_returns'].rolling(window=20).std()
    
    # Normalize Intraday Range Growth by Volatility
    df['normalized_intraday_growth'] = df['intraday_range_growth'] / df['volatility']
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'].pct_change(5)
    
    # Combine with Intraday Range Growth
    df['combined_factors'] = df['normalized_intraday_growth'] + df['price_momentum']
    
    # Final Factor
    df['final_factor'] = df['combined_factors'] * df['adjusted_volume_weighted_avg']
    
    # Drop rows with NaN values caused by rolling window
    df.dropna(inplace=True)
    
    # Return the final factor as a pandas Series
    return df['final_factor']
```
