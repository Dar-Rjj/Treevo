```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Log Return
    df['log_return'] = np.log(df['close'] / df['open'])
    
    # Calculate Intraday High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Combine Intraday Log Return and High-Low Range
    df['combined_factor'] = df['log_return'] * df['high_low_range']
    
    # Smooth using Exponential Moving Average (EMA)
    df['ema_smoothed'] = df['combined_factor'].ewm(span=14, min_periods=14).mean()
    
    # Apply Volume Weighting
    df['volume_weighted'] = df['ema_smoothed'] * df['volume']
    
    # Incorporate Previous Day's Closing Gap
    df['prev_day_gap'] = df['open'] - df['close'].shift(1)
    df['gap_adjusted'] = df['volume_weighted'] + df['prev_day_gap']
    
    # Integrate Long-Term Log Momentum
    long_term_days = 50
    df['long_term_log_return'] = np.log(df['close'] / df['close'].shift(long_term_days))
    df['normalized_long_term'] = df['long_term_log_return'] / df['high_low_range']
    
    # Include Adaptive Volatility Component
    rolling_window = 20
    df['intraday_log_returns_rolling_std'] = df['log_return'].rolling(window=rolling_window, min_periods=rolling_window).std()
    
    # Calculate Average True Range (ATR)
    df['true_range'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    df['atr'] = df['true_range'].rolling(window=14, min_periods=14).mean()
    df['combined_volatility'] = (df['intraday_log_returns_rolling_std'] + df['atr']) / 2
    
    # Adjust Volatility Component with Volume
    df['volatility_volume_adjusted'] = df['combined_volatility'] * df['volume']
    
    # Final Factor Calculation
    df['final_factor'] = (df['gap_adjusted'] + df['normalized_long_term'] + df['volatility_volume_adjusted'])
    
    # Apply Quadratic Non-Linear Transformation
    df['heuristic_factor'] = np.log(1 + df['final_factor'] ** 2)
    
    return df['heuristic_factor']
```
