```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on intraday volatility adjusted return.

    Parameters:
    df (pd.DataFrame): A pandas DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].

    Returns:
    pd.Series: A pandas Series indexed by (date) representing the factor values.
    """
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']

    # Calculate Base Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Adjust Close-to-Open Return with Intraday Range
    adjusted_return = close_to_open_return / intraday_range.replace(0, 1e-10)  # Avoid division by zero
    
    # Multiply by Volume to get the final factor
    factor = adjusted_return * df['volume']
    
    return factor
```
