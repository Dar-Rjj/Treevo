```python
import pandas as pd
import numpy as np

def heuristics_v2(df, span=10):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Adjust Close-to-Open Return with Intraday Range
    adjusted_return = close_to_open_return / np.where(intraday_range != 0, intraday_range, 1)
    
    # Weight by Volume
    volume_weighted_return = adjusted_return * df['volume']
    
    # Exponential Moving Average of Adjusted Returns
    ema_adjusted_return = volume_weighted_return.ewm(span=span, adjust=False).mean()
    
    return ema_adjusted_return
```
