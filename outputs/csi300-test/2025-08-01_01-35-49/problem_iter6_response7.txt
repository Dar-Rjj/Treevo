import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    volume_weighted_intraday_return = df['volume_weighted_intraday_return'].rolling(window=10).sum()
    
    # Calculate High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Weighted High-Low Range
    df['weighted_high_low_range'] = df['high_low_range'] * df['amount']
    
    # Calculate Multi-Day Momentum
    multi_day_momentum = df['close'].pct_change(periods=5)
    
    # Integrate Volume-Weighted Intraday Return, Weighted High-Low Range, and Multi-Day Momentum
    integrated_factor = (volume_weighted_intraday_return + df['weighted_high_low_range']) * multi_day_momentum
    
    return integrated_factor
