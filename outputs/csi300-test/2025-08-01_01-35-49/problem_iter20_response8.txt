```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine Intraday Return and High-Low Range
    combined_factor = intraday_return * high_low_range
    combined_factor_ema = combined_factor.ewm(span=14).mean()
    
    # Apply Volume Weighting
    volume_weighted_factor = combined_factor_ema * df['volume']
    
    # Incorporate Previous Day's Closing Gap
    prev_day_close_gap = df['open'] - df['close'].shift(1)
    volume_weighted_factor_with_gap = volume_weighted_factor + prev_day_close_gap
    
    # Integrate Long-Term Momentum
    long_term_return = df['close'] - df['close'].shift(50)
    normalized_long_term_return = long_term_return / high_low_range
    
    # Include Enhanced Dynamic Volatility Component
    rolling_std = intraday_return.rolling(window=20).std()
    atr = (df['high'] - df['low']).ewm(span=14, adjust=False).mean()
    combined_volatility = (rolling_std + atr) / 2
    adjusted_volatility = combined_volatility * df['volume']
    
    # Incorporate Market Sentiment
    # Note: This assumes a column 'news_sentiment_ratio' is available in the DataFrame
    news_sentiment_ratio = df.get('news_sentiment_ratio', 0)
    
    # Incorporate Liquidity Measures
    average_daily_volume = df['volume'].rolling(window=30).mean()
    turnover_ratio = df['volume'] / average_daily_volume
    
    # Incorporate Sector-Specific Trends
    # Note: This assumes a column 'sector_index' is available in the DataFrame
    sector_index_momentum = (df.get('sector_index', 0) - df.get('sector_index', 0).shift(50))
    
    # Final Factor Calculation
    final_factor = (
        volume_weighted_factor_with_gap +
        normalized_long_term_return +
        adjusted_volatility +
        news_sentiment_ratio +
        turnover_ratio +
        sector_index_momentum
    )
    
    # Apply Non-Linear Transformation
    final_factor_transformed = np.log1p(final_factor)
    
    return final_factor_transformed
```
