def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily range to capture volatility
    daily_range = df['high'] - df['low']
    
    # Calculate a weighted close based on volume for a more accurate price movement indicator
    weighted_close = (df['close'] * df['volume']) / df['volume'].rolling(window=5).mean()
    
    # Calculate typical price for money flow analysis
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate money flow
    money_flow = typical_price * df['volume']
    
    # Calculate positive and negative money flow
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Calculate the money flow ratio for buying and selling pressure
    money_flow_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum().fillna(1)
    
    # Incorporate the dynamic daily range as a measure of volatility
    dynamic_volatility = daily_range.rolling(window=20).std() + 1e-7
    volatility_factor = 1 / dynamic_volatility
    
    # Calculate a momentum factor using the difference between the current close and the exponential moving average
    exp_momentum = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    
    # Combine these factors in a heuristic way to create a novel alpha factor
    alpha_factor = (weighted_close / (df['open'] + daily_range) * money_flow_ratio * volatility_factor * exp_momentum).fillna(0)
    
    return alpha_factor
