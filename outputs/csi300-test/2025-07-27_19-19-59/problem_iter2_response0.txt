```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price
    co_diff = df['close'] - df['open']
    
    # Calculate the average of high and low prices
    avg_hl = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of the volume to the amount
    vol_to_amt_ratio = df['volume'] / (df['amount'] + 1e-7)
    
    # Calculate the volatility as the range divided by the previous day's close
    volatility = (df['high'] - df['low']) / df['close'].shift(1).fillna(1)
    
    # Calculate the transaction intensity as the ratio of volume to the average price
    trans_intensity = df['volume'] / (avg_hl + 1e-7)
    
    # Novel alpha factor: (Close-Open difference * Volume-to-Amount Ratio * Volatility * Transaction Intensity) / Average of High and Low
    novel_factor = (co_diff * vol_to_amt_ratio * volatility * trans_intensity) / (avg_hl + 1e-7)
    return novel_factor
```
