```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily range to capture volatility
    daily_range = df['high'] - df['low']
    
    # Adaptive window for moving average based on recent volatility
    adaptive_window = (daily_range.rolling(window=10).std() * 5).fillna(20).astype(int)
    adaptive_window = adaptive_window.apply(lambda x: max(1, min(x, 30)))  # Cap the window size between 1 and 30
    
    # Calculate a weighted close based on volume for a more accurate price movement indicator
    weighted_close = (df['close'] * df['volume']) / df['volume'].rolling(window=adaptive_window, min_periods=1).mean()
    
    # Calculate typical price for money flow analysis
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate money flow
    money_flow = typical_price * df['volume']
    
    # Calculate positive and negative money flow
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Calculate the money flow ratio for buying and selling pressure with an adaptive window
    money_flow_ratio = (positive_money_flow.rolling(window=adaptive_window, min_periods=1).sum() /
                        negative_money_flow.rolling(window=adaptive_window, min_periods=1).sum().fillna(1))
    
    # Incorporate the daily range as a measure of dynamic, volume-weighted volatility
    volatility_factor = 1 / (daily_range + 1e-7)
    dynamic_volatility = (volatility_factor * df['volume']).rolling(window=adaptive_window, min_periods=1).mean()
    
    # Calculate a momentum factor using the difference between the current close and the adaptive moving average
    momentum_factor = df['close'] - df['close'].rolling(window=adaptive_window, min_periods=1).mean()
    
    # Additional factor: Volume-adjusted momentum
    volume_adjusted_momentum = momentum_factor * (df['volume'] / df['volume'].rolling(window=adaptive_window, min_periods=1).mean())
    
    # Simplify the combination of factors by using a weighted sum
    alpha_factor = (weighted_close * 0.4 + money_flow_ratio * 0.3 + dynamic_volatility * 0.1 + volume_adjusted_momentum * 0.2).fillna(0)
    
    return alpha_factor
```
