```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily range to capture volatility
    daily_range = df['high'] - df['low']
    
    # Calculate typical price for money flow analysis
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate money flow
    money_flow = typical_price * df['volume']
    
    # Calculate positive and negative money flow
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Calculate the money flow ratio for buying and selling pressure with a fixed window
    money_flow_ratio = (positive_money_flow.rolling(window=20).sum() / 
                        negative_money_flow.rolling(window=20).sum().fillna(1))
    
    # Incorporate the daily range as a measure of volatility
    volatility_factor = 1 / (daily_range + 1e-7)
    
    # Adaptive window for momentum based on standard deviation
    adaptive_window = (df['close'].rolling(window=20).std() / df['close']).clip(lower=5, upper=50).astype(int)
    
    # Calculate a momentum factor using the difference between the current close and the moving average with adaptive window
    momentum_factor = df['close'] - df['close'].rolling(window=adaptive_window).mean()
    
    # Consider market trend by adding a trend-following factor with adaptive window
    trend_factor = df['close'].pct_change(periods=adaptive_window).rolling(window=adaptive_window).mean()
    
    # Additional factor: Volume-adjusted momentum
    volume_adjusted_momentum = momentum_factor * (df['volume'] / df['volume'].rolling(window=10).mean())
    
    # Combine these factors in a heuristic way to create a novel alpha factor
    alpha_factor = (money_flow_ratio * volatility_factor * volume_adjusted_momentum * trend_factor).fillna(0)
    
    return alpha_factor
```
