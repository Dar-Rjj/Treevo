```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price
    price_diff = df['close'] - df['open']
    
    # Calculate the average of high and low price
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the signed volume based on the sign of the price difference
    signed_volume = df['volume'] * (price_diff / abs(price_diff)).fillna(0)
    
    # Calculate the factor by dividing the signed volume with the average price
    factor = signed_volume / (avg_price + 1e-7)
    
    return factor
```
