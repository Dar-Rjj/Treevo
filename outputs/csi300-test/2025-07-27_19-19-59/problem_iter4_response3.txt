```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily range to capture volatility
    daily_range = df['high'] - df['low']
    
    # Calculate a smoothed close price using an exponential moving average (EMA)
    smoothed_close = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate a weighted close based on volume for more accurate price movement indicator
    weighted_close = (df['close'] * df['volume']) / df['volume'].rolling(window=3).mean()
    
    # Calculate the money flow ratio for buying and selling pressure
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_flow_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    
    # Calculate momentum by comparing the current close price to the EMA of the last 5 closes
    momentum = df['close'] / df['close'].ewm(span=5, adjust=False).mean()
    
    # Combine these factors in a heuristic way to create a novel alpha factor
    alpha_factor = (smoothed_close / (df['open'] + daily_range) * money_flow_ratio * momentum).fillna(0)
    
    return alpha_factor
```
