```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily range to capture volatility
    daily_range = df['high'] - df['low']
    
    # Calculate a dynamically weighted close, using volume for more accurate price movement
    vol_window = 5
    dynamic_vol = df['volume'].rolling(window=vol_window).mean()
    weighted_close = (df['close'] * df['volume']) / dynamic_vol
    
    # Calculate typical price for money flow analysis
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate money flow
    money_flow = typical_price * df['volume']
    
    # Calculate positive and negative money flow with a shorter window
    mf_window = 10
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0).rolling(window=mf_window).sum()
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0).rolling(window=mf_window).sum().fillna(1)
    
    # Calculate the money flow ratio for buying and selling pressure
    money_flow_ratio = positive_money_flow / negative_money_flow
    
    # Integrate dynamic, volume-weighted volatility
    vol_factor = 1 / (daily_range.rolling(window=vol_window).std() + 1e-7)
    
    # Calculate a short-term momentum factor
    mom_window = 10
    momentum_factor = (df['close'] - df['close'].rolling(window=mom_window).mean()) / df['close'].rolling(window=mom_window).std()
    
    # Combine these factors in a heuristic way to create a novel alpha factor
    alpha_factor = (weighted_close / (df['open'] + daily_range) * money_flow_ratio * vol_factor * momentum_factor).fillna(0)
    
    return alpha_factor
```
