```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily range to capture volatility
    daily_range = df['high'] - df['low']
    
    # Calculate a weighted close based on volume for a more accurate price movement indicator
    weighted_close = (df['close'] * df['volume']) / df['volume'].rolling(window=3).mean()
    
    # Calculate typical price for money flow analysis
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate money flow
    money_flow = typical_price * df['volume']
    
    # Calculate positive and negative money flow
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Calculate the money flow ratio for buying and selling pressure
    money_flow_ratio = positive_money_flow.rolling(window=14).sum() / (negative_money_flow.rolling(window=14).sum() + 1e-7)
    
    # Incorporate the average true range as a measure of dynamic volatility
    true_range = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs()
    true_range += (df['high'] - df['close'].shift()).abs()
    true_range += (df['low'] - df['close'].shift()).abs()
    average_true_range = true_range.rolling(window=14).mean()
    dynamic_volatility_factor = 1 / (average_true_range + 1e-7)
    
    # Combine these factors in a heuristic way to create a novel alpha factor
    alpha_factor = (weighted_close / (df['open'] + daily_range) * money_flow_ratio * dynamic_volatility_factor).fillna(0)
    
    return alpha_factor
```
