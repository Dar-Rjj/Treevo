```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily range to capture volatility
    daily_range = df['high'] - df['low']
    
    # Dynamic window for moving average based on the 20-day simple moving average of volume
    dynamic_window = (df['volume'].rolling(window=20, min_periods=1).mean() / df['volume']).fillna(1).astype(int)
    
    # Calculate a weighted close based on volume for a more accurate price movement indicator
    weighted_close = (df['close'] * df['volume']) / df['volume'].rolling(window=dynamic_window, min_periods=1).mean()
    
    # Calculate typical price for money flow analysis
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate money flow
    money_flow = typical_price * df['volume']
    
    # Calculate positive and negative money flow
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Calculate the money flow ratio for buying and selling pressure with a dynamic window
    money_flow_ratio = (positive_money_flow.rolling(window=dynamic_window, min_periods=1).sum() /
                        negative_money_flow.rolling(window=dynamic_window, min_periods=1).sum().fillna(1))
    
    # Incorporate the daily range as a measure of dynamic, volume-weighted volatility
    volatility_factor = 1 / (daily_range + 1e-7)
    dynamic_volatility = (volatility_factor * df['volume']).rolling(window=dynamic_window, min_periods=1).mean()
    
    # Calculate a momentum factor using the difference between the current close and the dynamic moving average
    momentum_factor = df['close'] - df['close'].rolling(window=dynamic_window, min_periods=1).mean()
    
    # Additional factor: Volume-adjusted momentum
    volume_adjusted_momentum = momentum_factor * (df['volume'] / df['volume'].rolling(window=dynamic_window, min_periods=1).mean())
    
    # Combine these factors in a heuristic way to create a novel alpha factor
    alpha_factor = (weighted_close / (df['open'] + daily_range) * money_flow_ratio * dynamic_volatility * volume_adjusted_momentum).fillna(0)
    
    return alpha_factor
```
