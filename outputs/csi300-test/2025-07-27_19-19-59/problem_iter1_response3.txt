```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    co_diff = df['close'] - df['open']
    # Calculate the average of high and low prices
    avg_hl = (df['high'] + df['low']) / 2
    # Calculate the ratio of the volume to the amount
    vol_to_amt_ratio = df['volume'] / (df['amount'] + 1e-7)
    
    # Novel alpha factor: (Close-Open difference * Volume-to-Amount Ratio) / Average of High and Low
    novel_factor = (co_diff * vol_to_amt_ratio) / (avg_hl + 1e-7)
    return novel_factor
```
