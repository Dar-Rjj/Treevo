def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily range to capture volatility
    daily_range = df['high'] - df['low']
    
    # Adaptive window for weighted close
    adaptive_window = df['volume'].rolling(window=10).mean()
    weighted_close = (df['close'] * df['volume']) / adaptive_window
    
    # Calculate typical price for money flow analysis
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate money flow
    money_flow = typical_price * df['volume']
    
    # Calculate positive and negative money flow
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Dynamic window size for money flow ratio
    window_size = df['volume'].rolling(window=5).median().astype(int) * 2
    window_size = window_size.clip(lower=10, upper=30)
    
    # Calculate the money flow ratio with dynamic window
    money_flow_ratio = (positive_money_flow.rolling(window=window_size, min_periods=10).sum() / 
                        negative_money_flow.rolling(window=window_size, min_periods=10).sum().fillna(1))
    
    # Incorporate the daily range as a measure of volatility
    volatility_factor = 1 / (daily_range + 1e-7)
    
    # Calculate a momentum factor using logarithmic returns
    momentum_factor = np.log(df['close']) - np.log(df['close']).rolling(window=50, min_periods=25).mean()
    
    # Consider market trend by adding a trend-following factor
    trend_factor = df['close'].pct_change(periods=20).rolling(window=10, min_periods=5).mean()
    
    # Additional factor: Volume-adjusted momentum
    volume_adjusted_momentum = momentum_factor * (df['volume'] / df['volume'].rolling(window=10, min_periods=5).mean())
    
    # Add market sentiment (assuming a column 'sentiment' is available, e.g., from news or social media analysis)
    sentiment_factor = df.get('sentiment', 1.0)
    
    # Add macroeconomic indicator (assuming a column 'macro_indicator' is available, e.g., GDP growth rate)
    macro_factor = df.get('macro_indicator', 1.0)
    
    # Combine these factors in a heuristic way to create a novel alpha factor
    alpha_factor = (weighted_close * money_flow_ratio * volatility_factor * volume_adjusted_momentum * trend_factor * sentiment_factor * macro_factor).fillna(0)
    
    return alpha_factor
