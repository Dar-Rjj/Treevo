```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    co_diff = df['close'] - df['open']
    
    # Calculate the average of high and low prices
    avg_hl = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of the volume to the amount
    vol_to_amt_ratio = df['volume'] / (df['amount'] + 1e-7)
    
    # Calculate the trend strength using a moving average
    trend_strength = df['close'].rolling(window=10).mean() - df['close'].rolling(window=30).mean()
    
    # Novel alpha factor: (Close-Open difference * Volume-to-Amount Ratio * Trend Strength) / Average of High and Low
    novel_factor = (co_diff * vol_to_amt_ratio * trend_strength) / (avg_hl + 1e-7)
    
    # Smooth the factor with a moving average for stability
    smoothed_factor = novel_factor.rolling(window=5).mean().fillna(0)
    
    return smoothed_factor
```
