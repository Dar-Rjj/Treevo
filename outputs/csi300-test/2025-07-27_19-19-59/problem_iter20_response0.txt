```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily range to capture volatility
    daily_range = df['high'] - df['low']
    
    # Calculate a weighted close based on volume for a more accurate price movement indicator
    weighted_close = (df['close'] * df['volume']) / df['volume'].rolling(window=10).mean()
    
    # Calculate typical price for money flow analysis
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate money flow
    money_flow = typical_price * df['volume']
    
    # Calculate positive and negative money flow
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Calculate the money flow ratio for buying and selling pressure with an adaptive window
    money_flow_ratio = (positive_money_flow.rolling(window=20, min_periods=10).sum() / 
                        negative_money_flow.rolling(window=20, min_periods=10).sum().fillna(1))
    
    # Incorporate the daily range as a measure of volatility
    volatility_factor = 1 / (daily_range + 1e-7)
    
    # Calculate a momentum factor using the difference between the current close and the adaptive moving average
    short_window = 5
    long_window = 50
    momentum_factor = df['close'] - df['close'].ewm(span=long_window, adjust=False).mean()
    
    # Consider market trend by adding a more sophisticated trend-following factor
    trend_factor = df['close'].pct_change(periods=20).ewm(span=10, adjust=False).mean()
    
    # Additional factor: Volume-adjusted momentum
    volume_adjusted_momentum = momentum_factor * (df['volume'] / df['volume'].rolling(window=10, min_periods=5).mean())
    
    # Incorporate market sentiment using a simple RSI (Relative Strength Index)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine these factors in a heuristic way to create a novel alpha factor
    alpha_factor = (weighted_close * money_flow_ratio * volatility_factor * volume_adjusted_momentum * trend_factor * rsi).fillna(0)
    
    return alpha_factor
```
