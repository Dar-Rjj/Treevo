```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices
    price_diff = df['close'] - df['open']
    
    # Calculate the weighted average of volume and amount to capture the flow intensity
    volume_amount_weighted = (df['volume'] * df['amount']) / (df['volume'] + 1e-7)
    
    # Combine the two signals, capturing the momentum and flow intensity together
    factor = price_diff * volume_amount_weighted
    
    return factor
```
