import pandas as pd

def heuristics_v2(df):
    # Long-term momentum: 6-month return
    long_momentum = df['close'].pct_change(120)
    
    # Medium-term volatility trend: 30-day moving average of standard deviation over the last 60 days
    vol_trend_60 = df['close'].pct_change().rolling(window=60).std().rolling(window=30).mean()
    
    # Short-term liquidity shift: Change in the 5-day minimum volume
    min_vol_5 = df['volume'].rolling(window=5).min()
    liquidity_shift = (min_vol_5 - min_vol_5.shift(1)) / min_vol_5.shift(1)
    
    # Composite heuristic score
    heuristics_matrix = 0.4 * long_momentum + 0.4 * (1/vol_trend_60) + 0.2 * liquidity_shift
    
    return heuristics_matrix
