def heuristics_v2(df):
    # Mean reversion factor: 30-day return
    mean_reversion = -df['close'].pct_change(30)
    
    # Volatility factor: Average intraday range over the last 60 days
    avg_intraday_range = (df['high'] - df['low']).rolling(window=60).mean()
    
    # Liquidity factor: 10-day change in volume
    vol_change_10 = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10)
    
    # Composite heuristic score
    heuristics_matrix = 0.5 * mean_reversion + 0.3 * (1/avg_intraday_range) + 0.2 * vol_change_10
    
    return heuristics_matrix
