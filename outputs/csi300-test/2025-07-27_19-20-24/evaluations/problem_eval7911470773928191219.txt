import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate RSI (Relative Strength Index) for 14 days
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate ATR (Average True Range) for 14 days
    tr = pd.DataFrame(index=df.index, columns=['tr'])
    tr['tr'] = np.maximum(np.maximum((df['high'] - df['low']), 
                          abs(df['high'] - df['close'].shift(1))), 
                          abs(df['low'] - df['close'].shift(1)))
    atr = tr['tr'].rolling(window=14).mean()
    
    # Calculate the ratio of current volume to its 30-day minimum
    min_volume_30 = df['volume'].rolling(window=30).min()
    vol_ratio = df['volume'] / min_volume_30
    
    # Create a heuristic matrix using a weighted sum of the indicators
    heuristics_matrix = (rsi * 0.4 + atr * 0.3 + vol_ratio * 0.3)
    
    return heuristics_matrix
