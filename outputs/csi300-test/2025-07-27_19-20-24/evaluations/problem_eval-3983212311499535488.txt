def heuristics_v2(df):
    # Momentum factor: 20-day return
    momentum = df['close'].pct_change(20)
    
    # Volatility factor: Standard deviation of daily returns over the last 30 days
    daily_returns = df['close'].pct_change()
    std_dev_30 = daily_returns.rolling(window=30).std()
    
    # Liquidity factor: 5-day change in volume
    volume_change_5 = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Composite heuristic score
    heuristics_matrix = 0.5 * momentum + 0.3 * (1/std_dev_30) + 0.2 * volume_change_5
    
    return heuristics_matrix
