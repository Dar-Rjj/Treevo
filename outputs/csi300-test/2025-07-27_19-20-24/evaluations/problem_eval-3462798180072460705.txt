def heuristics_v2(df):
    # Momentum factor: 30-day return
    momentum = df['close'].pct_change(30)
    
    # Volatility factor: Standard deviation of daily returns over the last 60 days
    daily_returns = df['close'].pct_change()
    vol_60 = daily_returns.rolling(window=60).std()
    
    # Liquidity factor: 10-day change in volume
    vol_change_10 = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10)
    
    # Composite heuristic score
    heuristics_matrix = 0.4 * momentum + 0.4 * (1/vol_60) + 0.2 * vol_change_10
    
    return heuristics_matrix
