def heuristics_v2(df):
    # Momentum factor: 20-day return
    momentum = df['close'].pct_change(20)
    
    # Volatility factor: Average True Range over the last 50 days
    true_range = df['high'] - df['low']
    atr_50 = true_range.rolling(window=50).mean()
    
    # Liquidity factor: 5-day change in volume
    volume_change_5 = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Composite heuristic score
    heuristics_matrix = 0.4 * momentum + 0.4 * (1/atr_50) + 0.2 * volume_change_5
    
    return heuristics_matrix
