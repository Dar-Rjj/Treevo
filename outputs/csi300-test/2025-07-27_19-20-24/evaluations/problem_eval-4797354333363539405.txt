def heuristics_v2(df):
    # Momentum factor: 50-day return
    momentum = df['close'].pct_change(50)
    
    # Volatility factor: Coefficient of variation (standard deviation/mean) of intraday range over the last 20 days
    intraday_range = (df['high'] - df['low'])
    vol_coeff_variation = intraday_range.rolling(window=20).std() / intraday_range.rolling(window=20).mean()
    
    # Liquidity factor: 10-day percentage change in volume
    vol_change_10 = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10)
    
    # Composite heuristic score
    heuristics_matrix = 0.4 * momentum + 0.4 * (1/vol_coeff_variation) + 0.2 * vol_change_10
    
    return heuristics_matrix
