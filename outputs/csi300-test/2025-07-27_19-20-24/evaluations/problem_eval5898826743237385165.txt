def heuristics_v2(df):
    # Mean reversion factor: 60-day return
    mean_reversion = -df['close'].pct_change(60)
    
    # Volatility factor: Average intraday range over the last 30 days
    avg_intraday_range = (df['high'] - df['low']).rolling(window=30).mean()
    
    # Liquidity factor: 5-day change in volume
    vol_change_5 = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Composite heuristic score
    heuristics_matrix = 0.4 * mean_reversion + 0.4 * (1/avg_intraday_range) + 0.2 * vol_change_5
    
    return heuristics_matrix
