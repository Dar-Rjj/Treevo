def heuristics_v2(df):
    # Momentum factor: 10-day return
    momentum = df['close'].pct_change(10)
    
    # Volatility factor: Average True Range over the last 30 days
    true_range = df['high'] - df['low']
    atr_30 = true_range.rolling(window=30).mean()
    
    # Liquidity factor: 5-day change in volume
    vol_change_5 = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Composite heuristic score
    heuristics_matrix = 0.4 * momentum + 0.4 * (1/atr_30) + 0.2 * vol_change_5
    
    return heuristics_matrix
