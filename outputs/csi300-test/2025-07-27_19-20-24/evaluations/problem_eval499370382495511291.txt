def heuristics_v2(df):
    # Calculate the logarithmic return for 15 days
    log_return_15 = (df['close'] / df['close'].shift(15)).apply(np.log)
    
    # Calculate the 30-day moving average of the closing price
    ma_30 = df['close'].rolling(window=30).mean()
    
    # Calculate the ratio of the 30-day moving average to the current closing price
    ma_ratio = ma_30 / df['close']
    
    # Create a composite heuristic
    heuristics_matrix = log_return_15 + ma_ratio
    
    return heuristics_matrix
