def heuristics_v2(df):
    # Mean Reversion factor: 20-day percentage change from the 20-day moving average
    ma_20 = df['close'].rolling(window=20).mean()
    mean_reversion = (df['close'] - ma_20) / ma_20
    
    # Trend Strength factor: 50-day directional movement indicator
    trend_strength = df['close'].rolling(window=50).apply(lambda x: (x[-1] - x[0]) / x[0])
    
    # Turnover Rate factor: 7-day average of the daily ratio of volume to amount
    turnover_rate_7 = (df['volume'] / df['amount']).rolling(window=7).mean()
    
    # Composite heuristic score
    heuristics_matrix = 0.4 * mean_reversion + 0.4 * trend_strength + 0.2 * turnover_rate_7
    
    return heuristics_matrix
