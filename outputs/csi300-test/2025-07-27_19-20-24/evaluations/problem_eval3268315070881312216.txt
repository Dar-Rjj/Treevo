def heuristics_v2(df):
    # Calculate rate of change (Momentum Indicator) for 15 days
    roc_15 = df['close'].pct_change(periods=15)
    
    # Calculate standard deviation (Volatility Indicator) for 30 days
    std_30 = df['close'].rolling(window=30).std()
    
    # Calculate percentage change in volume (Liquidity Indicator) for 10 days
    vol_change_10 = df['volume'].pct_change(periods=10)
    
    # Create a composite heuristic with specific weights
    heuristics_matrix = (0.4 * roc_15 + 0.3 * std_30 + 0.3 * vol_change_10)
    
    return heuristics_matrix
