def heuristics_v2(df):
    # Momentum factor: 30-day return
    momentum = df['close'].pct_change(30)
    
    # Volatility factor: Standard deviation of daily returns over the last 60 days
    daily_returns = df['close'].pct_change()
    std_dev_60 = daily_returns.rolling(window=60).std()
    
    # Liquidity factor: 10-day change in amount
    amount_change_10 = (df['amount'] - df['amount'].shift(10)) / df['amount'].shift(10)
    
    # Composite heuristic score
    heuristics_matrix = 0.4 * momentum + 0.4 * (1/std_dev_60) + 0.2 * amount_change_10
    
    return heuristics_matrix
