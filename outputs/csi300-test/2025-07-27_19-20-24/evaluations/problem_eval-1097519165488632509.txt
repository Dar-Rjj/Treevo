def heuristics_v2(df):
    daily_return = df['close'].pct_change()
    ema_daily_return = daily_return.ewm(span=20, adjust=False).mean()
    vol_std_20 = df['volume'].rolling(window=20).std()
    volume_ratio = df['volume'] / vol_std_20
    heuristics_matrix = ema_daily_return * volume_ratio
    return heuristics_matrix
