def heuristics_v2(df):
    # Momentum factor: 20-day return
    momentum = df['close'].pct_change(20)
    
    # Volatility factor: Average of the absolute value of daily returns over the last 40 days
    daily_returns = df['close'].pct_change()
    avg_abs_returns_40 = daily_returns.abs().rolling(window=40).mean()
    
    # Liquidity factor: 5-day change in volume
    volume_change_5 = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Composite heuristic score
    heuristics_matrix = 0.5 * momentum + 0.3 * (1/avg_abs_returns_40) + 0.2 * volume_change_5
    
    return heuristics_matrix
