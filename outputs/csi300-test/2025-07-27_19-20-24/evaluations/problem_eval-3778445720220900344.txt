def heuristics_v2(df):
    # Calculate 30-day moving average of the daily returns
    daily_return = df['close'].pct_change()
    ma_daily_return_30 = daily_return.rolling(window=30).mean()
    
    # Calculate 14-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate 5-day moving average of the RSI
    ma_rsi_5 = rsi.rolling(window=5).mean()
    
    # Calculate 20-day EMA of the close price
    ema_close_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Composite heuristic score
    heuristics_matrix = 0.3 * ma_daily_return_30 + 0.4 * (rsi / ma_rsi_5) + 0.3 * (df['close'] - ema_close_20)
    
    return heuristics_matrix
