def heuristics_v2(df):
    daily_return = df['close'].pct_change()
    highest_high = df['high'].rolling(window=20).max()
    performance_ratio = daily_return / highest_high
    
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    liquidity_ratio = df['volume'] / avg_volume_10
    
    heuristics_matrix = 0.7 * performance_ratio + 0.3 * liquidity_ratio
    return heuristics_matrix
