import pandas as pd

def heuristics_v2(df):
    # Relative strength: 60-day return of the stock divided by the 60-day return of the sector
    stock_return_60 = df['close'].pct_change(60)
    sector_return_60 = df['sector_close'].pct_change(60)  # Assuming 'sector_close' column is available in the DataFrame
    relative_strength = stock_return_60 / sector_return_60
    
    # Price stability: 10-day coefficient of variation (std/mean) of the closing prices
    price_stability = df['close'].rolling(window=10).std() / df['close'].rolling(window=10).mean()
    
    # Volume momentum: 5-day return of the trading volume
    volume_momentum = df['volume'].pct_change(5)
    
    # Composite heuristic score
    heuristics_matrix = 0.4 * relative_strength + 0.3 * (1 / price_stability) + 0.3 * volume_momentum
    
    return heuristics_matrix
