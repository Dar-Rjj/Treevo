def heuristics_v2(df):
    # Intraday strength factor: 5-day average of the ratio of closing price to opening price
    intra_strength = (df['close'] / df['open']).rolling(window=5).mean()
    
    # Long-term interest factor: 90-day EMA of trading volume
    ema_90_volume = df['volume'].ewm(span=90, adjust=False).mean()
    
    # Composite heuristic score
    heuristics_matrix = 0.6 * intra_strength + 0.4 * (ema_90_volume / df['volume'])
    
    return heuristics_matrix
