import pandas as pd

def heuristics_v2(df):
    # Long-term P/E ratio change: 12-month percentage change in P/E ratio
    pe_ratio = df['close'] / df['eps']
    long_pe_change = pe_ratio.pct_change(240)
    
    # Medium-term price volatility: 60-day standard deviation of daily returns
    price_volatility = df['close'].pct_change().rolling(window=60).std()
    
    # Short-term trading volume momentum: 5-day moving average of daily volume percentage change
    volume_momentum = df['volume'].pct_change().rolling(window=5).mean()
    
    # Composite heuristic score
    heuristics_matrix = 0.3 * long_pe_change + 0.4 * (1/price_volatility) + 0.3 * volume_momentum
    
    return heuristics_matrix
