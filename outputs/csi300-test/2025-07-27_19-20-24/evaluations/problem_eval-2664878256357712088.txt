def heuristics_v2(df):
    # Calculate the logarithmic returns (Momentum Indicator) for 10 days
    log_returns_10 = (df['close'] / df['close'].shift(10)).apply(np.log)
    
    # Calculate the coefficient of variation (Risk-Adjusted Volatility Indicator) for 20 days
    avg_returns_20 = df['close'].pct_change().rolling(window=20).mean()
    std_20 = df['close'].pct_change().rolling(window=20).std()
    coef_var_20 = std_20 / avg_returns_20
    
    # Calculate the rate of change in volume (Liquidity Indicator) for 5 days
    vol_rate_of_change_5 = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Create a composite heuristic
    heuristics_matrix = (log_returns_10 + 1 / coef_var_20 + vol_rate_of_change_5) / 3
    
    return heuristics_matrix
