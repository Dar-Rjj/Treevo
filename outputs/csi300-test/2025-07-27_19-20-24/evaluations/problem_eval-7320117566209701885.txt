def heuristics_v2(df):
    # Momentum factor: 20-day return
    momentum = df['close'].pct_change(20)
    
    # Volatility factor: Average of absolute daily returns over the last 50 days
    daily_returns_abs = df['close'].pct_change().abs()
    avg_ret_50 = daily_returns_abs.rolling(window=50).mean()
    
    # Liquidity factor: 15-day change in volume
    vol_change_15 = (df['volume'] - df['volume'].shift(15)) / df['volume'].shift(15)
    
    # Composite heuristic score
    heuristics_matrix = 0.3 * momentum + 0.5 * (1/avg_ret_50) + 0.2 * vol_change_15
    
    return heuristics_matrix
