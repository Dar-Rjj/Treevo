def heuristics_v2(df):
    # Medium-term momentum: 6-month return
    med_momentum = df['close'].pct_change(120)
    
    # Short-term volatility trend: 30-day moving average of standard deviation over the last 90 days
    vol_trend_90 = df['close'].pct_change().rolling(window=90).std().rolling(window=30).mean()
    
    # Very short-term liquidity shift: Change in the 5-day average volume
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    liquidity_shift = (avg_vol_5 - avg_vol_5.shift(1)) / avg_vol_5.shift(1)
    
    # Composite heuristic score
    heuristics_matrix = 0.4 * med_momentum + 0.4 * (1/vol_trend_90) + 0.2 * liquidity_shift
    
    return heuristics_matrix
