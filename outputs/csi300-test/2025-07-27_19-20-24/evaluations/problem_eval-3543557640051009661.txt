import pandas as pd

def heuristics_v2(df):
    close_momentum = df['close'].pct_change(periods=20)
    volume_avg = df['volume'].rolling(window=20).mean()
    volume_diff = (df['volume'] - volume_avg) / volume_avg
    heuristics_matrix = 0.5 * close_momentum + 0.5 * volume_diff
    return heuristics_matrix
