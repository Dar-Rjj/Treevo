def heuristics_v2(df):
    # Momentum factor: 20-day return
    momentum = df['close'].pct_change(20)
    
    # Stability factor: Average of daily returns over the last 50 days
    daily_returns = df['close'].pct_change()
    avg_return_50 = daily_returns.rolling(window=50).mean()
    
    # Liquidity factor: 5-day change in volume
    vol_change_5 = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Composite heuristic score
    heuristics_matrix = 0.5 * momentum + 0.3 * avg_return_50 + 0.2 * vol_change_5
    
    return heuristics_matrix
