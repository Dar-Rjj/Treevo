def heuristics_v2(df):
    # Exponential Moving Average (EMA) of daily returns over the last 20 days
    daily_returns = df['close'].pct_change()
    ema_20 = daily_returns.ewm(span=20, adjust=False).mean()
    
    # Volume Momentum: 30-day to 60-day volume ratio
    volume_30 = df['volume'].rolling(window=30).mean()
    volume_60 = df['volume'].rolling(window=60).mean()
    volume_ratio = volume_30 / volume_60
    
    # Composite heuristic score
    heuristics_matrix = 0.5 * ema_20 + 0.5 * volume_ratio
    
    return heuristics_matrix
