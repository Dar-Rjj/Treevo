import pandas as pd

def heuristics_v2(df):
    # 60-day Sharpe Ratio: Mean return over 60 days divided by the standard deviation of returns over 60 days
    daily_returns = df['close'].pct_change()
    sharpe_ratio_60 = (daily_returns.rolling(window=60).mean()) / (daily_returns.rolling(window=60).std())
    
    # 30-day Price Range: High minus Low over 30 days
    price_range_30 = (df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()) / df['close']
    
    # 5-day Exponential Moving Average of Volume
    ema_vol_5 = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Composite heuristic score
    heuristics_matrix = 0.4 * sharpe_ratio_60 + 0.35 * price_range_30 + 0.25 * (ema_vol_5 / df['volume'])
    
    return heuristics_matrix
