import pandas as pd

def heuristics_v2(df):
    # Long-term momentum: 12-month return
    long_momentum = df['close'].pct_change(240)
    
    # Medium-term volatility trend: 90-day moving average of standard deviation over the last 180 days
    vol_trend_180 = df['close'].pct_change().rolling(window=180).std().rolling(window=90).mean()
    
    # Short-term liquidity shift: Change in the 10-day average volume
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    liquidity_shift = (avg_vol_10 - avg_vol_10.shift(1)) / avg_vol_10.shift(1)
    
    # Composite heuristic score
    heuristics_matrix = 0.5 * long_momentum + 0.3 * (1/vol_trend_180) + 0.2 * liquidity_shift
    
    return heuristics_matrix
