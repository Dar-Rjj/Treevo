    # Calculate simple moving average for closing prices
    sma_close = df['close'].rolling(window=10).mean()
    
    # Calculate weighted moving average for closing prices
    weights = [0.1, 0.2, 0.3, 0.4]
    wma_close = df['close'].rolling(window=len(weights)).apply(lambda x: (x * pd.Series(weights, index=x.index)).sum(), raw=False)
    
    # Calculate a simple momentum indicator
    momentum = df['close'] - df['close'].shift(5)
    
    # Volume trend over the last 10 days
    volume_trend = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10)
    
    # Combine the factors into a single heuristics matrix
    heuristics_matrix = (sma_close + wma_close + momentum) * volume_trend
    
    return heuristics_matrix
