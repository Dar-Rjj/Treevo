def heuristics_v2(df):
    # Momentum factor: 20-day return
    momentum = df['close'].pct_change(20)
    
    # Volatility factor: Mean absolute deviation of daily returns over the last 50 days
    daily_returns = df['close'].pct_change()
    mad_50 = daily_returns.rolling(window=50).apply(lambda x: np.mean(np.abs(x - np.mean(x))), raw=True)
    
    # Liquidity factor: 15-day change in volume
    vol_change_15 = (df['volume'] - df['volume'].shift(15)) / df['volume'].shift(15)
    
    # Composite heuristic score
    heuristics_matrix = 0.5 * momentum + 0.3 * (1/mad_50) + 0.2 * vol_change_15
    
    return heuristics_matrix
