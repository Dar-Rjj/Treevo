def heuristics_v2(df):
    daily_return = df['close'].pct_change()
    diff_high_low = (df['high'] - df['low']).abs()
    diff_close_open = (df['close'] - df['open']).abs()
    avg_diff = (diff_high_low + diff_close_open).rolling(window=20).mean()
    heuristics_matrix = daily_return / avg_diff
    return heuristics_matrix
