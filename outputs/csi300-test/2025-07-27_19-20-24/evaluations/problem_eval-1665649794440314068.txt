import pandas as pd

def heuristics_v2(df):
    # Calculate rate of change (Momentum Indicator) for 10 days
    roc_10 = df['close'].pct_change(periods=10)
    
    # Calculate mean absolute deviation (Volatility Indicator) for 20 days
    mad_20 = df['close'].rolling(window=20).apply(lambda x: (x - x.mean()).abs().mean())
    
    # Calculate the ratio of current volume to the average volume over 5 days (Liquidity Indicator)
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    vol_ratio_5 = df['volume'] / avg_vol_5
    
    # Create a composite heuristic
    heuristics_matrix = (roc_10 + mad_20 + vol_ratio_5) / 3
    
    return heuristics_matrix
