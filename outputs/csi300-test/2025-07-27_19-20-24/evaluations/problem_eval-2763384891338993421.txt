import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 10-day EMA of closing prices
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Ratio of 5-day EMA to 10-day EMA
    ema_ratio = ema_5 / ema_10
    
    # 30-day percentage change in daily trading amount
    amount_change = df['amount'].pct_change(30)
    
    # Composite heuristic score
    heuristics_matrix = ema_ratio + amount_change
    
    return heuristics_matrix
