def heuristics_v2(df):
    # Trend factor: 20-day MACD
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = short_ema - long_ema
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    trend_factor = macd_line - signal_line
    
    # Volatility factor: 90-day standard deviation of daily returns
    daily_returns = df['close'].pct_change()
    vol_90 = daily_returns.rolling(window=90).std()
    
    # Momentum factor: 5-day RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=5).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=5).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Composite heuristic score
    heuristics_matrix = 0.4 * trend_factor + 0.3 * (1/vol_90) + 0.3 * rsi
    
    return heuristics_matrix
