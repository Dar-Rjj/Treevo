def heuristics_v2(df):
    # Momentum factor: RSI over the last 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volatility factor: Average True Range (ATR) over the last 14 days
    tr = pd.DataFrame({'H-L': df['high'] - df['low'], 
                       'H-PC': abs(df['high'] - df['close'].shift(1)), 
                       'L-PC': abs(df['low'] - df['close'].shift(1))})
    atr = tr.max(axis=1).rolling(window=14).mean()
    
    # Liquidity factor: Current volume divided by the 20-day moving average of volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume_20
    
    # Composite heuristic score
    heuristics_matrix = 0.5 * rsi + 0.3 * (1 / atr) + 0.2 * volume_ratio
    
    return heuristics_matrix
