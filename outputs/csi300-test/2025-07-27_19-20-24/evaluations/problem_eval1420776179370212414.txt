import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change over 5, 10, and 20 days
    roc_5 = df['close'].pct_change(5)
    roc_10 = df['close'].pct_change(10)
    roc_20 = df['close'].pct_change(20)
    
    # Calculate the volume-adjusted price movement
    adjusted_price_movement = (df['close'] - df['open']) / df['volume']
    
    # Combine the factors into a composite heuristic
    heuristics_matrix = (roc_5 + roc_10 + roc_20) * adjusted_price_movement
    
    return heuristics_matrix
