import pandas as pd

def heuristics_v2(df):
    # Calculate rate of change (Momentum Indicator) for 10 days
    roc_10 = df['close'].pct_change(periods=10)
    
    # Calculate standard deviation (Volatility Indicator) for 20 days
    std_20 = df['close'].rolling(window=20).std()
    
    # Calculate percentage change in volume (Liquidity Indicator) for 5 days
    vol_change_5 = df['volume'].pct_change(periods=5)
    
    # Create a composite heuristic
    heuristics_matrix = (roc_10 + std_20 + vol_change_5) / 3
    
    return heuristics_matrix
