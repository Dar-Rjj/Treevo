def heuristics_v2(df):
    # Calculate rate of change (Momentum Indicator) for 15 days
    roc_15 = df['close'].pct_change(periods=15)
    
    # Calculate standard deviation (Volatility Indicator) for 30 days
    std_30 = df['close'].rolling(window=30).std()
    
    # Calculate percentage change in volume (Liquidity Indicator) for 7 days
    vol_change_7 = df['volume'].pct_change(periods=7)
    
    # Create a composite heuristic
    heuristics_matrix = (roc_15 + std_30 + vol_change_7) / 3
    
    return heuristics_matrix
