import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of closing price to the 30-day average high price
    avg_high_30 = df['high'].rolling(window=30).mean()
    close_to_avg_high = df['close'] / avg_high_30
    
    # Calculate the Relative Strength Index (RSI) over the past 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the change in trading amount compared to the 5-day moving average
    avg_amount_5 = df['amount'].rolling(window=5).mean()
    amount_change = (df['amount'] - avg_amount_5) / avg_amount_5
    
    # Combine the indicators into a heuristics matrix
    heuristics_matrix = 0.4 * close_to_avg_high + 0.4 * rsi + 0.2 * amount_change
    
    return heuristics_matrix
