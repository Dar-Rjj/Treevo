import pandas as pd

def heuristics_v2(df):
    # Calculate 50-day weighted moving average of close price
    wma_50 = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate 20-day exponential moving average of volume
    ema_volume_20 = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Momentum factor: difference between close price and 50-day WMA
    momentum_factor = df['close'] - wma_50
    
    # Volume support factor: recent EMA of volume
    volume_support = ema_volume_20
    
    # Composite heuristic score
    heuristics_matrix = 0.7 * momentum_factor + 0.3 * volume_support
    
    return heuristics_matrix
