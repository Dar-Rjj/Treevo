import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Exponential Weighted Squared Returns Over N Days
    df['squared_return'] = df['daily_return'] ** 2
    df['ewma_squared_returns'] = df['squared_return'].ewm(span=N).mean()
    
    # Volume Adjusted Close Price
    df['moving_avg_close'] = df['close'].rolling(window=M).mean()
    df['volume_adjusted_close'] = df['moving_avg_close'] * np.log(df['volume'])
    
    # High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Integrated Momentum
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    df['momentum'] = df['close'] - df['20_day_avg_close']
    
    # High-Low Momentum
    df['5_day_avg_high'] = df['high'].rolling(window=5).mean()
    df['5_day_avg_low'] = df['low'].rolling(window=5).mean()
    df['high_momentum'] = df['high'] - df['5_day_avg_high']
    df['low_momentum'] = df['low'] - df['5_day_avg_low']
    df['integrated_momentum'] = df['momentum'] + df['high_momentum'] + df['low_momentum']
    
    # Volume Adjusted Daily Return
    df['volume_adjusted_daily_return'] = df['volume'] * df['daily_return']
    
    # Intraday Range to Volume Ratio
    df['range_to_volume_ratio'] = df['intraday_range'] / df['volume']
    
    # Closing Price to High-Low Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['high_low_diff']
    
    # Intraday Volatility
    df['intraday_volatility'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Volume Trend
    df['3_day_avg_volume'] = df['volume'].rolling(window=3).mean()
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['volume_trend'] = df['3_day_avg_volume'] - df['7_day_avg_volume']
    
    # Combined Indicator
    df['combined_indicator'] = (df['ewma_squared_returns'] * df['volume_adjusted_close']) / df['high_low_diff']
    
    # Rolling Window Summations
    df['sum_volume_adjusted_daily_return'] = df['volume_adjusted_daily_return'].rolling(window=5).sum()
    df['sum_range_to_volume_ratio'] = df['range_to_volume_ratio'].rolling(window=5).sum()
    df['sum_close_to_range_ratio'] = df['close_to_range_ratio'].rolling(window=5).sum()
    
    # Combined Integrated Momentum, Intraday Volatility, and Volume Trend
    df['momentum_volatility_trend'] = (-1 * df['integrated_momentum'] + df['intraday_volatility']) * df['volume_trend']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['combined_indicator'] 
                          + df['sum_volume_adjusted_daily_return'] 
                          + df['sum_range_to_volume_ratio'] 
                          + df['sum_close_to_range_ratio']) * df['momentum_volatility_trend']
    
    return df['alpha_factor'].dropna()
