```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement
    df['DailyRange'] = df['high'] - df['low']
    df['Midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Multi-Day Average Midpoint (5-day window)
    df['AverageMidpoint'] = df['Midpoint'].rolling(window=5).mean()
    
    # Compute Volume Adjusted Momentum
    df['CloseToCloseReturn'] = df['close'].pct_change()
    df['VolumeAdjustedMomentum'] = df['CloseToCloseReturn'] * df['volume']
    
    # Calculate Enhanced Price Envelopment
    df['MultiDayRange'] = df['DailyRange'].rolling(window=5).sum()
    df['MultiDayVolume'] = df['volume'].rolling(window=5).sum()
    df['BaseEnvelopment'] = df['MultiDayRange'] / df['MultiDayVolume']
    df['WeightedSum'] = (df['high'] + df['low']) / 2
    df['EnhancedPriceEnvelopment'] = df['BaseEnvelopment'] + df['WeightedSum']
    
    # Combine Factors
    df['AlphaFactor'] = df['VolumeAdjustedMomentum'] * df['EnhancedPriceEnvelopment']
    
    return df['AlphaFactor']
```
