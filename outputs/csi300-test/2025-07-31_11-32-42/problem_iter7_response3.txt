```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Factors
    df['1d_price_momentum'] = df['close'].pct_change()
    df['5d_price_momentum'] = df['close'].pct_change(periods=5)
    df['20d_price_momentum'] = df['close'].pct_change(periods=20)
    
    # Calculate Volume Activity Indicators
    df['1d_volume_diff'] = df['volume'].diff()
    df['5d_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['20d_avg_volume'] = df['volume'].rolling(window=20).mean()
    
    # Calculate Daily Momentum
    df['20d_avg_close'] = df['close'].rolling(window=20).mean()
    df['daily_momentum'] = df['close'] - df['20d_avg_close']
    
    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Combine Momentum and Volume Activities
    df['1d_mom_vol'] = df['1d_price_momentum'] * df['1d_volume_diff']
    df['5d_mom_vol'] = df['5d_price_momentum'] * df['5d_avg_volume']
    df['20d_mom_vol'] = df['20d_price_momentum'] * df['20d_avg_volume']
    
    # Combine All Factors
    df['combined_factors'] = (df['1d_mom_vol'] + 
                              df['5d_mom_vol'] + 
                              df['20d_mom_vol'] - 
                              df['daily_momentum'] + 
                              df['intraday_volatility'])
    
    return df['combined_factors']
```
