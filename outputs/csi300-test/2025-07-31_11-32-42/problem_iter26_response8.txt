```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Integrated Momentum
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    df['integrated_momentum'] = df['close'] - df['20_day_avg_close']
    
    # Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Open-to-Close Movement
    df['open_to_close_movement'] = df['close'] - df['open']
    
    # Integrated Intraday and Open-to-Close Metric
    df['integrated_metric'] = df['intraday_volatility'] / df['open_to_close_movement']
    
    # Volume Trend
    df['3_day_avg_volume'] = df['volume'].rolling(window=3).mean()
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['volume_trend'] = df['3_day_avg_volume'] - df['7_day_avg_volume']
    
    # Advanced Volume Impact
    df['advanced_volume_impact'] = np.cbrt(df['volume']) * df['integrated_metric']
    
    # Adjusted Price Range as a Stability Measure
    df['adjusted_price_range'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    df['stability_adjusted_impact'] = df['advanced_volume_impact'] / df['adjusted_price_range']
    
    # Dynamic Price Momentum
    df['5_day_avg_close'] = df['close'].rolling(window=5).mean()
    df['5_day_close_range'] = df['close'].rolling(window=5).max() - df['close'].rolling(window=5).min()
    df['dynamic_momentum_5'] = (df['close'] - df['5_day_avg_close']) / df['5_day_close_range']
    
    # Additional Momentum Indicators
    df['10_day_avg_close'] = df['close'].rolling(window=10).mean()
    df['10_day_close_range'] = df['close'].rolling(window=10).max() - df['close'].rolling(window=10).min()
    df['dynamic_momentum_10'] = (df['close'] - df['10_day_avg_close']) / df['10_day_close_range']
    
    # Integrate Momentum Indicators
    df['integrated_momentum_indicators'] = (df['dynamic_momentum_5'] + df['dynamic_momentum_10']) / 2
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['integrated_momentum'] + 
        df['intraday_volatility'] + 
        df['volume_trend']
    ) * df['stability_adjusted_impact'] / df['adjusted_price_range'] + df['integrated_momentum_indicators'] - (df['close'].pct_change(periods=5) - df['close'].pct_change(periods=10))
    
    return df['final_alpha_factor'].dropna()
```
