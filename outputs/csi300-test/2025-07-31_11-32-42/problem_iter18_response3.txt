```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Ratio
    intraday_high_low_ratio = df['high'] / df['low']
    
    # Calculate Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Opening Gaps
    combined_metric = intraday_high_low_ratio * open_close_return
    
    # Adjust by Volume
    volume_adjusted_combined_metric = combined_metric * df['volume']
    
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Daily Returns
    daily_returns = df['close'] - df['open']
    
    # Volume Adjusted Daily Return
    volume_adjusted_daily_return = daily_returns * df['volume']
    
    # Intraday Range to Volume Ratio
    intraday_range_to_volume_ratio = intraday_high_low_range / df['volume']
    
    # Closing Price to High-Low Range Ratio
    closing_price_to_high_low_range_ratio = df['close'] / intraday_high_low_range
    
    # Combine Intraday Factors over a 5-day rolling window
    sum_volume_adjusted_daily_return = volume_adjusted_daily_return.rolling(window=5).sum()
    sum_intraday_range_to_volume_ratio = intraday_range_to_volume_ratio.rolling(window=5).sum()
    sum_closing_price_to_high_low_range_ratio = closing_price_to_high_low_range_ratio.rolling(window=5).sum()
    
    # Calculate Integrated Momentum
    avg_20_day_close = df['close'].rolling(window=20).mean()
    momentum_20_day = df['close'] - avg_20_day_close
    high_momentum = df['high'] - df['high'].rolling(window=5).mean()
    low_momentum = df['low'] - df['low'].rolling(window=5).mean()
    integrated_momentum = high_momentum + low_momentum
    
    # Combine Integrated Momentum and Intraday Volatility
    combined_momentum_and_volatility = -momentum_20_day + integrated_momentum
    
    # Introduce New Factor: Price to Volume Elasticity
    avg_5_day_close = df['close'].rolling(window=5).mean()
    avg_5_day_volume = df['volume'].rolling(window=5).mean()
    price_to_volume_elasticity = avg_5_day_close / avg_5_day_volume
    
    # Measure Volume Trend
    avg_3_day_volume = df['volume'].rolling(window=3).mean()
    avg_7_day_volume = df['volume'].rolling(window=7).mean()
    volume_trend = avg_3_day_volume - avg_7_day_volume
    
    # Introduce a New Factor: Price-Earnings to Volume Ratio
    avg_5_day_amount = df['amount'].rolling(window=5).mean()
    price_earnings_to_volume_ratio = avg_5_day_close / avg_5_day_amount
    
    # Final Alpha Factor
    final_alpha_factor = (
        sum_volume_adjusted_daily_return +
        sum_intraday_range_to_volume_ratio +
        sum_closing_price_to_high_low_range_ratio +
        volume_adjusted_combined_metric +
        combined_momentum_and_volatility * volume_trend +
        price_to_volume_elasticity +
        price_earnings_to_volume_ratio
    )
    
    return final_alpha_factor
```
