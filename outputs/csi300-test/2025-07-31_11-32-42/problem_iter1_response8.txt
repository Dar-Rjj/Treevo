```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Momentum
    high_5_day_ma = df['high'].rolling(window=5).mean()
    low_5_day_ma = df['low'].rolling(window=5).mean()
    
    high_momentum = df['high'] - high_5_day_ma
    low_momentum = df['low'] - low_5_day_ma
    
    high_low_momentum = high_momentum + low_momentum
    
    # Measure Volume Trend
    volume_3_day_ma = df['volume'].rolling(window=3).mean()
    volume_7_day_ma = df['volume'].rolling(window=7).mean()
    
    volume_trend = volume_3_day_ma - volume_7_day_ma
    
    # Combine Momentum and Volume Trend
    factor = high_low_momentum * volume_trend
    
    return factor
```
