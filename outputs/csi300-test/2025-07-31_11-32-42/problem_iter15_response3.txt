```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_volatility / (open_to_close_movement + 1e-6)  # Adding small value to avoid division by zero
    
    # Incorporate Advanced Volume Impact
    volume_impact = np.cbrt(df['volume'])
    integrated_volume_adjusted = integrated_metric * volume_impact
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_ratio = (df['high'] - df['low']) / (df['open'] + df['close'] + 1e-6)  # Adding small value to avoid division by zero
    stability_adjusted_metric = integrated_volume_adjusted / (price_range_ratio + 1e-6)  # Adding small value to avoid division by zero
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    volume_weighted_open = (df['open'] * df['volume']).sum() / df['volume'].sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (volume_weighted_close - volume_weighted_open) / (df['high'].max() - df['low'].min())
    
    # Introduce Dynamic Price Momentum
    close_5d_ma = df['close'].rolling(window=5).mean()
    dynamic_price_momentum = (df['close'] - close_5d_ma) / (df['close'].rolling(window=5).max() - df['close'].rolling(window=5).min())
    
    # Combine all components into the final factor
    final_factor = stability_adjusted_metric * vwapchl * dynamic_price_momentum
    
    return final_factor
```
