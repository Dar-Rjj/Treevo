```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 20-Day Average Close
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    
    # Calculate Daily Momentum
    df['daily_momentum'] = df['close'] - df['20_day_avg_close']
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate 5-Day Moving Average of High-Low Differences
    df['intraday_volatility'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Combine Momentum and Intraday Volatility
    df['combined_mom_vol'] = -1 * df['daily_momentum'] + df['intraday_volatility']
    
    # Calculate Intraday High-Low Spread
    df['intraday_range'] = df['high'] - df['low']
    
    # Measure Close Location within Intraday Range
    df['close_location'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Adjust for Volume Intensity
    df['weighted_reversal_potential'] = df['volume'] * df['close_location']
    
    # Combine All Factors
    df['alpha_factor'] = df['combined_mom_vol'] + df['weighted_reversal_potential']
    
    return df['alpha_factor']
```
