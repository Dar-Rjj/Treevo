import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Change
    vw_close = df['close'] * df['volume']
    vw_open = df['open'] * df['volume']
    vw_close_5d = vw_close.rolling(window=5).sum()
    vw_open_5d = vw_open.rolling(window=5).sum()
    vw_c_o_change = vw_close_5d - vw_open_5d

    # Calculate Volume-Weighted Intraday High-Low Spread
    vw_high = df['high'] * df['volume']
    vw_low = df['low'] * df['volume']
    vw_high_5d = vw_high.rolling(window=5).sum()
    vw_low_5d = vw_low.rolling(window=5).sum()
    vw_hl_spread = vw_high_5d - vw_low_5d

    # Combine Volume-Weighted Changes
    combined_vw_changes = vw_c_o_change + vw_hl_spread

    # Adjust for Volume-Weighted Average Price
    vw_average_price = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    adjusted_vw_changes = combined_vw_changes - vw_average_price

    # Incorporate Price Trend Factor
    price_trend = df['close'].rolling(window=20).mean()
    factor_values = adjusted_vw_changes - price_trend

    return factor_values
