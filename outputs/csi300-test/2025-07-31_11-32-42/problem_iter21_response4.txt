import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_volatility / open_to_close_movement.replace(0, np.nan)
    
    # Incorporate Advanced Volume Impact
    volume_impact = integrated_metric * (df['volume'] ** (1/3))
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    volume_adjusted_metric = volume_impact / price_range_stability.replace(0, np.nan)
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    volume_weighted_open = (df['open'] * df['volume']).sum() / df['volume'].sum()
    
    # Synthesize VWAPCHL into the Model
    vwap_chl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'])
    
    # Introduce Dynamic Price Momentum
    five_day_moving_avg = df['close'].rolling(window=5).mean()
    five_day_momentum = (df['close'] - five_day_moving_avg) / (df['close'].rolling(window=5).max() - df['close'].rolling(window=5).min()).replace(0, np.nan)
    
    # Enhance with Additional Momentum Indicators
    ten_day_ema = df['close'].ewm(span=10, adjust=False).mean()
    ten_day_momentum = (df['close'] - ten_day_ema) / (df['close'].rolling(window=10).max() - df['close'].rolling(window=10).min()).replace(0, np.nan)
    
    # Integrate Momentum Indicators
    balanced_momentum_signal = (five_day_momentum + ten_day_momentum) / 2
    
    # Final Alpha Factor
    alpha_factor = volume_adjusted_metric * vwap_chl * balanced_momentum_signal
    return alpha_factor
