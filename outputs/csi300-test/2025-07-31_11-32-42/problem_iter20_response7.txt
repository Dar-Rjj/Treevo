```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_metric = intraday_high_low_spread - open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    log_volume = np.log(df['volume'] + 1)
    volume_adjusted_metric = integrated_metric * log_volume
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = volume_adjusted_metric / price_range_stability
    
    # Calculate Volume-Weighted Prices
    vwap_close = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_open = (df['open'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Synthesize VWAPCHL into the Model
    vwap_chl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    
    # Evaluate Volume-Weighted Momentum
    vwap = (df['close'] * df['volume'] + df['open'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_momentum = vwap - vwap.shift(1)
    
    # Introduce Dynamic Weighting
    vol_5day_ma = df['volume'].rolling(window=5).mean()
    vol_5day_std = df['volume'].rolling(window=5).std()
    z_score = (df['volume'] - vol_5day_ma) / vol_5day_std
    z_score_weighted_metric = volume_adjusted_metric * z_score
    
    # Introduce Momentum Component
    n = 10  # Example: 10-day SMA
    sma_n = df['close'].rolling(window=n).mean()
    momentum_component = df['close'] - sma_n
    
    # Integrate Momentum with Volume-Adjusted Metrics
    momentum_integrated_metric = momentum_component * stability_adjusted_metric
    
    # Final Integration
    final_factor = (
        stability_adjusted_metric 
        * vwap_chl 
        + momentum_integrated_metric 
        * vwap_momentum 
        * z_score_weighted_metric
    )
    
    return final_factor
```
