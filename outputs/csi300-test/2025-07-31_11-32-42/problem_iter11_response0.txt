```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using (open, high, low, close, amount, volume) data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the volume-weighted typical price
    vwap = (df['amount'] / df['volume']).rolling(window=20).mean()
    
    # Calculate the rolling average of the typical price
    typical_price_rolling_mean = typical_price.rolling(window=20).mean()
    
    # Calculate the rolling standard deviation of the typical price
    typical_price_rolling_std = typical_price.rolling(window=20).std()
    
    # Calculate the normalized typical price
    normalized_typical_price = (typical_price - typical_price_rolling_mean) / typical_price_rolling_std
    
    # Calculate the factor
    factor = normalized_typical_price * vwap
    
    return factor
```
