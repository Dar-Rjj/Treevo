```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-day Price Return
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Compute 20-day Price Return
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Compute 5-day Average Volume
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    
    # Compute 10-day Average Volume
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    
    # Subtract 20-day Price Return from 5-day Price Return
    df['price_momentum_diff'] = df['5_day_return'] - df['20_day_return']
    
    # Subtract 10-day Average Volume from 5-day Average Volume
    df['volume_surge'] = df['5_day_avg_volume'] - df['10_day_avg_volume']
    
    # Multiply the two results
    df['alpha_factor'] = df['price_momentum_diff'] * df['volume_surge']
    
    return df['alpha_factor']
```
