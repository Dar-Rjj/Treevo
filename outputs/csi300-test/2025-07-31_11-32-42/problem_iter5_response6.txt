```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Factors
    df['1_day_price_momentum'] = df['close'] / df['close'].shift(1) - 1
    df['5_day_price_momentum'] = df['close'] / df['close'].shift(5) - 1
    df['20_day_price_momentum'] = df['close'] / df['close'].shift(20) - 1

    # Calculate Volume Activity Indicators
    df['1_day_volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()

    # Calculate Daily Momentum
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    df['daily_momentum'] = df['close'] - df['20_day_avg_close']

    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['5_day_intraday_volatility'] = df['high_low_diff'].rolling(window=5).mean()

    # Combine Momentum and Volume Activities
    df['combined_1_day'] = df['1_day_price_momentum'] * df['1_day_volume_diff']
    df['combined_5_day'] = df['5_day_price_momentum'] * df['5_day_avg_volume']
    df['combined_20_day'] = df['20_day_price_momentum'] * df['20_day_avg_volume']
    df['combined_momentum_volume'] = df['combined_1_day'] + df['combined_5_day'] + df['combined_20_day']

    # Calculate Daily Price Movement
    df['daily_range'] = df['high'] - df['low']
    df['midpoint'] = (df['high'] + df['low']) / 2

    # Calculate Multi-Day Average Midpoint
    df['5_day_midpoint_avg'] = df['midpoint'].rolling(window=5).mean()

    # Compute Volume Adjusted Momentum
    df['close_to_close_return'] = df['close'] / df['close'].shift(1) - 1
    df['volume_adjusted_momentum'] = df['close_to_close_return'] * df['volume']

    # Calculate Enhanced Price Envelopment
    df['multi_day_range_sum'] = df['daily_range'].rolling(window=5).sum()
    df['multi_day_volume_sum'] = df['volume'].rolling(window=5).sum()
    df['base_envelopment'] = df['multi_day_range_sum'] / df['multi_day_volume_sum']
    df['weighted_sum'] = (df['high'] + df['low']) / 2
    df['enhanced_price_envelopment'] = df['base_envelopment'] + df['weighted_sum']

    # Combine All Momentum and Volume Factors
    df['combined_factors'] = df['combined_momentum_volume'] - df['daily_momentum'] + df['5_day_intraday_volatility']

    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['combined_factors'] * df['volume_adjusted_momentum'] * df['enhanced_price_envelopment']

    return df['final_alpha_factor'].dropna()
```
