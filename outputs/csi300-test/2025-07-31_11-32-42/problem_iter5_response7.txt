```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_volatility / open_to_close_movement
    
    # Incorporate Volume Impact
    volume_adjusted_metric = integrated_metric * df['volume']
    
    # Factor in Price Range as a Stability Measure
    price_range_ratio = (df['high'] - df['low']) / df['open']
    stability_adjusted_metric = volume_adjusted_metric / price_range_ratio
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    volume_weighted_open = (df['open'] * df['volume']).sum() / df['volume'].sum()
    
    # Synthesize VWAPCHL into the Model
    vwap_chl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'])
    
    # Final Alpha Factor
    alpha_factor = stability_adjusted_metric * vwap_chl
    
    return alpha_factor
```
