import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using past and current data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       indexed by date.
    
    Returns:
    pd.Series: The calculated alpha factor indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 5-day moving average of the close price
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate the 10-day exponentially weighted moving average of the close price
    df['10_day_ewma'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the price range (high - low)
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the average volume over the last 5 days
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the relative strength index (RSI) over a 14-day period
    def rsi(series, period=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    df['rsi_14'] = rsi(df['close'])
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['5_day_ma'] - df['10_day_ewma']) / df['price_range'] * df['rsi_14'] * (df['volume'] / df['5_day_avg_volume'])
    
    # Return the alpha factor
    return df['alpha_factor']
