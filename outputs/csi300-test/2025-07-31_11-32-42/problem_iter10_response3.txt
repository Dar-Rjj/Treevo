import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Measure Close Location within Intraday Range
    df['close_location'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Adjust for Volume Intensity
    df['weighted_reversal'] = df['volume'] * df['close_location']
    
    # Incorporate Open-Price Impact
    df['open_location'] = (df['open'] - df['low']) / df['intraday_range']
    df['combined_weight'] = df['weighted_reversal'] + (1 - df['open_location'])
    
    # Incorporate Previous Day's Momentum
    df['prev_close'] = df['close'].shift(1)
    df['momentum'] = (df['open'] - df['prev_close']) * df['combined_weight']
    
    # Final Intraday Reversal and Momentum Potential
    df['intraday_reversal_momentum'] = df['combined_weight'] + df['momentum']
    
    return df['intraday_reversal_momentum'].dropna()
