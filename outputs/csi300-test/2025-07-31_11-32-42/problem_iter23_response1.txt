```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 20-Day Average Close
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    
    # Subtract 20-Day Average Close from Today's Close
    df['momentum_20_day'] = df['close'] - df['20_day_avg_close']
    
    # Compute 5-day Price Return
    df['price_return_5_day'] = df['close'] / df['close'].shift(5) - 1
    
    # Compute 20-day Price Return
    df['price_return_20_day'] = df['close'] / df['close'].shift(20) - 1
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    df['close_to_open_change'] = df['close'] - df['open']
    
    # Combine Intraday and Opening Gaps
    df['combined_intraday_open_gap'] = df['intraday_high_low_spread'] - df['close_to_open_change']
    
    # Weight by Volume
    df['volume_weighted_combined_change'] = df['combined_intraday_open_gap'] * df['volume']
    
    # Calculate 5-Day Moving Average of High-Low Differences
    df['5_day_avg_high_low_diff'] = df['intraday_high_low_spread'].rolling(window=5).mean()
    
    # Measure Close Location within Intraday Range
    df['close_location_ratio'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Adjust for Volume Intensity
    df['weighted_intraday_reversal_potential'] = df['close_location_ratio'] * df['volume']
    
    # Calculate Volume Surge
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    
    # Calculate Volume Adjusted Momentum
    df['volume_adjusted_momentum'] = (df['close'] / df['close'].shift(1) - 1) * df['volume']
    
    # Synthesize All Factors
    df['daily_momentum'] = df['momentum_20_day'] * -1
    df['intraday_volatility'] = df['intraday_high_low_spread'] * df['volume']
    df['short_term_momentum_diff'] = df['price_return_5_day'] - df['price_return_20_day']
    df['long_term_momentum_diff'] = df['price_return_20_day'] - (df['close'] / df['close'].shift(50) - 1)
    df['synthesized_factors'] = (
        df['daily_momentum'] +
        df['intraday_volatility'] +
        df['weighted_intraday_reversal_potential'] +
        df['short_term_momentum_diff'] * df['long_term_momentum_diff'] +
        df['volume_adjusted_momentum']
    )
    
    # Incorporate Intraday High-Low Spread Ratio
    df['5_day_avg_high'] = df['high'].rolling(window=5).mean()
    df['5_day_avg_low'] = df['low'].rolling(window=5).mean()
    df['intraday_high_low_spread_ratio'] = df['5_day_avg_high'] / df['5_day_avg_low']
    df['synthesized_factors'] += df['intraday_high_low_spread_ratio'] * df['intraday_volatility']
    
    # Incorporate Price Momentum with Volume
    df['price_return_10_day'] = df['close'] / df['close'].shift(10) - 1
    df['volume_weighted_price_momentum_10_day'] = df['price_return_10_day'] * df['10_day_avg_volume']
    df['synthesized_factors'] += df['volume_weighted_price_momentum_10_day']
    
    # Final Alpha Factor
    alpha_factor = df['synthesized_factors']
    
    return alpha_factor
```
