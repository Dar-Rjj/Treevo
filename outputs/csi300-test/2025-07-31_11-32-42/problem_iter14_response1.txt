```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Midpoint
    df['midpoint'] = (df['high'] + df['low']) / 2
    
    # Multi-Day Average Midpoint (5-day window)
    df['avg_midpoint'] = df['midpoint'].rolling(window=5).mean()
    
    # Compute Volume Adjusted Momentum
    df['volume_adjusted_momentum'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Daily Returns
    df['daily_returns'] = df['close'] - df['open']
    
    # Volume Adjusted Daily Return
    df['volume_adjusted_daily_return'] = df['volume'] * df['daily_returns']
    
    # Intraday Range to Volume Ratio
    df['intraday_range_to_volume_ratio'] = df['intraday_range'] / df['volume']
    
    # Closing Price to High-Low Range Ratio
    df['closing_price_to_high_low_range_ratio'] = df['close'] / df['intraday_range']
    
    # Enhanced Price Envelopment
    sum_daily_ranges = df['intraday_range'].rolling(window=5).sum()
    sum_volumes = df['volume'].rolling(window=5).sum()
    df['enhanced_price_envelopment'] = (sum_daily_ranges / sum_volumes) + df['midpoint']
    
    # Volatility-Adjusted Momentum
    df['std_daily_returns'] = df['daily_returns'].rolling(window=5).std()
    df['volatility_adjusted_momentum'] = df['volume_adjusted_momentum'] / df['std_daily_returns']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    df['integrated_volatility'] = df['intraday_range'] / df['daily_returns']
    
    # Incorporate Advanced Volume Impact
    df['advanced_volume_impact'] = df['integrated_volatility'] * np.cbrt(df['volume'])
    
    # Factor in Adjusted Price Range as a Stability Measure
    df['adjusted_price_range'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    df['volume_adjusted_metric'] = df['advanced_volume_impact'] / df['adjusted_price_range']
    
    # Calculate Volume-Weighted Prices
    df['vwap_close'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_open'] = (df['open'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Synthesize VWAPCHL into the Model
    df['vwapchl'] = (df['vwap_close'] - df['vwap_open']) / df['intraday_range']
    
    # Introduce Dynamic Price Momentum
    df['3day_ma_close'] = df['close'].rolling(window=3).mean()
    df['dynamic_price_momentum'] = (df['close'] - df['3day_ma_close']) / df['close'].rolling(window=3).std()
    
    # Combine Factors
    combined_factors = (
        df['volume_adjusted_momentum'] * df['enhanced_price_envelopment'] +
        df['volume_adjusted_daily_return'].rolling(window=5).sum() +
        df['intraday_range_to_volume_ratio'].rolling(window=5).sum() +
        df['closing_price_to_high_low_range_ratio'].rolling(window=5).sum() +
        df['volatility_adjusted_momentum'].rolling(window=5).sum() +
        df['volume_adjusted_metric'].rolling(window=5).sum() +
        df['dynamic_price_momentum'].rolling(window=5).sum()
    )
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors
    
    return final_alpha_factor
```
