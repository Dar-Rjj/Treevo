```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_volatility / (open_to_close_movement + 1e-6)  # add small constant to avoid division by zero
    
    # Incorporate Advanced Volume Impact
    volume_impact = np.cbrt(df['volume']) * integrated_metric
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'] + 1e-6)  # add small constant to avoid division by zero
    volume_adjusted_metric = volume_impact / (price_range_stability + 1e-6)  # add small constant to avoid division by zero
    
    # Synthesize VWAPCHL into the Model
    vwap_close = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_open = (df['open'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_chl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    
    # Introduce Dynamic Price Momentum
    close_5_day_ma = df['close'].rolling(window=5).mean()
    dynamic_momentum_5 = (df['close'] - close_5_day_ma) / (df['close'].rolling(window=5).max() - df['close'].rolling(window=5).min() + 1e-6)  # add small constant to avoid division by zero
    
    # Enhance with Additional Momentum Indicators
    close_10_day_ma = df['close'].rolling(window=10).mean()
    dynamic_momentum_10 = (df['close'] - close_10_day_ma) / (df['close'].rolling(window=10).max() - df['close'].rolling(window=10).min() + 1e-6)  # add small constant to avoid division by zero
    
    # Integrate Momentum Indicators
    combined_momentum = (dynamic_momentum_5 + dynamic_momentum_10) / 2
    
    # Final Alpha Factor
    alpha_factor = volume_adjusted_metric * vwap_chl * combined_momentum
    
    return alpha_factor
```
