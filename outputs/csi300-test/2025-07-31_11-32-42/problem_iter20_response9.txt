```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Integrated Momentum
    df['20-Day_Avg_Close'] = df['close'].rolling(window=20).mean()
    df['Integrated_Momentum'] = df['close'] - df['20-Day_Avg_Close']
    
    # Intraday Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['5-Day_High_Low_Mean'] = df['High_Low_Diff'].rolling(window=5).mean()
    df['Intraday_Volatility'] = df['5-Day_High_Low_Mean']
    
    # Open-to-Close Movement
    df['Open_to_Close_Move'] = df['close'] - df['open']
    
    # Integrated Intraday and Open-to-Close Metric
    df['Integrated_Intraday_Open_to_Close'] = df['Intraday_Volatility'] / df['Open_to_Close_Move']
    
    # Volume Trend
    df['3-Day_Volume_MA'] = df['volume'].rolling(window=3).mean()
    df['7-Day_Volume_MA'] = df['volume'].rolling(window=7).mean()
    df['Volume_Trend'] = df['3-Day_Volume_MA'] - df['7-Day_Volume_MA']
    
    # Advanced Volume Impact
    df['Cubed_Root_Volume'] = np.cbrt(df['volume'])
    df['Advanced_Volume_Impact'] = df['Cubed_Root_Volume'] * df['Integrated_Intraday_Open_to_Close']
    
    # Adjusted Price Range as a Stability Measure
    df['Adjusted_Price_Range'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    df['Stability_Measure'] = df['Advanced_Volume_Impact'] / df['Adjusted_Price_Range']
    
    # Dynamic Price Momentum
    df['5-Day_Close_MA'] = df['close'].rolling(window=5).mean()
    df['5-Day_Close_Range'] = df['close'].rolling(window=5).max() - df['close'].rolling(window=5).min()
    df['Dynamic_Price_Momentum_5'] = (df['close'] - df['5-Day_Close_MA']) / df['5-Day_Close_Range']
    
    # Additional Momentum Indicators
    df['10-Day_Close_MA'] = df['close'].rolling(window=10).mean()
    df['10-Day_Close_Range'] = df['close'].rolling(window=10).max() - df['close'].rolling(window=10).min()
    df['Dynamic_Price_Momentum_10'] = (df['close'] - df['10-Day_Close_MA']) / df['10-Day_Close_Range']
    
    # Integrate Momentum Indicators
    df['Integrated_Momentum_Indicators'] = (df['Dynamic_Price_Momentum_5'] + df['Dynamic_Price_Momentum_10']) / 2
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        (df['Integrated_Momentum'] + df['Intraday_Volatility'] + df['Volume_Trend']) * 
        df['Advanced_Volume_Impact'] / 
        df['Stability_Measure'] + 
        df['Integrated_Momentum_Indicators'] - 
        (df['close'] / df['close'].shift(5) - 1) + 
        (df['close'] / df['close'].shift(10) - 1)
    )
    
    return df['Final_Alpha_Factor']
```
