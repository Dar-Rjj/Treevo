import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Combine Intraday and Opening Gaps
    combined_intraday_open_gap = intraday_high_low_spread - close_to_open_change
    
    # Weight by Volume
    weighted_combined_change = combined_intraday_open_gap * df['volume']
    
    # Calculate Daily Price Movement
    daily_price_movement = (df['high'] + df['low']) / 2
    
    # Calculate Multi-Day Average Midpoint
    multi_day_avg_midpoint = daily_price_movement.rolling(window=5).mean()
    
    # Compute Volume Adjusted Momentum
    close_to_close_return = df['close'].pct_change()
    volume_adjusted_momentum = close_to_close_return * df['volume']
    
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_high_low_range / open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    advanced_volume_impact = integrated_metric * df['volume'].pow(0.5)
    
    # Factor in Adjusted Price Range as a Stability Measure
    adjusted_price_range = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_measure = advanced_volume_impact / adjusted_price_range
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    volume_weighted_open = (df['open'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'])
    
    # Combine Factors
    combined_factors = (
        volume_adjusted_momentum +
        weighted_combined_change.rolling(window=5).sum() +
        advanced_volume_impact.rolling(window=5).sum() +
        stability_measure.rolling(window=5).sum() +
        vwapchl.rolling(window=5).sum()
    )
    
    return combined_factors
