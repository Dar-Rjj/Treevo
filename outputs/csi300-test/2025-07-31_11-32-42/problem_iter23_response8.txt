```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor to predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: Factor values indexed by (date).
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily returns
    df['ma_10'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the daily returns
    df['ma_30'] = df['daily_return'].rolling(window=30).mean()
    
    # Calculate the standard deviation of the last 30 days' daily returns
    df['std_30'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the 10-day moving average of the trading volume
    df['vol_ma_10'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the trading volume
    df['vol_ma_30'] = df['volume'].rolling(window=30).mean()
    
    # Calculate the 10-day exponential moving average (EMA) of the closing price
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponential moving average (EMA) of the closing price
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the ratio of the current volume to the 30-day moving average of the volume
    df['vol_ratio'] = df['volume'] / df['vol_ma_30']
    
    # Calculate the price-to-volume momentum
    df['mom'] = (df['close'] / df['close'].shift(5)) * df['vol_ratio']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (
        (df['ma_10'] - df['ma_30']) / df['std_30'] + 
        (df['ema_10'] - df['ema_30']) / df['close'] +
        df['mom']
    )
    
    return df['alpha_factor']
```
