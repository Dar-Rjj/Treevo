```python
import pandas as pd

def heuristics_v2(df):
    # Calculate moving averages (SMA and EMA)
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['EMA_50'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Evaluate price momentum
    df['Price_Change_1D'] = df['close'].pct_change(1)
    df['Price_Change_5D'] = df['close'].pct_change(5)
    
    df['ROC_1D'] = df['Price_Change_1D']
    df['ROC_5D'] = df['Price_Change_5D']
    
    # Utilize volume and amount data
    df['Avg_Volume_5D'] = df['volume'].rolling(window=5).mean()
    df['Avg_Volume_10D'] = df['volume'].rolling(window=10).mean()
    
    df['Volume_Spike_5D'] = (df['volume'] / df['Avg_Volume_5D']) - 1
    df['Volume_Spike_10D'] = (df['volume'] / df['Avg_Volume_10D']) - 1
    
    # On-Balance Volume (OBV)
    df['OBV'] = (df['close'].diff() > 0).astype(int) * df['volume'] - (df['close'].diff() < 0).astype(int) * df['volume']
    df['OBV'] = df['OBV'].cumsum()
    
    # Incorporate high and low prices for volatility analysis
    df['Daily_Range'] = df['high'] - df['low']
    
    df['TR'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    df['ATR_14D'] = df['TR'].rolling(window=14).mean()
    
    # Analyze opening and closing prices
    df['Open_Close_Diff'] = df['close'] - df['open']
    df['Percentage_Gap'] = df['Open_Close_Diff'] / df['open']
    
    # Combine multiple factors
    # Example: Combining price momentum, volume spikes, and OBV
    df['Alpha_Factor'] = (
        0.3 * df['Price_Change_5D'] + 
        0.3 * df['Volume_Spike_5D'] + 
        0.4 * df['OBV'].pct_change(5)
    )
    
    return df['Alpha_Factor'].dropna()
```
