```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrated Intraday and Opening Gaps
    integrated_gap = intraday_high_low_spread - open_to_close_movement
    
    # Advanced Volume Impact
    log_volume = np.log(df['volume'] + 1)
    volume_impact = integrated_gap * log_volume
    
    # Adjusted Price Range as a Stability Measure
    adjusted_price_range = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = volume_impact / adjusted_price_range
    
    # Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).rolling(window=5).sum()
    volume_weighted_open = (df['open'] * df['volume']).rolling(window=5).sum()
    
    # VWAPCHL into the Model
    vwapchl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'])
    
    # Volume-Weighted Momentum
    vw_average_price = (df['close'] * df['volume'] + df['open'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    change_in_vw_average_price = vw_average_price - vw_average_price.shift(1)
    
    # Dynamic Weighting
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    five_day_std_volume = df['volume'].rolling(window=5).std()
    z_score = (df['volume'] - five_day_avg_volume) / five_day_std_volume
    z_score_weighted_metric = stability_adjusted_metric * z_score
    
    # Momentum Component
    sma_close = df['close'].rolling(window=5).mean()
    momentum_component = df['close'] - sma_close
    
    # Final Integration
    final_factor = (stability_adjusted_metric * vwapchl + momentum_component) * change_in_vw_average_price * z_score_weighted_metric
    
    return final_factor
```
