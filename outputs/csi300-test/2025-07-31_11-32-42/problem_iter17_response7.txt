```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_volatility / (open_to_close_movement + 1e-6)  # Adding a small epsilon to avoid division by zero
    
    # Incorporate Advanced Volume Impact
    volume_impact = integrated_metric * np.cbrt(df['volume'])
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'] + 1e-6)
    volume_adjusted_metric = volume_impact / (price_range_stability + 1e-6)
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    volume_weighted_open = (df['open'] * df['volume']).sum() / df['volume'].sum()
    
    # Synthesize VWAPCHL into the Model
    vwap_chl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'] + 1e-6)
    
    # Introduce Dynamic Price Momentum
    close_5_day_ma = df['close'].rolling(window=5).mean()
    close_5_day_range = df['close'].rolling(window=5).max() - df['close'].rolling(window=5).min()
    momentum_5_day = (df['close'] - close_5_day_ma) / (close_5_day_range + 1e-6)
    
    # Enhance with Additional Momentum Indicators
    close_10_day_ma = df['close'].rolling(window=10).mean()
    close_10_day_range = df['close'].rolling(window=10).max() - df['close'].rolling(window=10).min()
    momentum_10_day = (df['close'] - close_10_day_ma) / (close_10_day_range + 1e-6)
    
    # Integrate Momentum Indicators
    balanced_momentum = (momentum_5_day + momentum_10_day) / 2
    
    # Final Alpha Factor
    alpha_factor = volume_adjusted_metric * vwap_chl * balanced_momentum
    
    return alpha_factor
```
