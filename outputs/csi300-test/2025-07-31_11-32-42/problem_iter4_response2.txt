```python
import pandas as pd

def heuristics_v2(df, N=10, M=5):
    # Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Squared Returns Over N Days
    df['squared_returns'] = df['daily_return'].pow(2)
    df['sum_squared_returns'] = df['squared_returns'].rolling(window=N).sum()
    
    # Volume Weighted Close Price
    df['moving_avg_close'] = df['close'].rolling(window=M).mean()
    df['volume_weighted_close'] = df['moving_avg_close'] * df['volume']
    
    # High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Combined Indicator
    df['combined_indicator'] = (df['sum_squared_returns'] * df['volume_weighted_close']) / df['high_low_range']
    
    return df['combined_indicator']
```
