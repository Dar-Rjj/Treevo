```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Measure Close Location within Intraday Range
    close_location = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Adjust for Volume Intensity
    weighted_intraday_reversal_potential = close_location * df['volume']
    
    # Calculate Short-Term Price Momentum
    short_term_momentum_5d = df['close'].pct_change(5)
    short_term_momentum_10d = df['close'].pct_change(10)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum_20d = df['close'].pct_change(20)
    long_term_momentum_50d = df['close'].pct_change(50)
    
    # Calculate Volume Surge
    volume_5d_avg = df['volume'].rolling(window=5).mean()
    volume_10d_avg = df['volume'].rolling(window=10).mean()
    
    # Combine Factors
    momentum_diff_short = short_term_momentum_5d - short_term_momentum_10d
    momentum_diff_long = long_term_momentum_20d - long_term_momentum_50d
    volume_surge_diff = volume_5d_avg - volume_10d_avg
    
    combined_factor = (momentum_diff_short * momentum_diff_long * volume_surge_diff) + weighted_intraday_reversal_potential
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Combine Intraday and Opening Gaps
    combined_change = intraday_high_low_spread - close_to_open_change
    
    # Weight by Volume
    volume_weighted_combined_change = combined_change * df['volume']
    
    # Add to the Integrated Intraday and Momentum Reversal Indicator
    final_factor = combined_factor + volume_weighted_combined_change
    
    return final_factor
```
