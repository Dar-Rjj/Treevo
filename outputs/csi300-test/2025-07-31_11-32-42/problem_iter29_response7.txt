import pandas as pd

def heuristics_v2(df):
    """
    This function computes a novel alpha factor based on the provided DataFrame.
    The factor is based on the weighted sum of the following:
    1. Relative strength index (RSI) of the close price.
    2. Volume-weighted average price (VWAP).
    3. A momentum factor based on the change in the close price over a certain period.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as 'date'.
    
    Returns:
    pd.Series: Series with the computed alpha factor, indexed by 'date'.
    """
    # Calculate the RSI
    def compute_rsi(prices, window=14):
        delta = prices.diff(1)
        up, down = delta.copy(), delta.copy()
        up[up < 0] = 0
        down[down > 0] = 0
        roll_up = up.rolling(window).mean()
        roll_down = down.abs().rolling(window).mean()
        RS = roll_up / roll_down
        return 100.0 - (100.0 / (1.0 + RS))
    
    rsi = compute_rsi(df['close'])

    # Calculate the VWAP
    vwap = (df['amount'] * df['close']).cumsum() / df['volume'].cumsum()

    # Calculate the momentum factor
    momentum = df['close'] / df['close'].shift(14) - 1

    # Combine the factors into a single alpha factor
    alpha_factor = 0.5 * rsi + 0.3 * vwap + 0.2 * momentum
    return alpha_factor
