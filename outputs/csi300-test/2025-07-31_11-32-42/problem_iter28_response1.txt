```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Metrics
    integrated_metric = intraday_volatility + open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    volume_impact = integrated_metric * np.cbrt(df['volume'])
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = volume_impact / price_range_stability
    
    # Calculate Volume-Weighted Prices
    vwap_close = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    vwap_open = (df['open'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Synthesize VWAPCHL into the Model
    vwap_chl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    
    # Introduce Dynamic Price Momentum
    close_5dma = df['close'].rolling(window=5).mean()
    dynamic_momentum = (df['close'] - close_5dma) / df['close'].rolling(window=5).std()
    
    # Daily Momentum
    close_20dma = df['close'].rolling(window=20).mean()
    daily_momentum = df['close'] - close_20dma
    
    # Combine Momentum and Intraday Volatility
    combined_momentum = -1 * daily_momentum + intraday_volatility
    
    # Measure Close Location within Intraday Range
    close_location = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Adjust for Volume Intensity
    volume_intensity = df['volume'] * close_location
    
    # Evaluate Volume-Weighted Momentum
    vwap = (df['close'] * df['volume'] + df['open'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_change = vwap - vwap.shift(1)
    
    # Final Integration
    final_factor = (
        stability_adjusted_metric * vwap_chl * vwap_change *
        (df['close'].pct_change(5) - df['close'].pct_change(10)) *
        (df['close'].pct_change(20) - df['close'].pct_change(50)) *
        vwap_change
    )
    
    return final_factor
```
