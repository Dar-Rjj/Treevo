```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Daily Midpoint
    daily_midpoint = (df['high'] + df['low']) / 2
    
    # Multi-Day Average Midpoint (5-day window)
    multi_day_avg_midpoint = daily_midpoint.rolling(window=5).mean()
    
    # Volume Adjusted Momentum
    volume_adjusted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Volume Adjusted Daily Return
    volume_adjusted_daily_return = df['volume'] * open_close_return
    
    # Intraday Range to Volume Ratio
    intraday_range_volume_ratio = intraday_high_low_range / df['volume']
    
    # Closing Price to High-Low Range Ratio
    closing_price_high_low_ratio = df['close'] / intraday_high_low_range
    
    # Enhanced Price Envelopment
    sum_daily_ranges = intraday_high_low_range.rolling(window=5).sum()
    sum_volumes = df['volume'].rolling(window=5).sum()
    enhanced_price_envelopment = (sum_daily_ranges / sum_volumes) + multi_day_avg_midpoint
    
    # Volatility-Adjusted Momentum
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    std_dev_5_day = daily_returns.rolling(window=5).std()
    volatility_adjusted_momentum = volume_adjusted_momentum / std_dev_5_day
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_high_low_range / open_close_return
    
    # Advanced Volume Impact
    advanced_volume_impact = integrated_metric * (df['volume'] ** (1/3))
    
    # Adjusted Price Range as a Stability Measure
    adjusted_price_range = (df['high'] - df['low']) / (df['open'] + df['close'])
    adjusted_metric = advanced_volume_impact / adjusted_price_range
    
    # Volume-Weighted Prices
    vwap_close = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_open = (df['open'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Synthesize VWAPCHL into the Model
    vwap_chl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    
    # Dynamic Price Momentum
    close_3_day_ma = df['close'].rolling(window=3).mean()
    dynamic_price_momentum = (df['close'] - close_3_day_ma) / df['close'].rolling(window=3).std()
    
    # Final Alpha Factor
    alpha_factor = (
        volume_adjusted_momentum * enhanced_price_envelopment +
        volume_adjusted_daily_return.rolling(window=5).sum() +
        intraday_range_volume_ratio.rolling(window=5).sum() +
        closing_price_high_low_ratio.rolling(window=5).sum() +
        volatility_adjusted_momentum.rolling(window=5).sum() +
        adjusted_metric.rolling(window=5).sum() +
        dynamic_price_momentum.rolling(window=5).sum() +
        vwap_chl
    )
    
    return alpha_factor
```
