```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_volatility / (open_to_close_movement + 1e-6)  # Add small epsilon to avoid division by zero
    
    # Incorporate Advanced Volume Impact
    volume_adjusted_metric = integrated_metric * (df['volume'] ** (1/3))
    
    # Factor in Adjusted Price Range as a Stability Measure
    adjusted_price_range = (df['high'] - df['low']) / (df['open'] + df['close'] + 1e-6)  # Add small epsilon to avoid division by zero
    stability_adjusted_metric = volume_adjusted_metric / adjusted_price_range
    
    # Calculate Volume-Weighted Prices
    vwap_close = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    vwap_open = (df['open'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Synthesize VWAPCHL into the Model
    vwapc_minus_vwapo = vwap_close - vwap_open
    vwapc_minus_vwapo_normalized = vwapc_minus_vwapo / (df['high'] - df['low'] + 1e-6)  # Add small epsilon to avoid division by zero
    
    # Introduce Dynamic Price Momentum
    close_5_day_ma = df['close'].rolling(window=5).mean()
    dynamic_momentum_5_day = (df['close'] - close_5_day_ma) / (df['close'].rolling(window=5).max() - df['close'].rolling(window=5).min() + 1e-6)  # Add small epsilon to avoid division by zero
    
    # Enhance with Additional Momentum Indicators
    close_10_day_ema = df['close'].ewm(span=10, adjust=False).mean()
    dynamic_momentum_10_day = (df['close'] - close_10_day_ema) / (df['close'].rolling(window=10).max() - df['close'].rolling(window=10).min() + 1e-6)  # Add small epsilon to avoid division by zero
    
    # Integrate and Weight Momentum Indicators
    weighted_momentum = 0.6 * dynamic_momentum_5_day + 0.4 * dynamic_momentum_10_day
    
    # Final Alpha Factor
    alpha_factor = (stability_adjusted_metric + vwapc_minus_vwapo_normalized + weighted_momentum) / 3
    
    return alpha_factor
```
