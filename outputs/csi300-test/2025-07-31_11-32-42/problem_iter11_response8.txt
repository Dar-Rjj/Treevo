import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Evaluate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_gap = intraday_high_low_spread - close_to_open_change
    
    # Incorporate Advanced Volume Impact
    advanced_volume_impact = integrated_gap * df['volume'].pow(0.5)
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = advanced_volume_impact / price_range_stability
    
    # Weight by Volume
    volume_weighted_combined = stability_adjusted_metric * df['volume']
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    volume_weighted_open = (df['open'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'])
    
    # Final Alpha Factor
    alpha_factor = volume_weighted_combined * vwapchl
    
    return alpha_factor
