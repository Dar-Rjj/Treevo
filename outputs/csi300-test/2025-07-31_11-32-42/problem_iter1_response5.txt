```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-day Price Momentum
    df['1_day_momentum'] = df['close'].pct_change()
    
    # Calculate 5-day Price Momentum
    df['5_day_momentum'] = df['close'].pct_change(periods=5)
    
    # Calculate 20-day Price Momentum
    df['20_day_momentum'] = df['close'].pct_change(periods=20)
    
    # Calculate 1-day Volume Difference
    df['1_day_volume_diff'] = df['volume'].diff()
    
    # Calculate 5-day Average Volume
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    
    # Calculate 20-day Average Volume
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()
    
    # Combine Factors
    df['1_day_factor'] = df['1_day_momentum'] * df['1_day_volume_diff']
    df['5_day_factor'] = df['5_day_momentum'] * df['5_day_avg_volume']
    df['20_day_factor'] = df['20_day_momentum'] * df['20_day_avg_volume']
    
    # Composite Factor
    df['composite_factor'] = (df['1_day_factor'] + df['5_day_factor'] + df['20_day_factor']) / 3
    
    return df['composite_factor'].dropna()
```
