```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 20-Day Average Close
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    
    # Daily Momentum
    df['daily_momentum'] = df['close'] - df['20_day_avg_close']
    
    # Short-Term (5-day) Price Return
    df['5_day_return'] = df['close'] / df['close'].shift(5) - 1
    
    # Long-Term (20-day) Price Return
    df['20_day_return'] = df['close'] / df['close'].shift(20) - 1
    
    # Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['5_day_volatility'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Combine Momentum and Intraday Volatility
    df['momentum_volatility'] = -df['daily_momentum'] + df['5_day_volatility']
    
    # Intraday High-Low Spread
    df['intraday_range'] = df['high'] - df['low']
    
    # Measure Close Location within Intraday Range
    df['close_proportional_distance'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Adjust for Volume Intensity
    df['weighted_intraday_reversal'] = df['volume'] * df['close_proportional_distance']
    
    # Volume Surge
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_surge'] = df['5_day_avg_volume'] - df['10_day_avg_volume']
    
    # Combine Short-Term and Long-Term Price Momentum
    df['price_momentum_diff'] = df['5_day_return'] - df['20_day_return']
    
    # Synthesize All Factors
    df['alpha_factor'] = (df['momentum_volatility'] + df['weighted_intraday_reversal']) * df['price_momentum_diff'] * df['volume_surge']
    
    # Drop rows with NaN values
    df.dropna(inplace=True)
    
    return df['alpha_factor']
```
