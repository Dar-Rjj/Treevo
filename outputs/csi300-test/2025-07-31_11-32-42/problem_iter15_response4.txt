import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Evaluate Close-to-Open Change
    close_open_change = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_metric = intraday_spread - close_open_change
    
    # Incorporate Advanced Volume Impact
    volume_adjusted_metric = integrated_metric * (df['volume'] ** 0.5)
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_stability_ratio = (df['high'] - df['low']) / (df['open'] + df['close'])
    adjusted_metric = volume_adjusted_metric / price_stability_ratio
    
    # Weight by Volume
    volume_weighted_metric = adjusted_metric * df['volume']
    
    # Calculate Volume-Weighted Prices
    vwap_close = (df['close'] * df['volume']).rolling(window=20, min_periods=1).sum() / df['volume'].rolling(window=20, min_periods=1).sum()
    vwap_open = (df['open'] * df['volume']).rolling(window=20, min_periods=1).sum() / df['volume'].rolling(window=20, min_periods=1).sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchhl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    
    # Final Alpha Factor
    alpha_factor = volume_weighted_metric * vwapchhl
    
    return alpha_factor
