```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement
    df['Daily_Range'] = df['high'] - df['low']
    df['Midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Multi-Day Average Midpoint (5-day window)
    df['Avg_Midpoint'] = df['Midpoint'].rolling(window=5).mean()
    
    # Compute Volume Adjusted Momentum
    df['Close_to_Close_Return'] = df['close'].pct_change()
    df['Volume_Adjusted_Momentum'] = df['Close_to_Close_Return'] * df['volume']
    
    # Calculate Price Envelopment
    df['Multi_Day_Range'] = df['Daily_Range'].rolling(window=5).sum()
    df['Multi_Day_Volume'] = df['volume'].rolling(window=5).sum()
    df['Price_Envelopment'] = df['Multi_Day_Range'] / df['Multi_Day_Volume']
    
    # Combine Factors
    df['Alpha_Factor'] = df['Volume_Adjusted_Momentum'] * df['Price_Envelopment']
    
    return df['Alpha_Factor']
```
