import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_gap = intraday_high_low_spread - open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    volume_impact = np.log(df['volume'] + 1)
    integrated_volume_adjusted = integrated_gap * volume_impact
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = integrated_volume_adjusted / price_stability
    
    # Calculate Volume-Weighted Prices
    vwap_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    vwap_open = (df['open'] * df['volume']).sum() / df['volume'].sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    
    # Final Integration
    final_factor = stability_adjusted_metric * vwapchl
    
    return final_factor
