```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor using open, high, low, close, amount, and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series with the alpha factor values indexed by date.
    """
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the average daily trading amount
    avg_amount = df['amount'].rolling(window=20).mean()
    
    # Calculate the average daily trading volume
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ratio of current day's volume to the average volume
    volume_ratio = df['volume'] / avg_volume
    
    # Calculate the ratio of current day's amount to the average amount
    amount_ratio = df['amount'] / avg_amount
    
    # Combine all factors into a single alpha factor
    alpha_factor = (daily_return * rsi * volume_ratio * amount_ratio).dropna()
    
    return alpha_factor
```
