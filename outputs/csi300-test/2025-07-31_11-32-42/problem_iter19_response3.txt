```python
import pandas as pd

def heuristics(df):
    # Integrated Momentum
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    df['integrated_momentum'] = df['close'] - df['20_day_avg_close']

    # Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['5_day_avg_high_low_diff'] = df['high_low_diff'].rolling(window=5).mean()
    df['intraday_volatility'] = df['high_low_diff'] / df['5_day_avg_high_low_diff']

    # Volume Trend
    df['3_day_avg_volume'] = df['volume'].rolling(window=3).mean()
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['volume_trend'] = df['3_day_avg_volume'] - df['7_day_avg_volume']

    # Combine Factors
    df['combined_factors'] = (df['integrated_momentum'] + df['intraday_volatility']) * df['volume_trend']

    # Short-term Price Momentum Difference
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['10_day_return'] = (df['close'] / df['close'].shift(10)) - 1
    df['short_term_momentum_diff'] = df['5_day_return'] - df['10_day_return']

    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factors'] * df['short_term_momentum_diff']

    return df['alpha_factor'].dropna()
```
