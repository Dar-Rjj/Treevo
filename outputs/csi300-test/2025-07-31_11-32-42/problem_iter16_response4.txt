```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Combine Intraday and Opening Gaps
    combined_change = intraday_high_low_spread - close_to_open_change
    
    # Weight by Volume
    weighted_combined_change = combined_change * df['volume']
    
    # Calculate Daily Price Movement
    daily_price_movement = df['high'] - df['low']
    
    # Calculate Multi-Day Average Midpoint (5-day window)
    midpoints = (df['high'] + df['low']) / 2
    multi_day_avg_midpoint = midpoints.rolling(window=5).mean()
    
    # Compute Volume Adjusted Momentum
    close_to_close_return = df['close'].pct_change()
    volume_adjusted_momentum = close_to_close_return * df['volume']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    integrated_volatility = intraday_high_low_spread / open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    advanced_volume_impact = integrated_volatility * np.sqrt(df['volume'])
    
    # Factor in Adjusted Price Range as a Stability Measure
    adjusted_price_range = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_measure = advanced_volume_impact / adjusted_price_range
    
    # Combine Factors
    rolling_weighted_combined_change = weighted_combined_change.rolling(window=5).sum()
    rolling_advanced_volume_impact = advanced_volume_impact.rolling(window=5).sum()
    rolling_stability_measure = stability_measure.rolling(window=5).sum()
    
    final_alpha_factor = (volume_adjusted_momentum * multi_day_avg_midpoint +
                          rolling_weighted_combined_change +
                          rolling_advanced_volume_impact +
                          rolling_stability_measure)
    
    return final_alpha_factor
```
