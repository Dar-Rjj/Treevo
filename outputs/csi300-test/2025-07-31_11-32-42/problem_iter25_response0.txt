```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_volatility / (open_to_close_movement + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Incorporate Advanced Volume Impact
    volume_impact = integrated_metric * (df['volume']) ** (1/3)
    
    # Factor in Adjusted Price Range as a Stability Measure
    stability_measure = (df['high'] - df['low']) / (df['open'] + df['close'] + 1e-6)  # Adding a small constant to avoid division by zero
    volume_adjusted_stability = volume_impact / (stability_measure + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Calculate Volume-Weighted Prices
    vwap_close = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    vwap_open = (df['open'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Synthesize VWAPCHL into the Model
    vwap_chl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    
    # Introduce Dynamic Price Momentum
    five_day_momentum = (df['close'] - df['close'].rolling(window=5).mean()) / (df['close'].rolling(window=5).max() - df['close'].rolling(window=5).min())
    
    # Enhance with Additional Momentum Indicators
    ten_day_ema = df['close'].ewm(span=10, adjust=False).mean()
    ten_day_momentum = (df['close'] - ten_day_ema) / (df['close'].rolling(window=10).max() - df['close'].rolling(window=10).min())
    
    # Integrate Momentum Indicators
    balanced_momentum_signal = (five_day_momentum + ten_day_momentum) / 2
    
    # Final Alpha Factor
    alpha_factor = volume_adjusted_stability * vwap_chl * balanced_momentum_signal
    
    return alpha_factor
```
