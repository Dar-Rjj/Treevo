```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_range = df['high'] - df['low']
    
    # Measure Close Location within Intraday Range
    close_proportional_distance = (df['close'] - df['low']) / intraday_range
    
    # Adjust for Volume Intensity
    weighted_intraday_reversal = close_proportional_distance * df['volume']
    
    # Incorporate Open-Price Impact
    open_proportional_distance = (df['open'] - df['low']) / intraday_range
    combined_weight = weighted_intraday_reversal + (1 - open_proportional_distance)
    
    # Incorporate Previous Day's Momentum
    previous_close = df['close'].shift(1)
    price_change = df['open'] - previous_close
    final_factor = combined_weight * price_change
    
    return final_factor
```
