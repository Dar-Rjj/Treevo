```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_gap = intraday_spread - open_to_close
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = integrated_gap / price_range
    
    # Incorporate Advanced Volume Impact
    volume_impact = np.sqrt(df['volume']) * stability_adjusted_metric
    
    # Calculate Volume-Weighted Prices
    vwc = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    vwo = (df['open'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (vwc - vwo) / (df['high'] - df['low'])
    
    # Evaluate Volume-Weighted Momentum
    vwap = ((df['close'] * df['volume'] + df['open'] * df['volume']) / (df['volume'] + df['volume'])).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    vwap_change = vwap - vwap.shift(1)
    
    # Introduce Dynamic Weighting
    std_vol_5d = df['volume'].rolling(window=5).std()
    vol_ma_5d = df['volume'].rolling(window=5).mean()
    z_score = (df['volume'] - vol_ma_5d) / std_vol_5d
    z_score_weighted_metric = z_score * volume_impact
    
    # Final Integration
    final_factor = volume_impact * vwapchl * vwap_change * z_score_weighted_metric
    
    return final_factor
```
