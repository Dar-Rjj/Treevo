```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Integrated Momentum
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    df['integrated_momentum'] = df['close'] - df['20_day_avg_close']

    # Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['5_day_avg_high_low_diff'] = df['high_low_diff'].rolling(window=5).mean()
    df['intraday_volatility'] = df['high_low_diff'] - df['5_day_avg_high_low_diff']

    # Volume Trend
    df['3_day_avg_volume'] = df['volume'].rolling(window=3).mean()
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['volume_trend'] = df['3_day_avg_volume'] - df['7_day_avg_volume']

    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']

    # Evaluate Open-to-Close Movement
    df['open_to_close_movement'] = df['close'] - df['open']

    # Integrate Intraday and Opening Gaps
    df['integrated_gap'] = df['intraday_high_low_spread'] - df['open_to_close_movement']

    # Incorporate Advanced Volume Impact
    df['log_volume'] = np.log(df['volume'] + 1)
    df['volume_adjusted_metric'] = df['integrated_gap'] * df['log_volume']

    # Factor in Adjusted Price Range as a Stability Measure
    df['price_range_stability'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    df['stability_adjusted_metric'] = df['volume_adjusted_metric'] / df['price_range_stability']

    # Synthesize VWAPCHL into the Model
    df['volume_weighted_close'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['volume_weighted_open'] = (df['open'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['vwapchl'] = (df['volume_weighted_close'] - df['volume_weighted_open']) / (df['high'] - df['low'])
    df['vwapchl_adjusted_metric'] = df['stability_adjusted_metric'] * df['vwapchl']

    # Combine Factors
    df['combined_factor'] = (df['integrated_momentum'] + df['intraday_volatility']) * df['volume_trend'] * df['vwapchl_adjusted_metric']

    # Subtract 10-day Price Return from 5-day Price Return
    df['5_day_price_return'] = (df['close'] / df['close'].shift(5) - 1) * 100
    df['10_day_price_return'] = (df['close'] / df['close'].shift(10) - 1) * 100
    df['short_term_momentum_diff'] = df['5_day_price_return'] - df['10_day_price_return']

    # Multiply short-term price momentum difference with the volume surge difference
    df['final_factor'] = df['short_term_momentum_diff'] * df['combined_factor']

    return df['final_factor']
```
