```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Factors
    df['1_day_momentum'] = df['close'].pct_change()
    df['5_day_momentum'] = df['close'].pct_change(5)
    df['20_day_momentum'] = df['close'].pct_change(20)

    # Calculate Volume Activity Indicators
    df['1_day_volume_diff'] = df['volume'].diff()
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()

    # Calculate Daily Momentum
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    df['daily_momentum'] = df['close'] - df['20_day_avg_close']

    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['5_day_high_low_avg'] = df['high_low_diff'].rolling(window=5).mean()

    # Combine Momentum and Volume Activities
    df['1_day_mom_vol'] = df['1_day_momentum'] * df['1_day_volume_diff']
    df['5_day_mom_vol'] = df['5_day_momentum'] * df['5_day_avg_volume']
    df['20_day_mom_vol'] = df['20_day_momentum'] * df['20_day_avg_volume']

    # Combine All Factors
    df['combined_factors'] = (df['1_day_mom_vol'] + 
                               df['5_day_mom_vol'] + 
                               df['20_day_mom_vol'] - 
                               df['daily_momentum'] + 
                               df['5_day_high_low_avg'])

    return df['combined_factors']
```
