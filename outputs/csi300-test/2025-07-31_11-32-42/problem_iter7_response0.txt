```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Midpoint
    df['midpoint'] = (df['high'] + df['low']) / 2
    
    # Compute Multi-Day Average Midpoint (5 days window)
    df['avg_midpoint_5d'] = df['midpoint'].rolling(window=5).mean()
    
    # Compute Volume Adjusted Momentum
    df['close_to_close_return'] = df['close'].pct_change()
    df['volume_adjusted_momentum'] = df['close_to_close_return'] * df['volume']
    
    # Calculate Short-Term and Long-Term Price Momentum
    df['price_return_5d'] = (df['close'] / df['close'].shift(5)) - 1
    df['price_return_10d'] = (df['close'] / df['close'].shift(10)) - 1
    df['price_return_20d'] = (df['close'] / df['close'].shift(20)) - 1
    df['price_return_50d'] = (df['close'] / df['close'].shift(50)) - 1
    
    # Calculate Volume Surge
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    
    # Calculate Price Envelopment
    df['daily_range'] = df['high'] - df['low']
    df['multi_day_range'] = df['daily_range'].rolling(window=5).sum()
    df['multi_day_volume'] = df['volume'].rolling(window=5).sum()
    df['price_envelopment'] = df['multi_day_range'] / df['multi_day_volume']
    
    # Combine Factors
    df['short_term_price_momentum_diff'] = df['price_return_5d'] - df['price_return_10d']
    df['long_term_price_momentum_diff'] = df['price_return_20d'] - df['price_return_50d']
    df['volume_surge_diff'] = df['avg_volume_5d'] - df['avg_volume_10d']
    
    df['factor'] = (df['short_term_price_momentum_diff'] * 
                    df['long_term_price_momentum_diff'] * 
                    df['volume_surge_diff'] * 
                    df['volume_adjusted_momentum'] * 
                    df['price_envelopment'])
    
    return df['factor']
```
