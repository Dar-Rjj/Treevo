```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between today's close and yesterday's close for short-term momentum
    df['close_diff'] = df['close'].diff()
    
    # Compute the average close price over a trailing window to assess medium-term movement
    window_size = 20
    df['avg_close_20'] = df['close'].rolling(window=window_size).mean()
    
    # Evaluate the ratio of today's close to the minimum close in a trailing window to measure resilience
    df['resilience_ratio'] = df['close'] / df['close'].rolling(window=window_size).min()
    
    # Determine the upper and lower shadow lengths (high - close, open - low) to gauge intra-day volatility
    df['upper_shadow'] = df['high'] - df['close']
    df['lower_shadow'] = df['open'] - df['low']
    
    # Assess the candle body size (close - open) to capture daily price range
    df['candle_body'] = df['close'] - df['open']
    
    # Measure the change in volume from the previous day to detect shifts in buying or selling pressure
    df['volume_change'] = df['volume'].diff()
    
    # Identify days where volume significantly exceeds the trailing average as potential signal indicators
    df['avg_volume_20'] = df['volume'].rolling(window=window_size).mean()
    df['volume_signal'] = (df['volume'] > df['avg_volume_20'] * 1.5).astype(int)
    
    # Develop a factor that combines the momentum of close prices with significant volume changes
    df['momentum_volume_factor'] = df['close_diff'] * df['volume_signal']
    
    # Create a signal when a large price move is accompanied by an unusually high volume
    df['large_price_move'] = (df['close_diff'].abs() > df['close_diff'].abs().rolling(window=window_size).mean() * 1.5).astype(int)
    df['composite_signal'] = df['large_price_move'] * df['volume_signal']
    
    # Final alpha factor: Combine the resilience ratio and the composite signal
    df['alpha_factor'] = df['resilience_ratio'] * df['composite_signal']
    
    return df['alpha_factor']
```
