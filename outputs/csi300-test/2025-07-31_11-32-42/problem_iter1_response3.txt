import pandas as pd

def heuristics_v2(df, N=10, M=5):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Sum of Squared Returns Over N Days
    df['squared_returns'] = df['daily_return'] ** 2
    df['sum_squared_returns'] = df['squared_returns'].rolling(window=N).sum()
    
    # Volume Weighted Moving Average
    df['moving_average_close'] = df['close'].rolling(window=M).mean()
    df['volume_weighted_ma'] = df['moving_average_close'] * df['volume']
    
    # High-Low Range Indicator
    df['high_low_range'] = df['high'] - df['low']
    
    # Combine Indicators
    df['combined_value'] = (df['sum_squared_returns'] * df['volume_weighted_ma']) / df['high_low_range']
    
    # Drop NaN values and return the final factor
    df = df.dropna()
    return df['combined_value']
