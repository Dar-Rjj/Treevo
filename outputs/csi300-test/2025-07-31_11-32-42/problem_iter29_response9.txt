```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Integrated Momentum
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    df['integrated_momentum'] = df['close'] - df['20_day_avg_close']

    # Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['5_day_avg_high_low_diff'] = df['high_low_diff'].rolling(window=5).mean()

    # Open-to-Close Movement
    df['open_to_close_movement'] = df['close'] - df['open']

    # Integrated Intraday and Open-to-Close Metric
    df['integrated_intraday_open_to_close'] = df['5_day_avg_high_low_diff'] / df['open_to_close_movement']

    # Volume Trend
    df['3_day_avg_volume'] = df['volume'].rolling(window=3).mean()
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['volume_trend'] = df['3_day_avg_volume'] - df['7_day_avg_volume']

    # Advanced Volume Impact
    df['cube_root_volume'] = df['volume'] ** (1/3)
    df['advanced_volume_impact'] = df['cube_root_volume'] * df['integrated_intraday_open_to_close']

    # Adjusted Price Range as a Stability Measure
    df['adjusted_price_range'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    df['adjusted_advanced_volume_impact'] = df['advanced_volume_impact'] / df['adjusted_price_range']

    # Dynamic Price Momentum
    df['5_day_avg_close'] = df['close'].rolling(window=5).mean()
    df['5_day_close_range_max'] = df['close'].rolling(window=5).max() - df['close'].rolling(window=5).min()
    df['5_day_momentum'] = (df['close'] - df['5_day_avg_close']) / df['5_day_close_range_max']

    # Additional Momentum Indicators
    df['10_day_avg_close'] = df['close'].rolling(window=10).mean()
    df['10_day_close_range_max'] = df['close'].rolling(window=10).max() - df['close'].rolling(window=10).min()
    df['10_day_momentum'] = (df['close'] - df['10_day_avg_close']) / df['10_day_close_range_max']

    # Integrate Momentum Indicators
    df['integrated_momentum_indicators'] = (df['5_day_momentum'] + df['10_day_momentum']) / 2

    # Final Alpha Factor
    df['5_day_return'] = df['close'].pct_change(5)
    df['10_day_return'] = df['close'].pct_change(10)
    df['final_alpha_factor'] = (
        df['integrated_momentum'] +
        df['5_day_avg_high_low_diff'] +
        df['volume_trend']
    ) * df['adjusted_advanced_volume_impact'] / df['adjusted_price_range'] + df['integrated_momentum_indicators'] - (df['10_day_return'] - df['5_day_return'])

    return df['final_alpha_factor']
```
