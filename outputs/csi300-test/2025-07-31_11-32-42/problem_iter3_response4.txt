```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced Price Momentum Factors
    df['1_day_price_mom'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5_day_price_mom'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['20_day_price_mom'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)

    # Calculate Volume Activity Indicators
    df['1_day_vol_diff'] = df['volume'] - df['volume'].shift(1)
    df['5_day_avg_vol'] = df['volume'].rolling(window=5).mean()
    df['20_day_avg_vol'] = df['volume'].rolling(window=20).mean()

    # Calculate Trade Amount Momentum
    df['1_day_amount_mom'] = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    df['5_day_amount_mom'] = (df['amount'] - df['amount'].shift(5)) / df['amount'].shift(5)
    df['20_day_amount_mom'] = (df['amount'] - df['amount'].shift(20)) / df['amount'].shift(20)

    # Combine Factors
    factor = (
        df['1_day_price_mom'] * df['1_day_vol_diff'] +
        df['5_day_price_mom'] * df['5_day_avg_vol'] +
        df['20_day_price_mom'] * df['20_day_avg_vol'] +
        df['1_day_amount_mom'] + df['1_day_price_mom'] +
        df['5_day_amount_mom'] + df['5_day_price_mom'] +
        df['20_day_amount_mom'] + df['20_day_price_mom']
    )

    return factor
```
