```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Integrated Momentum
    df['20_Day_Avg_Close'] = df['close'].rolling(window=20).mean()
    df['5_Day_Moving_Avg_High'] = df['high'].rolling(window=5).mean()
    df['5_Day_Moving_Avg_Low'] = df['low'].rolling(window=5).mean()
    df['High_Low_Momentum'] = (df['5_Day_Moving_Avg_High'] - df['5_Day_Moving_Avg_Low']).diff().sum()

    # Intraday Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['5_Day_Moving_Avg_HL_Diff'] = df['High_Low_Diff'].rolling(window=5).mean()
    
    # Volume Trend
    df['3_Day_Moving_Avg_Volume'] = df['volume'].rolling(window=3).mean()
    df['7_Day_Moving_Avg_Volume'] = df['volume'].rolling(window=7).mean()
    df['Volume_Trend'] = df['3_Day_Moving_Avg_Volume'] - df['7_Day_Moving_Avg_Volume']
    
    # Daily Price Movement
    df['Daily_Range'] = df['high'] - df['low']
    df['Midpoint'] = (df['high'] + df['low']) / 2
    
    # Multi-Day Average Midpoint
    df['5_Day_Avg_Midpoint'] = df['Midpoint'].rolling(window=5).mean()
    
    # Volume Adjusted Momentum
    df['Close_to_Close_Return'] = df['close'].pct_change()
    df['Volume_Adjusted_Momentum'] = df['Close_to_Close_Return'] * df['volume']
    
    # Short-Term and Long-Term Price Momentum
    df['5_Day_Price_Return'] = df['close'].pct_change(5)
    df['10_Day_Price_Return'] = df['close'].pct_change(10)
    df['20_Day_Price_Return'] = df['close'].pct_change(20)
    df['50_Day_Price_Return'] = df['close'].pct_change(50)
    
    # Volume Surge
    df['5_Day_Avg_Volume'] = df['volume'].rolling(window=5).mean()
    df['10_Day_Avg_Volume'] = df['volume'].rolling(window=10).mean()
    df['Volume_Surge'] = df['5_Day_Avg_Volume'] - df['10_Day_Avg_Volume']
    
    # Price Envelopment
    df['Multi_Day_Range'] = df['Daily_Range'].rolling(window=5).sum()
    df['Multi_Day_Volume'] = df['volume'].rolling(window=5).sum()
    df['Price_Envelopment'] = df['Multi_Day_Range'] * df['Multi_Day_Volume']
    
    # Combine Factors
    df['Combined_Factor'] = (-1 * df['Close_to_Close_Return']) + \
                            df['5_Day_Moving_Avg_HL_Diff'] * \
                            df['Volume_Trend'] - \
                            (df['10_Day_Price_Return'] - df['5_Day_Price_Return']) - \
                            (df['50_Day_Price_Return'] - df['20_Day_Price_Return']) - \
                            (df['10_Day_Avg_Volume'] - df['5_Day_Avg_Volume']) * \
                            (df['10_Day_Price_Return'] - df['5_Day_Price_Return']) * \
                            df['Volume_Surge'] * \
                            df['Volume_Adjusted_Momentum'] * \
                            df['Price_Envelopment']
    
    return df['Combined_Factor'].dropna()
```
