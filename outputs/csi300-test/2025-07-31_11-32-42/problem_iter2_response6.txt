```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement
    df['Daily_Range'] = df['high'] - df['low']
    df['Midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Multi-Day Average Midpoint
    window = 5
    df['Multi_Day_Midpoint_Avg'] = df['Midpoint'].rolling(window=window).mean()
    
    # Compute Volume Adjusted Momentum
    df['Close_to_Close_Return'] = df['close'].pct_change()
    df['Volume_Adjusted_Momentum'] = df['Close_to_Close_Return'] * df['volume']
    
    # Calculate Short-Term Price Momentum
    df['5_day_Price_Return'] = df['close'] / df['close'].shift(5) - 1
    df['10_day_Price_Return'] = df['close'] / df['close'].shift(10) - 1
    
    # Calculate Long-Term Price Momentum
    df['20_day_Price_Return'] = df['close'] / df['close'].shift(20) - 1
    df['50_day_Price_Return'] = df['close'] / df['close'].shift(50) - 1
    
    # Calculate Volume Surge
    df['5_day_Avg_Volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_Avg_Volume'] = df['volume'].rolling(window=10).mean()
    
    # Calculate Price Envelopment
    df['Multi_Day_Range'] = df['Daily_Range'].rolling(window=window).sum()
    df['Multi_Day_Volume'] = df['volume'].rolling(window=window).sum()
    df['Price_Envelopment'] = df['Multi_Day_Range'] / df['Multi_Day_Volume']
    
    # Combine Factors
    df['Short_Term_Momentum_Diff'] = df['5_day_Price_Return'] - df['10_day_Price_Return']
    df['Long_Term_Momentum_Diff'] = df['20_day_Price_Return'] - df['50_day_Price_Return']
    df['Volume_Surge_Diff'] = df['5_day_Avg_Volume'] - df['10_day_Avg_Volume']
    
    df['Momentum_Multiplier'] = df['Short_Term_Momentum_Diff'] * df['Long_Term_Momentum_Diff']
    df['Volume_Surge_Multiplier'] = df['Momentum_Multiplier'] * df['Volume_Surge_Diff'] * df['Volume_Adjusted_Momentum']
    
    df['Alpha_Factor'] = df['Volume_Surge_Multiplier'] * df['Price_Envelopment']
    
    return df['Alpha_Factor'].dropna()
```
