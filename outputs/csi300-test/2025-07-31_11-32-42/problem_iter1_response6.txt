```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['10_day_return'] = (df['close'] / df['close'].shift(10)) - 1
    
    # Calculate Long-Term Price Momentum
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['50_day_return'] = (df['close'] / df['close'].shift(50)) - 1
    
    # Calculate Volume Surge
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    
    # Combine Factors
    df['short_term_momentum_diff'] = df['5_day_return'] - df['10_day_return']
    df['long_term_momentum_diff'] = df['20_day_return'] - df['50_day_return']
    df['volume_surge_diff'] = df['5_day_avg_volume'] - df['10_day_avg_volume']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['short_term_momentum_diff'] * 
                          df['long_term_momentum_diff'] * 
                          df['volume_surge_diff'])
    
    return df['alpha_factor'].dropna()
```
