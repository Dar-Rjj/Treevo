```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open'].shift(1)
    
    # Combine Intraday and Opening Gaps
    combined_change = intraday_high_low_spread - close_to_open_change
    
    # Weight by Volume
    weighted_combined_change = combined_change * df['volume']
    
    # Compute Volume Adjusted Momentum
    close_to_close_return = df['close'].pct_change()
    volume_adjusted_momentum = close_to_close_return * df['volume']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    integrated_volatility = intraday_high_low_spread / open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    advanced_volume_impact = np.log(df['volume'] + 1) * integrated_volatility
    
    # Factor in Adjusted Price Range as a Stability Measure
    adjusted_price_range = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = advanced_volume_impact / adjusted_price_range
    
    # Synthesize VWAPCHL into the Model
    vwap_close = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    vwap_open = (df['open'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    vwapchl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    vwapchl_adjusted_metric = vwapchl * stability_adjusted_metric
    
    # Integrated Momentum
    average_20day_close = df['close'].rolling(window=20).mean()
    integrated_momentum = df['close'] - average_20day_close
    
    # Intraday Volatility
    high_low_diff = df['high'] - df['low']
    avg_5day_high_low_diff = high_low_diff.rolling(window=5).mean()
    
    # Volume Trend
    avg_3day_volume = df['volume'].rolling(window=3).mean()
    avg_7day_volume = df['volume'].rolling(window=7).mean()
    volume_trend = avg_3day_volume - avg_7day_volume
    
    # Combine Factors
    combined_factors = (
        volume_adjusted_momentum * vwapchl_adjusted_metric +
        weighted_combined_change.rolling(window=5).sum() +
        advanced_volume_impact.rolling(window=5).sum() +
        (1 / adjusted_price_range).rolling(window=5).sum() +
        integrated_momentum + avg_5day_high_low_diff
    ) * volume_trend
    
    # Short-term price momentum difference
    short_term_momentum_diff = df['close'].pct_change(5) - df['close'].pct_change(10)
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors * short_term_momentum_diff
    
    return final_alpha_factor
```
