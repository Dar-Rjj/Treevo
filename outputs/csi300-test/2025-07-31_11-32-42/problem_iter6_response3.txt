```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Price Momentum
    df['5_day_price_return'] = df['close'].pct_change(5)
    df['10_day_price_return'] = df['close'].pct_change(10)
    df['20_day_price_return'] = df['close'].pct_change(20)
    df['50_day_price_return'] = df['close'].pct_change(50)

    # Calculate Volume Surge and Average Volume
    df['1_day_volume_diff'] = df['volume'].diff()
    df['5_day_ema_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['10_day_ema_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['20_day_ema_volume'] = df['volume'].ewm(span=20, adjust=False).mean()

    # Combine Factors
    df['short_term_momentum_diff'] = df['5_day_price_return'] - df['10_day_price_return']
    df['long_term_momentum_diff'] = df['20_day_price_return'] - df['50_day_price_return']
    df['volume_trend_diff'] = df['5_day_ema_volume'] - df['10_day_ema_volume']
    
    df['factor'] = (
        (df['short_term_momentum_diff'] * df['long_term_momentum_diff']) +
        (df['5_day_price_return'] * df['5_day_ema_volume']) +
        (df['20_day_price_return'] * df['20_day_ema_volume']) +
        (df['close'].pct_change() * df['1_day_volume_diff'])
    )

    return df['factor']
```
