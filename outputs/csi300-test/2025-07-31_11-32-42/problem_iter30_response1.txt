```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A pandas Series indexed by (date) representing the factor values.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day moving average of the money flow
    money_flow_10d_ma = money_flow.rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the volume
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the typical price
    typical_price_10d_ma = typical_price.rolling(window=10).mean()
    
    # Calculate the money flow ratio
    money_flow_ratio = money_flow_10d_ma / (typical_price_10d_ma * volume_10d_ma)
    
    # Calculate the 10-day rolling standard deviation of the close price
    close_price_std = df['close'].rolling(window=10).std()
    
    # Calculate the final factor
    factor = money_flow_ratio * (df['close'] - df['close'].shift(1)) / close_price_std
    
    return factor
```
