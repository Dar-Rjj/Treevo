import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_volatility - open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    volume_impact = (df['volume'] ** 0.5) * integrated_metric
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    volume_adjusted_stability = volume_impact / price_range_stability
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).cumsum()
    volume_weighted_open = (df['open'] * df['volume']).cumsum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'])
    
    # Introduce Dynamic Price Momentum
    close_5_day_ma = df['close'].rolling(window=5).mean()
    dynamic_price_momentum = (df['close'] - close_5_day_ma) / df['close'].rolling(window=5).max()
    
    # Evaluate Volume-Weighted Momentum
    volume_weighted_price = (df['close'] * df['volume'] + df['open'] * df['volume']).cumsum() / df['volume'].cumsum()
    volume_weighted_momentum = volume_weighted_price - volume_weighted_price.shift(1)
    
    # Final Integration
    final_factor = volume_adjusted_stability * vwapchl * volume_weighted_momentum
    
    return final_factor
