import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement and Midpoint
    df['Daily_Range'] = df['high'] - df['low']
    df['Midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Multi-Day Average Midpoint
    window = 5
    df['Multi_Day_Midpoint'] = df['Midpoint'].rolling(window=window).mean()
    
    # Compute Volume Adjusted Close-to-Close Return
    df['Close_to_Close_Return'] = df['close'] / df['close'].shift(1) - 1
    df['Volume_Adjusted_Return'] = df['Close_to_Close_Return'] * df['volume']
    
    # Calculate Multi-Day Range and Volume
    df['Multi_Day_Range'] = df['Daily_Range'].rolling(window=window).sum()
    df['Multi_Day_Volume'] = df['volume'].rolling(window=window).sum()
    
    # Combine Volume-Adjusted Returns with Price Envelopment
    df['Price_Envelopment'] = df['Multi_Day_Range'] / df['Multi_Day_Volume']
    df['Combined_Factor'] = df['Volume_Adjusted_Return'] * df['Price_Envelopment']
    
    # Calculate Multi-Period Price Momentum
    df['1_day_Price_Momentum'] = df['close'] / df['close'].shift(1) - 1
    df['5_day_Price_Momentum'] = df['close'] / df['close'].shift(5) - 1
    df['20_day_Price_Momentum'] = df['close'] / df['close'].shift(20) - 1
    
    # Calculate Multi-Period Volume Activity
    df['1_day_Volume_Difference'] = df['volume'] - df['volume'].shift(1)
    df['5_day_Average_Volume'] = df['volume'].rolling(window=5).mean()
    df['20_day_Average_Volume'] = df['volume'].rolling(window=20).mean()
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Combined_Factor'] +
        df['1_day_Price_Momentum'] * df['1_day_Volume_Difference'] +
        df['5_day_Price_Momentum'] * df['5_day_Average_Volume'] +
        df['20_day_Price_Momentum'] * df['20_day_Average_Volume']
    )
    
    return df['Final_Alpha_Factor'].dropna()
