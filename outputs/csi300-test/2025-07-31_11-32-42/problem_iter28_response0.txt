```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_gap = intraday_high_low_spread - open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    log_volume = np.log(df['volume'] + 1)
    volume_adjusted_metric = integrated_gap * log_volume
    
    # Factor in Adjusted Price Range as a Stability Measure
    adjusted_price_range = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = volume_adjusted_metric / adjusted_price_range
    
    # Incorporate Momentum Factor
    return_5_days = (df['close'].shift(5) - df['close']) / df['close']
    momentum_factor = stability_adjusted_metric * return_5_days
    
    # Add Intraday Volatility
    high_low_diff_5_day_ma = df['high'] - df['low']
    intraday_volatility = high_low_diff_5_day_ma.rolling(window=5).mean()
    volatility_adjusted_metric = momentum_factor + (intraday_high_low_spread - intraday_volatility)
    
    # Integrate Volume Trend
    volume_3d_ma = df['volume'].rolling(window=3).mean()
    volume_7d_ma = df['volume'].rolling(window=7).mean()
    volume_trend = volume_3d_ma - volume_7d_ma
    volume_trend_adjusted_metric = volatility_adjusted_metric * volume_trend
    
    # Short-Term Price Return
    price_return_5d = (df['close'] / df['close'].shift(5)) - 1
    price_return_10d = (df['close'] / df['close'].shift(10)) - 1
    short_term_price_return = price_return_5d - price_return_10d
    
    # Final Integration
    final_alpha_factor = volume_trend_adjusted_metric * short_term_price_return
    
    return final_alpha_factor
```
