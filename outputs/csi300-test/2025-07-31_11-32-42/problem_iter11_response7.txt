import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close) prices, amount, and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: Factor values indexed by (date).
    """
    # Calculate the daily return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate the average true range (ATR)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = abs(df['high'] - df['close'].shift(1))
    df['tr3'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the money flow multiplier
    df['mf_multiplier'] = (df['close'] - df['low'] - (df['high'] - df['close'])) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    df['mf_volume'] = df['mf_multiplier'] * df['volume']
    
    # Calculate the money flow ratio
    df['mf_ratio'] = df['mf_volume'].rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = df['daily_return'] * df['atr'] * df['mf_ratio']
    
    # Return the alpha factor
    return df['alpha_factor']
