```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Integrated Momentum
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    df['integrated_momentum'] = df['close'] - df['20_day_avg_close']
    
    df['5_day_avg_high'] = df['high'].rolling(window=5).mean()
    df['5_day_avg_low'] = df['low'].rolling(window=5).mean()
    df['high_momentum'] = df['high'] - df['5_day_avg_high']
    df['low_momentum'] = df['low'] - df['5_day_avg_low']
    df['high_low_momentum'] = df['high_momentum'] + df['low_momentum']
    
    # Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['5_day_avg_high_low_diff'] = df['high_low_diff'].rolling(window=5).mean()
    df['intraday_volatility'] = df['5_day_avg_high_low_diff']
    
    # Volume Trend
    df['3_day_avg_volume'] = df['volume'].rolling(window=3).mean()
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['volume_trend'] = df['3_day_avg_volume'] - df['7_day_avg_volume']
    
    # Daily Price Movement
    df['daily_range'] = df['high'] - df['low']
    df['midpoint'] = (df['high'] + df['low']) / 2
    
    # Multi-Day Average Midpoint
    df['5_day_avg_midpoint'] = df['midpoint'].rolling(window=5).mean()
    
    # Volume Adjusted Momentum
    df['close_to_close_return'] = df['close'].pct_change()
    df['volume_adjusted_momentum'] = df['close_to_close_return'] * df['volume']
    
    # Short-Term and Long-Term Price Momentum
    df['5_day_price_return'] = df['close'] / df['close'].shift(5) - 1
    df['10_day_price_return'] = df['close'] / df['close'].shift(10) - 1
    df['20_day_price_return'] = df['close'] / df['close'].shift(20) - 1
    df['50_day_price_return'] = df['close'] / df['close'].shift(50) - 1
    
    # Volume Surge
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_surge'] = df['5_day_avg_volume'] - df['10_day_avg_volume']
    
    # Price Envelopment
    df['multi_day_range'] = df['daily_range'].rolling(window=5).sum()
    df['multi_day_volume'] = df['volume'].rolling(window=5).sum()
    df['price_envelopment'] = df['multi_day_range'] / df['multi_day_volume']
    
    # Combine Factors
    factor = (
        df['integrated_momentum'] +
        df['intraday_volatility']
    ) * df['volume_trend'] - \
    (df['10_day_price_return'] - df['5_day_price_return']) * \
    (df['50_day_price_return'] - df['20_day_price_return']) * \
    df['volume_surge'] * \
    df['volume_adjusted_momentum'] * \
    df['price_envelopment']
    
    return factor
```
