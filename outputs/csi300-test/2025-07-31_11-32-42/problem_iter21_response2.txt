import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_volatility + open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    volume_impact = df['volume'].apply(lambda x: x ** (1/3))
    integrated_volume_adjusted = integrated_metric * volume_impact
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_ratio = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = integrated_volume_adjusted / price_range_ratio
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    volume_weighted_open = (df['open'] * df['volume']).sum() / df['volume'].sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'])
    
    # Introduce Dynamic Price Momentum
    close_5_day_ma = df['close'].rolling(window=5).mean()
    close_5_day_std = df['close'].rolling(window=5).std()
    dynamic_price_momentum = (df['close'] - close_5_day_ma) / close_5_day_std
    
    # Evaluate Volume-Weighted Momentum
    volume_weighted_average_price = (df['close'] * df['volume'] + df['open'] * df['volume']).cumsum() / df['volume'].cumsum()
    volume_weighted_momentum = volume_weighted_average_price.diff().fillna(0)
    
    # Final Integration
    final_factor = stability_adjusted_metric * vwapchl * volume_weighted_momentum
    
    return final_factor
