import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 20-Day Average Close (t-19, t-18, ..., t)
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    
    # Subtract from Today's Close (t) for Long-Term Momentum
    df['long_term_momentum'] = df['close'] - df['20_day_avg_close']
    
    # Compute 5-Day Moving Average of Close Price (t-4, t-3, ..., t)
    df['5_day_avg_close'] = df['close'].rolling(window=5).mean()
    
    # Subtract 5-Day Moving Average from Current Close Price (t)
    df['5_day_diff_close'] = df['close'] - df['5_day_avg_close']
    
    # Divide by Maximum 5-Day Close Price Range for Short-Term Momentum
    df['short_term_momentum'] = df['5_day_diff_close'] / df['close'].rolling(window=5).max()
    
    # Calculate 10-Day Moving Average of Close Price (t-9, t-8, ..., t)
    df['10_day_avg_close'] = df['close'].rolling(window=10).mean()
    
    # Subtract 10-Day Moving Average from Current Close Price (t)
    df['10_day_diff_close'] = df['close'] - df['10_day_avg_close']
    
    # Divide by Maximum 10-Day Close Price Range for Medium-Term Momentum
    df['medium_term_momentum'] = df['10_day_diff_close'] / df['close'].rolling(window=10).max()
    
    # Add 5-Day and 10-Day Momentum Indicators
    df['momentum_sum'] = df['short_term_momentum'] + df['medium_term_momentum']
    
    # Divide by 2 for Balanced Momentum Signal
    df['balanced_momentum'] = df['momentum_sum'] / 2
    
    # Subtract 10-day Price Return from 5-day Price Return for Trend Continuation
    df['price_return_5d'] = df['close'].pct_change(5)
    df['price_return_10d'] = df['close'].pct_change(10)
    df['trend_continuation'] = df['price_return_5d'] - df['price_return_10d']
    
    # Determine High-Low Difference (t)
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute 5-Day Moving Average of High-Low Differences (t-4, t-3, ..., t)
    df['5_day_avg_high_low_diff'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Assess the candle body size (close - open) to capture daily price range
    df['candle_body_size'] = df['close'] - df['open']
    
    # Evaluate Open-to-Close Movement
    df['open_to_close_movement'] = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    df['intraday_volatility_ratio'] = df['5_day_avg_high_low_diff'] / df['open_to_close_movement']
    
    # Measure 3-day Moving Average of Volume (t-2, t-1, t)
    df['3_day_avg_volume'] = df['volume'].rolling(window=3).mean()
    
    # Measure 7-day Moving Average of Volume (t-6, t-5, ..., t)
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    
    # Calculate the Difference between 3-day and 7-day Moving Averages
    df['volume_difference'] = df['3_day_avg_volume'] - df['7_day_avg_volume']
    
    # Multiply Integrated Metric by Cube Root of Volume (t)
    df['advanced_volume_impact'] = df['intraday_volatility_ratio'] * np.cbrt(df['volume'])
    
    # Incorporate Advanced Volume Impact
    df['integrated_metric'] = df['advanced_volume_impact'] * df['volume_difference']
    
    # Compute (High - Low) / (Open + Close) (t)
    df['adjusted_price_range'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    
    # Divide Advanced Volume Impact by this Ratio
    df['stability_adjusted_metric'] = df['integrated_metric'] / df['adjusted_price_range']
    
    # Calculate Volume-Weighted Prices
    df['vwap_close'] = (df['close'] * df['volume']).rolling(window=1).
