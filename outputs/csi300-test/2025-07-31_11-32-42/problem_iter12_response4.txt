```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_volatility / (open_to_close_movement + 1e-8)  # Add small epsilon to avoid division by zero
    
    # Incorporate Advanced Volume Impact
    volume_impact = integrated_metric * (df['volume'] ** (1/3))
    
    # Factor in Adjusted Price Range as a Stability Measure
    adjusted_price_range = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_measure = volume_impact / (adjusted_price_range + 1e-8)  # Add small epsilon to avoid division by zero
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    volume_weighted_open = (df['open'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'])
    
    # Introduce Dynamic Price Momentum
    close_3d_ma = df['close'].rolling(window=3).mean()
    price_momentum = (df['close'] - close_3d_ma) / df['close'].rolling(window=3).std()
    
    # Combine all factors
    alpha_factor = stability_measure * vwapchl * price_momentum
    
    return alpha_factor
```
