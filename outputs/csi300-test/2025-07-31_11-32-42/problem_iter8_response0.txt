```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Price Momentum
    df['5_day_return'] = df['close'].pct_change(5)
    df['10_day_return'] = df['close'].pct_change(10)
    df['20_day_return'] = df['close'].pct_change(20)
    df['50_day_return'] = df['close'].pct_change(50)
    
    # Calculate Volume Surge and Average Volume
    df['1_day_vol_diff'] = df['volume'].diff()
    df['5_day_ema_vol'] = df['volume'].ewm(span=5).mean()
    df['10_day_ema_vol'] = df['volume'].ewm(span=10).mean()
    df['20_day_ema_vol'] = df['volume'].ewm(span=20).mean()
    
    # Calculate Daily Range and Midpoint
    df['daily_range'] = df['high'] - df['low']
    df['midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Multi-Day Average Midpoint
    df['5_day_avg_midpoint'] = df['midpoint'].rolling(window=5).mean()
    
    # Compute Volume Adjusted Momentum
    df['close_to_close_return'] = df['close'].pct_change()
    df['vol_adj_momentum'] = df['close_to_close_return'] * df['volume']
    
    # Calculate Enhanced Price Envelopment
    df['sum_daily_ranges_5'] = df['daily_range'].rolling(window=5).sum()
    df['sum_volumes_5'] = df['volume'].rolling(window=5).sum()
    df['base_envelopment'] = df['sum_daily_ranges_5'] / df['sum_volumes_5']
    df['weighted_sum_high_low'] = (df['high'] + df['low']) / 2
    df['enhanced_price_envelopment'] = df['base_envelopment'] + df['weighted_sum_high_low']
    
    # Combine Factors
    df['price_momentum_diff_5_10'] = df['5_day_return'] - df['10_day_return']
    df['price_momentum_diff_20_50'] = df['20_day_return'] - df['50_day_return']
    df['volume_diff_5_10_ema'] = df['5_day_ema_vol'] - df['10_day_ema_vol']
    df['factor_1'] = df['price_momentum_diff_5_10'] * df['price_momentum_diff_20_50']
    df['factor_2'] = df['close'].pct_change() * df['1_day_vol_diff']
    df['factor_3'] = df['5_day_return'] * df['5_day_ema_vol']
    df['factor_4'] = df['20_day_return'] * df['20_day_ema_vol']
    df['factor_5'] = df['vol_adj_momentum'] * df['enhanced_price_envelopment']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['factor_1'] + df['factor_2'] + df['factor_3'] + df['factor_4'] + df['factor_5']
    
    return df['alpha_factor']
```
