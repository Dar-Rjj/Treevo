```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_range = df['high'] - df['low']
    
    # Measure Close Location within Intraday Range
    close_proportional_distance = (df['close'] - df['low']) / intraday_range
    
    # Adjust for Volume Intensity
    weighted_intraday_reversal = close_proportional_distance * df['volume']
    
    # Incorporate Price Momentum
    previous_close = df['close'].shift(1)
    momentum_indicator = (df['open'] - previous_close) / previous_close
    
    # Combine all components to create the final factor
    alpha_factor = weighted_intraday_reversal + momentum_indicator
    
    return alpha_factor
```
