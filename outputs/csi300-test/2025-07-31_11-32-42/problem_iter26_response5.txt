import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Combine Intraday and Opening Gaps
    combined_change = intraday_high_low_spread - close_to_open_change
    
    # Weight by Volume
    volume_weighted_change = combined_change * df['volume']
    
    # Compute Volume Adjusted Momentum
    close_to_close_return = df['close'].pct_change()
    volume_adjusted_momentum = close_to_close_return * df['volume']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    integrated_volatility = intraday_high_low_spread / open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    log_volume = np.log(df['volume'] + 1)
    advanced_volume_impact = integrated_volatility * log_volume
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = advanced_volume_impact / price_range_stability
    
    # Synthesize VWAPCHL into the Model
    vwap_close = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    vwap_open = (df['open'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    vwap_chl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    vwapchhl_factor = vwap_chl * stability_adjusted_metric
    
    # Combine Factors
    combined_factors = (
        volume_adjusted_momentum * vwapchhl_factor +
        volume_weighted_change.rolling(window=5).sum() +
        advanced_volume_impact.rolling(window=5).sum() +
        price_range_stability.rolling(window=5).sum()
    )
    
    # Add Relative Strength Index (RSI) to the Model
    def calculate_rsi(prices, window=14):
        delta = prices.diff(1)
        gain = (delta.where(delta > 0, 0)).fillna(0)
        loss = (-delta.where(delta < 0, 0)).fillna(0)
        avg_gain = gain.rolling(window=window).mean()
        avg_loss = loss.rolling(window=window).mean()
        rs = avg_gain / avg_loss
        rsi = 100 - (100 / (1 + rs))
        return rsi
    
    rsi = calculate_rsi(df['close'], window=14)
    
    # Include Price Trend Consistency
    ma_3 = df['close'].rolling(window=3).mean()
    ma_10 = df['close'].rolling(window=10).mean()
    trend_consistency = np.where(ma_3 > ma_10, 1, -1)
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors * rsi * trend_consistency
    
    return final_alpha_factor
