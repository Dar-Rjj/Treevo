```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_volatility / (open_to_close_movement + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Incorporate Advanced Volume Impact
    volume_adjusted_metric = integrated_metric * (df['volume'] ** (1/3))
    
    # Factor in Adjusted Price Range as a Stability Measure
    stability_measure = (df['high'] - df['low']) / (df['open'] + df['close'] + 1e-6)  # Adding a small constant to avoid division by zero
    adjusted_metric = volume_adjusted_metric / (stability_measure + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Calculate Volume-Weighted Prices
    vwap_close = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    vwap_open = (df['open'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    
    # Introduce Dynamic Price Momentum
    close_5d_ma = df['close'].rolling(window=5).mean()
    max_5d_close_range = df['close'].rolling(window=5).max() - df['close'].rolling(window=5).min()
    momentum_5d = (df['close'] - close_5d_ma) / (max_5d_close_range + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Enhance with Additional Momentum Indicators
    close_10d_ema = df['close'].ewm(span=10, adjust=False).mean()
    max_10d_close_range = df['close'].rolling(window=10).max() - df['close'].rolling(window=10).min()
    momentum_10d = (df['close'] - close_10d_ema) / (max_10d_close_range + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Integrate and Weight Momentum Indicators
    weighted_momentum = 0.6 * momentum_5d + 0.4 * momentum_10d
    
    # Examine Closing Prices and Momentum
    short_term_momentum = df['close'].diff()
    medium_term_momentum = df['close'].rolling(window=20).mean()
    balanced_momentum_signal = (momentum_5d + momentum_10d) / 2
    
    # Analyze Intraday and Open-to-Close Movements
    upper_shadow = df['high'] - df['close']
    lower_shadow = df['open'] - df['low']
    candle_body_size = df['close'] - df['open']
    
    # Leverage Volume Data for Market Sentiment
    volume_change = df['volume'].pct_change()
    
    # Combine Price and Volume to Create Composite Indicators
    composite_indicator = (short_term_momentum + volume_change) / 2
    
    # Final Alpha Factor
    alpha_factor = (adjusted_metric + vwapchl + weighted_momentum + balanced_momentum_signal + composite_indicator)
    
    return alpha_factor
```
