```python
import pandas as pd

def heuristics_v2(df):
    # Compute daily price return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute high-low range to identify volatility
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute volume change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Identify volume-price momentum
    df['volume_price_momentum'] = df['daily_return'] * df['volume_change'].fillna(0)
    
    # Determine bullish or bearish engulfing pattern
    df['bullish_engulfing'] = ((df['open'] < df['close']) & (df['open'].shift(1) > df['close'].shift(1)) &
                               (df['close'] > df['open'].shift(1)) & (df['open'] < df['close'].shift(1)))
    df['bearish_engulfing'] = ((df['open'] > df['close']) & (df['open'].shift(1) < df['close'].shift(1)) &
                               (df['close'] < df['open'].shift(1)) & (df['open'] > df['close'].shift(1)))
    
    # Calculate doji presence
    df['doji'] = (abs(df['open'] - df['close']) / df['high_low_range']) < 0.05
    
    # Compute typical price
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate money flow
    df['money_flow'] = df['typical_price'] * df['volume']
    
    # Calculate money flow ratio over a 14-day period
    positive_money_flow = df[df['daily_return'] > 0]['money_flow'].rolling(window=14).sum()
    negative_money_flow = df[df['daily_return'] < 0]['money_flow'].rolling(window=14).sum().abs()
    df['money_flow_ratio'] = positive_money_flow / negative_money_flow
    df['money_flow_ratio'].fillna(0, inplace=True)
    
    # Combine factors into a single alpha factor
    alpha_factor = (df['volume_price_momentum'] + 
                    (df['bullish_engulfing'].astype(int) - df['bearish_engulfing'].astype(int)) + 
                    (df['doji'].astype(int) * -0.5) + 
                    df['money_flow_ratio'])
    
    return alpha_factor
```
