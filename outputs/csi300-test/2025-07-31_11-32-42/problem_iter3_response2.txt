import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_range = df['high'] - df['low']
    
    # Measure Close Location within Intraday Range
    proportional_distance_close = (df['close'] - df['low']) / intraday_range
    
    # Adjust for Volume Intensity
    weighted_intraday_reversal_potential = proportional_distance_close * df['volume']
    
    # Incorporate Open-Price Impact
    proportional_distance_open = (df['open'] - df['low']) / intraday_range
    open_price_impact = proportional_distance_open * df['volume']
    
    # Combine with Close-Adjusted Intraday Reversal
    intraday_reversal_potential_index = weighted_intraday_reversal_potential + open_price_impact
    
    return intraday_reversal_potential_index
