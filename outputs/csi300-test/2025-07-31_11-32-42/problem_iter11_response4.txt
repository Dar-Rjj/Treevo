import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate the average volume over the last 10 days
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the relative strength index (RSI) over the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14d'] = 100 - (100 / (1 + rs))
    
    # Calculate the money flow index (MFI) over the last 14 days
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    df['mfi_14d'] = 100 - (100 / (1 + money_ratio))
    
    # Combine the factors to create the final heuristic
    df['heuristic_factor'] = (
        df['daily_return'] * 0.5 +
        (df['avg_volume_10d'] / df['volume']) * 0.3 +
        (df['rsi_14d'] / 100) * 0.1 +
        (df['mfi_14d'] / 100) * 0.1
    )
    
    return df['heuristic_factor']
