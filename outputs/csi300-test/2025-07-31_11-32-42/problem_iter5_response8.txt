```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Change
    close_vol_weighted = (df['close'] * df['volume']).sum()
    open_vol_weighted = (df['open'] * df['volume']).sum()
    vol_weighted_close_open_change = close_vol_weighted - open_vol_weighted
    
    # Calculate Volume-Weighted Intraday High-Low Spread
    high_vol_weighted = (df['high'] * df['volume']).sum()
    low_vol_weighted = (df['low'] * df['volume']).sum()
    vol_weighted_high_low_spread = high_vol_weighted - low_vol_weighted
    
    # Combine Volume-Weighted Changes
    combined_vol_weighted_changes = vol_weighted_close_open_change - vol_weighted_high_low_spread
    
    # Normalize by (High - Low)
    high_low_diff = df['high'] - df['low']
    normalized_factor = combined_vol_weighted_changes / high_low_diff
    
    return normalized_factor
```
