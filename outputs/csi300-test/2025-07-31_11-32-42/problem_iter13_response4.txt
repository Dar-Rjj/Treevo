```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Integrated Momentum
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    df['integrated_momentum'] = df['close'] - df['20_day_avg_close']
    
    df['5_day_high_ma'] = df['high'].rolling(window=5).mean()
    df['5_day_low_ma'] = df['low'].rolling(window=5).mean()
    df['high_momentum'] = df['high'] - df['5_day_high_ma']
    df['low_momentum'] = df['low'] - df['5_day_low_ma']
    df['high_low_momentum'] = df['high_momentum'] + df['low_momentum']
    df['integrated_momentum'] += df['high_low_momentum']

    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_diff'].rolling(window=5).mean()

    # Measure Volume Trend
    df['3_day_volume_ma'] = df['volume'].rolling(window=3).mean()
    df['7_day_volume_ma'] = df['volume'].rolling(window=7).mean()
    df['volume_trend'] = df['3_day_volume_ma'] - df['7_day_volume_ma']

    # Combine Factors
    df['5_day_price_return'] = df['close'].pct_change(5)
    df['10_day_price_return'] = df['close'].pct_change(10)
    df['20_day_price_return'] = df['close'].pct_change(20)
    df['50_day_price_return'] = df['close'].pct_change(50)

    df['short_term_price_momentum'] = df['5_day_price_return'] - df['10_day_price_return']
    df['long_term_price_momentum'] = df['20_day_price_return'] - df['50_day_price_return']

    df['5_day_volume_ma'] = df['volume'].rolling(window=5).mean()
    df['10_day_volume_ma'] = df['volume'].rolling(window=10).mean()
    df['volume_surge'] = df['5_day_volume_ma'] - df['10_day_volume_ma']

    df['price_envelopment'] = (df['high'] + df['low']) / 2.0 - df['close']

    df['alpha_factor'] = (
        df['integrated_momentum'] +
        df['intraday_volatility'] *
        df['volume_trend'] -
        df['short_term_price_momentum'] * df['long_term_price_momentum'] * 
        df['volume_surge'] * 
        df['price_envelopment']
    )

    return df['alpha_factor']
```
