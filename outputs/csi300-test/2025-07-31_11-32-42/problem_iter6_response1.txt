```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Daily Returns
    daily_returns = df['close'] - df['open']
    
    # Volume Adjusted Daily Return
    volume_adjusted_return = daily_returns * df['volume']
    
    # Intraday Range to Volume Ratio
    range_to_volume_ratio = intraday_range / df['volume']
    
    # Closing Price to High-Low Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Sum of Volume Adjusted Daily Return over a 10-day rolling window
    sum_volume_adjusted_return = volume_adjusted_return.rolling(window=10).sum()
    
    # Sum of Intraday Range to Volume Ratio over a 10-day rolling window
    sum_range_to_volume_ratio = range_to_volume_ratio.rolling(window=10).sum()
    
    # Sum of Closing Price to High-Low Range Ratio over a 10-day rolling window
    sum_close_to_range_ratio = close_to_range_ratio.rolling(window=10).sum()
    
    # Average Trading Volume over a 20-day rolling window
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # High-Low Range to Average Volume Ratio
    range_to_avg_volume_ratio = intraday_range / avg_volume
    
    # Combine Factors
    combined_factors = (
        sum_volume_adjusted_return + 
        sum_range_to_volume_ratio + 
        sum_close_to_range_ratio + 
        range_to_avg_volume_ratio
    )
    
    return combined_factors
```
