import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Combine Intraday and Opening Gaps
    combined_change = intraday_high_low_spread - close_to_open_change
    
    # Weight by Volume
    weighted_combined_change = combined_change * df['volume']
    
    # Calculate Daily Price Movement
    daily_range = df['high'] - df['low']
    midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Multi-Day Average Midpoint
    multi_day_avg_midpoint = midpoint.rolling(window=5).mean()
    
    # Compute Volume Adjusted Momentum
    close_to_close_return = df['close'].pct_change()
    volume_adjusted_momentum = close_to_close_return * df['volume']
    
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Daily Returns
    daily_returns = df['close'] - df['open']
    
    # Volume Adjusted Daily Return
    volume_adjusted_daily_return = daily_returns * df['volume']
    
    # Intraday Range to Volume Ratio
    intraday_range_to_volume_ratio = intraday_high_low_range / df['volume']
    
    # Closing Price to High-Low Range Ratio
    closing_price_to_high_low_range_ratio = df['close'] / intraday_high_low_range
    
    # Calculate Enhanced Price Envelopment
    multi_day_range = daily_range.rolling(window=5).sum()
    multi_day_volume = df['volume'].rolling(window=5).sum()
    base_envelopment = multi_day_range / multi_day_volume
    weighted_sum = (df['high'] + df['low']) / 2
    final_envelopment = base_envelopment + weighted_sum
    
    # Combine Factors
    factor1 = volume_adjusted_momentum * final_envelopment
    factor2 = volume_adjusted_daily_return.rolling(window=5).sum()
    factor3 = intraday_range_to_volume_ratio.rolling(window=5).sum()
    factor4 = closing_price_to_high_low_range_ratio.rolling(window=5).sum()
    
    # Final Alpha Factor
    alpha_factor = factor1 + factor2 + factor3 + factor4
    
    return alpha_factor
