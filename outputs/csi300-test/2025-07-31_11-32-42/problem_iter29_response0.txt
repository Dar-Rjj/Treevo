```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Integrated Momentum
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    df['integrated_momentum'] = df['close'] - df['20_day_avg_close']
    
    # Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['5_day_ma_high_low_diff'] = df['high_low_diff'].rolling(window=5).mean()
    df['intraday_volatility'] = df['high_low_diff'] - df['5_day_ma_high_low_diff']
    
    # Open-to-Close Movement
    df['open_to_close_movement'] = df['close'] - df['open']
    
    # Volume Trend
    df['3_day_ma_volume'] = df['volume'].rolling(window=3).mean()
    df['7_day_ma_volume'] = df['volume'].rolling(window=7).mean()
    df['volume_trend'] = df['3_day_ma_volume'] - df['7_day_ma_volume']
    
    # Advanced Volume Impact
    df['cube_root_volume'] = df['volume'] ** (1/3)
    df['advanced_volume_impact'] = df['cube_root_volume'] * df['intraday_volatility'] / df['open_to_close_movement']
    
    # Adjusted Price Range
    df['adjusted_price_range'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    df['adjusted_advanced_volume_impact'] = df['advanced_volume_impact'] / df['adjusted_price_range']
    
    # Dynamic Price Momentum
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    df['dynamic_price_momentum'] = (df['close'] - df['5_day_ma_close']) / (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min())
    
    # Additional Momentum Indicators
    df['10_day_ma_close'] = df['close'].rolling(window=10).mean()
    df['additional_momentum_indicators'] = (df['close'] - df['10_day_ma_close']) / (df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min())
    
    # Intraday High-Low Spread and Open-to-Close Movement
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['integrated_intraday_gaps'] = df['intraday_high_low_spread'] - df['open_to_close_movement']
    
    # Advanced Volume Impact with Log
    df['log_volume'] = np.log(df['volume'] + 1)
    df['integrated_volume_adjusted_metric'] = df['log_volume'] * df['integrated_intraday_gaps']
    df['stability_adjusted_metric'] = df['integrated_volume_adjusted_metric'] / df['adjusted_price_range']
    
    # Volume-Weighted Prices
    df['volume_weighted_close'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['volume_weighted_open'] = (df['open'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['vwap_chl'] = (df['volume_weighted_close'] - df['volume_weighted_open']) / (df['high'] - df['low'])
    
    # Momentum Factor
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['momentum_factor'] = df['5_day_return'] * df['stability_adjusted_metric']
    
    # Final Integration
    df['factor'] = (
        df['integrated_momentum'] +
        df['intraday_volatility'] +
        df['volume_trend'] *
        df['adjusted_advanced_volume_impact'] /
        df['adjusted_price_range'] +
        df['dynamic_price_momentum'] +
        df['additional_momentum_indicators'] +
        df['stability_adjusted_metric'] *
        df['vwap_chl'] *
        df['momentum_factor']
    )
    
    return df['factor']
```
