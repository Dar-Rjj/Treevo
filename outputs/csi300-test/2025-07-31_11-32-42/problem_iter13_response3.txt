```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Midpoint
    df['Midpoint'] = (df['high'] + df['low']) / 2
    
    # Multi-Day Average Midpoint
    window = 5
    df['AvgMidpoint'] = df['Midpoint'].rolling(window=window).mean()
    
    # Compute Volume Adjusted Momentum
    df['VolumeAdjustedMomentum'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Intraday High-Low Range
    df['IntradayRange'] = df['high'] - df['low']
    
    # Daily Returns
    df['DailyReturns'] = df['close'] - df['open']
    
    # Volume Adjusted Daily Return
    df['VolumeAdjustedDailyReturn'] = df['volume'] * df['DailyReturns']
    
    # Intraday Range to Volume Ratio
    df['IntradayRangeToVolumeRatio'] = df['IntradayRange'] / df['volume']
    
    # Closing Price to High-Low Range Ratio
    df['CloseToRangeRatio'] = df['close'] / df['IntradayRange']
    
    # Enhanced Price Envelopment
    sum_daily_ranges = df['IntradayRange'].rolling(window=window).sum()
    sum_volumes = df['volume'].rolling(window=window).sum()
    df['EnhancedPriceEnvelopment'] = (sum_daily_ranges / sum_volumes) + df['Midpoint']
    
    # Volatility-Adjusted Momentum
    std_daily_returns = df['DailyReturns'].rolling(window=window).std()
    df['VolatilityAdjustedMomentum'] = df['VolumeAdjustedMomentum'] / std_daily_returns
    
    # Combine Factors
    df['CombinedFactor1'] = df['VolumeAdjustedMomentum'] * df['EnhancedPriceEnvelopment']
    df['CombinedFactor2'] = df['VolumeAdjustedDailyReturn'].rolling(window=window).sum()
    df['CombinedFactor3'] = df['IntradayRangeToVolumeRatio'].rolling(window=window).sum()
    df['CombinedFactor4'] = df['CloseToRangeRatio'].rolling(window=window).sum()
    df['CombinedFactor5'] = df['VolatilityAdjustedMomentum'].rolling(window=window).sum()
    
    # Final Alpha Factor
    df['AlphaFactor'] = (df['CombinedFactor1'] + 
                         df['CombinedFactor2'] + 
                         df['CombinedFactor3'] + 
                         df['CombinedFactor4'] + 
                         df['CombinedFactor5'])
    
    return df['AlphaFactor']
```
