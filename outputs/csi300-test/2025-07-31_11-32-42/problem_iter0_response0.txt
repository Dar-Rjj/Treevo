```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Volume Weighted Close-to-Open Return.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: Series with the factor values indexed by date.
    """
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(1) - df['open']) / df['open']
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume'].shift(1)
    
    return volume_weighted_return
```
