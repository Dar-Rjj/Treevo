import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_volatility / (open_to_close_movement + 1e-8)
    
    # Incorporate Advanced Volume Impact
    volume_impact = integrated_metric * np.cbrt(df['volume'])
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'] + 1e-8)
    stability_adjusted_metric = volume_impact / (price_range_stability + 1e-8)
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    volume_weighted_open = (df['open'] * df['volume']).sum() / df['volume'].sum()
    
    # Synthesize VWAPCHL into the Model
    vwap_chl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'] + 1e-8)
    
    # Introduce Dynamic Price Momentum
    close_5d_ma = df['close'].rolling(window=5).mean()
    dynamic_price_momentum = (df['close'] - close_5d_ma) / (df['close'].rolling(window=5).max() - df['close'].rolling(window=5).min() + 1e-8)
    
    # Calculate 20-Day Average Close
    close_20d_avg = df['close'].rolling(window=20).mean()
    
    # Daily Momentum
    daily_momentum = df['close'] - close_20d_avg
    
    # Compute 5-day Price Return
    price_return_5d = df['close'].pct_change(periods=5)
    
    # Compute 20-day Price Return
    price_return_20d = df['close'].pct_change(periods=20)
    
    # Calculate 5-Day Moving Average of High-Low Differences
    high_low_diff_5d_avg = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Combine Momentum and Intraday Volatility
    momentum_intraday_volatility = -daily_momentum + intraday_volatility
    
    # Measure Close Location within Intraday Range
    close_location = (df['close'] - df['low']) / (df['high'] - df['low'] + 1e-8)
    
    # Adjust for Volume Intensity
    weighted_intraday_reversal_potential = close_location * df['volume']
    
    # Calculate Volume Surge
    volume_5d_avg = df['volume'].rolling(window=5).mean()
    volume_10d_avg = df['volume'].rolling(window=10).mean()
    
    # Calculate Volume Adjusted Momentum
    close_to_close_return = df['close'].pct_change()
    volume_adjusted_momentum = close_to_close_return * df['volume']
    
    # Synthesize All Factors
    combined_factors = (
        momentum_intraday_volatility +
        weighted_intraday_reversal_potential -
        (price_return_5d - price_return_20d) *
        (price_return_20d - df['close'].pct_change(periods=50)) *
        volume_adjusted_momentum
    )
    
    # Incorporate Intraday High-Low Spread Over Time
    high_5d_avg = df['high'].rolling(window=5).mean()
    low_5d_avg = df['low'].rolling(window=5).mean()
    
    final_factor = combined_factors * (high_5d_avg - low_5d_avg)
    
    return final_factor
