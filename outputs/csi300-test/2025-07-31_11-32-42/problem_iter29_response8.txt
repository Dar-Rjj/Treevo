```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Integrated Momentum and Trend Analysis
    sma_50 = df['close'].rolling(window=50).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    trend_reversal = sma_50 - ema_20
    
    avg_close_20 = df['close'].rolling(window=20).mean()
    close_diff_20 = df['close'] - avg_close_20
    
    # Enhanced Momentum Indicators
    roc_14 = (df['close'] / df['close'].shift(14) - 1) * 100
    
    ma_close_5 = df['close'].rolling(window=5).mean()
    dynamic_momentum_5 = (df['close'] - ma_close_5) / df['close'].rolling(window=5).max()
    
    ma_close_10 = df['close'].rolling(window=10).mean()
    dynamic_momentum_10 = (df['close'] - ma_close_10) / df['close'].rolling(window=10).max()
    
    # Integrated Intraday Volatility
    high_low_diff = df['high'] - df['low']
    avg_high_low_diff_5 = high_low_diff.rolling(window=5).mean()
    
    # Volume Analysis and Trends
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    volume_roc_10 = (df['volume'] / df['volume'].shift(10) - 1) * 100
    
    ma_volume_3 = df['volume'].rolling(window=3).mean()
    ma_volume_7 = df['volume'].rolling(window=7).mean()
    volume_diff_3_7 = ma_volume_3 - ma_volume_7
    
    # Advanced Volume Impact
    cube_root_volume = df['volume'] ** (1/3)
    intraday_open_close_metric = (df['high'] - df['low']) / (df['open'] + df['close'])
    advanced_volume_impact = (high_low_diff / intraday_open_close_metric) * cube_root_volume
    
    # Adjusted Price Range as a Stability Measure
    adjusted_price_range = (df['high'] - df['low']) / (df['open'] + df['close'])
    
    # Final Alpha Factor
    integrated_momentum = (trend_reversal + close_diff_20) / 2
    combined_factors = integrated_momentum + avg_high_low_diff_5 + volume_diff_3_7
    final_factor = (combined_factors * advanced_volume_impact) / adjusted_price_range
    balanced_momentum = (dynamic_momentum_5 + dynamic_momentum_10) / 2
    final_alpha_factor = final_factor + balanced_momentum + (roc_14.shift(5) - roc_14.shift(10))
    
    return final_alpha_factor
```
