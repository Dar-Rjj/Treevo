```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using (open, high, low, close, amount, volume) data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series with the calculated alpha factor values indexed by (date).
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the 20-day moving average of the typical price
    ma_20 = typical_price.rolling(window=20).mean()
    
    # Calculate the 20-day rolling standard deviation of the typical price
    std_20 = typical_price.rolling(window=20).std()
    
    # Calculate the 20-day rolling sum of the volume
    volume_sum_20 = df['volume'].rolling(window=20).sum()
    
    # Calculate the 20-day rolling sum of the amount
    amount_sum_20 = df['amount'].rolling(window=20).sum()
    
    # Calculate the price-to-volume ratio
    price_volume_ratio = (typical_price - ma_20) / (std_20 * volume_sum_20)
    
    # Calculate the price-to-amount ratio
    price_amount_ratio = (typical_price - ma_20) / (std_20 * amount_sum_20)
    
    # Combine the two ratios to form the final factor
    alpha_factor = (price_volume_ratio + price_amount_ratio) / 2
    
    return alpha_factor
```
