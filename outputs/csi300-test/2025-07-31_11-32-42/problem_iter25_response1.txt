```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Integrated Momentum
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    df['integrated_momentum'] = df['close'] - df['20_day_avg_close']
    
    # Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['5_day_avg_high_low_diff'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Open-to-Close Movement
    df['open_to_close'] = df['close'] - df['open']
    
    # Integrated Intraday and Open-to-Close Metric
    df['integrated_intraday_metric'] = df['5_day_avg_high_low_diff'] / df['open_to_close']
    
    # Volume Trend
    df['3_day_avg_volume'] = df['volume'].rolling(window=3).mean()
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['volume_trend'] = df['3_day_avg_volume'] - df['7_day_avg_volume']
    
    # Advanced Volume Impact
    df['cube_root_volume'] = np.cbrt(df['volume'])
    df['advanced_volume_impact'] = df['integrated_intraday_metric'] * df['cube_root_volume']
    
    # Adjusted Price Range as a Stability Measure
    df['adjusted_price_range'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    df['adjusted_advanced_volume_impact'] = df['advanced_volume_impact'] / df['adjusted_price_range']
    
    # Dynamic Price Momentum
    df['5_day_avg_close'] = df['close'].rolling(window=5).mean()
    df['5_day_max_close'] = df['close'].rolling(window=5).max()
    df['dynamic_price_momentum'] = (df['close'] - df['5_day_avg_close']) / df['5_day_max_close']
    
    # Additional Momentum Indicators
    df['10_day_avg_close'] = df['close'].rolling(window=10).mean()
    df['10_day_max_close'] = df['close'].rolling(window=10).max()
    df['additional_momentum'] = (df['close'] - df['10_day_avg_close']) / df['10_day_max_close']
    
    # 5-day and 10-day Price Return
    df['5_day_price_return'] = df['close'].pct_change(periods=5)
    df['10_day_price_return'] = df['close'].pct_change(periods=10)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['integrated_momentum'] +
        df['5_day_avg_high_low_diff'] +
        df['volume_trend'] *
        df['adjusted_advanced_volume_impact'] /
        df['adjusted_price_range'] +
        df['dynamic_price_momentum'] +
        df['additional_momentum'] -
        (df['5_day_price_return'] - df['10_day_price_return'])
    )
    
    return df['final_alpha_factor']
```
