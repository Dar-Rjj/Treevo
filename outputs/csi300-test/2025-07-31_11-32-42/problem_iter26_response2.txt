```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Combine Intraday and Opening Gaps
    combined_change = intraday_high_low_spread - close_to_open_change
    
    # Weight by Volume
    volume_weighted_combined_change = combined_change * df['volume']
    
    # Compute Volume Adjusted Momentum
    close_to_close_return = df['close'].pct_change()
    volume_adjusted_momentum = close_to_close_return * df['volume']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    integrated_volatility = intraday_high_low_spread / open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    log_volume = np.log(df['volume'] + 1)
    advanced_volume_impact = log_volume * integrated_volatility
    
    # Factor in Adjusted Price Range as a Stability Measure
    adjusted_price_range = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = advanced_volume_impact / adjusted_price_range
    
    # Synthesize VWAPCHL into the Model
    vwap_close = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    vwap_open = (df['open'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    vwap_chl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    vwapchhl_stability_adjusted = vwap_chl * stability_adjusted_metric
    
    # Combine Factors
    combined_factors = (
        volume_adjusted_momentum * vwapchhl_stability_adjusted +
        volume_weighted_combined_change.rolling(window=5).sum() +
        advanced_volume_impact.rolling(window=5).sum() +
        adjusted_price_range.rolling(window=5).sum()
    )
    
    # Add Relative Strength Index (RSI) to the Model
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    final_alpha_factor = combined_factors * rsi
    
    return final_alpha_factor
```
