```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Daily Returns
    daily_returns = df['close'] - df['open']
    
    # Volume Adjusted Daily Return
    volume_adjusted_return = daily_returns * df['volume']
    
    # Intraday Range to Volume Ratio
    range_to_volume_ratio = intraday_range / df['volume']
    
    # Sum of Volume Adjusted Daily Return over a 5-day rolling window
    volume_adjusted_return_rolling = volume_adjusted_return.rolling(window=5).sum()
    
    # Sum of Intraday Range to Volume Ratio over a 5-day rolling window
    range_to_volume_ratio_rolling = range_to_volume_ratio.rolling(window=5).sum()
    
    # Combine Factors
    combined_factors = (volume_adjusted_return_rolling + range_to_volume_ratio_rolling) / 2
    
    return combined_factors
```
