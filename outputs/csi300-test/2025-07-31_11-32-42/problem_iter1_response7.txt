```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Volume-Weighted Average Price Change over High and Low (VWAPCHL).
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A series of VWAPCHL values indexed by date.
    """
    # Volume-Weighted Close
    volume_weighted_close = (df['close'] * df['volume']).cumsum()
    
    # Volume-Weighted Open
    volume_weighted_open = (df['open'] * df['volume']).cumsum()
    
    # Difference between High and Low Prices
    high_low_diff = df['high'] - df['low']
    
    # VWAPCHL
    vwapchl = (volume_weighted_close - volume_weighted_open) / high_low_diff
    
    return vwapchl
```
