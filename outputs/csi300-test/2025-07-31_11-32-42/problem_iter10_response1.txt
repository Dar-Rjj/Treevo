```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day Price Returns
    df['5_day_price_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_price_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate 5-day and 10-day Average Volumes
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    
    # Calculate 5-Day Moving Average of High-Low Differences
    df['high_low_diff'] = df['high'] - df['low']
    df['5_day_high_low_volatility'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Momentum Difference
    df['momentum_difference'] = df['5_day_price_return'] - df['20_day_price_return']
    
    # Volume Difference
    df['volume_difference'] = df['5_day_avg_volume'] - df['10_day_avg_volume']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['momentum_difference'] * df['volume_difference']) + df['5_day_high_low_volatility']
    
    return df['final_alpha_factor'].dropna()
```
