import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=5):
    # Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Exponential Weighted Squared Returns Over N Days
    df['squared_returns'] = df['daily_return'] ** 2
    df['ewma_squared_returns'] = df['squared_returns'].ewm(span=N, adjust=False).mean()
    
    # Volume Adjusted Close Price
    df['moving_avg_close'] = df['close'].rolling(window=M).mean()
    df['volume_adjusted_close'] = df['moving_avg_close'] * np.log(df['volume'])
    
    # High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Combined Indicator
    df['combined_indicator'] = (df['ewma_squared_returns'] * df['volume_adjusted_close']) / df['high_low_diff']
    
    # Drop any rows with NaN values
    df.dropna(inplace=True)
    
    return df['combined_indicator']
