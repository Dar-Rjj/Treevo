import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) indexed by date.
    
    Returns:
    pd.Series: A pandas Series indexed by date representing the factor values.
    """
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the typical price
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the 10-day moving average of the typical price
    df['typical_price_10ma'] = df['typical_price'].rolling(window=10).mean()
    
    # Calculate the 10-day rolling standard deviation of the typical price
    df['typical_price_10std'] = df['typical_price'].rolling(window=10).std()
    
    # Calculate the volume weighted average price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the VWAP
    df['vwap_10ma'] = df['vwap'].rolling(window=10).mean()
    
    # Calculate the 10-day rolling standard deviation of the VWAP
    df['vwap_10std'] = df['vwap'].rolling(window=10).std()
    
    # Calculate the factor as a combination of the above metrics
    df['factor'] = (df['close_pct_change'] * df['typical_price_10ma'] / df['typical_price_10std']) + \
                   (df['vwap_10ma'] / df['vwap_10std'])
    
    # Return the factor as a Series
    return df['factor']
