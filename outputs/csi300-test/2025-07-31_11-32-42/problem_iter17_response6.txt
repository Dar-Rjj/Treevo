```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_metric = intraday_high_low_spread - open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    volume_impact = np.log(df['volume'] + 1)
    integrated_volume_adjusted = integrated_metric * volume_impact
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = integrated_volume_adjusted / price_stability
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).sum()
    volume_weighted_open = (df['open'] * df['volume']).sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (volume_weighted_close - volume_weighted_open) / (df['high'].max() - df['low'].min())
    
    # Introduce Enhanced VWAPCHL
    rolling_avg_vwapchl = vwapchl.rolling(window=n).mean()
    enhanced_vwapchl = vwapchl - rolling_avg_vwapchl
    
    # Final Integration
    final_factor = stability_adjusted_metric * enhanced_vwapchl
    
    return final_factor
```
