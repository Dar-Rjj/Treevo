import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Factors
    df['1d_price_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5d_price_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['20d_price_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume Activity Indicators
    df['1d_volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['5d_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['20d_avg_volume'] = df['volume'].rolling(window=20).mean()
    
    # Calculate Daily Momentum
    df['20d_avg_close'] = df['close'].rolling(window=20).mean()
    df['daily_momentum'] = df['close'] - df['20d_avg_close']
    
    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['5d_intraday_volatility'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Combine Momentum and Volume Activities
    df['combined_1d'] = df['1d_price_momentum'] * df['1d_volume_diff']
    df['combined_5d'] = df['5d_price_momentum'] * df['5d_avg_volume']
    df['combined_20d'] = df['20d_price_momentum'] * df['20d_avg_volume']
    
    # Combine All Factors
    df['combined_factors'] = df['combined_1d'] + df['combined_5d'] + df['combined_20d']
    df['daily_momentum_adj'] = -1 * df['daily_momentum']
    
    df['composite_factor'] = df['combined_factors'] + df['daily_momentum_adj'] + df['5d_intraday_volatility']
    
    return df['composite_factor'].dropna()
