```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Daily Returns
    daily_returns = df['close'] - df['open']
    
    # Volume Adjusted Daily Return
    volume_adjusted_daily_return = df['volume'] * daily_returns
    
    # Intraday Range to Volume Ratio
    intraday_range_to_volume_ratio = intraday_high_low_range / df['volume']
    
    # Closing Price to High-Low Range Ratio
    closing_price_to_high_low_range_ratio = df['close'] / intraday_high_low_range
    
    # Combine Intraday Factors
    sum_volume_adjusted_daily_return = volume_adjusted_daily_return.rolling(window=5).sum()
    sum_intraday_range_to_volume_ratio = intraday_range_to_volume_ratio.rolling(window=5).sum()
    sum_closing_price_to_high_low_range_ratio = closing_price_to_high_low_range_ratio.rolling(window=5).sum()
    combined_intraday_factors = (sum_volume_adjusted_daily_return + 
                                 sum_intraday_range_to_volume_ratio + 
                                 sum_closing_price_to_high_low_range_ratio)
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Combine Intraday and Opening Gaps
    combined_intraday_and_opening_gaps = intraday_high_low_spread - close_to_open_change
    
    # Weight by Volume
    weighted_intraday_and_opening_gaps = df['volume'] * combined_intraday_and_opening_gaps
    
    # Calculate Integrated Momentum
    twenty_day_avg_close = df['close'].rolling(window=20).mean()
    momentum_20_day = df['close'] - twenty_day_avg_close
    high_momentum = df['high'] - df['high'].rolling(window=5).mean()
    low_momentum = df['low'] - df['low'].rolling(window=5).mean()
    high_low_momentum = high_momentum + low_momentum
    integrated_momentum = momentum_20_day + high_low_momentum
    
    # Calculate Intraday Volatility
    high_low_diff = df['high'] - df['low']
    intraday_volatility = high_low_diff.rolling(window=5).mean()
    
    # Measure Volume Trend
    three_day_avg_volume = df['volume'].rolling(window=3).mean()
    seven_day_avg_volume = df['volume'].rolling(window=7).mean()
    volume_trend = three_day_avg_volume - seven_day_avg_volume
    
    # Combine Integrated Momentum and Intraday Volatility
    combined_momentum_volatility = -1 * integrated_momentum + intraday_volatility
    
    # Introduce New Factor: Price to Volume Elasticity
    five_day_avg_close = df['close'].rolling(window=5).mean()
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    price_to_volume_elasticity = five_day_avg_close / five_day_avg_volume
    
    # Final Alpha Factor
    final_alpha_factor = (combined_intraday_factors + 
                          weighted_intraday_and_opening_gaps + 
                          combined_momentum_volatility * volume_trend + 
                          price_to_volume_elasticity)
    
    return final_alpha_factor
```
