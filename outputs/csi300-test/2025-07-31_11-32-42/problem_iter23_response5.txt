import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['ma_10_return'] = df['return'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the daily return
    df['ma_30_return'] = df['return'].rolling(window=30).mean()
    
    # Calculate the 10-day standard deviation of the daily return
    df['std_10_return'] = df['return'].rolling(window=10).std()
    
    # Calculate the 30-day standard deviation of the daily return
    df['std_30_return'] = df['return'].rolling(window=30).std()
    
    # Calculate the 10-day moving average of the volume
    df['ma_10_volume'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the volume
    df['ma_30_volume'] = df['volume'].rolling(window=30).mean()
    
    # Calculate the ratio of the 10-day moving average of the volume to the 30-day moving average of the volume
    df['volume_ratio'] = df['ma_10_volume'] / df['ma_30_volume']
    
    # Calculate the volatility-adjusted return
    df['vol_adj_return'] = df['return'] / df['std_30_return']
    
    # Calculate the final alpha factor
    df['alpha_factor'] = (df['vol_adj_return'] * df['volume_ratio']) - (df['ma_10_return'] - df['ma_30_return'])
    
    return df['alpha_factor']
