```python
import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term price momentum
    df['5_day_price_return'] = df['close'].pct_change(5)
    df['10_day_price_return'] = df['close'].pct_change(10)
    df['20_day_price_return'] = df['close'].pct_change(20)
    df['50_day_price_return'] = df['close'].pct_change(50)

    # Calculate volume surge and average volume
    df['1_day_volume_diff'] = df['volume'].diff()
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()

    # Combine factors
    df['short_term_momentum_diff'] = df['5_day_price_return'] - df['10_day_price_return']
    df['long_term_momentum_diff'] = df['20_day_price_return'] - df['50_day_price_return']
    df['volume_diff_5_vs_10'] = df['5_day_avg_volume'] - df['10_day_avg_volume']

    # Multiplicative combinations
    df['factor1'] = df['short_term_momentum_diff'] * df['long_term_momentum_diff']
    df['factor2'] = df['5_day_price_return'] * df['1_day_volume_diff']
    df['factor3'] = df['5_day_price_return'] * df['5_day_avg_volume']
    df['factor4'] = df['20_day_price_return'] * df['20_day_avg_volume']

    # Final alpha factor
    df['alpha_factor'] = (df['factor1'] + df['factor2'] + df['factor3'] + df['factor4']) / 4.0

    return df['alpha_factor'].dropna()
```
