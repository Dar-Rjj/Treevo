```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_metric = intraday_high_low_spread - open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    log_volume = np.log(df['volume'] + 1)
    volume_adjusted_metric = integrated_metric * log_volume
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = volume_adjusted_metric / price_range_stability
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).rolling(window=20, min_periods=1).sum() / df['volume'].rolling(window=20, min_periods=1).sum()
    volume_weighted_open = (df['open'] * df['volume']).rolling(window=20, min_periods=1).sum() / df['volume'].rolling(window=20, min_periods=1).sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'])
    
    # Incorporate Momentum Factor
    momentum_factor = (df['close'].shift(5) / df['close']) - 1
    momentum_adjusted_metric = stability_adjusted_metric * momentum_factor
    
    # Add Volume-Weighted Moving Average (VWMA) Factors
    vwma_5 = (df['close'] * df['volume']).rolling(window=5, min_periods=1).sum() / df['volume'].rolling(window=5, min_periods=1).sum()
    vwma_20 = (df['close'] * df['volume']).rolling(window=20, min_periods=1).sum() / df['volume'].rolling(window=20, min_periods=1).sum()
    
    # Integrate VWMA into the Model
    vwma_factor = vwma_5 - vwma_20
    vwma_adjusted_metric = momentum_adjusted_metric * vwma_factor
    
    # Final Integration
    final_factor = stability_adjusted_metric * vwapchl * momentum_factor * vwma_factor
    
    return final_factor
```
