```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Combine Intraday and Opening Gaps
    combined_change = intraday_high_low_spread - close_to_open_change
    
    # Weight by Volume
    weighted_change = combined_change * df['volume']
    
    # Compute Volume Adjusted Momentum
    close_to_close_return = df['close'].pct_change()
    volume_adjusted_momentum = close_to_close_return * df['volume']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    integrated_metric = intraday_high_low_spread / open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    log_volume = np.log(df['volume'] + 1)
    advanced_volume_impact = integrated_metric * log_volume
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = advanced_volume_impact / price_range_stability
    
    # Synthesize VWAPCHL into the Model
    vwap_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    vwap_open = (df['open'] * df['volume']).sum() / df['volume'].sum()
    vwap_chl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    vwap_adjusted_metric = vwap_chl * stability_adjusted_metric
    
    # Integrated Momentum
    average_20_day_close = df['close'].rolling(window=20).mean()
    integrated_momentum = df['close'] - average_20_day_close
    
    # Intraday Volatility
    high_low_diff = df['high'] - df['low']
    moving_avg_5_day_high_low = high_low_diff.rolling(window=5).mean()
    
    # Volume Trend
    moving_avg_3_day_volume = df['volume'].rolling(window=3).mean()
    moving_avg_7_day_volume = df['volume'].rolling(window=7).mean()
    volume_trend = moving_avg_3_day_volume - moving_avg_7_day_volume
    
    # Combine Factors
    sum_weighted_intraday_opening_gaps = weighted_change.rolling(window=5).sum()
    sum_advanced_volume_impact = advanced_volume_impact.rolling(window=5).sum()
    sum_price_range_stability = price_range_stability.rolling(window=5).sum()
    combined_factors = (volume_adjusted_momentum * vwap_adjusted_metric +
                        sum_weighted_intraday_opening_gaps +
                        sum_advanced_volume_impact +
                        sum_price_range_stability +
                        integrated_momentum + moving_avg_5_day_high_low) * volume_trend
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors * (df['close'].pct_change(5) - df['close'].pct_change(10))
    
    return final_alpha_factor
```
