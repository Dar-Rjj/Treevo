import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the provided market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       and index as date.
    
    Returns:
    pd.Series: A pandas Series indexed by date representing the factor values.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day moving average of money flow
    money_flow_10d_ma = money_flow.rolling(window=10).mean()
    
    # Calculate the 10-day moving average of volume
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day moving average of typical price
    typical_price_10d_ma = typical_price.rolling(window=10).mean()
    
    # Calculate the money flow ratio
    money_flow_ratio = money_flow_10d_ma / (typical_price_10d_ma * volume_10d_ma)
    
    # Calculate the 10-day standard deviation of the money flow ratio
    money_flow_ratio_10d_std = money_flow_ratio.rolling(window=10).std()
    
    # Normalize the money flow ratio
    normalized_money_flow_ratio = (money_flow_ratio - money_flow_ratio.rolling(window=10).mean()) / money_flow_ratio_10d_std
    
    # Return the final factor
    return normalized_money_flow_ratio
