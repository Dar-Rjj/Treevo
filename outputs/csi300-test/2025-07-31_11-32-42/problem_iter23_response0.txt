import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the provided market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with the alpha factor values indexed by (date).
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the closing price
    df['ma_10_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the closing price
    df['ma_30_close'] = df['close'].rolling(window=30).mean()
    
    # Calculate the 10-day exponential moving average of the closing price
    df['ema_10_close'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponential moving average of the closing price
    df['ema_30_close'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the momentum factor (difference between 10-day and 30-day moving averages)
    df['momentum_factor'] = df['ma_10_close'] - df['ma_30_close']
    
    # Calculate the relative strength (RSI) over 14 days
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the volume-weighted average price (VWAP)
    df['vwap'] = (df['amount'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the factor as a combination of the momentum factor and RSI
    df['alpha_factor'] = df['momentum_factor'] * (1 - df['rsi'] / 100)
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
