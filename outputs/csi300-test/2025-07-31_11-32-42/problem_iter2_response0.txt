```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day Price Returns
    df['5_day_return'] = df['close'].pct_change(5)
    df['20_day_return'] = df['close'].pct_change(20)
    
    # Calculate 5-day and 10-day Average Volumes
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate 5-Day Moving Average of High-Low Differences
    df['5_day_avg_high_low_diff'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Combine Factors
    df['short_term_momentum_diff'] = df['5_day_return'] - df['20_day_return']
    df['volume_surge_diff'] = df['5_day_avg_volume'] - df['10_day_avg_volume']
    df['momentum_volume_factor'] = df['short_term_momentum_diff'] * df['volume_surge_diff']
    df['daily_momentum'] = -df['5_day_return']
    df['intraday_volatility'] = df['5_day_avg_high_low_diff']
    
    df['combined_factor'] = df['daily_momentum'] + df['intraday_volatility'] + df['momentum_volume_factor']
    
    return df['combined_factor'].dropna()
```
