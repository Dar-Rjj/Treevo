```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Daily Returns
    df['daily_return'] = df['close'] - df['open']
    
    # Calculate 20-Day Average Close
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    
    # Calculate Integrated Momentum
    df['integrated_momentum'] = df['close'] - df['20_day_avg_close']
    df['5_day_ma_high'] = df['high'].rolling(window=5).mean()
    df['5_day_ma_low'] = df['low'].rolling(window=5).mean()
    df['high_low_momentum'] = (df['high'] - df['5_day_ma_high']) + (df['low'] - df['5_day_ma_low'])
    
    # Volume Adjusted Daily Return
    df['volume_adjusted_daily_return'] = df['volume'] * df['daily_return']
    
    # Intraday Range to Volume Ratio
    df['intraday_range_to_volume_ratio'] = df['intraday_range'] / df['volume']
    
    # Closing Price to High-Low Range Ratio
    df['closing_price_to_intraday_range_ratio'] = df['close'] / df['intraday_range']
    
    # Calculate Intraday Volatility
    df['high_low_difference'] = df['high'] - df['low']
    df['5_day_avg_high_low_difference'] = df['high_low_difference'].rolling(window=5).mean()
    
    # Measure Volume Trend
    df['3_day_vol_avg'] = df['volume'].rolling(window=3).mean()
    df['7_day_vol_avg'] = df['volume'].rolling(window=7).mean()
    df['volume_trend'] = df['3_day_vol_avg'] - df['7_day_vol_avg']
    
    # Combine Factors
    df['sum_volume_adjusted_daily_return'] = df['volume_adjusted_daily_return'].rolling(window=5).sum()
    df['sum_intraday_range_to_volume_ratio'] = df['intraday_range_to_volume_ratio'].rolling(window=5).sum()
    df['sum_closing_price_to_intraday_range_ratio'] = df['closing_price_to_intraday_range_ratio'].rolling(window=5).sum()
    
    # Combine Integrated Momentum, Intraday Volatility, and Volume Trend
    df['combined_factors'] = -1 * df['integrated_momentum'] + df['high_low_momentum'] + df['5_day_avg_high_low_difference']
    df['combined_factors'] *= df['volume_trend']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['sum_volume_adjusted_daily_return'] + 
                                df['sum_intraday_range_to_volume_ratio'] + 
                                df['sum_closing_price_to_intraday_range_ratio']) * df['combined_factors']
    
    return df['final_alpha_factor']
```
