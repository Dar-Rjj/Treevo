```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_metric = intraday_spread - open_close_movement
    
    # Incorporate Advanced Volume Impact
    volume_impact = np.sqrt(df['volume']) * integrated_metric
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_ratio = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = volume_impact / price_range_ratio
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).rolling(window=5, min_periods=1).sum() / df['volume'].rolling(window=5, min_periods=1).sum()
    volume_weighted_open = (df['open'] * df['volume']).rolling(window=5, min_periods=1).sum() / df['volume'].rolling(window=5, min_periods=1).sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'])
    
    # Evaluate Volume-Weighted Momentum
    volume_weighted_avg_price = (df['close'] * df['volume'] + df['open'] * df['volume']).rolling(window=5, min_periods=1).sum() / (2 * df['volume'].rolling(window=5, min_periods=1).sum())
    change_in_volume_weighted_avg_price = volume_weighted_avg_price - volume_weighted_avg_price.shift(1)
    
    # Introduce Dynamic Weighting
    five_day_vol_ma = df['volume'].rolling(window=5, min_periods=1).mean()
    five_day_vol_std = df['volume'].rolling(window=5, min_periods=1).std()
    z_score = (df['volume'] - five_day_vol_ma) / five_day_vol_std
    z_score_weighted_metric = stability_adjusted_metric * z_score
    
    # Final Integration
    final_factor = (stability_adjusted_metric * vwapchl * change_in_volume_weighted_avg_price * z_score_weighted_metric).fillna(0)
    
    return final_factor
```
