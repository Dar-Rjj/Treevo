```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_gap = intraday_spread - open_to_close
    
    # Incorporate Advanced Volume Impact
    volume_impact = np.sqrt(df['volume']) * integrated_gap
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = volume_impact / price_stability
    
    # Calculate Volume-Weighted Prices
    vwap_close = (df['close'] * df['volume']).rolling(window=len(df), min_periods=1).sum() / df['volume'].rolling(window=len(df), min_periods=1).sum()
    vwap_open = (df['open'] * df['volume']).rolling(window=len(df), min_periods=1).sum() / df['volume'].rolling(window=len(df), min_periods=1).sum()
    
    # Synthesize VWAPCHL into the Model
    vwap_chl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    
    # Final Integration
    final_factor = stability_adjusted_metric * vwap_chl
    
    return final_factor
```
