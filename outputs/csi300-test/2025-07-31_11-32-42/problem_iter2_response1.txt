```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_range = df['high'] - df['low']
    
    # Measure Close Location within Intraday Range
    proportional_distance = (df['close'] - df['low']) / intraday_range
    
    # Adjust for Volume Intensity
    weighted_intraday_reversal = proportional_distance * df['volume']
    
    # Calculate Short-Term Price Momentum
    short_term_momentum_5 = (df['close'] / df['close'].shift(5)) - 1
    short_term_momentum_10 = (df['close'] / df['close'].shift(10)) - 1
    
    # Calculate Long-Term Price Momentum
    long_term_momentum_20 = (df['close'] / df['close'].shift(20)) - 1
    long_term_momentum_50 = (df['close'] / df['close'].shift(50)) - 1
    
    # Calculate Volume Surge
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    
    # Combine Factors
    momentum_diff_short = short_term_momentum_5 - short_term_momentum_10
    momentum_diff_long = long_term_momentum_20 - long_term_momentum_50
    volume_surge_diff = avg_volume_5 - avg_volume_10
    
    combined_momentum = (momentum_diff_short * momentum_diff_long) * volume_surge_diff
    final_factor = combined_momentum + weighted_intraday_reversal
    
    return final_factor
```
