```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Price Momentum
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Volume Surge
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    
    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['5_day_moving_avg_high_low_diff'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Combine Momentum and Volume Differences
    df['momentum_diff'] = df['5_day_return'] - df['20_day_return']
    df['volume_surge_diff'] = df['5_day_avg_volume'] - df['10_day_avg_volume']
    df['momentum_volume_factor'] = df['momentum_diff'] * df['volume_surge_diff']
    
    # Calculate Daily Returns
    df['daily_return'] = df['close'] - df['open']
    
    # Volume Adjusted Daily Return
    df['volume_adjusted_return'] = df['volume'] * df['daily_return']
    
    # Intraday Range to Volume Ratio
    df['intraday_range_to_volume_ratio'] = df['high_low_diff'] / df['volume']
    
    # Closing Price to High-Low Range Ratio
    df['close_to_high_low_range_ratio'] = df['close'] / df['high_low_diff']
    
    # Sum of Volume Adjusted Daily Return over a 10-day rolling window
    df['sum_volume_adjusted_return'] = df['volume_adjusted_return'].rolling(window=10).sum()
    
    # Sum of Intraday Range to Volume Ratio over a 10-day rolling window
    df['sum_intraday_range_to_volume_ratio'] = df['intraday_range_to_volume_ratio'].rolling(window=10).sum()
    
    # Sum of Closing Price to High-Low Range Ratio over a 10-day rolling window
    df['sum_close_to_high_low_range_ratio'] = df['close_to_high_low_range_ratio'].rolling(window=10).sum()
    
    # Average Trading Volume over a 20-day rolling window
    df['avg_trading_volume'] = df['volume'].rolling(window=20).mean()
    
    # High-Low Range to Average Volume Ratio
    df['high_low_range_to_avg_volume_ratio'] = df['high_low_diff'] / df['avg_trading_volume']
    
    # Combine Intraday and Volume Adjusted Factors
    df['combined_intraday_volume_factors'] = (
        df['sum_volume_adjusted_return'] +
        df['sum_intraday_range_to_volume_ratio'] +
        df['sum_close_to_high_low_range_ratio'] +
        df['high_low_range_to_avg_volume_ratio']
    )
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['momentum_volume_factor'] +
        df['5_day_moving_avg_high_low_diff'] +
        df['combined_intraday_volume_factors']
    )
    
    return df['final_alpha_factor']
```
