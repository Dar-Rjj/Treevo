```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Combine Intraday and Opening Gaps
    combined_change = intraday_high_low_spread - close_to_open_change
    
    # Weight by Volume
    volume_weighted_change = combined_change * df['volume']
    
    # Compute Volume Adjusted Momentum
    close_to_close_return = (df['close'] / df['close'].shift(1)) - 1
    volume_adjusted_momentum = close_to_close_return * df['volume']
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = volume_weighted_change / price_range_stability
    
    # Synthesize VWAPCHL into the Model
    vwap_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    vwap_open = (df['open'] * df['volume']).sum() / df['volume'].sum()
    vwap_chl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    vwapchls_adjusted_metric = vwap_chl * stability_adjusted_metric
    
    # Combine Factors
    combined_factor = (volume_adjusted_momentum * vwapchls_adjusted_metric).rolling(window=5).sum()
    combined_factor += (combined_change * df['volume']).rolling(window=5).sum()
    combined_factor += (price_range_stability).rolling(window=5).sum()
    
    # Add Relative Strength Index (RSI) to the Model
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    combined_factor *= rsi
    
    # Include Price Trend Consistency
    ma3 = df['close'].rolling(window=3).mean()
    ma10 = df['close'].rolling(window=10).mean()
    trend_consistency = np.where(ma3 > ma10, 1, -1)
    
    # Final Alpha Factor
    final_alpha_factor = combined_factor * trend_consistency
    
    return final_alpha_factor
```
