import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_range = df['high'] - df['low']
    
    # Measure Close Location within Intraday Range
    close_location = (df['close'] - df['low']) / intraday_range
    
    # Adjust for Volume Intensity
    volume_adjusted_close_location = close_location * df['volume']
    
    # Incorporate Open-Price Impact
    open_distance = (df['open'] - df['low']) / intraday_range
    open_adjusted_close_location = volume_adjusted_close_location + open_distance
    
    # Incorporate Previous Day's Momentum
    prev_close = df['close'].shift(1)
    momentum = df['open'] - prev_close
    final_intraday_reversal_momentum = open_adjusted_close_location * momentum
    
    # Calculate Daily Price Movement and Midpoint
    daily_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Multi-Day Average Midpoint (e.g., 5 days)
    multi_day_midpoint = daily_midpoint.rolling(window=5).mean()
    
    # Compute Volume Adjusted Momentum
    close_to_close_return = df['close'].pct_change()
    volume_adjusted_momentum = close_to_close_return * df['volume']
    
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Daily Returns
    daily_returns = df['close'] - df['open']
    
    # Volume Adjusted Daily Return
    volume_adjusted_daily_return = daily_returns * df['volume']
    
    # Intraday Range to Volume Ratio
    intraday_range_volume_ratio = intraday_high_low_range / df['volume']
    
    # Closing Price to High-Low Range Ratio
    closing_price_high_low_range_ratio = df['close'] / intraday_high_low_range
    
    # Calculate Enhanced Price Envelopment
    multi_day_range = intraday_high_low_range.rolling(window=5).sum()
    multi_day_volume = df['volume'].rolling(window=5).sum()
    base_envelopment = multi_day_range / multi_day_volume
    weighted_sum = (df['high'] + df['low']) / 2
    final_envelopment = base_envelopment + weighted_sum
    
    # Combine Factors
    combined_factor_1 = volume_adjusted_momentum * final_envelopment
    combined_factor_2 = volume_adjusted_daily_return.rolling(window=5).sum()
    combined_factor_3 = intraday_range_volume_ratio.rolling(window=5).sum()
    combined_factor_4 = closing_price_high_low_range_ratio.rolling(window=5).sum()
    
    # Final Alpha Factor
    alpha_factor = (combined_factor_1 
                    + combined_factor_2 
                    + combined_factor_3 
                    + combined_factor_4)
    
    return alpha_factor.dropna()
