```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range and Midpoint
    df['daily_range'] = df['high'] - df['low']
    df['midpoint'] = (df['high'] + df['low']) / 2
    
    # Choose a window (e.g., 5 days)
    window = 5
    
    # Calculate Multi-Day Average Midpoint
    df['avg_midpoint'] = df['midpoint'].rolling(window=window).mean()
    
    # Compute Volume Adjusted Momentum
    df['close_to_close_return'] = df['close'].pct_change()
    df['volume_adjusted_momentum'] = df['close_to_close_return'] * df['volume']
    
    # Calculate Enhanced Price Envelopment
    df['sum_daily_ranges'] = df['daily_range'].rolling(window=window).sum()
    df['sum_volumes'] = df['volume'].rolling(window=window).sum()
    df['base_envelopment'] = df['sum_daily_ranges'] / df['sum_volumes']
    df['weighted_sum_high_low'] = (df['high'] + df['low']) / 2
    df['enhanced_price_envelopment'] = df['base_envelopment'] + df['weighted_sum_high_low']
    
    # Combine Factors
    df['alpha_factor'] = df['volume_adjusted_momentum'] * df['enhanced_price_envelopment']
    
    return df['alpha_factor']
```
