```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_gap = intraday_spread - open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    volume_impact = np.sqrt(df['volume']) * integrated_gap
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = volume_impact / price_range
    
    # Calculate Volume-Weighted Prices
    vwap_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    vwap_open = (df['open'] * df['volume']).sum() / df['volume'].sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    
    # Evaluate Volume-Weighted Momentum
    vwap = (df['close'] * df['volume'] + df['open'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_momentum = vwap - vwap.shift(1)
    
    # Final Integration
    factor = stability_adjusted_metric * vwapchl * vwap_momentum
    
    return factor
```
