```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_gap = intraday_high_low_spread - open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    log_volume = np.log(df['volume'] + 1)
    volume_adjusted_metric = integrated_gap * log_volume
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = volume_adjusted_metric / price_range_stability
    
    # Calculate Volume-Weighted Prices
    vwap_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    vwap_open = (df['open'] * df['volume']).sum() / df['volume'].sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    
    # Incorporate Momentum Factor
    momentum_factor = (df['close'].shift(5) / df['close']).fillna(1) - 1
    momentum_adjusted_metric = stability_adjusted_metric * momentum_factor
    
    # Introduce Mean Reversion Component
    sma_20 = df['close'].rolling(window=20).mean()
    mean_reversion_component = (sma_20 - df['close']) / df['close'].rolling(window=20).std()
    
    # Add a Volatility Adjustment
    daily_returns = df['close'].pct_change()
    historical_volatility = daily_returns.rolling(window=20).std()
    volatility_adjustment = historical_volatility * momentum_adjusted_metric
    
    # Final Integration
    final_factor = (momentum_adjusted_metric * vwapchl + mean_reversion_component + volatility_adjustment)
    
    return final_factor
```
