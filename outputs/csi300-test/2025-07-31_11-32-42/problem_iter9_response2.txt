```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Change
    close_weighted = (df['close'] * df['volume']).sum()
    open_weighted = (df['open'] * df['volume']).sum()
    volume_weighted_close_open_change = close_weighted - open_weighted

    # Calculate Volume-Weighted Intraday High-Low Spread
    high_weighted = (df['high'] * df['volume']).sum()
    low_weighted = (df['low'] * df['volume']).sum()
    volume_weighted_high_low_spread = high_weighted - low_weighted

    # Combine Volume-Weighted Changes
    combined_volume_weighted_changes = volume_weighted_close_open_change + volume_weighted_high_low_spread

    # Calculate Volume-Weighted Average Price
    volume_weighted_average_price = (df['close'] * df['volume']).sum() / df['volume'].sum()

    # Adjust for Volume-Weighted Average Price
    adjusted_factor = combined_volume_weighted_changes - volume_weighted_average_price

    return pd.Series(adjusted_factor, index=df.index)
```
