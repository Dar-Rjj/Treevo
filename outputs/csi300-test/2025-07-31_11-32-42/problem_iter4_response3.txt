```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day moving averages of High and Low prices
    df['high_5_day_ma'] = df['high'].rolling(window=5).mean()
    df['low_5_day_ma'] = df['low'].rolling(window=5).mean()
    
    # Calculate High-Low Momentum
    df['high_momentum'] = df['high'] - df['high_5_day_ma']
    df['low_momentum'] = df['low'] - df['low_5_day_ma']
    df['high_low_momentum'] = df['high_momentum'] + df['low_momentum']
    
    # Compute 3-day and 7-day moving averages of Volume
    df['volume_3_day_ma'] = df['volume'].rolling(window=3).mean()
    df['volume_7_day_ma'] = df['volume'].rolling(window=7).mean()
    
    # Calculate difference between 3-day and 7-day moving averages of Volume
    df['volume_trend'] = df['volume_3_day_ma'] - df['volume_7_day_ma']
    
    # Combine High-Low Momentum and Volume Trend
    df['alpha_factor'] = df['high_low_momentum'] * df['volume_trend']
    
    return df['alpha_factor']
```
