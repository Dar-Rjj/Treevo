```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=10, P=14, Q=10, R=10, S=10, T=14, U=20):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_gap = intraday_spread - open_close_movement
    
    # Incorporate Advanced Volume Impact
    log_volume = np.log(df['volume'] + 1)
    volume_adjusted_metric = integrated_gap * log_volume
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = volume_adjusted_metric / price_range_stability
    
    # Calculate Volume-Weighted Prices
    vwap_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    vwap_open = (df['open'] * df['volume']).sum() / df['volume'].sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (vwap_close - vwap_open) / (df['high'] - df['low'])
    vwapchl_adjusted_metric = stability_adjusted_metric * vwapchl
    
    # Incorporate Momentum Indicators
    sma = df['close'].rolling(window=N).mean()
    roc = df['close'].pct_change(M)
    daily_gain = df['close'].diff().where(df['close'].diff() > 0, 0)
    daily_loss = -df['close'].diff().where(df['close'].diff() < 0, 0)
    avg_gain = daily_gain.rolling(window=P).mean()
    avg_loss = daily_loss.rolling(window=P).mean()
    rs = avg_gain / avg_loss
    mom = df['close'].diff(Q)
    
    # Incorporate Volume-Related Features
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    vroc = df['volume'].pct_change(S)
    
    # OBV
    obv = df['volume'].copy()
    obv[0] = 0
    for i in range(1, len(obv)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv.iloc[i] = obv.iloc[i-1] + df['volume'].iloc[i]
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv.iloc[i] = obv.iloc[i-1] - df['volume'].iloc[i]
        else:
            obv.iloc[i] = obv.iloc[i-1]
    
    # Generate Volatility-Based Factors
    tr = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    atr = tr.rolling(window=T).mean()
    std_dev = df['close'].rolling(window=U).std()
    
    # Create Open-Close Relationship Factors
    ocr = df['close'] / df['open']
    gs = df['open'] - df['close'].shift(1)
    
    # Final Integration
    volatility_adjustment = df['close'].pct_change().rolling(window=20).std()
    final_factor = vwapchl_adjusted_metric * (sma + roc + rs + mom) + vwap + vroc + obv + atr + std_dev + ocr + gs
    final_factor = final_factor / volatility_adjustment
    
    return final_factor
```
