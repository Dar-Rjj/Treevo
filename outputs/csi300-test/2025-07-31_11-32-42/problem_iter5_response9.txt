```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Combine Intraday and Opening Gaps
    combined_change = intraday_high_low_spread - close_to_open_change
    
    # Weight by Volume
    weighted_combined_change = combined_change * df['volume']
    
    # Calculate Enhanced Price Momentum Factors
    price_momentum_1d = df['close'] - df['close'].shift(1)
    price_momentum_5d = df['close'] - df['close'].shift(5)
    price_momentum_20d = df['close'] - df['close'].shift(20)
    
    # Calculate Volume Activity Indicators
    volume_difference_1d = df['volume'] - df['volume'].shift(1)
    volume_average_5d = df['volume'].rolling(window=5).mean()
    volume_average_20d = df['volume'].rolling(window=20).mean()
    
    # Calculate Trade Amount Momentum
    amount_momentum_1d = df['amount'] - df['amount'].shift(1)
    amount_momentum_5d = df['amount'] - df['amount'].shift(5)
    amount_momentum_20d = df['amount'] - df['amount'].shift(20)
    
    # Integrate All Components
    integrated_1d = price_momentum_1d * volume_difference_1d + amount_momentum_1d
    integrated_5d = price_momentum_5d * volume_average_5d + amount_momentum_5d
    integrated_20d = price_momentum_20d * volume_average_20d + amount_momentum_20d
    
    # Final Composite Factor
    composite_factor = (integrated_1d + integrated_5d + integrated_20d + weighted_combined_change)
    
    return composite_factor
```
