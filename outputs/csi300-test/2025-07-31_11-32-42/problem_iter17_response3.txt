```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Close Location within Intraday Range
    df['close_location'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Weighted Intraday Reversal Potential
    df['weighted_intraday_reversal'] = df['close_location'] * df['volume']
    
    # Open-Price Impact
    df['open_price_impact'] = (df['open'] - df['low']) / df['intraday_range']
    
    # Final Intraday Reversal and Momentum Potential
    df['price_change'] = df['open'] - df['close'].shift(1)
    df['final_intraday_momentum'] = df['weighted_intraday_reversal'] * df['price_change']
    
    # Calculate 20-Day Average Close
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    
    # Subtract 20-Day Average Close from Today's Close
    df['close_minus_20d_avg'] = df['close'] - df['20_day_avg_close']
    
    # Short-Term vs Long-Term Price Momentum
    df['5_day_return'] = df['close'] / df['close'].shift(5) - 1
    df['20_day_return'] = df['close'] / df['close'].shift(20) - 1
    
    # Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['5_day_volatility'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Combine Momentum and Intraday Volatility
    df['momentum_and_volatility'] = -df['5_day_return'] + df['5_day_volatility']
    
    # Adjust for Volume Intensity
    df['volume_adjusted_momentum'] = (df['close'] / df['close'].shift(1) - 1) * df['volume']
    
    # Calculate Volume Surge
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    
    # Enhanced Short-Term Momentum
    df['enhanced_short_term_momentum'] = df['5_day_return'] * df['5_day_avg_volume']
    
    # Enhanced Long-Term Momentum
    df['enhanced_long_term_momentum'] = df['20_day_return'] * df['10_day_avg_volume']
    
    # Enhanced Momentum Difference
    df['enhanced_momentum_diff'] = df['enhanced_short_term_momentum'] - df['enhanced_long_term_momentum']
    
    # Incorporate Intraday Range into Volume Adjusted Momentum
    df['vol_adj_momentum_range'] = df['volume_adjusted_momentum'] / df['intraday_range']
    
    # Synthesize All Factors
    df['alpha_factor'] = (
        df['momentum_and_volatility'] 
        + df['weighted_intraday_reversal']
        + (df['5_day_return'] - df['20_day_return'])
        - (df['20_day_return'] - df['close'].pct_change(50))
        * df['enhanced_momentum_diff']
        * df['vol_adj_momentum_range']
    )
    
    return df['alpha_factor']
```
