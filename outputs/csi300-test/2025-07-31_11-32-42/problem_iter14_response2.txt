```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       indexed by date.
    
    Returns:
    pd.Series: A series of alpha factor values indexed by date.
    """
    # Calculate daily returns
    df['returns'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of closing prices
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of closing prices
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the difference between the 10-day and 30-day moving averages
    df['ma_diff'] = df['ma_10'] - df['ma_30']
    
    # Calculate the 10-day standard deviation of returns
    df['std_10'] = df['returns'].rolling(window=10).std()
    
    # Calculate the 30-day standard deviation of returns
    df['std_30'] = df['returns'].rolling(window=30).std()
    
    # Calculate the ratio of 10-day to 30-day standard deviations
    df['std_ratio'] = df['std_10'] / df['std_30']
    
    # Calculate the 10-day rolling sum of volume
    df['vol_sum_10'] = df['volume'].rolling(window=10).sum()
    
    # Calculate the 30-day rolling sum of volume
    df['vol_sum_30'] = df['volume'].rolling(window=30).sum()
    
    # Calculate the ratio of 10-day to 30-day rolling sums of volume
    df['vol_ratio'] = df['vol_sum_10'] / df['vol_sum_30']
    
    # Combine the different factors into a single alpha factor
    df['alpha_factor'] = (df['ma_diff'] + df['std_ratio'] + df['vol_ratio']) / 3.0
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
```
