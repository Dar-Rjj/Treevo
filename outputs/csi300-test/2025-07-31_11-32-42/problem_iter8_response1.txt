```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Daily Returns
    daily_returns = df['close'] - df['open']
    
    # Volume Adjusted Daily Return
    volume_adjusted_daily_return = daily_returns * df['volume']
    
    # Intraday Range to Volume Ratio
    intraday_range_to_volume_ratio = intraday_high_low_range / df['volume']
    
    # Closing Price to High-Low Range Ratio
    closing_price_to_high_low_range_ratio = df['close'] / intraday_high_low_range
    
    # Combine Intraday Factors
    intraday_factors_sum = (volume_adjusted_daily_return.rolling(window=5).sum() +
                            intraday_range_to_volume_ratio.rolling(window=5).sum() +
                            closing_price_to_high_low_range_ratio.rolling(window=5).sum())
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Combine Intraday and Opening Gaps
    combined_intraday_opening_gaps = intraday_high_low_spread - close_to_open_change
    
    # Weight by Volume
    weighted_intraday_opening_gaps = combined_intraday_opening_gaps * df['volume']
    
    # Calculate Integrated Momentum
    twenty_day_avg_close = df['close'].rolling(window=20).mean()
    integrated_momentum = df['close'] - twenty_day_avg_close
    high_low_momentum = (df['high'] - df['high'].rolling(window=5).mean()) + (df['low'] - df['low'].rolling(window=5).mean())
    
    # Calculate Intraday Volatility
    high_low_difference = df['high'] - df['low']
    five_day_volatility = high_low_difference.rolling(window=5).mean()
    
    # Measure Volume Trend
    three_day_volume_avg = df['volume'].rolling(window=3).mean()
    seven_day_volume_avg = df['volume'].rolling(window=7).mean()
    volume_trend = three_day_volume_avg - seven_day_volume_avg
    
    # Combine Integrated Momentum and Intraday Volatility
    combined_momentum_volatility = -1 * integrated_momentum + high_low_momentum + five_day_volatility
    
    # Final Alpha Factor
    final_alpha_factor = (intraday_factors_sum + 
                          weighted_intraday_opening_gaps * 
                          combined_momentum_volatility * 
                          volume_trend)
    
    return final_alpha_factor
```
