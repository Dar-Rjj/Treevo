```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and indexed by date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the daily return
    daily_return = df['close'].pct_change()

    # Calculate the 10-day moving average of the close price
    ma_10 = df['close'].rolling(window=10).mean()

    # Calculate the 30-day moving average of the close price
    ma_30 = df['close'].rolling(window=30).mean()

    # Calculate the volume weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).rolling(window=10).mean()

    # Calculate the 10-day standard deviation of the close price
    std_10 = df['close'].rolling(window=10).std()

    # Calculate the 30-day standard deviation of the close price
    std_30 = df['close'].rolling(window=30).std()

    # Calculate the 10-day momentum
    momentum_10 = df['close'] / df['close'].shift(10) - 1

    # Calculate the 30-day momentum
    momentum_30 = df['close'] / df['close'].shift(30) - 1

    # Calculate the 10-day range (high - low)
    range_10 = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()

    # Calculate the 30-day range (high - low)
    range_30 = df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()

    # Calculate the 10-day relative strength index (RSI)
    gain = df['close'].diff().clip(lower=0)
    loss = -df['close'].diff().clip(upper=0)
    avg_gain = gain.rolling(window=10).mean()
    avg_loss = loss.rolling(window=10).mean()
    rs = avg_gain / avg_loss
    rsi_10 = 100 - (100 / (1 + rs))

    # Combine the factors into a single alpha factor
    alpha_factor = (
        (daily_return + momentum_10 + momentum_30) * 0.5 +
        (vwap - ma_10) / std_10 * 0.3 +
        (range_10 / range_30) * 0.2
    )

    return alpha_factor
```
