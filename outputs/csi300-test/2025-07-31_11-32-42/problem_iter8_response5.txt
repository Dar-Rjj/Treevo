```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_range = df['high'] - df['low']
    
    # Measure Close Location within Intraday Range
    close_location = (df['close'] - df['low']) / intraday_range
    
    # Adjust for Volume Intensity
    weighted_intraday_reversal = close_location * df['volume']
    
    # Calculate Short-Term Price Momentum
    short_term_momentum_5 = df['close'].pct_change(5)
    short_term_momentum_10 = df['close'].pct_change(10)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum_20 = df['close'].pct_change(20)
    long_term_momentum_50 = df['close'].pct_change(50)
    
    # Calculate Volume Surge
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    
    # Combine Factors
    short_term_diff = short_term_momentum_5 - short_term_momentum_10
    long_term_diff = long_term_momentum_20 - long_term_momentum_50
    volume_surge_diff = avg_volume_5 - avg_volume_10
    combined_factors = (short_term_diff * long_term_diff) * volume_surge_diff + weighted_intraday_reversal
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Combine Intraday and Opening Gaps
    combined_gaps = intraday_range - close_to_open_change
    
    # Weight by Volume
    volume_weighted_gaps = combined_gaps * df['volume']
    
    # Add to the Integrated Intraday and Momentum Reversal Indicator
    final_factor = combined_factors + volume_weighted_gaps
    
    return final_factor
```
