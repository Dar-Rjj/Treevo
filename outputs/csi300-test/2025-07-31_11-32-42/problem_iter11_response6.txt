```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Daily Returns
    daily_returns = df['close'] - df['open']
    
    # Volume Adjusted Daily Return
    vol_adj_daily_return = df['volume'] * daily_returns
    
    # Intraday Range to Volume Ratio
    range_vol_ratio = intraday_range / df['volume']
    
    # Closing Price to High-Low Range Ratio
    close_high_low_ratio = df['close'] / intraday_range
    
    # Measure Close Location within Intraday Range
    close_location = (df['close'] - df['low']) / intraday_range
    
    # Adjust for Volume Intensity
    weighted_intraday_reversal = df['volume'] * close_location
    
    # Calculate Short-Term Price Momentum
    short_term_momentum_5d = df['close'].pct_change(5)
    short_term_momentum_10d = df['close'].pct_change(10)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum_20d = df['close'].pct_change(20)
    long_term_momentum_50d = df['close'].pct_change(50)
    
    # Calculate Volume Surge
    vol_surge_5d = df['volume'].rolling(window=5).mean()
    vol_surge_10d = df['volume'].rolling(window=10).mean()
    
    # Combine Momentum and Volume Factors
    momentum_diff_short = short_term_momentum_5d - short_term_momentum_10d
    momentum_diff_long = long_term_momentum_20d - long_term_momentum_50d
    volume_diff = vol_surge_5d - vol_surge_10d
    combined_momentum_volume = (momentum_diff_short * momentum_diff_long) * volume_diff
    
    # Sum of Volume Adjusted Daily Return over a 10-day rolling window
    sum_vol_adj_daily_return = vol_adj_daily_return.rolling(window=10).sum()
    
    # Sum of Intraday Range to Volume Ratio over a 10-day rolling window
    sum_range_vol_ratio = range_vol_ratio.rolling(window=10).sum()
    
    # Sum of Closing Price to High-Low Range Ratio over a 10-day rolling window
    sum_close_high_low_ratio = close_high_low_ratio.rolling(window=10).sum()
    
    # Integrate Weighted Intraday Reversal Potential
    integrated_factor = (
        sum_vol_adj_daily_return + 
        sum_range_vol_ratio + 
        sum_close_high_low_ratio + 
        weighted_intraday_reversal + 
        combined_momentum_volume
    )
    
    # Calculate Close-to-Open Change
    close_open_change = df['close'] - df['open']
    
    # Combine Intraday and Opening Gaps
    combined_change = intraday_range - close_open_change
    
    # Weight by Volume
    volume_weighted_change = df['volume'] * combined_change
    
    # Add to the Integrated Intraday and Momentum Reversal Indicator
    final_factor = integrated_factor + volume_weighted_change
    
    return final_factor
```
