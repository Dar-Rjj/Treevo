```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Integrated Momentum
    # 20-Day Average Close
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    # Subtract 20-Day Average Close from Today's Close
    df['momentum_20_day'] = df['close'] - df['20_day_avg_close']
    
    # High-Low Momentum
    df['5_day_avg_high'] = df['high'].rolling(window=5).mean()
    df['5_day_avg_low'] = df['low'].rolling(window=5).mean()
    df['high_momentum'] = df['high'] - df['5_day_avg_high']
    df['low_momentum'] = df['low'] - df['5_day_avg_low']
    df['high_low_momentum'] = df['high_momentum'] + df['low_momentum']
    
    # Combine to get Integrated Momentum
    df['integrated_momentum'] = df['momentum_20_day'] + df['high_low_momentum']

    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_diff'].rolling(window=5).mean()

    # Measure Volume Trend
    df['3_day_avg_volume'] = df['volume'].rolling(window=3).mean()
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['volume_trend'] = df['3_day_avg_volume'] - df['7_day_avg_volume']

    # Combine Integrated Momentum, Intraday Volatility, and Volume Trend
    df['alpha_factor'] = (df['integrated_momentum'] * -1 + df['intraday_volatility']) * df['volume_trend']

    return df['alpha_factor']
```
