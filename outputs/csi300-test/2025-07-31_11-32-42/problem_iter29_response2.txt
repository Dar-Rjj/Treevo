```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_gap = intraday_high_low_spread - open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    log_volume = np.log(df['volume'] + 1)
    volume_impact = integrated_gap * log_volume
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_range_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = volume_impact / price_range_stability
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    volume_weighted_open = (df['open'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Synthesize VWAPCHL into the Model
    vwap_chl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'])
    
    # Incorporate Momentum Factor
    momentum_factor = (df['close'].shift(5) / df['close']) - 1
    momentum_adjusted_metric = stability_adjusted_metric * momentum_factor
    
    # Final Integration
    final_factor = stability_adjusted_metric * vwap_chl * momentum_factor
    
    return final_factor
```
