import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday Volatility and Open-to-Close Movement
    integrated_metric = intraday_volatility / (open_to_close_movement + 1e-6)  # Avoid division by zero
    
    # Incorporate Advanced Volume Impact
    volume_impact = integrated_metric * (df['volume'] ** 0.5)
    
    # Factor in Adjusted Price Range as a Stability Measure
    adjusted_price_range = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = volume_impact / (adjusted_price_range + 1e-6)  # Avoid division by zero
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    volume_weighted_open = (df['open'] * df['volume']).sum() / df['volume'].sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'])
    
    # Introduce Dynamic Price Momentum
    close_5_day_ma = df['close'].rolling(window=5).mean()
    dynamic_momentum = (df['close'] - close_5_day_ma) / (df['close'].rolling(window=5).max() - df['close'].rolling(window=5).min())
    
    # Calculate 20-Day Average Close
    close_20_day_avg = df['close'].rolling(window=20).mean()
    
    # Daily Momentum
    daily_momentum = df['close'] - close_20_day_avg
    
    # Compute 5-day Price Return
    price_return_5_day = df['close'] / df['close'].shift(5) - 1
    
    # Compute 20-day Price Return
    price_return_20_day = df['close'] / df['close'].shift(20) - 1
    
    # Calculate 5-Day Moving Average of High-Low Differences
    high_low_diff_5_day_ma = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Combine Momentum and Intraday Volatility
    combined_momentum_volatility = -daily_momentum + intraday_volatility
    
    # Measure Close Location within Intraday Range
    close_location = (df['close'] - df['low']) / (df['high'] - df['low'] + 1e-6)  # Avoid division by zero
    
    # Adjust for Volume Intensity
    weighted_intraday_reversal_potential = close_location * df['volume']
    
    # Calculate Volume Surge
    volume_5_day_avg = df['volume'].rolling(window=5).mean()
    volume_10_day_avg = df['volume'].rolling(window=10).mean()
    
    # Calculate Volume Adjusted Momentum
    close_to_close_return = df['close'] / df['close'].shift(1) - 1
    volume_adjusted_momentum = close_to_close_return * df['volume']
    
    # Synthesize All Factors
    final_factor = (
        combined_momentum_volatility +
        weighted_intraday_reversal_potential +
        (price_return_5_day - price_return_20_day) -
        (price_return_20_day - df['close'].pct_change(50)) *
        (price_return_5_day - price_return_20_day) *
        volume_adjusted_momentum
    )
    
    # Incorporate Intraday High-Low Spread Over Time
    high_5_day_avg = df['high'].rolling(window=5).mean()
    low_5_day_avg = df['low'].rolling(window=5).mean()
    intraday_high_low_spread = high_5_day_avg - low_5_day_avg
    
    final_factor += intraday_high_low_spread
    
    return final_factor
