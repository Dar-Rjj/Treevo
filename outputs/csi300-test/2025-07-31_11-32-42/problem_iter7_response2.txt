```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Price Momentum
    df['5_day_Price_Return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_Price_Return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Volume Surge
    df['5_day_Avg_Volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_Avg_Volume'] = df['volume'].rolling(window=10).mean()
    
    # Calculate Intraday Volatility
    df['High_Low_Difference'] = df['high'] - df['low']
    df['5_day_Moving_Avg_High_Low_Diff'] = df['High_Low_Difference'].rolling(window=5).mean()
    
    # Combine Factors
    df['Momentum_Difference'] = df['5_day_Price_Return'] - df['20_day_Price_Return']
    df['Volume_Surge_Difference'] = df['5_day_Avg_Volume'] - df['10_day_Avg_Volume']
    df['Combined_Factor'] = (df['Momentum_Difference'] * df['Volume_Surge_Difference']) + df['5_day_Moving_Avg_High_Low_Diff']
    
    return df['Combined_Factor']
```
