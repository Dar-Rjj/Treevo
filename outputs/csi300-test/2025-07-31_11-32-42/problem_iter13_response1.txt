```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Close Location within Intraday Range
    df['close_location'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Weighted Intraday Reversal Potential
    df['weighted_reversal_potential'] = df['close_location'] * df['volume']
    
    # Open-Price Impact
    df['open_price_impact'] = (df['open'] - df['low']) / df['intraday_range']
    
    # Price Change from Previous Day's Close to Current Day's Open
    df['price_change'] = df['open'] - df['close'].shift(1)
    
    # Final Intraday Reversal and Momentum Potential
    df['intraday_reversal_momentum'] = df['weighted_reversal_potential'] * df['price_change']
    
    return df['intraday_reversal_momentum']
```
