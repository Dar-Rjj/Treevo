```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=30):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Evaluate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Integrate Intraday and Opening Gaps
    integrated_metric = intraday_high_low_spread - open_to_close_movement
    
    # Incorporate Advanced Volume Impact
    volume_impact = np.log(df['volume'] + 1)
    integrated_volume_adjusted_metric = integrated_metric * volume_impact
    
    # Factor in Adjusted Price Range as a Stability Measure
    price_stability = (df['high'] - df['low']) / (df['open'] + df['close'])
    stability_adjusted_metric = integrated_volume_adjusted_metric / price_stability
    
    # Calculate Volume-Weighted Prices
    volume_weighted_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    volume_weighted_open = (df['open'] * df['volume']).sum() / df['volume'].sum()
    
    # Synthesize VWAPCHL into the Model
    vwapchl = (volume_weighted_close - volume_weighted_open) / (df['high'] - df['low'])
    
    # Introduce Momentum Component
    sma_close = df['close'].rolling(window=N).mean()
    momentum_component = df['close'] - sma_close
    
    # Calculate Relative Strength
    n_day_cumulative_returns = (df['close'] / df['close'].shift(N)) - 1
    max_n_day_cumulative_returns = n_day_cumulative_returns.rolling(window=M).max()
    relative_strength = n_day_cumulative_returns / max_n_day_cumulative_returns
    
    # Integrate Momentum and Relative Strength with Volume-Adjusted Metrics
    momentum_and_strength_integrated = momentum_component * stability_adjusted_metric * relative_strength
    
    # Final Integration
    final_factor = stability_adjusted_metric * vwapchl + momentum_and_strength_integrated
    
    return final_factor
```
