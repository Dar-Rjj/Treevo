```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Integrated Momentum
    df['20_day_ema_close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['integrated_momentum'] = df['close'] - df['20_day_ema_close']

    # Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['5_day_moving_avg_high_low_diff'] = df['high_low_diff'].rolling(window=5).mean()

    # Open-to-Close Movement
    df['open_to_close_movement'] = df['close'] - df['open']

    # Integrated Intraday and Open-to-Close Metric
    df['integrated_intraday_metric'] = df['5_day_moving_avg_high_low_diff'] / df['open_to_close_movement']

    # Volume Trend
    df['3_day_ema_volume'] = df['volume'].ewm(span=3, adjust=False).mean()
    df['7_day_ema_volume'] = df['volume'].ewm(span=7, adjust=False).mean()
    df['volume_trend'] = df['3_day_ema_volume'] - df['7_day_ema_volume']

    # Advanced Volume Impact
    df['advanced_volume_impact'] = np.sqrt(df['volume']) * df['integrated_intraday_metric']

    # Adjusted Price Range as a Stability Measure
    df['adjusted_price_range'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    df['advanced_volume_impact_adjusted'] = df['advanced_volume_impact'] / df['adjusted_price_range']

    # Dynamic Price Momentum
    df['5_day_ema_close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['5_day_momentum'] = (df['close'] - df['5_day_ema_close']) / df['close'].rolling(window=5).max()

    # Additional Momentum Indicators
    df['10_day_ema_close'] = df['close'].ewm(span=10, adjust=False).mean()
    df['10_day_momentum'] = (df['close'] - df['10_day_ema_close']) / df['close'].rolling(window=10).max()

    # Integrate Momentum Indicators
    df['integrated_momentum_indicators'] = (df['5_day_momentum'] + df['10_day_momentum']) / 2

    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['integrated_momentum'] * 
        df['5_day_moving_avg_high_low_diff'] * 
        df['volume_trend'] * 
        df['advanced_volume_impact_adjusted'] * 
        df['integrated_momentum_indicators'] - 
        (df['close'].pct_change(5) - df['close'].pct_change(10))
    )

    return df['final_alpha_factor']
```
