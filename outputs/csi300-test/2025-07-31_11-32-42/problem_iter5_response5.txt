```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement
    df['daily_range'] = df['high'] - df['low']
    df['midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Multi-Day Average Midpoint (5-day window)
    df['avg_midpoint'] = df['midpoint'].rolling(window=5).mean()
    
    # Compute Volume Adjusted Momentum
    df['close_to_close_return'] = df['close'].pct_change()
    df['volume_adjusted_momentum'] = df['close_to_close_return'] * df['volume']
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Daily Returns
    df['daily_return'] = df['close'] - df['open']
    
    # Volume Adjusted Daily Return
    df['volume_adjusted_daily_return'] = df['daily_return'] * df['volume']
    
    # Intraday Range to Volume Ratio
    df['intraday_range_to_volume_ratio'] = df['intraday_range'] / df['volume']
    
    # Closing Price to High-Low Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['intraday_range']
    
    # Calculate Enhanced Price Envelopment
    df['multi_day_range'] = df['daily_range'].rolling(window=5).sum()
    df['multi_day_volume'] = df['volume'].rolling(window=5).sum()
    df['base_envelopment'] = df['multi_day_range'] / df['multi_day_volume']
    df['weighted_sum'] = (df['high'] + df['low']) / 2
    df['enhanced_envelopment'] = df['base_envelopment'] + df['weighted_sum']
    
    # Combine Factors
    df['combined_factor_1'] = df['volume_adjusted_momentum'] * df['enhanced_envelopment']
    df['combined_factor_2'] = df['volume_adjusted_daily_return'].rolling(window=5).sum()
    df['combined_factor_3'] = df['intraday_range_to_volume_ratio'].rolling(window=5).sum()
    df['combined_factor_4'] = df['close_to_range_ratio'].rolling(window=5).sum()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_factor_1'] + 
                                df['combined_factor_2'] + 
                                df['combined_factor_3'] + 
                                df['combined_factor_4'])
    
    return df['final_alpha_factor']
```
