```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Intraday Reversal Potential Index.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: Series with the Intraday Reversal Potential Index values, indexed by date.
    """
    # Calculate Intraday High-Low Spread
    intraday_range = df['high'] - df['low']
    
    # Measure Close Location within Intraday Range
    proportional_distance = (df['close'] - df['low']) / intraday_range
    
    # Adjust for Volume Intensity
    weighted_intraday_reversal_potential = df['volume'] * proportional_distance
    
    return weighted_intraday_reversal_potential
```
