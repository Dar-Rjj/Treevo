```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using (open, high, low, close, amount, volume) data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) indexed by date.
    
    Returns:
    pd.Series: A Series of alpha factor values indexed by date.
    """
    
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day moving average of the money flow
    money_flow_10d_ma = money_flow.rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the volume
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    
    # Calculate the normalized money flow
    normalized_money_flow = money_flow_10d_ma / volume_10d_ma
    
    # Calculate the 10-day standard deviation of the close price
    close_10d_std = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day mean of the close price
    close_10d_mean = df['close'].rolling(window=10).mean()
    
    # Calculate the 10-day volatility
    close_10d_volatility = close_10d_std / close_10d_mean
    
    # Calculate the final alpha factor
    alpha_factor = (normalized_money_flow - close_10d_volatility) / close_10d_volatility
    
    return alpha_factor
```
