```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Measure Close Location within Intraday Range
    close_location = (df['close'] - df['low']) / intraday_high_low_spread
    
    # Adjust for Volume Intensity
    weighted_intraday_reversal = close_location * df['volume']
    
    # Calculate Short-Term and Long-Term Price Momentum
    short_term_momentum = df['close'].pct_change(5)
    long_term_momentum = df['close'].pct_change(10)
    very_long_term_momentum = df['close'].pct_change(20)
    extremely_long_term_momentum = df['close'].pct_change(50)
    
    # Calculate Volume Surge
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    ten_day_avg_volume = df['volume'].rolling(window=10).mean()
    
    # Combine Momentum and Volume Factors
    momentum_difference_short = short_term_momentum - long_term_momentum
    momentum_difference_long = very_long_term_momentum - extremely_long_term_momentum
    volume_surge_difference = five_day_avg_volume - ten_day_avg_volume
    combined_factors = (momentum_difference_short * momentum_difference_long) * volume_surge_difference
    
    # Integrate Weighted Intraday Reversal Potential
    integrated_intraday_momentum_indicator = combined_factors + weighted_intraday_reversal
    
    # Calculate Daily Range
    daily_range = intraday_high_low_spread
    five_day_avg_daily_range = daily_range.rolling(window=5).mean()
    
    # Calculate Daily Midpoint
    daily_midpoint = (df['high'] + df['low']) / 2
    five_day_avg_midpoint = daily_midpoint.rolling(window=5).mean()
    
    # Volume Adjusted Momentum
    volume_adjusted_momentum = df['close'].pct_change() * df['volume']
    
    # Multi-Day Volume
    multi_day_volume = df['volume'].rolling(window=5).mean()
    
    # Multi-Day Range
    multi_day_range = daily_range.rolling(window=5).mean()
    
    # Combine Factors
    daily_momentum = df['close'].pct_change()
    intraday_volatility = (daily_range - five_day_avg_daily_range) / five_day_avg_daily_range
    volume_trend = (multi_day_volume - five_day_avg_volume) / five_day_avg_volume
    price_envelopment = (df['close'] - five_day_avg_midpoint) / five_day_avg_midpoint
    
    combined_factors_final = (
        -1 * daily_momentum +
        intraday_volatility * volume_trend *
        (short_term_momentum - long_term_momentum) *
        (very_long_term_momentum - extremely_long_term_momentum) *
        (five_day_avg_volume - ten_day_avg_volume) *
        volume_adjusted_momentum * price_envelopment
    )
    
    # Final Integrated Factor
    final_factor = integrated_intraday_momentum_indicator * combined_factors_final
    
    return final_factor
```
