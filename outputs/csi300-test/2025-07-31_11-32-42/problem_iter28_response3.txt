```python
import pandas as pd

def heuristics_v2(df):
    # Integrated Momentum
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    df['integrated_momentum'] = df['close'] - df['20_day_avg_close']

    # Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['5_day_volatility_ma'] = df['high_low_diff'].rolling(window=5).mean()
    df['intraday_volatility'] = df['high_low_diff'] - df['5_day_volatility_ma']

    # Open-to-Close Movement
    df['open_to_close_movement'] = df['close'] - df['open']

    # Volume Trend
    df['3_day_vol_ma'] = df['volume'].rolling(window=3).mean()
    df['7_day_vol_ma'] = df['volume'].rolling(window=7).mean()
    df['volume_trend'] = df['3_day_vol_ma'] - df['7_day_vol_ma']

    # Advanced Volume Impact
    df['cubed_root_volume'] = df['volume'] ** (1/3)
    df['advanced_volume_impact'] = df['cubed_root_volume'] * (df['intraday_volatility'] / df['open_to_close_movement'])

    # Adjusted Price Range
    df['adjusted_price_range'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    df['adjusted_price_ratio'] = df['advanced_volume_impact'] / df['adjusted_price_range']

    # Dynamic Price Momentum
    df['5_day_close_ma'] = df['close'].rolling(window=5).mean()
    df['dynamic_price_momentum'] = (df['close'] - df['5_day_close_ma']) / (df['close'].rolling(window=5).max() - df['close'].rolling(window=5).min())

    # Additional Momentum Indicators
    df['10_day_close_ma'] = df['close'].rolling(window=10).mean()
    df['additional_momentum'] = (df['close'] - df['10_day_close_ma']) / (df['close'].rolling(window=10).max() - df['close'].rolling(window=10).min())

    # 5-day and 10-day Price Return
    df['5_day_price_return'] = df['close'].pct_change(periods=5)
    df['10_day_price_return'] = df['close'].pct_change(periods=10)

    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['integrated_momentum'] + 
        df['intraday_volatility'] + 
        df['volume_trend'] 
    ) * df['advanced_volume_impact'] / df['adjusted_price_ratio'] + \
    df['dynamic_price_momentum'] + \
    df['additional_momentum'] - \
    (df['5_day_price_return'] - df['10_day_price_return'])

    return df['final_alpha_factor']
```
