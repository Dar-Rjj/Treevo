```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 20-Day Average Close
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    
    # Subtract 20-Day Average Close from Today's Close
    df['daily_momentum'] = df['close'] - df['20_day_avg_close']
    
    # Compute 5-day Price Return
    df['5_day_return'] = df['close'] / df['close'].shift(5) - 1
    
    # Compute 20-day Price Return
    df['20_day_return'] = df['close'] / df['close'].shift(20) - 1
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate 5-Day Moving Average of High-Low Differences
    df['intraday_volatility'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Combine Momentum and Intraday Volatility
    df['momentum_intraday_volatility'] = -df['daily_momentum'] + df['intraday_volatility']
    
    # Calculate Intraday High-Low Spread
    df['intraday_range'] = df['high'] - df['low']
    
    # Measure Close Location within Intraday Range
    df['proportional_distance'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Adjust for Volume Intensity
    df['weighted_intraday_reversal_potential'] = df['volume'] * df['proportional_distance']
    
    # Calculate 5-day Average Volume
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    
    # Calculate 10-day Average Volume
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    
    # Calculate Close-to-Close Return
    df['close_to_close_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Multiply Close-to-Close Return by Volume
    df['volume_adjusted_momentum'] = df['close_to_close_return'] * df['volume']
    
    # Synthesize All Factors
    df['synthesized_factors'] = (
        df['momentum_intraday_volatility'] +
        df['weighted_intraday_reversal_potential'] +
        (df['5_day_return'] - df['20_day_return']) +
        ((df['close'] / df['close'].shift(5)) - (df['close'] / df['close'].shift(50))) * 
        ((df['close'] / df['close'].shift(20)) - (df['close'] / df['close'].shift(50)))
    ) * df['volume_adjusted_momentum']
    
    # Calculate 5-Day Average High
    df['5_day_avg_high'] = df['high'].rolling(window=5).mean()
    
    # Calculate 5-Day Average Low
    df['5_day_avg_low'] = df['low'].rolling(window=5).mean()
    
    # Calculate Intraday High-Low Spread Ratio
    df['intraday_high_low_spread_ratio'] = df['5_day_avg_high'] / df['5_day_avg_low']
    
    # Incorporate Intraday High-Low Spread Ratio
    df['final_factor'] = df['synthesized_factors'] + df['intraday_high_low_spread_ratio'] * df['intraday_volatility']
    
    return df['final_factor'].dropna()
```
