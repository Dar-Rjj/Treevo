```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day moving averages of High and Low prices
    df['5_day_avg_high'] = df['high'].rolling(window=5).mean()
    df['5_day_avg_low'] = df['low'].rolling(window=5).mean()

    # Calculate High-Low Momentum
    df['high_momentum'] = df['high'] - df['5_day_avg_high']
    df['low_momentum'] = df['low'] - df['5_day_avg_low']
    df['high_low_momentum'] = df['high_momentum'] + df['low_momentum']

    # Calculate Close Price Momentum
    df['1_day_close_momentum'] = df['close'].diff(1)
    df['5_day_close_momentum'] = df['close'] - df['close'].shift(5)
    df['20_day_close_momentum'] = df['close'] - df['close'].shift(20)

    # Calculate 3-day and 7-day moving averages of Volume
    df['3_day_avg_volume'] = df['volume'].rolling(window=3).mean()
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()

    # Calculate the difference between 3-day and 7-day moving averages of Volume
    df['volume_trend'] = df['3_day_avg_volume'] - df['7_day_avg_volume']

    # Calculate 1-day Volume Difference
    df['1_day_volume_difference'] = df['volume'].diff(1)

    # Calculate 5-day and 20-day Average Volume
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()

    # Combine Momentum, Volume Trend, and Volume Activity
    df['combined_high_low_volume'] = df['high_low_momentum'] * df['1_day_volume_difference']
    df['combined_1_day_close_volume'] = df['1_day_close_momentum'] * df['1_day_volume_difference']
    df['combined_5_day_close_volume'] = df['5_day_close_momentum'] * df['5_day_avg_volume']
    df['combined_20_day_close_volume'] = df['20_day_close_momentum'] * df['20_day_avg_volume']

    # Final alpha factor
    df['alpha_factor'] = (df['combined_high_low_volume'] + df['combined_1_day_close_volume'] 
                          + df['combined_5_day_close_volume'] + df['combined_20_day_close_volume'])

    return df['alpha_factor']
```
