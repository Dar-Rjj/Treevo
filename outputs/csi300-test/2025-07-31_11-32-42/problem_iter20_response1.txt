import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using past and current data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) indexed by date.
    
    Returns:
    pd.Series: A pandas Series representing the factor values indexed by date.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the close price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the 10-day moving average of the volume
    df['vol_ma_10'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the volume
    df['vol_ma_30'] = df['volume'].rolling(window=30).mean()
    
    # Calculate the volume-weighted price
    df['vwp'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Calculate the 10-day moving average of the volume-weighted price
    df['vwp_ma_10'] = df['vwp'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the volume-weighted price
    df['vwp_ma_30'] = df['vwp'].rolling(window=30).mean()
    
    # Calculate the factor value
    df['factor'] = (df['ma_10'] - df['ma_30']) / df['ma_30'] * (df['vol_ma_10'] / df['vol_ma_30']) * (df['vwp_ma_10'] / df['vwp_ma_30'])
    
    # Return the factor values
    return df['factor']
