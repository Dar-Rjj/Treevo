```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Price Momentum
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['10_day_return'] = (df['close'] / df['close'].shift(10)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['60_day_return'] = (df['close'] / df['close'].shift(60)) - 1

    # Calculate Volume Surge and Average Volume
    df['1_day_volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['5_day_ema_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['10_day_ema_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['30_day_ema_volume'] = df['volume'].ewm(span=30, adjust=False).mean()

    # Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)

    # Exponential Weighted Squared Returns Over N Days
    df['squared_daily_return'] = df['daily_return'] ** 2
    df['exp_weighted_squared_returns'] = df['squared_daily_return'].ewm(span=30, adjust=False).mean()

    # Volume Adjusted Close Price
    df['moving_avg_close'] = df['close'].rolling(window=30).mean()
    df['log_volume'] = np.log(df['volume'])
    df['volume_adjusted_close'] = df['moving_avg_close'] * df['log_volume']

    # High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']

    # Price Range
    df['price_range'] = df['high'] / df['low']

    # Combine Factors
    df['price_momentum_diff_1'] = df['5_day_return'] - df['10_day_return']
    df['price_momentum_diff_2'] = df['20_day_return'] - df['60_day_return']
    df['volume_diff_1'] = df['5_day_ema_volume'] - df['10_day_ema_volume']
    
    factor = (
        (df['price_momentum_diff_1'] * df['price_momentum_diff_2']) +
        (df['5_day_return'] * df['1_day_volume_diff']) +
        (df['20_day_return'] * df['5_day_ema_volume']) +
        (df['exp_weighted_squared_returns'] * df['volume_adjusted_close'])
    ) / df['high_low_diff'] * df['price_range']

    # Additional Factor: Volume Volatility
    df['5_day_vol_std'] = df['volume'].rolling(window=5).std()
    df['10_day_vol_std'] = df['volume'].rolling(window=10).std()
    df['30_day_vol_std'] = df['volume'].rolling(window=30).std()

    factor += df['5_day_vol_std'] + df['10_day_vol_std'] + df['30_day_vol_std']

    return factor.dropna()
```
