```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Daily Returns
    daily_returns = df['close'] - df['open']
    
    # Volume Adjusted Daily Return
    volume_adjusted_daily_return = daily_returns * df['volume']
    
    # Intraday Range to Volume Ratio
    intraday_range_to_volume_ratio = intraday_high_low_range / df['volume']
    
    # Closing Price to High-Low Range Ratio
    closing_price_to_high_low_range_ratio = df['close'] / intraday_high_low_range
    
    # Sum of Volume Adjusted Daily Return over a 10-day rolling window
    sum_volume_adjusted_daily_return = volume_adjusted_daily_return.rolling(window=10).sum()
    
    # Sum of Intraday Range to Volume Ratio over a 10-day rolling window
    sum_intraday_range_to_volume_ratio = intraday_range_to_volume_ratio.rolling(window=10).sum()
    
    # Sum of Closing Price to High-Low Range Ratio over a 10-day rolling window
    sum_closing_price_to_high_low_range_ratio = closing_price_to_high_low_range_ratio.rolling(window=10).sum()
    
    # High-Low Range to Average Volume Ratio
    average_volume_20_day = df['volume'].rolling(window=20).mean()
    high_low_range_to_average_volume_ratio = intraday_high_low_range / average_volume_20_day
    
    # Final Alpha Factor
    alpha_factor = (sum_volume_adjusted_daily_return + 
                    sum_intraday_range_to_volume_ratio + 
                    sum_closing_price_to_high_low_range_ratio + 
                    high_low_range_to_average_volume_ratio)
    
    return alpha_factor
```
