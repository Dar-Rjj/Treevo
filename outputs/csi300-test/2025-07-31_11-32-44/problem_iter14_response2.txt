import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['1_Period_Return'] = df['close'].pct_change()
    df['5_Period_MA_Momentum'] = df['1_Period_Return'].rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low']) / 2
    df['Typical_Price_Volume'] = df['Typical_Price'] * df['volume']
    df['Cumulative_TP_Volume'] = df['Typical_Price_Volume'].cumsum()
    df['Cumulative_Volume'] = df['volume'].cumsum()
    df['VWAP'] = df['Cumulative_TP_Volume'] / df['Cumulative_Volume']
    
    # Calculate VWAP to Close Difference
    df['VWAP_Close_Diff'] = df['close'] - df['VWAP']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['Momentum_VWAP_Combined'] = df['Daily_Momentum'] * df['VWAP_Close_Diff']
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Confirmed_Factor'] = df['Momentum_VWAP_Combined'] * np.log(df['Volume_Change'] + 1)
    
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate N-day Close Price Percent Change
    N = 5
    df['N_Day_PCT_Change'] = df['close'].pct_change(periods=N)
    
    # Compute M-day Average of N-day Close Price Percent Change
    M = 5
    df['M_Day_Avg_N_Day_PCT_Change'] = df['N_Day_PCT_Change'].rolling(window=M).mean()
    
    # Combine Intraday High-Low Spread and Momentum
    df['Intraday_High_Low_Momentum'] = df['Intraday_High_Low_Spread'] * df['M_Day_Avg_N_Day_PCT_Change']
    
    # Adjust with Volume
    df['Volume_Weight'] = df['volume'].shift(1)
    df['Smoothed_Volume_Weight'] = df['Volume_Weight'].ewm(span=5, adjust=False).mean()
    df['Adjusted_Intraday_HL_Momentum'] = df['Intraday_High_Low_Momentum'] * df['Smoothed_Volume_Weight']
    
    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Introduce Intraday Range Volatility
    df['Intraday_Range'] = df['high'] - df['low']
    P = 5
    df['P_Day_STD_Intraday_Range'] = df['Intraday_Range'].rolling(window=P).std()
    
    # Synthesize All Factors
    df['Synthesized_Factor'] = (df['Volume_Confirmed_Factor'] * df['Intraday_Reversal'] * df['P_Day_STD_Intraday_Range'])
    
    return df['Synthesized_Factor']
