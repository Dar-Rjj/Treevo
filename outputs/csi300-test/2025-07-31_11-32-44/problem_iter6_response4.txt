```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Momentum
    prev_close = df['close'].shift(1)
    current_close = df['close']
    momentum = current_close / prev_close - 1
    
    # Combine High-Low Range and Momentum
    combined_factor = high_low_range * momentum
    
    # Confirm with Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_confirmation = (df['volume'] > avg_volume).astype(int)
    
    # Introduce Price Volatility
    daily_price_change = (df['close'] - df['open']) / df['open']
    price_volatility = daily_price_change.rolling(window=20).std()
    
    # Adjust by Price Volatility
    final_factor = combined_factor * volume_confirmation / (price_volatility + 1e-6)
    
    return final_factor
```
