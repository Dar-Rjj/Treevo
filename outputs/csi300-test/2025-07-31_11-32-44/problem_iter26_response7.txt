import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series representing the alpha factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily returns
    df['10_day_ma_return'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 30-day volatility of the daily returns
    df['30_day_volatility'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the money flow index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_flow_sum = positive_flow.rolling(window=14).sum()
    negative_flow_sum = negative_flow.rolling(window=14).sum()
    money_ratio = positive_flow_sum / negative_flow_sum
    df['mfi'] = 100 - (100 / (1 + money_ratio))
    
    # Calculate the price momentum
    df['momentum'] = df['close'] / df['close'].shift(10) - 1
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['10_day_ma_return'] 
                          + df['mfi'] 
                          + df['momentum'] / (1 + df['30_day_volatility']))
    
    return df['alpha_factor'].dropna()
