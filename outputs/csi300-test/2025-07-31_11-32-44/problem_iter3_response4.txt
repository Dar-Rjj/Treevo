```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    df['previous_day_high_low_spread'] = df['intraday_high_low_spread'].shift(1)
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['intraday_momentum'] = (df['intraday_high_low_spread'] - df['previous_day_high_low_spread']) * df['volume_ratio']
    
    # Calculate Intraday Reversal
    df['close_open_diff'] = df['close'] - df['open']
    df['previous_day_close_open_diff'] = df['close_open_diff'].shift(1)
    df['intraday_reversal'] = (df['close_open_diff'] - df['previous_day_close_open_diff']) * df['intraday_momentum']
    
    # Calculate Volume Trend
    df['5_day_vol_ma'] = df['volume'].rolling(window=5).mean()
    df['10_day_vol_ma'] = df['volume'].rolling(window=10).mean()
    df['volume_trend'] = (df['5_day_vol_ma'] - df['10_day_vol_ma']) / df['10_day_vol_ma']
    
    # Adjust Intraday Reversal by Volume Trend
    df['intraday_reversal_adjusted'] = df['intraday_reversal'] * df['volume_trend']
    
    return df['intraday_reversal_adjusted']
```
