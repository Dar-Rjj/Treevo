```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 20-Day Raw Momentum
    df['20Day_Raw_Momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['1_Period_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5_Period_MA_Momentum'] = df['1_Period_Return'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Generate Initial Alpha Factor
    df['Initial_Alpha_Factor'] = df['High_Low_Range'] * df['5_Period_MA_Momentum']
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Confirmed_Alpha'] = df['Initial_Alpha_Factor'] * np.log(df['Volume_Change'] + 1)
    
    # Calculate VWAP
    df['VWAP'] = (df[['high', 'low']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['VWAP_Close_Diff'] = df['close'] - df['VWAP']
    df['Combined_Momentum_VWAP'] = df['Daily_Momentum'] * df['VWAP_Close_Diff']
    
    # Integrate with Intraday and Trend Adjustments
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    df['Volume_Adjusted_Intraday_Return'] = df['Intraday_Return'] / df['volume']
    df['Enhanced_Price_Momentum'] = df['20Day_Raw_Momentum'] * df['Volume_Adjusted_Intraday_Return']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Volume_Confirmed_Alpha'] * df['Combined_Momentum_VWAP'] * df['Enhanced_Price_Momentum']
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * np.sign(df['Volume_Change'])
    
    return df['Final_Alpha_Factor']
```
