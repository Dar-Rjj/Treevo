import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Daily Momentum
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # 5-Period Moving Average of Price Momentum
    df['1_Period_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5_Period_MA_Momentum'] = df['1_Period_Return'].rolling(window=5).mean()
    
    # Volume Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low']) / 2
    df['TP_Volume'] = df['Typical_Price'] * df['volume']
    df['Cum_TP_Volume'] = df['TP_Volume'].rolling(window=5).sum()
    df['Total_Volume'] = df['volume'].rolling(window=5).sum()
    df['VWAP'] = df['Cum_TP_Volume'] / df['Total_Volume']
    
    # VWAP-Close Difference
    df['VWAP_Close_Diff'] = df['VWAP'] - df['close']
    
    # Combined Momentum and VWAP-Close Difference
    df['Combined_Momentum_VWAP'] = df['Daily_Momentum'] * df['VWAP_Close_Diff']
    
    # Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Volume Confirmation
    df['Volume_Confirmed_Factor'] = df['Combined_Momentum_VWAP'] * np.log(df['Volume_Change'] + 1)
    
    # Intraday High-Low Spread
    df['Intraday_HL_Spread'] = df['high'] - df['low']
    
    # Adjusted Intraday High-Low Spread
    df['Adjusted_Intraday_HL_Spread'] = df['Intraday_HL_Spread'] * df['5_Period_MA_Momentum']
    df['Volume_EMA'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['Adjusted_Intraday_HL_Spread'] = df['Adjusted_Intraday_HL_Spread'] / df['Volume_EMA']
    
    # Intraday Reversal
    df['Intraday_Reversal'] = df['close'] - df['open'] - (df['close'].shift(1) - df['open'].shift(1))
    
    # Intraday Range Volatility
    P = 5  # You can choose a different period if needed
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Range_Volatility'] = df['Intraday_Range'].rolling(window=P).std()
    
    # Synthesized Alpha Factor
    df['Synthesized_Alpha_Factor'] = df['Volume_Confirmed_Factor'] * df['Intraday_Reversal']
    df['Synthesized_Alpha_Factor'] = df['Synthesized_Alpha_Factor'] * df['Intraday_Range_Volatility']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Synthesized_Alpha_Factor'].rolling(window=7).mean() + df['Combined_Momentum_VWAP']
    
    return df['Final_Alpha_Factor'].dropna()
