```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['avg_price'] = (df['high'] + df['low']) / 2
    df['total_volume'] = df['volume']
    df['vwap_sum'] = (df['avg_price'] * df['volume']).rolling(window=20, min_periods=1).sum()
    df['vwap'] = df['vwap_sum'] / df['total_volume'].rolling(window=20, min_periods=1).sum()
    
    # Calculate VWAP to Close Difference
    df['vwap_close_diff'] = df['close'] - df['vwap']
    
    # Combine Momentum and VWAP-Close Difference
    df['cumulative_momentum_impact_factor'] = df['daily_momentum'] * df['vwap_close_diff']
    
    return df['cumulative_momentum_impact_factor']
```
