```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=5, lag=1):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate N-day Close Price Percent Change
    n_day_percent_change = df['close'].pct_change(periods=N)
    
    # Compute M-day Average of N-day Close Price Percent Change
    m_day_average_n_day_percent_change = n_day_percent_change.rolling(window=M).mean()
    
    # Combine Intraday High-Low Spread and Momentum
    combined_momentum = intraday_high_low_spread * m_day_average_n_day_percent_change
    
    # Adjust with Volume
    volume_change = df['volume'].pct_change(periods=lag)
    average_volume = df['volume'].rolling(window=M).mean()
    volume_adjusted_momentum = combined_momentum * (df['volume'] / average_volume)
    
    # Calculate Intraday Reversal
    close_open_diff = df['close'] - df['open']
    prev_close_open_diff = df['close'].shift(1) - df['open'].shift(1)
    intraday_reversal = close_open_diff - prev_close_open_diff
    
    # Compute Z-score of Volume Change
    volume_change_zscore = (volume_change - volume_change.rolling(window=M).mean()) / volume_change.rolling(window=M).std()
    
    # Multiply Adjusted Momentum by Intraday Reversal and Volume Reversal Z-score
    alpha_factor = volume_adjusted_momentum * intraday_reversal * volume_change_zscore
    
    return alpha_factor
```
