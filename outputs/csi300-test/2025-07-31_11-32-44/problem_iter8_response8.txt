```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 1-Period Return
    df['1_period_return'] = df['close'].pct_change()
    
    # Compute 5-Period Moving Average of 1-Period Returns
    df['5_period_ma_return'] = df['1_period_return'].rolling(window=5).mean()
    
    # Compute 5-day Exponential Moving Average of 1-Period Returns
    df['5_day_ema_return'] = df['1_period_return'].ewm(span=5, adjust=False).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Generate Initial Alpha Factor
    df['initial_alpha_factor'] = df['high_low_range'] * df['5_period_ma_return']
    
    # Confirm with Volume
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    # Apply Volume Confirmation
    df['confirmed_alpha_factor'] = df['initial_alpha_factor'] * np.log(df['volume_change'] + 1)
    
    # Calculate Volume Adjusted Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    df['volume_adjusted_intraday_return'] = df['intraday_return'] / df['volume']
    
    # Integrate Price Momentum and Intraday Return
    df['integrated_factor'] = df['5_day_ema_return'] * df['volume_adjusted_intraday_return']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['confirmed_alpha_factor'] * np.sign(df['volume_change'])
    
    # Integrate with Trend Continuity and Filter
    df['trend_continuity'] = np.sign(df['close'] - df['close'].shift(1))
    df['combined_factor'] = df['final_alpha_factor'] * df['trend_continuity']
    df['alpha_factor'] = df['combined_factor'].apply(lambda x: x if abs(x) > 0.5 else 0)
    
    return df['alpha_factor']
```
