```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Parameters
    N = 10  # Number of days for close price percent change
    threshold = 0.7  # Threshold for filtering low signal-to-noise ratio
    
    # Calculate Enhanced Raw Momentum
    df['N_day_close_change'] = df['close'].pct_change(periods=N)
    df['raw_momentum'] = (df['close'] / df['close'].shift(30)) - 1
    df['enhanced_raw_momentum'] = df['N_day_close_change'] * df['raw_momentum']
    
    # Calculate Volume-Weighted Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    df['volume_weight'] = df['volume'] ** (1/3)
    df['volume_weighted_intraday_return'] = df['intraday_return'] / df['volume_weight']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Combine Enhanced Raw Momentum, Volume-Weighted Intraday Return, and High-Low Range
    df['combined_factor'] = df['enhanced_raw_momentum'] * df['volume_weighted_intraday_return'] * df['high_low_range']
    
    # Confirm with Volume Delta
    df['volume_delta'] = df['volume'] - df['volume'].shift(30)
    df['absolute_volume_delta'] = df['volume_delta'].abs()
    df['confirmed_factor'] = df['combined_factor'] * df['absolute_volume_delta']
    
    # Integrate with Trend Continuity and Filter
    df['trend_continuity'] = np.sign(df['close'] - df['close'].shift(1))
    df['final_factor'] = df['confirmed_factor'] * df['trend_continuity']
    df['final_factor'] = df['final_factor'].apply(lambda x: x if abs(x) > threshold else 0)
    
    return df['final_factor'].dropna()
```
