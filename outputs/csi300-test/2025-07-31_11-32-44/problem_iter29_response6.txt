```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Raw Momentum
    N = 5
    df['close_pct_change'] = df['close'].pct_change(periods=N)
    df['exp_decay_momentum'] = df['close'] / df['close'].shift(45) - 1

    # Calculate Volume-Adjusted Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    df['volume_adjusted_volatility'] = df['intraday_volatility'] / np.log(df['volume'])

    # Combine Momentum, Intraday Volatility, and High-Low Range
    df['combined_factor'] = df['exp_decay_momentum'] * df['volume_adjusted_volatility'] * (df['high'] - df['low'])

    # Confirm with Absolute Volume Delta
    df['volume_delta'] = df['volume'] - df['volume'].shift(45)
    df['absolute_volume_delta'] = df['volume_delta'].abs()
    df['combined_factor'] *= df['absolute_volume_delta']

    # Integrate with Trend Continuity and Filter
    df['trend_continuity'] = np.sign(df['close'] - df['close'].shift(1))
    df['alpha_factor'] = df['combined_factor'] * df['trend_continuity']
    df['alpha_factor'] = df['alpha_factor'].where(df['alpha_factor'].abs() >= 0.8, 0)

    # Analyze Prices for Momentum and Trends
    df['price_momentum'] = df['close'] - df['close'].shift(1)
    df['moving_avg_crossover'] = np.sign(df['close'] - df['close'].rolling(window=20).mean())

    # Use Volume to Identify Market Sentiment
    n_days = 20
    df['avg_volume'] = df['volume'].rolling(window=n_days).mean()
    df['volume_spike_indicator'] = df['volume'] / df['avg_volume']
    df['volume_price_confirmation'] = df['price_momentum'] * (df['volume'] > df['avg_volume'])

    # Evaluate Open and Close Prices for Gaps
    df['gap_analysis'] = df['open'] - df['close'].shift(1)
    df['gap_fill_indicator'] = (df['open'] - df['close']).abs() < 0.5 * df['gap_analysis'].abs()

    # Combine Indicators for a Single Alpha Factor
    weights = {
        'price_momentum': 0.3,
        'moving_avg_crossover': 0.2,
        'volume_spike_indicator': 0.2,
        'gap_fill_indicator': 0.3
    }
    df['composite_alpha_factor'] = (
        weights['price_momentum'] * df['price_momentum'] +
        weights['moving_avg_crossover'] * df['moving_avg_crossover'] +
        weights['volume_spike_indicator'] * df['volume_spike_indicator'] +
        weights['gap_fill_indicator'] * df['gap_fill_indicator'].astype(int)
    )
    df['weighted_composite_alpha_factor'] = df['alpha_factor'] * df['composite_alpha_factor']

    return df['weighted_composite_alpha_factor']
```
