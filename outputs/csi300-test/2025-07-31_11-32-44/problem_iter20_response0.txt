```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate 5-Period Moving Average of Returns
    returns_1_period = df['close'].pct_change()
    ma_5_returns = returns_1_period.rolling(window=5).mean()
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Generate Base Alpha Factor
    base_alpha = raw_momentum * ma_5_returns * high_low_range
    
    # Confirm with Volume
    volume_delta = df['volume'] - df['volume'].shift(20)
    abs_volume_delta = volume_delta.abs()
    sma_5_abs_volume_delta = abs_volume_delta.rolling(window=5).mean()
    
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Price Momentum
    ma_5_price_momentum = (df['close'] - df['close'].shift(1)).pct_change().rolling(window=5).mean()
    
    # Combine Daily Momentum and 5-Period Moving Average of Returns
    combined_momentum = daily_momentum + ma_5_returns
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    vwap_close_diff = vwap - df['close']
    
    # Adjust Combined Alpha Factor
    adjusted_alpha = base_alpha * sma_5_abs_volume_delta * vwap_close_diff
    
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    prev_intraday_spread = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = intraday_spread - prev_intraday_spread
    
    # Generate Additional Alpha Factor Component
    amount_delta = df['amount'] - df['amount'].shift(20)
    additional_alpha_component = ma_5_returns * np.sign(volume_delta + amount_delta)
    
    # Combine All Components for Final Alpha Factor
    final_alpha = adjusted_alpha + intraday_momentum + additional_alpha_component + combined_momentum
    
    # Further Adjust Combined Alpha Factor
    final_alpha *= sma_5_abs_volume_delta * amount_delta.abs()
    
    # Apply 5-day Exponential Moving Average
    ema_5_final_alpha = final_alpha.ewm(span=5, adjust=False).mean()
    
    return ema_5_final_alpha
```
