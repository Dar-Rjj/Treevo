```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    intraday_high_low_momentum = (df['high'] - df['low']) / df['low']
    
    # Integrate Price Momentum
    df['previous_close'] = df['close'].shift(1)
    price_momentum = df['close'] - df['previous_close']
    
    # Combine Intraday High-Low Momentum and Price Momentum
    combined_momentum = intraday_high_low_momentum * price_momentum
    
    # Confirm with Volume
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    volume_confirmation = df['volume'] / df['average_volume']
    
    # Final factor
    factor = combined_momentum * volume_confirmation
    
    return factor
```
