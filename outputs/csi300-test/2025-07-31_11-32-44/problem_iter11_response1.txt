```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Integrated Intraday Adjusted Return
    df['Intraday_Range'] = (df['high'] - df['low']) / df['low']
    df['Intraday_Volatility'] = abs(df['close'] - df['open']) / ((df['high'] + df['low']) / 2)
    df['Intraday_Adjusted_Return'] = df['Intraday_Range'] / df['volume']
    df['Trend_Continuity'] = np.sign(df['close'] - df['close'].shift(1))
    df['Intraday_Adjusted_Return'] *= df['Trend_Continuity']

    # Calculate Multi-Period Price Momentum
    df['1_Period_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5_Period_Momentum'] = df['1_Period_Return'].rolling(window=5).mean()

    # Combine Intraday and Multi-Period Momentum
    df['Combined_Momentum'] = df['Intraday_Adjusted_Return'] * df['5_Period_Momentum']

    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    df['Prev_Day_High_Low_Spread'] = df['Intraday_High_Low_Spread'].shift(1)
    df['Intraday_Momentum'] = df['Intraday_High_Low_Spread'] - df['Prev_Day_High_Low_Spread']

    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    df['VWAP_Close_Diff'] = df['close'] - df['VWAP']

    # Calculate Volume Reversal Z-score
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Rolling_Mean'] = df['Volume_Change'].rolling(window=20).mean()
    df['Volume_Rolling_Std'] = df['Volume_Change'].rolling(window=20).std()
    df['Volume_Reversal_Z_Score'] = (df['Volume_Change'] - df['Volume_Rolling_Mean']) / df['Volume_Rolling_Std']

    # Calculate Amount Change
    df['Amount_Change'] = df['amount'] - df['amount'].shift(1)

    # Generate Alpha Factor Component
    df['Volume_Amount_Confirmation'] = np.sign(df['Volume_Change'] + df['Amount_Change'])
    df['Alpha_Factor_Component'] = df['5_Period_Momentum'] * df['Volume_Amount_Confirmation']

    # Apply Volume and Amount Confirmation
    df['Combined_Factor'] = (df['Combined_Momentum'] * df['VWAP_Close_Diff'] * df['Volume_Reversal_Z_Score']) + df['Alpha_Factor_Component']

    return df['Combined_Factor']
```
