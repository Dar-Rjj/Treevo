import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # 5-Period Moving Average of Price Momentum
    price_momentum_5 = (daily_momentum.rolling(window=5).sum() / 5)
    
    # Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combined Momentum and VWAP-Close Difference
    combined_momentum_vwap = daily_momentum * (vwap - df['close'])
    
    # Confirmed with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    confirmed_with_volume = combined_momentum_vwap * np.log(volume_change + 1)
    
    # Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Intraday Reversal
    intraday_reversal = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Intraday Range Volatility
    p_day = 14
    intraday_range_volatility = (df['high'] - df['low']).rolling(window=p_day).std()
    
    # Synthesized Alpha Factor
    adjusted_momentum = confirmed_with_volume * intraday_reversal * intraday_range_volatility
    
    # Stochastic Oscillator
    low_14 = df['low'].rolling(window=14).min()
    high_14 = df['high'].rolling(window=14).max()
    k_line = (df['close'] - low_14) / (high_14 - low_14)
    d_line = k_line.rolling(window=3).mean()
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_momentum * d_line
    final_alpha_factor = final_alpha_factor * np.sign(volume_change)
    
    return final_alpha_factor
