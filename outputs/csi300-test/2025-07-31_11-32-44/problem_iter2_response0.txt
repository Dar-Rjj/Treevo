```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['close'] - df['open']).abs() / ((df['high'] + df['low']) / 2)
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Weight by Volume to Capture Interest
    volume_adjusted_return = intraday_return / df['volume']
    
    # Integrate with Trend Continuity
    trend_continuity = df['close'].diff().fillna(0).apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate Intraday Momentum
    previous_high_low_spread = high_low_spread.shift(1).fillna(0)
    intraday_momentum = high_low_spread - previous_high_low_spread
    
    # Calculate Intraday Reversal
    close_open_diff = (df['close'] - df['open'])
    previous_close_open_diff = close_open_diff.shift(1).fillna(0)
    intraday_reversal = (close_open_diff - previous_close_open_diff) * intraday_momentum
    
    # Combine all components into the final factor
    factor = (intraday_volatility * volume_adjusted_return * trend_continuity * intraday_momentum * intraday_reversal).dropna()
    
    return factor
```
