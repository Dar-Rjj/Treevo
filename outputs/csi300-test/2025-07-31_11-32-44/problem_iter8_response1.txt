```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Price Momentum
    price_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    moving_avg_momentum = price_returns.rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    vwap_close_diff = df['close'] - vwap
    
    # Combine Daily Momentum and VWAP-Close Difference
    combined_factor = daily_momentum * vwap_close_diff
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    confirmed_factor = combined_factor * np.log(volume_change + 1)
    
    # Synthesize with Intraday Reversal
    intraday_reversal = df['close'] - df['open'] - (df['close'].shift(1) - df['open'].shift(1))
    synthesized_factor = confirmed_factor + intraday_reversal
    
    # Final Alpha Factor with final confirmation
    final_alpha = synthesized_factor * np.sign(volume_change)
    
    return final_alpha
```
