```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Momentum
    price_momentum = (df['close'] / df['close'].shift(1)) - 1
    
    # Combine High-Low Range and Momentum
    combined_factor = high_low_range * price_momentum
    
    # Calculate Average Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Confirm with Volume
    volume_confirmation = (df['volume'] > avg_volume).astype(int)
    
    # Final factor: combine the high-low range momentum with volume confirmation
    final_factor = combined_factor * volume_confirmation
    
    return final_factor
```
