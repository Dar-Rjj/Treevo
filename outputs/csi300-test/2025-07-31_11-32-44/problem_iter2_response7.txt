```python
import pandas as pd

def heuristics_v2(df):
    # Compute 1-Period Return using Close price
    df['return_1_period'] = df['close'] / df['close'].shift(1) - 1
    
    # Compute 5-Period Moving Average of 1-Period Returns
    df['momentum_5_period'] = df['return_1_period'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Combine High-Low Range and Momentum
    df['combined_factor'] = df['high_low_range'] * df['momentum_5_period']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Apply Volume Confirmation
    df['alpha_factor'] = df['combined_factor'] * df['volume_change'].apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
