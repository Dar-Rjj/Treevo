```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['DailyMomentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['HL_Vol'] = ((df['high'] + df['low']) / 2) * df['volume']
    df['TotalVolume'] = df['volume'].rolling(window=5, min_periods=1).sum()
    df['TotalHL_Vol'] = df['HL_Vol'].rolling(window=5, min_periods=1).sum()
    df['VWAP'] = df['TotalHL_Vol'] / df['TotalVolume']
    
    # Calculate VWAP to Close Difference
    df['VWAP_CloseDiff'] = df['close'] - df['VWAP']
    
    # Combine Momentum and VWAP-Close Difference
    df['CombinedMomentumVWAP'] = df['DailyMomentum'] * df['VWAP_CloseDiff']
    
    # Calculate Intraday High-Low Spread
    df['IntradayHighLowSpread'] = df['high'] - df['low']
    
    # Calculate Intraday Reversal
    df['CloseOpenDiff'] = df['close'] - df['open']
    df['PrevDayCloseOpenDiff'] = df['CloseOpenDiff'].shift(1)
    df['IntradayReversal'] = (df['CloseOpenDiff'] - df['PrevDayCloseOpenDiff']) * df['DailyMomentum']
    
    # Synthesize Enhanced Intraday and Cumulative Momentum Factor
    df['EnhancedIntradayCumulativeMomentum'] = df['CombinedMomentumVWAP'] + df['IntradayReversal']
    
    return df['EnhancedIntradayCumulativeMomentum']
```
