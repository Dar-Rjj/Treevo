```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['DailyMomentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 20-Day Raw Momentum
    df['20DayRawMomentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['1PeriodReturn'] = df['close'].pct_change()
    df['5PeriodMA'] = df['1PeriodReturn'].rolling(window=5).mean()
    
    # Calculate 5-day Exponential Moving Average of 1-Period Returns
    df['5DayEMA'] = df['1PeriodReturn'].ewm(span=5, adjust=False).mean()
    
    # Calculate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Generate Initial Alpha Factor
    df['InitialAlphaFactor'] = df['HighLowRange'] * df['5PeriodMA']
    
    # Confirm with Volume
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    df['VolumeConfirmation'] = np.log(df['VolumeChange'] + 1)
    df['InitialAlphaFactor'] *= df['VolumeConfirmation']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['HL_Volume'] = ((df['high'] + df['low']) / 2) * df['volume']
    df['TotalVolume'] = df['volume'].rolling(window=1).sum()
    df['VWAP'] = df['HL_Volume'] / df['TotalVolume']
    
    # Calculate VWAP to Close Difference
    df['VWAP_CloseDiff'] = df['close'] - df['VWAP']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['CombinedMomentum'] = df['DailyMomentum'] * df['VWAP_CloseDiff']
    
    # Integrate with Intraday and Trend Adjustments
    df['IntradayReturn'] = (df['high'] - df['low']) / df['low']
    df['VolumeAdjustedIntradayReturn'] = df['IntradayReturn'] / df['volume']
    df['EnhancedPriceMomentum'] = df['CombinedMomentum'] * df['VolumeAdjustedIntradayReturn']
    
    # Calculate Volume Delta
    df['VolumeDelta'] = df['volume'] - df['volume'].shift(20)
    
    # Adjust with Volume Delta
    df['AdjustedResult'] = df['EnhancedPriceMomentum'] * abs(df['VolumeDelta'])
    
    # Integrate Price Delta
    df['PriceDelta'] = df['close'] - df['close'].shift(1)
    df['IntegratedResult'] = df['AdjustedResult'] + df['PriceDelta']
    
    # Apply 7-day Simple Moving Average
    df['7DaySMA'] = df['IntegratedResult'].rolling(window=7).mean()
    
    # Calculate True Range
    df['TrueRange'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    
    # Calculate Price Volatility
    df['LogReturns'] = np.log(df['close'] / df['close'].shift(1))
    df['PriceVolatility'] = df['LogReturns'].rolling(window=20).var()
    
    # Enhance Combined Components
    df['EnhancedCombinedComponents'] = df['InitialAlphaFactor'] * df['TrueRange'] * np.sqrt(df['PriceVolatility'])
    
    # Apply 5-day Exponential Moving Average
    df['FinalAlphaFactor'] = df['EnhancedCombinedComponents'].ewm(span=5, adjust=False).mean()
    
    # Final Confirmation
    df['FinalAlphaFactor'] *= np.sign(df['VolumeChange'])
    df['FinalAlphaFactor'] *= np.sign(df['close'] - df['close'].shift(1))
    
    # Filter out low signal-to-noise ratio
    threshold = 0.5
    df['FinalAlphaFactor'] = df['FinalAlphaFactor'].apply(lambda x: x if abs(x) > threshold else 0)
    
    return df['FinalAlphaFactor']
```
