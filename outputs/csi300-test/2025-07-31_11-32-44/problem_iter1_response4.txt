```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Momentum
    df['prev_close'] = df['close'].shift(1)
    momentum = df['close'] - df['prev_close']
    
    # Normalize High-Low Range
    normalized_high_low_range = high_low_range / df['close']
    
    # Combine High-Low Range and Momentum
    combined_factor = normalized_high_low_range * momentum
    
    # Confirm with Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_confirmation = (df['volume'] > avg_volume).astype(int)
    
    # Final Alpha Factor
    alpha_factor = combined_factor * volume_confirmation
    
    return alpha_factor
```
