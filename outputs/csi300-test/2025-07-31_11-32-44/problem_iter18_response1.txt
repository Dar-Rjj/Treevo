```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate N-day Close Price Percent Change (N=5)
    n = 5
    n_day_percent_change = df['close'].pct_change(periods=n)
    
    # Compute M-day Average of N-day Close Price Percent Change (M=10)
    m = 10
    m_day_avg_momentum = n_day_percent_change.rolling(window=m).mean()
    
    # Combine Intraday High-Low Spread and Momentum
    combined_momentum = intraday_high_low_spread * m_day_avg_momentum
    
    # Adjust with Volume
    volume_weight = df['volume'].shift(1).ewm(span=5, adjust=False).mean()
    adjusted_momentum = combined_momentum * volume_weight
    
    # Calculate Intraday Reversal
    intraday_reversal = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Introduce Intraday Range Volatility (P=5)
    p = 5
    intraday_range = df['high'] - df['low']
    intraday_range_volatility = intraday_range.rolling(window=p).std()
    
    # Incorporate Intraday Opening Gap (Q=5)
    q = 5
    daily_opening_gap = df['open'] - df['close'].shift(1)
    opening_gap_volatility = daily_opening_gap.rolling(window=q).std()
    
    # Combine Adjusted Momentum, Intraday Reversal, Intraday Range Volatility, and Opening Gap
    combined_factor = adjusted_momentum * intraday_reversal * intraday_range_volatility + opening_gap_volatility
    
    # Final Alpha Factor
    final_alpha_factor = combined_factor
    
    return final_alpha_factor
```
