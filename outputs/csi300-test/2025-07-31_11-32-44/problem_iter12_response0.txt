```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Intraday Volatility (Average True Range over a period)
    df['True_Range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()
    
    # Calculate Intraday Momentum
    df['Previous_Day_Spread'] = df['Intraday_High_Low_Spread'].shift(1)
    df['Intraday_Momentum'] = df['Intraday_High_Low_Spread'] - df['Previous_Day_Spread']
    
    # Calculate Intraday Reversal
    df['Previous_Day_Close_Open_Diff'] = (df['close'] - df['open']).shift(1)
    df['Intraday_Reversal'] = (df['close'] - df['open']) - df['Previous_Day_Close_Open_Diff']
    df['Intraday_Reversal'] = df['Intraday_Reversal'] * df['Intraday_Momentum']
    
    # Calculate Volume Trend (Exponential Moving Average of Volume)
    df['Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Compute Volume Reversal Component
    df['Volume_Reversal_Sentiment'] = np.where(df['volume'] > df['Volume_EMA'], 1, -1)
    df['Volume_Reversal_Component'] = df['Volume_Reversal_Sentiment'] * df['Intraday_Return']
    
    # Introduce Intraday Gap Factor
    df['Previous_Day_Close'] = df['close'].shift(1)
    df['Intraday_Gap_Factor'] = (df['open'] - df['Previous_Day_Close']) / df['Previous_Day_Close']
    
    # Incorporate Transaction Amount
    df['Amount_SMA'] = df['amount'].rolling(window=20).mean()
    df['Transaction_Amount_Sentiment'] = np.where(df['amount'] > df['Amount_SMA'], 1, -1)
    df['Transaction_Amount_Component'] = df['Transaction_Amount_Sentiment'] * df['Intraday_Return']
    
    # Calculate Intraday Price Efficiency
    df['Intraday_Price_Efficiency_1'] = (df['high'] - df['close']) / (df['high'] - df['low'])
    df['Intraday_Price_Efficiency_2'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    df['Intraday_Price_Efficiency'] = (df['Intraday_Price_Efficiency_1'] + df['Intraday_Price_Efficiency_2']) / 2
    
    # Incorporate Intraday Price Efficiency
    df['Previous_Day_Efficiency'] = df['Intraday_Price_Efficiency'].shift(1)
    df['Intraday_Price_Efficiency_Component'] = (df['Intraday_Price_Efficiency'] - df['Previous_Day_Efficiency']) * df['Intraday_Return']
    
    # Introduce Intraday Volume-Price Synchronization
    df['Previous_Day_Volume'] = df['volume'].shift(1)
    df['Volume_Price_Synchronization'] = np.where(df['volume'] > df['Previous_Day_Volume'], 1, -1)
    df['Volume_Price_Synchronization_Component'] = df['Volume_Price_Synchronization'] * df['Intraday_Return']
    
    # Combine All Components
    df['Alpha_Factor'] = (
        df['Intraday_Return'] * df['Intraday_Range'] * df['ATR'] +
        df['Intraday_Reversal'] +
        df['Volume_Reversal_Component'] +
        df['Intraday_Gap_Factor'] +
        df['Transaction_Amount_Component'] +
        df['Intraday_Price_Efficiency_Component'] +
        df['Volume_Price_Synchronization_Component']
    )
    
    return df['Alpha_Factor']
```
