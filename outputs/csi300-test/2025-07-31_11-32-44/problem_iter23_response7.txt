import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 20-Day Raw Momentum
    raw_momentum_20 = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate 5-Period Moving Average of Price Momentum
    price_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    ma_price_momentum_5 = price_momentum.rolling(window=5).mean()
    
    # Calculate 5-day Exponential Moving Average of 1-Period Returns
    ema_price_momentum_5 = price_momentum.ewm(span=5, adjust=False).mean()
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Generate Initial Alpha Factor
    initial_alpha_factor = high_low_range * ma_price_momentum_5
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    confirmed_alpha_factor = initial_alpha_factor * np.log(volume_change + 1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    vwap_close_diff = df['close'] - vwap
    
    # Combine Daily Momentum and VWAP-Close Difference
    combined_momentum_vwap = daily_momentum * vwap_close_diff
    
    # Integrate with Intraday and Trend Adjustments
    intraday_return = (df['high'] - df['low']) / df['low']
    volume_adjusted_intraday_return = intraday_return / df['volume']
    integrated_factor = combined_momentum_vwap * volume_adjusted_intraday_return
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(30)
    
    # Adjust with Volume Delta
    adjusted_factor = integrated_factor * abs(volume_delta)
    
    # Integrate Price Delta
    price_delta = df['close'] - df['close'].shift(1)
    adjusted_factor += price_delta
    
    # Apply 7-day Simple Moving Average
    final_factor = adjusted_factor.rolling(window=7).mean()
    
    # Calculate Intraday Reversal
    intraday_reversal = df['close'] - df['open'] - (df['close'].shift(1) - df['open'].shift(1))
    
    # Synthesize Enhanced Intraday and Cumulative Momentum Factor
    synthesized_factor = final_factor + intraday_reversal
    
    # Final Alpha Factor
    threshold = 0.5
    volume_change_sign = np.sign(volume_change)
    close_diff_sign = np.sign(df['close'] - df['close'].shift(1))
    final_alpha_factor = synthesized_factor * volume_change_sign * close_diff_sign
    final_alpha_factor = np.where(abs(final_alpha_factor) < threshold, 0, final_alpha_factor)
    
    return final_alpha_factor
