```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] / df['close'].shift(20) - 1
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(20)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Amount Delta
    amount_delta = df['amount'] - df['amount'].shift(20)
    
    # Combine Components
    combined_factor = raw_momentum * high_low_range * abs(volume_delta) * abs(amount_delta)
    
    # Apply 5-day Exponential Moving Average
    ema_5_days = combined_factor.ewm(span=5, adjust=False).mean()
    
    return ema_5_days
```
