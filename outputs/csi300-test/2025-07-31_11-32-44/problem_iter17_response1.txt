```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 20-Day Raw Momentum
    df['20DayRawMomentum'] = df['close'].pct_change(20)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['1PeriodReturn'] = df['close'].pct_change()
    df['5PeriodMA'] = df['1PeriodReturn'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Generate Initial Alpha Factor
    df['InitialAlphaFactor'] = df['HighLowRange'] * df['5PeriodMA']
    
    # Confirm with Volume
    df['VolumeDelta'] = (df['volume'] - df['volume'].shift(20)).abs()
    df['ConfirmedAlphaFactor'] = df['InitialAlphaFactor'] * df['VolumeDelta']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    df['VWAPCloseDiff'] = df['close'] - df['VWAP']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['CombinedMomentum'] = df['20DayRawMomentum'] * df['VWAPCloseDiff']
    
    # Integrate with Intraday and Trend Adjustments
    df['IntradayReturn'] = (df['high'] - df['low']) / df['low']
    df['VolumeAdjustedIntradayReturn'] = df['IntradayReturn'] / df['volume']
    df['IntegratedFactor'] = df['CombinedMomentum'] * df['VolumeAdjustedIntradayReturn']
    
    # Apply 5-day Exponential Moving Average
    df['EMA_5'] = df['IntegratedFactor'].ewm(span=5, adjust=False).mean()
    
    # Integrate with Trend Continuity and Filter
    df['TrendContinuity'] = (df['close'] > df['close'].shift(1)).astype(int) * 2 - 1
    df['FinalAlphaFactor'] = df['TrendContinuity'] * df['EMA_5']
    
    # Introduce a threshold to filter out low signal-to-noise ratio
    threshold = 0.5
    df['FinalAlphaFactor'] = df['FinalAlphaFactor'].apply(lambda x: x if abs(x) > threshold else 0)
    
    return df['FinalAlphaFactor']
```
