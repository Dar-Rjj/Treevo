```python
import pandas as pd

def heuristics_v2(df):
    # Compute 1-Period Return using Close price
    df['1_period_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Compute 5-Period Moving Average of 1-Period Return
    df['5_period_moving_avg'] = df['1_period_return'].rolling(window=5).mean()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Generate Alpha Factor
    df['alpha_factor'] = df['5_period_moving_avg'] * df['volume_change'].apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    return df['alpha_factor']
```
