```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Raw Momentum
    df['raw_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume-Weighted Intraday Momentum
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_weighted_intraday_momentum'] = df['intraday_high_low_spread'] * df['avg_volume_5d']
    
    # Combine Intraday and Raw Momentum
    df['prev_day_intraday_high_low_spread'] = df['intraday_high_low_spread'].shift(1)
    df['combined_intraday_raw_momentum'] = (df['intraday_high_low_spread'] - df['prev_day_intraday_high_low_spread']) * df['raw_momentum']
    
    # Calculate Volume Delta
    df['volume_delta'] = df['volume'] - df['volume'].shift(20)
    
    # Adjust with Volume Delta
    df['adjusted_combined_momentum'] = df['combined_intraday_raw_momentum'] * df['volume_delta'].abs()
    
    # Apply 5-day Simple Moving Average
    df['sma_5d'] = df['adjusted_combined_momentum'].rolling(window=5).mean()
    
    # Calculate Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['high'] + df['low']) / 2 * df['volume'] / df['volume'].sum()
    
    # Calculate VWAP to Close Difference
    df['vwap_close_diff'] = df['close'] - df['vwap']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['combined_daily_momentum_vwap'] = df['daily_momentum'] * df['vwap_close_diff']
    
    # Calculate Intraday Reversal
    df['intraday_reversal'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Synthesize Enhanced Intraday and Cumulative Momentum Factor
    df['enhanced_intraday_cumulative_momentum'] = df['combined_daily_momentum_vwap'] + df['intraday_reversal']
    
    # Introduce Intraday Gap Factor
    df['intraday_gap_factor'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine All Components
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_range'] = (df['high'] - df['low']) / df['low']
    df['intraday_volatility'] = df['intraday_range'].rolling(window=5).std()
    
    df['alpha_factor'] = (df['intraday_return'] * df['intraday_range'] * df['intraday_volatility'] + 
                          df['intraday_reversal'] + 
                          df['vwap_close_diff'] * df['daily_momentum'] + 
                          df['intraday_gap_factor'] + 
                          df['sma_5d'] + 
                          df['enhanced_intraday_cumulative_momentum'])
    
    return df['alpha_factor']
```
