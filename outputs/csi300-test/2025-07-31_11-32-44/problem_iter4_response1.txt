```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['DailyMomentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['TypicalPrice'] = (df['high'] + df['low']) / 2
    df['TypicalPriceVolume'] = df['TypicalPrice'] * df['volume']
    df['CumulativeTypicalPriceVolume'] = df['TypicalPriceVolume'].cumsum()
    df['CumulativeVolume'] = df['volume'].cumsum()
    df['VWAP'] = df['CumulativeTypicalPriceVolume'] / df['CumulativeVolume']
    
    # Combine Momentum and VWAP-Close Difference
    df['VWAPCloseDifference'] = df['close'] - df['VWAP']
    df['Factor'] = df['DailyMomentum'] * df['VWAPCloseDifference']
    
    return df['Factor']
```
