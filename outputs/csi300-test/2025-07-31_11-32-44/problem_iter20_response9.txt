```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] / df['close'].shift(20)
    
    # Calculate Enhanced Raw Momentum
    n_day_returns = (df['close'].pct_change(periods=5) + df['close'].pct_change(periods=10) + df['close'].pct_change(periods=21)) / 3
    exp_weighted_momentum = df['close'] / df['close'].shift(30).ewm(span=30, adjust=False).mean()
    enhanced_raw_momentum = n_day_returns * exp_weighted_momentum
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(20)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Calculate Volume-Adjusted Intraday Volatility
    volume_adjusted_intraday_volatility = intraday_volatility / np.sqrt(df['volume'])
    
    # Combine Enhanced Raw Momentum, Volume-Adjusted Intraday Volatility, and High-Low Range
    combined_factor = enhanced_raw_momentum * volume_adjusted_intraday_volatility * high_low_range
    
    # Adjust with Volume Delta
    combined_factor *= abs(volume_delta)
    
    # Integrate with Trend Continuity and Filter
    trend_continuity = np.sign(df['close'] - df['close'].shift(1))
    combined_factor_filtered = combined_factor * trend_continuity
    combined_factor_filtered[abs(combined_factor_filtered) < 0.7] = 0
    
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Returns
    returns_1_period = df['close'].pct_change(periods=1)
    moving_average_returns = returns_1_period.rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    vwap_close_diff = df['close'] - vwap
    
    # Combine Daily Momentum and VWAP-Close Difference
    momentum_vwap_combined = daily_momentum * vwap_close_diff
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_confirmation = momentum_vwap_combined * np.log(volume_change + 1)
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Previous Day's Intraday High-Low Spread
    prev_intraday_high_low_spread = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = intraday_high_low_spread - prev_intraday_high_low_spread
    
    # Generate Alpha Factor Component
    amount_change = df['amount'] - df['amount'].shift(1)
    alpha_factor_component = moving_average_returns * np.sign(volume_change + amount_change)
    
    # Synthesize Enhanced Intraday and Cumulative Momentum Factor
    synthesized_factor = volume_confirmation + intraday_momentum
    
    # Final Alpha Factor
    final_alpha_factor = synthesized_factor + combined_factor_filtered + alpha_factor_component
    
    return final_alpha_factor
```
