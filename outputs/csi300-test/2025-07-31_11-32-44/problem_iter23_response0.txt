```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Adjusted Return
    intraday_range = (df['high'] - df['low']) / df['low']
    intraday_volatility = abs(df['close'] - df['open']) / ((df['high'] + df['low']) / 2)
    volume_weight = (df['volume'] / df['volume'].rolling(window=5).max()) * np.sign(df['close'] - df['close'].shift(1))
    intraday_adjusted_return = intraday_range * volume_weight

    # Calculate Multi-Period Price Momentum
    one_period_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    five_period_momentum = one_period_return.rolling(window=5).mean()
    
    # Combine Intraday and Multi-Period Momentum
    combined_momentum = intraday_adjusted_return * five_period_momentum

    # Generate Alpha Factor Component
    volume_change = df['volume'] - df['volume'].shift(1)
    amount_change = df['amount'] - df['amount'].shift(1)
    sign_volume_amount_change = np.sign(volume_change + amount_change)
    alpha_factor_component = five_period_momentum * sign_volume_amount_change
    
    # Apply Final Combination and Confirmation
    vwap = (0.5 * (df['high'] + df['low']) * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_close_diff = df['close'] - vwap
    volume_reversal_score = (df['volume'] - df['volume'].rolling(window=5).mean()) / (df['volume'].rolling(window=5).max() - df['volume'].rolling(window=5).min())
    final_combination = (combined_momentum * vwap_close_diff) * volume_reversal_score
    
    # Add Intraday Momentum
    high_low_spread = df['high'] - df['low']
    intraday_momentum = high_low_spread - high_low_spread.shift(1)
    final_alpha_factor = final_combination + intraday_momentum
    
    return final_alpha_factor
```
