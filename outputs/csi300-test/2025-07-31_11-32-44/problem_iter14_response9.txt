```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['DailyMomentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['1PeriodReturn'] = df['close'].pct_change()
    df['5PeriodMAMomentum'] = df['1PeriodReturn'].rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    df['VWAPtoCloseDiff'] = df['close'] - df['VWAP']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['CombinedFactor'] = df['DailyMomentum'] * df['VWAPtoCloseDiff']
    
    # Confirm with Volume
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    df['VolumeConfirmedFactor'] = df['CombinedFactor'] * np.log(df['VolumeChange'] + 1)
    
    # Synthesize with Intraday Reversal
    df['IntradayReversal'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    df['SynthesizedFactor'] = df['VolumeConfirmedFactor'] * df['IntradayReversal']
    
    # Integrate with Stochastic Oscillator
    high_14 = df['high'].rolling(window=14).max()
    low_14 = df['low'].rolling(window=14).min()
    df['%K'] = (df['close'] - low_14) / (high_14 - low_14)
    df['%D'] = df['%K'].rolling(window=3).mean()
    df['StochasticOscillatorFactor'] = df['SynthesizedFactor'] * df['%D']
    
    # Calculate Raw Momentum
    df['RawMomentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate Volume Delta
    df['VolumeDelta'] = df['volume'] - df['volume'].shift(20)
    
    # Calculate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Combine High-Low Range and Raw Momentum
    df['CombinedMomentum'] = df['RawMomentum'] * df['HighLowRange']
    
    # Adjust with Volume Delta
    df['AdjustedMomentum'] = df['CombinedMomentum'] * np.abs(df['VolumeDelta'])
    df['SMA5_AdjustedMomentum'] = df['AdjustedMomentum'].rolling(window=5).mean()
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = df['StochasticOscillatorFactor'] + df['SMA5_AdjustedMomentum']
    df['FinalAlphaFactor'] = df['FinalAlphaFactor'] * np.sign(df['VolumeChange'])
    
    return df['FinalAlphaFactor']
```
