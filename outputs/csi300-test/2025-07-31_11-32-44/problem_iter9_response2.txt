```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate N-day Close Price Percent Change (e.g., N=5)
    n = 5
    n_day_close_price_percent_change = df['close'].pct_change(n)
    
    # Compute M-day Average of N-day Close Price Percent Change (e.g., M=10)
    m = 10
    m_day_average_n_day_close_price_percent_change = n_day_close_price_percent_change.rolling(window=m).mean()
    
    # Combine Intraday High-Low Spread and Momentum
    adjusted_momentum = intraday_high_low_spread * m_day_average_n_day_close_price_percent_change
    
    # Adjust with Volume
    volume_lag = 5
    volume_weight = df['volume'].rolling(window=volume_lag).mean()
    normalized_volume_weight = volume_weight / df['volume']
    
    # Calculate Intraday Reversal
    intraday_reversal = df['close'] - df['open'] - (df['close'].shift(1) - df['open'].shift(1))
    
    # Combine Adjusted Momentum and Intraday Reversal
    volume_weight_z_score = (normalized_volume_weight - normalized_volume_weight.rolling(window=m).mean()) / normalized_volume_weight.rolling(window=m).std()
    final_factor = adjusted_momentum * intraday_reversal * volume_weight_z_score
    
    return final_factor
```
