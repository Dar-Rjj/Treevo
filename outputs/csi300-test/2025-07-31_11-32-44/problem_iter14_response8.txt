```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 20-Day Raw Momentum
    df['20_day_raw_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['1_period_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5_period_moving_avg'] = df['1_period_return'].rolling(window=5).mean()
    
    # Calculate 5-day Exponential Moving Average of 1-Period Returns
    df['5_day_ema'] = df['1_period_return'].ewm(span=5, adjust=False).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Generate Initial Alpha Factor
    df['initial_alpha_factor'] = df['high_low_range'] * df['5_period_moving_avg']
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['confirmed_alpha_factor'] = df['initial_alpha_factor'] * np.log(df['volume_change'] + 1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap_numerator'] = ((df['high'] + df['low']) / 2) * df['volume']
    df['vwap'] = df['vwap_numerator'].cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    df['vwap_close_diff'] = df['close'] - df['vwap']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['combined_momentum_vwap'] = df['daily_momentum'] * df['vwap_close_diff']
    
    # Integrate with Intraday and Trend Adjustments
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    df['volume_adjusted_intraday_return'] = df['intraday_return'] / df['volume']
    df['integrated_momentum'] = df['combined_momentum_vwap'] * df['volume_adjusted_intraday_return']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['confirmed_alpha_factor'] * np.sign(df['volume_change'])
    df['final_alpha_factor'] *= np.sign(df['close'] - df['close'].shift(1))
    
    # Introduce a threshold to filter out low signal-to-noise ratio
    df['final_alpha_factor'] = np.where(np.abs(df['final_alpha_factor']) < 0.5, 0, df['final_alpha_factor'])
    
    return df['final_alpha_factor']
```
