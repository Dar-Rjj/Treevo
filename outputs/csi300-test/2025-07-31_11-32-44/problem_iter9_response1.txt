```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high_low_spread'] - df['high_low_spread'].shift(1)
    
    # Calculate Volume-Weighted Intraday Momentum
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_weighted_intraday_momentum'] = df['intraday_momentum'] * df['avg_volume_5d']
    
    # Calculate Intraday Reversal
    df['close_open_diff'] = df['close'] - df['open']
    df['previous_close_open_diff'] = df['close_open_diff'].shift(1)
    df['intraday_reversal'] = df['close_open_diff'] - df['previous_close_open_diff']
    
    # Calculate Volume-Weighted Intraday Reversal
    df['volume_weighted_intraday_reversal'] = df['intraday_reversal'] * df['avg_volume_5d']
    
    # Calculate Cumulative Intraday Reversal
    df['cumulative_intraday_reversal'] = df['intraday_reversal'].rolling(window=5).sum()
    
    # Final factor: Combine all the components
    alpha_factor = (
        df['volume_weighted_intraday_momentum'] + 
        df['volume_weighted_intraday_reversal'] + 
        df['cumulative_intraday_reversal']
    )
    
    return alpha_factor
```
