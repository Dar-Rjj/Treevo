import numpy as np

def heuristics_v2(df):

    daily_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    momentum_5ma = daily_momentum.rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Daily Momentum and VWAP-Close Difference
    combined_factor_1 = daily_momentum * (vwap - df['close'])

    volume_change = np.log(df['volume'] / df['volume'].shift(1) + 1)

    intraday_high_low_spread = df['high'] - df['low']

    n_day_percent_change = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    m_day_average = n_day_percent_change.rolling(window=5).mean()
    
    # Combine Intraday High-Low Spread and Momentum
    combined_intraday_momentum = intraday_high_low_spread * m_day_average
    
    # Adjust with Smoothed Volume
    smoothed_volume = df['volume'].ewm(span=5, adjust=False).mean()
    combined_intraday_momentum = combined_intraday_momentum * (smoothed_volume / df['volume'])
    
    # Calculate Intraday Reversal
    intraday_reversal = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))

    p_day_std = intraday_high_low_spread.rolling(window=5).std()

    synthesized_factors = combined_factor_1 * intraday_reversal * p_day_std

    d_line = k_line.rolling(window=3).mean()

    synthesized_factors = synthesized_factors * np.sign(volume_change)

    high_low_range = df['high'] - df['low']

    intraday_volatility = (df['high'] - df['low']) / df['close']

    volume_adjusted_intraday_volatility = intraday_volatility / np.sqrt(df['volume'])
    
    # Combine Enhanced Raw Momentum, Volume-Adjusted Intraday Volatility, and High-Low Range
    combined_factor_2 = enhanced_raw_momentum * volume_adjusted_intraday_volatility * high_low_range

    combined_factor_2 = combined_factor_2 * np.abs(volume_delta)

    combined_factor_2 = np.where(np.abs(combined_factor_2) < threshold, 0, combined_factor_2)
