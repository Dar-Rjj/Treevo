```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Raw Momentum
    N = 5
    df['close_pct_change'] = df['close'].pct_change(periods=N)
    df['exp_decay_momentum'] = (df['close'] - df['close'].shift(45)) / df['close'].shift(45)

    # Calculate Volume-Adjusted Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    df['log_volume'] = np.log(df['volume'])
    df['vol_adj_intraday_volatility'] = df['intraday_volatility'] / df['log_volume']

    # Combine Momentum, Intraday Volatility, and High-Low Range
    df['combined_factor'] = df['exp_decay_momentum'] * df['vol_adj_intraday_volatility'] * (df['high'] - df['low'])

    # Confirm with Absolute Volume Delta
    df['volume_delta'] = df['volume'] - df['volume'].shift(45)
    df['abs_volume_delta'] = df['volume_delta'].abs()
    df['combined_factor'] *= df['abs_volume_delta']

    # Integrate with Trend Continuity and Filter
    df['trend_continuity'] = np.sign(df['close'] - df['close'].shift(1))
    df['alpha_factor'] = df['combined_factor'] * df['trend_continuity']
    
    # Apply Threshold
    df['alpha_factor'] = df['alpha_factor'].where(np.abs(df['alpha_factor']) >= 0.8, 0)

    return df['alpha_factor']
```
