```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Adjusted Return
    intraday_range = (df['high'] - df['low']) / df['low']
    intraday_volatility = abs(df['close'] - df['open']) / ((df['high'] + df['low']) / 2)
    volume_weighted_intraday_return = (intraday_range * intraday_volatility) / df['volume']
    trend_continuity = (df['close'] - df['close'].shift(1)).apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    intraday_adjusted_return = volume_weighted_intraday_return * trend_continuity

    # Calculate Multi-Period Price Momentum
    one_period_return = df['close'].pct_change()
    five_period_moving_average = one_period_return.rolling(window=5).mean()
    multi_period_momentum = five_period_moving_average

    # High-Low Range Confirmation
    high_low_range = (df['high'] - df['low']) / df['low']

    # Generate Alpha Factor
    combined_momentum = intraday_adjusted_return * multi_period_momentum
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_confirmation = volume_change.apply(lambda x: 1 if x > 0 else -1)

    alpha_factor = combined_momentum * volume_confirmation

    return alpha_factor
```
