```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Return using Close price
    df['1_period_return'] = df['close'].pct_change()
    
    # Shift Close price by 1 period
    df['close_shifted'] = df['close'].shift(1)
    
    # Compute 5-Period Moving Average of Price Momentum
    df['price_momentum'] = df['close'] - df['close_shifted']
    df['5_period_momentum_ma'] = df['price_momentum'].rolling(window=5).mean()
    
    # Moving Average Crossover
    df['20_day_ma'] = df['close'].rolling(window=20).mean()
    df['crossover_signal'] = (df['close'] > df['20_day_ma']).astype(int) - (df['close'] < df['20_day_ma']).astype(int)
    
    # Volume Spike Indicator
    df['volume_spike'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Volume-Price Confirmation
    df['volume_price_confirmation'] = (df['1_period_return'] > 0) & (df['volume'] > df['volume'].shift(1))
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Gap Analysis
    df['gap'] = df['open'] - df['close'].shift(1)
    df['gap_filled'] = (df['open'] - df['close']).abs() < df['gap'].abs()
    
    # Composite Alpha Factor
    df['composite_alpha'] = (
        0.3 * df['price_momentum'] +
        0.2 * df['crossover_signal'] +
        0.2 * df['volume_spike'] +
        0.1 * df['gap_filled'].astype(int)
    )
    
    # VWAP
    df['vwap'] = (df[['high', 'low']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # VWAP-Close Difference
    df['vwap_close_diff'] = df['vwap'] - df['close']
    
    # Intraday High-Low Spread
    df['intraday_spread'] = df['high'] - df['low']
    
    # Adjust with Volume
    df['volume_weight'] = df['volume']
    df['smoothed_volume_weight'] = df['volume_weight'].ewm(span=5, adjust=False).mean()
    
    # Intraday Reversal
    df['intraday_reversal'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Intraday Range Volatility
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['intraday_range'].rolling(window=5).std()
    
    # Synthesize All Factors
    df['adjusted_momentum'] = df['5_period_momentum_ma'] * df['intraday_reversal']
    df['combined_factor'] = df['adjusted_momentum'] * df['intraday_volatility']
    
    # Finalize the Alpha Factor
    df['final_alpha'] = (
        0.5 * df['combined_factor'] +
        0.3 * df['composite_alpha'] +
        0.2 * df['vwap_close_diff']
    )
    
    # Apply 7-day Simple Moving Average to the Final Integrated Result
    df['alpha_factor'] = df['final_alpha'].rolling(window=7).mean()
    
    return df['alpha_factor']
```
