import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', 'amount', 'volume' and index 'date'.
    
    Returns:
    pd.Series: A Series with the calculated factor values, indexed by 'date'.
    """
    # Calculate the percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Calculate the intraday range (high - low)
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate the volume-weighted average price (VWAP)
    df['vwap'] = (df['amount'] / df['volume'])
    
    # Calculate the 10-day moving average of the VWAP
    df['vwap_10_ma'] = df['vwap'].rolling(window=10).mean()
    
    # Calculate the ratio of the current VWAP to its 10-day moving average
    df['vwap_ratio'] = df['vwap'] / df['vwap_10_ma']
    
    # Calculate the 5-day mean of the intraday range
    df['intraday_range_5_mean'] = df['intraday_range'].rolling(window=5).mean()
    
    # Final factor: combine the VWAP ratio and the normalized intraday range
    df['factor'] = (df['vwap_ratio'] - 1) * (df['intraday_range'] / df['intraday_range_5_mean'])
    
    return df['factor']
