import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 20-Day Raw Momentum
    df['20D_Raw_Momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['1D_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5D_MA_Momentum'] = df['1D_Return'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Generate Initial Alpha Factor
    df['Initial_Alpha_Factor'] = df['High_Low_Range'] * df['5D_MA_Momentum']
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Confirmed_Alpha_Factor'] = df['Initial_Alpha_Factor'] * np.log(df['Volume_Change'] + 1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df[['high', 'low']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    df['VWAP_Close_Diff'] = df['close'] - df['VWAP']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['Combined_Momentum_VWAP'] = df['Daily_Momentum'] * df['VWAP_Close_Diff']
    
    # Integrate with Intraday and Trend Adjustments
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    df['Volume_Adjusted_Intraday_Return'] = df['Intraday_Return'] / df['volume']
    df['Enhanced_Price_Momentum'] = df['5D_MA_Momentum'] * df['Volume_Adjusted_Intraday_Return']
    
    # Calculate Volume Delta
    df['Volume_Delta'] = df['volume'] - df['volume'].shift(20)
    
    # Adjust with Volume Delta
    df['Adjusted_Result'] = df['Combined_Momentum_VWAP'] * abs(df['Volume_Delta'])
    
    # Integrate Price Delta
    df['Price_Delta'] = df['close'] - df['close'].shift(1)
    df['Final_Integrated_Result'] = df['Adjusted_Result'] + df['Price_Delta']
    
    # Apply 7-day Simple Moving Average
    df['SMA_7D'] = df['Final_Integrated_Result'].rolling(window=7).mean()
    
    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = (df['close'].shift(1) - df['open'].shift(1))
    
    # Synthesize Enhanced Intraday and Cumulative Momentum Factor
    df['Synthesized_Factor'] = df['SMA_7D'] + df['Intraday_Reversal']
    
    # Final Alpha Factor
    df['Sign_Volume_Change'] = np.sign(df['Volume_Change'])
    df['Sign_Close_Change'] = np.sign(df['close'] - df['close'].shift(1))
    df['Final_Alpha_Factor'] = df['Synthesized_Factor'] * df['Sign_Volume_Change'] * df['Sign_Close_Change']
    
    # Introduce a threshold to filter out low signal-to-noise ratio
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].apply(lambda x: x if abs(x) > 0.5 else 0)
    
    return df['Final_Alpha_Factor']
