import pandas as pd

def heuristics_v2(df):
    """
    This function generates a novel and interpretable alpha factor based on the provided DataFrame.
    The alpha factor is calculated using a combination of price and volume data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series with the alpha factor values indexed by (date).
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the volume-weighted typical price
    volume_weighted_typical_price = typical_price * df['volume']
    
    # Calculate the rolling sum of the volume-weighted typical price
    rolling_sum_volume_weighted_typical_price = volume_weighted_typical_price.rolling(window=20).sum()
    
    # Calculate the rolling sum of the volume
    rolling_sum_volume = df['volume'].rolling(window=20).sum()
    
    # Calculate the volume-weighted price
    volume_weighted_price = rolling_sum_volume_weighted_typical_price / rolling_sum_volume
    
    # Calculate the price momentum
    price_momentum = typical_price.pct_change(periods=10)
    
    # Calculate the alpha factor
    alpha_factor = (volume_weighted_price - typical_price) * price_momentum
    
    return alpha_factor
