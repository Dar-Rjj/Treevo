```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['1_period_return'] = df['close'].pct_change()
    df['5_period_moving_avg'] = df['1_period_return'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Generate Initial Alpha Factor
    df['initial_alpha_factor'] = df['high_low_range'] * df['5_period_moving_avg']
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['confirmed_alpha_factor'] = df['initial_alpha_factor'] * np.log(df['volume_change'] + 1)
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Previous Day's Intraday High-Low Spread
    df['prev_intraday_high_low_spread'] = (df['high'].shift(1) - df['low'].shift(1))
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['intraday_high_low_spread'] - df['prev_intraday_high_low_spread']
    
    # Combine Intraday and Price Momentum
    df['combined_alpha_factor'] = df['confirmed_alpha_factor'] + df['intraday_momentum']
    
    # Additional Confirmation with Amount
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['amount_confirmed_alpha_factor'] = df['combined_alpha_factor'] * np.sign(df['amount_change'] + 1)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['amount_confirmed_alpha_factor'] * np.sign(df['volume_change'] + df['amount_change'])
    
    return df['final_alpha_factor']
```
