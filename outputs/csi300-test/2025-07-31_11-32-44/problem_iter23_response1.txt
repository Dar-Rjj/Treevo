```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['DailyMomentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 20-Day Raw Momentum
    df['20DayRawMomentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['1PeriodReturn'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5PeriodMA'] = df['1PeriodReturn'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Generate Initial Alpha Factor
    df['InitialAlphaFactor'] = df['HighLowRange'] * df['5PeriodMA']
    
    # Confirm with Volume
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    df['ConfirmedAlphaFactor'] = df['InitialAlphaFactor'] * np.log(df['VolumeChange'] + 1)
    
    # Calculate Volume-Adjusted Intraday Volatility
    df['IntradayVolatility'] = (df['high'] - df['low']) / df['close']
    df['VolumeAdjustedVolatility'] = df['IntradayVolatility'] / np.sqrt(df['volume'])
    
    # Combine Enhanced Raw Momentum, Volume-Adjusted Intraday Volatility, and High-Low Range
    df['EnhancedRawMomentum'] = df['20DayRawMomentum'] * df['5PeriodMA']
    df['CombinedResult'] = df['EnhancedRawMomentum'] * df['VolumeAdjustedVolatility'] * df['HighLowRange']
    
    # Calculate Volume Delta
    df['VolumeDelta'] = df['volume'] - df['volume'].shift(30)
    
    # Adjust with Volume Delta
    df['AdjustedCombinedResult'] = df['CombinedResult'] * df['VolumeDelta'].abs()
    
    # Integrate with Intraday and Trend Adjustments
    df['IntradayReturn'] = (df['high'] - df['low']) / df['low']
    df['VolumeAdjustedIntradayReturn'] = df['IntradayReturn'] / df['volume']
    df['IntegratedMomentum'] = df['EnhancedRawMomentum'] * df['VolumeAdjustedIntradayReturn']
    
    # Calculate VWAP
    df['VWAP'] = (df['high'] + df['low']) / 2 * df['volume']
    df['VWAP'] = df['VWAP'].cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    df['VWAPCloseDiff'] = df['close'] - df['VWAP']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['MomentumVWAPFactor'] = df['DailyMomentum'] * df['VWAPCloseDiff']
    
    # Calculate Intraday Reversal
    df['IntradayReversal'] = df['close'] - df['open'] - (df['close'].shift(1) - df['open'].shift(1))
    
    # Synthesize Enhanced Intraday and Cumulative Momentum Factor
    df['SynthesizedFactor'] = df['MomentumVWAPFactor'] + df['IntradayReversal']
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = df['SynthesizedFactor'] * np.sign(df['VolumeChange']) * np.sign(df['close'] - df['close'].shift(1))
    
    # Apply threshold to filter out low signal-to-noise ratio
    threshold = 0.6
    df['FinalAlphaFactor'] = df['FinalAlphaFactor'].apply(lambda x: x if abs(x) >= threshold else 0)
    
    return df['FinalAlphaFactor'].dropna()
```
