```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Parameters
    N = 10  # Number of days for close price percent change
    decay_factor = 0.99  # Exponential decay factor
    threshold = 0.8  # Threshold for filtering low signal-to-noise ratio

    # Calculate Enhanced Raw Momentum
    df['Close_Price_Pct_Change'] = df['close'].pct_change(periods=N)
    df['Exp_Weighted_Momentum'] = (df['close'] / df['close'].shift(45)).pow(decay_factor)
    df['Enhanced_Raw_Momentum'] = df['Close_Price_Pct_Change'] * df['Exp_Weighted_Momentum']

    # Calculate Volume-Adjusted Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['close']
    df['Log_Volume'] = np.log(df['volume'])
    df['Volume_Adjusted_Volatility'] = df['Intraday_Volatility'] / df['Log_Volume']

    # Calculate High-Low Range
    df['High_Low_Range'] = (df['high'] - df['low'])

    # Combine Enhanced Raw Momentum, Volume-Adjusted Intraday Volatility, and High-Low Range
    df['Combined_Factor'] = (df['Enhanced_Raw_Momentum'] * 
                             df['Volume_Adjusted_Volatility'] * 
                             df['High_Low_Range'])

    # Confirm with Volume Delta
    df['Volume_Delta'] = df['volume'] - df['volume'].shift(45)
    df['Combined_Factor'] = df['Combined_Factor'] * df['Volume_Delta'].abs()

    # Integrate with Trend Continuity and Filter
    df['Trend_Continuity'] = np.sign(df['close'] - df['close'].shift(1))
    df['Alpha_Factor'] = df['Combined_Factor'] * df['Trend_Continuity']
    
    # Apply threshold to filter out low signal-to-noise ratio
    df['Alpha_Factor'] = df['Alpha_Factor'].where(df['Alpha_Factor'].abs() >= threshold, 0)

    return df['Alpha_Factor']
```
