```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate N-day Price Momentum (e.g., 5-day momentum)
    n_day_momentum = df['close'].pct_change(periods=5)
    
    # Calculate M-day Averaged N-day Price Momentum (e.g., 10-day average of 5-day momentum)
    m_day_averaged_n_day_momentum = n_day_momentum.rolling(window=10).mean()
    
    # Combined Momentum
    combined_momentum = daily_momentum + m_day_averaged_n_day_momentum
    
    # Calculate 1-Period Return
    one_period_return = df['close'] - df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Returns
    five_period_ma_returns = one_period_return.rolling(window=5).sum() / 5
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Generate Initial Alpha Factor
    initial_alpha_factor = high_low_range * five_period_ma_returns
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    confirmed_alpha_factor = initial_alpha_factor * np.log(volume_change + 1)
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Previous Day's Intraday High-Low Spread
    prev_intraday_high_low_spread = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = intraday_high_low_spread - prev_intraday_high_low_spread
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # VWAP to Close Difference
    vwap_close_difference = df['close'] - vwap
    
    # Volume Reversal Z-score
    volume_reversal_z_score = (volume_change - volume_change.rolling(window=30).mean()) / volume_change.rolling(window=30).std()
    
    # Calculate Amount Change
    amount_change = df['amount'] - df['amount'].shift(1)
    
    # Generate Alpha Factor Component
    alpha_factor_component = five_period_ma_returns * np.sign(volume_change + amount_change)
    
    # Combine All Components for Final Alpha Factor
    final_alpha_factor = (combined_momentum * vwap_close_difference) * volume_reversal_z_score
    final_alpha_factor += confirmed_alpha_factor
    final_alpha_factor += intraday_momentum
    final_alpha_factor += alpha_factor_component
    
    return final_alpha_factor
```
