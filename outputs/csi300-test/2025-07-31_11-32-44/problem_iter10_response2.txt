```python
import pandas as pd

def heuristics(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] / df['close'].shift(20) - 1
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(20)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine High-Low Range and Raw Momentum
    combined_result = raw_momentum * high_low_range
    
    # Adjust with Volume Delta
    adjusted_result = combined_result * volume_delta.abs()
    
    # Apply 5-day Simple Moving Average
    alpha_factor = adjusted_result.rolling(window=5).mean()
    
    return alpha_factor
```
