```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'].pct_change(periods=20)
    
    # Calculate Volume Delta
    volume_delta = (df['volume'] - df['volume'].shift(20)).abs()
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Delta
    price_delta = df['close'] - df['close'].shift(1)
    
    # Combine High-Low Range and Raw Momentum
    combined_momentum = raw_momentum * high_low_range
    
    # Adjust with Volume Delta
    adjusted_result = combined_momentum * volume_delta
    
    # Integrate Price Delta
    integrated_result = adjusted_result + price_delta
    
    # Apply 7-day Simple Moving Average
    sma_7 = integrated_result.rolling(window=7).mean()
    
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    vwap_close_diff = df['close'] - vwap
    
    # Combine Momentum and VWAP-Close Difference
    combined_momentum_vwap = daily_momentum * vwap_close_diff
    
    # Calculate Intraday Reversal
    intraday_reversal = (df['close'].shift(1) - df['open'].shift(1))
    
    # Synthesize Enhanced Intraday and Cumulative Momentum Factor
    synthesized_factor = combined_momentum_vwap - intraday_reversal
    
    # Final Alpha Factor
    final_alpha_factor = sma_7 + synthesized_factor
    
    return final_alpha_factor
```
