```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility using Average True Range
    df['true_range'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high_low_spread'] - df['high_low_spread'].shift(1)
    
    # Calculate Intraday Reversal
    df['intraday_reversal'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    df['intraday_reversal'] *= df['intraday_momentum']
    
    # Calculate Volume Trend using Exponential Moving Average of Volume
    df['volume_ema'] = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Compute Volume Reversal Component
    df['volume_reversal'] = np.where(df['volume'] > df['volume_ema'], 1, -1) * df['intraday_return']
    
    # Introduce Intraday Gap Factor
    df['intraday_gap'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Incorporate Transaction Amount
    df['amount_sma'] = df['amount'].rolling(window=14).mean()
    df['amount_trend'] = np.where(df['amount'] > df['amount_sma'], 1, -1) * df['intraday_return']
    
    # Calculate Intraday Price Efficiency
    df['price_efficiency_high'] = (df['high'] - df['close']) / (df['high'] - df['low'])
    df['price_efficiency_low'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    df['price_efficiency'] = df['price_efficiency_high'] + df['price_efficiency_low']
    
    # Incorporate Intraday Price Efficiency
    df['price_efficiency_diff'] = df['price_efficiency'] - df['price_efficiency'].shift(1)
    df['price_efficiency_component'] = df['price_efficiency_diff'] * df['intraday_return']
    
    # Combine All Components
    factor = (df['intraday_return'] * df['intraday_range'] * df['atr'] 
              + df['intraday_reversal'] 
              + df['volume_reversal'] 
              + df['intraday_gap'] 
              + df['amount_trend'] 
              + df['price_efficiency_component'])
    
    return factor.dropna()
```
