import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Volume-Weighted Intraday Momentum
    avg_volume_last_5_days = df['volume'].rolling(window=5).mean()
    volume_weighted_intraday_momentum = intraday_high_low_spread * avg_volume_last_5_days
    
    # Combine Intraday and Raw Momentum
    combined_momentum = (intraday_high_low_spread - intraday_high_low_spread.shift(1)) * raw_momentum
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(20)
    
    # Adjust with Volume Delta
    adjusted_momentum = combined_momentum * abs(volume_delta)
    
    # Apply 5-day Simple Moving Average
    alpha_factor = adjusted_momentum.rolling(window=5).mean()
    
    return alpha_factor
