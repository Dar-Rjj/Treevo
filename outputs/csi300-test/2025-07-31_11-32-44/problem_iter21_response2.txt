```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute N-day Close Price Percent Change
    N = 10
    df['close_percent_change'] = df['close'].pct_change(N)
    
    # Calculate Exponential Weighted Momentum
    ewm_alpha = 0.5
    df['ewm_momentum'] = (df['close'] / df['close'].shift(30)).apply(lambda x: x ** ewm_alpha)
    
    # Enhanced Raw Momentum
    df['enhanced_raw_momentum'] = df['close_percent_change'] * df['ewm_momentum']
    
    # Compute Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Volume-Adjusted Intraday Volatility
    df['log_volume'] = np.log(df['volume'])
    df['vol_adj_intraday_volatility'] = df['intraday_volatility'] / df['log_volume']
    
    # High-Low Range
    df['high_low_range'] = df['high'] / df['low']
    
    # Combine Enhanced Raw Momentum, Volume-Adjusted Intraday Volatility, and High-Low Range
    df['combined_factor'] = df['enhanced_raw_momentum'] * df['vol_adj_intraday_volatility'] * df['high_low_range']
    
    # Confirm with Volume Delta
    df['volume_delta'] = (df['volume'] - df['volume'].shift(30)).abs()
    df['confirmed_factor'] = df['combined_factor'] * df['volume_delta']
    
    # Trend Continuity and Filter
    df['trend_continuity'] = np.sign(df['close'] - df['close'].shift(1))
    threshold = 0.7
    df['alpha_factor'] = df['confirmed_factor'] * df['trend_continuity']
    df['alpha_factor'] = df['alpha_factor'].apply(lambda x: x if abs(x) > threshold else 0)
    
    return df['alpha_factor']
```
