```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low']
    
    # Integrate Price Momentum
    price_momentum = df['close'].diff()
    
    # Combine Intraday High-Low Momentum and Price Momentum
    combined_momentum = intraday_momentum * price_momentum
    
    # Confirm with Volume
    # Calculate Average Volume using a moving average
    window_size = 20  # You can adjust this window size
    avg_volume = df['volume'].rolling(window=window_size).mean()
    
    # Compare with Average Volume
    volume_confirmation = df['volume'] / avg_volume
    
    # Final factor: Multiply combined momentum by volume confirmation
    factor = combined_momentum * volume_confirmation
    
    return factor
```
