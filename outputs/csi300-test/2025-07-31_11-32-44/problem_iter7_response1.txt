```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=5):
    # Calculate Daily Momentum
    df['DailyMomentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate N-day Price Momentum
    df['NDayMomentum'] = df['close'].pct_change(N)
    
    # Calculate M-day Averaged N-day Price Momentum
    df['MDayAveragedNDayMomentum'] = df['NDayMomentum'].rolling(window=M).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['TypicalPrice'] = (df['high'] + df['low']) / 2
    df['TypicalPriceVolume'] = df['TypicalPrice'] * df['volume']
    df['CumulativeTPV'] = df['TypicalPriceVolume'].cumsum()
    df['CumulativeVolume'] = df['volume'].cumsum()
    df['VWAP'] = df['CumulativeTPV'] / df['CumulativeVolume']
    
    # VWAP to Close Difference
    df['VWAPCloseDiff'] = df['close'] - df['VWAP']
    
    # Volume Reversal Z-score
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    df['RollingVolStd'] = df['VolumeChange'].rolling(window=N).std()
    df['RollingVolMean'] = df['VolumeChange'].rolling(window=N).mean()
    df['VolumeReversalZScore'] = (df['VolumeChange'] - df['RollingVolMean']) / df['RollingVolStd']
    
    # Combine All Factors
    df['CombinedFactor'] = (df['MDayAveragedNDayMomentum'] * df['VWAPCloseDiff']) * df['VolumeReversalZScore']
    
    return df['CombinedFactor']
```
