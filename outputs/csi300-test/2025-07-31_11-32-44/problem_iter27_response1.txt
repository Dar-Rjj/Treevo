```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['5_Period_MA_Momentum'] = df['Daily_Momentum'].rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['Momentum_VWAP_Combined'] = df['Daily_Momentum'] * (df['VWAP'] - df['close'])
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] / df['volume'].shift(1)
    df['Confirmed_Factor'] = df['Momentum_VWAP_Combined'] * np.log(df['Volume_Change'] + 1)
    
    # Calculate Intraday High-Low Spread
    df['Intraday_Spread'] = df['high'] - df['low']
    
    # Compute M-day Average of N-day Close Price Percent Change
    df['N_day_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['M_day_Avg_N_day_Return'] = df['N_day_Return'].rolling(window=5).mean()
    
    # Combine Intraday High-Low Spread and Momentum
    df['Adjusted_Momentum'] = df['Intraday_Spread'] * df['M_day_Avg_N_day_Return']
    
    # Adjust with Smoothed Volume
    df['EMA_Volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['Adjusted_Momentum'] = df['Adjusted_Momentum'] * (df['EMA_Volume'] / df['volume'].shift(1))
    
    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Introduce Intraday Range Volatility
    df['Range_Volatility'] = (df['high'] - df['low']).rolling(window=5).std()
    
    # Synthesize All Factors
    df['Synthesized_Factor'] = df['Confirmed_Factor'] * df['Intraday_Reversal'] * df['Range_Volatility']
    
    # Integrate with Stochastic Oscillator
    df['%K'] = (df['close'] - df['low'].rolling(window=14).min()) / (df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min())
    df['%D'] = df['%K'].rolling(window=3).mean()
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Synthesized_Factor'] * np.sign(df['Volume_Change']) * df['close'].rolling(window=100).mean()
    
    return df['Final_Alpha_Factor'].dropna()
```
