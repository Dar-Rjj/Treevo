```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Momentum
    df['range'] = df['high'] - df['low']
    df['normalized_range'] = df['range'] / df['close'].shift(1)
    
    df['open_close_diff'] = df['close'] - df['open']
    df['normalized_open_close_diff'] = df['open_close_diff'] / df['close'].shift(1)
    
    df['average_momentum'] = (df['normalized_range'] + df['normalized_open_close_diff']) / 2
    
    # Incorporate Volume Impact
    df['volume_rate_of_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    df['alpha_factor'] = df['average_momentum'] * df['volume_rate_of_change']
    
    return df['alpha_factor']
```
