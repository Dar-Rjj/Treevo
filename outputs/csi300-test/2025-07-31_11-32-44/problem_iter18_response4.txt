import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(20)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate True Range
    true_range = df[['high', 'low', 'close']].rolling(window=2).apply(
        lambda x: max(x[0] - x[1], abs(x[0] - x[2]), abs(x[1] - x[2])),
        raw=True
    )
    
    # Combine Components
    combined_factor = raw_momentum * true_range * abs(volume_delta)
    
    # Apply 5-day Exponential Moving Average
    ema_5_day = combined_factor.ewm(span=5, adjust=False).mean()
    
    return ema_5_day
