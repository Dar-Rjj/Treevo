```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['DailyMomentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['HLVolumeWeighted'] = (df['high'] + df['low']) / 2 * df['volume']
    df['VWAP'] = df['HLVolumeWeighted'].rolling(window=20, min_periods=1).sum() / df['volume'].rolling(window=20, min_periods=1).sum()
    
    # Calculate VWAP to Close Difference
    df['VWAPCloseDiff'] = df['close'] - df['VWAP']
    
    # Combine Momentum and VWAP-Close Difference
    df['CombinedMomentumVWAP'] = df['DailyMomentum'] * df['VWAPCloseDiff']
    
    # Calculate Intraday Reversal
    df['IntradayReversal'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Synthesize Enhanced Intraday and Cumulative Momentum Factor
    df['EnhancedIntradayCumulativeMomentum'] = df['CombinedMomentumVWAP'] + df['IntradayReversal']
    
    # Calculate Raw Momentum
    df['RawMomentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume Delta
    df['VolumeDelta'] = df['volume'] - df['volume'].shift(20)
    
    # Calculate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Combine High-Low Range and Raw Momentum
    df['CombinedHighLowMomentum'] = df['RawMomentum'] * df['HighLowRange']
    
    # Adjust with Volume Delta
    df['AdjustedMomentum'] = df['CombinedHighLowMomentum'] * abs(df['VolumeDelta'])
    df['SMA_AdjustedMomentum'] = df['AdjustedMomentum'].rolling(window=5, min_periods=1).mean()
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = df['EnhancedIntradayCumulativeMomentum'] + df['SMA_AdjustedMomentum']
    
    return df['FinalAlphaFactor']
```
