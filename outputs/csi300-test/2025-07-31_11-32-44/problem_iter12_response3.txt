```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Parameters
    N = 10  # N-day Close Price Percent Change
    M = 5   # M-day Average of N-day Close Price Percent Change
    lag = 2  # Lag for Volume Weight
    EMA_span = 5  # Span for Exponential Moving Average on Volume Weight
    P = 10  # P-day Standard Deviation of Intraday Range

    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']

    # Calculate N-day Close Price Percent Change
    df['N_day_Close_Pct_Change'] = df['close'].pct_change(N)

    # Compute M-day Average of N-day Close Price Percent Change
    df['M_day_Avg_N_day_Close_Pct_Change'] = df['N_day_Close_Pct_Change'].rolling(window=M).mean()

    # Combine Intraday High-Low Spread and Momentum
    df['Intraday_Momentum'] = df['Intraday_High_Low_Spread'] * df['M_day_Avg_N_day_Close_Pct_Change']

    # Adjust with Volume
    df['Volume_Weight'] = df['volume'].shift(lag)
    df['Smoothed_Volume_Weight'] = df['Volume_Weight'].ewm(span=EMA_span, adjust=False).mean()
    df['Adjusted_Momentum'] = df['Intraday_Momentum'] * df['Smoothed_Volume_Weight']

    # Calculate Intraday Reversal
    df['Close_Open_Diff'] = df['close'] - df['open']
    df['Prev_Close_Open_Diff'] = df['Close_Open_Diff'].shift(1)
    df['Intraday_Reversal'] = df['Close_Open_Diff'] - df['Prev_Close_Open_Diff']

    # Introduce Intraday Range Volatility
    df['Intraday_Range'] = df['high'] - df['low']
    df['P_day_Std_Intraday_Range'] = df['Intraday_Range'].rolling(window=P).std()

    # Combine Adjusted Momentum, Intraday Reversal, and Intraday Range Volatility
    df['Combined_Factor'] = (df['Adjusted_Momentum'] * df['Intraday_Reversal']) * df['P_day_Std_Intraday_Range']

    # Return the factor values as a pandas Series
    return df['Combined_Factor'].dropna()
```
