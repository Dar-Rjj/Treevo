```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Momentum
    df['raw_momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate 5-Period Moving Average of Returns
    df['1_period_return'] = df['close'].pct_change()
    df['5_period_ma_returns'] = df['1_period_return'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Generate Base Alpha Factor
    df['base_alpha_factor'] = df['raw_momentum'] * df['5_period_ma_returns'] * df['high_low_range']
    
    # Confirm with Volume
    df['volume_delta'] = df['volume'] - df['volume'].shift(20)
    df['abs_volume_delta'] = df['volume_delta'].abs()
    df['avg_abs_volume_delta'] = df['abs_volume_delta'].rolling(window=5).mean()
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_day_intraday_high_low_spread'] = df['intraday_high_low_spread'].shift(1)
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['intraday_high_low_spread'] - df['prev_day_intraday_high_low_spread']
    
    # Generate Additional Alpha Factor Component
    df['amount_delta'] = df['amount'] - df['amount'].shift(20)
    df['sign_vol_amount_change'] = np.sign(df['volume_delta'] + df['amount_delta'])
    df['additional_alpha_component'] = df['5_period_ma_returns'] * df['sign_vol_amount_change']
    
    # Combine All Components for Final Alpha Factor
    df['confirmed_alpha_factor'] = df['base_alpha_factor'] * df['avg_abs_volume_delta']
    df['final_alpha_factor'] = (df['confirmed_alpha_factor'] + 
                                df['intraday_momentum'] + 
                                df['additional_alpha_component'])
    
    # Adjust Combined Alpha Factor
    df['abs_amount_delta'] = df['amount_delta'].abs()
    df['adjusted_final_alpha_factor'] = (df['final_alpha_factor'] * 
                                         df['avg_abs_volume_delta'] * 
                                         df['abs_amount_delta'])
    
    # Apply 5-day Exponential Moving Average
    df['ema_adjusted_alpha_factor'] = df['adjusted_final_alpha_factor'].ewm(span=5).mean()
    
    return df['ema_adjusted_alpha_factor'].dropna()
```
