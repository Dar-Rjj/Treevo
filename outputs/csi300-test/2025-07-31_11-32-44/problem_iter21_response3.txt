import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with (date) as index representing the factor values.
    """
    # Calculate the intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the daily volatility
    df['daily_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Calculate the volume-weighted intraday return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * (df['volume'] / df['volume'].rolling(window=21).mean())
    
    # Calculate the smoothed volume-weighted intraday return
    df['smoothed_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=5).mean()
    
    # Calculate the relative strength of the intraday return
    df['relative_strength'] = df['intraday_return'] / df['intraday_return'].rolling(window=21).std()
    
    # Combine the factors into a final alpha factor
    df['alpha_factor'] = (
        df['smoothed_volume_weighted_intraday_return'] * 
        df['relative_strength'] * 
        df['daily_volatility']
    )
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
