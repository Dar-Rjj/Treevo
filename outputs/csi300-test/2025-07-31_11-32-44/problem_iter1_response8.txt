```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Momentum
    df['raw_momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate Volume Delta
    df['volume_delta'] = df['volume'] - df['volume'].shift(20)
    
    # Combine Momentum and Volume
    df['combined_value'] = df['raw_momentum'] * df['volume_delta']
    df['combined_value'] = df['combined_value'].where(df['volume_delta'] >= 0, df['combined_value'] * -1)
    
    # Add a smoothing factor (5-day simple moving average)
    df['alpha_factor'] = df['combined_value'].rolling(window=5).mean()
    
    return df['alpha_factor']
```
