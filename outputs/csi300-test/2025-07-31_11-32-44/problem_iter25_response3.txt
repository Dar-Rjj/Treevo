```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['1_Period_Return'] = df['close'].pct_change()
    df['5_Period_MA_Return'] = df['1_Period_Return'].rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low']) / 2
    df['VWAP'] = (df['Typical_Price'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['VWAP_Close_Diff'] = df['close'] - df['VWAP']
    df['Momentum_VWAP_Combined'] = df['Daily_Momentum'] * df['VWAP_Close_Diff']
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Confirmed_Factor'] = df['Momentum_VWAP_Combined'] * np.log(df['Volume_Change'] + 1)
    
    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['close'] - df['close'].shift(20)
    
    # Combine Intraday and Raw Momentum
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    df['Prev_Day_Intraday_High_Low_Spread'] = df['Intraday_High_Low_Spread'].shift(1)
    df['Intraday_Raw_Momentum_Combined'] = df['Intraday_High_Low_Spread'] - df['Prev_Day_Intraday_High_Low_Spread']
    df['Intraday_Raw_Momentum_Combined'] *= df['Raw_Momentum']
    
    # Adjust with Volume Delta
    df['Volume_Delta'] = df['volume'] - df['volume'].shift(20)
    df['Adjusted_Result'] = df['Intraday_Raw_Momentum_Combined'] * df['Volume_Delta'].abs()
    
    # Apply 5-day Simple Moving Average
    df['5_day_SMA_Adjusted'] = df['Adjusted_Result'].rolling(window=5).mean()
    
    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = df['close'] - df['open'] - (df['close'].shift(1) - df['open'].shift(1))
    
    # Introduce Intraday Range Volatility
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Range_Volatility'] = df['Intraday_Range'].rolling(window=5).std()
    
    # Synthesize All Factors
    df['Synthesized_Factor'] = df['Volume_Confirmed_Factor'] * df['Intraday_Reversal'] * df['Intraday_Range_Volatility']
    
    # Integrate Price Delta
    df['Price_Delta'] = df['close'] - df['close'].shift(1)
    df['Integrated_Result'] = df['Synthesized_Factor'] + df['Price_Delta']
    
    # Apply 7-day Simple Moving Average
    df['7_day_SMA_Integrated'] = df['Integrated_Result'].rolling(window=7).mean()
    
    # Generate Alpha Factor Component
    df['Alpha_Factor_Component'] = df['5_Period_MA_Return'] * np.sign(df['Volume_Change'] + (df['amount'] - df['amount'].shift(1)))
    
    return df['7_day_SMA_Integrated'] * df['Alpha_Factor_Component']
```
