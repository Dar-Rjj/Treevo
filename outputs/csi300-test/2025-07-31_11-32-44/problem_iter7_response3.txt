```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Return
    df['return_1_period'] = df['close'].pct_change()
    
    # Calculate 10-Period Price Momentum
    df['momentum_10_period'] = df['return_1_period'].rolling(window=10).sum() / 10
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Generate Alpha Factor
    df['alpha_factor'] = df['momentum_10_period'] * np.log(df['volume_change'] + 1)
    
    return df['alpha_factor']
```
