```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return_high'] = df['high'].pct_change()
    df['intraday_return_low'] = df['low'].pct_change()
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Volatility (Average True Range)
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Volume Trend
    df['volume_ema'] = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Compute Volume Reversal Component
    df['volume_reversal_sentiment'] = df.apply(lambda x: 1 if x['volume'] > x['volume_ema'] else -1, axis=1)
    df['volume_reversal_component'] = df['intraday_return_high'] * df['volume_reversal_sentiment']
    
    # Enhance Volume Reversal Component
    df['volume_reversal_adjustment'] = df['intraday_return_high'] * df['intraday_range']
    df['enhanced_volume_reversal'] = df['volume_reversal_component'] * np.sign(df['volume_reversal_adjustment'])
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['intraday_return_high'] * df['intraday_range']
    df['intraday_momentum'] -= df['intraday_high_low_spread'].shift(1)
    
    # Calculate Intraday Reversal
    df['intraday_reversal'] = df['close'] - df['open']
    df['intraday_reversal'] -= (df['close'].shift(1) - df['open'].shift(1))
    df['intraday_reversal'] *= df['intraday_momentum']
    
    # Combine Indicators
    df['alpha_factor'] = df['intraday_momentum'] + df['enhanced_volume_reversal'] + df['intraday_reversal']
    
    return df['alpha_factor'].dropna()
```
