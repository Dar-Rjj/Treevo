```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Determine Intraday Volatility
    intraday_volatility = abs(df['close'] - df['open']) / ((df['high'] + df['low']) / 2)
    
    # Weight by Volume to Capture Interest
    volume_weighted_return = intraday_return / df['volume']
    
    # Integrate with Trend Continuity
    # Use the sign of (Today's Close - Previous Day's Close)
    trend_continuity = df['close'].diff().fillna(0).apply(lambda x: 1 if x >= 0 else -1)
    
    # Final factor calculation
    factor = volume_weighted_return * trend_continuity
    
    return factor
```
