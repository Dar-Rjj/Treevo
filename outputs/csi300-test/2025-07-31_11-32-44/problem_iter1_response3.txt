```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    df['previous_day_spread'] = df['intraday_high_low_spread'].shift(1)
    df['intraday_momentum'] = df['intraday_high_low_spread'] - df['previous_day_spread']
    
    # Calculate Intraday Reversal
    df['close_open_diff'] = df['close'] - df['open']
    df['previous_day_close_open_diff'] = df['close_open_diff'].shift(1)
    df['intraday_reversal'] = (df['close_open_diff'] - df['previous_day_close_open_diff']) * df['intraday_momentum']
    
    # Drop intermediate columns
    df.drop(columns=['intraday_high_low_spread', 'previous_day_spread', 'close_open_diff', 'previous_day_close_open_diff'], inplace=True)
    
    return df['intraday_reversal']
```
