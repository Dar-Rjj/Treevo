```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Determine Intraday Volatility
    intraday_volatility = (df['close'] - df['open']).abs() / ((df['high'] + df['low']) / 2)
    
    # Weight by Volume to Capture Interest
    volume_weighted_return = intraday_return / df['volume']
    
    # Integrate with Trend Continuity
    previous_close = df['close'].shift(1)
    trend_continuity = (df['close'] - previous_close).apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Combine all components
    factor = volume_weighted_return * trend_continuity
    
    return factor
```
