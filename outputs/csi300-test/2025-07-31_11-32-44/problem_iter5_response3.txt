```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] / df['close'].shift(20) - 1
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(20)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine High-Low Range and Momentum
    combined_momentum = high_low_range * raw_momentum
    
    # Confirm with Volume
    confirmed_momentum = combined_momentum * abs(volume_delta)
    
    # Apply 5-day Simple Moving Average
    alpha_factor = confirmed_momentum.rolling(window=5).mean()
    
    return alpha_factor
```
