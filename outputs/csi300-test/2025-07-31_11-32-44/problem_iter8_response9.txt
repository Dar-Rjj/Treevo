```python
import pandas as pd

def heuristics_v2(df):
    # Parameters
    N = 10  # Number of days for close price percent change
    threshold = 0.5  # Threshold to filter out low signal-to-noise ratio
    
    # Calculate Enhanced Raw Momentum
    df['close_percent_change'] = df['close'].pct_change(N)
    df['raw_momentum'] = (df['close'] / df['close'].shift(20))
    df['enhanced_raw_momentum'] = df['close_percent_change'] * df['raw_momentum']
    
    # Calculate Volume Adjusted Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    df['volume_adjusted_intraday_return'] = df['intraday_return'] / df['volume']
    
    # Calculate High-Low Range
    df['high_low_range'] = (df['high'] - df['low'])
    
    # Combine Enhanced Raw Momentum, Volume Adjusted Intraday Return, and High-Low Range
    df['combined_factor'] = (df['enhanced_raw_momentum'] * 
                             df['volume_adjusted_intraday_return'] * 
                             df['high_low_range'])
    
    # Confirm with Volume Delta
    df['volume_delta'] = df['volume'] - df['volume'].shift(20)
    df['combined_factor'] *= abs(df['volume_delta'])
    
    # Integrate with Trend Continuity and Filter
    df['trend_continuity'] = df['close'].diff().apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    df['alpha_factor'] = df['combined_factor'] * df['trend_continuity']
    df['alpha_factor'] = df['alpha_factor'].apply(lambda x: x if abs(x) >= threshold else 0)
    
    return df['alpha_factor']
```
