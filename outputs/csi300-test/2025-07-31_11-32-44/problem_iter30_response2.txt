```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 20-Day Raw Momentum
    df['20_Day_Raw_Momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['1_Period_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5_Period_MA_Momentum'] = df['1_Period_Return'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Generate Initial Alpha Factor
    df['Initial_Alpha_Factor'] = df['High_Low_Range'] * df['5_Period_MA_Momentum']
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Confirmed_Alpha'] = df['Initial_Alpha_Factor'] * np.log(df['Volume_Change'] + 1)
    
    # Calculate Volume-Adjusted Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['close']
    df['Volume_Adjusted_Volatility'] = df['Intraday_Volatility'] / np.sqrt(df['volume'])
    
    # Combine Enhanced Raw Momentum, Volume-Adjusted Intraday Volatility, and High-Low Range
    df['Combined_Factor'] = df['20_Day_Raw_Momentum'] * df['Volume_Adjusted_Volatility'] * df['High_Low_Range']
    
    # Calculate Volume Delta
    df['Volume_Delta'] = df['volume'] - df['volume'].shift(30)
    
    # Adjust with Volume Delta
    df['Volume_Adjusted_Combined_Factor'] = df['Combined_Factor'] * abs(df['Volume_Delta'])
    
    # Integrate with Intraday and Trend Adjustments
    df['Volume_Adjusted_Intraday_Return'] = ((df['high'] - df['low']) / df['low']) / df['volume']
    df['Enhanced_Price_Momentum'] = df['20_Day_Raw_Momentum'] * df['Volume_Adjusted_Intraday_Return']
    
    # Calculate VWAP
    df['VWAP'] = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    df['VWAP_Close_Difference'] = df['close'] - df['VWAP']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['Momentum_VWAP_Factor'] = df['Daily_Momentum'] * df['VWAP_Close_Difference']
    
    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = (df['close'].shift(1) - df['open'].shift(1))
    
    # Synthesize Enhanced Intraday and Cumulative Momentum Factor
    df['Synthesized_Factor'] = df['Momentum_VWAP_Factor'] + df['Intraday_Reversal']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Volume_Adjusted_Combined_Factor'] * df['Synthesized_Factor'] * np.sign(df['Volume_Change']) * np.sign(df['close'] - df['close'].shift(1))
    
    # Apply threshold to filter out low signal-to-noise ratio
    threshold = 0.6
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].apply(lambda x: x if abs(x) > threshold else 0)
    
    return df['Final_Alpha_Factor']
```
