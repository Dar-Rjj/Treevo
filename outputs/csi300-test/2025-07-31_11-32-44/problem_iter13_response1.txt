```python
import pandas as pd

def heuristics_v2(df, N=10, M=5, threshold=0.5):
    """
    Generate an integrated alpha factor that combines N-day close price percent change,
    M-day exponential moving average, and volume-adjusted intraday return.
    
    Parameters:
    - df: pandas DataFrame with columns (open, high, low, close, amount, volume)
    - N: Number of days for the close price percent change
    - M: Number of days for the exponential moving average
    - threshold: Threshold to filter out small values
    
    Returns:
    - pandas Series representing the factor values
    """
    # Compute N-day Close Price Percent Change
    close_price_change = df['close'] / df['close'].shift(N) - 1
    
    # Apply M-day Exponential Moving Average to N-day Percent Change
    ema_close_price_change = close_price_change.ewm(span=M, adjust=False).mean()
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Adjusted Intraday Return
    volume_adjusted_intraday_return = intraday_return / df['volume']
    
    # Combine Metrics
    combined_factor = ema_close_price_change * volume_adjusted_intraday_return
    
    # Filter and set to zero if the absolute value is below the threshold
    combined_factor = combined_factor.where(combined_factor.abs() >= threshold, 0)
    
    return combined_factor
```
