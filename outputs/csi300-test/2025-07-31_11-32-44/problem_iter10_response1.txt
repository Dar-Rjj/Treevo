```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility using Average True Range
    df['true_range'] = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Intraday Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['intraday_momentum'] = df['high_low_spread'] - df['high_low_spread'].shift(1)
    
    # Calculate Intraday Reversal
    df['prev_close_open_diff'] = (df['close'].shift(1) - df['open'].shift(1))
    df['intraday_reversal'] = (df['close'] - df['open']) - df['prev_close_open_diff']
    
    # Calculate Volume Trend using Exponential Moving Average
    df['volume_ema'] = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Compute Volume Reversal Component
    df['volume_reversal'] = df['intraday_return'] * np.where(df['volume'] > df['volume_ema'], 1, -1)
    
    # Introduce Intraday Gap Factor
    df['intraday_gap'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Incorporate Transaction Amount using Simple Moving Average
    df['amount_sma'] = df['amount'].rolling(window=14).mean()
    df['transaction_amount_sentiment'] = df['intraday_return'] * np.where(df['amount'] > df['amount_sma'], 1, -1)
    
    # Combine All Components
    df['alpha_factor'] = (
        df['intraday_return'] * 
        df['intraday_range'] * 
        df['atr'] + 
        df['intraday_reversal'] + 
        df['volume_reversal'] + 
        df['intraday_gap'] + 
        df['transaction_amount_sentiment']
    )
    
    return df['alpha_factor']
```
