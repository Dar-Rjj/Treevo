import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Raw Momentum
    raw_momentum = df['close'] / df['close'].shift(20)
    
    # Calculate Volume-Weighted Intraday Momentum
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_weighted_intraday_momentum = intraday_high_low_spread * avg_volume_5d
    
    # Combine Intraday and Raw Momentum
    combined_momentum = (intraday_high_low_spread - intraday_high_low_spread.shift(1)) * raw_momentum
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(20)
    
    # Adjust with Volume Delta
    adjusted_momentum = combined_momentum * volume_delta.abs()
    
    # Apply 5-day Simple Moving Average
    sma_5d = adjusted_momentum.rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Calculate Intraday Volatility (Average True Range over a period)
    true_range = pd.DataFrame({
        'hl': df['high'] - df['low'],
        'hc': (df['high'] - df['close'].shift(1)).abs(),
        'lc': (df['low'] - df['close'].shift(1)).abs()
    }).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    intraday_volatility = atr
    
    # Calculate Volume Trend (Exponential Moving Average of Volume)
    volume_ema = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Compute Volume Reversal Component
    volume_reversal = np.where(df['volume'] > volume_ema, 1, -1) * intraday_return
    
    # Introduce Intraday Gap Factor
    intraday_gap_factor = (df['open'] / df['close'].shift(1)) - 1
    
    # Combine All Components
    factor = (intraday_return * intraday_volatility +
              volume_reversal +
              intraday_gap_factor +
              sma_5d)
    
    return factor
