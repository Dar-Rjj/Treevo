import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['HL_Avg_Vol'] = ((df['high'] + df['low']) / 2) * df['volume']
    df['Total_Vol'] = df['volume'].rolling(window=1, min_periods=1).sum()
    df['VWAP'] = df['HL_Avg_Vol'].rolling(window=1, min_periods=1).sum() / df['Total_Vol']
    
    # Calculate VWAP to Close Difference
    df['VWAP_Close_Diff'] = df['close'] - df['VWAP']
    
    # Combine Momentum and VWAP-Close Difference
    df['Momentum_VWAP_Factor'] = df['Momentum'] * df['VWAP_Close_Diff']
    
    # Calculate Intraday Reversal
    df['Close_Open_Diff'] = df['close'] - df['open']
    df['Intraday_Reversal'] = df['Close_Open_Diff'] - df['Close_Open_Diff'].shift(1)
    
    # Synthesize Enhanced Intraday and Cumulative Momentum Factor
    df['Enhanced_Momentum_Factor'] = df['Momentum_VWAP_Factor'] + df['Intraday_Reversal']
    
    return df['Enhanced_Momentum_Factor']
