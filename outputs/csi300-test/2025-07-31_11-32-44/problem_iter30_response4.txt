import pandas as pd

def heuristics_v2(df):
    # Calculate moving averages of close prices
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    df['10_day_ma_close'] = df['close'].rolling(window=10).mean()
    df['20_day_ma_close'] = df['close'].rolling(window=20).mean()
    
    # Calculate momentum of close prices
    df['10_day_momentum'] = df['close'].pct_change(periods=10)
    df['20_day_momentum'] = df['close'].pct_change(periods=20)
    df['50_day_momentum'] = df['close'].pct_change(periods=50)
    
    # Calculate standard deviation of daily returns
    df['daily_returns'] = df['close'].pct_change()
    df['10_day_volatility'] = df['daily_returns'].rolling(window=10).std()
    df['20_day_volatility'] = df['daily_returns'].rolling(window=20).std()
    df['50_day_volatility'] = df['daily_returns'].rolling(window=50).std()
    
    # Calculate moving averages of volume
    df['5_day_ma_volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_ma_volume'] = df['volume'].rolling(window=10).mean()
    df['20_day_ma_volume'] = df['volume'].rolling(window=20).mean()
    
    # Calculate volume trends
    df['10_day_volume_trend'] = df['volume'].rolling(window=10).apply(lambda x: (x[-1] - x[0]) / x[0], raw=True)
    df['20_day_volume_trend'] = df['volume'].rolling(window=20).apply(lambda x: (x[-1] - x[0]) / x[0], raw=True)
    df['50_day_volume_trend'] = df['volume'].rolling(window=50).apply(lambda x: (x[-1] - x[0]) / x[0], raw=True)
    
    # Calculate the ratio of current volume to the average volume
    df['volume_ratio_10_day'] = df['volume'] / df['10_day_ma_volume']
    df['volume_ratio_20_day'] = df['volume'] / df['20_day_ma_volume']
    df['volume_ratio_50_day'] = df['volume'] / df['50_day_ma_volume']
    
    # Calculate the range (high - low)
    df['range'] = df['high'] - df['low']
    df['10_day_range_trend'] = df['range'].rolling(window=10).apply(lambda x: (x[-1] - x[0]) / x[0], raw=True)
    df['20_day_range_trend'] = df['range'].rolling(window=20).apply(lambda x: (x[-1] - x[0]) / x[0], raw=True)
    df['50_day_range_trend'] = df['range'].rolling(window=50).apply(lambda x: (x[-1] - x[0]) / x[0], raw=True)
    
    # Calculate the average true range (ATR)
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x['high'], x['close']) - min(x['low'], x['close']), axis=1)
    df['10_day_atr'] = df['true_range'].rolling(window=10).mean()
    df['20_day_atr'] = df['true_range'].rolling(window=20).mean()
    df['50_day_atr'] = df['true_range'].rolling(window=50).mean()
    
    # Examine the relationship between open and close prices
    df['gap'] = df['open'] - df['close'].shift(1)
    df['gap_trend'] = df['gap'].rolling(window=10).apply(lambda x: (x[-1] - x[0]) / x[0], raw=True)
    
    df['percent_change'] = (df['close'] - df['open']) / df['open']
    df['percent_change_trend'] = df['percent_change'].rolling(window=10).apply(lambda x: (x[-1] - x[0]) / x[0], raw=True)
