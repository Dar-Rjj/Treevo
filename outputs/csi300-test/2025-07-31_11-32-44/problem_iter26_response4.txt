import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series with the calculated alpha factor values indexed by (date).
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the intraday range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate the average true range over a 10-day rolling window
    df['atr_10'] = df['intraday_range'].rolling(window=10).mean()
    
    # Calculate the volume weighted average price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=10).mean()
    
    # Calculate the normalized volume
    df['norm_volume'] = (df['volume'] - df['volume'].rolling(window=10).min()) / (df['volume'].rolling(window=10).max() - df['volume'].rolling(window=10).min())
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['daily_return'] * df['norm_volume']) + (df['intraday_range'] / df['atr_10']) + (df['close'] / df['vwap'])
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
