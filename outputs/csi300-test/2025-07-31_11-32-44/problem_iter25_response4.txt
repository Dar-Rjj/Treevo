```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['5_Period_Momentum_MA'] = df['Daily_Momentum'].rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap_numerator = ((df['high'] + df['low']) / 2) * df['volume']
    vwap_denominator = df['volume'].rolling(window=1, min_periods=1).sum()
    df['VWAP'] = vwap_numerator.rolling(window=1, min_periods=1).sum() / vwap_denominator
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['Momentum_VWAP_Combined'] = df['Daily_Momentum'] * (df['VWAP'] - df['close'])
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'].pct_change(fill_method=None)
    df['Momentum_VWAP_Volume_Confirmed'] = df['Momentum_VWAP_Combined'] * np.log(df['Volume_Change'] + 1)
    
    # Calculate Intraday High-Low Spread
    df['Intraday_HL_Spread'] = df['high'] - df['low']
    
    # Compute M-day Average of N-day Close Price Percent Change
    df['N_Day_Close_Pct_Change'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['M_Day_Avg_N_Day_Close_Pct_Change'] = df['N_Day_Close_Pct_Change'].rolling(window=10).mean()
    
    # Combine Intraday High-Low Spread and Momentum
    df['HL_Spread_Momentum_Combined'] = df['Intraday_HL_Spread'] * df['M_Day_Avg_N_Day_Close_Pct_Change']
    
    # Adjust with Smoothed Volume
    df['Smoothed_Volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['HL_Spread_Momentum_Adjusted'] = df['HL_Spread_Momentum_Combined'] / df['Smoothed_Volume']
    
    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Introduce Intraday Range Volatility
    df['P_Day_HL_STD'] = df['Intraday_HL_Spread'].rolling(window=14).std()
    
    # Synthesize All Factors
    df['Adjusted_Momentum'] = df['Momentum_VWAP_Volume_Confirmed'] * df['HL_Spread_Momentum_Adjusted']
    df['Intraday_Factor'] = df['Adjusted_Momentum'] * df['Intraday_Reversal'] * df['P_Day_HL_STD']
    
    # Integrate with Stochastic Oscillator
    df['High_14'] = df['high'].rolling(window=14).max()
    df['Low_14'] = df['low'].rolling(window=14).min()
    df['%K'] = (df['close'] - df['Low_14']) / (df['High_14'] - df['Low_14']) * 100
    df['%D'] = df['%K'].rolling(window=3).mean()
    
    # Final Alpha Factor
    df['Final_Alpha'] = df['Intraday_Factor'] * df['%D']
    
    # Apply sign of Volume Change as a final confirmation
    df['Final_Alpha'] *= np.sign(df['Volume_Change'])
    
    return df['Final_Alpha']
```
