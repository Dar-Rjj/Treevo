```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['price_return'] = df['close'].pct_change()
    df['5_period_momentum_ma'] = df['price_return'].rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['high'] + df['low']) / 2 * df['volume']
    df['vwap'] = df['vwap'].cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    df['vwap_close_diff'] = df['close'] - df['vwap']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['momentum_vwap_diff'] = df['daily_momentum'] * df['vwap_close_diff']
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_confirmation'] = df['momentum_vwap_diff'] * np.log(df['volume_change'] + 1)
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate N-day Close Price Percent Change (N=5)
    df['5_day_close_pct_change'] = df['close'].pct_change(periods=5)
    
    # Compute M-day Average of N-day Close Price Percent Change (M=10)
    df['10_day_avg_5_day_pct_change'] = df['5_day_close_pct_change'].rolling(window=10).mean()
    
    # Combine Intraday High-Low Spread and Momentum
    df['momentum_intraday_spread'] = df['intraday_high_low_spread'] * df['10_day_avg_5_day_pct_change']
    
    # Adjust with Volume
    df['volume_weight'] = df['volume'].shift(1)  # Lag 1
    df['smoothed_volume_weight'] = df['volume_weight'].ewm(span=10, adjust=False).mean()
    df['adjusted_momentum_intraday_spread'] = df['momentum_intraday_spread'] * df['smoothed_volume_weight']
    
    # Calculate Intraday Reversal
    df['intraday_reversal'] = df['close'] - df['open'] - (df['close'].shift(1) - df['open'].shift(1))
    
    # Introduce Intraday Range Volatility
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_range_volatility'] = df['intraday_range'].rolling(window=14).std()
    
    # Synthesize All Factors
    df['synthesized_factor'] = df['volume_confirmation'] * df['adjusted_momentum_intraday_spread'] * df['intraday_reversal'] * df['intraday_range_volatility']
    
    # Integrate with Stochastic Oscillator
    n = 14
    df['low_n'] = df['low'].rolling(window=n).min()
    df['high_n'] = df['high'].rolling(window=n).max()
    df['%K'] = (df['close'] - df['low_n']) / (df['high_n'] - df['low_n'])
    df['%D'] = df['%K'].rolling(window=3).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['synthesized_factor'] * np.sign(df['volume_change'])
    
    return df['final_alpha_factor']
```
