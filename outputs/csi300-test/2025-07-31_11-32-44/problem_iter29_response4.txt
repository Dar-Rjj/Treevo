```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Daily Momentum
    df['DailyMomentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # 5-Period Moving Average of Price Momentum
    df['1PeriodReturn'] = df['DailyMomentum']
    df['5PeriodMA'] = df['1PeriodReturn'].rolling(window=5).mean()
    
    # Volume Weighted Average Price (VWAP)
    df['TypicalPrice'] = (df['high'] + df['low']) / 2
    df['TP_Volume'] = df['TypicalPrice'] * df['volume']
    df['Cumulative_TP_Volume'] = df['TP_Volume'].rolling(window=5).sum()
    df['Cumulative_Volume'] = df['volume'].rolling(window=5).sum()
    df['VWAP'] = df['Cumulative_TP_Volume'] / df['Cumulative_Volume']
    
    # VWAP-Close Difference
    df['VWAP_Close_Diff'] = df['VWAP'] - df['close']
    
    # Combined Momentum and VWAP-Close Difference
    df['Combined_Mom_VWAP'] = df['DailyMomentum'] * df['VWAP_Close_Diff']
    
    # Volume Change
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    
    # Volume Spike Indicator
    n_days = 5
    df['AvgVolume'] = df['volume'].rolling(window=n_days).mean()
    df['VolumeSpike'] = df['volume'] / df['AvgVolume']
    
    # Volume-Price Confirmation
    df['VolumePriceConfirmation'] = df['VolumeSpike'] * (df['close'] - df['close'].shift(1))
    
    # Volume Confirmation
    df['VolumeConfirmation'] = df['Combined_Mom_VWAP'] * np.log(df['VolumeChange'] + 1)
    
    # Intraday High-Low Spread
    df['IntradaySpread'] = df['high'] - df['low']
    
    # Adjusted Intraday High-Low Spread
    df['AdjustedIntradaySpread'] = df['IntradaySpread'] * df['5PeriodMA']
    df['VolumeEMA'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['AdjustedIntradaySpread'] = df['AdjustedIntradaySpread'] * df['VolumeEMA']
    
    # Intraday Reversal
    df['IntradayReversal'] = df['close'] - df['open'] - (df['close'].shift(1) - df['open'].shift(1))
    
    # Intraday Range Volatility
    P = 5
    df['IntradayRange'] = df['high'] - df['low']
    df['IntradayRangeVolatility'] = df['IntradayRange'].rolling(window=P).std()
    
    # Gap Analysis
    df['Gap'] = df['open'] - df['close'].shift(1)
    
    # Gap Fill Indicator
    df['GapFill'] = (df['open'] - df['close'].shift(1)) - (df['close'] - df['open'])
    
    # Composite Alpha Factor
    weights = {'DailyMomentum': 0.2, '5PeriodMA': 0.2, 'VolumePriceConfirmation': 0.3, 'GapFill': 0.3}
    df['CompositeAlpha'] = (df['DailyMomentum'] * weights['DailyMomentum'] +
                            df['5PeriodMA'] * weights['5PeriodMA'] +
                            df['VolumePriceConfirmation'] * weights['VolumePriceConfirmation'] +
                            df['GapFill'] * weights['GapFill'])
    
    # Synthesized Alpha Factor
    df['SynthesizedAlpha'] = df['VolumeConfirmation'] * df['IntradayReversal'] * df['IntradayRangeVolatility']
    
    # Final Alpha Factor
    df['FinalAlpha'] = df['SynthesizedAlpha'].rolling(window=7).mean() + df['Combined_Mom_VWAP']
    
    return df['FinalAlpha'].dropna()
```
