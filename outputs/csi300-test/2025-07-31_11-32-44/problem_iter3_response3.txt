```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 1-Period Return
    df['1_period_return'] = df['close'].pct_change()
    
    # Compute 10-Period Moving Average of 1-Period Returns
    df['10_period_moving_avg'] = df['1_period_return'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['10_period_moving_avg']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Generate Alpha Factor
    df['alpha_factor'] = df['price_momentum'] * np.log(df['volume_change'] + 1)
    
    return df['alpha_factor']
```
