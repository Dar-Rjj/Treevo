```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Return
    df['1_period_return'] = df['close'].pct_change()
    
    # Calculate 5-Period Moving Average of Returns
    df['5_period_moving_avg_return'] = df['1_period_return'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Generate Initial Alpha Factor
    df['initial_alpha_factor'] = df['high_low_range'] * df['5_period_moving_avg_return']
    
    # Confirm with Volume
    df['volume_change'] = df['volume'].diff()
    df['confirmed_alpha_factor'] = df['initial_alpha_factor'] * np.log(df['volume_change'] + 1)
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Previous Day's Intraday High-Low Spread
    df['prev_intraday_high_low_spread'] = df['high_low_range'].shift(1)
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['intraday_high_low_spread'] - df['prev_intraday_high_low_spread']
    
    # Calculate Amount Change
    df['amount_change'] = df['amount'].diff()
    
    # Generate Additional Alpha Factor Component
    df['additional_alpha_factor'] = df['5_period_moving_avg_return'] * np.sign(df['volume_change'] + df['amount_change'])
    
    # Combine All Components for Final Alpha Factor
    df['final_alpha_factor'] = (df['confirmed_alpha_factor'] 
                                + df['intraday_momentum'] 
                                + df['additional_alpha_factor'])
    
    return df['final_alpha_factor']
```
