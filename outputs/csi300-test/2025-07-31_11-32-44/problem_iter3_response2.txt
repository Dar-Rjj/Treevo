```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=21, M=10, lag=5, lookback=20, threshold=0.5):
    """
    Calculate the Price Momentum and Volume Reversal Factor with Enhanced Indicators.
    
    Parameters:
    - df: pandas DataFrame with columns (open, high, low, close, amount, volume)
    - N: Number of days for the N-day Close Price Percent Change (e.g., 21)
    - M: Number of days for the M-day Exponential Moving Average (e.g., 10)
    - lag: Number of days for the Volume Change (e.g., 5)
    - lookback: Number of days for the rolling cumulative standard deviation and mean (e.g., 20)
    - threshold: Threshold to filter out low signal-to-noise ratio (e.g., 0.5)
    
    Returns:
    - alpha_factor: pandas Series with the calculated alpha factor
    """
    
    # Calculate N-day Close Price Percent Change
    df['N_day_close_pct_change'] = df['close'].pct_change(N)
    
    # Compute M-day Exponential Moving Average of N-day Close Price Percent Change
    df['price_momentum'] = df['N_day_close_pct_change'].ewm(span=M).mean()
    
    # Compute Volume Change
    df['volume_change'] = df['volume'].pct_change(lag)
    
    # Compute Cumulative Z-score of Volume Change
    df['volume_mean'] = df['volume_change'].rolling(window=lookback).mean()
    df['volume_std'] = df['volume_change'].rolling(window=lookback).std()
    df['volume_z_score'] = (df['volume_change'] - df['volume_mean']) / df['volume_std']
    
    # Combine Price Momentum and Volume Reversal
    df['combined_factor'] = df['price_momentum'] * df['volume_z_score']
    
    # Introduce a threshold to filter out low signal-to-noise ratio
    df['alpha_factor'] = df['combined_factor'].apply(lambda x: x if abs(x) >= threshold else 0)
    
    return df['alpha_factor']
```
