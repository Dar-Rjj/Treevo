```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Return
    df['1_period_return'] = df['close'].pct_change()
    
    # Price Momentum: Difference between today's close and yesterday's close
    df['price_momentum'] = df['close'] - df['close'].shift(1)
    
    # 5-Period Moving Average of Price Momentum
    df['5_period_momentum_ma'] = df['price_momentum'].rolling(window=5).mean()
    
    # Intraday Reversal
    df['intraday_reversal'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Volume Spike Indicator: Ratio of today's volume to the average volume over the past 5 days
    df['volume_spike'] = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Volume-Price Confirmation
    df['volume_price_confirmation'] = (df['close'] > df['close'].shift(1)).astype(int) * (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Volume Confirmation
    df['volume_confirmation'] = np.log(df['volume_change'] + 1)
    
    # Volume Weight
    df['volume_weight'] = df['volume'].rolling(window=5).mean()
    
    # Smooth Volume Weight with EMA
    df['smooth_volume_weight'] = df['volume_weight'].ewm(span=5, adjust=False).mean()
    
    # Gap Analysis: Measure the gap between today's open and yesterday's close
    df['gap'] = df['open'] - df['close'].shift(1)
    
    # Gap Fill Indicator: Check if the gap was filled during the day by comparing the open and close prices
    df['gap_fill'] = (df['open'] > df['close'].shift(1)) & (df['close'] < df['close'].shift(1))
    df['gap_fill'] = df['gap_fill'].astype(int)
    
    # Intraday High-Low Spread
    df['intraday_spread'] = df['high'] - df['low']
    
    # Intraday Range Volatility: P-day Standard Deviation of Intraday Range
    df['intraday_range_volatility'] = df['intraday_spread'].rolling(window=5).std()
    
    # Composite Alpha Factor
    df['composite_alpha'] = (
        df['price_momentum'] +
        df['5_period_momentum_ma'] +
        df['intraday_reversal'] * df['volume_confirmation'] +
        df['intraday_range_volatility'] +
        df['volume_spike'] * df['volume_price_confirmation']
    )
    
    # Volume Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # VWAP to Close Difference
    df['vwap_close_diff'] = df['close'] - df['vwap']
    
    # Combine Momentum and VWAP-Close Difference
    df['final_factor'] = df['price_momentum'] * df['vwap_close_diff']
    
    # Apply 7-day Simple Moving Average to the Final Integrated Result
    df['final_alpha'] = df['final_factor'].rolling(window=7).mean()
    
    return df['final_alpha']
```
