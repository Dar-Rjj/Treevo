```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 20-Day Raw Momentum
    df['raw_momentum_20'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['price_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['momentum_ma_5'] = df['price_momentum'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Generate Initial Alpha Factor
    df['initial_alpha_factor'] = df['high_low_range'] * df['momentum_ma_5']
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_confirmation'] = np.log(df['volume_change'] + 1)
    df['alpha_factor_with_volume'] = df['initial_alpha_factor'] * df['volume_confirmation']
    
    # Calculate Volume-Adjusted Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    df['volatility_adjusted'] = df['intraday_volatility'] / np.sqrt(df['volume'])
    
    # Combine Enhanced Raw Momentum, Volume-Adjusted Intraday Volatility, and High-Low Range
    df['combined_factor'] = df['raw_momentum_20'] * df['volatility_adjusted'] * df['high_low_range']
    
    # Calculate Volume Delta
    df['volume_delta'] = df['volume'] - df['volume'].shift(30)
    df['combined_factor_adjusted'] = df['combined_factor'] * abs(df['volume_delta'])
    
    # Integrate with Intraday and Trend Adjustments
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    df['volume_adjusted_intraday_return'] = df['intraday_return'] / df['volume']
    df['enhanced_momentum'] = df['raw_momentum_20'] * df['volume_adjusted_intraday_return']
    
    # Calculate VWAP
    df['vwap'] = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    df['vwap_close_diff'] = df['close'] - df['vwap']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['momentum_vwap_factor'] = df['daily_momentum'] * df['vwap_close_diff']
    
    # Calculate Intraday Reversal
    df['intraday_reversal'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Synthesize Enhanced Intraday and Cumulative Momentum Factor
    df['synthesized_factor'] = df['momentum_vwap_factor'] + df['intraday_reversal']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['combined_factor_adjusted'] * df['alpha_factor_with_volume']
    df['final_alpha_factor'] *= np.sign(df['volume_change'])
    df['final_alpha_factor'] *= np.sign(df['close'] - df['close'].shift(1))
    
    # Apply threshold to filter out low signal-to-noise ratio
    df['final_alpha_factor'] = df['final_alpha_factor'].apply(lambda x: x if abs(x) > 0.6 else 0)
    
    return df['final_alpha_factor']
```
