import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['intraday_high_low_spread'] - df['intraday_high_low_spread'].shift(1)
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_momentum'] * df['volume']
    
    # Calculate Intraday Reversal
    df['intraday_reversal'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Calculate Volume-Weighted Intraday Reversal
    df['volume_weighted_intraday_reversal'] = df['intraday_reversal'] * df['volume']
    
    # Combine the two volume-weighted indicators to form the final factor
    df['alpha_factor'] = (df['volume_weighted_intraday_momentum'] + df['volume_weighted_intraday_reversal']) / 2
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
