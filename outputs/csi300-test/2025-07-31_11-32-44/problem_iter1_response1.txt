```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=5, lag=1):
    # Calculate Price Momentum
    # Compute N-day Close Price Percent Change
    close_price = df['close']
    n_day_percent_change = close_price.pct_change(periods=N)
    
    # Compute M-day Average of N-day Close Price Percent Change
    m_day_avg_percent_change = n_day_percent_change.rolling(window=M).mean()
    
    # Calculate Volume Reversal
    # Compute Volume Change
    volume = df['volume']
    volume_change = volume.pct_change(periods=lag)
    
    # Compute Z-score of Volume Change
    rolling_mean = volume_change.rolling(window=M).mean()
    rolling_std = volume_change.rolling(window=M).std()
    volume_z_score = (volume_change - rolling_mean) / rolling_std
    
    # Combine Price Momentum and Volume Reversal
    # Multiply Price Momentum by Volume Reversal Z-score
    alpha_factor = m_day_avg_percent_change * volume_z_score
    
    return alpha_factor
```
