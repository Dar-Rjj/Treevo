```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Difference
    volume_diff = (df['volume'] - df['volume'].shift(20)).abs()
    
    # Combine Components
    combined_component = raw_momentum * high_low_range * volume_diff
    
    # Apply 5-day Simple Moving Average
    sma_5 = combined_component.rolling(window=5).mean()
    
    return sma_5
```
