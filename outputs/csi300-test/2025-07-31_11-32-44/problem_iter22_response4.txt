```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['1_Period_Return'] = df['close'].pct_change()
    df['5_Period_MA_Momentum'] = df['1_Period_Return'].rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    df['VWAP'] = vwap
    
    # Calculate VWAP to Close Difference
    df['VWAP_Close_Diff'] = df['close'] - df['VWAP']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['Combined_Momentum_VWAP'] = df['Daily_Momentum'] * df['VWAP_Close_Diff']
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Confirmed_Factor'] = df['Combined_Momentum_VWAP'] * np.log(df['Volume_Change'] + 1)
    
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate N-day Close Price Percent Change
    N = 5
    df[f'{N}_day_Close_Price_Pct_Change'] = df['close'].pct_change(periods=N)
    
    # Compute M-day Average of N-day Close Price Percent Change
    M = 5
    df[f'{M}_day_Avg_{N}_day_Close_Price_Pct_Change'] = df[f'{N}_day_Close_Price_Pct_Change'].rolling(window=M).mean()
    
    # Combine Intraday High-Low Spread and Momentum
    df['Adjusted_Momentum'] = df['Intraday_High_Low_Spread'] * df[f'{M}_day_Avg_{N}_day_Close_Price_Pct_Change']
    
    # Adjust with Volume
    volume_weight = df['volume'].shift(1)  # Use a lag of 1
    smooth_volume_weight = volume_weight.ewm(span=5, adjust=False).mean()
    df['Volume_Adjusted_Momentum'] = df['Adjusted_Momentum'] * smooth_volume_weight
    
    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Introduce Intraday Range Volatility
    df['Intraday_Range'] = df['high'] - df['low']
    P = 5
    df['Intraday_Range_Volatility'] = df['Intraday_Range'].rolling(window=P).std()
    
    # Synthesize All Factors
    df['Synthesized_Factor'] = df['Volume_Adjusted_Momentum'] * df['Intraday_Reversal'] * df['Intraday_Range_Volatility']
    
    # Integrate with Stochastic Oscillator
    n = 14
    df['%K'] = (df['close'] - df['low'].rolling(window=n).min()) / (df['high'].rolling(window=n).max() - df['low'].rolling(window=n).min())
    df['%D'] = df['%K'].rolling(window=3).mean()
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Synthesized_Factor'] * df['%D']
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * np.sign(df['Volume_Change'])
    
    return df['Final_Alpha_Factor']
```
