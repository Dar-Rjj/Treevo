```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] / df['close'].shift(20)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(20)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine Initial Components
    combined_component = raw_momentum * high_low_range * abs(volume_delta)
    
    # Apply 5-day Simple Moving Average
    sma_5 = combined_component.rolling(window=5).mean()
    
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Price Momentum
    one_period_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    ma_5_price_momentum = one_period_return.rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    vwap_close_diff = df['close'] - vwap
    
    # Combine Daily Momentum and VWAP-Close Difference
    combined_daily_momentum_vwap = daily_momentum * vwap_close_diff
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    confirmed_factor = combined_daily_momentum_vwap * np.log(volume_change + 1)
    
    # Integrate with Stochastic Oscillator
    k_line = (df['close'] - df['low'].rolling(window=14).min()) / (df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min())
    d_line = k_line.rolling(window=3).mean()
    integrated_factor = confirmed_factor * d_line
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    prev_day_intraday_high_low_spread = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = intraday_high_low_spread - prev_day_intraday_high_low_spread
    
    # Generate Alpha Factor Component
    amount_change = df['amount'] - df['amount'].shift(1)
    alpha_factor_component = ma_5_price_momentum * np.sign(volume_change + amount_change)
    
    # Synthesize All Factors
    final_alpha_factor = intraday_momentum + integrated_factor + alpha_factor_component
    final_alpha_factor = final_alpha_factor * np.sign(volume_change)
    
    # Calculate Relative Strength Index (RSI)
    gain = (df['close'] - df['close'].shift(1)).apply(lambda x: x if x > 0 else 0)
    loss = (df['close'] - df['close'].shift(1)).apply(lambda x: abs(x) if x < 0 else 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Incorporate RSI into Final Alpha Factor
    final_alpha_factor = final_alpha_factor * rsi
    
    return final_alpha_factor
```
