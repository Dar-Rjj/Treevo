```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Return using Close price
    df['1-period_return'] = df['close'].pct_change()
    
    # Shift Close price by 1 period
    df['shifted_close'] = df['close'].shift(1)
    
    # Compute 5-Period Moving Average of Price Momentum
    df['price_momentum'] = df['close'] - df['shifted_close']
    df['5_period_ma_momentum'] = df['price_momentum'].rolling(window=5).mean()
    
    # Moving Average Crossover
    df['ma_20'] = df['close'].rolling(window=20).mean()
    df['crossover'] = (df['close'] > df['ma_20']).astype(int)
    
    # Volume Spike Indicator
    avg_volume = df['volume'].rolling(window=20).mean()
    df['volume_spike'] = df['volume'] / avg_volume
    
    # Volume-Price Confirmation
    df['volume_price_confirmation'] = (df['close'] > df['shifted_close']) * (df['volume'] > df['volume'].shift(1))
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Gap Analysis
    df['gap'] = df['open'] - df['shifted_close']
    df['gap_fill'] = (df['open'] > df['shifted_close']) & (df['close'] <= df['shifted_close'])
    
    # VWAP Calculation
    df['vwap'] = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_close_diff'] = df['close'] - df['vwap']
    
    # Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Combine Intraday High-Low Spread and 5-Period Moving Average of Price Momentum
    df['intraday_momentum'] = df['intraday_high_low_spread'] * df['5_period_ma_momentum']
    
    # Adjust with Volume
    df['volume_weight'] = df['volume'].shift(1)
    df['smoothed_volume_weight'] = df['volume_weight'].ewm(span=5, adjust=False).mean()
    df['adjusted_intraday_momentum'] = df['intraday_momentum'] * df['smoothed_volume_weight']
    
    # Intraday Reversal
    df['intraday_reversal'] = df['close'] - df['open'] - (df['shifted_close'] - df['open'].shift(1))
    
    # Intraday Range Volatility
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_range_volatility'] = df['intraday_range'].rolling(window=5).std()
    
    # Synthesize All Factors
    df['synthesized_factor'] = (df['adjusted_intraday_momentum'] * df['intraday_reversal'] * df['intraday_range_volatility'])
    
    # Final Alpha Factor
    df['alpha_factor'] = df['synthesized_factor'] * df['vwap_close_diff']
    df['final_alpha_factor'] = df['alpha_factor'].rolling(window=7).mean()
    
    return df['final_alpha_factor']
```
