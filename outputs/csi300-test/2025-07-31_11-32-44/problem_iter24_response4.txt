import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Enhanced Raw Momentum
    n_day_percent_change = df['close'].pct_change(periods=21)
    ewm_momentum = df['close'].ewm(span=30, adjust=False).mean()
    enhanced_raw_momentum = n_day_percent_change * ewm_momentum
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(20)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Calculate Volume-Adjusted Intraday Volatility
    volume_adjusted_intraday_volatility = intraday_volatility / np.sqrt(df['volume'])
    
    # Combine Enhanced Raw Momentum, Volume-Adjusted Intraday Volatility, and High-Low Range
    combined_factor = enhanced_raw_momentum * volume_adjusted_intraday_volatility * high_low_range
    
    # Adjust with Volume Delta
    adjusted_factor = combined_factor * np.abs(volume_delta)
    
    # Integrate with Trend Continuity and Filter
    trend_continuity = np.sign(df['close'] - df['close'].shift(1))
    filtered_factor = adjusted_factor * trend_continuity
    filtered_factor = np.where(np.abs(filtered_factor) < 0.7, 0, filtered_factor)
    
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Returns
    returns_1_period = df['close'].pct_change()
    ma_5_returns = returns_1_period.rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    vwap_close_difference = df['close'] - vwap
    
    # Combine Daily Momentum and VWAP-Close Difference
    momentum_vwap_factor = daily_momentum * vwap_close_difference
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    confirmed_factor = momentum_vwap_factor * np.log(volume_change + 1)
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Previous Day's Intraday High-Low Spread
    prev_intraday_high_low_spread = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = intraday_high_low_spread - prev_intraday_high_low_spread
    
    # Generate Alpha Factor Component
    alpha_factor_component = ma_5_returns * np.sign(volume_change + df['amount'].pct_change().fillna(0))
    
    # Synthesize Enhanced Intraday and Cumulative Momentum Factor
    synthesized_factor = confirmed_factor + intraday_momentum
    
    # Final Alpha Factor
    final_alpha_factor = synthesized_factor + alpha_factor_component
    
    # Introduce a Dynamic Threshold Based on Recent Market Volatility
    rolling_std = df['close'].rolling(window=20).std()
    dynamic_threshold = 2 * rolling_std
    final_alpha_factor = np.where(np.abs(final_alpha_factor) > dynamic_threshold, final_alpha_factor, 0)
    
    return pd.Series(final_alpha_factor, index=df.index)
