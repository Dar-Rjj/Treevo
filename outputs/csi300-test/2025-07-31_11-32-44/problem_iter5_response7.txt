```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Return
    df['return_1_period'] = df['close'].pct_change()
    
    # Compute 5-Period Moving Average of 1-Period Returns
    df['price_momentum'] = df['return_1_period'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Generate Initial Alpha Factor
    df['initial_alpha_factor'] = df['high_low_range'] * df['price_momentum']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Apply Volume Confirmation
    df['confirmed_alpha_factor'] = df['initial_alpha_factor'] * np.log(df['volume_change'] + 1)
    
    # Final Alpha Factor with sign of Volume Change as a final confirmation
    df['final_alpha_factor'] = df['confirmed_alpha_factor'] * np.sign(df['volume_change'])
    
    return df['final_alpha_factor']
```
