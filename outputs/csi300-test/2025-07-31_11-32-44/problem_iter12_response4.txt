```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute N-day Close Price Percent Change
    N = 10
    close_price_pct_change = df['close'].pct_change(N)
    
    # Compute Raw Momentum
    raw_momentum = df['close'] / df['close'].shift(30) - 1
    
    # Combine to get Enhanced Raw Momentum
    enhanced_raw_momentum = close_price_pct_change * raw_momentum
    
    # Compute Volume-Weighted Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    volume_weighted_intraday_return = intraday_return / np.sqrt(df['volume'])
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine Enhanced Raw Momentum, Volume-Weighted Intraday Return, and High-Low Range
    combined_factor = enhanced_raw_momentum * volume_weighted_intraday_return * high_low_range
    
    # Confirm with Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(30)
    combined_factor = combined_factor * np.abs(volume_delta)
    
    # Integrate with Trend Continuity and Filter
    trend_continuity = np.sign(df['close'] - df['close'].shift(1))
    combined_factor = combined_factor * trend_continuity
    
    # Apply threshold to filter out low signal-to-noise ratio
    threshold = 0.7
    combined_factor = combined_factor.where(np.abs(combined_factor) > threshold, 0)
    
    return combined_factor
```
