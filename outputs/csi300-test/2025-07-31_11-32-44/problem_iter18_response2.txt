```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['close'] - df['close'].shift(20)
    
    # Combine Intraday and Raw Momentum
    df['combined_momentum'] = (df['intraday_high_low_spread'] - df['intraday_high_low_spread'].shift(1)) * df['raw_momentum']
    
    # Calculate Volume Delta
    df['volume_delta'] = df['volume'] - df['volume'].shift(5)
    
    # Adjust with Volume Delta
    df['adjusted_momentum'] = df['combined_momentum'] * df['volume_delta'].abs()
    
    # Apply 10-day Simple Moving Average
    df['alpha_factor'] = df['adjusted_momentum'].rolling(window=10).mean()
    
    return df['alpha_factor'].dropna()
```
