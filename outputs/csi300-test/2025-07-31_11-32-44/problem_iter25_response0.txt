```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap_numerator = ((df['high'] + df['low']) / 2) * df['volume']
    vwap = vwap_numerator.cumsum() / df['volume'].cumsum()
    
    # Combine Daily Momentum and VWAP
    combined_momentum_vwap = daily_momentum * (df['close'] - vwap)
    
    # Calculate Intraday Reversal
    intraday_reversal = df['close'].shift(1) - df['open'].shift(1)
    
    # Synthesize Momentum and Reversal Factor
    synthesized_momentum_reversal = combined_momentum_vwap + intraday_reversal
    
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(20)
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    prev_intraday_high_low_spread = intraday_high_low_spread.shift(1)
    
    # Combine Intraday and Raw Momentum
    combined_intraday_raw_momentum = (intraday_high_low_spread - prev_intraday_high_low_spread) * raw_momentum
    
    # Adjust with Volume Delta
    adjusted_intraday_raw_momentum = combined_intraday_raw_momentum * abs(volume_delta)
    
    # Introduce Intraday Gap Factor
    intraday_gap_factor = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Final Alpha Factor
    final_alpha_factor = synthesized_momentum_reversal + adjusted_intraday_raw_momentum + intraday_gap_factor
    
    return final_alpha_factor
```
