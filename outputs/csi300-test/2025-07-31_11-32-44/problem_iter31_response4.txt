```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the short-term (5 days) and long-term (20 days) moving averages of closing prices
    df['short_ma'] = df['close'].rolling(window=5).mean()
    df['long_ma'] = df['close'].rolling(window=20).mean()
    
    # Generate a momentum signal by subtracting the current closing price from the short-term moving average
    df['momentum_signal'] = df['short_ma'] - df['close']
    
    # Determine the strength of the trend by comparing the short-term and long-term moving averages
    df['trend_strength'] = df['short_ma'] - df['long_ma']
    
    # Calculate the standard deviation of daily returns over a 10-day period to assess volatility
    df['daily_returns'] = df['close'].pct_change()
    df['volatility'] = df['daily_returns'].rolling(window=10).std()
    
    # Calculate the ratio of the current closing price to the 52-week high
    df['52_week_high'] = df['close'].rolling(window=252).max()
    df['price_to_52_week_high_ratio'] = df['close'] / df['52_week_high']
    
    # Compute the on-balance volume (OBV)
    df['obv'] = (df['close'].diff() / df['close'].abs()).fillna(0) * df['volume']
    df['obv'] = df['obv'].cumsum()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['momentum_signal'] + df['trend_strength'] - df['volatility'] 
                          + (1 - df['price_to_52_week_high_ratio']) + df['obv'])
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
