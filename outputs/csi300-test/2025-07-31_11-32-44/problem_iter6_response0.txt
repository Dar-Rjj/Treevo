```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['DailyMomentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['MidPrice'] = (df['high'] + df['low']) / 2
    df['MidPriceVolume'] = df['MidPrice'] * df['volume']
    total_volume = df['volume'].sum()
    vwap = df['MidPriceVolume'].sum() / total_volume
    df['VWAP'] = vwap
    
    # Calculate VWAP to Close Ratio
    df['VWAPCloseRatio'] = df['VWAP'] / df['close']
    
    # Combine Momentum and VWAP-Close Ratio
    df['CombinedMomentumVWAP'] = df['DailyMomentum'] * df['VWAPCloseRatio']
    
    # Calculate Intraday High-Low Spread
    df['IntradayHighLowSpread'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Reversal
    df['IntradayReversal'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Synthesize Enhanced Intraday and Cumulative Momentum Factor
    df['FinalAlphaFactor'] = df['CombinedMomentumVWAP'] + df['IntradayReversal']
    
    # Incorporate Intraday High-Low Spread
    df['EnhancedFinalAlphaFactor'] = df['FinalAlphaFactor'] * df['IntradayHighLowSpread']
    
    return df['EnhancedFinalAlphaFactor']
```
