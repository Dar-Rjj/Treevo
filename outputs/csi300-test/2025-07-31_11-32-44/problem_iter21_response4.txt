import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Adjusted Return
    intraday_range = (df['high'] - df['low']) / df['low']
    intraday_volatility = (df['close'] - df['open']).abs() / ((df['high'] + df['low']) / 2)
    volume_weight = intraday_range * (df['volume'] / df['volume'].rolling(window=20).max()) * (df['close'] - df['close'].shift(1)).apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate Multi-Period Price Momentum
    one_period_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    five_period_momentum = one_period_return.rolling(window=5).mean()
    
    # Combine Intraday and Multi-Period Momentum
    combined_momentum = volume_weight * five_period_momentum
    
    # Generate Alpha Factor Component
    volume_change = df['volume'] - df['volume'].shift(1)
    amount_change = df['amount'] - df['amount'].shift(1)
    momentum_component = five_period_momentum * (volume_change + amount_change).apply(lambda x: 1 if x > 0 else -1)
    
    # Apply Final Combination and Confirmation
    vwap = ((df['high'] + df['low']) / 2 * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    vwap_close_diff = df['close'] - vwap
    mean_volume_5d = df['volume'].rolling(window=5).mean()
    max_volume_5d = df['volume'].rolling(window=5).max()
    min_volume_5d = df['volume'].rolling(window=5).min()
    volume_reversal_score = (df['volume'] - mean_volume_5d) / (max_volume_5d - min_volume_5d)
    
    final_factor = (combined_momentum * vwap_close_diff) * volume_reversal_score + (df['high'] - df['low']) - (df['high'].shift(1) - df['low'].shift(1))
    
    return final_factor
