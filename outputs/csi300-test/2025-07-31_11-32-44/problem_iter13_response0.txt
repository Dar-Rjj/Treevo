```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility using Average True Range
    df['tr'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    df['atr'] = df['tr'].rolling(window=14).mean()
    
    # Calculate Intraday Reversal
    df['previous_close_open_diff'] = df['close'].shift(1) - df['open'].shift(1)
    df['intraday_reversal'] = df['close'] - df['open'] - df['previous_close_open_diff']
    
    # Calculate Volume Trend using Exponential Moving Average of Volume
    df['volume_ema'] = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Compute Volume Reversal Component
    df['volume_sentiment'] = np.where(df['volume'] > df['volume_ema'], 1, -1)
    df['volume_reversal_component'] = df['volume_sentiment'] * df['intraday_return']
    
    # Introduce Intraday Gap Factor
    df['intraday_gap'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine All Components
    df['alpha_factor'] = (df['intraday_return'] * df['intraday_range'] * df['atr'] +
                           df['intraday_reversal'] + 
                           df['volume_reversal_component'] + 
                           df['intraday_gap'])
    
    return df['alpha_factor']
```
