import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor from the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the close price
    df['close_10_ma'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the close price
    df['close_10_std'] = df['close'].rolling(window=10).std()
    
    # Calculate the Z-score of the close price
    df['close_z_score'] = (df['close'] - df['close_10_ma']) / df['close_10_std']
    
    # Calculate the volume-weighted average price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=10).mean()
    
    # Calculate the percentage difference between the close price and VWAP
    df['close_vwap_diff_pct'] = (df['close'] - df['vwap']) / df['vwap']
    
    # Calculate the alpha factor
    df['alpha_factor'] = df['close_z_score'] * df['close_vwap_diff_pct'] * df['close_pct_change']
    
    return df['alpha_factor']
