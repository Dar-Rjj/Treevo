```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['close'] / df['close'].shift(20)
    
    # Calculate Volume-Weighted Intraday Momentum
    df['avg_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['vol_weighted_intraday_mom'] = df['intraday_high_low_spread'] * df['avg_volume_5_days']
    
    # Combine Intraday and Raw Momentum
    df['prev_day_intraday_spread'] = df['intraday_high_low_spread'].shift(1)
    df['combined_momentum'] = (df['intraday_high_low_spread'] - df['prev_day_intraday_spread']) * df['raw_momentum']
    
    # Calculate Volume Delta
    df['volume_delta'] = df['volume'] - df['volume'].shift(20)
    
    # Adjust with Volume Delta
    df['adjusted_momentum'] = df['combined_momentum'] * abs(df['volume_delta'])
    
    # Apply 5-day Simple Moving Average
    df['sma_5_days'] = df['adjusted_momentum'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Volatility (Average True Range)
    df['true_range'] = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    df['intraday_volatility'] = df['true_range'].rolling(window=5).mean()
    
    # Calculate Intraday Reversal
    df['prev_close_open_diff'] = (df['close'] - df['open']).shift(1)
    df['intraday_reversal'] = (df['close'] - df['open']) - df['prev_close_open_diff']
    
    # Calculate Volume Trend (Exponential Moving Average of Volume)
    df['volume_ema'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Compute Volume Reversal Component
    df['volume_sentiment'] = np.where(df['volume'] > df['volume_ema'], 1, -1)
    df['volume_reversal'] = df['intraday_return'] * df['volume_sentiment']
    
    # Introduce Intraday Gap Factor
    df['prev_close'] = df['close'].shift(1)
    df['intraday_gap_factor'] = (df['open'] - df['prev_close']) / df['prev_close']
    
    # Combine All Components
    df['alpha_factor'] = (df['intraday_return'] * df['intraday_range'] *
                          df['intraday_volatility'] + df['intraday_reversal'] +
                          df['volume_reversal'] + df['intraday_gap_factor'] +
                          df['sma_5_days'])
    
    return df['alpha_factor']
```
