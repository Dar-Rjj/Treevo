```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high_low_spread'] - df['high_low_spread'].shift(1)
    
    # Calculate Intraday Reversal
    df['close_open_diff'] = df['close'] - df['open']
    df['prev_close_open_diff'] = df['close_open_diff'].shift(1)
    df['intraday_reversal'] = (df['close_open_diff'] - df['prev_close_open_diff']) * df['intraday_momentum']
    
    # Generate the final factor
    factor = df['intraday_reversal']
    
    return factor
```
