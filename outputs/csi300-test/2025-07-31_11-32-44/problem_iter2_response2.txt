import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Momentum
    N = 21  # Choose N (e.g., 5, 10, 21)
    M = 10  # Choose M (e.g., 5, 10, 21)
    n_day_percent_change = df['close'].pct_change(periods=N).dropna()
    m_day_avg_momentum = n_day_percent_change.rolling(window=M).mean()
    
    # Combine High-Low Range and Momentum
    adjusted_momentum = high_low_range * m_day_avg_momentum
    
    # Adjust with Volume
    volume_lag = 5  # Choose a lag (e.g., 1, 2, 5)
    volume_change = df['volume'].pct_change(periods=volume_lag).dropna()
    avg_volume = df['volume'].rolling(window=20).mean()  # Historical average volume
    volume_reversal = volume_change - (df['volume'] / avg_volume)
    
    # Compute Z-score of Volume Change
    volume_change_zscore = (volume_change - volume_change.rolling(window=20).mean()) / volume_change.rolling(window=20).std()
    
    # Multiply Adjusted Momentum by Volume Reversal Z-score
    final_factor = adjusted_momentum * volume_change_zscore
    
    return final_factor
