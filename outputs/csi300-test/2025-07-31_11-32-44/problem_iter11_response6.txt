```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Returns
    df['Return_1p'] = df['close'].pct_change()
    df['MA_Return_5p'] = df['Return_1p'].rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Momentum and VWAP-Close Difference
    df['VWAP_Close_Diff'] = df['close'] - df['VWAP']
    df['Momentum_VWAP_Factor'] = df['Daily_Momentum'] * df['VWAP_Close_Diff']
    
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Generate Initial Alpha Factor
    df['Initial_Alpha_Factor'] = df['Intraday_High_Low_Spread'] * df['MA_Return_5p']
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Confirmed_Alpha_Factor'] = df['Initial_Alpha_Factor'] * np.log(df['Volume_Change'] + 1)
    
    # Calculate Previous Day's Intraday High-Low Spread
    df['Prev_Intraday_High_Low_Spread'] = df['Intraday_High_Low_Spread'].shift(1)
    
    # Calculate Intraday Momentum
    df['Intraday_Momentum'] = df['Intraday_High_Low_Spread'] - df['Prev_Intraday_High_Low_Spread']
    
    # Calculate Amount Change
    df['Amount_Change'] = df['amount'] - df['amount'].shift(1)
    
    # Generate Additional Alpha Factor Component
    df['Additional_Alpha_Factor'] = df['MA_Return_5p'] * np.sign(df['Volume_Change'] + df['Amount_Change'])
    
    # Synthesize Intermediate Alpha Factor
    df['Synthesized_Intermediate_Alpha_Factor'] = df['Momentum_VWAP_Factor'] * df['Intraday_High_Low_Spread']
    
    # Combine All Components for Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Confirmed_Alpha_Factor'] 
                                + df['Intraday_Momentum'] 
                                + df['Additional_Alpha_Factor'] 
                                + df['Synthesized_Intermediate_Alpha_Factor'])
    
    return df['Final_Alpha_Factor']
```
