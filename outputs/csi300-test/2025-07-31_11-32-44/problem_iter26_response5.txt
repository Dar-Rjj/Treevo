import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, amount, volume) indexed by date.
    
    Returns:
    pd.Series: A Series representing the factor values indexed by date.
    """
    # Calculate daily returns
    df['returns'] = df['close'].pct_change()
    
    # Calculate 10-day moving average of the closing price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate 30-day moving average of the closing price
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the distance between the current close and the 10-day moving average
    df['dist_ma_10'] = df['close'] - df['ma_10']
    
    # Calculate the distance between the current close and the 30-day moving average
    df['dist_ma_30'] = df['close'] - df['ma_30']
    
    # Calculate the standard deviation of the last 10 days' returns
    df['std_10'] = df['returns'].rolling(window=10).std()
    
    # Calculate the standard deviation of the last 30 days' returns
    df['std_30'] = df['returns'].rolling(window=30).std()
    
    # Calculate the ratio of the 10-day and 30-day standard deviations
    df['std_ratio'] = df['std_10'] / df['std_30']
    
    # Calculate the average volume over the last 10 days
    df['avg_vol_10'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the average volume over the last 30 days
    df['avg_vol_30'] = df['volume'].rolling(window=30).mean()
    
    # Calculate the ratio of the 10-day and 30-day average volumes
    df['vol_ratio'] = df['avg_vol_10'] / df['avg_vol_30']
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['dist_ma_10'] / df['std_10']) * df['vol_ratio'] * df['std_ratio']
    
    # Return the alpha factor as a Series
    return df['alpha_factor'].dropna()
