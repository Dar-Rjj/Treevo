import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['price_momentum'] = df['close'].pct_change()
    
    # Determine Moving Average Crossover
    n = 10  # Example: 10-day moving average
    df['ma_close'] = df['close'].rolling(window=n).mean()
    df['crossover_signal'] = (df['close'] > df['ma_close']).astype(int)
    
    # Compute Volume Spike Indicator
    m = 20  # Example: 20-day average volume
    df['avg_volume'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = df['volume'] / df['avg_volume']
    
    # Assess Volume-Price Confirmation
    df['price_increase'] = (df['close'] > df['open']).astype(int)
    df['volume_price_confirmation'] = (df['volume_spike'] > 1) & (df['price_increase'] == 1)
    df['volume_price_confirmation'] = df['volume_price_confirmation'].astype(int)
    
    # Measure Gap Analysis
    df['gap'] = df['open'] - df['close'].shift(1)
    
    # Check if the gap was filled during the day
    df['gap_fill'] = (df['open'] > df['close'].shift(1)) & (df['close'] <= df['close'].shift(1))
    df['gap_fill'] |= (df['open'] < df['close'].shift(1)) & (df['close'] >= df['close'].shift(1))
    df['gap_fill_indicator'] = df['gap_fill'].astype(int)
    
    # Integrate into a single score (Composite Alpha Factor)
    # Example weights
    weight_momentum = 0.3
    weight_crossover = 0.2
    weight_volume_spike = 0.2
    weight_volume_price = 0.15
    weight_gap_fill = 0.15
    
    df['composite_alpha_factor'] = (
        weight_momentum * df['price_momentum'] +
        weight_crossover * df['crossover_signal'] +
        weight_volume_spike * df['volume_spike'] +
        weight_volume_price * df['volume_price_confirmation'] +
        weight_gap_fill * df['gap_fill_indicator']
    )
    
    return df['composite_alpha_factor']
