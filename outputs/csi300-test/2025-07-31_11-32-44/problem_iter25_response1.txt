```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate 5-day Close Price Percent Change
    close_5d_pct_change = df['close'].pct_change(periods=5)
    
    # Compute 10-day Average of 5-day Close Price Percent Change
    momentum_10d_avg = close_5d_pct_change.rolling(window=10).mean()
    
    # Combine Intraday High-Low Spread and Momentum
    combined_momentum = intraday_high_low_spread * momentum_10d_avg
    
    # Adjust with Volume
    volume_weight = df['volume'] / df['volume'].rolling(window=1).max()
    smoothed_volume_weight = volume_weight.ewm(span=10, adjust=False).mean()
    adjusted_momentum = combined_momentum * smoothed_volume_weight
    
    # Calculate Intraday Reversal
    prev_day_close_open_diff = (df['close'].shift(1) - df['open'].shift(1)).fillna(0)
    intraday_reversal = (df['close'] - df['open']) - prev_day_close_open_diff
    
    # Introduce Intraday Range Volatility
    intraday_range = df['high'] - df['low']
    intraday_range_volatility = intraday_range.rolling(window=10).std()
    
    # Incorporate Intraday Opening Gap
    daily_opening_gap = df['open'] - df['close'].shift(1).fillna(0)
    opening_gap_volatility = daily_opening_gap.rolling(window=10).std()
    
    # Combine Adjusted Momentum, Intraday Reversal, Intraday Range Volatility, and Opening Gap
    combined_factor = adjusted_momentum * intraday_reversal * (1 + intraday_range_volatility) + opening_gap_volatility
    
    return combined_factor
```
