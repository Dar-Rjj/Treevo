```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] / df['close'].shift(20) - 1
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(20)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine High-Low Range and Raw Momentum
    combined_result = raw_momentum * high_low_range
    
    # Adjust with Volume Delta
    adjusted_result = combined_result * np.abs(volume_delta)
    
    # Integrate Price Delta
    price_delta = df['close'] - df['close'].shift(1)
    integrated_result = adjusted_result + price_delta
    
    # Apply 7-day Simple Moving Average
    sma_7_result = integrated_result.rolling(window=7).mean()
    
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate VWAP
    vwap = ((df['high'] + df['low']) / 2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    vwap_close_diff = vwap - df['close']
    
    # Combine Daily Momentum and VWAP-Close Difference
    momentum_vwap_factor = daily_momentum * vwap_close_diff
    
    # Confirm with Volume
    volume_change = df['volume'] / df['volume'].shift(1)
    confirmed_factor = momentum_vwap_factor * np.log(volume_change + 1)
    
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Calculate Previous Day's Intraday High-Low Spread
    prev_intraday_spread = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = intraday_spread - prev_intraday_spread
    
    # Generate Alpha Factor Component
    returns_5ma = (df['close'] / df['close'].shift(1) - 1).rolling(window=5).mean()
    volume_amount_change = (df['volume'] - df['volume'].shift(1)) + (df['amount'] - df['amount'].shift(1))
    alpha_factor_component = returns_5ma * np.sign(volume_amount_change)
    
    # Synthesize Enhanced Intraday and Cumulative Momentum Factor
    synthesized_factor = confirmed_factor + intraday_momentum
    
    # Final Alpha Factor
    final_alpha_factor = sma_7_result + synthesized_factor + alpha_factor_component
    
    return final_alpha_factor
```
