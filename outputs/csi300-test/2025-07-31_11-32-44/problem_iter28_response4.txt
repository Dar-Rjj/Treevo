```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] / df['close'].shift(20) - 1
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate True Range
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Price Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    price_volatility = log_returns.rolling(window=20).sum().apply(np.sqrt)
    
    # Calculate Volume Trend
    volume_trend = df['volume'].shift(5) / df['volume'].shift(20)
    
    # Combine Components
    combined_factor = raw_momentum * true_range * price_volatility * volume_trend
    
    # Apply 5-day Exponential Moving Average
    ema_factor = combined_factor.ewm(span=5, adjust=False).mean()
    
    return ema_factor
```
