```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Volume Weighted Close-to-Open Return.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: Series with the factor values indexed by date.
    """
    # Calculate close-to-open return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight the return by volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Normalize the factor to make it interpretable
    normalized_factor = volume_weighted_return / df['volume'].rolling(window=20).mean()
    
    return normalized_factor
```
