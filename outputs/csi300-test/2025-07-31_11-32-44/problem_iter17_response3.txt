```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['1_Period_Return'] = df['close'] / df['close'].shift(1) - 1
    df['5_Period_MA_Momentum'] = df['1_Period_Return'].rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low']) / 2
    df['VWAP'] = (df['Typical_Price'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    df['VWAP_Close_Diff'] = df['close'] - df['VWAP']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['Combined_Factor'] = df['Daily_Momentum'] * df['VWAP_Close_Diff']
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Confirmed_Factor'] = df['Combined_Factor'] * np.log(df['Volume_Change'] + 1)
    
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate N-day Close Price Percent Change (e.g., N=5)
    N = 5
    df['N_Day_Pct_Change'] = df['close'].pct_change(N)
    
    # Compute M-day Average of N-day Close Price Percent Change (e.g., M=5)
    M = 5
    df['M_Day_Avg_N_Day_Pct_Change'] = df['N_Day_Pct_Change'].rolling(window=M).mean()
    
    # Combine Intraday High-Low Spread and Momentum
    df['Momentum_Adjusted_Spread'] = df['Intraday_High_Low_Spread'] * df['M_Day_Avg_N_Day_Pct_Change']
    
    # Adjust with Volume
    lag = 1
    df['Volume_Weight'] = df['volume'].shift(lag)
    ema_span = 5
    df['Smoothed_Volume_Weight'] = df['Volume_Weight'].ewm(span=ema_span, adjust=False).mean()
    df['Adjusted_Momentum'] = df['Momentum_Adjusted_Spread'] * df['Smoothed_Volume_Weight']
    
    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = df['close'] - df['open'] - (df['close'].shift(1) - df['open'].shift(1))
    
    # Introduce Intraday Range Volatility
    df['Intraday_Range'] = df['high'] - df['low']
    P = 5
    df['Intraday_Range_Volatility'] = df['Intraday_Range'].rolling(window=P).std()
    
    # Synthesize All Factors
    df['Synthesized_Factor'] = df['Volume_Confirmed_Factor'] * df['Intraday_Reversal'] * df['Intraday_Range_Volatility']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Synthesized_Factor'] + df['7_Day_SMA_Result']
    df['7_Day_SMA_Result'] = df['Final_Alpha_Factor'].rolling(window=7).mean()
    df['Final_Alpha_Factor'] = df['Daily_Momentum'] * df['VWAP_Close_Diff'] + df['7_Day_SMA_Result']
    
    return df['Final_Alpha_Factor'].dropna()
```
