```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Momentum
    df['raw_momentum'] = df['close'] / df['close'].shift(20) - 1
    
    # Calculate Volume Delta
    df['volume_delta'] = df['volume'] - df['volume'].shift(20)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate True Range
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Price Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    df['price_volatility'] = (log_returns.rolling(window=20).sum() ** 2)
    
    # Combine Components
    df['combined_factor'] = df['raw_momentum'] * df['true_range'] * df['volume_delta'].abs() * np.sqrt(df['price_volatility'])
    
    # Apply 5-day Exponential Moving Average
    df['alpha_factor'] = df['combined_factor'].ewm(span=5, adjust=False).mean()
    
    return df['alpha_factor']
```
