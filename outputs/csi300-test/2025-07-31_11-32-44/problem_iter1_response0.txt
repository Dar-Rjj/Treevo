```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()
    
    # Volume-weight each return
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    
    # Apply Exponential Moving Average (EMA) to volume-weighted returns
    ema_window = 20  # You can adjust this window as needed
    ema_factor = 2 / (ema_window + 1)
    df['ewma_returns'] = df['volume_weighted_return'].ewm(alpha=ema_factor, min_periods=ema_window).mean()
    
    # Apply low price filter
    df['alpha_factor'] = df.apply(
        lambda row: 0 if row['low'] < df.loc[row.name - pd.Timedelta(days=1), 'close'] else row['ewma_returns'],
        axis=1
    )
    
    return df['alpha_factor']
```
