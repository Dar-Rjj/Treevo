```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low'])
    
    # Calculate Intraday Volatility using Average True Range (ATR)
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    df['intraday_volatility'] = df['atr_14']
    
    # Calculate Volume Trend
    df['volume_ema_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Compute Volume Reversal Component
    df['volume_reversal'] = np.where(df['volume'] > df['volume_ema_5'], 1, -1) * df['intraday_return']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['intraday_return'] * df['intraday_range'] * df['intraday_volatility']
    
    # Enhance Volume Reversal Component
    df['volume_reversal_adjustment'] = df['volume_reversal'] * df['intraday_momentum']
    df['enhanced_volume_reversal'] = np.sign(df['volume_reversal_adjustment'])
    
    # Combine Intraday Momentum and Enhanced Volume Reversal
    df['factor'] = df['intraday_momentum'] + df['enhanced_volume_reversal']
    
    return df['factor']
```
