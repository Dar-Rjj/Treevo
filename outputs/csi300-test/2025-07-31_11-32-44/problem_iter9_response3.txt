```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap_numerator = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    vwap = vwap_numerator.rolling(window=5, min_periods=1).sum() / df['volume'].rolling(window=5, min_periods=1).sum()
    
    # Combine Momentum and VWAP-Open Difference
    vwap_open_diff = vwap - df['open']
    combined_factor = daily_momentum * vwap_open_diff
    
    # Incorporate Intraday Reversal
    intraday_high_low_spread = df['high'] - df['low']
    intraday_reversal = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Synthesize Final Alpha Factor
    average_volume = df['volume'].rolling(window=5, min_periods=1).mean()
    final_alpha_factor = (combined_factor * intraday_high_low_spread + intraday_reversal) * (df['volume'] / average_volume)
    
    return final_alpha_factor
```
