import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['1_Period_Return'] = df['close'] / df['close'].shift(1) - 1
    df['5_Period_MA_Momentum'] = df['1_Period_Return'].rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low']) / 2
    df['VWAP'] = (df['Typical_Price'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    df['VWAP_Close_Diff'] = df['close'] - df['VWAP']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['Combined_Momentum_VWAP'] = df['Daily_Momentum'] * df['VWAP_Close_Diff']
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Confirmed_Factor'] = df['Combined_Momentum_VWAP'] * np.log(df['Volume_Change'] + 1)
    
    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['close'] - df['close'].shift(20)
    
    # Combine Intraday and Raw Momentum
    df['Intraday_HL_Spread'] = df['high'] - df['low']
    df['Intraday_HL_Spread_Change'] = df['Intraday_HL_Spread'] - df['Intraday_HL_Spread'].shift(1)
    df['Intraday_Raw_Momentum'] = df['Intraday_HL_Spread_Change'] * df['Raw_Momentum']
    
    # Calculate Volume Delta
    df['Volume_Delta'] = df['volume'] - df['volume'].shift(20)
    df['Adjusted_Momentum'] = df['Intraday_Raw_Momentum'] * df['Volume_Delta'].abs()
    
    # Apply 5-day Simple Moving Average
    df['SMA_5_Adjusted_Momentum'] = df['Adjusted_Momentum'].rolling(window=5).mean()
    
    # Calculate Intraday Reversal
    df['Close_Open_Diff'] = df['close'] - df['open']
    df['Intraday_Reversal'] = df['Close_Open_Diff'] - df['Close_Open_Diff'].shift(1)
    
    # Introduce Intraday Range Volatility
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Range_Volatility'] = df['Intraday_Range'].rolling(window=5).std()
    
    # Synthesize All Factors
    df['Synthesized_Factor'] = df['SMA_5_Adjusted_Momentum'] * df['Intraday_Reversal'] * df['Intraday_Range_Volatility']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Combine High-Low Range and Raw Momentum
    df['HL_Range_Raw_Momentum'] = df['Raw_Momentum'] * df['High_Low_Range']
    
    # Integrate Price Delta
    df['Price_Delta'] = df['close'] - df['close'].shift(1)
    df['Integrated_Result'] = df['Synthesized_Factor'] + df['Price_Delta']
    
    # Apply 7-day Simple Moving Average
    df['SMA_7_Integrated_Result'] = df['Integrated_Result'].rolling(window=7).mean()
    
    # Generate Alpha Factor Component
    df['Alpha_Factor_Component'] = df['5_Period_MA_Momentum'] * np.sign(df['Volume_Change'] + df['amount'] - df['amount'].shift(1))
    
    # Add Intraday Momentum
    df['Intraday_Momentum'] = df['close'] - df['open']
    df['Final_Alpha_Factor'] = df['SMA_7_Integrated_Result'] + df['Intraday_Momentum']
    
    return df['Final_Alpha_Factor'].dropna()
