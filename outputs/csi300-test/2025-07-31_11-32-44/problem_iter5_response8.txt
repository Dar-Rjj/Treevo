```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Combined Momentum
    df['DailyMomentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    N = 21  # Example: 21-day momentum
    df[f'{N}DayMomentum'] = df['close'].pct_change(N)
    M = 5  # Example: 5-day average of 21-day momentum
    df[f'{M}DayAveraged{N}DayMomentum'] = df[f'{N}DayMomentum'].rolling(window=M).mean()

    # Calculate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']

    # Calculate Volume Weighted Average Price (VWAP)
    df['HL_Volume'] = ((df['high'] + df['low']) / 2) * df['volume']
    df['TotalVolume'] = df['volume']
    df['VWAP'] = df['HL_Volume'].cumsum() / df['TotalVolume'].cumsum()

    # Calculate VWAP to Close Difference
    df['VWAPCloseDiff'] = df['close'] - df['VWAP']

    # Combine High-Low Range and Momentum
    df['AdjustedHighLowMomentum'] = df['HighLowRange'] * df[f'{M}DayAveraged{N}DayMomentum']

    # Calculate Volume Reversal
    lag = 1  # Example: 1-day volume change
    df['VolumeChange'] = df['volume'] - df['volume'].shift(lag)
    df['VolumeZScore'] = (df['VolumeChange'] - df['VolumeChange'].rolling(window=20).mean()) / df['VolumeChange'].rolling(window=20).std()

    # Adjust with Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    df['VolumeAdjustment'] = df['volume'] / avg_volume
    df['AdjustedHighLowMomentum'] *= df['VolumeAdjustment']

    # Combine All Factors
    df['AlphaFactor'] = (df[f'{M}DayAveraged{N}DayMomentum'] * 
                         df['VWAPCloseDiff'] * 
                         df['AdjustedHighLowMomentum'] * 
                         df['VolumeZScore'])

    return df['AlphaFactor']
```
