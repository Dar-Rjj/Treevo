```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low']) / 2
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # VWAP-Close Difference
    vwap_close_diff = df['close'] - vwap
    
    # Combine Momentum and VWAP-Close Difference
    momentum_vwap_factor = daily_momentum * vwap_close_diff
    
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Reversal
    intraday_reversal = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (momentum_vwap_factor * high_low_spread) + intraday_reversal
    
    return final_alpha_factor
```
