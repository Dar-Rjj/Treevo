```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-Period Moving Average of Price Momentum
    df['1_Period_Return'] = df['close'].pct_change()
    df['5_Period_MA_Momentum'] = df['1_Period_Return'].rolling(window=5).mean()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df['high'] + df['low']) / 2 * df['volume']
    df['VWAP'] = df['VWAP'].cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP to Close Difference
    df['VWAP_Close_Diff'] = df['close'] - df['VWAP']
    
    # Combine Daily Momentum and VWAP-Close Difference
    df['Combined_Factor'] = df['Daily_Momentum'] * df['VWAP_Close_Diff']
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Confirmed_Factor'] = df['Combined_Factor'] * np.log(df['Volume_Change'] + 1)
    
    # Synthesize with Intraday Reversal and High-Low Spread
    df['Intraday_Reversal'] = df['close'] - df['open'] - (df['close'].shift(1) - df['open'].shift(1))
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    df['Prev_Intraday_High_Low_Spread'] = df['high'].shift(1) - df['low'].shift(1)
    df['Intraday_Momentum'] = df['Intraday_High_Low_Spread'] - df['Prev_Intraday_High_Low_Spread']
    
    df['Synthesized_Factor'] = (df['Volume_Confirmed_Factor'] +
                                df['Intraday_Reversal'] +
                                df['Intraday_Momentum'])
    
    # Integrate with Stochastic Oscillator
    high_14 = df['high'].rolling(window=14).max()
    low_14 = df['low'].rolling(window=14).min()
    df['%K'] = (df['close'] - low_14) / (high_14 - low_14)
    df['%D'] = df['%K'].rolling(window=3).mean()
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Synthesized_Factor'] * np.sign(df['Volume_Change'])
    
    return df['Final_Alpha_Factor']
```
