import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       indexed by date.
    
    Returns:
    pd.Series: A series of the calculated alpha factor, indexed by date.
    """
    # Calculate the daily return
    daily_return = df['close'].pct_change()

    # Calculate the average true range (ATR) over a 14-day window
    tr = df[['high', 'low', 'close']].join(df['close'].shift(1).rename('prev_close'))
    tr['h-l'] = tr['high'] - tr['low']
    tr['h-pc'] = abs(tr['high'] - tr['prev_close'])
    tr['l-pc'] = abs(tr['low'] - tr['prev_close'])
    tr['tr'] = tr[['h-l', 'h-pc', 'l-pc']].max(axis=1)
    atr = tr['tr'].rolling(window=14).mean()

    # Calculate the relative volume
    avg_volume = df['volume'].rolling(window=30).mean()
    relative_volume = df['volume'] / avg_volume

    # Calculate the factor
    alpha_factor = (daily_return * atr) / relative_volume

    return alpha_factor
