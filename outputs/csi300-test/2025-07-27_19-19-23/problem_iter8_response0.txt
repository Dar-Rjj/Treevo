```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Identify Volume Spikes
    volume_change = df['volume'].pct_change()
    spike_threshold = 0.20  # 20% increase
    spike_indicator = (volume_change > spike_threshold).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    hl_spike = high_low_diff * spike_indicator
    hl_spike_sum = hl_spike.rolling(window=10).sum()
    
    # Calculate Daily Log Return
    log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume Spikes
    adjusted_return = log_return * (1 + spike_indicator * 0.5)  # Weight for spike
    smoothed_adjusted_return = adjusted_return.ewm(span=10, adjust=False).mean()
    
    # Calculate Average High-Low Difference
    avg_high_low_diff = high_low_diff.rolling(window=10).mean()
    
    # Calculate Volume Rate of Change (VROC)
    vroc = (df['volume'] / df['volume'].shift(10)) - 1
    
    # Multiply Price Momentum by VROC
    combined_momentum = avg_high_low_diff * vroc
    
    # Adjust Combined Momentum for Volume Spikes
    adjusted_combined_momentum = combined_momentum * (1 + spike_indicator * 0.5)  # Weight for spike
    smoothed_adjusted_combined_momentum = adjusted_combined_momentum.ewm(span=10, adjust=False).mean()
    
    # Cumulative Adjusted Momentum Factor
    cumulative_adjusted_momentum = (1 + smoothed_adjusted_return).cumprod() * (1 + smoothed_adjusted_combined_momentum).cumprod()
    
    return cumulative_adjusted_momentum
```
