import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement
    df['price_movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['price_movement'] * df['volume']
    
    # Aggregate Volume Adjusted Price Movements over a 30-day rolling window
    df['aggregated_volume_adjusted_price_movement'] = df['volume_adjusted_price_movement'].rolling(window=30).sum()
    
    # Calculate the Rolling Average of Aggregated Values with Exponential Weighting for Smoothing
    df['ewma_5'] = df['aggregated_volume_adjusted_price_movement'].ewm(span=5, adjust=False).mean()
    df['ewma_10'] = df['aggregated_volume_adjusted_price_movement'].ewm(span=10, adjust=False).mean()
    
    # Derive Momentum from the Rolling Average
    df['momentum'] = df['ewma_5'] - df['ewma_10']
    
    return df['momentum']
