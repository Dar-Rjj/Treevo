import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']).diff()
    
    # Identify Volume and Price Spikes
    df['vol_20d_ma'] = df['volume'].rolling(window=20).mean()
    df['rel_vol'] = df['volume'] / df['vol_20d_ma']
    df['price_z_score'] = (df['high'] - df['low']) / df['high'].rolling(window=20).std()
    spike_threshold = 1.5
    df['spike'] = ((df['rel_vol'] > spike_threshold) | (df['price_z_score'] > spike_threshold)).astype(int)
    
    # Adjust Momentum for Spikes
    df['adj_log_return'] = df['log_return'] * (1 + 0.5 * df['spike'])
    
    # Smooth the Adjusted Returns
    df['smoothed_adj_return'] = df['adj_log_return'].ewm(span=7).mean()
    
    # Cumulative Momentum Factor
    df['cum_momentum'] = (1 + df['smoothed_adj_return']).cumprod()
    
    # Combine High-Low Difference with Spikes
    df['high_low_diff'] = df['high'] - df['low']
    df['high_low_spike'] = df['high_low_diff'] * df['spike']
    df['sum_high_low_spike'] = df['high_low_spike'].rolling(window=10).sum()
    
    # Calculate Volume-Adjusted Price Movement
    df['daily_price_move'] = df['close'] - df['open']
    df['vol_adj_price_move'] = df['daily_price_move'] * df['volume']
    cap = df['vol_adj_price_move'].rolling(window=30).std() * 3
    df['vol_adj_price_move'] = df['vol_adj_price_move'].clip(-cap, cap)
    
    # Calculate Rolling Momentum Indicator
    df['agg_vol_adj_price_move'] = df['vol_adj_price_move'].rolling(window=30).sum()
    df['ema_5'] = df['agg_vol_adj_price_move'].ewm(span=5).mean()
    df['ema_15'] = df['agg_vol_adj_price_move'].ewm(span=15).mean()
    df['rolling_momentum'] = df['ema_5'] - df['ema_15']
    
    # Combined Momentum Factor
    df['combined_momentum'] = df['cum_momentum'] * df['sum_high_low_spike']
    
    # Volume Trend: Volume Rate of Change (VROC)
    m = 14
    df['vroc'] = df['volume'].pct_change(periods=m)
    df['vroc_smoothed'] = df['vroc'].ewm(span=7).mean()
    
    # Additional Factor: Volatility-Adjusted Momentum
    df['daily_volatility'] = df['close'].rolling(window=20).std()
    df['norm_log_return'] = df['log_return'] / df['daily_volatility']
    df['smoothed_norm_return'] = df['norm_log_return'].ewm(span=7).mean()
    
    # Enhanced Combined Factor
    df['enhanced_combined_factor'] = df['combined_momentum'] + df['smoothed_norm_return']
    
    # Final Combined Alpha Factor
    df['final_combined_alpha'] = df['combined_momentum'] * df['vroc_smoothed'] + df['enhanced_combined_factor']
    
    return df['final_combined_alpha']
