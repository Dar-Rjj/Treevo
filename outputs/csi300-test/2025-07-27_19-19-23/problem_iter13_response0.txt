```python
import pandas as pd
import numpy as np

def heuristics_v2(df, spike_threshold=2.0, lookback_period=10, alpha=0.5):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume Spikes
    rolling_std = df['volume'].rolling(window=lookback_period).std()
    df['volume_z_score'] = (df['volume'] - df['volume'].rolling(window=lookback_period).mean()) / rolling_std
    df['spike_indicator'] = (df['volume_z_score'] > spike_threshold).astype(int)
    
    # Adjust Momentum for Volume Spikes
    weight = 0.8  # Example weight, can be adjusted
    df['adjusted_log_return'] = df['log_return'] * (1 - df['spike_indicator'] * (1 - weight))
    
    # Smooth the Adjusted Returns
    df['smoothed_adjusted_returns'] = df['adjusted_log_return'].ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Combine High-Low Difference and Volume Spikes
    df['high_low_spike'] = df['high_low_diff'] * df['spike_indicator']
    df['summed_high_low_spike'] = df['high_low_spike'].rolling(window=lookback_period).sum()
    
    # Cumulative Momentum Factor
    cumulative_factor = (df['smoothed_adjusted_returns'] + 1).cumprod() + df['summed_high_low_spike']
    
    return cumulative_factor
```
