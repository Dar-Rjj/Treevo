```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=10, spike_threshold=0.2, smoothing_span=5):
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Sum High-Low Differences over last n days
    df['sum_high_low_diff'] = df['high_low_diff'].rolling(window=n).sum()
    
    # Calculate Volume Rate of Change (VROC)
    df['vroc'] = df['volume'].pct_change(periods=m)
    
    # Identify Volume Spikes
    df['daily_volume_pct_change'] = df['volume'].pct_change()
    df['spike_indicator'] = (df['daily_volume_pct_change'] > spike_threshold).astype(int)
    
    # Multiply VROC by Spike Indicator
    df['vroc_spike_product'] = df['vroc'] * df['spike_indicator']
    
    # Calculate Average High-Low Difference
    df['avg_high_low_diff'] = df['sum_high_low_diff'] / n
    
    # Combine Price and Volume Factors
    df['combined_momentum'] = df['sum_high_low_diff'] * df['vroc_spike_product']
    df['sum_combined_momentum'] = df['combined_momentum'].rolling(window=n).sum()
    
    # Adjust Combined Momentum for Volume Spikes
    df['adjusted_combined_momentum'] = df['sum_combined_momentum'] * df['spike_indicator']
    
    # Smooth the Adjusted Combined Momentum
    df['smoothed_adjusted_momentum'] = df['adjusted_combined_momentum'].ewm(span=smoothing_span).mean()
    
    # Cumulative Adjusted Momentum Factor
    daily_returns = (df['smoothed_adjusted_momentum'] + 1).cumprod()
    
    return daily_returns
```
