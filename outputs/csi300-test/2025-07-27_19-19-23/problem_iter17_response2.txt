import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=10, z_score_threshold=2.0, vroc_period=10, ema_span=5):
    # Calculate Daily Price Movement
    df['Daily_Price_Movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['Volume_Adjusted_Movement'] = df['Daily_Price_Movement'] * df['volume']
    
    # Identify Volume Spikes
    df['Volume_Z_Score'] = (df['volume'] - df['volume'].rolling(window=lookback_period).mean()) / df['volume'].rolling(window=lookback_period).std()
    df['Spike_Indicator'] = (df['Volume_Z_Score'] > z_score_threshold).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    df['High_Low_Diff'] = df['high'] - df['low']
    df['High_Low_Vol_Spike'] = df['High_Low_Diff'] * df['Spike_Indicator']
    df['Sum_High_Low_Vol_Spike'] = df['High_Low_Vol_Spike'].rolling(window=lookback_period).sum()
    
    # Adjust Momentum for Volume Spikes
    df['Log_Return'] = np.log(df['close'] / df['open'])
    df['Adjusted_Log_Return'] = np.where(df['Spike_Indicator'] == 1, df['Log_Return'] * 1.5, df['Log_Return'])
    
    # Smooth the Adjusted Returns
    df['Smoothed_Returns'] = df['Adjusted_Log_Return'].ewm(span=ema_span, adjust=False).mean()
    
    # Incorporate High-Low Range into Momentum
    df['High_Low_Range'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['Adjusted_Momentum'] = df['Smoothed_Returns'] * df['High_Low_Range']
    
    # Combine High-Low and Price Momentum
    df['Combined_Momentum'] = df['Sum_High_Low_Vol_Spike'] + df['Adjusted_Momentum']
    
    # Volume Trend
    df['VROC'] = df['volume'] / df['volume'].shift(vroc_period) - 1
    
    # Combined Factor
    df['Combined_Factor'] = df['Combined_Momentum'] * df['VROC']
    
    # Cumulative Momentum Factor
    df['Cumulative_Momentum_Factor'] = (df['Combined_Factor'] + 1).cumprod()
    
    return df['Cumulative_Momentum_Factor']
