```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))

    # Identify Volume Spikes
    df['volume_z_score'] = (df['volume'] - df['volume'].rolling(window=20).mean()) / df['volume'].rolling(window=20).std()
    spike_threshold = 2.0  # Example threshold
    df['is_spike'] = df['volume_z_score'] > spike_threshold

    # Adjust Momentum for Volume Spikes
    weight = 0.5  # Example weight
    df['adjusted_return'] = df['log_return'] * (1 - df['is_spike'] + df['is_spike'] * weight)

    # Smooth the Adjusted Returns using Exponential Moving Average
    df['smoothed_return'] = df['adjusted_return'].ewm(span=20, min_periods=20).mean()

    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (1 + df['smoothed_return']).cumprod().fillna(1)
    
    return df['cumulative_momentum']
```
