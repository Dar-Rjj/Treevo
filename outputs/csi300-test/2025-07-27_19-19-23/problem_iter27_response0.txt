```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market data.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
        
    Returns:
        pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the price change during the day
    df['price_change'] = df['close'] - df['open']
    
    # Calculate the normalized volume
    df['norm_volume'] = (df['volume'] - df['volume'].mean()) / df['volume'].std()
    
    # Calculate the volatility
    df['volatility'] = df['close'].rolling(window=5).std()
    
    # Calculate the momentum
    df['momentum'] = df['close'].pct_change(periods=10)
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['price_change'] / df['volatility']) * df['norm_volume'] * df['momentum']
    
    # Return the alpha factor as a Series
    return df['alpha_factor'].dropna()
```
