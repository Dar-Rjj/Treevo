import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    window = 30
    df['volume_zscore'] = (df['volume'] - df['volume'].rolling(window=window).mean()) / df['volume'].rolling(window=window).std()
    df['price_zscore'] = ((df['high'] + df['low']) / 2 - df['close'].rolling(window=window).mean()) / df['close'].rolling(window=window).std()
    spike_threshold = 3.0
    df['spike'] = (np.abs(df['volume_zscore']) > spike_threshold) | (np.abs(df['price_zscore']) > spike_threshold)
    
    # Adjust Momentum for Volume and Price Spikes
    weight = 1.5
    df['adjusted_return'] = df['log_return'] * np.where(df['spike'], weight, 1)
    
    # Smooth the Adjusted Returns
    span = 10
    df['smoothed_return'] = df['adjusted_return'].ewm(span=span).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (df['smoothed_return'] + 1).cumprod()
    
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_movement'] = df['daily_price_movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    rolling_window = 30
    df['rolling_agg_vol_adj'] = df['volume_adjusted_movement'].rolling(window=rolling_window).sum()
    df['rolling_ewma_5'] = df['rolling_agg_vol_adj'].ewm(span=5).mean()
    df['rolling_ewma_15'] = df['rolling_agg_vol_adj'].ewm(span=15).mean()
    df['momentum_indicator'] = df['rolling_ewma_5'] - df['rolling_ewma_15']
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Combine High-Low Difference with Volume and Price Spikes
    lookback_period = 10
    df['spiked_high_low_diff'] = df['high_low_diff'] * df['spike']
    df['summed_spiked_high_low_diff'] = df['spiked_high_low_diff'].rolling(window=lookback_period).sum()
    
    # Combine High-Low and Price Momentum
    df['combined_momentum'] = df['cumulative_momentum'] + df['summed_spiked_high_low_diff']
    
    # Volume Trend
    m = 10
    df['vroc'] = (df['volume'] - df['volume'].shift(m)) / df['volume'].shift(m)
    
    # Combined Factor
    df['combined_factor'] = df['combined_momentum'] * df['vroc']
    
    # Additional Factor: Volatility-Adjusted Momentum
    volatility_lookback = 20
    df['daily_volatility'] = df['close'].rolling(window=volatility_lookback).std()
    df['normalized_return'] = df['log_return'] / df['daily_volatility']
    df['smoothed_normalized_return'] = df['normalized_return'].ewm(span=span).mean()
    
    # Enhanced Combined Factor
    df['enhanced_combined_factor'] = df['combined_factor'] + df['smoothed_normalized_return']
    
    return df['enhanced_combined_factor']
