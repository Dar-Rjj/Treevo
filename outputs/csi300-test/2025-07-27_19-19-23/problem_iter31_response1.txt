```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']).diff()

    # Identify Volume and Price Spikes
    df['vol_20d_ma'] = df['volume'].rolling(window=20).mean()
    df['relative_vol'] = df['volume'] / df['vol_20d_ma']
    
    df['z_score_price'] = (df['high'] - df['low']) / df['high'].rolling(window=20).std()
    df['spike'] = ((df['relative_vol'] > 2) | (df['z_score_price'] > 2)).astype(int)

    # Adjust Momentum for Spikes
    spike_weight = 1.5  # You can adjust this weight
    df['adjusted_log_return'] = df['log_return'] * (1 + (df['spike'] * (spike_weight - 1)))

    # Smooth the Adjusted Returns
    df['smoothed_adjusted_returns'] = df['adjusted_log_return'].ewm(span=10, min_periods=10).mean()

    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (df['smoothed_adjusted_returns'] + 1).cumprod()

    # Combine High-Low Difference with Spikes
    df['high_low_diff'] = df['high'] - df['low']
    df['hl_diff_with_spike'] = df['high_low_diff'] * df['spike']
    df['summed_hl_diff_with_spike'] = df['hl_diff_with_spike'].rolling(window=10).sum()

    # Combined Momentum Factor
    df['combined_momentum'] = df['cumulative_momentum'] * df['summed_hl_diff_with_spike']

    # Volume-Adjusted Price Movement
    df['daily_price_movement'] = df['close'] - df['open']
    df['volume_adjusted_price_movement'] = df['daily_price_movement'] * df['volume']

    # Rolling Momentum Indicator
    df['rolling_volume_adjusted_price_movement'] = df['volume_adjusted_price_movement'].rolling(window=30).sum()
    df['ema_rolling_volume_adjusted_price_movement_5'] = df['rolling_volume_adjusted_price_movement'].ewm(span=5, min_periods=5).mean()
    df['ema_rolling_volume_adjusted_price_movement_10'] = df['rolling_volume_adjusted_price_movement'].ewm(span=10, min_periods=10).mean()
    df['rolling_momentum'] = df['ema_rolling_volume_adjusted_price_movement_5'] - df['ema_rolling_volume_adjusted_price_movement_10']

    # Volume Trend
    m = 14  # You can adjust this period
    df['vroc'] = df['volume'].pct_change(periods=m)
    df['smoothed_vroc'] = df['vroc'].ewm(span=10, min_periods=10).mean()

    # Final Combined Alpha Factor
    df['final_combined_alpha_factor'] = df['combined_momentum'] * df['smoothed_vroc']

    return df['final_combined_alpha_factor']
```
