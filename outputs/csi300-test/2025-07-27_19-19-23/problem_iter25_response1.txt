import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    df['vol_zscore'] = (df['volume'] - df['volume'].rolling(window=20).mean()) / df['volume'].rolling(window=20).std()
    df['price_zscore'] = (df[['high', 'low']].mean(axis=1) - df[['high', 'low']].mean(axis=1).rolling(window=20).mean()) / df[['high', 'low']].mean(axis=1).rolling(window=20).std()
    spike_threshold = 2.0
    df['spike'] = ((df['vol_zscore'] > spike_threshold) | (df['price_zscore'] > spike_threshold)).astype(int)
    
    # Adjust Momentum for Volume and Price Spikes
    weight = 0.5
    df['adjusted_log_return'] = df['log_return'] * (1 + df['spike'] * (weight - 1))
    
    # Smooth the Adjusted Returns
    df['smoothed_log_return'] = df['adjusted_log_return'].ewm(span=10, adjust=False).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (df['smoothed_log_return'] + 1).cumprod()
    df['cumulative_momentum'] = df['cumulative_momentum'] / df['cumulative_momentum'].iloc[0]
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['aggregated_vol_adj_price'] = df['volume_adjusted_price_movement'].rolling(window=30).sum()
    df['exp_weighted_aggregated_vol_adj_price'] = df['aggregated_vol_adj_price'].ewm(span=30, adjust=False).mean()
    df['momentum_indicator'] = df['exp_weighted_aggregated_vol_adj_price'].ewm(span=5, adjust=False).mean() - df['exp_weighted_aggregated_vol_adj_price'].ewm(span=15, adjust=False).mean()
    
    # Combine High-Low Difference with Volume and Price Spikes
    df['high_low_diff'] = df['high'] - df['low']
    df['spike_high_low_diff'] = df['high_low_diff'] * df['spike']
    df['sum_spike_high_low_diff'] = df['spike_high_low_diff'].rolling(window=10).sum()
    
    # Combined Factor
    df['combined_factor'] = (df['cumulative_momentum'] + df['sum_spike_high_low_diff']) * df['momentum_indicator']
    
    # Additional Factor: Volatility-Adjusted Momentum
    df['daily_volatility'] = df['close'].rolling(window=20).std()
    df['normalized_log_return'] = df['log_return'] / df['daily_volatility']
    df['smoothed_normalized_log_return'] = df['normalized_log_return'].ewm(span=10, adjust=False).mean()
    df['volatility_adjusted_momentum'] = (df['smoothed_normalized_log_return'] + 1).cumprod()
    df['volatility_adjusted_momentum'] = df['volatility_adjusted_momentum'] / df['volatility_adjusted_momentum'].iloc[0]
    
    # Final Factor
    df['final_factor'] = df['combined_factor'] * df['volatility_adjusted_momentum']
    
    return df['final_factor']
