import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=10, vol_lookback=20, ema_span=5, vroc_period=10, spike_threshold=2.0):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Identify Volume Spikes
    volume_zscore = (df['volume'] - df['volume'].rolling(window=lookback_period).mean()) / df['volume'].rolling(window=lookback_period).std()
    spike_indicator = (volume_zscore > spike_threshold).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    combined_diff_spike = (high_low_diff * spike_indicator).rolling(window=lookback_period).sum()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume Spikes
    adjusted_daily_log_return = daily_log_return * (1 + (spike_indicator - 1) * 0.5)
    
    # Smooth the Adjusted Returns
    smoothed_adjusted_returns = adjusted_daily_log_return.ewm(span=ema_span).mean()
    
    # Price Momentum: Weighted Average High-Low Difference
    weights = np.arange(1, lookback_period + 1)
    weighted_high_low_diff = (high_low_diff.rolling(window=lookback_period).apply(lambda x: np.sum(x * weights), raw=True) / np.sum(weights))
    
    # Volume Trend: Volume Rate of Change (VROC)
    vroc = df['volume'].pct_change(periods=vroc_period)
    
    # Combined Factor: Combine High-Low and Price Momentum
    combined_factor = combined_diff_spike + smoothed_adjusted_returns
    
    # Combine with Volume Trend
    combined_factor *= vroc
    
    # Cumulative Momentum Factor
    cumulative_momentum = (1 + smoothed_adjusted_returns).cumprod()
    
    # Additional Factor: Volatility-Adjusted Momentum
    daily_volatility = df['close'].rolling(window=vol_lookback).std()
    normalized_returns = daily_log_return / daily_volatility
    smoothed_normalized_returns = normalized_returns.ewm(span=ema_span).mean()
    
    # Enhanced Combined Factor
    enhanced_combined_factor = combined_factor + smoothed_normalized_returns
    
    # Final Factor
    final_factor = (enhanced_combined_factor * cumulative_momentum) + smoothed_adjusted_returns
    
    return final_factor
