```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement
    df['Daily_Price_Movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['Volume_Adjusted_Price_Movement'] = df['Daily_Price_Movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['Aggregated_Volume_Adjusted_Price_Movement'] = df['Volume_Adjusted_Price_Movement'].rolling(window=30).sum()
    df['Rolling_Average_Aggregated_VAPM'] = df['Aggregated_Volume_Adjusted_Price_Movement'].ewm(span=30, adjust=False).mean()
    df['5_day_EWMA'] = df['Rolling_Average_Aggregated_VAPM'].ewm(span=5, adjust=False).mean()
    df['10_day_EWMA'] = df['Rolling_Average_Aggregated_VAPM'].ewm(span=10, adjust=False).mean()
    df['Momentum'] = df['5_day_EWMA'] - df['10_day_EWMA']
    
    # Calculate High-Low Difference
    df['High_Low_Difference'] = df['high'] - df['low']
    
    # Identify Volume Spikes
    df['Volume_Z_Score'] = (df['volume'] - df['volume'].rolling(window=30).mean()) / df['volume'].rolling(window=30).std()
    df['Volume_Spike_Indicator'] = (df['Volume_Z_Score'] > 2).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    df['High_Low_Spike_Adjusted'] = df['High_Low_Difference'] * df['Volume_Spike_Indicator']
    df['Sum_High_Low_Spike_Adjusted'] = df['High_Low_Spike_Adjusted'].rolling(window=10).sum()
    
    # Adjust Momentum for Volume Spikes
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    weight = 1.5  # Example weight
    df['Adjusted_Log_Return'] = np.where(df['Volume_Spike_Indicator'] == 1, df['Log_Return'] * weight, df['Log_Return'])
    
    # Smooth the Adjusted Returns
    df['Smoothed_Adjusted_Returns'] = df['Adjusted_Log_Return'].ewm(span=10, adjust=False).mean()
    
    # Volume Trend
    m = 14  # Example period for VROC
    df['VROC'] = df['volume'].pct_change(periods=m) * 100
    
    # Combined Factor
    df['Combined_Factor'] = (df['Sum_High_Low_Spike_Adjusted'] + df['Smoothed_Adjusted_Returns']) * df['VROC']
    
    # Cumulative Momentum Factor
    df['Cumulative_Momentum_Factor'] = (df['Combined_Factor'] + 1).cumprod()
    
    return df['Cumulative_Momentum_Factor']
```
