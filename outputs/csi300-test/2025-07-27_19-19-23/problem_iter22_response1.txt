```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume Spikes
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / df['volume_ma_20']
    spike_threshold = 2.0  # Example threshold
    df['spike_indicator'] = (relative_volume > spike_threshold).astype(int)
    
    # Adjust Momentum for Volume Spikes
    weight = 2.0  # Example weight
    df['adjusted_log_return'] = df['log_return'] * (1 + df['spike_indicator'] * (weight - 1))
    
    # Smooth the Adjusted Returns using Exponential Moving Average
    smoothing_span = 10
    df['smoothed_adjusted_return'] = df['adjusted_log_return'].ewm(span=smoothing_span, adjust=False).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (df['smoothed_adjusted_return'] + 1).cumprod()
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Combine High-Low Difference and Volume Spikes
    lookback_period = 10
    df['high_low_diff_spike'] = df['high_low_diff'] * df['spike_indicator']
    df['sum_high_low_diff_spike'] = df['high_low_diff_spike'].rolling(window=lookback_period).sum()
    
    # Calculate Volume Trend (Volume Rate of Change - VROC)
    vroc_period = 10
    df['vroc'] = df['volume'].pct_change(periods=vroc_period)
    
    # Combined Factor
    df['combined_factor'] = (df['cumulative_momentum'] + df['sum_high_low_diff_spike']) * df['vroc']
    
    return df['combined_factor']
```
