```python
import pandas as pd

def heuristics_v2(df, lookback=10, spike_threshold=0.2):
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate Daily Volume Percentage Change
    df['volume_pct_change'] = df['volume'].pct_change()
    
    # Identify Volume Spikes
    df['volume_spike'] = (df['volume_pct_change'] > spike_threshold).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    df['high_low_with_spike'] = df['high_low_diff'] * df['volume_spike']
    
    # Sum over the fixed lookback period
    df['factor_value'] = df['high_low_with_spike'].rolling(window=lookback).sum()
    
    return df['factor_value']
```
