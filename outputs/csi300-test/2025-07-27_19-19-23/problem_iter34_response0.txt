```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=10, spike_threshold=2.0, e_span=5, vroc_period=10, vol_lookback=20):
    # Calculate High-Low Difference
    df['HighLowDiff'] = df['high'] - df['low']
    
    # Identify Volume Spikes
    df['VolumeZScore'] = (df['volume'] - df['volume'].rolling(window=lookback_period).mean()) / df['volume'].rolling(window=lookback_period).std()
    df['VolumeSpike'] = df['VolumeZScore'].apply(lambda x: 1 if x > spike_threshold else 0)
    
    # Combine High-Low Difference and Volume Spikes
    df['HL_VolumeSpike'] = df['HighLowDiff'] * df['VolumeSpike']
    df['HL_VolumeSpike_Sum'] = df['HL_VolumeSpike'].rolling(window=lookback_period).sum()
    
    # Calculate Daily Log Return
    df['LogReturn'] = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume Spikes
    df['AdjustedReturn'] = df['LogReturn'] * (1 + df['VolumeSpike'] * 0.5)  # Weight for Spike
    
    # Smooth the Adjusted Returns
    df['SmoothedReturn'] = df['AdjustedReturn'].ewm(span=e_span).mean()
    
    # Price Momentum
    weights = np.arange(1, lookback_period + 1)
    df['WeightedHighLowDiff'] = (df['HighLowDiff'].rolling(window=lookback_period).apply(lambda x: np.sum(x * weights), raw=True))
    
    # Volume Trend
    df['VROC'] = df['volume'].pct_change(periods=vroc_period)
    
    # Combined Factor
    df['CombinedFactor'] = df['SmoothedReturn'] + df['WeightedHighLowDiff'] * df['VROC']
    
    # Additional Factor: Volatility-Adjusted Momentum
    df['DailyVolatility'] = df['LogReturn'].rolling(window=vol_lookback).std()
    df['NormalizedReturn'] = df['LogReturn'] / df['DailyVolatility']
    df['SmoothedNormalizedReturn'] = df['NormalizedReturn'].ewm(span=e_span).mean()
    
    # Enhanced Combined Factor
    df['EnhancedCombinedFactor'] = df['CombinedFactor'] + df['SmoothedNormalizedReturn']
    
    # Final Factor
    df['CumulativeMomentum'] = df['EnhancedCombinedFactor'].expanding(min_periods=lookback_period).sum()
    df['FinalFactor'] = df['EnhancedCombinedFactor'] * df['CumulativeMomentum'] + df['SmoothedReturn']
    
    return df['FinalFactor']
```
