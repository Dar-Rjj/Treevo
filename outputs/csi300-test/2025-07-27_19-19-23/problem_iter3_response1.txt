import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement
    df['price_movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['price_movement'] * df['volume']
    
    # Aggregate Volume Adjusted Price Movements over a 30-day rolling window
    df['rolling_sum'] = df['volume_adjusted_price_movement'].rolling(window=30).sum()
    
    # Calculate the Rolling Average of Aggregated Values
    df['rolling_avg_30'] = df['rolling_sum'] / df['volume'].rolling(window=30).sum()
    
    # Derive Momentum from the Rolling Average
    df['momentum'] = df['rolling_avg_30'] - df['rolling_avg_30'].rolling(window=5).mean()
    
    # Return the momentum factor
    return df['momentum'].dropna()
