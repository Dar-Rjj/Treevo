    df['z_score_price'] = (df['high'] - df['low']) / ((df['high'] + df['low']) * 0.5).rolling(window=20).std()

    weight = 1.5

    cumulative_momentum.iloc[0] = 1
    df['cumulative_momentum'] = cumulative_momentum

    df['high_low_diff'] = df['high'] - df['low']
