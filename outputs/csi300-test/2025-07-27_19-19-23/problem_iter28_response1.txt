```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=10, spike_threshold=2.0, ema_span=5, vroc_period=10, volatility_lookback=10):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Identify Volume Spikes
    rolling_std_volume = df['volume'].rolling(window=lookback_period).std()
    z_score_volume = (df['volume'] - df['volume'].rolling(window=lookback_period).mean()) / rolling_std_volume
    volume_spike = (z_score_volume > spike_threshold).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    high_low_spike = (high_low_diff * volume_spike).rolling(window=lookback_period).sum()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume Spikes
    weight = 1.5  # Example weight
    adjusted_daily_log_return = daily_log_return.where(volume_spike == 0, daily_log_return * weight)
    
    # Smooth the Adjusted Returns
    smoothed_returns = adjusted_daily_log_return.ewm(span=ema_span, adjust=False).mean()
    
    # Price Momentum
    weighted_high_low_diff = (high_low_diff * np.arange(1, len(high_low_diff) + 1)).rolling(window=lookback_period).sum() / lookback_period
    
    # Volume Trend: Volume Rate of Change (VROC)
    vroc = df['volume'].pct_change(periods=vroc_period)
    
    # Combined Factor
    combined_factor = (high_low_spike + smoothed_returns) * vroc
    
    # Additional Factor: Volatility-Adjusted Momentum
    daily_volatility = df['close'].rolling(window=volatility_lookback).std()
    normalized_returns = daily_log_return / daily_volatility
    smoothed_normalized_returns = normalized_returns.ewm(span=ema_span, adjust=False).mean()
    
    # Incorporate Volatility-Adjusted Momentum
    enhanced_combined_factor = combined_factor + smoothed_normalized_returns
    
    return enhanced_combined_factor
```
