```python
import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Average High-Low Difference over last n days
    price_momentum = high_low_diff.rolling(window=n).mean()
    
    # Volume Rate of Change (VROC) for period m
    vroc = (df['volume'] / df['volume'].shift(m) - 1) * 100
    
    # Combined Factor: Multiply Price Momentum and VROC
    combined_factor = price_momentum * vroc
    
    return combined_factor
```
