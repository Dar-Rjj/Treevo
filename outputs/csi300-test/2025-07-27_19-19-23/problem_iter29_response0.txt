```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Identify Volume Spikes
    z_score_volume = (df['volume'] - df['volume'].rolling(window=30).mean()) / df['volume'].rolling(window=30).std()
    volume_spike = (z_score_volume > 2).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    hl_diff_spiked = high_low_diff * volume_spike
    summed_hl_diff_spiked = hl_diff_spiked.rolling(window=10).sum()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume Spikes
    weighted_daily_log_return = daily_log_return * (1 + (volume_spike * 0.5))
    
    # Smooth the Adjusted Returns
    smoothed_returns = weighted_daily_log_return.ewm(span=10, adjust=False).mean()
    
    # Price Momentum
    weighted_high_low_diff = (high_low_diff * np.arange(1, 11)).rolling(window=10).sum() / 55
    
    # Volume Trend (VROC)
    vroc = df['volume'].pct_change(periods=5) * 100
    
    # Combined Factor
    combined_factor = summed_hl_diff_spiked + smoothed_returns
    combined_with_volume_trend = combined_factor * vroc
    
    # Volatility-Adjusted Momentum
    daily_volatility = df['close'].rolling(window=30).std()
    normalized_returns = daily_log_return / daily_volatility
    smoothed_normalized_returns = normalized_returns.ewm(span=10, adjust=False).mean()
    
    # Enhanced Combined Factor
    enhanced_combined_factor = combined_with_volume_trend + smoothed_normalized_returns
    
    # Calculate Daily Price Movement
    daily_price_movement = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    volume_adjusted_price_movement = daily_price_movement * df['volume']
    
    # Calculate Rolling Momentum Indicator
    rolling_momentum = volume_adjusted_price_movement.ewm(span=30, adjust=False).mean()
    momentum_indicator = rolling_momentum.ewm(span=5, adjust=False).mean() - rolling_momentum.ewm(span=10, adjust=False).mean()
    
    # Improved Combined Factor
    final_factor = enhanced_combined_factor + momentum_indicator
    
    return final_factor
```
