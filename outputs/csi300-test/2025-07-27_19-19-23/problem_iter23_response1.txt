```python
import pandas as pd
import numpy as np

def heuristics_v2(df, spike_threshold=2.0, volume_spike_threshold=1.2, momentum_weight=0.5, ema_span=10, vroc_period=10):
    # Calculate Daily Log Return
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    df['volume_zscore'] = (df['volume'] - df['volume'].rolling(window=20).mean()) / df['volume'].rolling(window=20).std()
    df['price_zscore'] = (df[['high', 'low']].mean(axis=1) - df[['high', 'low']].mean(axis=1).rolling(window=20).mean()) / df[['high', 'low']].mean(axis=1).rolling(window=20).std()
    df['is_spike'] = ((df['volume_zscore'] > spike_threshold) | (df['price_zscore'] > spike_threshold)).astype(int)
    
    # Adjust Momentum for Volume and Price Spikes
    df['adjusted_daily_log_return'] = df['daily_log_return'] * np.where(df['is_spike'], momentum_weight, 1)
    
    # Smooth the Adjusted Returns
    df['smoothed_adjusted_returns'] = df['adjusted_daily_log_return'].ewm(span=ema_span).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum_factor'] = (df['smoothed_adjusted_returns'] + 1).cumprod().fillna(1)
    
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['daily_price_movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['aggregated_volume_adjusted_price_movement'] = df['volume_adjusted_price_movement'].rolling(window=30).sum()
    df['rolling_momentum_indicator'] = df['aggregated_volume_adjusted_price_movement'].ewm(span=ema_span).mean()
    
    # Calculate High-Low Difference
    df['high_low_difference'] = df['high'] - df['low']
    
    # Combine High-Low Difference with Volume and Price Spikes
    df['combined_high_low_spike'] = df['high_low_difference'] * df['is_spike']
    df['summed_combined_high_low_spike'] = df['combined_high_low_spike'].rolling(window=10).sum()
    
    # Combine High-Low and Price Momentum
    df['combined_high_low_price_momentum'] = df['summed_combined_high_low_spike'] + df['cumulative_momentum_factor']
    
    # Volume Trend
    df['vroc'] = (df['volume'] - df['volume'].shift(vroc_period)) / df['volume'].shift(vroc_period)
    
    # Combined Factor
    df['combined_factor'] = df['combined_high_low_price_momentum'] * df['vroc']
    
    # Additional Factor: Volatility-Adjusted Momentum
    df['daily_volatility'] = df['close'].rolling(window=20).std()
    df['normalized_daily_log_return'] = df['daily_log_return'] / df['daily_volatility']
    df['smoothed_normalized_returns'] = df['normalized_daily_log_return'].ewm(span=ema_span).mean()
    
    # Enhanced Combined Factor
    df['enhanced_combined_factor'] = df['combined_factor'] + df['smoothed_normalized_returns']
    
    # Identify Volume Spikes
    df['relative_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['is_volume_spike'] = (df['relative_volume'] > volume_spike_threshold).astype(int)
    
    # Adjust Momentum for Volume Spikes
    df['final_adjusted_return'] = df['daily_log_return'] * np.where(df['is_volume_spike'], momentum_weight, 1)
    
    # Final Factor
    df['final_factor'] = df['enhanced_combined_factor'] * (df['final_adjusted_return'] + 1)
    
    return df['final_factor']
```
