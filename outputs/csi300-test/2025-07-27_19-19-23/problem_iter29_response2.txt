```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    df['volume_20_ma'] = df['volume'].rolling(window=20).mean()
    df['relative_volume'] = df['volume'] / df['volume_20_ma']
    
    df['volume_z_score'] = (df['volume'] - df['volume'].rolling(window=20).mean()) / df['volume'].rolling(window=20).std()
    df['price_z_score'] = (df['high'] - df['low']) / (df['high'] - df['low']).rolling(window=20).std()
    
    spike_threshold = 2.0
    df['spike'] = ((df['volume_z_score'] > spike_threshold) | (df['relative_volume'] > spike_threshold)).astype(int)
    
    # Adjust Momentum for Volume and Price Spikes
    weight = 1.5
    df['adjusted_log_return'] = df['daily_log_return'] * (df['spike'] * weight + (1 - df['spike']))
    
    # Smooth the Adjusted Returns
    df['smoothed_adjusted_returns'] = df['adjusted_log_return'].ewm(span=10).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (df['smoothed_adjusted_returns'] + 1).cumprod()
    
    # High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Combine High-Low Difference with Volume and Price Spikes
    df['high_low_spike'] = df['high_low_diff'] * df['spike']
    df['summed_high_low_spike'] = df['high_low_spike'].rolling(window=10).sum()
    
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['close'] - df['open']
    
    # Volume-Adjusted Price Movement
    cap = 10000
    df['volume_adjusted_price_movement'] = df['daily_price_movement'] * df['volume']
    df['volume_adjusted_price_movement'] = df['volume_adjusted_price_movement'].clip(upper=cap)
    
    # Rolling Momentum Indicator
    df['aggregated_volume_adjusted_price_movement'] = df['volume_adjusted_price_movement'].rolling(window=30).sum()
    df['rolling_momentum'] = df['aggregated_volume_adjusted_price_movement'].ewm(span=10).mean() - df['aggregated_volume_adjusted_price_movement'].ewm(span=5).mean()
    
    # Combined Momentum and High-Low Difference
    df['combined_momentum_and_high_low'] = df['cumulative_momentum'] * df['summed_high_low_spike']
    
    # Volume Trend: Volume Rate of Change (VROC)
    m = 14
    df['vroc'] = (df['volume'] - df['volume'].shift(m)) / df['volume'].shift(m) * 100
    df['smoothed_vroc'] = df['vroc'].ewm(span=10).mean()
    
    # Final Combined Alpha Factor
    df['final_combined_alpha_factor'] = df['combined_momentum_and_high_low'] * df['smoothed_vroc']
    
    # Additional Factor: Volatility-Adjusted Momentum
    df['daily_volatility'] = df['close'].rolling(window=20).std()
    df['normalized_daily_log_return'] = df['daily_log_return'] / df['daily_volatility']
    df['smoothed_normalized_returns'] = df['normalized_daily_log_return'].ewm(span=10).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['final_combined_alpha_factor'] * df['smoothed_normalized_returns']
    
    return df['alpha_factor']
```
