```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    df['rolling_volume_20'] = df['volume'].rolling(window=20).mean()
    df['relative_volume'] = df['volume'] / df['rolling_volume_20']
    df['high_low_z_score'] = (df['high'] - df['low']) / df['close'].rolling(window=20).std()
    df['spike'] = (df['relative_volume'] > 1.5) | (df['high_low_z_score'] > 2)
    
    # Adjust Momentum for Spikes
    weight = 1.5  # Example weight
    df['adjusted_log_return'] = df['log_return'] * np.where(df['spike'], weight, 1)
    
    # Smooth the Adjusted Returns
    df['smoothed_returns'] = df['adjusted_log_return'].ewm(span=10).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (1 + df['smoothed_returns']).cumprod().fillna(1)
    
    # Combine High-Low Difference with Spikes
    df['high_low_diff'] = df['high'] - df['low']
    df['high_low_diff_spikes'] = df['high_low_diff'] * df['spike']
    df['sum_high_low_diff_spikes'] = df['high_low_diff_spikes'].rolling(window=10).sum()
    
    # Calculate Volume-Adjusted Price Movement
    df['price_movement'] = df['close'] - df['open']
    df['volume_adjusted_price_movement'] = df['price_movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['aggregated_volume_adjusted'] = df['volume_adjusted_price_movement'].rolling(window=30).sum()
    df['ewma_5'] = df['aggregated_volume_adjusted'].ewm(span=5).mean()
    df['ewma_15'] = df['aggregated_volume_adjusted'].ewm(span=15).mean()
    df['rolling_momentum_indicator'] = df['ewma_5'] - df['ewma_15']
    
    # Combined Momentum Factor
    df['combined_momentum'] = df['cumulative_momentum'] * df['sum_high_low_diff_spikes']
    
    # Volume Trend: Volume Rate of Change (VROC)
    m = 14  # Example period
    df['vroc'] = (df['volume'] - df['volume'].shift(m)) / df['volume'].shift(m)
    df['smoothed_vroc'] = df['vroc'].ewm(span=10).mean()
    
    # Additional Factor: Volatility-Adjusted Momentum
    df['daily_volatility'] = df['close'].rolling(window=20).std()
    df['normalized_log_return'] = df['log_return'] / df['daily_volatility']
    df['smoothed_normalized_returns'] = df['normalized_log_return'].ewm(span=10).mean()
    
    # Enhanced Combined Factor
    df['enhanced_combined_factor'] = df['combined_momentum'] + df['smoothed_normalized_returns']
    
    # Final Combined Alpha Factor
    df['final_alpha_factor'] = df['combined_momentum'] * df['smoothed_vroc'] + df['enhanced_combined_factor']
    
    return df['final_alpha_factor']
```
