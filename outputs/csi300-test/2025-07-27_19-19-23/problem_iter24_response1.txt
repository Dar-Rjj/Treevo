```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    window = 30
    df['volume_zscore'] = (df['volume'] - df['volume'].rolling(window).mean()) / df['volume'].rolling(window).std()
    df['price_zscore'] = (df['high'] - df['low']) / (df['high'] - df['low']).rolling(window).std()
    spike_threshold = 2.0
    df['spike'] = ((df['volume_zscore'] > spike_threshold) | (df['price_zscore'] > spike_threshold)).astype(int)
    
    # Adjust Momentum for Volume and Price Spikes
    weight = 0.5
    df['adjusted_return'] = df['log_return'] * (1 - weight * df['spike'])
    
    # Smooth the Adjusted Returns
    span = 10
    df['smoothed_return'] = df['adjusted_return'].ewm(span=span, adjust=False).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (df['smoothed_return'] + 1).cumprod()
    
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['close'] - df['open']
    
    # Calculate Volume-Adjusted Price Movement
    cap = 1000000
    df['volume_adjusted_movement'] = df['daily_price_movement'] * df['volume']
    df['volume_adjusted_movement'] = df['volume_adjusted_movement'].clip(upper=cap)
    
    # Calculate Rolling Momentum Indicator
    rolling_window = 30
    df['agg_volume_adjusted_movement'] = df['volume_adjusted_movement'].rolling(rolling_window).sum()
    df['rolling_avg_agg_values'] = df['agg_volume_adjusted_movement'].ewm(span=span, adjust=False).mean()
    df['momentum'] = df['rolling_avg_agg_values'].ewm(span=5, adjust=False).mean() - df['rolling_avg_agg_values'].ewm(span=10, adjust=False).mean()
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Combine High-Low Difference with Volume and Price Spikes
    lookback_period = 10
    df['high_low_diff_spiked'] = df['high_low_diff'] * df['spike']
    df['high_low_diff_spiked_sum'] = df['high_low_diff_spiked'].rolling(lookback_period).sum()
    
    # Price Momentum
    df['weighted_high_low_diff'] = df['high_low_diff'].ewm(span=span, adjust=False).mean()
    
    # Volume Trend
    vroc_period = 14
    df['vroc'] = (df['volume'] / df['volume'].shift(vroc_period) - 1) * 100
    df['smoothed_vroc'] = df['vroc'].ewm(span=span, adjust=False).mean()
    
    # Combined Factor
    df['combined_factor'] = df['cumulative_momentum'] + df['high_low_diff_spiked_sum'] + df['weighted_high_low_diff']
    df['combined_factor'] = df['combined_factor'] * df['smoothed_vroc']
    
    # Cumulative Adjusted Momentum Factor
    df['cumulative_adjusted_momentum'] = (df['combined_factor'] + 1).cumprod()
    
    return df['cumulative_adjusted_momentum']
```
