import pandas as pd
import numpy as np
from scipy.stats import zscore

def heuristics_v2(df, lookback_period=10, spike_threshold=2.0, ema_span=5, vroc_period=10):
    # Calculate High-Low Difference
    df['High_Low_Diff'] = df['high'] - df['low']
    
    # Identify Volume and Price Spikes
    df['Volume_ZScore'] = (df['volume'] - df['volume'].rolling(window=lookback_period).mean()) / df['volume'].rolling(window=lookback_period).std()
    df['Price_ZScore'] = (df['close'] - df['close'].rolling(window=lookback_period).mean()) / df['close'].rolling(window=lookback_period).std()
    df['Spike_Indicator'] = ((df['Volume_ZScore'].abs() > spike_threshold) | (df['Price_ZScore'].abs() > spike_threshold)).astype(int)
    
    # Combine High-Low Difference and Volume and Price Spikes
    df['HL_Diff_Spike'] = df['High_Low_Diff'] * df['Spike_Indicator']
    df['Sum_HL_Diff_Spike'] = df['HL_Diff_Spike'].rolling(window=lookback_period).sum()
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust Momentum for Volume and Price Spikes
    df['Adjusted_Return'] = df['Daily_Log_Return'] * (1 + 0.5 * df['Spike_Indicator'])
    
    # Smooth the Adjusted Returns
    df['Smoothed_Adjusted_Return'] = df['Adjusted_Return'].ewm(span=ema_span).mean()
    
    # Cumulative Momentum Factor
    df['Cumulative_Momentum'] = (1 + df['Smoothed_Adjusted_Return']).cumprod()
    
    # Calculate Daily Price Movement
    df['Daily_Price_Movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['Volume_Adjusted_Price_Movement'] = df['Daily_Price_Movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['Aggregated_Volume_Adjusted_PM'] = df['Volume_Adjusted_Price_Movement'].rolling(window=30).sum()
    df['Rolling_Average'] = df['Aggregated_Volume_Adjusted_PM'].ewm(span=5).mean()
    df['Momentum'] = df['Aggregated_Volume_Adjusted_PM'].ewm(span=5).mean() - df['Aggregated_Volume_Adjusted_PM'].ewm(span=10).mean()
    
    # Combine High-Low and Price Momentum
    df['Combined_Factor'] = df['Sum_HL_Diff_Spike'] + df['Smoothed_Adjusted_Return']
    
    # Volume Trend (VROC)
    df['VROC'] = df['volume'].pct_change(periods=vroc_period)
    
    # Combined Factor with Volume Trend
    df['Combined_Factor_with_VT'] = df['Combined_Factor'] * df['VROC']
    
    # Additional Factor: Volatility-Adjusted Momentum
    df['Daily_Volatility'] = df['close'].rolling(window=lookback_period).std()
    df['Normalized_Return'] = df['Daily_Log_Return'] / df['Daily_Volatility']
    df['Smoothed_Normalized_Return'] = df['Normalized_Return'].ewm(span=ema_span).mean()
    
    # Enhanced Combined Factor
    df['Enhanced_Combined_Factor'] = df['Combined_Factor_with_VT'] + df['Smoothed_Normalized_Return']
    
    return df['Enhanced_Combined_Factor']
