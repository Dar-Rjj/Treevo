```python
import pandas as pd
import numpy as np

def heuristics_v2(df, threshold=2.0, ema_span=10, roc_period=10):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    df['volume_zscore'] = (df['volume'] - df['volume'].rolling(window=30).mean()) / df['volume'].rolling(window=30).std()
    df['price_zscore'] = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2).rolling(window=30).std()
    
    df['spike_indicator'] = (np.abs(df['volume_zscore']) > threshold) | (np.abs(df['price_zscore']) > threshold)
    
    # Adjust Momentum for Volume and Price Spikes
    spike_weight = 0.5  # Example weight; can be adjusted
    df['adjusted_log_return'] = df['log_return'] * (1 - df['spike_indicator'] * (1 - spike_weight))
    
    # Smooth the Adjusted Returns
    df['smoothed_returns'] = df['adjusted_log_return'].ewm(span=ema_span, adjust=False).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (1 + df['smoothed_returns']).cumprod()
    
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['daily_price_movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['rolling_avg_30'] = df['volume_adjusted_price_movement'].rolling(window=30).sum() / df['volume'].rolling(window=30).sum()
    df['rolling_avg_5'] = df['volume_adjusted_price_movement'].ewm(span=5, adjust=False).mean()
    df['rolling_avg_10'] = df['volume_adjusted_price_movement'].ewm(span=10, adjust=False).mean()
    df['rolling_momentum'] = df['rolling_avg_5'] - df['rolling_avg_10']
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Combine High-Low Difference with Volume and Price Spikes
    df['combined_high_low_diff'] = df['high_low_diff'] * df['spike_indicator']
    df['summed_high_low_diff'] = df['combined_high_low_diff'].rolling(window=10).sum()
    
    # Combine High-Low and Price Momentum
    df['combined_momentum'] = df['cumulative_momentum'] + df['summed_high_low_diff']
    
    # Volume Trend (VROC)
    df['vroc'] = df['volume'].pct_change(periods=roc_period)
    
    # Combined Factor
    df['final_factor'] = df['combined_momentum'] * df['vroc']
    
    return df['final_factor']
```
