```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=5, spike_threshold=3, weight_on_spike=0.5, ema_span=7):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Average High-Low Difference
    avg_high_low_diff = high_low_diff.rolling(window=n).mean()
    
    # Calculate Daily Log Return
    log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Integrated Momentum
    integrated_momentum = (avg_high_low_diff * log_return).fillna(0)
    
    # Compute Volume Rate of Change (VROC)
    vroc = df['volume'].pct_change(periods=m).fillna(0)
    
    # Multiply Integrated Momentum by VROC
    combined_momentum = integrated_momentum * vroc
    
    # Identify Volume Spikes
    rolling_std = df['volume'].rolling(window=n).std()
    z_score = (df['volume'] - df['volume'].rolling(window=n).mean()) / rolling_std
    is_spike = (z_score.abs() > spike_threshold)
    
    # Adjust Combined Momentum for Volume Spikes
    adjusted_momentum = combined_momentum.where(~is_spike, combined_momentum * weight_on_spike)
    
    # Smooth the Adjusted Combined Momentum
    smoothed_momentum = adjusted_momentum.ewm(span=ema_span, adjust=False).mean()
    
    # Cumulative Adjusted Momentum Factor
    cumulative_adjusted_momentum = (smoothed_momentum + 1).cumprod()
    
    return cumulative_adjusted_momentum
```
