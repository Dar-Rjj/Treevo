```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Difference
    df['High-Low'] = df['high'] - df['low']
    
    # Identify Volume Spikes
    df['Volume_ZScore'] = (df['volume'] - df['volume'].rolling(window=20).mean()) / df['volume'].rolling(window=20).std()
    spike_threshold = 2.0
    df['Spike_Indicator'] = (df['Volume_ZScore'] > spike_threshold).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    lookback_period = 10
    df['High-Low_Spike'] = df['High-Low'] * df['Spike_Indicator']
    df['Sum_High-Low_Spike'] = df['High-Low_Spike'].rolling(window=lookback_period).sum()
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume Spikes
    weight = 1.5
    df['Adjusted_Daily_Log_Return'] = df['Daily_Log_Return'] * (1 + (weight - 1) * df['Spike_Indicator'])
    
    # Smooth the Adjusted Returns
    df['Smoothed_Adjusted_Returns'] = df['Adjusted_Daily_Log_Return'].ewm(span=lookback_period, adjust=False).mean()
    
    # Price Momentum
    df['Weighted_High-Low'] = df['High-Low'].rolling(window=lookback_period).apply(lambda x: np.sum(x * np.arange(lookback_period, 0, -1) / np.sum(np.arange(lookback_period, 0, -1))), raw=True)
    
    # Volume Trend (VROC)
    m = 10
    df['VROC'] = df['volume'].pct_change(periods=m)
    
    # Combined Factor
    df['Combined_Factor'] = df['Sum_High-Low_Spike'] + df['Smoothed_Adjusted_Returns']
    df['Combined_Factor'] = df['Combined_Factor'] * df['VROC']
    
    # Additional Factor: Volatility-Adjusted Momentum
    volatility_lookback = 20
    df['Daily_Volatility'] = df['close'].rolling(window=volatility_lookback).std()
    df['Normalized_Return'] = df['Daily_Log_Return'] / df['Daily_Volatility']
    df['Smoothed_Normalized_Return'] = df['Normalized_Return'].ewm(span=volatility_lookback, adjust=False).mean()
    
    # Enhanced Combined Factor
    df['Enhanced_Combined_Factor'] = df['Combined_Factor'] + df['Smoothed_Normalized_Return']
    
    # Calculate Daily Price Movement
    df['Daily_Price_Movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['Volume_Adjusted_Price_Movement'] = df['Daily_Price_Movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    rolling_window = 30
    df['Aggregated_Volume_Adjusted'] = df['Volume_Adjusted_Price_Movement'].rolling(window=rolling_window).sum()
    df['Rolling_Average'] = df['Aggregated_Volume_Adjusted'].ewm(span=rolling_window, adjust=False).mean()
    df['Momentum_Indicator'] = df['Rolling_Average'].ewm(span=5, adjust=False).mean() - df['Rolling_Average'].ewm(span=10, adjust=False).mean()
    
    # Improved Combined Factor
    df['Improved_Combined_Factor'] = df['Enhanced_Combined_Factor'] + df['Momentum_Indicator']
    
    return df['Improved_Combined_Factor']
```
