```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Identify Volume and Price Spikes
    df['volume_zscore'] = (df['volume'] - df['volume'].rolling(window=20).mean()) / df['volume'].rolling(window=20).std()
    df['relative_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['price_zscore'] = (df['close'] - df['close'].rolling(window=20).mean()) / df['close'].rolling(window=20).std()
    
    # Classify as Spike if either Z-Score or Relative Volume exceeds a threshold
    df['spike_indicator'] = ((df['volume_zscore'] > 3) | (df['relative_volume'] > 2)).astype(int)
    
    # Combine High-Low Difference and Volume-Price Spikes
    df['hl_spiked'] = df['high_low_diff'] * df['spike_indicator']
    df['sum_hl_spiked'] = df['hl_spiked'].rolling(window=10).sum()
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume-Price Spikes
    df['adjusted_log_return'] = df['log_return'] * (1.5 * df['spike_indicator'] + 1 * (1 - df['spike_indicator']))
    
    # Smooth the Adjusted Returns
    df['smoothed_adjusted_returns'] = df['adjusted_log_return'].ewm(span=7).mean()
    
    # Price Momentum
    df['weighted_high_low'] = (df['high_low_diff'].rolling(window=10).apply(lambda x: np.sum(x * np.arange(1, len(x) + 1))))
    
    # Volume Trend
    df['vroc'] = df['volume'].pct_change(periods=14)
    df['smoothed_vroc'] = df['vroc'].ewm(span=7).mean()
    
    # Combined Factor
    df['combined_factor'] = df['sum_hl_spiked'] + df['smoothed_adjusted_returns']
    df['final_factor'] = df['combined_factor'] * df['smoothed_vroc']
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (df['final_factor'] + 1).cumprod()
    
    # Additional Factor: Volatility-Adjusted Momentum
    df['daily_volatility'] = df['close'].rolling(window=20).std()
    df['normalized_log_return'] = df['log_return'] / df['daily_volatility']
    df['smoothed_normalized_returns'] = df['normalized_log_return'].ewm(span=7).mean()
    
    # Enhanced Combined Factor
    df['enhanced_combined_factor'] = df['final_factor'] + df['smoothed_normalized_returns']
    
    return df['enhanced_combined_factor']
```
