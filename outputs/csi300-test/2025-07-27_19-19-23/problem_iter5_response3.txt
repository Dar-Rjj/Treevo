import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=10, spike_threshold=2.0, weight=0.5, ema_span=5):
    # Calculate High-Low Difference
    df['High-Low'] = df['high'] - df['low']
    
    # Calculate Average High-Low Difference
    df['Avg_High-Low'] = df['High-Low'].rolling(window=n).mean()
    
    # Calculate Daily Log Return
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume Spikes using Z-Score
    df['Volume_ZScore'] = (df['volume'] - df['volume'].rolling(window=n).mean()) / df['volume'].rolling(window=n).std()
    df['Spike_Indicator'] = (df['Volume_ZScore'] > spike_threshold).astype(int)
    
    # Calculate Volume Rate of Change (VROC)
    df['VROC'] = df['volume'].pct_change(periods=m)
    
    # Combine High-Low Difference and Volume Spikes
    df['High-Low_Spike_Adjusted'] = df['High-Low'] * df['Spike_Indicator']
    df['Sum_High-Low_Spike'] = df['High-Low_Spike_Adjusted'].rolling(window=10).sum()
    
    # Adjust Momentum for Volume Spikes
    df['Adj_Log_Return'] = df['Log_Return'] * (df['Spike_Indicator'] * weight + (1 - df['Spike_Indicator']))
    df['Smoothed_Adj_Return'] = df['Adj_Log_Return'].ewm(span=ema_span, min_periods=ema_span).mean()
    
    # Multiply Price Momentum by VROC
    df['Combined_Momentum'] = df['Avg_High-Low'] * df['VROC']
    
    # Adjust Combined Momentum for Volume Spikes
    df['Adj_Combined_Momentum'] = df['Combined_Momentum'] * (df['Spike_Indicator'] * weight + (1 - df['Spike_Indicator']))
    df['Smoothed_Adj_Combined_Momentum'] = df['Adj_Combined_Momentum'].ewm(span=ema_span, min_periods=ema_span).mean()
    
    # Cumulative Adjusted Momentum Factor
    df['Cumulative_Factor'] = (1 + df['Smoothed_Adj_Return']).cumprod() * (1 + df['Smoothed_Adj_Combined_Momentum']).cumprod()
    
    # Normalize to starting value
    df['Cumulative_Factor'] /= df['Cumulative_Factor'].iloc[0]
    
    return df['Cumulative_Factor']
