import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_days=10, spike_threshold=2.0, smoothing_span=5, vroc_period=10, weight=0.5):
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Identify Volume Spikes
    df['volume_zscore'] = (df['volume'] - df['volume'].rolling(window=lookback_days).mean()) / df['volume'].rolling(window=lookback_days).std()
    df['volume_spike'] = df['volume_zscore'] > spike_threshold
    
    # Combine High-Low Difference and Volume Spikes
    df['high_low_diff_spike'] = df['high_low_diff'] * df['volume_spike']
    df['sum_high_low_diff_spike'] = df['high_low_diff_spike'].rolling(window=lookback_days).sum()
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume Spikes
    df['adjusted_log_return'] = df['log_return'] * (1 + (df['volume_spike'] * (weight - 1)))
    
    # Incorporate High-Low Range into Momentum
    df['high_low_range'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['momentum_adjusted'] = df['adjusted_log_return'] * df['high_low_range']
    
    # Smooth the Adjusted Returns
    df['smoothed_momentum'] = df['momentum_adjusted'].ewm(span=smoothing_span, adjust=False).mean()
    
    # Calculate Average High-Low Difference
    df['avg_high_low_diff'] = df['high_low_diff'].rolling(window=lookback_days).mean()
    
    # Calculate Volume Rate of Change (VROC)
    df['vroc'] = (df['volume'] / df['volume'].shift(vroc_period)) - 1
    
    # Multiply Price Momentum by VROC
    df['combined_momentum'] = df['smoothed_momentum'] * df['vroc']
    
    # Adjust Combined Momentum for Volume Spikes
    df['adjusted_combined_momentum'] = df['combined_momentu m'] * (1 + (df['volume_spike'] * (weight - 1)))
    df['smoothed_adjusted_combined_momentum'] = df['adjusted_combined_momentum'].ewm(span=smoothing_span, adjust=False).mean()
    
    # Cumulative Adjusted Momentum Factor
    df['cumulative_momentum'] = (df['smoothed_adjusted_combined_momentum'] + 1).cumprod()
    
    return df['cumulative_momentum']
