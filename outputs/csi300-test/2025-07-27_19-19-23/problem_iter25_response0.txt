```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement
    daily_price_movement = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    volume_adjusted_movement = daily_price_movement * df['volume']
    
    # Calculate Rolling Momentum Indicator
    rolling_momentum = volume_adjusted_movement.rolling(window=30).sum()
    ewma_5 = rolling_momentum.ewm(span=5).mean()
    ewma_10 = rolling_momentum.ewm(span=10).mean()
    momentum_indicator = ewma_5 - ewma_10
    
    # Calculate High-Low Difference
    high_low_difference = df['high'] - df['low']
    
    # Identify Volume Spikes
    z_score_volume = (df['volume'] - df['volume'].rolling(window=30).mean()) / df['volume'].rolling(window=30).std()
    spike_indicator = (z_score_volume > 2.0).astype(int)  # 2.0 is a common threshold for Z-Score
    
    # Combine High-Low Difference and Volume Spikes
    high_low_spike = (high_low_difference * spike_indicator).rolling(window=10).sum()
    
    # Adjust Momentum for Volume Spikes
    log_returns = np.log(df['close'] / df['close'].shift(1))
    adjusted_log_returns = np.where(spike_indicator == 1, log_returns * 0.8, log_returns)
    
    # Smooth the Adjusted Returns
    smoothed_returns = adjusted_log_returns.ewm(span=10).mean()
    
    # Volume Trend (VROC for period m)
    vroc = df['volume'].pct_change(periods=10)
    
    # Combined Factor
    combined_factor = (momentum_indicator + high_low_spike + smoothed_returns) * vroc
    
    return pd.Series(combined_factor, index=df.index)
```
