import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    window = 30
    df['volume_z_score'] = (df['volume'] - df['volume'].rolling(window=window).mean()) / df['volume'].rolling(window=window).std()
    df['price_z_score'] = (df[['high', 'low']].mean(axis=1) - df[['high', 'low']].mean(axis=1).rolling(window=window).mean()) / df[['high', 'low']].mean(axis=1).rolling(window=window).std()
    spike_threshold = 2.0
    df['spike_indicator'] = ((df['volume_z_score'].abs() > spike_threshold) | (df['price_z_score'].abs() > spike_threshold)).astype(int)
    
    # Adjust Momentum for Volume and Price Spikes
    spike_weight = 0.5
    df['adjusted_log_return'] = df['log_return'] * (1 - df['spike_indicator'] * (1 - spike_weight))
    
    # Smooth the Adjusted Returns
    ema_span = 10
    df['smoothed_log_return'] = df['adjusted_log_return'].ewm(span=ema_span, adjust=False).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (1 + df['smoothed_log_return']).cumprod()
    
    # Calculate Daily Price Movement
    df['price_movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['price_movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    rolling_window = 30
    df['aggregated_volume_adjusted_price_movement'] = df['volume_adjusted_price_movement'].rolling(window=rolling_window).sum()
    df['rolling_momentum'] = df['aggregated_volume_adjusted_price_movement'].ewm(span=5, adjust=False).mean() - df['aggregated_volume_adjusted_price_movement'].ewm(span=10, adjust=False).mean()
    
    # Calculate High-Low Difference
    df['high_low_difference'] = df['high'] - df['low']
    
    # Combine High-Low Difference with Volume and Price Spikes
    lookback_period = 10
    df['combined_high_low'] = (df['high_low_difference'] * df['spike_indicator']).rolling(window=lookback_period).sum()
    
    # Combine High-Low and Price Momentum
    df['combined_momentum'] = df['cumulative_momentum'] + df['combined_high_low']
    
    # Volume Trend
    vroc_period = 10
    df['vroc'] = df['volume'].pct_change(periods=vroc_period)
    
    # Combined Factor
    df['combined_factor'] = df['combined_momentum'] * df['vroc']
    
    # Additional Factor: Volatility-Adjusted Momentum
    volatility_lookback = 20
    df['daily_volatility'] = df['close'].rolling(window=volatility_lookback).std()
    df['normalized_log_return'] = df['log_return'] / df['daily_volatility']
    df['smoothed_normalized_log_return'] = df['normalized_log_return'].ewm(span=ema_span, adjust=False).mean()
    
    # Enhanced Combined Factor
    df['enhanced_combined_factor'] = df['combined_factor'] + df['smoothed_normalized_log_return']
    
    return df['enhanced_combined_factor']
