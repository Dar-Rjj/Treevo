import pandas as pd
import numpy as np

def heuristics_v2(df, spike_threshold=2.0, momentum_span=30, vroc_period=10):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']).diff()
    
    # Identify Volume and Price Spikes
    df['vol_z_score'] = (df['volume'] - df['volume'].rolling(window=30).mean()) / df['volume'].rolling(window=30).std()
    df['price_z_score'] = (df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)) / df[['high', 'low']].min(axis=1).rolling(window=30).std()
    df['spike_indicator'] = ((df['vol_z_score'] > spike_threshold) | (df['price_z_score'] > spike_threshold)).astype(int)
    
    # Adjust Momentum for Volume and Price Spikes
    df['adjusted_log_return'] = df['log_return'] * (1 + df['spike_indicator'] * 0.5)
    
    # Smooth the Adjusted Returns
    df['smoothed_adjusted_return'] = df['adjusted_log_return'].ewm(span=momentum_span).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (1 + df['smoothed_adjusted_return']).cumprod()
    
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['daily_price_movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['aggregated_volume_adjusted_movement'] = df['volume_adjusted_price_movement'].rolling(window=30).sum()
    df['smoothed_aggregated_movement'] = df['aggregated_volume_adjusted_movement'].ewm(span=30).mean()
    df['rolling_momentum'] = df['smoothed_aggregated_movement'].ewm(span=5).mean() - df['smoothed_aggregated_movement'].ewm(span=10).mean()
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Combine High-Low Difference with Volume and Price Spikes
    df['combined_high_low_spikes'] = df['high_low_diff'] * df['spike_indicator']
    df['summed_combined_high_low_spikes'] = df['combined_high_low_spikes'].rolling(window=10).sum()
    
    # Price Momentum
    df['weighted_high_low_diff'] = df['high_low_diff'].rolling(window=10).apply(lambda x: np.sum(x * np.arange(1, len(x) + 1) / len(x)), raw=True)
    
    # Volume Trend
    df['vroc'] = df['volume'].pct_change(periods=vroc_period)
    
    # Combined Factor
    df['combined_factor'] = df['cumulative_momentum'] * df['vroc']
    
    return df['combined_factor']
