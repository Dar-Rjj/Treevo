```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    window = 20
    df['vol_zscore'] = (df['volume'] - df['volume'].rolling(window=window).mean()) / df['volume'].rolling(window=window).std()
    df['price_zscore'] = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2).rolling(window=window).std()
    
    spike_threshold = 2.0
    df['spike'] = (df['vol_zscore'] > spike_threshold) | (df['price_zscore'] > spike_threshold)
    
    # Adjust Momentum for Volume and Price Spikes
    weight = 1.5
    df['adjusted_log_return'] = np.where(df['spike'], df['log_return'] * weight, df['log_return'])
    
    # Smooth the Adjusted Returns
    ema_span = 10
    df['smoothed_returns'] = df['adjusted_log_return'].ewm(span=ema_span, min_periods=ema_span).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (df['smoothed_returns'] + 1).cumprod().fillna(1)
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate Rolling Momentum Indicator
    rolling_window = 30
    df['rolling_volume_adjusted_price'] = df['volume_adjusted_price_movement'].rolling(window=rolling_window).sum()
    df['rolling_avg_30'] = df['rolling_volume_adjusted_price'].ewm(span=rolling_window, min_periods=rolling_window).mean()
    df['momentum_indicator'] = df['rolling_volume_adjusted_price'].ewm(span=5, min_periods=5).mean() - df['rolling_volume_adjusted_price'].ewm(span=15, min_periods=15).mean()
    
    # Combine High-Low Difference with Volume and Price Spikes
    lookback_period = 10
    df['high_low_diff'] = (df['high'] - df['low']) * df['spike']
    df['sum_high_low_diff'] = df['high_low_diff'].rolling(window=lookback_period).sum()
    
    # Combined Factor
    df['combined_factor'] = df['cumulative_momentum'] + df['sum_high_low_diff']
    
    # Additional Factor: Volatility-Adjusted Momentum
    volatility_lookback = 20
    df['daily_volatility'] = df['close'].pct_change().rolling(window=volatility_lookback).std()
    df['normalized_return'] = df['log_return'] / df['daily_volatility']
    df['smoothed_normalized_return'] = df['normalized_return'].ewm(span=ema_span, min_periods=ema_span).mean()
    
    # Enhanced Combined Factor
    df['enhanced_combined_factor'] = df['combined_factor'] + df['smoothed_normalized_return']
    
    return df['enhanced_combined_factor']
```
