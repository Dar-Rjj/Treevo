```python
import pandas as pd
import numpy as np

def heuristics_v2(df, threshold=2.0, weight=0.5, ema_span=10):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume Spikes
    rolling_std = df['volume'].rolling(window=30).std()
    df['z_score'] = (df['volume'] - df['volume'].rolling(window=30).mean()) / rolling_std
    df['spike'] = df['z_score'] > threshold
    
    # Adjust Momentum for Volume Spikes
    df['adjusted_log_return'] = np.where(df['spike'], df['log_return'] * weight, df['log_return'])
    
    # Smooth the Adjusted Returns
    df['smoothed_returns'] = df['adjusted_log_return'].ewm(span=ema_span, min_periods=ema_span).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (df['smoothed_returns'] + 1).cumprod().fillna(1)
    
    return df['cumulative_momentum']
```
