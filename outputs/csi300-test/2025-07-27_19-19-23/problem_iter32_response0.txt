```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    df['vol_20ma'] = df['volume'].rolling(window=20).mean()
    df['relative_vol'] = df['volume'] / df['vol_20ma']
    df['z_score_vol'] = (df['volume'] - df['volume'].rolling(window=20).mean()) / df['volume'].rolling(window=20).std()
    df['z_score_price'] = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2).rolling(window=20).std()
    df['spike'] = (df['relative_vol'] > 1.5) | (df['z_score_vol'] > 2) | (df['z_score_price'] > 2)
    
    # Adjust Momentum for Volume and Price Spikes
    spike_weight = 1.5
    df['adjusted_log_return'] = df['log_return'] * (1 + (df['spike'] * (spike_weight - 1)))
    
    # Smooth the Adjusted Returns
    span = 5
    df['smoothed_adj_returns'] = df['adjusted_log_return'].ewm(span=span, adjust=False).mean()
    
    # Cumulative Momentum Factor
    df['cum_momentum'] = (df['smoothed_adj_returns'] + 1).cumprod()
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Combine High-Low Difference with Volume and Price Spikes
    lookback_period = 10
    df['spiked_high_low_diff'] = df['high_low_diff'] * df['spike']
    df['sum_spiked_high_low_diff'] = df['spiked_high_low_diff'].rolling(window=lookback_period).sum()
    
    # Calculate Daily Price Movement
    df['price_movement'] = df['close'] - df['open']
    
    # Calculate Volume-Adjusted Price Movement
    cap = 3
    df['vol_adjusted_price_movement'] = df['price_movement'] * df['volume'].clip(upper=df['volume'].quantile(0.99))
    
    # Calculate Rolling Momentum Indicator
    rolling_window = 30
    df['aggregated_vol_adj_price_movement'] = df['vol_adjusted_price_movement'].rolling(window=rolling_window).sum()
    df['ema_5'] = df['aggregated_vol_adj_price_movement'].ewm(span=5, adjust=False).mean()
    df['ema_15'] = df['aggregated_vol_adj_price_movement'].ewm(span=15, adjust=False).mean()
    df['momentum_indicator'] = df['ema_5'] - df['ema_15']
    
    # Combined Momentum and High-Low Difference
    df['combined_momentum'] = df['cum_momentum'] + df['sum_spiked_high_low_diff']
    
    # Volume Trend
    m = 10
    df['vroc'] = df['volume'].pct_change(periods=m)
    smoothing_factor = 5
    df['smoothed_vroc'] = df['vroc'].ewm(span=smoothing_factor, adjust=False).mean()
    
    # Final Combined Alpha Factor
    df['final_combined_factor'] = df['combined_momentum'] * df['smoothed_vroc']
    
    # Additional Factor: Volatility-Adjusted Momentum
    volatility_lookback = 20
    df['daily_volatility'] = df['close'].rolling(window=volatility_lookback).std()
    df['normalized_log_return'] = df['log_return'] / df['daily_volatility']
    df['smoothed_normalized_returns'] = df['normalized_log_return'].ewm(span=span, adjust=False).mean()
    
    # Enhanced Combined Factor
    df['enhanced_combined_factor'] = df['final_combined_factor'] + df['smoothed_normalized_returns']
    
    return df['enhanced_combined_factor']
```
