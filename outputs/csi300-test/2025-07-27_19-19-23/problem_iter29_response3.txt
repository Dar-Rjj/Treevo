import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Calculate the rolling average of the percentage change
    df['rolling_avg_pct_change_close'] = df['pct_change_close'].rolling(window=10).mean()
    
    # Calculate the relative volume, which is a ratio of current volume to 10-day average volume
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['relative_volume'] = df['volume'] / df['avg_volume_10d']
    
    # Calculate the money flow index (MFI) for 14 days
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_mf = positive_flow.rolling(window=14).sum()
    negative_mf = negative_flow.rolling(window=14).sum()
    money_flow_ratio = positive_mf / negative_mf
    df['MFI'] = 100 - (100 / (1 + money_flow_ratio))
    
    # Combine all factors into one by normalizing and summing them
    normalized_rolling_avg = (df['rolling_avg_pct_change_close'] - df['rolling_avg_pct_change_close'].mean()) / df['rolling_avg_pct_change_close'].std()
    normalized_relative_vol = (df['relative_volume'] - df['relative_volume'].mean()) / df['relative_volume'].std()
    normalized_mfi = (df['MFI'] - df['MFI'].mean()) / df['MFI'].std()
    factor = normalized_rolling_avg + normalized_relative_vol + normalized_mfi
    
    return factor.dropna()
