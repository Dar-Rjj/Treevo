```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=10, vol_lookback=10, vroc_period=10, ema_span=5, spike_threshold=2.0):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Identify Volume Spikes
    rolling_vol_std = df['volume'].rolling(window=lookback_period).std()
    z_scores = (df['volume'] - df['volume'].rolling(window=lookback_period).mean()) / rolling_vol_std
    spike_indicator = (z_scores > spike_threshold).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    combined_high_low = (high_low_diff * spike_indicator).rolling(window=lookback_period).sum()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume Spikes
    adjusted_momentum = daily_log_return * (spike_indicator * 0.5 + 0.5)  # Weighted by 0.5 if spike
    
    # Smooth the Adjusted Returns
    smoothed_adjusted_returns = adjusted_momentum.ewm(span=ema_span).mean()
    
    # Price Momentum: Weighted Average High-Low Difference
    weighted_high_low_diff = (high_low_diff * np.arange(1, lookback_period + 1)).rolling(window=lookback_period).sum() / lookback_period
    
    # Volume Trend: Volume Rate of Change (VROC)
    vroc = df['volume'].pct_change(periods=vroc_period)
    
    # Combined Factor: High-Low and Price Momentum with Volume Trend
    combined_factor = (combined_high_low + weighted_high_low_diff) * vroc
    
    # Cumulative Momentum Factor
    cumulative_momentum = (adjusted_momentum + 1).cumprod()
    
    # Additional Factor: Volatility-Adjusted Momentum
    daily_volatility = df['close'].rolling(window=vol_lookback).std()
    normalized_returns = daily_log_return / daily_volatility
    smoothed_normalized_returns = normalized_returns.ewm(span=ema_span).mean()
    
    # Enhanced Combined Factor
    enhanced_combined_factor = combined_factor + smoothed_normalized_returns
    
    # Final Factor
    final_factor = enhanced_combined_factor * cumulative_momentum + smoothed_adjusted_returns
    
    return final_factor
```
