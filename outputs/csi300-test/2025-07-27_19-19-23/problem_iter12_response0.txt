```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    # Detect Volume Spikes
    rolling_vol_mean = df['volume'].rolling(window=20).mean()
    rolling_vol_std = df['volume'].rolling(window=20).std()
    df['vol_zscore'] = (df['volume'] - rolling_vol_mean) / rolling_vol_std
    volume_spike_threshold = 2.0  # Example threshold, can be adjusted
    df['is_volume_spike'] = df['vol_zscore'].abs() > volume_spike_threshold
    
    # Detect Price Spikes
    df['range'] = df['high'] - df['low']
    range_ma = df['range'].rolling(window=20).mean()
    price_spike_threshold_multiplier = 2.0  # Example multiplier, can be adjusted
    df['is_price_spike'] = df['range'] > price_spike_threshold_multiplier * range_ma
    
    # Classify as Spike if either Volume or Price Spike is detected
    df['is_spike'] = df['is_volume_spike'] | df['is_price_spike']
    
    # Adjust Momentum for Volume and Price Spikes
    spike_weight = 0.5  # Example weight, can be adjusted
    df['adjusted_log_return'] = np.where(df['is_spike'], df['log_return'] * spike_weight, df['log_return'])
    
    # Smooth the Adjusted Returns using Exponential Moving Average
    ema_span = 20  # Example span, can be adjusted
    df['smoothed_adjusted_log_return'] = df['adjusted_log_return'].ewm(span=ema_span, min_periods=ema_span).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (df['smoothed_adjusted_log_return'] + 1).cumprod().fillna(1)
    
    return df['cumulative_momentum']
```
