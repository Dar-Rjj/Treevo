```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement
    daily_price_movement = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    volume_adjusted_price_movement = daily_price_movement * df['volume']
    
    # Calculate Rolling Momentum Indicator
    rolling_30_sum = volume_adjusted_price_movement.rolling(window=30).sum()
    ewma_5 = rolling_30_sum.ewm(span=5).mean()
    ewma_10 = rolling_30_sum.ewm(span=10).mean()
    momentum = ewma_5 - ewma_10
    
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Identify Volume Spikes
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / avg_volume_20
    spike_threshold = 1.20  # 20% increase
    spike_indicator = (relative_volume > spike_threshold).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    combined_high_low_spike = (high_low_diff * spike_indicator).rolling(window=10).sum()
    
    # Adjust Momentum for Volume Spikes
    log_returns = np.log(df['close'] / df['close'].shift(1))
    adjusted_log_returns = log_returns * (1 + spike_indicator * 0.5)  # Weight of 0.5 for spikes
    
    # Smooth the Adjusted Returns
    smoothed_returns = adjusted_log_returns.ewm(span=10).mean()
    
    # Volume Trend
    VROC_period = 10
    VROC = df['volume'].pct_change(periods=VROC_period)
    
    # Combined Factor
    combined_factor = (combined_high_low_spike + smoothed_returns) * VROC
    
    # Cumulative Momentum Factor
    cumulative_momentum_factor = (1 + combined_factor).cumprod()
    
    return cumulative_momentum_factor
```
