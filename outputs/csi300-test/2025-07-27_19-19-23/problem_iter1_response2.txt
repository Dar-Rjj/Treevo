```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']).diff()
    
    # Identify Volume Spikes
    df['volume_zscore'] = (df['volume'] - df['volume'].rolling(window=20).mean()) / df['volume'].rolling(window=20).std()
    spike_threshold = 2.0
    df['spike_indicator'] = (df['volume_zscore'] > spike_threshold).astype(int)
    
    # Adjust Momentum for Volume Spikes
    weight = 0.5  # Weight to apply if there is a volume spike
    df['adjusted_return'] = df['log_return'] * (1 - df['spike_indicator'] * (1 - weight))
    
    # Smooth the Adjusted Returns using Exponential Moving Average
    df['smoothed_return'] = df['adjusted_return'].ewm(span=10, min_periods=5).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (df['smoothed_return'] + 1).cumprod() * 1.0
    
    return df['cumulative_momentum']
```
