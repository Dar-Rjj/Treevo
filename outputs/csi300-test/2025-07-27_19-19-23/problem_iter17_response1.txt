import pandas as pd
import numpy as np

def heuristics_v2(df, threshold=3, spike_weight=0.5, ema_span=20, lookback_period=10, vroc_period=10, volatility_lookback=20):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    df['volume_zscore'] = (df['volume'] - df['volume'].rolling(window=lookback_period).mean()) / df['volume'].rolling(window=lookback_period).std()
    df['price_zscore'] = (df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)).rolling(window=lookback_period).std()
    df['spike'] = ((df['volume_zscore'] > threshold) | (df['price_zscore'] > threshold)).astype(int)
    
    # Adjust Momentum for Volume and Price Spikes
    df['adjusted_log_return'] = df['log_return'] * (1 + (spike_weight - 1) * df['spike'])
    
    # Smooth the Adjusted Returns
    df['smoothed_adjusted_returns'] = df['adjusted_log_return'].ewm(span=ema_span, min_periods=ema_span).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (df['smoothed_adjusted_returns'] + 1).cumprod()
    
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['daily_price_movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['aggregated_volume_adjusted'] = df['volume_adjusted_price_movement'].rolling(window=30).sum()
    df['rolling_ema_5'] = df['aggregated_volume_adjusted'].ewm(span=5, min_periods=5).mean()
    df['rolling_ema_10'] = df['aggregated_volume_adjusted'].ewm(span=10, min_periods=10).mean()
    df['rolling_momentum'] = df['rolling_ema_5'] - df['rolling_ema_10']
    
    # Calculate High-Low Difference
    df['high_low_difference'] = df['high'] - df['low']
    
    # Combine High-Low Difference with Volume and Price Spikes
    df['spike_adjusted_high_low'] = df['high_low_difference'] * df['spike']
    df['summed_high_low'] = df['spike_adjusted_high_low'].rolling(window=lookback_period).sum()
    
    # Combine High-Low and Price Momentum
    df['combined_momentum'] = df['summed_high_low'] + df['smoothed_adjusted_returns']
    
    # Volume Trend
    df['vroc'] = df['volume'].pct_change(periods=vroc_period)
    
    # Combined Factor
    df['combined_factor'] = df['combined_momentum'] * df['vroc']
    
    # Additional Factor: Volatility-Adjusted Momentum
    df['daily_volatility'] = df['close'].rolling(window=volatility_lookback).std()
    df['normalized_log_return'] = df['log_return'] / df['daily_volatility']
    df['smoothed_normalized_returns'] = df['normalized_log_return'].ewm(span=ema_span, min_periods=ema_span).mean()
    
    # Enhanced Combined Factor
    df['enhanced_combined_factor'] = df['combined_factor'] + df['smoothed_normalized_returns']
    
    return df['enhanced_combined_factor']
