```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Identify Volume Spikes
    z_score = (df['volume'] - df['volume'].rolling(window=20).mean()) / df['volume'].rolling(window=20).std()
    spike_indicator = (z_score > 3).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    high_low_spiked = high_low_diff * spike_indicator
    high_low_sum = high_low_spiked.rolling(window=10).sum()
    
    # Calculate Daily Log Return
    log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume Spikes
    weighted_log_return = log_return * (1 + 0.5 * spike_indicator)  # Increase weight by 50% if spike
    
    # Smooth the Adjusted Returns
    smoothed_returns = weighted_log_return.ewm(span=10, adjust=False).mean()
    
    # Price Momentum
    price_momentum = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Volume Trend
    vroc = (df['volume'] / df['volume'].shift(14) - 1) * 100  # VROC for 14 days
    
    # Combined Factor
    combined_factor = (high_low_sum + smoothed_returns) * vroc
    
    # Cumulative Momentum Factor
    cumulative_momentum = (1 + combined_factor).cumprod()
    
    return cumulative_momentum
```
