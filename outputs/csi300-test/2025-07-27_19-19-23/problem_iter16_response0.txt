```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement
    df['Daily_Price_Movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['Volume_Adjusted_Price_Movement'] = df['Daily_Price_Movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['30Day_Aggregated_Volume_Adjusted_Movement'] = df['Volume_Adjusted_Price_Movement'].rolling(window=30).sum()
    df['30Day_ExpWeighted_Avg'] = df['30Day_Aggregated_Volume_Adjusted_Movement'].ewm(span=30, adjust=False).mean()
    df['10Day_ExpWeighted_Avg'] = df['30Day_Aggregated_Volume_Adjusted_Movement'].ewm(span=10, adjust=False).mean()
    df['5Day_ExpWeighted_Avg'] = df['30Day_Aggregated_Volume_Adjusted_Movement'].ewm(span=5, adjust=False).mean()
    df['Momentum'] = df['5Day_ExpWeighted_Avg'] - df['10Day_ExpWeighted_Avg']
    
    # Calculate High-Low Difference
    df['High_Low_Diff'] = df['high'] - df['low']
    
    # Identify Volume Spikes
    df['Volume_Z_Score'] = (df['volume'] - df['volume'].rolling(window=30).mean()) / df['volume'].rolling(window=30).std()
    df['Spike_Indicator'] = df['Volume_Z_Score'].apply(lambda x: 1 if x > 2 else 0)
    
    # Combine High-Low Difference and Volume Spikes
    df['HL_Spike_Product'] = df['High_Low_Diff'] * df['Spike_Indicator']
    df['10Day_HL_Spike_Sum'] = df['HL_Spike_Product'].rolling(window=10).sum()
    
    # Adjust Momentum for Volume Spikes
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Adjusted_Log_Return'] = df['Log_Return'] * (1 + 0.5 * df['Spike_Indicator'])
    
    # Smooth the Adjusted Returns
    df['Smoothed_Adjusted_Returns'] = df['Adjusted_Log_Return'].ewm(span=10, adjust=False).mean()
    
    # Volume Trend
    df['VROC'] = df['volume'] / df['volume'].shift(5) - 1
    
    # Combined Factor
    df['Combined_Factor'] = df['Momentum'] + df['10Day_HL_Spike_Sum'] + df['Smoothed_Adjusted_Returns']
    df['Combined_Factor'] = df['Combined_Factor'] * df['VROC']
    
    # Cumulative Momentum Factor
    df['Cumulative_Momentum_Factor'] = df['Combined_Factor'].add(1).cumprod().fillna(1)
    
    return df['Cumulative_Momentum_Factor']
```
