```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement
    df['price_movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['price_movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['rolling_sum_price_movement'] = df['volume_adjusted_price_movement'].rolling(window=20).sum()
    df['rolling_momentum_avg'] = df['rolling_sum_price_movement'].rolling(window=20).mean()
    df['momentum'] = df['rolling_momentum_avg'] - df['rolling_momentum_avg'].shift(1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted Volatility
    df['volume_adjusted_volatility'] = df['daily_volatility'] * df['volume']
    
    # Calculate Rolling Volatility Indicator
    df['rolling_sum_volatility'] = df['volume_adjusted_volatility'].rolling(window=20).sum()
    df['rolling_volatility_avg'] = df['rolling_sum_volatility'].rolling(window=20).mean()
    df['volatility_momentum'] = df['rolling_volatility_avg'] - df['rolling_volatility_avg'].shift(1)
    
    # Combine Momentum and Volatility Indicators
    df['combined_factor'] = df['momentum'] + df['volatility_momentum']
    
    return df['combined_factor']
```
