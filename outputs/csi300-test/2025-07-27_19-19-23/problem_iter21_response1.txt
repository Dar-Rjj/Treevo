```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['daily_price_movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['volume_adjusted_rolling_sum'] = df['volume_adjusted_price_movement'].rolling(window=30).sum()
    df['ewma_5'] = df['volume_adjusted_rolling_sum'].ewm(span=5).mean()
    df['ewma_10'] = df['volume_adjusted_rolling_sum'].ewm(span=10).mean()
    df['momentum'] = df['ewma_5'] - df['ewma_10']
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Identify Volume Spikes
    df['volume_z_score'] = (df['volume'] - df['volume'].rolling(window=30).mean()) / df['volume'].rolling(window=30).std()
    spike_threshold = 2.0  # Z-Score threshold for volume spikes
    df['spike_indicator'] = np.where(df['volume_z_score'] > spike_threshold, 1, 0)
    
    # Combine High-Low Difference and Volume Spikes
    df['hl_diff_spikes'] = df['high_low_diff'] * df['spike_indicator']
    df['hl_diff_spikes_sum'] = df['hl_diff_spikes'].rolling(window=10).sum()
    
    # Adjust Momentum for Volume Spikes
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))
    weight = 1.5  # Weight for volume spikes
    df['adjusted_daily_log_return'] = np.where(df['spike_indicator'] == 1, df['daily_log_return'] * weight, df['daily_log_return'])
    
    # Smooth the Adjusted Returns
    df['smoothed_adjusted_returns'] = df['adjusted_daily_log_return'].ewm(span=10).mean()
    
    # Volume Trend (VROC)
    m = 14  # VROC period
    df['vroc'] = df['volume'].pct_change(periods=m)
    
    # Combined Factor
    df['combined_factor'] = df['hl_diff_spikes_sum'] + df['smoothed_adjusted_returns']
    
    # Integrate with Volume Trend
    df['integrated_factor'] = df['combined_factor'] * df['vroc']
    
    # Cumulative Momentum Factor
    df['cumulative_momentum_factor'] = (df['integrated_factor'] + 1).cumprod()
    
    return df['cumulative_momentum_factor']
```
