import pandas as pd
import numpy as np

def heuristics_v2(df, lookback=10, spike_threshold=2.0, weight=0.5, span=7):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Identify Volume Spikes
    z_score = (df['volume'] - df['volume'].rolling(window=lookback).mean()) / df['volume'].rolling(window=lookback).std()
    volume_spike = (z_score > spike_threshold).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    combined_high_low = (high_low_diff * volume_spike).rolling(window=lookback).sum()
    
    # Calculate Daily Log Return
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume Spikes
    adjusted_log_returns = log_returns * (1 + (volume_spike * (weight - 1)))
    
    # Incorporate High-Low Range into Momentum
    high_low_range = (df['high'] - df['low']) / df['close'].shift(1)
    momentum_factor = adjusted_log_returns * high_low_range
    
    # Smooth the Adjusted Returns
    smoothed_momentum = momentum_factor.ewm(span=span).mean()
    
    # Adjust Combined Momentum for Volume Spikes
    adjusted_combined_momentum = smoothed_momentum * (1 + (volume_spike * (weight - 1)))
    smoothed_adjusted_combined_momentum = adjusted_combined_momentum.ewm(span=span).mean()
    
    # Cumulative Adjusted Momentum Factor
    cumulative_adjusted_momentum = (combined_high_low + smoothed_adjusted_combined_momentum).cumsum()
    
    return cumulative_adjusted_momentum
