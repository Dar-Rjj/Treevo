```python
import pandas as pd
import numpy as np

def heuristics_v2(df, spike_threshold=2.0, e_span=20):
    """
    Calculate a novel alpha factor that adjusts price momentum by volume spikes and high-low range.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    spike_threshold (float): Z-Score threshold to classify volume spikes
    e_span (int): Span for the exponential moving average
    
    Returns:
    pd.Series: Alpha factor values
    """
    
    # 1. Calculate Daily Log Return
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # 2. Identify Volume Spikes
    volume_zscore = (df['volume'] - df['volume'].rolling(window=e_span).mean()) / df['volume'].rolling(window=e_span).std()
    volume_spike = (volume_zscore > spike_threshold).astype(int)
    
    # 3. Adjust Momentum for Volume Spikes
    weighted_log_returns = log_returns * (1 - 0.5 * volume_spike)  # Assume weight = 0.5 for simplicity
    
    # 4. Incorporate High-Low Range into Momentum
    high_low_range = (df['high'] - df['low']) / df['close'].shift(1)
    adjusted_log_returns = weighted_log_returns * high_low_range
    
    # 5. Smooth the Adjusted Returns
    smoothed_returns = adjusted_log_returns.ewm(span=e_span, adjust=False).mean()
    
    # 6. Cumulative Momentum Factor
    cumulative_factor = (1 + smoothed_returns).cumprod()
    
    return cumulative_factor
```
