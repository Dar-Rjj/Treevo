```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    df['vol_20d_avg'] = df['volume'].rolling(window=20).mean()
    df['rel_vol'] = df['volume'] / df['vol_20d_avg']
    df['z_score_vol'] = (df['volume'] - df['volume'].rolling(window=20).mean()) / df['volume'].rolling(window=20).std()
    df['z_score_price'] = (df['high'] - df['low']) / (df['high'] - df['low']).rolling(window=20).std()
    spike_threshold = 2.0
    df['spike_indicator'] = (df['rel_vol'] > spike_threshold) | (abs(df['z_score_vol']) > spike_threshold) | (abs(df['z_score_price']) > spike_threshold)
    
    # Adjust Momentum for Volume and Price Spikes
    weight = 1.5
    df['adjusted_log_return'] = df['log_return'] * (1 + (df['spike_indicator'] * (weight - 1)))
    
    # Smooth the Adjusted Returns
    df['smoothed_returns'] = df['adjusted_log_return'].ewm(span=7).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (1 + df['smoothed_returns']).cumprod()
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Combine High-Low Difference with Volume and Price Spikes
    df['spiked_high_low_diff'] = df['high_low_diff'] * df['spike_indicator']
    df['summed_spiked_high_low_diff'] = df['spiked_high_low_diff'].rolling(window=10).sum()
    
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['close'] - df['open']
    
    # Calculate Volume-Adjusted Price Movement
    df['volume_adjusted_movement'] = df['daily_price_movement'] * df['volume']
    cap = df['volume_adjusted_movement'].rolling(window=20).std() * 3
    df['capped_volume_adjusted_movement'] = np.clip(df['volume_adjusted_movement'], -cap, cap)
    
    # Calculate Rolling Momentum Indicator
    df['aggregated_movement'] = df['capped_volume_adjusted_movement'].rolling(window=30).sum()
    df['5d_ewma'] = df['aggregated_movement'].ewm(span=5).mean()
    df['10d_ewma'] = df['aggregated_movement'].ewm(span=10).mean()
    df['momentum_indicator'] = df['5d_ewma'] - df['10d_ewma']
    
    # Combined Momentum and High-Low Difference
    df['combined_momentum'] = df['cumulative_momentum'] * df['summed_spiked_high_low_diff']
    
    # Volume Trend
    df['vroc'] = (df['volume'] - df['volume'].shift(14)) / df['volume'].shift(14)
    df['smoothed_vroc'] = df['vroc'].ewm(span=7).mean()
    
    # Final Combined Alpha Factor
    df['alpha_factor'] = df['combined_momentum'] * df['smoothed_vroc']
    
    # Additional Factor: Volatility-Adjusted Momentum
    df['daily_volatility'] = df['close'].rolling(window=20).std()
    df['normalized_log_return'] = df['log_return'] / df['daily_volatility']
    df['smoothed_normalized_returns'] = df['normalized_log_return'].ewm(span=7).mean()
    
    # New Factor: Momentum Reversal Indicator
    df['short_term_momentum'] = df['log_return'].ewm(span=5).mean()
    df['long_term_momentum'] = df['log_return'].ewm(span=60).mean()
    df['momentum_reversal'] = df['long_term_momentum'] - df['short_term_momentum']
    
    # Final Combined Alpha Factor
    df['final_alpha_factor'] = df['alpha_factor'] * df['smoothed_normalized_returns'] * df['momentum_reversal']
    
    return df['final_alpha_factor']
```
