```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement
    df['Daily_Price_Movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['Volume_Adjusted_Price_Movement'] = df['Daily_Price_Movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    ewm_5 = df['Volume_Adjusted_Price_Movement'].ewm(span=5).mean()
    ewm_10 = df['Volume_Adjusted_Price_Movement'].ewm(span=10).mean()
    df['Momentum_Indicator'] = ewm_5 - ewm_10
    
    # Calculate High-Low Difference
    df['High_Low_Diff'] = df['high'] - df['low']
    
    # Identify Volume Spikes
    df['Volume_Z_Score'] = (df['volume'] - df['volume'].rolling(window=30).mean()) / df['volume'].rolling(window=30).std()
    df['Volume_Spike'] = df['Volume_Z_Score'] > 2.0  # Z-Score threshold for spike
    
    # Combine High-Low Difference and Volume Spikes
    df['High_Low_Spike'] = df['High_Low_Diff'] * df['Volume_Spike']
    df['High_Low_Spike_Sum'] = df['High_Low_Spike'].rolling(window=10).sum()
    
    # Adjust Momentum for Volume Spikes
    df['Adjusted_Momentum'] = df['Momentum_Indicator'].where(~df['Volume_Spike'], df['Momentum_Indicator'] * 0.5)
    
    # Smooth the Adjusted Returns
    df['Smoothed_Adjusted_Momentum'] = df['Adjusted_Momentum'].ewm(span=10).mean()
    
    # Volume Trend (VROC)
    df['VROC'] = df['volume'].pct_change(periods=10)
    
    # Combined Factor
    df['Combined_Factor'] = (df['High_Low_Spike_Sum'] + df['Smoothed_Adjusted_Momentum']) * df['VROC']
    
    # Calculate Daily Log Return
    df['Log_Return'] = np.log(df['close']).diff()
    
    # Calculate Volume Adjusted Log Return
    df['Volume_Adjusted_Log_Return'] = df['Log_Return'] * df['volume']
    
    # Calculate Rolling Log Return Momentum
    ewm_log_5 = df['Volume_Adjusted_Log_Return'].ewm(span=5).mean()
    ewm_log_10 = df['Volume_Adjusted_Log_Return'].ewm(span=10).mean()
    df['Log_Return_Momentum'] = ewm_log_5 - ewm_log_10
    
    # Combine Log Return Momentum and Volume Spikes
    df['Adjusted_Log_Return_Momentum'] = df['Log_Return_Momentum'].where(~df['Volume_Spike'], df['Log_Return_Momentum'] * 0.5)
    
    # Smooth the Combined Log Return Momentum
    df['Smoothed_Log_Return_Momentum'] = df['Adjusted_Log_Return_Momentum'].ewm(span=10).mean()
    
    # Integrated Combined Factor
    df['Integrated_Combined_Factor'] = (df['High_Low_Spike_Sum'] + df['Smoothed_Adjusted_Momentum'] + df['Smoothed_Log_Return_Momentum']) * df['VROC']
    
    return df['Integrated_Combined_Factor']
```
