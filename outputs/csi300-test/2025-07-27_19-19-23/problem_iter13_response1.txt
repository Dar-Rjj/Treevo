```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['daily_price_movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['30_day_sum'] = df['volume_adjusted_price_movement'].rolling(window=30).sum()
    df['5_day_ewma'] = df['volume_adjusted_price_movement'].ewm(span=5, adjust=False).mean()
    df['10_day_ewma'] = df['volume_adjusted_price_movement'].ewm(span=10, adjust=False).mean()
    df['momentum'] = df['5_day_ewma'] - df['10_day_ewma']
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Identify Volume Spikes
    df['volume_pct_change'] = df['volume'].pct_change()
    spike_threshold = 0.20
    df['spike_indicator'] = (df['volume_pct_change'] > spike_threshold).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    df['high_low_with_spike'] = df['high_low_diff'] * df['spike_indicator']
    df['10_day_sum_high_low_with_spike'] = df['high_low_with_spike'].rolling(window=10).sum()
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume Spikes
    df['adjusted_log_return'] = df['log_return'] * df['spike_indicator'] + df['log_return'] * (1 - df['spike_indicator'])
    df['smoothed_adjusted_return'] = df['adjusted_log_return'].ewm(span=10, adjust=False).mean()
    
    # Combined Factor
    df['combined_factor'] = df['momentum'] + df['smoothed_adjusted_return'] + df['10_day_sum_high_low_with_spike']
    
    # Calculate Volume Rate of Change (VROC)
    df['vroc'] = df['volume'].pct_change(periods=20)  # Assuming m=20 for VROC
    
    # Final Combined Factor
    df['final_factor'] = df['combined_factor'] * df['vroc']
    
    return df['final_factor']
```
