```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['daily_price_movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['volume_adjusted_momentum_30d'] = df['volume_adjusted_price_movement'].rolling(window=30).sum()
    df['ewma_5d'] = df['volume_adjusted_momentum_30d'].ewm(span=5).mean()
    df['ewma_10d'] = df['volume_adjusted_momentum_30d'].ewm(span=10).mean()
    df['momentum'] = df['ewma_5d'] - df['ewma_10d']
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Identify Volume Spikes
    df['volume_pct_change'] = df['volume'].pct_change()
    spike_threshold = 0.20  # 20% increase
    df['spike_indicator'] = (df['volume_pct_change'] > spike_threshold).astype(int)
    df['volume_zscore'] = (df['volume'] - df['volume'].rolling(window=30).mean()) / df['volume'].rolling(window=30).std()
    df['spike_indicator'] = (df['volume_zscore'] > 2).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    df['hl_spike_product'] = df['high_low_diff'] * df['spike_indicator']
    df['hl_spike_sum_10d'] = df['hl_spike_product'].rolling(window=10).sum()
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume Spikes
    df['adjusted_log_return'] = df['log_return'] * (1 + 0.5 * df['spike_indicator'])
    df['smoothed_adjusted_return'] = df['adjusted_log_return'].ewm(span=10).mean()
    
    # Combined Factor
    df['combined_factor'] = df['hl_spike_sum_10d'] + df['smoothed_adjusted_return']
    
    # Volume Trend (VROC)
    m = 14  # VROC period
    df['vroc'] = (df['volume'] / df['volume'].shift(m) - 1) * 100
    
    # Final Combined Factor with Volume Trend
    df['final_factor'] = df['combined_factor'] * df['vroc']
    
    return df['final_factor']
```
