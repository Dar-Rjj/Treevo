```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=10, spike_threshold=3, ewm_span=5, vroc_period=30, vol_lookback=30):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Identify Volume and Price Spikes
    z_score_volume = (df['volume'] - df['volume'].rolling(window=lookback_period).mean()) / df['volume'].rolling(window=lookback_period).std()
    z_score_price = (df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)).rolling(window=lookback_period).mean() / df[['high', 'low']].max(axis=1).rolling(window=lookback_period).std()
    spike_indicator = (z_score_volume.abs() > spike_threshold) | (z_score_price.abs() > spike_threshold)
    
    # Combine High-Low Difference and Volume and Price Spikes
    combined_high_low = (high_low_diff * spike_indicator).rolling(window=lookback_period).sum()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume and Price Spikes
    adjusted_daily_log_return = daily_log_return.where(~spike_indicator, daily_log_return * 0.5)
    
    # Smooth the Adjusted Returns
    smoothed_adjusted_returns = adjusted_daily_log_return.ewm(span=ewm_span).mean()
    
    # Cumulative Momentum Factor
    cumulative_momentum = (smoothed_adjusted_returns + 1).cumprod().fillna(1)
    
    # Calculate Daily Price Movement
    daily_price_movement = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    volume_adjusted_price_movement = daily_price_movement * df['volume']
    
    # Calculate Rolling Momentum Indicator
    rolling_momentum_30d = volume_adjusted_price_movement.rolling(window=30).sum()
    ewma_5d = rolling_momentum_30d.ewm(span=5).mean()
    ewma_10d = rolling_momentum_30d.ewm(span=10).mean()
    momentum_indicator = ewma_5d - ewma_10d
    
    # Combine High-Low and Price Momentum
    combined_momentum = combined_high_low + cumulative_momentum + momentum_indicator
    
    # Volume Trend
    vroc = (df['volume'] - df['volume'].shift(vroc_period)) / df['volume'].shift(vroc_period)
    
    # Combined Factor
    combined_factor = combined_momentum * vroc
    
    # Additional Factor: Volatility-Adjusted Momentum
    daily_volatility = df['close'].rolling(window=vol_lookback).std()
    normalized_returns = daily_log_return / daily_volatility
    smoothed_normalized_returns = normalized_returns.ewm(span=ewm_span).mean()
    
    # Enhanced Combined Factor
    enhanced_combined_factor = combined_factor + smoothed_normalized_returns
    
    return enhanced_combined_factor
```
