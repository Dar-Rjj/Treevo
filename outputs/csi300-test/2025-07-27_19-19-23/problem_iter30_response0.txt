import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    df['vol_20_day_ma'] = df['volume'].rolling(window=20).mean()
    df['rel_vol'] = df['volume'] / df['vol_20_day_ma']
    
    df['price_z_score'] = (df['high'] - df['low']) / df['high'].rolling(window=20).std()
    
    spike_threshold = 2.0
    df['spike_indicator'] = ((df['rel_vol'] > spike_threshold) | (df['price_z_score'] > spike_threshold)).astype(int)
    
    # Adjust Momentum for Spikes
    df['adjusted_log_return'] = df['log_return'] * (1 + 0.5 * df['spike_indicator'])
    
    # Smooth the Adjusted Returns
    df['smoothed_returns'] = df['adjusted_log_return'].ewm(span=7).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (1 + df['smoothed_returns']).cumprod()
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Combine High-Low Difference with Spikes
    df['high_low_spike'] = df['high_low_diff'] * df['spike_indicator']
    df['sum_high_low_spike'] = df['high_low_spike'].rolling(window=10).sum()
    
    # Calculate Daily Price Movement
    df['daily_price_move'] = df['close'] - df['open']
    
    # Calculate Volume-Adjusted Price Movement
    df['vol_adj_price_move'] = df['daily_price_move'] * df['volume']
    cap = 3 * df['vol_adj_price_move'].std()
    df['vol_adj_price_move'] = np.clip(df['vol_adj_price_move'], -cap, cap)
    
    # Calculate Rolling Momentum Indicator
    df['agg_vol_adj_price_move'] = df['vol_adj_price_move'].rolling(window=30).sum()
    df['rolling_momentum'] = df['agg_vol_adj_price_move'].ewm(com=0.5).mean() - df['agg_vol_adj_price_move'].ewm(span=10).mean()
    
    # Combined Momentum Factor
    df['combined_momentum'] = df['cumulative_momentum'] * df['sum_high_low_spike']
    
    # Volume Trend (VROC)
    m = 14
    df['vroc'] = df['volume'] / df['volume'].shift(m) - 1
    df['vroc_smoothed'] = df['vroc'].ewm(span=7).mean()
    
    # Final Combined Alpha Factor
    df['alpha_factor'] = df['combined_momentum'] * df['vroc_smoothed']
    
    return df['alpha_factor']
