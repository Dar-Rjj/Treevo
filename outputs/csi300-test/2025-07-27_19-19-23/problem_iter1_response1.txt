```python
import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a Volume and Price Momentum Factor.
    The factor is calculated by first determining the daily price movement using
    the difference between close and open prices. Then, it multiplies this
    movement by the volume to get the volume-adjusted price movement. Finally,
    it calculates a 20-day rolling sum of the volume-adjusted price movements
    and subtracts the previous day's rolling average from the current day's
    to derive the momentum.

    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.

    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """

    # Calculate Daily Price Movement
    df['price_movement'] = df['close'] - df['open']

    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['price_movement'] * df['volume']

    # Aggregate Volume Adjusted Price Movements over a 20-day rolling window
    df['rolling_sum'] = df['volume_adjusted_price_movement'].rolling(window=20).sum()

    # Calculate the Rolling Average of Aggregated Values
    df['rolling_avg'] = df['rolling_sum'].rolling(window=20).mean()

    # Derive Momentum from the Rolling Average
    df['momentum'] = df['rolling_avg'] - df['rolling_avg'].shift(1)

    # Drop rows with NaN values
    df.dropna(inplace=True)
    
    return df['momentum']
```
