```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume and Price Spikes
    df['rolling_volume_20'] = df['volume'].rolling(window=20).mean()
    df['relative_volume'] = df['volume'] / df['rolling_volume_20']
    
    df['z_score_price'] = (df['high'] - df['low']) / df[['high', 'low']].rolling(window=20).std().mean(axis=1)
    
    spike_threshold = 2.0
    df['spike_indicator'] = ((df['relative_volume'] > spike_threshold) | (np.abs(df['z_score_price']) > spike_threshold)).astype(int)
    
    # Adjust Momentum for Spikes
    df['adjusted_log_return'] = df['log_return'] * (1 + 0.5 * df['spike_indicator'])
    
    # Smooth the Adjusted Returns
    df['smoothed_returns'] = df['adjusted_log_return'].ewm(span=7).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum'] = (1 + df['smoothed_returns']).cumprod()
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Combine High-Low Difference with Spikes
    df['spiked_high_low_diff'] = df['high_low_diff'] * df['spike_indicator']
    df['sum_spiked_high_low_diff'] = df['spiked_high_low_diff'].rolling(window=10).sum()
    
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['close'] - df['open']
    
    # Calculate Volume-Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['daily_price_movement'] * df['volume']
    cap = df['volume_adjusted_price_movement'].rolling(window=20).std() * 3
    df['capped_volume_adjusted_price_movement'] = df['volume_adjusted_price_movement'].clip(-cap, cap)
    
    # Calculate Rolling Momentum Indicator
    df['aggregated_vol_adj_movement'] = df['capped_volume_adjusted_price_movement'].rolling(window=30).sum()
    df['rolling_momentum'] = df['aggregated_vol_adj_movement'].ewm(span=5).mean() - df['aggregated_vol_adj_movement'].ewm(span=10).mean()
    
    # Combined Momentum Factor
    df['combined_momentum'] = df['cumulative_momentum'] * df['sum_spiked_high_low_diff']
    
    # Volume Trend
    df['vroc'] = (df['volume'] - df['volume'].shift(14)) / df['volume'].shift(14) * 100
    df['smoothed_vroc'] = df['vroc'].ewm(span=7).mean()
    
    # Final Combined Alpha Factor
    df['final_combined_factor'] = df['combined_momentum'] * df['smoothed_vroc']
    
    # Additional Factor: Volatility-Adjusted Momentum
    df['daily_volatility'] = df['close'].rolling(window=20).std()
    df['normalized_log_return'] = df['log_return'] / df['daily_volatility']
    df['smoothed_normalized_returns'] = df['normalized_log_return'].ewm(span=7).mean()
    
    # Enhanced Combined Factor
    df['enhanced_combined_factor'] = df['final_combined_factor'] + df['smoothed_normalized_returns']
    
    return df['enhanced_combined_factor']
```
