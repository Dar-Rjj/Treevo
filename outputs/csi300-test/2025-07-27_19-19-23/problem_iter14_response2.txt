import pandas as pd
import numpy as np

def heuristics_v2(df, volume_spike_threshold=2.0, lookback_period=10, short_ema_span=5, long_ema_span=10, vroc_period=30):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Volume Spikes
    rolling_std = df['volume'].rolling(window=lookback_period).std()
    df['z_score_volume'] = (df['volume'] - df['volume'].rolling(window=lookback_period).mean()) / rolling_std
    df['volume_spike'] = (df['z_score_volume'] > volume_spike_threshold).astype(int)
    
    # Adjust Momentum for Volume Spikes
    weight = 1.5  # You can adjust this weight based on your analysis
    df['adjusted_log_return'] = df['log_return'] * (1 + (df['volume_spike'] * (weight - 1)))
    
    # Smooth the Adjusted Returns
    df['smoothed_adjusted_returns'] = df['adjusted_log_return'].ewm(span=short_ema_span).mean()
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Combine High-Low Difference and Volume Spikes
    df['spike_high_low_diff'] = df['high_low_diff'] * df['volume_spike']
    df['summed_high_low_diff'] = df['spike_high_low_diff'].rolling(window=lookback_period).sum()
    
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['daily_price_movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['agg_vol_adj_price_movement'] = df['volume_adjusted_price_movement'].rolling(window=30).sum()
    df['rolling_momentum'] = df['agg_vol_adj_price_movement'].ewm(span=short_ema_span).mean() - df['agg_vol_adj_price_movement'].ewm(span=long_ema_span).mean()
    
    # Cumulative Momentum Factor
    df['cumulative_momentum_factor'] = df['summed_high_low_diff'] + df['smoothed_adjusted_returns']
    
    # Combine with Volume Trend
    df['vroc'] = (df['volume'] / df['volume'].shift(vroc_period) - 1) * 100
    df['cumulative_momentum_factor'] = (df['cumulative_momentum_factor'] + 1) * (df['vroc'] + 1)
    
    return df['cumulative_momentum_factor']
