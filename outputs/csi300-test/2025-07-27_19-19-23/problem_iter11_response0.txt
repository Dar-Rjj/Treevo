```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=10, spike_threshold=2.0, smooth_span=5, vroc_period=10):
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Identify Volume and Price Spikes
    df['volume_z_score'] = (df['volume'] - df['volume'].rolling(window=lookback_period).mean()) / df['volume'].rolling(window=lookback_period).std()
    df['price_z_score'] = (df['high'] - df['high'].rolling(window=lookback_period).mean()) / df['high'].rolling(window=lookback_period).std() + \
                          (df['low'] - df['low'].rolling(window=lookback_period).mean()) / df['low'].rolling(window=lookback_period).std()
    
    df['spike_indicator'] = ((df['volume_z_score'] > spike_threshold) | (df['price_z_score'] > spike_threshold)).astype(int)
    
    # Combine High-Low Difference with Volume and Price Spikes
    df['adjusted_high_low_diff'] = df['high_low_diff'] * df['spike_indicator']
    df['summed_adjusted_high_low_diff'] = df['adjusted_high_low_diff'].rolling(window=lookback_period).sum()
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Log Return for Volume and Price Spikes
    df['adjusted_log_return'] = df['log_return'] * (1 + df['spike_indicator'] * 0.5)
    df['smoothed_adjusted_returns'] = df['adjusted_log_return'].ewm(span=smooth_span).mean()
    
    # Price Momentum
    df['avg_high_low_diff'] = df['high_low_diff'].rolling(window=lookback_period).mean()
    
    # Volume Trend
    df['vroc'] = (df['volume'] - df['volume'].shift(vroc_period)) / df['volume'].shift(vroc_period)
    
    # Combined Factor
    df['combined_factor'] = (df['summed_adjusted_high_low_diff'] + df['smoothed_adjusted_returns']) * df['vroc']
    
    # Cumulative Momentum Factor
    df['cumulative_momentum_factor'] = (1 + df['combined_factor']).cumprod()
    
    return df['cumulative_momentum_factor']
```
