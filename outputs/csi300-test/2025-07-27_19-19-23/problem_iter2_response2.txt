```python
import pandas as pd

def heuristics_v2(df, n=10, m=5, spike_threshold=0.2):
    # Compute High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Sum High-Low Differences over last n days
    df['sum_high_low_diff'] = df['high_low_diff'].rolling(window=n).sum()
    
    # Compute Volume Rate of Change (VROC) for period m
    df['vroc'] = df['volume'].pct_change(periods=m)
    
    # Calculate Daily Volume Percentage Change
    df['volume_pct_change'] = df['volume'].pct_change()
    
    # Identify Volume Spikes
    df['volume_spike'] = (df['volume_pct_change'] > spike_threshold).astype(int)
    
    # Multiply VROC by Spike Indicator
    df['vroc_spike_product'] = df['vroc'] * df['volume_spike']
    
    # Combine Price and Volume Factors
    df['factor_value'] = df['sum_high_low_diff'] * df['vroc_spike_product']
    
    # Sum over a fixed lookback period (e.g., 10 days)
    df['alpha_factor'] = df['factor_value'].rolling(window=n).sum()
    
    return df['alpha_factor']
```
