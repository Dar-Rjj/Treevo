import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_days=10, spike_threshold=20, z_score_threshold=2, ema_span=5):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Identify Volume Spikes
    daily_vol_change = df['volume'].pct_change().fillna(0)
    spike_indicator = (daily_vol_change > (spike_threshold / 100)).astype(int)
    
    # Calculate Z-Score of Volume
    rolling_mean = df['volume'].rolling(window=lookback_days).mean()
    rolling_std = df['volume'].rolling(window=lookback_days).std()
    z_score = (df['volume'] - rolling_mean) / rolling_std
    spike_z_score = (z_score.abs() > z_score_threshold).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    combined_high_low = (high_low_diff * spike_indicator).rolling(window=lookback_days).sum()
    
    # Calculate Daily Log Return
    log_return = np.log(df['close'] / df['close'].shift(1)).fillna(0)
    
    # Adjust Momentum for Volume Spikes
    adjusted_log_return = log_return * (spike_z_score + 1)
    smoothed_log_return = adjusted_log_return.ewm(span=ema_span, adjust=False).mean()
    
    # Cumulative Momentum Factor
    cumulative_momentum = (smoothed_log_return + 1).cumprod()
    
    # Normalize to starting value
    cumulative_momentum = cumulative_momentum / cumulative_momentum.iloc[0]
    
    return cumulative_momentum
