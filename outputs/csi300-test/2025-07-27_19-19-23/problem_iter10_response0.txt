```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=10, spike_threshold=0.2, smooth_span=5, vroc_period=10):
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Identify Volume Spikes
    df['volume_pct_change'] = df['volume'].pct_change()
    df['spike_indicator'] = (df['volume_pct_change'] > spike_threshold).astype(int)
    
    # Combine High-Low Difference with Volume Spikes
    df['hl_spike'] = df['high_low_diff'] * df['spike_indicator']
    df['hl_spike_sum'] = df['hl_spike'].rolling(window=lookback_period).sum()
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Log Return for Volume Spikes
    df['adjusted_log_return'] = df['log_return'] * (1 + df['spike_indicator'] * spike_threshold)
    df['smoothed_log_return'] = df['adjusted_log_return'].ewm(span=smooth_span).mean()
    
    # Price Momentum
    df['price_momentum'] = df['high_low_diff'].rolling(window=lookback_period).mean()
    
    # Volume Trend
    df['vroc'] = df['volume'].pct_change(periods=vroc_period)
    
    # Combined Factor
    df['combined_factor'] = (df['hl_spike_sum'] + df['smoothed_log_return']) * df['vroc']
    
    return df['combined_factor']
```
