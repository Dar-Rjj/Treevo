```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))

    # Identify Volume and Price Spikes
    df['vol_20d_ma'] = df['volume'].rolling(window=20).mean()
    df['rel_vol'] = df['volume'] / df['vol_20d_ma']
    
    df['vol_zscore'] = (df['volume'] - df['volume'].rolling(window=20).mean()) / df['volume'].rolling(window=20).std()
    df['price_zscore'] = (df[['high', 'low']].mean(axis=1) - df[['high', 'low']].mean(axis=1).rolling(window=20).mean()) / df[['high', 'low']].mean(axis=1).rolling(window=20).std()
    
    df['spike_indicator'] = ((df['rel_vol'] > 1.5) | (np.abs(df['vol_zscore']) > 2) | (np.abs(df['price_zscore']) > 2)).astype(int)

    # Adjust Momentum for Volume and Price Spikes
    df['adj_log_return'] = df['log_return'] * (1 + 0.5 * df['spike_indicator'])

    # Smooth the Adjusted Returns
    df['smoothed_adj_log_return'] = df['adj_log_return'].ewm(span=7).mean()

    # Cumulative Momentum Factor
    df['cum_momentum'] = (1 + df['smoothed_adj_log_return']).cumprod()

    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']

    # Combine High-Low Difference with Volume and Price Spikes
    df['hl_with_spikes'] = df['high_low_diff'] * df['spike_indicator']
    df['sum_hl_with_spikes'] = df['hl_with_spikes'].rolling(window=10).sum()

    # Calculate Daily Price Movement
    df['daily_price_move'] = df['close'] - df['open']

    # Calculate Volume-Adjusted Price Movement
    df['volume_adj_price_move'] = df['daily_price_move'] * df['volume']
    cap = df['volume_adj_price_move'].rolling(window=30).std() * 3
    df['volume_adj_price_move'] = df['volume_adj_price_move'].clip(lower=-cap, upper=cap)

    # Calculate Rolling Momentum Indicator
    df['agg_volume_adj_price_move'] = df['volume_adj_price_move'].rolling(window=30).sum()
    df['ewma_5'] = df['agg_volume_adj_price_move'].ewm(span=5).mean()
    df['ewma_10'] = df['agg_volume_adj_price_move'].ewm(span=10).mean()
    df['momentum'] = df['ewma_5'] - df['ewma_10']

    # Combined Momentum and High-Low Difference
    df['combined_momentum'] = df['cum_momentum'] + df['sum_hl_with_spikes']

    # Volume Trend
    df['vroc'] = (df['volume'] - df['volume'].shift(14)) / df['volume'].shift(14) * 100
    df['vroc_smoothed'] = df['vroc'].ewm(span=7).mean()

    # Final Combined Alpha Factor
    df['final_factor'] = df['combined_momentum'] * df['vroc_smoothed']

    # Additional Factor: Volatility-Adjusted Momentum
    df['daily_volatility'] = df['close'].rolling(window=20).std()
    df['norm_log_return'] = df['log_return'] / df['daily_volatility']
    df['smoothed_norm_log_return'] = df['norm_log_return'].ewm(span=7).mean()

    # Enhanced Combined Factor
    df['enhanced_factor'] = df['final_factor'] + df['smoothed_norm_log_return']

    return df['enhanced_factor']
```
