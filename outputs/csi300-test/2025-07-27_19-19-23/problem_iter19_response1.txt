import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement
    df['Daily_Price_Movement'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Movement
    df['Volume_Adjusted_Price_Movement'] = df['Daily_Price_Movement'] * df['volume']
    
    # Calculate Rolling Momentum Indicator
    df['Aggregated_Volume_Adjusted_Movement'] = df['Volume_Adjusted_Price_Movement'].rolling(window=30).sum()
    df['30_day_EWMA'] = df['Aggregated_Volume_Adjusted_Movement'].ewm(span=30).mean()
    df['10_day_EWMA'] = df['Aggregated_Volume_Adjusted_Movement'].ewm(span=10).mean()
    df['5_day_EWMA'] = df['Aggregated_Volume_Adjusted_Movement'].ewm(span=5).mean()
    df['Momentum'] = df['5_day_EWMA'] - df['10_day_EWMA']
    
    # Calculate High-Low Difference
    df['High_Low_Diff'] = df['high'] - df['low']
    
    # Identify Volume Spikes
    df['Volume_Z_Score'] = (df['volume'] - df['volume'].rolling(window=30).mean()) / df['volume'].rolling(window=30).std()
    df['Spike_Indicator'] = (df['Volume_Z_Score'] > 2).astype(int)  # 2 standard deviations
    
    # Combine High-Low Difference and Volume Spikes
    df['High_Low_Diff_Spike'] = df['High_Low_Diff'] * df['Spike_Indicator']
    df['Sum_High_Low_Diff_Spike'] = df['High_Low_Diff_Spike'].rolling(window=10).sum()
    
    # Adjust Momentum for Volume Spikes
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Adjusted_Return'] = np.where(df['Spike_Indicator'] == 1, df['Daily_Log_Return'] * 1.5, df['Daily_Log_Return'])
    
    # Smooth the Adjusted Returns
    df['Smoothed_Adjusted_Returns'] = df['Adjusted_Return'].ewm(span=10).mean()
    
    # Volume Trend
    df['VROC'] = df['volume'] / df['volume'].shift(30) - 1  # VROC for period 30
    
    # Combined Factor
    df['Combined_Factor'] = df['Sum_High_Low_Diff_Spike'] + df['Smoothed_Adjusted_Returns']
    df['Combined_Factor_with_Volume_Trend'] = df['Combined_Factor'] * df['VROC']
    
    # Cumulative Momentum Factor
    df['Cumulative_Momentum_Factor'] = (df['Smoothed_Adjusted_Returns'] + 1).cumprod()
    
    return df['Combined_Factor_with_Volume_Trend']
