```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Identify Volume Spikes
    rolling_std = df['volume'].rolling(window=20).std()
    z_score = (df['volume'] - df['volume'].rolling(window=20).mean()) / rolling_std
    spike_indicator = (z_score > 3).astype(int)
    
    # Combine High-Low Difference and Volume Spikes
    adjusted_high_low_diff = high_low_diff * spike_indicator
    summed_high_low_diff = adjusted_high_low_diff.rolling(window=10).sum()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Adjust Momentum for Volume Spikes
    weighted_daily_log_return = daily_log_return * (1 + 0.5 * spike_indicator)
    
    # Smooth the Adjusted Returns
    smoothed_returns = weighted_daily_log_return.ewm(span=10, adjust=False).mean()
    
    # Price Momentum
    price_momentum = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Volume Trend: Volume Rate of Change (VROC)
    vroc = (df['volume'] - df['volume'].shift(20)) / df['volume'].shift(20) * 100
    
    # Combined Factor
    combined_factor = (summed_high_low_diff + smoothed_returns) * vroc
    
    # Additional Factor: Volatility-Adjusted Momentum
    daily_volatility = df['close'].pct_change().rolling(window=20).std()
    normalized_returns = daily_log_return / daily_volatility
    smoothed_normalized_returns = normalized_returns.ewm(span=10, adjust=False).mean()
    
    # Enhanced Combined Factor
    enhanced_combined_factor = combined_factor + smoothed_normalized_returns
    
    return enhanced_combined_factor
```
