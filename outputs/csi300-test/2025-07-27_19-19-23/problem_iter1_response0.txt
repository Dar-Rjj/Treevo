import pandas as pd

def heuristics_v2(df, n=20, m=10):
    # Compute High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute Average High-Low Difference over last n days
    df['avg_high_low_diff'] = df['high_low_diff'].rolling(window=n).mean()
    
    # Compute Volume Rate of Change (VROC) for period m
    df['vroc'] = df['volume'] / df['volume'].shift(m) - 1
    
    # Multiply Price Momentum by VROC
    df['price_momentum'] = df['avg_high_low_diff']
    df['heuristic_factor'] = df['price_momentum'] * df['vroc']
    
    return df['heuristic_factor'].dropna()
