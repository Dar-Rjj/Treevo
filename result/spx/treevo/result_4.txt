def heuristics_v2(df):
    # Compute High-Low difference
    high_low_diff = df['high'] - df['low']
    
    # Choose lookback period and smoothing factor
    lookback_period = 14
    smoothing_factor = 0.2
    
    # Compute the Simple Moving Average (SMA) of the High-Low difference
    sma_high_low_diff = high_low_diff.rolling(window=lookback_period).mean()
    
    # Calculate Raw Momentum: current day's high-low difference - SMA
    raw_momentum = high_low_diff - sma_high_low_diff
    
    # Apply Exponential Weighting for smoothing
    smoothed_momentum = raw_momentum.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Add offset to shift to positive range
    min_value = 0
    max_possible_value = (df['high'].max() - df['low'].min()) - sma_high_low_diff.min()
    offset = -smoothed_momentum.min() + min_value
    
    # Scale to desired range (0 to 100)
    bounded_momentum = (smoothed_momentum + offset) * (100 / (max_possible_value + offset))
    
    return bounded_momentum
