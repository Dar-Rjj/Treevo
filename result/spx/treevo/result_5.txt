def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_momentum'] = df['open'] - df['low'] - df['high_low_diff']

    # Compute Next Day Open to Current Day Close
    df['next_day_open'] = df['open'].shift(-1)
    df['next_day_open_close_diff'] = df['next_day_open'] - df['close']

    # Combine for Initial Alpha Signal
    df['initial_alpha_signal'] = (df['intraday_momentum'] + df['next_day_open_close_diff']) * df['volume']

    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)

    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)

    # Compute Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['daily_price_change'].rolling(window=5).sum() / 5

    # Compute Weighted Volume Indicator
    df['signed_volume_change'] = df['volume_change'] * df['volume_change'].apply(lambda x: 1 if x >= 0 else -1)
    df['weighted_volume_indicator'] = df['signed_volume_change'].rolling(window=5).mean()

    # Final Alpha Signal
    df['final_alpha_signal'] = df['initial_alpha_signal'] + (df['smoothed_price_momentum'] * df['weighted_volume_indicator'])

    return df['final_alpha_signal'].dropna()
