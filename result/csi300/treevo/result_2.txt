def heuristics_v2(df):
    '''
    change
    factors.notna() & returns.notna() & np.isfinite(factors) & np.isfinite(returns)
    to
    factors.notna() & returns.notna()
    '''
    # Calculate simple moving averages for different time windows
    short_sma = df['close'].rolling(window=10).mean()
    long_sma = df['close'].rolling(window=50).mean()
    
    # Compute the difference between short-term and long-term SMA to identify crossover signals
    sma_diff = short_sma - long_sma
    
    # Identify price momentum using rate of change (ROC) over different periods
    roc_10 = df['close'].pct_change(periods=10)
    roc_50 = df['close'].pct_change(periods=50)
    
    # Assess volume trends by calculating the ratio of current volume to the average volume over a period
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Analyze the relative strength index (RSI) to gauge overbought or oversold conditions
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Examine the relationship between open, high, low, and close prices
    daily_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Develop a composite factor by combining multiple indicators
    composite_factor = (sma_diff + roc_10 + volume_ratio + rsi + daily_range + close_open_diff) / 6
    
    # Consider the amount traded in relation to price changes
    amount_price_ratio = df['amount'] / df['close'].pct_change().abs()
    
    # Incorporate the accumulation/distribution line (ADL) to assess buying and selling pressure
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    adl = (money_flow_multiplier * df['volume']).cumsum()
    
    # Evaluate the volatility of the stock using the standard deviation of returns
    returns = df['close'].pct_change()
    volatility = returns.rolling(window=20).std()
    
    # Create a custom indicator by combining high, low, and close prices
    custom_indicator = (df['high'] + df['low'] + (2 * df['close'])) / 4
    
    # Combine all factors into a final alpha factor
    alpha_factor = (composite_factor + amount_price_ratio + adl.pct_change() + volatility + custom_indicator) / 5

    return alpha_factor
