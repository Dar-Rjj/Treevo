def heuristics_v2(df, n=10):
    # Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open'].shift(1)
    
    # Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combined Factor
    combined_factor = intraday_momentum + close_to_open_return + price_change
    
    # Volume Weight
    average_volume = df['volume'].rolling(window=n).mean()
    volume_weight = df['volume'] / average_volume
    weighted_combined_factor = combined_factor * volume_weight
    
    # Adjust by Volume and Price Spikes
    volume_spike = df['volume'] / df['volume'].shift(1)
    price_spike = (df['close'] - df['close'].shift(1)).abs() / df['close'].shift(1)
    
    # Cap at Threshold to Avoid Extreme Values
    volume_spike_threshold = 5
    price_spike_threshold = 0.05
    volume_spike_factor = volume_spike.clip(upper=volume_spike_threshold)
    price_spike_factor = price_spike.clip(upper=price_spike_threshold)
    
    combined_spike = volume_spike_factor * price_spike_factor
    
    # Final Alpha Factor
    final_alpha_factor = weighted_combined_factor / combined_spike
    
    return final_alpha_factor
