def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['prev_close'] = df['close'].shift(1)
    df['close_to_prev_close_return'] = (df['close'] - df['prev_close']) / df['prev_close']
    
    # Calculate High-Low Breakout Intensity
    df['breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['breakout_intensity'] = df['breakout_intensity'].apply(lambda x: max(x, 0))
    
    # Volume Weighted Intraday Return
    df['volume_weighted_intraday_return'] = ((df['close'] - df['open']) / df['open']) * (df['volume'] / df['volume'].rolling(window=5).mean())
    
    # Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine Intraday, Close-to-PreviousClose Returns, Breakout Intensity, and Volume Weighted Intraday Return
    combined_metrics = (df['intraday_return'] + df['close_to_prev_close_return'] + df['breakout_intensity'] + df['volume_weighted_intraday_return']) * df['volume']
    
    # Intermediate Factor
    avg_close_5_days = df['close'].rolling(window=5).mean()
    std_dev_close_5_days = df['close'].rolling(window=5).std()
    df['intermediate_factor'] = (df['volume_increase_ratio'] * (df['close'] - avg_close_5_days) * (std_dev_close_5_days / df['close'].shift(1)))
    
    # Adjust for Volatility
    rolling_volatility = (df['high'] - df['low']).rolling(window=5).std()
    adjusted_combined_metrics = combined_metrics / rolling_volatility
    
    # Final Composite Factor
    final_composite_factor = adjusted_combined_metrics + df['intermediate_factor']
    
    return final_composite_factor
