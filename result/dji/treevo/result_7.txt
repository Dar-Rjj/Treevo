def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] / df['open']) - 1
    daily_volatility = (df['high'] - df['low']).rolling(window=5).std()
    intraday_momentum = (high_low_diff * daily_volatility + open_close_return) / (daily_volatility + 1)

    # Apply Volume and Price Shock Filter
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    price_shock = abs(df['close'] - df['close'].shift(1))
    historical_price_range = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    price_shock_ratio = price_shock / historical_price_range
    shock_filter = (volume_ratio > 1.5) & (amount_ratio > 1.5) & (price_shock_ratio > 0.05)
    intraday_momentum_filtered = intraday_momentum.where(shock_filter, other=np.nan)

    # Measure Volume Synchronization
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    volume_shock_filter = (volume_ratio > 1.5)
    synchronization_value = np.sign(volume_change * price_change).where(volume_shock_filter, other=np.nan)

    # Trend Confirmation
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    crossover_signal = np.where(short_ema > long_ema, 1, -1)

    # Integrate with Intraday Momentum
    adjusted_intraday_momentum = intraday_momentum_filtered * crossover_signal

    # Calculate On-Balance Volume (OBV)
    obv = pd.Series(index=df.index, dtype='float64')
    obv[0] = 0
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv[i] = obv.iloc[i-1] + df['volume'].iloc[i]
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv[i] = obv.iloc[i-1] - df['volume'].iloc[i]
        else:
            obv[i] = obv.iloc[i-1]

    # Weight OBV by EMA Difference
    ema_difference = abs(short_ema - long_ema)
    final_obv_factor = obv * ema_difference

    # Synthesize Final Alpha Factor
    final_alpha_factor = adjusted_intraday_momentum + synchronization_value + final_obv_factor
    return final_alpha_factor
