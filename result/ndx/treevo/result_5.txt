def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Price Momentum
    price_momentum = (df['close'].shift(-1) - df['close']) / df['close']
    
    # Combine Spread and Momentum
    combined_factor = high_low_spread * price_momentum
    
    # Volume Adjust Factor
    volume_adjusted_factor = combined_factor / df['volume']
    
    return volume_adjusted_factor
