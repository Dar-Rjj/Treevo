def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Range
    enhanced_range = (df['high'] - df['low']) / df['close']
    
    # Calculate Price Change
    price_change = df['close'].shift(-1) - df['open']
    
    # Combine Enhanced High-to-Low Range with Price Change
    combined_value = (enhanced_range * 100) + price_change
    
    # Smoothing and Initial Momentum Calculation
    ema_combined = combined_value.ewm(span=7).mean()
    initial_momentum = ema_combined - ema_combined.shift(7)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Multi-Day Price Momentum
    momentum_3d = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    momentum_7d = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    momentum_21d = (df['close'] - df['close'].shift(21)) / df['close'].shift(21)
    
    # Weighted Average the Momentum
    weighted_momentum = (3 * momentum_3d + momentum_7d + momentum_21d) / 31
    
    # Adjust for Volume Changes
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_adjusted_momentum = weighted_momentum * (1 + volume_change)
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + volume_adjusted_momentum
    
    # Calculate Volume-Weighted Price Momentum
    weighted_close = df['close'].rolling(window=5).mean()
    volume_weighted_momentum = (df['close'] - weighted_close) * daily_log_return
    
    # Introduce High-Frequency Volatility
    true_range = pd.DataFrame({
        'HL': df['high'] - df['low'],
        'HC': (df['high'] - df['close'].shift(1)).abs(),
        'LC': (df['low'] - df['close'].shift(1)).abs()
    }).max(axis=1)
    sma_true_range = true_range.rolling(window=15).mean()
    
    # Adjust for High-Frequency Volatility
    volatility_adjusted_momentum = volume_weighted_momentum / sma_true_range
    
    # Final Factor Construction
    final_factor = intermediate_factor + volatility_adjusted_momentum
    
    return final_factor
