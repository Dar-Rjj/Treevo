def heuristics_v2(df):
    # Calculate the cube of the high-to-low price ratio
    cube_high_low_ratio = (df['high'] / df['low']) ** 3
    
    # Calculate the natural logarithm of the difference between the close and open prices
    log_close_open_diff = np.log(np.abs(df['close'] - df['open']))
    
    # Calculate the cube root of the sum of trading amount and volume
    cbrt_amount_volume_sum = np.cbrt(df['amount'] + df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * cube_high_low_ratio + w2 * log_close_open_diff + w3 * cbrt_amount_volume_sum
    
    return heuristics_matrix
